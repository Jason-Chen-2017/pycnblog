```python
import matplotlib.pyplot as plt

def attention_multiplier(P, N, k_P, k_N, P_0, N_0):
    L = 1 / (1 + np.exp(-k_P * (P - P_0))) * 1 / (1 + np.exp(-k_N * (N - N_0)))

```
