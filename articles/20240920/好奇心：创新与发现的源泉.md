import numpy as np

def experiment():
    # 进行一次实验
    return np.random.binomial(n=1, p=0.5)

def optimize():
    # 优化算法
    p = 0.5
    for i in range(100):
        result = experiment()
        p = p * result + (1 - p) * (1 - result)
        p = p / (p + (1 - p))
    return p

def main():
    # 主函数
    optimal_p = optimize()
    print(f"Optimal probability: {optimal_p}")

if __name__ == "__main__":
    main()
```
