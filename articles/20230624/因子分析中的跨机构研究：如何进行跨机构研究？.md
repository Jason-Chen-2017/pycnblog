```
import pandas as pd

# 读取数据
data = pd.read_csv('data.csv')

# 将数据进行分组
grouped_data = data.groupby(['class_num', 'year']).mean()

# 计算各个因子的值
因子_values = grouped_data[['class_num', 'year']].values

# 进行相关性分析和比较
print(因子_values.corr())
print(因子_values.mean())
print(因子_values.std())
```
