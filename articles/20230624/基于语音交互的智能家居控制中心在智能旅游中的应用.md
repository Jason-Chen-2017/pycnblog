```python
def handle_agent_input(agent_id, input_str):
    agent_url = get_agent_url(agent_id)
    if not agent_url:
        return False
    agent_name = input_str
    agent_info = requests.get(agent_url).json()
    if not agent_info:
        return False
    if agent_name!= agent_info['name']:
        return False
    result = requests.post(f'https://api.example.com/agent/{agent_id}/submit', data={'name': agent_name})
    if not result.status_code == 200:
        return False
    return True
```
