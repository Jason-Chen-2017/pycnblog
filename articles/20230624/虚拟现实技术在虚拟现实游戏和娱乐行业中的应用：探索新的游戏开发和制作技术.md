import UnityEngine

class Virtual搏斗Scene : MonoBehaviour
{
    public float speed = 10.0f;
    public float time = 10.0f;
    public float distance = 10.0f;
    public float jump = 2.0f;
    public float jumpForce = 5.0f;
    public float leftHandForce = 3.0f;
    public float rightHandForce = 2.0f;

    private Rigidbody rb;
    private Vector3[] moveVectors;
    private bool isJumping = false;

    void Start()
    {
        rb = GetComponent<Rigidbody>();
        moveVectors = new Vector3[4];
    }

    void Update()
    {
        // Move left
        if (Input.GetButtonDown("Left"))
        {
            rb.AddForce(Vector3.up * speed * Time.deltaTime, ForceMode.Impulse);
        }
        
        // Move right
        if (Input.GetButtonDown("Right"))
        {
            rb.AddForce(Vector3.down * speed * Time.deltaTime, ForceMode.Impulse);
        }
        
        // Jump
        if (isJumping)
        {
            rb.AddForce(Vector3.up * jump * time * Time.deltaTime, ForceMode.Impulse);
            isJumping = false;
        }
        
        // Jump Force
        if (Input.GetButtonDown("Jump"))
        {
            if (jumpForce > 0.0f)
            {
                rb.AddForce(Vector3.up * jumpForce * Time.deltaTime, ForceMode.Impulse);
                isJumping = true;
            }
            else
            {
                rb.AddForce(Vector3.down * jumpForce * Time.deltaTime, ForceMode.Impulse);
            }
        }
    }
}
