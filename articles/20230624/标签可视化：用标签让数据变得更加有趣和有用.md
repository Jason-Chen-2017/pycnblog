下面是一个简单的标签可视化应用示例，使用Tableau创建一个简单的标签桶和标签图表：

```
// Import the necessary libraries
import tns.tableau.client as tsClient
import tns.tableau.admin as tsAdmin

// Create a Tableau workspace
tsAdmin.createWorkspace(
    name: "Tableau workspace",
    user: "user",
    password: "password",
    computer: "computer"
)

// Create a sample data set
tsAdmin.createDataSet("Customer data set")

// Get the data set
tsAdmin.getTable("Customer data set", "Customer table")

// Select the first row of the customer table
customerTable = tsAdmin.getTable("Customer data set", "Customer table", [1])

// Create a new label column in the customer table
customerTable.addColumn(
    name: "Customer Label",
    type: tsAdmin.labels.String,
    length: 10,
    description: "Label for each customer"
)

// Create a new line chart
lineChart = customerTable.createLineChart(
    type: tsAdmin.labels.Series,
    data: customerTable.getRows(),
    name: "Customer Line Chart"
)

// Create a new bar chart
barChart = customerTable.createBarChart(
    type: tsAdmin.labels.Series,
    data: customerTable.getRows(),
    name: "Customer Bar Chart"
)

// Add the line and bar charts to the Tableau workspace
tsAdmin.updateTableauWorkspace(
    name: "Tableau workspace",
    data: lineChart.getRows(),
    chart: barChart.getRows()
)
```
