    """计算向量场大小"""
    return vector[0]

    """计算向量场方向"""
    return vector[1]

    """将向量场中的向量转换为向量，计算向量之间的 dot 乘积"""
    return np.dot(vector[0], vector[1])

    """向量之间的 dot 乘积"""
    return vector1[0] * vector2[0] + vector1[1] * vector2[1]

# 定义向量空间
vector_space = np.array([[2, 1], [1, 2]])

# 计算向量场的大小和方向
