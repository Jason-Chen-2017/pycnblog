### 4.1 数学模型构建

$$
TaxRate = \frac{\sum_{i=1}^{n} W_i \cdot V_i}{\sum_{i=1}^{n} W_i \cdot N_i}

```python
import numpy as np

def calculate_tax_rate(inheritors, asset_values, tax_policy):
    total_value = np.sum(asset_values)
    total_inheritors = len(inheritors)
    tax_base = total_value / total_inheritors
    
    tax_rates = []
    for i in range(total_inheritors):
        tax_rate = tax_policy * asset_values[i] / tax_base
        tax_rates.append(tax_rate)
    
    return tax_rates

def calculate_taxes(inheritors, asset_values, tax_rates):
    taxes = []
    for i in range(len(inheritors)):
        taxes.append(asset_values[i] * tax_rates[i])
    
    return taxes

def main():
    # 设置继承人信息
    inheritors = ['A', 'B', 'C']
    asset_values = [1000, 2000, 3000]
    tax_policy = 0.5  # 税率设定为50%

    # 计算数字遗产税率
    tax_rates = calculate_tax_rate(inheritors, asset_values, tax_policy)
    print("数字遗产税率：", tax_rates)

    # 计算税款
    taxes = calculate_taxes(inheritors, asset_values, tax_rates)
    print("继承人税款：", taxes)

if __name__ == '__main__':
    main()

3. **主函数**：在主函数中，我们设置了继承人信息、资产价值和税收政策，并调用上述两个函数，输出数字遗产税率和继承人税款。
