import numpy as np
import matplotlib.pyplot as plt

# 定义跨学科思维函数
def cross_disciplinary_thinking(x):
    # 对输入数据进行归一化处理
    x_normalized = x / np.linalg.norm(x)
    # 计算洞察力
    insight = np.sum(x_normalized)
    return insight

# 测试数据
x = np.array([0.8, 0.6, 0.5, 0.7])

# 计算洞察力
insight = cross_disciplinary_thinking(x)
