result = linprog(objective_function(C), constraints=constraints(T, R, Q), method='highs')

# 结果分析
print("最优生产成本：", result.x)
print("最优生产时间：", T[result.x.argmax()])
print("最优生产资源：", R[result.x.argmax()])
print("最优产品质量：", Q[result.x.argmax()])
```
