#### 5.2.2 用户画像生成

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.cluster import KMeans

# 生成用户画像
def generate_user_profile(user_id):
    # 查询用户行为数据
    user_actions = collection.find({"user_id": user_id})

    # 提取用户行为文本
    user_texts = [action["content"] for action in user_actions]

    # 构建 TF-IDF 向量器
    vectorizer = TfidfVectorizer(stop_words="english")

    # 训练向量器并转换文本为向量
    user_vectors = vectorizer.fit_transform(user_texts)

    # 使用 K 均值聚类生成用户画像
    kmeans = KMeans(n_clusters=5)
    user_profiles = kmeans.fit_predict(user_vectors.toarray())

    return user_profiles

# 示例
user_profiles = generate_user_profile("user123")
print(user_profiles)
```
