```python
import time
import serial
import requests

# 传感器数据采集
def read_sensor():
    # 假设使用Arduino读取传感器数据
    ser = serial.Serial('COM3', 9600)
    data = ser.readline()
    return data.decode('utf-8')

# 数据上传到云平台
def upload_data(data):
    # 假设使用API上传数据到AWS S3
    url = 'https://s3.amazonaws.com/your-bucket/data.txt'
    headers = {'Content-Type': 'text/plain'}
    requests.put(url, data=data, headers=headers)

# 主程序
def main():
    while True:
        data = read_sensor()
        upload_data(data)
        time.sleep(1)

if __name__ == '__main__':
    main()
```
