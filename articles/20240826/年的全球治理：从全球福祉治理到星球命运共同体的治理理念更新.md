import numpy as np

from cplex import Cplex

# 参数设置

w2 = 0.4
w3 = 0.3
R = 100
E = 50

# 约束条件
g1 = [3, 4, 5]

g3 = [2, 3, 4]

# 目标函数
f = w1 * (g1[0] + g1[1] + g1[2]) + w2 * (g2[0] + g2[1] + g2[2]) + w3 * (g3[0] + g3[1] + g3[2])

cplex = Cplex()
cplex.objective.set_linear(w1 * g1 + w2 * g2 + w3 * g3)
cplex.set线性约束(g1, g2, g3, R, E)
