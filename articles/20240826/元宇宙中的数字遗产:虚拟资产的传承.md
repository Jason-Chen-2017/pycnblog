### 5.2 源代码详细实现

以下是实现数字遗产传承智能合约的源代码示例（使用Solidity语言）：

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract DigitalLegacy {

    address private holder;
    address[] private inheritors;
    mapping(address => bool) private idVerification;

    constructor() {
        holder = msg.sender;
    }

    function addInheritor(address inheritor) public {
        require(msg.sender == holder, "Only the holder can add an inheritor");
        inheritors.push(inheritor);
    }

    function verifyId(address inheritor) public {
        require(msg.sender == inheritor, "Only the inheritor can verify their ID");
        idVerification[inheritor] = true;
    }

    function distributeLegacy() public {
        require(msg.sender == holder || msg.sender == address(this), "Only the holder or the contract can distribute the legacy");
        for (uint i = 0; i < inheritors.length; i++) {
            if (idVerification[inheritors[i]]) {
                payable(inheritors[i]).transfer(this.balance);
            }
        }
    }

}
```
