$$T = (2 + 3 + 4 + 5 + 6) + (5-1) \cdot \frac{80}{100} + (5-2) \cdot \frac{60}{100} + (5-3) \cdot \frac{40}{100} + (5-4) \cdot \frac{20}{100} + (5-5) \cdot \frac{10}{100}$$

$$T = 20 + 4 + 3 + 2 + 1 + 0 = 30$$

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

def calculate_total_time(tasks):
    total_time = 0
    for i, task in enumerate(tasks):
        total_time += task['completion_time'] + (len(tasks) - i - 1) * (task['priority'] / 100)
    return total_time

def plot_attention_map(tasks):
    attention_map = [task['priority'] for task in tasks]
    plt.bar(range(len(tasks)), attention_map)
    plt.xlabel('Task')
    plt.ylabel('Priority')
    plt.title('Attention Map')
    plt.show()

# 示例任务列表
tasks = [
    {'task_id': 1, 'completion_time': 2, 'priority': 80},
    {'task_id': 2, 'completion_time': 3, 'priority': 60},
    {'task_id': 3, 'completion_time': 4, 'priority': 40},
    {'task_id': 4, 'completion_time': 5, 'priority': 20},
    {'task_id': 5, 'completion_time': 6, 'priority': 10}
]

# 计算总任务完成时间
total_time = calculate_total_time(tasks)
print(f"Total time: {total_time} hours")

# 绘制注意义图
plot_attention_map(tasks)
```
