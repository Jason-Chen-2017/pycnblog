class MarriageCertification:
    def __init__(self):
        self.marriages = {}

    def register_account(self, user_id):
        self.marriages[user_id] = None

    def propose_marriage(self, user_id, partner_id):
        if user_id in self.marriages and partner_id in self.marriages:
            if self.marriages[user_id] is None and self.marriages[partner_id] is None:
                self.marriages[user_id] = partner_id
                self.marriages[partner_id] = user_id
                return "Marriage request accepted."
            else:
                return "Both users are already in a marriage."
        else:
            return "One or both users are not registered."
