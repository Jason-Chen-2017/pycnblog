class AttentionCapitalModel(nn.Module):
    def __init__(self, user_embedding_dim, content_embedding_dim):
        super(AttentionCapitalModel, self).__init__()
        self.user_embedding = nn.Embedding(user_embedding_dim, content_embedding_dim)
        self.content_embedding = nn.Embedding(content_embedding_dim, user_embedding_dim)
        self.attention = nn.Linear(user_embedding_dim, 1)

    def forward(self, user_ids, content_ids):
        user_embeddings = self.user_embedding(user_ids)
        content_embeddings = self.content_embedding(content_ids)
        attention_weights = self.attention(content_embeddings).squeeze(-1)
