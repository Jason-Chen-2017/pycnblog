def tsp(dist, n):
    best_cost = np.inf
    best_path = None
    for i in range(n):
        for j in range(i + 1, n):
            for k in range(n):
                if k != i and k != j:
                    dist[i][j] += dist[i][k] + dist[k][j]
                    path = [i, k, j]

def nlp(sentence, vocab, embeddings, n_layers, n_units, n_vocab):
