2. AWS Lambda：作为无服务器函数，用于处理和分析日志数据。
3. AWS EventBridge：作为无服务器事件驱动架构，用于将处理后的日志数据输出和交付给下游系统或用户。

以下是一个具体的代码实例：

```python
# 1. 创建DynamoDB表
import boto3
dynamodb = boto3.resource('dynamodb')
table = dynamodb.create_table(
    TableName='logs',
    KeySchema=[
        {
            'AttributeName': 'timestamp',
            'KeyType': 'HASH'
        }
    ],
    AttributeDefinitions=[
        {
            'AttributeName': 'timestamp',
            'AttributeType': 'N'
        },
        {
            'AttributeName': 'message',
            'AttributeType': 'S'
        }
    ],
    ProvisionedThroughput={
        'ReadCapacityUnits': 5,
        'WriteCapacityUnits': 5
    }
)
table.meta.client.get_waiter('table_exists').wait(TableName='logs')

# 2. 创建Lambda函数
import json
def lambda_handler(event, context):
    record = event['records'][0]
    message = record['message']
    timestamp = record['timestamp']
    dynamodb = boto3.resource('dynamodb')
    table = dynamodb.Table('logs')
    table.put_item(Item={
        'timestamp': timestamp,
        'message': message
    })

# 3. 创建EventBridge事件源
import time
import random
def generate_logs():
    while True:
        timestamp = int(time.time())
        message = f'This is a log message {random.randint(1, 100)}'
        print(message)
        event = {
            'records': [
                {
                    'message': message,
                    'timestamp': timestamp
                }
            ]
        }
        eventbridge = boto3.client('events')
        eventbridge.put_events(
            EventBatch=event
        )
        time.sleep(1)

generate_logs()
```
