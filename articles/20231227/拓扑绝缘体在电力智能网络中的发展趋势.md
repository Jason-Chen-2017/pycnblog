import matplotlib.pyplot as plt

# 定义拓扑绝缘体材料参数

                               'magneton': 9.284 * 1e-24,
                               'speed_of_light': 2.998 * 1e8}}

# 定义拓扑绝缘体结构参数
structure_params = {'thickness': 10 * 1e-9}

# 计算拓扑绝缘体电导率
def compute_conductivity(material, structure):
    electron_mass = material['electron_mass']
    magneton = material['magneton']
    speed_of_light = structure['thickness'] * material['speed_of_light']

    conductivity = (2 * magneton * speed_of_light) / (electron_mass * np.pi * structure['thickness'])

    return conductivity

# 计算拓扑绝缘体磁导率
def compute_magnetic_conductivity(material, structure):
    electron_mass = material['electron_mass']
    magneton = material['magneton']
    speed_of_light = structure['thickness'] * material['speed_of_light']

    magnetic_conductivity = (2 * electron_mass * speed_of_light) / (magneton * np.pi * structure['thickness'])

    return magnetic_conductivity

# 主程序
if __name__ == '__main__':
    material = material_params['Bi2Se3']
    structure = structure_params

    conductivity = compute_conductivity(material, structure)
    magnetic_conductivity = compute_magnetic_conductivity(material, structure)

    print('电导率:', conductivity)
    print('磁导率:', magnetic_conductivity)
