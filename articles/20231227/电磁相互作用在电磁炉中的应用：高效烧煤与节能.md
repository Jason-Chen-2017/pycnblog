import numpy as np

# 电磁力的产生和传播
def E_field(Q, r):
    E = (1 / (4 * np.pi * np.epsilon_0)) * (Q / r**2)
    return E

def B_field(J, dE_dt):
    B = J + np.mu_0 * np.epsilon_0 * dE_dt
    return B

# 电磁力的转化为炭煤的燃烧力
def power(Q, V, eta):
    P = eta * Q * V
    return P

# 炭煤的燃烧过程和节能优化
def heat_of_combustion(m, cp, dT):
    H = m * cp * dT
    return H

def efficiency(W_out, W_in):
    eta = W_out / W_in
    return eta

# 主程序
if __name__ == '__main__':
    Q = 1e-6  # Coulomb

    J = 1e-3  # A/m^2

    V = 110  # V

    H = heat_of_combustion(m, cp, dT)
