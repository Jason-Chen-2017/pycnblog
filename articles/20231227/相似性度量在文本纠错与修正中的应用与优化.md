def cosine_similarity(v, w):

## 4.3 Jaccard相似度示例

def jaccard_similarity(v, w):
    intersection = np.sum(np.minimum(v, w))
    union = np.sum(np.maximum(v, w))
    return intersection / union

w = np.array([4, 5, 6])
