class Task:
    def __init__(self, id, compute_resource):
        self.id = id
        self.compute_resource = compute_resource

    def execute(self):
        print(f"任务{self.id}开始执行")

class ComputeNode:
    def __init__(self, id, cpu):
        self.id = id
        self.cpu = cpu

    def allocate(self, task):
        self.cpu -= task.compute_resource
        print(f"任务{task.id}分配给计算节点{self.id}")

task1 = Task(1, 2)
task2 = Task(2, 4)
compute_node1 = ComputeNode(1, 10)
compute_node2 = ComputeNode(2, 8)

task1.execute()
task2.execute()
compute_node1.allocate(task1)
compute_node2.allocate(task2)
```

1. 边缘计算引擎的具体代码实例：

class EdgeComputingEngine:
    def __init__(self):
        self.tasks = []
        self.edge_devices = []

    def add_task(self, task):
        self.tasks.append(task)

    def add_edge_device(self, edge_device):
        self.edge_devices.append(edge_device)

    def assign_task(self):
        for task in self.tasks:
            for edge_device in self.edge_devices:
                if edge_device.can_handle(task):
                    edge_device.handle(task)
                    self.tasks.remove(task)
                    print(f"任务{task.id}分配给边缘设备{edge_device.id}")
                    break

class Task:
    def __init__(self, id, compute_resource):
        self.id = id
        self.compute_resource = compute_resource

class EdgeDevice:
    def __init__(self, id, cpu):
        self.id = id
        self.cpu = cpu

    def can_handle(self, task):
        return self.cpu >= task.compute_resource

    def handle(self, task):
        self.cpu -= task.compute_resource

edge_computing_engine = EdgeComputingEngine()
task1 = Task(1, 2)
task2 = Task(2, 4)
edge_device1 = EdgeDevice(1, 10)
edge_device2 = EdgeDevice(2, 8)

edge_computing_engine.add_task(task1)
edge_computing_engine.add_task(task2)
edge_computing_engine.add_edge_device(edge_device1)
edge_computing_engine.add_edge_device(edge_device2)

edge_computing_engine.assign_task()
```
