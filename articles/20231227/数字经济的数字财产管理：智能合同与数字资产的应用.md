```python
import hashlib
import hmac
import json
import time

class DigitalAsset:
    def __init__(self, private_key, public_key):
        self.private_key = private_key
        self.public_key = public_key

    def sign(self, data):
        signature = hmac.new(self.private_key, data.encode(), hashlib.sha256).digest()
        return signature

    def verify(self, data, signature):
        return hmac.compare_digest(signature, hmac.new(self.public_key, data.encode(), hashlib.sha256).digest())

    def transfer(self, to, amount):
        data = json.dumps({"to": to, "amount": amount})
        signature = self.sign(data)
        return {"data": data, "signature": signature}
```
