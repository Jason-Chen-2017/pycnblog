智能制造是一种利用人工智能、大数据、物联网、机器学习等新技术的制造方式，可以实现智能化的生产、智能化的质量控制、智能化的物流与运输等。智能制造的特点如下：

## 4.4 梯度下降代码实例

```python
import numpy as np

# 线性回归模型
def linear_regression(x, y, learning_rate=0.01, iterations=1000):
    m, n = len(x), len(x[0])
    X = np.column_stack((np.ones((m, 1)), x))
    theta = np.zeros((n, 1))
    y = np.array(y).reshape(-1, 1)

    for i in range(iterations):
        hypothesis = np.dot(X, theta)
        gradient = np.dot(X.T, (hypothesis - y)) / m
        theta -= learning_rate * gradient

    return theta

# 生成数据
x = np.random.rand(100, 1)
y = 2 * x + 1 + np.random.randn(100, 1)

# 训练模型
theta = linear_regression(x, y)

# 预测
x_predict = np.linspace(0, 1, 100)
y_predict = np.dot(np.column_stack((np.ones((100, 1)), x_predict)), theta)

# 绘图
plt.scatter(x, y, color='blue')
plt.plot(x_predict, y_predict, color='red')
plt.show()
```

import numpy as np

# 线性回归模型
def stochastic_gradient_descent(x, y, learning_rate=0.01, iterations=1000):
    m, n = len(x), len(x[0])
    X = np.column_stack((np.ones((m, 1)), x))
    theta = np.zeros((n, 1))
    y = np.array(y).reshape(-1, 1)

    for i in range(iterations):
        random_index = np.random.randint(m)
        Xi = X[random_index:random_index + 1]
        yi = y[random_index:random_index + 1]
        hypothesis = np.dot(Xi, theta)
        gradient = 2 * (hypothesis - yi)
        theta -= learning_rate * gradient

    return theta

# 生成数据

# 训练模型

# 预测
