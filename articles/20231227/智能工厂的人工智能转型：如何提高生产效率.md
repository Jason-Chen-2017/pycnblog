import numpy as np
from scipy.optimize import linprog

# 生产计划优化问题的目标函数
def objective_function(x):
    return -x[0] - x[1]

# 生产计划优化问题的约束条件
def constraint_function(x):
    return np.array([x[0] + x[1] - 100, -x[0] + 50, x[0] - x[1] - 50])

# 生产计划优化问题的变量范围
bounds = [(0, None), (0, None)]

# 解决生产计划优化问题
result = linprog(objective_function, bounds=bounds, constraints=constraint_function)

print("生产计划优化结果：")
print("最优解：", result.x)
print("最优值：", -result.fun)
```

from itertools import permutations

# 生产线调度优化问题的目标函数
def objective_function(schedule):
    return sum(schedule[i] * schedule[i + 1] for i in range(0, len(schedule), 2))

# 生产线调度优化问题的约束条件
def constraint_function(schedule):
    return np.array([schedule[i] - 60 for i in range(0, len(schedule), 2)])

# 生产线调度优化问题的变量范围
bounds = [(0, 120), (0, 120)]

# 生成所有可能的调度方案
schedules = list(permutations(range(0, 120, 2), 60))

# 找到最优的调度方案
best_schedule = min(schedules, key=objective_function)

print("生产线调度优化结果：")
print("最优调度方案：", best_schedule)
print("最优值：", objective_function(best_schedule))
```
