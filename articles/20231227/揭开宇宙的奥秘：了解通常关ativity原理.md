```python
import math

def entropy(probabilities):
    """
    计算熵
    """
    n = len(probabilities)
    entropy = 0
    for i in range(n):
        p = probabilities[i]
        entropy -= p * math.log2(p)
    return entropy

# 示例使用
probabilities = [0.1, 0.3, 0.2, 0.4]
print("熵:", entropy(probabilities))
```
