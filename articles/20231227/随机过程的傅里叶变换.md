# 4.具体代码实例和详细解释说明

以下是一个使用Python实现随机过程的傅里叶变换的代码示例：

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机过程
def random_process(seed, N, T):
    np.random.seed(seed)
    t = np.linspace(0, T, N)
    x = np.sin(2 * np.pi * 5 * t) + 0.5 * np.random.randn(N)
    return t, x

# 计算随机过程的傅里叶变换
def rpft(x, Fs, Nfft):
    x = x - np.mean(x)
    X = np.fft.fft(x, n=Nfft)
    P = np.abs(X)**2 / Nfft
    f = np.fft.fftfreq(Nfft, d=1/Fs)
    return f, P

# 生成随机过程
seed = 42
N = 1024
T = 1
t, x = random_process(seed, N, T)

# 计算随机过程的傅里叶变换
Fs = 1
Nfft = 2**12
f, P = rpft(x, Fs, Nfft)

# 绘制频域信号
plt.plot(f, P)
plt.xlabel('Frequency (Hz)')
plt.ylabel('Power Density (W/Hz)')
plt.title('Random Process Fourier Transform')
plt.show()
```

这个代码首先生成了一个随机过程，然后计算了该随机过程的傅里叶变换，最后绘制了频域信号。
