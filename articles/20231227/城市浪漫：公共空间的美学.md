```python
import numpy as np
from scipy.optimize import linprog

def space_layout_optimization(area, constraints):
    # 定义目标函数
    objective = np.array([1, 1])
    # 定义约束条件
    A = np.array([[1, 1]])
    b = np.array([area])
    # 使用线性规划求解
    result = linprog(objective, A_ub=A, b_ub=b, bounds=(0, None), method='highs')
    # 返回结果
    return result
```
