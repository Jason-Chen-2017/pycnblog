```python
import scipy.stats as stats

# 生成随机数据
np.random.seed(0)

# 计算样本均值和样本方差
sample_mean = np.mean(x)
sample_variance = np.var(x)

# 计算样本标准差
sample_std_dev = np.std(x)

# 计算Z分数
z_score = (sample_mean - 0) / (sample_std_dev / np.sqrt(1000))

# 计算P值

```
