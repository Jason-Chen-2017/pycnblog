```python
    'respect': lambda prompt: 'respect' in prompt,
    'honesty': lambda prompt: 'lie' not in prompt,
    'fairness': lambda prompt: 'discrimination' not in prompt
}

# 对提示进行判断
prompt = 'Please treat everyone with respect and honesty.'
for principle, rule in rules.items():
    if rule(prompt):
        print(f'The prompt is {principle}.')
    else:
        print(f'The prompt is not {principle}.')
```
