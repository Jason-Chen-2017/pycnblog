import re

class Calculator:
    def __init__(self):
        self.stack = []

    def tokenize(self, expression):
        tokens = re.split(r'[\d\.]+', expression)
        return [int(t) if t.isdigit() else float(t) if '.' in t else t for t in tokens]

    def parse(self, tokens):
        operators = ['+', '-', '*', '/']
        precedence = {'+': 1, '-': 1, '*': 2, '/': 2}
        stack = []
        for token in tokens:
            if token in operators:
                while stack and stack[-1] in operators and precedence[operators[0]] >= precedence[token]:
                    op2 = stack.pop()
                    op1 = stack.pop()
                    stack.append(op1 if operators[0] == '+' or operators[0] == '-' else op2 if operators[0] == '*' or operators[0] == '/' else op1)
                stack.append(token)
            else:
                stack.append(token)
        return stack

    def evaluate(self, tokens):
        stack = self.parse(tokens)
        while stack:
            op2 = stack.pop()
            op1 = stack.pop()
            if op1 is None:
                op1 = 0
            if op2 is None:
                op2 = 0
            if op1 is None and op2 is None:
                break
            if op1 is not None and op2 is not None:
                if op1 == '+' or op1 == '-':
                    stack.append(op1 + op2)
                elif op1 == '*' or op1 == '/':
                    stack.append(op1 * op2)
                elif op1 == '/':
                    stack.append(op1 / op2)
        return stack[0]

    def calculate(self, expression):
        tokens = self.tokenize(expression)
        return self.evaluate(tokens)
