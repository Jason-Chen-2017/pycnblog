```python

def create_rule(rule):
    rules.append(rule)

# 规则的修改
def modify_rule(rule):
    for r in rules:
        if r.condition == rule.condition:
            r.action = rule.action
            break

# 规则的删除
def delete_rule(rule):
    rules.remove(rule)

# 规则的查询
def query_rule(keyword):
    result = []
    for r in rules:
        if keyword in r.condition or keyword in r.action:
            result.append(r)
    return result
```
