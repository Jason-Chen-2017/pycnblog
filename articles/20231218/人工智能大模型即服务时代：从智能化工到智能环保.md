import numpy as np

# 生产线的参数
production_line_params = {
    'capacity': 100,
    'production_cost': 10,
    'time': 10
}

# 遗传算法的参数
population_size = 100
mutation_rate = 0.1
generations = 1000

# 初始化种群
population = np.random.randint(0, 100, size=(population_size, len(production_line_params)))

# 遗传算法的主循环
for generation in range(generations):
    # 计算种群的适应度
    fitness = np.array([calculate_fitness(params, production_line_params) for params in population])

    # 选择
    selected_indices = np.argsort(fitness)[-population_size:]
    selected_population = population[selected_indices]

    # 交叉
    crossover_rate = 0.8
    crossover_indices = np.random.rand(population_size) < crossover_rate
    offspring = np.empty(population_size)
    for i in range(population_size):
        if crossover_indices[i]:
            offspring[i] = np.random.choice(selected_population, len(production_line_params))
        else:
            offspring[i] = population[i]

    # 变异
    mutation_indices = np.random.rand(population_size) < mutation_rate
    for i in range(population_size):
        if mutation_indices[i]:
            offspring[i] = np.random.randint(0, 100, size=len(production_line_params))

    # 更新种群
    population = offspring

# 获取最佳解
best_params = population[np.argmax(fitness)]
print('最佳生产线参数:', best_params)
