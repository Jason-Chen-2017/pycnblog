```python
class RuleEngine:
    def __init__(self):
        self.rules = []

    def add_rule(self, rule):
        self.rules.append(rule)

    def execute(self, facts):
        for rule in self.rules:
            if rule.condition(facts):
                facts[rule.action] = rule.action(facts)
        return facts
```
