```javascript
// 计算器函数
function calculator(a, b, operator) {
  switch (operator) {
    case '+':
      return a + b;
    case '-':
      return a - b;
    case '*':
      return a * b;
    case '/':
      if (b === 0) {
        throw new Error('除数不能为0');
      }
      return a / b;
    default:
      throw new Error('无效的运算符');
  }
}

// 使用计算器函数
const result = calculator(10, 5, '+');
console.log(result); // 15
```
