```python
from itertools import permutations

def is_prime(n):
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True

def primes(n):
    return [x for x in range(2, n) if is_prime(x)]

def permutations_of_primes(n):
    return list(permutations(primes(n)))

def is_valid_password(password):
    return any(x in password for x in permutations_of_primes(8))
```
