import numpy as np

def calculate_gravity(M, m, r):
    G = 6.67430e-11
    F = G * M * m / r**2
    return F

def calculate_acceleration(M, m, r):
    F = calculate_gravity(M, m, r)
    a = F / m
    return a

def integrate(r0, v0, dt):
    t = 0
    while True:
        r = r0 + v0 * t + 0.5 * a * t**2
        t += dt
        yield r

v0 = 7.8e3
dt = 1e3

for r in integrate(r0, v0, dt):
    print(r)
