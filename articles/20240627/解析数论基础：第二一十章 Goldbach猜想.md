def is_prime(n):

        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:

        i += 6
    return True

def goldbach_conjecture(n):

        return False
    for p in range(2, n // 2 + 1):
        q = n - p
        if is_prime(p) and is_prime(q):
            return True
    return False

    if goldbach_conjecture(n):
