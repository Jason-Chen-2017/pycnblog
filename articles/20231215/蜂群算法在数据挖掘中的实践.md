# 4.具体代码实例和详细解释说明

以下是一个简单的蜂群算法实现示例：

```python
import numpy as np
import random

class ParticleSwarmOptimization:
    def __init__(self, num_particles, search_space, w, c1, c2, max_iter):
        self.num_particles = num_particles
        self.search_space = search_space
        self.w = w
        self.c1 = c1
        self.c2 = c2
        self.max_iter = max_iter
        self.positions = np.random.uniform(self.search_space[0], self.search_space[1], (self.num_particles, len(self.search_space)))
        self.velocities = np.zeros_like(self.positions)
        self.best_positions = np.copy(self.positions)
        self.best_fitnesses = np.zeros(self.num_particles)

    def fitness(self, position):
        return 1 / (1 + np.sqrt(np.sum(position**2)))

    def update_velocity(self, position, best_position, velocity, w, c1, c2, r1, r2):
        r1 = random.random()
        r2 = random.random()
        velocity = w * velocity + c1 * r1 * (best_position - position) + c2 * r2 * (best_position - position)
        return velocity

    def update_position(self, position, velocity):
        return position + velocity

    def run(self):
        for _ in range(self.max_iter):
            for i in range(self.num_particles):
                fitness = self.fitness(self.positions[i])
                if fitness > self.best_fitnesses[i]:
                    self.best_fitnesses[i] = fitness
                    self.best_positions[i] = np.copy(self.positions[i])

                r1 = random.random()
                r2 = random.random()

                self.velocities[i] = self.update_velocity(self.positions[i], self.best_positions[i], self.velocities[i], self.w, self.c1, self.c2, r1, r2)
                self.positions[i] = self.update_position(self.positions[i], self.velocities[i])

        best_fitness = np.max(self.best_fitnesses)
        best_position = self.best_positions[np.argmax(self.best_fitnesses)]
        return best_fitness, best_position

# 使用示例
num_particles = 50
search_space = (-5, 5)
w = 0.7
c1 = 1.5
c2 = 1.5
max_iter = 100

pso = ParticleSwarmOptimization(num_particles, search_space, w, c1, c2, max_iter)
best_fitness, best_position = pso.run()

print("最佳适应度：", best_fitness)
print("最佳位置：", best_position)
```
