```python
# 定义组合对象接口
class Component:
    def __init__(self):
        pass

    def operation(self):
        pass

# 定义具体的部分组合对象
class Leaf(Component):
    def operation(self):
        return "Leaf"

# 定义组合对象类
class Composite(Component):
    def __init__(self):
        self.children = []

    def add(self, child):
        self.children.append(child)

    def remove(self, child):
        self.children.remove(child)

    def operation(self):
        result = ""
        for child in self.children:
            result += child.operation()
        return result

# 创建组合对象
root = Composite()
leaf1 = Leaf()
leaf2 = Leaf()
root.add(leaf1)
root.add(leaf2)

# 调用组合对象的方法
print(root.operation())  # 输出：LeafLeaf
```
