```python
def min_spanning_tree(graph):
    n = len(graph)
    dp = [[float('inf')] * (n + 1) for _ in range(n + 1)]
    dp[0][0] = 0

    for i in range(1, n + 1):
        for j in range(1, n + 1):
            for k in range(1, n + 1):
                if k not in graph[i]:
                    continue
                dp[i][j] = min(dp[i][j], dp[k][j - 1] + graph[i][k])

    return dp[n][n - 1]
```
