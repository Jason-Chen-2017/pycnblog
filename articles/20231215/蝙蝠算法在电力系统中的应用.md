import random

# 初始化蝙蝠群

n_dimensions = 2

velocity = np.random.uniform(-1, 1, (n_bats, n_dimensions))

# 定义目标函数
def objective_function(x):
    return np.sum(x ** 2)

# 定义最佳解
best_solution = None

# 蝙蝠算法的迭代过程
for t in range(1000):
    # 更新速度
    for i in range(n_bats):
        a = random.random()
        b = random.random()
        velocity[i] = velocity[i] + (position[i] - best_solution) * a + (best_solution - position[i]) * b

    # 更新位置
    for i in range(n_bats):
        r = random.random()
        if r < 0.5:
            position[i] = position[i] + velocity[i]
        else:
            position[i] = position[i] - velocity[i]

    # 更新最佳解
    current_solution = position[np.argmin(objective_function(position))]
    if best_solution is None or objective_function(current_solution) < objective_function(best_solution):
        best_solution = current_solution

# 输出最佳解
print("最佳解:", best_solution)
