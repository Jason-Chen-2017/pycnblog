```python
import numpy as np
from scipy.optimize import linprog

# 城市需求、条件和目标信息

condition = np.array([1000, 2000, 3000])
objective = np.array([100, 200, 300])

# 城市空间布局、基础设施布局和交通运输系统的规划
x = np.array([100, 200, 300])

# 优化模型
c = np.hstack([-demand, condition, objective])
A = np.vstack([np.eye(3), np.ones(3), np.zeros(3)]).T
b = np.array([0, 0, 0])

# 调用linprog函数实现优化
result = linprog(c, A_ub=A, b_ub=b, method='simplex')

# 根据优化结果调整城市规划方案
x = result.x
```
