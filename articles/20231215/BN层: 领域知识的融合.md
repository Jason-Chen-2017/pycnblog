graph = Graph(rankdir='LR')

# 添加节点
for variable in bn.variables:
    node = graph.add_node(pydotplus.Node(variable.name))
    node['label'] = variable.name

# 添加边
for edge in bn.edges:
    graph.add_edge(pydotplus.Edge(edge.source.name, edge.destination.name))

# 保存图像
dot.set_graph_defaults(rankdir='LR')
dot.set(graph)

# 保存图像为SVG格式
svg = SVG(dot.create_svg())
