## 4.2 使用Python实现基于协同过滤的推荐算法

```python
import numpy as np
from scipy.spatial.distance import cosine

# 计算用户之间的相似度
def similarity(user_matrix):
    similarity_matrix = np.zeros((user_matrix.shape[0], user_matrix.shape[0]))
    for i in range(user_matrix.shape[0]):
        for j in range(i + 1, user_matrix.shape[0]):
            similarity_matrix[i, j] = cosine(user_matrix[i, :], user_matrix[j, :])
            similarity_matrix[j, i] = similarity_matrix[i, j]
    return similarity_matrix

# 进行推荐
def recommend(user_matrix, similarity_matrix, target_user, top_n):
    similarity_user = similarity_matrix[target_user, :]
    similarity_user = np.sort(similarity_user)[::-1]
    recommended_items = []
    for i in range(top_n):
        recommended_items.append(np.argmax(user_matrix[similarity_user[i], :]))
    return recommended_items

# 创建用户-物品矩阵

# 计算用户之间的相似度

# 进行推荐
target_user = 0

recommended_items = recommend(user_matrix, similarity_matrix, target_user, top_n)
```
