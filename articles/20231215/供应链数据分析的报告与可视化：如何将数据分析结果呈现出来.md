import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# 数据预处理
data = pd.read_csv('supplier_data.csv')
data = data.dropna()
data = data.fillna(0)
data = data.scale()

# 数据分析
correlation_matrix = data.corr()
sns.heatmap(correlation_matrix, annot=True, cmap='coolwarm')
plt.show()

# 可视化
sns.pairplot(data)
plt.show()

# 报告
report = 'In this report, we analyze the supplier data and find that the correlation between different variables is strong. We also find that the relationship between variables is non-linear. Based on these findings, we suggest that the company should focus on improving the efficiency of the supply chain.'
