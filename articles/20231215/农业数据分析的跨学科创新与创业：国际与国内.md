import numpy as np
from sklearn.preprocessing import StandardScaler

# 数据预处理
data = pd.read_csv('agriculture_data.csv')
data = data.dropna()

# 数据清洗
scaler = StandardScaler()
data = scaler.fit_transform(data)

# 数据分析
kmeans = KMeans(n_clusters=3)
kmeans.fit(data)

# 数据可视化
