import random

def play_music(file_name):

        data = f.read()
        for i in range(len(data)):
            print(data[i], end='')
            time.sleep(1)

    with open(file_name, 'r') as f:
        data = f.read()
        for i in range(len(data)):

            time.sleep(1)
