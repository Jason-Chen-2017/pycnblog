# 主循环

    # 更新粒子速度
    r1 = np.random.rand(n_particles, n_dimensions)
    r2 = np.random.rand(n_particles, n_dimensions)

    c2 = 2
    v = w * v + c1 * r1 * (p_best - x) + c2 * r2 * (g_best - x)

    # 更新粒子位置
    x = x + v
