# 4.具体代码实例和详细解释说明

以下是一个简单的规则设计器的代码实例，以及其详细解释说明：

```python
# 编写规则
rules = [
    {"condition": "age < 18", "action": "grant_discount"},
    {"condition": "age >= 18 and age < 60", "action": "grant_normal_discount"},
    {"condition": "age >= 60", "action": "grant_senior_discount"},
]

# 维护规则
def update_rule(rule, new_condition, new_action):
    rule["condition"] = new_condition
    rule["action"] = new_action

# 执行规则
def execute_rule(input_data, rules):
    for rule in rules:
        if eval(rule["condition"]):
            execute_action(rule["action"], input_data)
            return

# 返回结果
def execute_action(action, input_data):
    if action == "grant_discount":
        input_data["discount"] = 0.1
    elif action == "grant_normal_discount":
        input_data["discount"] = 0.05
    elif action == "grant_senior_discount":
        input_data["discount"] = 0.03

# 测试
input_data = {"age": 20}
execute_rule(input_data, rules)
print(input_data)  # {"age": 20, "discount": 0.1}
```
