```python
# 计算复杂度
def complexity(weights, values):
    return np.sum(weights * values)

# 计算可靠性
def reliability(fault_rate, recovery_time):
    return fault_rate / recovery_time

# 计算可扩展性
def scalability(throughput, average_latency):
    return throughput / average_latency

# 主程序
if __name__ == '__main__':
    # 设置质量评估指标
    weights = [0.4, 0.3, 0.3]
    values = [100, 200, 300]

    # 设置可靠性评估指标
    fault_rate = 0.01
    recovery_time = 10

    # 设置可扩展性评估指标
    throughput = 1000
    average_latency = 5

    # 计算质量、可靠性、可扩展性
    print('复杂度：', complexity(weights, values))
    print('可靠性：', reliability(fault_rate, recovery_time))
    print('可扩展性：', scalability(throughput, average_latency))
```
