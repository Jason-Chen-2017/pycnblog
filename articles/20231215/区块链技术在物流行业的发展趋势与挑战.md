### 4.1分布式账本代码实例

在这个代码实例中，我们将实现一个简单的分布式账本，用于记录物流信息。

```python
import hashlib
import json
from time import time

class DistributedLedger:
    def __init__(self):
        self.transactions = []

    def add_transaction(self, sender, recipient, amount):
        transaction = {
            'sender': sender,
            'recipient': recipient,
            'amount': amount,
            'timestamp': time()
        }
        self.transactions.append(transaction)

    def get_balance(self, address):
        balance = 0
        for transaction in self.transactions:
            if transaction['sender'] == address:
                balance -= transaction['amount']
            elif transaction['recipient'] == address:
                balance += transaction['amount']
        return balance

    def get_transactions(self):
        return self.transactions

# 使用示例
dl = DistributedLedger()
dl.add_transaction('Alice', 'Bob', 100)
dl.add_transaction('Alice', 'Charlie', 50)
dl.add_transaction('Bob', 'Alice', 20)
dl.add_transaction('Bob', 'Charlie', 30)
dl.add_transaction('Charlie', 'Bob', 10)
dl.add_transaction('Charlie', 'Alice', 20)
print(dl.get_balance('Alice'))  # 输出: 100
print(dl.get_balance('Bob'))  # 输出: 40
print(dl.get_balance('Charlie'))  # 输出: 60
print(dl.get_transactions())  # 输出: 所有交易记录
```

### 4.2加密算法代码实例

在这个代码实例中，我们将实现一个简单的加密算法，用于加密和解密物流信息。

```python
import hashlib
import hmac
import base64

def encrypt(message, key):
    encoded_key = base64.b64encode(key.encode('utf-8'))
    digest = hmac.new(encoded_key, message.encode('utf-8'), hashlib.sha256).digest()
    return base64.b64encode(digest)

def decrypt(encrypted_message, key):
    encoded_key = base64.b64encode(key.encode('utf-8'))
    decoded_encrypted_message = base64.b64decode(encrypted_message)
    digest = hmac.new(encoded_key, decoded_encrypted_message, hashlib.sha256).digest()
    return digest

# 使用示例
message = '物流信息'
key = '密钥'
encrypted_message = encrypt(message, key)
print(encrypted_message)  # 输出: 加密后的物流信息
decrypted_message = decrypt(encrypted_message, key)
print(decrypted_message)  # 输出: 原始的物流信息
```

```python
import time

def get_node_votes(node_addresses):
    votes = {}
    for address in node_addresses:
        with open(f'{address}.txt', 'r') as f:
            votes[address] = int(f.readline().strip())
    return votes

def get_winner(votes):
    max_votes = 0
    winner = None
    for address, votes in votes.items():
        if votes > max_votes:
            max_votes = votes
            winner = address
    return winner

# 使用示例

winner = get_winner(votes)
```
