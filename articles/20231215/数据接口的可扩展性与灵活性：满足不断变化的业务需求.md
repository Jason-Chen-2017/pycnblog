```python
@app.route('/users', methods=['GET'])

    users = [
        {'id': 1, 'name': 'John', 'email': 'john@example.com'},
        {'id': 2, 'name': 'Jane', 'email': 'jane@example.com'},
        {'id': 3, 'name': 'Alice', 'email': 'alice@example.com'}
```

这个代码实例是一个简单的RESTful API，用于获取用户列表。我们可以通过发送GET请求到`/users`端点来获取用户列表。

### 4.3 GraphQL API实例

以下是一个简单的GraphQL API实例：

```python
import graphene
from graphene import ObjectType, String, Field

class User(ObjectType):
    id = String()
    name = String()
    email = String()

class Query(ObjectType):
    users = Field(lambda: [User])

    def resolve_users(self, info):
        users = [
            {'id': 1, 'name': 'John', 'email': 'john@example.com'},
            {'id': 2, 'name': 'Jane', 'email': 'jane@example.com'},
            {'id': 3, 'name': 'Alice', 'email': 'alice@example.com'}
        ]
        return users

schema = graphene.Schema(query=Query)

# 创建GraphQL服务器
app = Flask(__name__)
app.config['GRAPHENE_SCHEMA'] = schema

# 创建GraphQL路由
@app.route('/graphql')
def graphql_endpoint():
    return schema.execute(request.get_json())

if __name__ == '__main__':
    app.run()
```

这个代码实例是一个简单的GraphQL API，用于获取用户列表。我们可以通过发送GraphQL请求到`/graphql`端点来获取用户列表。

```python
import grpc

import time

        users = [
            {'id': 1, 'name': 'John', 'email': 'john@example.com'},
            {'id': 2, 'name': 'Jane', 'email': 'jane@example.com'},
            {'id': 3, 'name': 'Alice', 'email': 'alice@example.com'}
        ]
        return users

# 定义gRPC服务器
class UserServiceServicer(UserService):
    pass

# 创建gRPC服务器
server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
user_service = UserServiceServicer()
server.add_insecure_port('[::]:50051')
server.start()

# 创建gRPC客户端
import user_service_pb2
import user_service_pb2_grpc

channel = grpc.insecure_channel('localhost:50051')
stub = user_service_pb2_grpc.UserServiceStub(channel)

# 调用gRPC服务
response = stub.GetUsers(user_service_pb2.GetUsersRequest())
print(response)

server.stop(0)
```
