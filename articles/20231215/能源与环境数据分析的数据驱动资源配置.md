```python
import pandas as pd
import numpy as np

# 数据清洗
data = pd.read_csv('energy_data.csv')
data = data.dropna()

# 数据转换
data['energy_type'] = data['energy_type'].astype('category')

# 数据归一化
data = (data - data.min()) / (data.max() - data.min())
```
