## 4.具体代码实例和详细解释说明

### 4.1电磁波的产生
```python
import numpy as np
import matplotlib.pyplot as plt

# 电磁波的产生
def generate_electromagnetic_wave(frequency, amplitude, duration):
    time = np.linspace(0, duration, num=1000)
    wave = amplitude * np.sin(2 * np.pi * frequency * time)
    return time, wave

# 测试
time, wave = generate_electromagnetic_wave(1000, 1, 1)
plt.plot(time, wave)
plt.xlabel('Time')
plt.ylabel('Amplitude')
plt.title('Electromagnetic Wave Generation')
plt.show()
```

### 4.2电磁波的传播
```python
import numpy as np
import matplotlib.pyplot as plt

# 电磁波的传播
def propagate_electromagnetic_wave(wave, speed, distance):
    time = np.linspace(0, distance / speed, num=1000)
    propagated_wave = wave * np.exp(-speed * time)
    return time, propagated_wave

# 测试
time, wave = propagate_electromagnetic_wave(wave, 3e8, 1)
plt.plot(time, wave)
plt.xlabel('Time')
plt.ylabel('Amplitude')
plt.title('Electromagnetic Wave Propagation')
plt.show()
```
