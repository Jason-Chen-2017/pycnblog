def sliding_window(arr, k):
    window_start = 0
    window_end = k - 1
    window_sum = sum(arr[window_start:window_end + 1])
    max_window_sum = window_sum

    while window_end < len(arr) - 1:
        window_end += 1
        window_start += 1
        window_sum = window_sum - arr[window_start - 1] + arr[window_end]
        if window_sum > max_window_sum:
            max_window_sum = window_sum

    return max_window_sum

arr = [1, 3, -1, -3, 5, 3, 6, 7]
k = 3
print(sliding_window(arr, k))
