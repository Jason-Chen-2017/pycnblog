def firefly_optimization(n, dim, max_iter, min_dist, alpha, beta0, gamma, rand):

    x = np.random.rand(n, dim)
    
    # 计算适应度
    f = np.sum(x, axis=1)
    
    # 主循环
    for t in range(max_iter):
        for i in range(n):
            # 更新飞行速度
            v = np.random.rand(dim)
            
            # 更新位置
            r1, r2 = rand(2)
            beta = np.exp(-min_dist * min_dist * gamma)
            delta = x[np.random.randint(n), :] - x[i, :]
            if np.dot(delta, f - f[i]) > 0:
                beta = 0
            x[i] = x[i] + v * beta + r1 * delta
            
            # 更新适应度
            f[i] = np.sum(x[i])
            
            # 更新最佳解
            if f[i] < f[np.argmin(f)]:
                x_best = x[i]
                f_best = f[i]
    
    return x_best, f_best
