### 4.2数据清洗

```python
import pandas as pd

# 读取数据源
df = pd.read_csv('data_source.csv')

# 数据清洗规则定义
rules = {
    'column1': {'type': 'int', 'min': 0, 'max': 100},
    'column2': {'type': 'str', 'length': (3, 10)}
}

# 数据清洗规则执行
for column, rule in rules.items():
    if rule['type'] == 'int':
        df[column] = df[column].astype(int)
        df[column] = df[column].clip(lower=rule['min'], upper=rule['max'])
    elif rule['type'] == 'str':
        df[column] = df[column].str.strip()
        df[column] = df[column].str.slice(rule['length'][0], rule['length'][1])

# 数据清洗结果评估
print(df.info())
```
