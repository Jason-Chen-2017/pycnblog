import hashlib

class Blockchain:
    def __init__(self):
        self.chain = []

    def create_block(self, data, timestamp):
        block = {
            'index': len(self.chain) + 1,
            'data': data,
            'timestamp': timestamp,
            'hash': self.hash(data, timestamp)
        }
        self.chain.append(block)

    def hash(self, data, timestamp):
        sha = hashlib.sha256()
        sha.update(str(data + timestamp).encode('utf-8'))
        return sha.hexdigest()

# 创建区块链网络

## 4.2 进行金融交易的代码实例

```python
class Transaction:
    def __init__(self, from_address, to_address, amount):
        self.from_address = from_address
        self.to_address = to_address
        self.amount = amount

    def sign(self, private_key):
        self.signature = hashlib.sha256(str(self.from_address + self.to_address + self.amount + private_key).encode('utf-8')).hexdigest()

    def verify(self, public_key):
        return hashlib.sha256(str(self.from_address + self.to_address + self.amount + public_key).encode('utf-8')).hexdigest() == self.signature

# 进行金融交易
transaction = Transaction('发送地址', '接收地址', '金额')
transaction.sign('私钥')
```
