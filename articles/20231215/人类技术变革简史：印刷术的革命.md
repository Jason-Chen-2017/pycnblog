def create_characters(n):
    characters = []
    for i in range(n):
        characters.append(random.choice('abcdefghijklmnopqrstuvwxyz'))
    return characters

def create_print_board(characters):
    print_board = []
    for character in characters:
        print_board.append(character)
    return print_board

def print_process(print_board, paper):
    print(print_board)
    print(paper)

characters = create_characters(10)
print_board = create_print_board(characters)
