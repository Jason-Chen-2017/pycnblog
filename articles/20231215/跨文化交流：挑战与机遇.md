```python
# 文化差异权重
w_cdi = np.array([0.3, 0.4, 0.3])

# 文化差异大小
d_cdi = np.array([1, 2, 3])

# 沟通效果权重
w_tei = np.array([0.3, 0.4, 0.3])

# 沟通效果大小
e_tei = np.array([1, 2, 3])

# 文化适应权重
w_cai = np.array([0.3, 0.4, 0.3])

# 文化适应大小
a_cai = np.array([1, 2, 3])

# 计算文化差异指数
cdi = np.sum(np.multiply(w_cdi, d_cdi)) / len(d_cdi)

# 计算沟通效果指数
tei = np.sum(np.multiply(w_tei, e_tei)) / len(e_tei)

# 计算文化适应指数
cai = np.sum(np.multiply(w_cai, a_cai)) / len(a_cai)

print("文化差异指数：", cdi)
print("沟通效果指数：", tei)
print("文化适应指数：", cai)
```
