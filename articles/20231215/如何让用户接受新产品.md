```python
import pandas as pd

# 读取产品特性数据
data = pd.read_csv('product_features.csv')

# 评估产品特性
def evaluate_features(data):
    # 计算产品特性得分
    feature_scores = data['feature'].apply(lambda x: x['score'])
    # 计算产品特性平均得分
    avg_score = feature_scores.mean()
    return avg_score

# 评估产品特性得分
avg_score = evaluate_features(data)
print('产品特性平均得分：', avg_score)
```
