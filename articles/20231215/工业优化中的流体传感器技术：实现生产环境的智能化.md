import numpy as np
from scipy.optimize import minimize

# 工业优化模型
def objective_function(x):
    # 目标函数
    return np.sum(x**2)

# 约束条件
def constraint_function(x):
    # 约束条件
    return x - 10

# 初始化参数
x0 = np.array([1.0, 1.0, 1.0])

# 优化算法
result = minimize(objective_function, x0, constraints={"type": "eq", "fun": constraint_function})
