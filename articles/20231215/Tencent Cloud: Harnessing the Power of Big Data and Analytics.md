### 4.2.1 大数据计算服务

```python
from pyspark import SparkContext

def create_spark_context(master, app_name):
    return SparkContext(master=master, app_name=app_name)

def execute_mapreduce(spark_context, map_func, reduce_func, data_path, output_path):
    data_rdd = spark_context.textFile(data_path)
    map_rdd = data_rdd.map(map_func)
    reduce_rdd = map_rdd.reduce(reduce_func)
    reduce_rdd.saveAsTextFile(output_path)

def stop_spark_context(spark_context):
    spark_context.stop()
```
### 4.2.2 数据流服务
```python
from pyflink.common.serialization import SimpleStringSchema
from pyflink.datastream import StreamExecutionEnvironment
from pyflink.table import StreamTableEnvironment, DataTypes

def create_stream_execution_environment():
    return StreamExecutionEnvironment.get_execution_environment()

def create_stream_table_environment(stream_execution_environment):
    return StreamTableEnvironment.create(stream_execution_environment)

def execute_window_join(stream_table_environment, left_data_path, right_data_path, result_path):
    left_table = stream_table_environment.from_path(left_data_path, schema=DataTypes.data_types().row(field0=DataTypes.STRING(), field1=DataTypes.STRING()), format='csv', field_delimiter=',')
    right_table = stream_table_environment.from_path(right_data_path, schema=DataTypes.data_types().row(field2=DataTypes.STRING(), field3=DataTypes.STRING()), format='csv', field_delimiter=',')
    result_table = left_table.join(right_table, left_table.field0 == right_table.field2).to_path(result_path, format='csv', field_delimiter=',')
    stream_table_environment.execute('window_join')
```
