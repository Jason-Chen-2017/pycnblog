    elif mode == 1:
        return np.array([0, 1])

# 计算弦的振动频率

    plt.xlabel('x')
    plt.ylabel('y')

    plt.show()

# 主函数
if __name__ == '__main__':
    length = 1.0  # 弦的长度
    mode = 0      # 振动模式
    frequency = string_vibration_frequency(length, mode)
    x, t, y = string_vibration_behavior(length, mode, frequency)
    plot_string_vibration(x, t, y)
