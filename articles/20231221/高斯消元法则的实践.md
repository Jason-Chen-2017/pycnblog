```python
import numpy as np

def gauss_elimination(A, b):
    n = len(A)
    for i in range(n):
        max_row = i
        for j in range(i, n):
            if abs(A[j][i]) > abs(A[max_row][i]):
                max_row = j
        A[[i, max_row]] = A[[max_row, i]]
        b[i], b[max_row] = b[max_row], b[i]

        if A[i][i] == 0:
            return None

        for j in range(i+1, n):
            factor = A[j][i] / A[i][i]
            A[j] = [A[j][k] - factor * A[i][k] for k in range(n)]
            b[j] -= factor * b[i]

    x = [0] * n
    for i in range(n):
        x[i] = b[i] / A[i][i]

    return x
```
