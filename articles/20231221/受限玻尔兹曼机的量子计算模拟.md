    energy = np.sum(state ** 2)
    entropy = -np.sum(state * np.log2(state))

        # 随机生成一个新的状态

        delta_entropy = new_entropy - entropy

        # 接受新状态

            energy = new_energy
            entropy = new_entropy

    return state, energy, entropy
