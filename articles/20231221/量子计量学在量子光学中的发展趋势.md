import scipy.linalg as la

# 定义光子相位编码矩阵
H = np.array([[1, 1], [1, -1]])

# 定义两个量子态
state1 = np.array([1, 0])

# 实现光子相位纠缠
state = la.dot(H, np.kron(state1, state2))
