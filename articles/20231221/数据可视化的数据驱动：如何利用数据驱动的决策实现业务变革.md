```python
import pandas as pd

plt.ylabel('y')
plt.title('Scatter Plot')
plt.show()

# 绘制一个直方图
plt.hist(df['x'], bins=5)
plt.xlabel('x')
plt.ylabel('Frequency')
plt.title('Histogram')
plt.show()

# 绘制一个箱线图
plt.boxplot(df['x'])
plt.xlabel('x')
plt.title('Box Plot')
plt.show()

# 绘制一个条形图
plt.bar(df['x'], df['y'])
plt.xlabel('x')
plt.ylabel('y')
plt.title('Bar Chart')
plt.show()
```
