# 使用高斯随机场模型进行预测
def gaussian_random_field(x, sigma_f, sigma_n):
    N = len(x)
    K = np.eye(N) + np.exp(-np.square(x[:, np.newaxis] - x[np.newaxis, :]) / (2 * sigma_f**2))
    f = np.linalg.cholesky(K)
    y_pred = f.dot(np.random.randn(N, 1)) + sigma_n * np.random.randn(N, 1)
    return y_pred

# 绘制预测结果
sigma_f = 1
sigma_n = 0.5
y_pred = gaussian_random_field(x, sigma_f, sigma_n)
plt.scatter(x, y)
plt.plot(x, y_pred, 'r-')
plt.xlabel('x')
plt.ylabel('y')
plt.title('高斯随机场预测')
plt.show()
