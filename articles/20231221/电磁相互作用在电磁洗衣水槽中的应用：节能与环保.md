import numpy as np
import matplotlib.pyplot as plt

# 电磁场强度的计算
def electric_field(charge, distance):
    epsilon_0 = 8.854e-12  # 空气的电容性
    electric_field = (1 / (4 * np.pi * epsilon_0)) * (charge / distance**2)
    return electric_field

# 电磁场方向的计算
def magnetic_field(current, distance):
    mu_0 = 4 * np.pi * 1e-7  # 空气的磁允性
    magnetic_field = mu_0 * current / distance
    return magnetic_field

# 电磁相互作用的计算
def electromagnetic_interaction(charge1, charge2, distance):
    k = 8.9875517923 * 10**9  # 电磁常数
    force = k * (charge1 * charge2 / distance**2)
    return force

# 节能效率的计算

    efficiency = output_work / input_work
    return efficiency

# 环保性能的计算
def environmental_performance(input_work, output_work):
    energy = input_work - output_work
    return energy
