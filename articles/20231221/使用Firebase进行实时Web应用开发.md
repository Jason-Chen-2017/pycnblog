```javascript
// 添加Firebase云存储的SDK
var config = {
  apiKey: "YOUR_API_KEY",
  authDomain: "YOUR_AUTH_DOMAIN",
  databaseURL: "YOUR_DATABASE_URL",
  storageBucket: "YOUR_STORAGE_BUCKET"
};
firebase.initializeApp(config);

var storageRef = storage.ref();
var filesRef = storageRef.child("images/");
var uploadTask = filesRef.putFile("path/to/file");

uploadTask.on("state_changed", function(snapshot) {
  var progress = (snapshot.bytesTransferred / snapshot.totalBytes) * 100;
  console.log("Upload is " + progress + "% done");
}, function(error) {
  console.log("Upload failed");
}, function() {
  console.log("Upload success");
});

// 下载文件
var downloadURL = uploadTask.snapshot.downloadURL;
console.log("File available at", downloadURL);
```

在这个代码实例中，我们首先添加了Firebase云存储的SDK。然后，我们使用`storage.ref()`来创建一个引用，并使用`child("images/")`来指定要上传的文件夹。接下来，我们使用`putFile("path/to/file")`来上传文件。最后，我们使用`snapshot.downloadURL`来获取下载文件的URL。
