```python
import os
import json

def get_legal_documents():
    url = 'https://www.example.com/legal_documents'
    response = requests.get(url)
    documents = json.loads(response.text)
    for document in documents:
        filename = document['name']
        content = document['content']
        with open(os.path.join('legal_documents', filename), 'w', encoding='utf-8') as file:
            file.write(content)

def main():
    while True:
        get_legal_documents()
        time.sleep(86400)

if __name__ == '__main__':
    main()
```
