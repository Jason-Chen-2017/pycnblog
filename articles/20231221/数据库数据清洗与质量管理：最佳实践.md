```
import pandas as pd
import numpy as np

# 读取数据
data = pd.read_csv('data.csv')

data['gender'] = data['gender'].map({'M': 1, 'F': 0})
data['income'] = data['income'].map(lambda x: x if x <= 50000 else np.nan)

data.to_csv('clean_data.csv', index=False)
```

## 4.2 数据质量检查代码实例

### 4.2.1 数据质量检查代码实例1

```
import pandas as pd

# 读取数据
data = pd.read_csv('clean_data.csv')

# 数据质量检查
def check_data_quality(data):
    # 检查缺失值
    missing_values = data.isnull().sum()
    # 检查重复值
    duplicated_values = data.duplicated().sum()
    # 检查异常值
    outliers = data[data > 1.5 * data.std()]

    return missing_values, duplicated_values, outliers

missing_values, duplicated_values, outliers = check_data_quality(data)
print('缺失值：', missing_values)
print('重复值：', duplicated_values)
print('异常值：', outliers)
```

## 4.3 数据纠正代码实例

### 4.3.1 数据纠正代码实例1

```
import pandas as pd

# 读取数据
data = pd.read_csv('clean_data.csv')

# 数据纠正
def correct_data(data):
    # 填充缺失值
    data['age'].fillna(data['age'].mean(), inplace=True)
    # 删除重复值
    data.drop_duplicates(inplace=True)
    # 删除异常值
    data.drop(data[data > 1.5 * data.std()].index, inplace=True)

    return data

data = correct_data(data)
print(data)
```
