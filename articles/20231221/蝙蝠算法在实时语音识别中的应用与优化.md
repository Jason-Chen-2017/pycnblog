5. 更新蝙蝠群的适应度：根据更新后的蝙蝠群，计算新的适应度。

def initialize_bat_population(population_size, search_space):

    for i in range(population_size):
        population.append(np.random.uniform(search_space[0], search_space[1]))
    return population

# 计算蝙蝠群的适应度
def calculate_fitness(population, fitness_function):
    fitness_values = []
    for individual in population:
        fitness_values.append(fitness_function(individual))
    return fitness_values

        if r1 > 0.5:
            velocity[i] = c1 * velocity[i] + c2 * np.random.uniform(-1, 1)
        else:
            velocity[i] = c1 * velocity[i]
        position[i] = position[i] + velocity[i]
        if r2 > 0.5:
            position[i] = np.random.uniform(0, 1) * fitness_values[i]
    return population, velocity, position

def main():
    # 初始化参数
    population_size = 50
    search_space = (-10, 10)
    max_iterations = 1000
    c1 = 0.5
    c2 = 1
    r1 = 0.2
    r2 = 0.2

    # 初始化蝙蝠群
    population = initialize_bat_population(population_size, search_space)

    # 主循环
    for iteration in range(max_iterations):
        # 计算蝙蝠群的适应度
        fitness_values = calculate_fitness(population, fitness_function)

        # 更新蝙蝠的速度和位置
        population, velocity, position = update_bat_velocity_position(population, velocity, position, fitness_values, c1, c2, r1, r2)

        # 判断终止条件

            print("Iteration:", iteration, "Best fitness:", min(fitness_values))
