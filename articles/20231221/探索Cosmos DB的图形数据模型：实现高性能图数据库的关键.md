```python
# 创建Cosmos客户端
client = CosmosClient("https://<your-account>.documents.azure.com:443/")

# 获取数据库
database = client.get_database_client("<your-database>")

# 获取容器
container = database.get_container_client("<your-container>")

# 创建节点
node = {
    "id": "1",
    "name": "Alice",
    "age": 30,
    "friends": ["2", "3"]
}

# 创建关系
relationship = {
    "source": "1",
    "target": "2",
    "weight": 1
}

# 添加节点
container.upsert_item(node)

# 添加关系
container.upsert_item(relationship)

# 查询节点
query = "SELECT * FROM c WHERE c.age > 25"
items = list(container.query_items(
    query=query,
    enable_cross_partition_query=True
))

# 打印结果
for item in items:
    print(item)
```
