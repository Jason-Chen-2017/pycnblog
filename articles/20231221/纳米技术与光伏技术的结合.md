```python
    output_power = input_power * efficiency * area
    return output_power

    cost = output_power / (life_time * efficiency)
    return cost

input_power = 1000  # W

cost = calculate_cost(output_power, 20, efficiency)

print("Output power: {:.2f} W".format(output_power))

```
