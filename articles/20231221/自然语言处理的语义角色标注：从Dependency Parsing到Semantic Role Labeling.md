import nltk
from nltk import pos_tag, word_tokenize
from nltk.corpus import wordnet

# 输入句子
sentence = "John gave Mary a book"

# 词法分析
tokens = word_tokenize(sentence)

# 词性标注
pos_tags = pos_tag(tokens)

# 依赖解析
dependency_parse = nltk.RegexpParser("SBAR|VP|NP|PP")
dependency_parse.parse(pos_tags)

# 语义角色标注
def semantic_role_labeling(sentence):
    # 词法分析
    tokens = word_tokenize(sentence)

    # 词性标注
    pos_tags = pos_tag(tokens)

    # 语义角色标注
    semantic_roles = {}
    for word, pos in pos_tags:
        if "VB" in pos:
            verb = word
            semantic_roles[verb] = {"subject": None, "object": None, "indirect_object": None}
        elif "NN" in pos or "NNS" in pos:
            noun = word
            if verb:
                if "direct_object" in semantic_roles[verb]:
                    semantic_roles[verb]["direct_object"].append(noun)
                else:
                    semantic_roles[verb]["direct_object"] = [noun]
        elif "IN" in pos:
            preposition = word
            if verb:
                if "indirect_object" in semantic_roles[verb]:
                    semantic_roles[verb]["indirect_object"].append(preposition)
                else:
                    semantic_roles[verb]["indirect_object"] = [preposition]
    return semantic_roles

semantic_roles = semantic_role_labeling(sentence)
print(semantic_roles)

在semantic_role_labeling函数中，我们首先对句子进行词法分析和词性标注。然后，我们遍历每个词和它的词性，并根据词性来识别语义角色。如果是动词，我们将其标注为主题、直接对象和间接对象。如果是名词，我们将其标注为直接对象。如果是介词，我们将其标注为间接对象。
