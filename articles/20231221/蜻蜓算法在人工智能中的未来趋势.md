import numpy as np

def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def hopfield(weights, thresholds, initial_states):
    num_neurons = weights.shape[0]
    states = np.zeros((num_neurons, 1))
    states[:, 0] = initial_states

    while True:
        new_states = np.zeros((num_neurons, 1))
        for i in range(num_neurons):
            new_states[i, 0] = sigmoid(np.dot(weights[i, :], states) + thresholds[i])

        if np.allclose(states, new_states, atol=1e-6):
            break

        states = new_states

    return states
