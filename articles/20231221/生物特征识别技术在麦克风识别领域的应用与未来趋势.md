```python
import sounddevice as sd
import numpy as np
import scipy.signal as signal
import sqlite3
from sklearn.metrics.pairwise import cosine_similarity

# 数据收集
def record(duration):
    return sd.rec(int(duration * sd.DEFAULT_SAMPLING_RATE), samplerate=sd.DEFAULT_SAMPLING_RATE, channels=1)

# 预处理
def preprocess(data):
    data = np.abs(data)
    data = signal.medfilt(data, kernel_size=3)
    data = data / np.max(data)
    return data

# 特征提取
def extract_features(data):
    f, t, Sxx = signal.spectrogram(data, fs=sd.DEFAULT_SAMPLING_RATE, window='hann', nperseg=256, noverlap=128)
    return f, t, Sxx

# 声纹模板存储
def store_template(f, t, Sxx):
    conn = sqlite3.connect('template.db')
    cursor = conn.cursor()
    cursor.execute('CREATE TABLE IF NOT EXISTS templates (frequency REAL, time REAL, energy REAL)')
    for i in range(len(f)):
        cursor.execute('INSERT INTO templates (frequency, time, energy) VALUES (?, ?, ?)', (f[i], t[i], Sxx[i]))
    conn.commit()
    conn.close()

# 声纹匹配
def match(data):
    conn = sqlite3.connect('template.db')
    cursor = conn.cursor()
    cursor.execute('SELECT * FROM templates')
    templates = cursor.fetchall()
    conn.close()
    data_f, data_t, data_Sxx = extract_features(data)
    similarities = []
    for f, t, Sxx in templates:
        similarity = cosine_similarity([[f, t, Sxx]], [[f, t, Sxx]])
        similarities.append(similarity[0][0])
    return similarities

# 识别结果输出
def recognize(similarities):
    max_similarity = max(similarities)
    index = similarities.index(max_similarity)
    print(f'The recognized person is {index + 1}')

if __name__ == '__main__':
    duration = 5  # 录音时长，单位秒
    data = record(duration)
    preprocessed_data = preprocess(data)
    store_template(preprocessed_data)
    recognized_similarities = match(preprocessed_data)
    recognize(recognized_similarities)
```
