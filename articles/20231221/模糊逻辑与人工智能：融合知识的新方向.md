import numpy as np

# 定义模糊集
A = {'a': 0.2, 'b': 0.5, 'c': 0.8}
B = {'x': 0.3, 'y': 0.7, 'z': 0.9}

# 定义模糊关系
R = np.array([[0.2, 0.5], [0.5, 0.8]])

# 定义模糊规则
rules = [
    {'if': {'A': 'a', 'B': 'x'}, 'then': 'C1'},
    {'if': {'A': 'a', 'B': 'y'}, 'then': 'C2'},
    {'if': {'A': 'b', 'B': 'x'}, 'then': 'C3'},
    {'if': {'A': 'b', 'B': 'y'}, 'then': 'C4'},
    {'if': {'A': 'c', 'B': 'x'}, 'then': 'C5'},
    {'if': {'A': 'c', 'B': 'y'}, 'then': 'C6'}

# 使用模糊逻辑运算符进行推理
def evaluate_rule(rule, A, B, R):
    if rule['if']['A'] and rule['if']['B']:
        return rule['then']
    else:
        return None

def decision_tree(A, B, R, rules):
    # 选择一个特征作为根节点
    best_feature = max(A, key=lambda x: len(rules[x]))
    # 根据该特征将数据集划分为多个子集
    subsets = {best_feature: [rule['if'][best_feature] for rule in rules if rule['if'][best_feature] in A[best_feature]]}
    # 对每个子集递归地应用决策树算法
    for subset in subsets.values():
        for x in subset:
            A[x] = False
        for y in subset:
            B[y] = False
        for z in subset:
            R[z] = False
        for rule in rules:
            if rule['if'][best_feature] not in subset:
                continue
            result = evaluate_rule(rule, A, B, R)
            if result:
                return result
    # 返回最终的决策
    return None

# 使用决策树进行预测
result = decision_tree(A, B, R, rules)
print(result)
