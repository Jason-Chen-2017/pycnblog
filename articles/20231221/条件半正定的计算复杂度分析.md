```python
    return np.array([2*x[0] + 4*x[1] - 4, 4*x[0] + 4*x[1]])

def line_search(x, alpha=0.01, beta=0.99):
    x_prev = x
    while True:
        x_new = x - alpha * gradient(x)
        if f(x_new) < f(x_prev):
            x_prev = x
            x = x_new
        else:
            alpha *= beta
    return x

x_optimal = line_search(x)
print(x_optimal)
```
