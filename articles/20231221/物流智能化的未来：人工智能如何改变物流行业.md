def greedy_scheduling(vehicles, orders):
    # 按距离排序
    orders.sort(key=lambda order: order['distance'])
    # 遍历所有订单
    for order in orders:
        # 找到距离最近的车辆
        nearest_vehicle = min(vehicles, key=lambda vehicle: vehicle['distance'])
        # 更新车辆状态
        nearest_vehicle['state'] = 'busy'
        # 更新订单状态
        order['state'] = 'delivered'
```

    offspring = crossover(selected)

    offspring = mutate(offspring)

from sklearn.cluster import KMeans

def bind_orders(orders):
    # 提取特征
    features = [order['features'] for order in orders]
    # 训练聚类模型
    model = KMeans(n_clusters=3)
    model.fit(features)
    # 分组
    for order in orders:
        cluster = model.labels[order['index']]
        order['cluster'] = cluster
```
