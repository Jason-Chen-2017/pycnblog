import os
import socket
import threading
import hashlib

# 发送文件
def send_file(file_path, dest_ip, dest_port):
    with open(file_path, 'rb') as f:
        file_size = os.path.getsize(file_path)
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.connect((dest_ip, dest_port))
        sock.sendall(file_size.to_bytes(8, 'big'))
        while True:
            data = f.read(1024)
            if not data:
                break
            sock.sendall(data)
        sock.close()

def receive_file(file_path, src_ip, src_port):
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.bind((src_ip, src_port))
    sock.listen(1)
    conn, addr = sock.accept()
    file_size = int.from_bytes(conn.recv(8), 'big')
    with open(file_path, 'wb') as f:
        while True:
            data = conn.recv(1024)
            if not data:
                break
            f.write(data)
    conn.close()

# 主函数
if __name__ == '__main__':
    send_file('test.txt', '127.0.0.1', 8080)
    receive_file('test.txt', '127.0.0.1', 8080)
