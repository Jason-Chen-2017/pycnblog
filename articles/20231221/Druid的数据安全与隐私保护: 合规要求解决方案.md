class User:
    def __init__(self, username, role):
        self.username = username
        self.role = role

class Resource:
    def __init__(self, resource_id, operations):
        self.resource_id = resource_id
        self.operations = operations

def check_access(user, resource):

        return True
    return False

    print("Access granted")

    print("Access denied")
