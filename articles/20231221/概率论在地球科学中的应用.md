# 泊松分布参数

# 时间步长

# 创建时间数组

# 创建概率数组
probability = np.zeros(time_range + 1)

# 计算概率分布
for t in time:
    probability[t] = np.exp(-lambda_ * t) * (lambda_ * t)**(time_step - 1) / factorial(time_step - 1)

# 绘制概率分布
plt.plot(time, probability)
plt.xlabel('Time')
plt.ylabel('Probability')
plt.title('Poisson Distribution of Earthquake Occurrence')
plt.show()
