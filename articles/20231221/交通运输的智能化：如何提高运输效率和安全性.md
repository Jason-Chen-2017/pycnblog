## 4.1 智能交通信息系统的代码实例

```python
import numpy as np
import cv2
import socket

# 数据收集
def collect_data():
    cap = cv2.VideoCapture(0)
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        cv2.imshow('frame', frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break
    cap.release()
    cv2.destroyAllWindows()

# 数据处理
def process_data(frame):
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    edges = cv2.Canny(blur, 50, 150)
    lines = cv2.HoughLinesP(edges, 1, np.pi / 180, 100, np.array([]), minLineLength=40, maxLineGap=5)
    return lines

# 信息传播
def send_data(lines):
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.connect(('localhost', 8080))
    s.send(np.array(lines))
    s.close()

if __name__ == '__main__':
    collect_data()
    cap = cv2.VideoCapture(0)
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        lines = process_data(frame)
        send_data(lines)
```

在这个代码实例中，我们首先通过摄像头捕获视频帧，然后对帧进行处理（如灰度化、模糊化、边缘检测、直线检测等），最后通过socket发送处理后的数据。

import random

# 交通流模型
def traffic_flow_model():
    num_cars = np.random.randint(50, 200)
    car_speeds = np.random.uniform(30, 80, num_cars)
    return num_cars, car_speeds

# 流控制策略
def traffic_light_control(num_cars, car_speeds):
    if num_cars < 50:
        return 'green'
    elif np.mean(car_speeds) < 50:
        return 'yellow'
    else:
        return 'red'

# 优化目标
def optimize_objective(num_cars, car_speeds):
    return np.mean(car_speeds)

if __name__ == '__main__':
    num_cars, car_speeds = traffic_flow_model()
    print('Num cars:', num_cars)
    print('Car speeds:', car_speeds)
    print('Traffic light control:', traffic_light_control(num_cars, car_speeds))
    print('Optimize objective:', optimize_objective(num_cars, car_speeds))

## 4.3 智能交通安全监控的代码实例

```python
import cv2
import numpy as np

# 图像处理
def preprocess_image(image):
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    return blur

# 目标检测
def detect_objects(image):
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    _, thresh = cv2.threshold(gray, 127, 255, cv2.THRESH_BINARY)
    contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    return contours

# 事件识别
def recognize_events(contours):
    for contour in contours:
        if np.min(contour[:, :, 0]) < 50:
            print('Event detected: car')
        elif np.min(contour[:, :, 0]) < 150:
            print('Event detected: pedestrian')
        else:
            print('Event detected: obstacle')

if __name__ == '__main__':
    preprocessed_image = preprocess_image(image)
    contours = detect_objects(preprocessed_image)
    recognize_events(contours)
```

在这个代码实例中，我们首先通过图像处理（如灰度化、模糊化等）对输入的图像进行处理，然后通过目标检测（如边缘检测、阈值化等）检测出交通场景中的目标，最后根据目标特征识别出交通事件（如车辆、行人、障碍物等）。
