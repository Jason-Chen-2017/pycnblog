## 4.3 优化算法的代码实例

```python
import numpy as np
from scipy.optimize import minimize

# 定义目标函数
def objective_function(x):
    return x[0]**2 + x[1]**2 + x[2]**2 + x[3]**2

# 定义约束条件
def constraint1(x):
    return x[0] + x[1] + x[2] - 100

def constraint2(x):
    return x[0] - x[1]

# 定义约束条件的对偶函数
def constraint1_dual(lambda_):
    return lambda_ * constraint1(x)

def constraint2_dual(lambda_):
    return lambda_ * constraint2(x)

# 定义约束条件列表
constraints = [
    {"type": "eq", "fun": constraint1_dual},
    {"type": "ineq", "fun": constraint2_dual}
]

# 定义初始值
initial_guess = np.array([1, 1, 1, 1])

# 使用Sequential Least Squares Programming（SLSQP）算法解决优化问题
result = minimize(objective_function, initial_guess, method="SLSQP", constraints=constraints)

# 打印结果
print(result)
```
