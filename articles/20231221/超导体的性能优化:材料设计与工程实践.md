import matplotlib.pyplot as plt

# 定义超导体的电导性模型

    if T <= Tc:
        return np.inf
    else:
        return 0

# 定义超导体的电阻模型
def superconductor_resistance(R, Tc):
    return R * (1 - np.heaviside(1 - T / Tc))

# 定义超导体的性能指标
def superconductor_performance(R, T, Tc):
    resistance = superconductor_resistance(R, Tc)
    conductivity = superconductor_conductivity(T, Tc)
    return resistance / conductivity

# 测试超导体性能优化
Tc = 20
R = 10

performance = superconductor_performance(R, T, Tc)

plt.plot(T, performance)
plt.xlabel('Temperature')
plt.ylabel('Performance')
plt.title('Superconductor Performance Optimization')
