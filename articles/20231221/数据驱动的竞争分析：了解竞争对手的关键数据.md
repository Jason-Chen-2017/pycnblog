# 4.具体代码实例和详细解释说明
以下是一个简单的Python代码实例，展示了如何使用Scrapy、pandas、scikit-learn和matplotlib进行数据收集、清洗、分析和可视化：

```python
import scrapy
import pandas as pd
from sklearn.cluster import KMeans
from sklearn.linear_model import LinearRegression
import matplotlib.pyplot as plt

# 1.数据收集
class CompetitorSpider(scrapy.Spider):
    name = 'competitor'
    start_urls = ['http://www.competitor.com/sales']

    def parse(self, response):
        sales_data = []
        for sale in response.css('div.sale'):
            sales_data.append({
                'date': sale.css('span.date::text').extract_first(),
                'sales': int(sale.css('span.sales::text').extract_first().replace(',', ''))
            })
        return sales_data

# 2.数据清洗
def clean_data(sales_data):
    df = pd.DataFrame(sales_data)
    df = df.drop_duplicates()
    df = df.fillna(0)
    return df

# 3.数据分析
def analyze_data(df):
    kmeans = KMeans(n_clusters=3)
    kmeans.fit(df[['sales']])
    sales_pred = LinearRegression().fit(df[['sales']], df['sales'])
    return kmeans, sales_pred

# 4.策略优化
def optimize_strategy(kmeans, sales_pred, df):
    df['cluster'] = kmeans.labels_
    df['pred'] = sales_pred.predict(df[['sales']])
    plt.scatter(df['sales'], df['pred'])
    plt.xlabel('Actual Sales')
    plt.ylabel('Predicted Sales')
    plt.show()

# 主程序
if __name__ == '__main__':
    sales_data = CompetitorSpider.parse()
    df = clean_data(sales_data)
    kmeans, sales_pred = analyze_data(df)
    optimize_strategy(kmeans, sales_pred, df)
```
