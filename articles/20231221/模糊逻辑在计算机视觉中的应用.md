### 4.1Python实现图像的边缘检测

```python
import cv2
import numpy as np

def sobel_edge_detection(image):
    # 转换为灰度图像
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    # 高斯滤波
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    # 计算梯度
    sobelx = cv2.Sobel(blur, cv2.CV_64F, 1, 0, ksize=5)
    sobely = cv2.Sobel(blur, cv2.CV_64F, 0, 1, ksize=5)
    # 计算边缘强度
    magnitude = np.sqrt(sobelx ** 2 + sobely ** 2)
    # 非极大值抑制
    _, thresh = cv2.threshold(magnitude, 0.01 * 255, 255, cv2.THRESH_BINARY)
    # 连通性分析
    contours, hierarchy = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    # 绘制边缘线
    for cnt in contours:
        x, y, w, h = cv2.boundingRect(cnt)
        if w > 10 and h > 10:
            cv2.rectangle(image, (x, y), (x + w, y + h), (0, 255, 0), 2)
    return image

# 读取图像
# 进行边缘检测
result = sobel_edge_detection(image)
# 显示结果
cv2.imshow('Edge Detection', result)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

```python
import cv2
import numpy as np

def object_detection(image, model):
    # 转换为灰度图像
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    # 高斯滤波
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    # 特征提取
    features = model.detectMultiScale(blur, 1.1, 3)
    # 绘制目标框
    for (x, y, w, h) in features:
        cv2.rectangle(image, (x, y), (x + w, y + h), (0, 255, 0), 2)
    return image

# 加载模型
model = cv2.CascadeClassifier('haarcascade_frontalface_default.xml')
# 读取图像
# 进行目标识别
result = object_detection(image, model)
# 显示结果
cv2.imshow('Object Detection', result)
```
