import numpy as np

def dense_circle_approximation(x, y, h):
    """
    密集圆近似
    """
    x_step = x[1] - x[0]
    y_step = y[1] - y[0]
    h_step = h[1] - h[0]

    x_mid = (x[0] + x[1]) / 2
    y_mid = (y[0] + y[1]) / 2
    h_mid = (h[0] + h[1]) / 2

    x_diff = x_mid - x[0]
    y_diff = y_mid - y[0]
    h_diff = h_mid - h[0]

    a = (y_diff - h_diff * x_step) / x_step
    b = y[0] - a * x[0]

    return a, b

def curvature(x, y, h):
    """
    曲率计算
    """
    a, b = dense_circle_approximation(x, y, h)
    x_diff = x[1] - x[0]
    y_diff = y[1] - y[0]
    h_diff = h[1] - h[0]

    T = np.array([a, b])
    r = np.array([x_diff, y_diff, h_diff])

    norm_T = np.linalg.norm(T)
    norm_r = np.linalg.norm(r)

    kappa = norm_T / norm_r

    return kappa
