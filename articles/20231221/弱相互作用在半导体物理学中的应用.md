```python
import numpy as np
import matplotlib.pyplot as plt

    E = np.zeros(r.shape)
    for i in range(r.size):
        E[i] = e**2 / (4 * np.pi * eps0 * r[i])
    return E

plt.xlabel('Distance (m)')
plt.ylabel('Electric Field (V/m)')
plt.title('Electron Bubble Formation')
plt.show()
```
