from sklearn import svm

def entropy(p):
    return -np.sum(p * np.log2(p))

# 生物资源的可持续利用
def linear_programming(c, A, b):
    return np.linalg.solve(A, b)

# 生物技术的应用
def support_vector_machine(X, y, C):
    clf = svm.SVC(C=C)
    clf.fit(X, y)
    return clf.predict(X)
