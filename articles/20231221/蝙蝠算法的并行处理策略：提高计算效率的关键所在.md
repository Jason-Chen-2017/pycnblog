import random

def bat_algorithm(dim, max_iter, N, k):

    x = np.random.rand(N, dim)
    v = np.zeros((N, dim))
    p = np.random.rand(N, dim)
    best_x = x[np.argmin(f(x))]
    best_v = v[np.argmin(f(x))]

    # 主循环
    for t in range(max_iter):
        # 更新蝙蝠的速度和位置
        for i in range(N):
            v[i] = v[i] + random.uniform(-1, 1) * p[i]
            x[i] = x[i] + v[i]

        # 评估蝙蝠群的 FITNESS
        fitness = f(x)

        # 更新最好的解
        if np.min(fitness) < np.min(f(best_x)):
            best_x = x[np.argmin(fitness)]
            best_v = v[np.argmin(fitness)]

    return best_x, f(best_x)

def f(x):
    # 定义优化问题
    return np.sum(x**2)

N = 50
k = 5

best_x, best_f = bat_algorithm(dim, max_iter, N, k)
