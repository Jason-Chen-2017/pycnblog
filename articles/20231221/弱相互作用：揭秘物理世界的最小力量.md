```python
import numpy as np

def weak_force(distance, charge1, charge2):
    constant = 1.0 / (4 * np.pi * 1e-7)
    force = constant * charge1 * charge2 / distance**2
    return force

distance = 1e-10  # 距离，以米为单位
charge1 = 1.6e-19  # 电荷1，以科尔斯为单位
charge2 = 1.6e-19  # 电荷2，以科尔斯为单位

force = weak_force(distance, charge1, charge2)
print("弱相互作用的力：", force, "新托ン")
```
