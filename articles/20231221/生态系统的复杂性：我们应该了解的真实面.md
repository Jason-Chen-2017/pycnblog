```python
import numpy as np
import matplotlib.pyplot as plt

def generate_species(n, p):
    species = np.zeros(n)
    for i in range(n):
        if np.random.rand() < p:
            species[i] = 1
    return species

def simulate_ecosystem(n, p, T, t_steps):
    species = np.zeros((T, t_steps))
    for t in range(t_steps):
        for i in range(T):
            if np.random.rand() < p:
                species[i, t] = 1
    return species

def plot_ecosystem(species):
    plt.imshow(species, cmap='binary')
    plt.colorbar()
    plt.show()

n = 100
p = 0.1
T = 10
t_steps = 100

species = simulate_ecosystem(n, p, T, t_steps)
plot_ecosystem(species)
```

```python
import numpy as np
import matplotlib.pyplot as plt

def generate_environment(n, p):
    environment = np.zeros(n)
    for i in range(n):
        if np.random.rand() < p:
            environment[i] = 1
    return environment

def simulate_ecosystem_dynamics(n, p, T, t_steps):
    environment = generate_environment(n, p)
    species = np.zeros((T, t_steps))
    for t in range(t_steps):
        for i in range(n):
            if np.random.rand() < p:
                species[i, t] = 1
    return species, environment

def plot_ecosystem_dynamics(species, environment):
    plt.subplot(2, 1, 1)
    plt.imshow(species, cmap='binary')
    plt.colorbar()
    plt.title('Species')
    plt.subplot(2, 1, 2)
    plt.imshow(environment, cmap='binary')
    plt.colorbar()
    plt.title('Environment')
    plt.show()

n = 100
p = 0.1
T = 10
t_steps = 100

species, environment = simulate_ecosystem_dynamics(n, p, T, t_steps)
plot_ecosystem_dynamics(species, environment)
```
