class Polynomial:
    def __init__(self, coefficients):
        self.coefficients = coefficients

    def __add__(self, other):
        return Polynomial([a + b for a, b in zip(self.coefficients, other.coefficients)])

    def __mul__(self, c):
        return Polynomial([a * c for a in self.coefficients])

    def integrate(self, a, b):
        result = Polynomial([0] * len(self.coefficients))
        for i, a_i in enumerate(self.coefficients):
            result.coefficients[i] = (a_i / (i + 1)) * (b ** (i + 1)) - (a ** (i + 1)) * (a_i / (i + 1))
        return result

    def limit(self, x):
        return Polynomial([a * x ** n for a, n in zip(self.coefficients, range(len(self.coefficients)))] if a else 0)

    def __str__(self):
        return " + ".join([f"{a}x^{n}" for a, n in zip(self.coefficients, range(len(self.coefficients)))])


p1 = Polynomial([1, 2, 3])
p2 = Polynomial([4, 5, 6])
p3 = p1 + p2
p4 = p1 * 2
p5 = p1.integrate(1, 2)
p6 = p1.limit(3)

print(p3)  # 7 + 7x + 8x^2
print(p4)  # 2 + 4x + 6x^2
print(p5)  # 0.5x + 0.5x^2 + 0.5x^3
