```python
def calculate_trust(trust_list):
    trust_sum = 0
    for trust in trust_list:
        trust_sum += trust
    return trust_sum

def calculate_goal(goal_list):
    goal_sum = 0
    for goal in goal_list:
        goal_sum += goal
    return goal_sum

def calculate_responsibility(responsibility_list):
    responsibility_sum = 0
    for responsibility in responsibility_list:
        responsibility_sum += responsibility
    return responsibility_sum

trust_list = [1, 2, 3, 4, 5]
goal_list = [6, 7, 8, 9, 10]
responsibility_list = [11, 12, 13, 14, 15]

trust_result = calculate_trust(trust_list)

responsibility_result = calculate_responsibility(responsibility_list)

print("信任度:", trust_result)
print("目标:", goal_result)
print("责任:", responsibility_result)
```
