## 4.1 机器学习代码实例

我们将通过一个简单的逻辑回归算法来展示机器学习的代码实例和详细解释。

```python
import numpy as np

# 定义数据
X = np.array([[1, 2], [2, 3], [3, 4], [4, 5]])
y = np.array([0, 1, 0, 1])

# 定义权重和偏置
w = np.array([0, 0])
b = 0

# 定义损失函数
def loss(y_true, y_pred):
    return np.sum((y_true - y_pred) ** 2)

# 定义梯度下降算法
def gradient_descent(X, y, w, b, learning_rate, iterations):
    for _ in range(iterations):
        y_pred = np.dot(X, w) + b
        loss_value = loss(y, y_pred)
        gradient_w = np.dot(X.T, (y_pred - y)) / len(y)
        gradient_b = np.mean(y_pred - y)
        w -= learning_rate * gradient_w
        b -= learning_rate * gradient_b
    return w, b

# 训练算法
w, b = gradient_descent(X, y, w, b, learning_rate=0.01, iterations=1000)

# 预测
def predict(X, w, b):
    return np.dot(X, w) + b

# 测试
X_test = np.array([[5, 6]])
y_test = predict(X_test, w, b)
print(y_test)
```

在上面的代码实例中，我们首先定义了数据和标签，然后定义了权重和偏置，接着定义了损失函数和梯度下降算法。接着我们训练了算法，并使用训练好的权重和偏置来进行预测。

## 4.2 深度学习代码实例

我们将通过一个简单的神经网络来展示深度学习的代码实例和详细解释。

```python
import numpy as np

# 定义数据
X = np.array([[1, 2], [2, 3], [3, 4], [4, 5]])
y = np.array([0, 1, 0, 1])

# 定义神经网络
class NeuralNetwork:
    def __init__(self, input_size, hidden_size, output_size):
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size
        self.weights1 = np.random.randn(input_size, hidden_size)
        self.weights2 = np.random.randn(hidden_size, output_size)
        self.bias1 = np.zeros((1, hidden_size))
        self.bias2 = np.zeros((1, output_size))

    def forward(self, X):
        self.a1 = np.dot(X, self.weights1) + self.bias1
        self.z1 = np.dot(self.a1, self.weights2) + self.bias2
        self.y_pred = 1 / (1 + np.exp(-self.z1))
        return self.y_pred

# 训练算法
def train(X, y, model, learning_rate, iterations):
    for _ in range(iterations):
        y_pred = model.forward(X)
        loss_value = loss(y, y_pred)
        gradient_weights1 = np.dot(X.T, (y_pred - y)) / len(y)
        gradient_bias1 = np.mean(y_pred - y)
        model.weights1 -= learning_rate * gradient_weights1
        model.bias1 -= learning_rate * gradient_bias1
    return model

# 创建神经网络
model = NeuralNetwork(input_size=2, hidden_size=2, output_size=1)

# 训练算法
model = train(X, y, model, learning_rate=0.01, iterations=1000)

# 预测
X_test = np.array([[5, 6]])
y_test = model.forward(X_test)
print(y_test)
```

在上面的代码实例中，我们首先定义了数据和标签，然后定义了神经网络的结构。接着我们训练了算法，并使用训练好的权重和偏置来进行预测。
