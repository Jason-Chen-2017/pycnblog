    # 删除缺失值

    data['age'] = data['age'].astype(int)

    return data

# 清洗数据

### 3.3 数据挖掘

通过以下代码实现数据挖掘：

```python
from sklearn.cluster import KMeans

# 数据挖掘
def cluster_data(data):
    # 聚类
    kmeans = KMeans(n_clusters=3)
    data['cluster'] = kmeans.fit_predict(data[['age', 'income']])

    return data

# 数据挖掘
data = cluster_data(data)
```
