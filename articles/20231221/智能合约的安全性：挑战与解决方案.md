## 4.1 代码实例

我们将通过一个简单的智能合约实例来详细解释智能合约的实现过程。这个智能合约实例是一个简单的加密货币交易智能合约，它可以用于交易、借贷、投资等。

```solidity
pragma solidity ^0.5.0;

contract SimpleCryptoTrade {
    address public owner;
    uint public balance;
    mapping(address => uint) public debt;

    event Transfer(address indexed from, address indexed to, uint value);
    event Deposit(address indexed from, uint value);
    event Withdraw(address indexed from, uint value);

    constructor() public {
        owner = msg.sender;
    }

    function deposit() public payable {
        require(msg.value > 0);
        balance += msg.value;
        debt[msg.sender] += msg.value;
        emit Deposit(msg.sender, msg.value);
    }

    function withdraw() public {
        require(msg.sender == owner);
        require(msg.value <= balance);
        balance -= msg.value;
        debt[msg.sender] -= msg.value;
        emit Withdraw(msg.sender, msg.value);
    }

    function transfer(address to, uint value) public {
        require(msg.sender == owner);
        require(debt[msg.sender] >= value);
        debt[msg.sender] -= value;
        debt[to] += value;
        emit Transfer(msg.sender, to, value);
    }
}
```
