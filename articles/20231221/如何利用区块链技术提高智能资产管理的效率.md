2. 分布式：区块链技术通过多个节点来实现数据存储和传输，这些节点可以在不同的地理位置，可以是个人、企业、组织等。
3. 透明度和可追溯性：区块链技术通过将数据存储在区块中，并通过加密技术来保护数据的安全性，从而实现数据的透明度和可追溯性。
4. 不可篡改性：区块链技术通过使用加密技术来保护数据的完整性，从而实现数据的不可篡改性。
5. 高效和可扩展性：区块链技术通过使用去中心化、分布式、透明度和不可篡改性等特点，实现了高效和可扩展性的数据存储和传输。

## 4.2 创建一个简单的智能合约

创建一个简单的智能合约的具体代码实例如下：

```solidity
pragma solidity ^0.5.16;

contract SimpleToken {
    address public owner;
    mapping(address => uint256) public balanceOf;
    uint256 public totalSupply;

    event Transfer(address indexed from, address indexed to, uint256 value);

    constructor() public {
        owner = msg.sender;
        totalSupply = 0;
    }

    function transfer(address to, uint256 value) public {
        require(msg.sender != 0x0);
        require(value > 0);

        balanceOf[msg.sender] -= value;
        totalSupply -= value;
        balanceOf[to] += value;
        totalSupply += value;

        emit Transfer(msg.sender, to, value);
    }

    function balanceOf(address tokenOwner) public view returns (uint256 balance) {
        return balanceOf[tokenOwner];
    }

    function totalSupply() public view returns (uint256 total) {
        return totalSupply;
    }
}
```
