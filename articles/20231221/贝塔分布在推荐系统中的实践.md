    beta = len(ratings) - np.mean(ratings) + 1
    return alpha, beta

# 贝塔分布的参数估计
def estimate_parameters(ratings, alpha, beta):
    likelihood = 0
    for rating in ratings:
        likelihood += beta.logpdf(rating; a=alpha, b=beta)
    return np.argmax(likelihood)

# 推荐算法
def recommend(user, item, alpha, beta):
    trust = (alpha - 1) / (alpha + beta - 2) * item + (beta - 1) / (alpha + beta - 2)
    return trust

# 示例数据

estimated_alpha, estimated_beta = estimate_parameters(ratings, alpha, beta)
trust = recommend(user=0, item=2, alpha=estimated_alpha, beta=estimated_beta)
print("用户0对项目2的信任度为：", trust)
