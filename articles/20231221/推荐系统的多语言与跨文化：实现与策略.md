# 4.具体代码实例和详细解释说明
在这里，我们将提供一个简单的协同过滤算法的Python代码实例，并进行详细解释。

```python
import numpy as np
from scipy.spatial.distance import cosine

# 用户行为数据
user_behavior = {
    'user1': ['item1', 'item2', 'item3'],
    'user2': ['item3', 'item4', 'item5'],
    'user3': ['item1', 'item5', 'item6']
}

# 计算用户相似度
def user_similarity(user1, user2):
    common_items = set(user1).intersection(set(user2))
    if len(common_items) == 0:
        return 0
    return 1 - cosine(user1, user2)

# 推荐物品
def recommend_items(user, items, similarity_matrix):
    similar_users = []
    for i, user in enumerate(items):
        if i == user:
            continue
        similarity = similarity_matrix[user][i]
        if similarity > 0:
            similar_users.append(i)
    return similar_users

# 主函数
def main():
    # 构建用户相似度矩阵
    similarity_matrix = np.zeros((3, 3))
    for i, user1 in enumerate(user_behavior.values()):
        for j, user2 in enumerate(user_behavior.values()):
            if i == j:
                continue
            similarity = user_similarity(user1, user2)
            similarity_matrix[i][j] = similarity
            similarity_matrix[j][i] = similarity

    # 推荐物品
    user = 'user1'
    items = user_behavior.keys()
    items.remove(user)
    recommended_items = recommend_items(user, items, similarity_matrix)
    print(f'为 {user} 推荐的物品：{recommended_items}')

if __name__ == '__main__':
    main()
```

在这个代码实例中，我们首先定义了用户行为数据，并计算了用户之间的相似度。然后，我们定义了一个推荐物品的函数，它根据用户的相似度来推荐物品。最后，我们调用主函数来实现推荐系统的功能。
