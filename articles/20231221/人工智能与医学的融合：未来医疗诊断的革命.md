### 4.1.1 图像分割实现

```python
import cv2
import numpy as np

# 读取医学图像

# 使用Canny边缘检测算法检测边缘
edges = cv2.Canny(image, 100, 200)

# 使用霍夫变换找到边缘的交点
lines = cv2.HoughLines(edges, 1, np.pi / 180, 200)

# 绘制边缘线条
for line in lines:
    rho, theta = line[0]
    a = np.cos(theta)
    b = np.sin(theta)
    x0 = a * rho
    y0 = b * rho
    x1 = int(x0 + 1000 * (-b))
    y1 = int(y0 + 1000 * (a))
    x2 = int(x0 - 1000 * (-b))
    y2 = int(y0 - 1000 * (a))
    cv2.line(image, (x1, y1), (x2, y2), (0, 255, 0), 1)

# 显示结果
cv2.imshow('Edge Map', image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
