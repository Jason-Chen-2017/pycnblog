# 4.具体代码实例和详细解释说明
## 4.1 数据清洗
```
CREATE TABLE customer (
  id INT,
  name STRING,
  age INT,
  gender STRING,
  city STRING
);

INSERT INTO TABLE customer
VALUES (1, 'John', 30, 'M', 'New York'),
       (2, 'Jane', 25, 'F', 'Los Angeles'),
       (3, 'Mike', 28, 'M', 'Chicago'),
       (4, 'Lisa', 22, 'F', 'Houston'),
       (5, 'Tom', 35, 'M', 'Phoenix');

SELECT AVG(age) AS average_age,
       MEDIAN(age) AS median_age,
       STDDEV(age) AS standard_deviation_age,
       VARIANCE(age) AS variance_age
FROM customer;
```
## 4.2 数据聚合
```
CREATE TABLE order (
  id INT,
  customer_id INT,
  order_date STRING,
  total_amount DECIMAL(10, 2)
);

INSERT INTO TABLE order
VALUES (1, 1, '2021-01-01', 100.00),
       (2, 2, '2021-01-02', 150.00),
       (3, 3, '2021-01-03', 200.00),
       (4, 4, '2021-01-04', 250.00),
       (5, 5, '2021-01-05', 300.00);

SELECT customer_id,
       COUNT(*) AS order_count,
       SUM(total_amount) AS total_amount
FROM order
GROUP BY customer_id;
```
## 4.3 数据挖掘
```
CREATE TABLE product (
  id INT,
  name STRING,
  category STRING,
  price DECIMAL(10, 2)
);

INSERT INTO TABLE product
VALUES (1, 'Laptop', 'Electronics', 999.99),
       (2, 'Smartphone', 'Electronics', 799.99),
       (3, 'TV', 'Electronics', 699.99),
       (4, 'Refrigerator', 'Appliance', 1999.99),
       (5, 'Washing Machine', 'Appliance', 1299.99);

SELECT category,
       COUNT(*) AS product_count,
       SUM(price) AS total_price
FROM product
GROUP BY category;
```
## 4.4 预测分析
```
CREATE TABLE sales (
  id INT,
  product_id INT,
  month STRING,
  sales_amount DECIMAL(10, 2)
);

INSERT INTO TABLE sales
VALUES (1, 1, '2021-01', 100.00),
       (2, 1, '2021-02', 150.00),
       (3, 1, '2021-03', 200.00),
       (4, 2, '2021-01', 250.00),
       (5, 2, '2021-02', 300.00);

SELECT product_id,
       month,
       sales_amount
FROM sales
WHERE month = '2021-01'
ORDER BY sales_amount DESC;
```
