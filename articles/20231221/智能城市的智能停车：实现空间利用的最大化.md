```python
import threading

def park_car(parking_space):
    # 处理停车空间
    print(f"车辆停放在空闲空间{parking_space}")

def find_free_parking_space():
    # 获取所有停车空间
    parking_spaces = [1, 2, 3, 4, 5]
    # 遍历所有停车空间
    for parking_space in parking_spaces:
        # 如果停车空间为空闲，则启动一个线程来处理车辆停放
        if parking_space == 0:
            t = threading.Thread(target=park_car, args=(parking_space,))
```
