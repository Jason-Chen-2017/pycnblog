import matplotlib.pyplot as plt

# 准粒子的波函数
def wave_function(x, y):
    return np.exp(-(x**2 + y**2) / 2)

# 准粒子的势能
def potential_energy(x, y):
    return -x**2 * y**2

# 准粒子的守恒定律
def conservation_law(x, y):
    return x**2 + y**2

# 计算电子在显示屏中的行为
def calculate_electron_behavior(x, y):
    wave_function_value = wave_function(x, y)
    potential_energy_value = potential_energy(x, y)
    conservation_law_value = conservation_law(x, y)
    return wave_function_value, potential_energy_value, conservation_law_value

# 绘制电子在显示屏中的行为

    wave_function_value, potential_energy_value, conservation_law_value = calculate_electron_behavior(x, y)
    plt.plot(x, wave_function_value, label='波函数')
    plt.plot(x, potential_energy_value, label='势能')
    plt.plot(x, conservation_law_value, label='守恒定律')
