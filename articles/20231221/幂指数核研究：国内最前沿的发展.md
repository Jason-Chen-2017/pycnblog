from scipy.optimize import curve_fit

y = np.array([2, 4, 6, 8, 10])

# 定义数学模型

    return a * x + b

# 使用scipy库中的curve_fit函数进行优化
popt, pcov = curve_fit(model, x, y)

# 输出优化结果
