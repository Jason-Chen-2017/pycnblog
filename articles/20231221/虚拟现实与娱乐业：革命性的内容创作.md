### 4.2 沉浸式交互代码实例

以下是一个简单的沉浸式交互代码实例：

```python
import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# 定义头部跟踪的函数
def head_tracking(x, y, z):
    # 计算头部的方向向量
    direction = np.array([x, y, z])
    # 计算头部的速度
    speed = np.linalg.norm(direction)
    # 更新头部的位置
    new_position = np.array([x, y, z])
    return new_position

# 创建三维图形
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# 绘制立方体
cube = ax.plot_trisurf(np.linspace(-1, 1, 10), np.linspace(-1, 1, 10), np.linspace(-1, 1, 10), cmap='viridis')

# 设置坐标轴范围
ax.set_xlim([-1, 1])
ax.set_ylim([-1, 1])
ax.set_zlim([-1, 1])

# 开始沉浸式交互
while True:
    # 获取用户的输入
    x, y, z = map(float, input('Enter x, y, z: ').split())
    # 更新头部的位置
    new_position = head_tracking(x, y, z)
    # 更新立方体的位置
    cube.set_poses([(new_position, [0, 0, 0, 1])])
    # 刷新图形
    plt.pause(0.1)

# 显示图形
plt.show()
```
