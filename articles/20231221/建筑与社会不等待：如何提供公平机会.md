import numpy as np
from scipy.optimize import minimize

# 定义目标函数
def objective_function(x):
    # 计算成本
    cost = np.sum(x**2)
    # 计算可持续性
    sustainability = np.sum(x)
    # 返回总评分
    return cost + sustainability

# 优化建筑设计
result = minimize(objective_function, x0=np.zeros(10), method='nelder-mead')

x_optimal = result.x
