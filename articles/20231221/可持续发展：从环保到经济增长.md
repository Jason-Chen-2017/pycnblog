def GDP(C, r, T):

    return np.sum([Y[t] / (1 + f) ** t for t in range(1, T + 1)])

def Innovation(Q, h, T):
    return np.sum([Q[t] / (1 + h) ** t for t in range(1, T + 1)])

    return np.sum([U[t] / (1 + l) ** t for t in range(1, T + 1)])
