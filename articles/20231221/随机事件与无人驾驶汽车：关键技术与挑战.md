# 数据分割和特征提取
rows, cols = blur.shape
objects = []
for i in range(0, rows, 20):
    for j in range(0, cols, 20):
        object = blur[i:i+20, j:j+20]
        objects.append(object)

# 对象检测和跟踪
for object in objects:
    # 对象检测
    contours, _ = cv2.findContours(object, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    # 对象跟踪
    for contour in contours:
        # 计算对象的中心点
        M = cv2.moments(contour)
        cX = int(M['m10'] / M['m00'])
        cY = int(M['m01'] / M['m00'])
        cv2.circle(img, (cX, cY), 5, (255, 0, 0), 2)

# 数据融合
result = cv2.addWeighted(img, 0.7, blur, 0.3, 0)
