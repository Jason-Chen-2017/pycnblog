import numpy as np

def dynamic_planning(dist, speed, time):
    """
    动态规划算法实现路径规划和控制
    
    :param dist: 距离列表
    :param speed: 速度列表
    :param time: 时间列表
    :return: 最佳路径和速度
    """
    dp = np.zeros((len(dist), len(time)))
    for i in range(len(dist)):
        for j in range(len(time)):
            if i == 0:
                dp[i][j] = dist[i] + speed[i] * time[j]
            elif j == 0:
                dp[i][j] = dp[i - 1][j]
            else:
                dp[i][j] = min(dp[i - 1][j], dp[i][j - 1]) + dist[i] + speed[i] * time[j]
    return dp
