### 4.3 案例实现

#### 4.3.1 数据清洗

```python
import pandas as pd

# 读取数据
order_df = pd.read_csv('order.csv')
inventory_df = pd.read_csv('inventory.csv')
vehicle_df = pd.read_csv('vehicle.csv')

# 去重
order_df = order_df.drop_duplicates()
inventory_df = inventory_df.drop_duplicates()
vehicle_df = vehicle_df.drop_duplicates()

# 合并
order_inventory_df = pd.merge(order_df, inventory_df, on='order_id')

# 分割
order_df = order_df[order_df['order_type'] == 'domestic']
inventory_df = inventory_df[inventory_df['warehouse'] == 'beijing']
vehicle_df = vehicle_df[vehicle_df['vehicle_type'] == 'truck']
```

#### 4.3.2 数据转换

```python
# 格式的统一
order_df['order_time'] = order_df['order_time'].str.strip()
inventory_df['expiration_date'] = inventory_df['expiration_date'].str.strip()
vehicle_df['vehicle_number'] = vehicle_df['vehicle_number'].str.strip()

# 类型转换
order_df['order_type'] = order_df['order_type'].astype('str')
inventory_df['warehouse'] = inventory_df['warehouse'].astype('str')
vehicle_df['vehicle_type'] = vehicle_df['vehicle_type'].astype('str')

# 单位的转换
order_df['weight'] = order_df['weight'].astype('float')
inventory_df['quantity'] = inventory_df['quantity'].astype('int')
vehicle_df['load_capacity'] = vehicle_df['load_capacity'].astype('int')
```

#### 4.3.3 数据整合

```python
# 连接
result_df = pd.concat([order_df, inventory_df, vehicle_df], axis=1)

# 内连接
result_df = pd.merge(order_df, inventory_df, on='order_id')

# 左连接
result_df = pd.merge(order_df, inventory_df, left_on='order_id', right_on='order_id')

# 右连接
result_df = pd.merge(order_df, inventory_df, left_on='order_id', right_on='order_id', how='right')

# 全连接
result_df = pd.merge(order_df, inventory_df, left_on='order_id', right_on='order_id', how='outer')

# 交叉连接
result_df = pd.merge(order_df, inventory_df, on='order_id', how='cross')
```
