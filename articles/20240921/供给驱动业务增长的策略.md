### 5.2 源代码详细实现

以下为供给驱动策略的 Python 源代码实现：

```python
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_squared_error
import matplotlib.pyplot as plt

# 5.2.1 数据采集与预处理
# 导入数据
data = pd.read_csv('data.csv')
data.head()

# 预处理数据
data['date'] = pd.to_datetime(data['date'])
data.set_index('date', inplace=True)
data.head()

# 提取特征
data['month'] = data.index.month
data['day_of_week'] = data.index.dayofweek
data.head()

# 5.2.2 模型构建与训练
# 划分训练集和测试集
train_data, test_data = train_test_split(data, test_size=0.2, shuffle=False)

# 构建模型
model = RandomForestRegressor(n_estimators=100)

# 训练模型
model.fit(train_data[['month', 'day_of_week']], train_data['sales'])

# 5.2.3 代码解读与分析
# 模型预测
predictions = model.predict(test_data[['month', 'day_of_week']])

# 评估模型效果
mse = mean_squared_error(test_data['sales'], predictions)
print('MSE:', mse)

# 5.2.4 运行结果展示
plt.figure(figsize=(10, 6))
plt.plot(test_data.index, test_data['sales'], label='Actual')
plt.plot(test_data.index, predictions, label='Predicted')
plt.legend()
plt.show()
```
