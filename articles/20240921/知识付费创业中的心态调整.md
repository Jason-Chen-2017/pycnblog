```python
# 定义心态调整函数
def adjust_mentality(self_cognition, psychological_adjustment, skill_improvement):
    return self_cognition * psychological_adjustment * skill_improvement

# 初始化参数

plt.xlabel('阶段')
plt.ylabel('调整结果')
plt.legend()
plt.show()

# 输出心态调整结果
print(f'心态调整结果：{mentality}')
```
