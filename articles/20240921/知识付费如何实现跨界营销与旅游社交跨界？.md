from sklearn.linear_model import LinearRegression

content_model = LinearRegression().fit(user_interest, user_behavior)

user_demand = demand_model.predict(user_behavior)

collaborated_content = collaboration_model.predict(user_behavior)

print("推荐内容：", recommended_content)
print("跨界合作内容：", collaborated_content)
