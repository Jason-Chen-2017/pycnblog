```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import MinMaxScaler
from sklearn.linear_model import LinearRegression
from sklearn.metrics import accuracy_score

# 读取数据
def read_data(data_file):
    data = []
    with open(data_file, 'r') as f:
        for line in f:
            data.append([float(x) for x in line.strip().split(',')])
    return data

# 预处理数据
def preprocess_data(data):
    scaler = MinMaxScaler()
    data = scaler.fit_transform(data)
    return data

# 分割训练集和测试集
def split_data(data):
    train_size = int(data.shape[0] * 0.8)
    test_size = len(data) - train_size
    return train_data, test_data

# 特征提取
def extract_features(data):
    features = []
    for i in range(data.shape[1]):
        feature = []
        for j in range(data.shape[0]):
            feature.append(data.iloc[j, i])
        features.append(feature)
    return features

# 训练模型
def train_model(data, features):
    model = LinearRegression()
    model.fit(features, data)
    return model

# 测试模型
def test_model(data, model):
    predictions = model.predict(features)
    true_values = data
    return predictions, true_values

# 计算模型性能指标
def calculate_accuracy(predictions, true_values):
    return accuracy_score(true_values, predictions)

# 主程序
if __name__ == '__main__':
    # 读取数据
    data_file = 'energy_data.csv'
    data = read_data(data_file)
    
    # 预处理数据
    preprocessed_data = preprocess_data(data)
    
    # 分割训练集和测试集
    train_data, test_data = split_data(preprocessed_data)
    
    # 特征提取
    features = extract_features(train_data)
    
    # 训练模型
    model = train_model(train_data, features)
    
    # 测试模型
    predictions, true_values = test_model(test_data, model)
    
    # 计算模型性能指标
    accuracy = calculate_accuracy(predictions, true_values)
    print('Accuracy:', accuracy)
```
