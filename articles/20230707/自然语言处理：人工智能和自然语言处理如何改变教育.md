import numpy as np
import pandas as pd
import re
import jieba
import spaCy

# 数据预处理
def preprocess(text):
    # 去除标点符号
    text = re.sub(r'\W+', '', text)
    # 去除数字
    text = re.sub(r'\d+', '', text)
    # 去除特殊字符
    text = re.sub(r'[\W]','', text)
    # 分词
    tokens = jieba.cut(text)
    # 去停用词
    tokens = [t for t in tokens if t not in ['是', '否', '的', '我', '你', '他', '她']]
    return''.join(tokens)

# 文本分类
def text_classify(text):
    # 加载预训练模型
    model = 'nltk.sentiment.vader'
    clf = TextClassifier(model)
    # 分析文本
    score = clf.polarity_scores(text)
    # 返回分类结果
    return clf.classify(text)

# 情感分析
def text_sentiment(text):
    # 加载预训练模型
    model = 'textBlob'
    clf = TextSentimentAnalyzer(model)
    # 分析文本
    score = clf.polarity_scores(text)
    # 返回情感结果
    return clf.sentiment.polarity
