import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import jupyter
from jupyter.notebook import Notebook
from PIL import Image
import h5py
import os

# 导入所需库
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import MinMaxScaler
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error
import numpy as np

# 读取数据
df = pd.read_csv('user_data.csv')

# 去噪处理
df = df[['userID', 'itemID', 'buy_time']].dropna()
df = df[['userID', 'itemID', 'buy_time']].dropna().values

# 预处理数据
X = df.drop(['userID', 'itemID', 'buy_time'], axis=1)
y = df['buy_price']
scaler = MinMaxScaler(feature_range=(0, 1))
X = scaler.fit_transform(X)

# 使用线性回归模型进行预测
model = LinearRegression()
model.fit(X, y)

# 预测
df['pred_buy_price'] = model.predict(X)

# 评估模型
mse = mean_squared_error(df['buy_price'], df['pred_buy_price'])
print('Mean Squared Error:', mse)

# 绘制数据
df.plot.scatter(x='userID', y='itemID', c=df.buy_price)
df.plot.scatter(x='userID', y='pred_buy_price', c=df.pred_buy_price)
