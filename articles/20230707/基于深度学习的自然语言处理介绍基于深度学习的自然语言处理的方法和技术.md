import torch
import torch.nn as nn
import torch.optim as optim

class NewsClassifier(nn.Module):
    def __init__(self):
        super(NewsClassifier, self).__init__()
        self.fc1 = nn.Linear(200, 128) # 200 个词的词袋模型
        self.fc2 = nn.Linear(128, 64) # 64 个词的词袋模型
        self.fc3 = nn.Linear(64, 1) # 1 个词的分类模型
        self.relu = nn.ReLU(inplace=True) # 返回输入的相对百分比

    def forward(self, x):
        x = self.relu(self.fc1(x))
        x = self.relu(self.fc2(x))
        x = self.relu(self.fc3(x))
        return x
