def preprocess_audio(audio_data):
    # 去除噪音
    noise = np.random.normal(0, 0.01, (int(audio_data.shape[0]), audio_data.shape[1]))
    audio_data = audio_data - noise
    # 对数据进行增强
    amplitude = 1 + 0.1 * np.random.normal()
    freq_range = 50000 * np.array([1, 2], dtype=audio_data.dtype)
    sharpness = 0.1 * np.random.normal()
    audio_data = audio_data + amplification * np.maximum(np.abs(freq_range), 0) + sharpness
    return audio_data

# 2. 特征提取

def extract_features(audio_data):
    # 1. 谱分析
    fft = np.fft.fft(audio_data)
    fft_power = np.fft.fftshift(fft)
    rms = np.sqrt(np.mean(np.abs(fft_power)))
    audio_features = np.abs(fft_power) / rms
    # 2. 语音自相关
    cov_matrix = np.cov(audio_data.T)
    cov_inv = np.linalg.inv(cov_matrix)
    audio_features = np.dot(cov_inv, audio_features.T)
    return audio_features
