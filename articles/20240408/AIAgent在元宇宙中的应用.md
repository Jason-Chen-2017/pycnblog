        engine="text-davinci-002",
        prompt=prompt,
        max_tokens=2048,
        n=1,
        stop=None,
        temperature=0.7,
    )
    return response.choices[0].text.strip()

def chat_with_agent():
    """
    与AIAgent虚拟助手进行对话
    """
    print("你好,我是AIAgent虚拟助手。很高兴为您服务,请问有什么我可以帮到您的吗?")
    
    while True:
        user_input = input("用户: ")
        if user_input.lower() == "exit":
            print("AIAgent: 非常感谢您的使用,再见!")
            break
        
        agent_response = generate_response(user_input)
        print("AIAgent:", agent_response)

    chat_with_agent()
