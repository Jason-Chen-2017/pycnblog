import numpy as np

def cos_similarity(x, y):
    """计算两个向量的余弦相似度"""
    dot_product = np.dot(x, y)
    norm_x = np.linalg.norm(x)
    norm_y = np.linalg.norm(y)
    return dot_product / (norm_x * norm_y)
