    audio = r.record(source)

# 使用AI大模型进行语音识别

# 根据文本分析情感状态

    emotion = '高兴'
elif '愤怒' in text:
    emotion = '愤怒'
elif '焦虑' in text:
    emotion = '焦虑'

    emotion = '其他'

# 输出情感状态
