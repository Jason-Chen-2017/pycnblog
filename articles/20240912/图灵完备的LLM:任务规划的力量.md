**源代码实例：**（Python代码）

```python
import openai

def plan_task(task_description):
    response = openai.Completion.create(
        engine="text-davinci-002",
        prompt=f"请根据以下任务描述生成任务规划：{task_description}",
        max_tokens=50
    )
    return response.choices[0].text.strip()

task_description = "客人需要预订一个座位，时间为明天晚上7点，餐厅要求是中餐，口味偏辣。"
task_plan = plan_task(task_description)
print(task_plan)
```
