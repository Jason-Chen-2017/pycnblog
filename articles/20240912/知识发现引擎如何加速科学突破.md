```python
import networkx as nx

def build_knowledge_graph(entities, relations):
    G = nx.Graph()
    
    for entity in entities:
        G.add_node(entity)
        
    for relation in relations:
        G.add_edge(relation[0], relation[1], relation=relation[2])
        
    return G

# 示例
```

```python
import networkx as nx
import spacy

nlp = spacy.load("en_core_web_sm")

def answer_question(graph, question):
    doc = nlp(question)
    query = []

    for token in doc:
        if token.pos_ in ["NOUN", "VERB", "ADP"]:
            query.append(token.text)

    for node in graph.nodes:
        if node in query:
            query.remove(node)

    for edge in graph.edges:
        if edge[2]["relation"] in query:
            query.remove(edge[2]["relation"])

    return query
```

```python
def expand_knowledge_graph(graph, entity, relations):
    graph.add_node(entity)

    for relation in relations:
        if relation[1] not in graph.nodes:
            graph.add_node(relation[1])
        graph.add_edge(entity, relation[1], relation=relation[2])

    return graph

# 示例
```

#### 4. 实现一个基于知识图谱的推荐系统。

```python
from sklearn.metrics.pairwise import cosine_similarity

def recommend_entities(graph, user_history, k=5):
    entity_vectors = []

    for entity in graph.nodes:
        vector = [0] * (len(graph.nodes) + 1)
        vector[graph.nodes.index(entity)] = 1
        entity_vectors.append(vector)

    user_vector = [0] * (len(graph.nodes) + 1)
    for entity in user_history:
        user_vector[graph.nodes.index(entity)] = 1

    similarity_matrix = cosine_similarity(entity_vectors, [user_vector])

    recommended_entities = sorted(range(len(similarity_matrix[0])), key=lambda i: similarity_matrix[0][i], reverse=True)[:k]

    return [graph.nodes[i] for i in recommended_entities]

# 示例

recommended_entities = recommend_entities(G, user_history)
print(recommended_entities)
```

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.model_selection import train_test_split
from sklearn.naive_bayes import MultinomialNB

def classify_text(graph, documents, labels, test_size=0.2):
    entity_vectors = []

    for entity in graph.nodes:
        vector = [0] * (len(graph.nodes) + 1)
        vector[graph.nodes.index(entity)] = 1
        entity_vectors.append(vector)

    X_train, X_test, y_train, y_test = train_test_split(entity_vectors, labels, test_size=test_size)

    vectorizer = TfidfVectorizer()
    X_train_tfidf = vectorizer.fit_transform(X_train)
    X_test_tfidf = vectorizer.transform(X_test)

    classifier = MultinomialNB()
    classifier.fit(X_train_tfidf, y_train)
    accuracy = classifier.score(X_test_tfidf, y_test)

    return accuracy

# 示例
documents = ["这是一本关于城市的书。", "这本书是关于作者的。"]
labels = ["城市", "作者"]
accuracy = classify_text(G, documents, labels)
print(accuracy)
```
