import numpy as np

def linear_regression(x, y):
    # 计算斜率和截距
    m = np.mean(x)
    b = np.mean(y - m*x)
    return m, b

# 测试数据
x = np.array([1, 2, 3, 4, 5])
y = np.array([2, 4, 5, 4, 5])

# 训练模型
m, b = linear_regression(x, y)

y_pred = m*x_new + b
print("预测值：", y_pred)
