**1. 编写一个简单的基于协同过滤的推荐算法。**

```python
# 基于用户-物品评分矩阵的协同过滤算法实现

def collaborative_filtering(train_data, user_id, k=5):
    # 计算用户相似度矩阵
    sim_matrix = cosine_similarity(train_data)

    # 计算用户与邻居的评分预测
    user_neighbors = sim_matrix[user_id]
    user_neighbors = sorted(user_neighbors, key=lambda x: x[1], reverse=True)[:k]
    predictions = []
    for neighbor, weight in user_neighbors:
        for item, rating in train_data[neighbor].items():
            predictions.append((item, rating * weight))
    return predictions
```

```python
# 基于物品属性相似度的内容推荐算法实现

def content_based_recommendation(train_data, item_id, k=5):
    # 计算物品属性相似度矩阵
    sim_matrix = jaccard_similarity(train_data)

    # 计算用户喜欢的物品与邻居的相似度
    item_neighbors = sim_matrix[item_id]
    item_neighbors = sorted(item_neighbors, key=lambda x: x[1], reverse=True)[:k]
    recommendations = []
    for neighbor, weight in item_neighbors:
        if neighbor not in train_data[item_id]:
            recommendations.append((neighbor, weight))
    return recommendations
```
