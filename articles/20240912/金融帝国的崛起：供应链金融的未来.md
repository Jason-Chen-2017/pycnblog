class OrderFundingSystem:
    def __init__(self):
        self.orders = {}
        self.loans = {}

    def submit_order(self, order):
        order_id = len(self.orders) + 1
        self.orders[order_id] = order
        return order_id

    def approve_order(self, order_id):
        if order_id in self.orders:
            self.orders[order_id]['status'] = 'approved'
        else:
            print("Invalid order ID")

    def disburse_funds(self, order_id):
        if order_id in self.orders and self.orders[order_id]['status'] == 'approved':
            self.loans[order_id] = self.orders[order_id]['amount']
            print(f"Funds disbursed for order {order_id}")
        else:
            print("Order not approved or does not exist")

    def repay_loan(self, loan_id, amount):
        if loan_id in self.loans and amount <= self.loans[loan_id]:
            self.loans[loan_id] -= amount
            if self.loans[loan_id] == 0:
                print(f"Loan {loan_id} repaid")
            else:
                print(f"Partial repayment for loan {loan_id}")
        else:
            print("Invalid loan ID or amount")

class RiskControlSystem:

        }
