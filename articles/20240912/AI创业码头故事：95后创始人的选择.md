```python
# 决策树算法实现示例

class DecisionTree:
    def __init__(self, max_depth=None):
        self.max_depth = max_depth

    def fit(self, X, y):
        # 构建决策树
        self.tree_ = self._build_tree(X, y)

    def _build_tree(self, X, y, depth=0):
        # 内部递归构建决策树
        if depth >= self.max_depth:
            return self._predict_leaves(y)
        
        # 选择最优特征和阈值
        best_split = self._find_best_split(X, y)
        if best_split is None:
            return self._predict_leaves(y)
        
        # 切分数据集
        left_idxs, right_idxs = self._split(X[:, best_split['feat']], best_split['thresh'])
        left_subtree = self._build_tree(X[left_idxs], y[left_idxs], depth+1)
        right_subtree = self._build_tree(X[right_idxs], y[right_idxs], depth+1)
        
        # 返回决策树的叶子节点
        return {'feat': best_split['feat'], 'thresh': best_split['thresh'], 'left': left_subtree, 'right': right_subtree}
    
    def _predict_leaves(self, y):
        # 预测每个样本的类别
        return [y[0]] * len(y)
    
    def _find_best_split(self, X, y):
        # 找到最佳特征和阈值
        best_split = None
        for feat_idx in range(X.shape[1]):
            unique_vals = np.unique(X[:, feat_idx])
            for val in unique_vals:
                left_idxs = X[:, feat_idx] < val
                right_idxs = X[:, feat_idx] >= val
                if left_idxs.any() and right_idxs.any():
                    gini_left = 1 - np.sum(left_idxs * y[left_idxs]) / np.sum(left_idxs)
                    gini_right = 1 - np.sum(right_idxs * y[right_idxs]) / np.sum(right_idxs)
                    gini = gini_left * np.sum(left_idxs) / len(y) + gini_right * np.sum(right_idxs) / len(y)
                    if best_split is None or gini < best_split['gini']:
                        best_split = {'feat': feat_idx, 'thresh': val, 'gini': gini}
        return best_split

    def _split(self, X, thresh):
        # 切分数据集
        left_idxs = X < thresh
        right_idxs = X >= thresh
        return left_idxs, right_idxs
    
    def predict(self, X):
        # 预测样本的类别
        return [self._predict(x) for x in X]
    
    def _predict(self, x):
        # 递归预测类别
        if not isinstance(self.tree_, dict):
            return self.tree_
        if x[self.tree_['feat']] <= self.tree_['thresh']:
            return self._predict(self.tree_['left'])
        else:
            return self._predict(self.tree_['right'])
```

**答案解析：** 该示例实现了决策树的基本算法，包括选择最佳分割、构建树结构和预测类别。

#### 7. 请实现一个简单的K-均值聚类算法。
```python
import numpy as np

def k_means_clustering(X, k, max_iters=100, tolerance=1e-4):
    # 初始化中心点
    centroids = X[np.random.choice(X.shape[0], k, replace=False)]
    
    for i in range(max_iters):
        # 计算每个样本到各个中心点的距离
        distances = np.linalg.norm(X[:, np.newaxis] - centroids, axis=2)
        
        # 分配样本到最近的中心点
        clusters = np.argmin(distances, axis=1)
        
        # 更新中心点
        new_centroids = np.array([X[clusters == j].mean(axis=0) for j in range(k)])
        
        # 检查收敛
        if np.linalg.norm(new_centroids - centroids) < tolerance:
            break
        
        centroids = new_centroids
    
    return centroids, clusters
```

**答案解析：** 该示例实现了K-均值聚类的基本算法，包括初始化中心点、分配样本和更新中心点。
