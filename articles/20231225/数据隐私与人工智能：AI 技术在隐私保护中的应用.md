## 4.1 数据脱敏

### 4.1.1 代码实例

```python
import random
import re

def anonymize_name(name):
    if re.match(r'^[A-Za-z]+$', name):
        return 'XX' + str(random.randint(1000, 9999))
    else:
        return name

def anonymize_id_card(id_card):
    if re.match(r'^\d{15}$', id_card) or re.match(r'^\d{17}$', id_card):
        return ''.join(random.sample('0123456789', 18))
    else:
        return id_card

def anonymize_phone_number(phone_number):
    if re.match(r'^\d{11}$', phone_number):
        return ''.join(random.sample('0123456789', 11))
    else:
        return phone_number

def anonymize_data(data):
    anonymized_data = {}
    for key, value in data.items():
        if key == 'name':
            anonymized_data[key] = anonymize_name(value)
        elif key == 'id_card':
            anonymized_data[key] = anonymize_id_card(value)
        elif key == 'phone_number':
            anonymized_data[key] = anonymize_phone_number(value)
        else:
            anonymized_data[key] = value
    return anonymized_data

data = {
    'name': 'John Doe',
    'id_card': '440322199001012301',
    'phone_number': '13911112222',
    'age': 30,
    'gender': 'male'
}

anonymized_data = anonymize_data(data)
print(anonymized_data)
```

### 4.1.2 详细解释说明

在上述代码中，我们首先导入了 `random` 和 `re` 模块，用于生成随机数据和正则表达式匹配。然后定义了四个脱敏函数，分别用于脱敏姓名、身份证号码、手机号码和其他数据。在脱敏函数中，我们使用了正则表达式来匹配需要脱敏的数据类型。然后定义了一个 `anonymize_data` 函数，用于将原始数据脱敏后返回。最后，我们测试了这个函数，将原始数据脱敏后打印出来。

def mask_data(data):
    masked_data = {}
    for key, value in data.items():
        if key == 'name':
            masked_data[key] = 'XX' + ''.join(random.sample(string.ascii_lowercase, 4))

## 4.3 加密

### 4.3.1 代码实例

```python
from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes
from Crypto.Util.Padding import pad, unpad

def encrypt_data(data, key):
    cipher = AES.new(key, AES.MODE_ECB)
    encrypted_data = cipher.encrypt(pad(data.encode('utf-8'), AES.block_size))
    return encrypted_data

def decrypt_data(encrypted_data, key):
    cipher = AES.new(key, AES.MODE_ECB)
    decrypted_data = unpad(cipher.decrypt(encrypted_data), AES.block_size).decode('utf-8')
    return decrypted_data

key = get_random_bytes(16)
data = {
    'name': 'John Doe',
    'id_card': '440322199001012301',
    'phone_number': '13911112222',
    'age': 30,
    'gender': 'male'
}

encrypted_data = encrypt_data(data, key)
print(encrypted_data)

decrypted_data = decrypt_data(encrypted_data, key)
print(decrypted_data)
```

### 4.3.2 详细解释说明

在上述代码中，我们首先导入了 `Crypto` 模块，用于加密和解密。然后定义了两个加密和解密函数，分别用于加密和解密数据。在加密函数中，我们使用了 AES 加密算法和 EC 模式来加密数据。在解密函数中，我们使用了 AES 解密算法和 EC 模式来解密数据。然后我们测试了这两个函数，将原始数据加密后解密后打印出来。
