```python

# 读取购买记录数据

# 读取聊天记录数据
chat_data = pd.read_csv('chat_data.csv')

# 合并数据
data = pd.concat([buy_data, chat_data], axis=1)

# 填充缺失值
data.fillna(0, inplace=True)

# 去除重复数据
data.drop_duplicates(inplace=True)
```
