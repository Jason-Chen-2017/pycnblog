import numpy as np
import matplotlib.pyplot as plt

def euler_method(f, x0, y0, h, t_end):
    t = np.arange(0, t_end, h)
    y = np.zeros_like(t)
    y[0] = y0

    for i in range(1, len(t)):
        y[i] = y[i - 1] + h * f(t[i - 1], y[i - 1])

    return t, y

def random_number(seed):
    mt = np.random.MersenneTwister(seed)
    return mt.randint(0, 100)

def main():
    # 微分方程：x' = -x
    f = lambda t, x: -x

    # 初始条件
    x0 = 1
    y0 = 0
    h = 0.1
    t_end = 10

    # 生成随机数
    seed = 42
    np.random.seed(seed)

    # 执行并行计算
    t, y = euler_method(f, x0, y0, h, t_end)

    # 绘制结果
    plt.plot(t, y)
    plt.xlabel('t')
    plt.ylabel('y')
    plt.title('Euler Method')
    plt.show()

if __name__ == '__main__':
    main()
