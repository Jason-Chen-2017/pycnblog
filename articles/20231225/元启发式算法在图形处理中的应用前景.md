import numpy as np

def rotate(x, y, angle):
    return x * np.cos(angle) - y * np.sin(angle), x * np.sin(angle) + y * np.cos(angle)

def fitness(x, y, angle):
    return -np.sum((x - np.mean(x)) ** 2) - np.sum((y - np.mean(y)) ** 2)

def generate(x, y, angle, mutation_rate):
    if random.random() < mutation_rate:
        angle += random.uniform(-0.1, 0.1)
    return x, y, angle

def main():
    x = np.random.rand(10)
    y = np.random.rand(10)
    angle = random.uniform(0, 2 * np.pi)
    best_angle = angle
    best_fitness = fitness(x, y, angle)

    for _ in range(1000):
        x_new, y_new, angle_new = generate(x, y, angle, 0.1)
        new_fitness = fitness(x_new, y_new, angle_new)
        if new_fitness > best_fitness:
            best_fitness = new_fitness
            best_angle = angle_new
            x = x_new
            y = y_new

    print("Best angle:", best_angle)
    print("Best fitness:", best_fitness)

if __name__ == "__main__":
    main()
