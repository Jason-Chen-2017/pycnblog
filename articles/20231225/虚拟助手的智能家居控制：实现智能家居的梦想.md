from spacy.tokens import Doc

# 实体识别
def named_entity_recognition(text):
    doc = Doc(text)
    matcher = Matcher(doc)
    matcher.add('PERSON', [{'POS': 'PROPN'}])
    matcher.add('ORGANIZATION', [{'POS': 'PROPN'}])
    matcher(doc)
    return [(ent.text, ent.label_) for ent in doc.ents]

# 关系抽取
def relation_extraction(text):
    doc = Doc(text)
    matcher = Matcher(doc)
    matcher.add('RELATION', [{'LOWER': 'birth'}, {'LOWER': 'death'}])
    matcher(doc)
    return [(ent.text, ent.label_) for ent in doc.ents]

# 实体链接
def entity_linking(text):
    doc = Doc(text)
    matcher = Matcher(doc)
    matcher.add('PERSON', [{'LOWER': 'person'}, {'LOWER': 'name'}])
    matcher.add('ORGANIZATION', [{'LOWER': 'organization'}, {'LOWER': 'name'}])
    matcher(doc)
    return [(ent.text, ent.label_) for ent in doc.ents]
