```

import org.apache.ignite.Ignition;

// 2. 创建一个 Ignite 实例
Ignite ignite = Ignition.start();

// 3. 创建一个缓存实例
IgniteCache<Integer, String> cache = ignite.cache("myCache");

// 4. 插入数据
cache.put(1, "Hello, World!");

// 5. 更新数据
cache.put(1, "Hello, Apache Ignite!");

// 6. 删除数据
cache.remove(1);

// 7. 查询数据
String value = cache.get(1);

// 8. 关闭 Ignite 实例
ignite.close();
```
