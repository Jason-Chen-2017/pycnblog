        self.b = np.random.randn(hidden_size)
        self.V = np.random.randn(output_size, hidden_size)
        self.c = np.random.randn(output_size)
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size

    def forward(self, v, h):

        self.v = v
        self.energy = np.dot(self.W, h) + np.dot(self.V, v) + self.b.T * h + self.c.T * v
        self.p_h = 1 / (1 + np.exp(-self.energy))
        self.p_v = 1 / (1 + np.exp(-self.energy))

    def sample(self, h):
        self.h = h
        self.energy = np.dot(self.W, h) + np.dot(self.V, self.v) + self.b.T * h + self.c.T * self.v
        self.p_h = 1 / (1 + np.exp(-self.energy))
        self.v = self.p_h / np.sum(self.p_h)

    def contrastive_divergence(self, v, h, g):

        self.sample(h)
        self.gradients_h = np.dot(self.W.T, (self.p_h - g * self.p_v))
        self.gradients_v = np.dot(self.V.T, (self.p_v - g * self.p_h))

        self.V += self.learning_rate * self.gradients_v

    def train(self, v, h, g, learning_rate):
