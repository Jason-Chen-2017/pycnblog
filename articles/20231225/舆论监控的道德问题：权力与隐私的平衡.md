## 4.1 自然语言处理

### 4.1.1 文本分类

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.pipeline import Pipeline
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

# 数据收集
data = ["这个政策很好", "这个政策很差", "这个电影很好", "这个电影很差"]

# 数据预处理
data_processed = [d.lower() for d in data]

# 特征提取
vectorizer = TfidfVectorizer()
features = vectorizer.fit_transform(data_processed)

# 模型训练
clf = MultinomialNB()
clf.fit(features, data)

# 模型评估
X_test, X_valid, y_test, y_valid = train_test_split(data_processed, data, test_size=0.2)
features_test = vectorizer.transform(X_test)
predictions = clf.predict(features_test)
print("Accuracy:", accuracy_score(y_test, predictions))

# 模型应用
new_data = "这个电影很好"
new_features = vectorizer.transform([new_data.lower()])
prediction = clf.predict(new_features)
print(prediction)
```

### 4.1.3 关键词提取

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.feature_extraction.text import TfidfTransformer
from sklearn.pipeline import Pipeline
from sklearn.datasets import fetch_20newsgroups

# 数据收集
data = fetch_20newsgroups(subset='all', categories=None, remove=('headers', 'footers', 'quotes'))

# 数据预处理
data_processed = [d.lower() for d in data]

# 特征提取
vectorizer = TfidfVectorizer()
features = vectorizer.fit_transform(data_processed)

# 关键词提取
extractor = TfidfTransformer()
keywords = extractor.fit_transform(features).toarray()

# 关键词筛选
vocabulary = vectorizer.get_feature_names_out()
keyword_dict = {vocabulary[i]: keywords[i].sum() for i in range(len(vocabulary))}
sorted_keywords = sorted(keyword_dict.items(), key=lambda x: x[1], reverse=True)

# 关键词应用
print(sorted_keywords[:10])
```
