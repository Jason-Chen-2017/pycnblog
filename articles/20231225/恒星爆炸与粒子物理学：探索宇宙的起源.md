import matplotlib.pyplot as plt

def f(x, p):
    return (x**2 + p**2) ** -3

def solve(L, T, N):
    x = np.linspace(0, L, N)
    u = np.zeros(N)
    for t in range(T):
        for n in range(1, N):
            A_n = (2 / (n * np.pi)) * (1 - np.cos(n * np.pi))
            u[n] = u[n] + A_n * np.sin(n * np.pi * x) * np.exp(-(n * np.pi / L) ** 2 * t)
    return x, u

L = 10

x, u = solve(L, T, N)

plt.plot(x, u)
plt.xlabel('x')
plt.ylabel('u(x, t)')
plt.title('Temperature distribution over time')
plt.show()
