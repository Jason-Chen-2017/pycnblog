```python
import re

def check_copyright(code):
    copyright_patterns = [
        r'\b(MIT|Apache|BSD)\b',
        r'\b(GPL|LGPL|AGPL)\b',
        r'\b(CC|Creative Commons)\b',
    ]
    for pattern in copyright_patterns:
        if re.search(pattern, code):
            return True
    return False

code = '''
# Your code here
'''

if check_copyright(code):
    print('This code may contain third-party resources.')
else:
    print('This code does not contain third-party resources.')
```
