import numpy as np

def pid_control(setpoint, process_value):
    error = setpoint - process_value
    integral = np.integrate.accumulate(error)
    derivative = (error - np.roll(error, 1)) / 1
    control = setpoint - (integral[-1] * 0.5) - (derivative * 0.5)
    return control
