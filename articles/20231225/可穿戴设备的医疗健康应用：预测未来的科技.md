## 4.1 数据预处理

```python
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import MinMaxScaler

# 加载数据
data = pd.read_csv('heart_rate.csv')

# 数据清洗
data = data.dropna()

# 数据归一化
scaler = MinMaxScaler()
data = scaler.fit_transform(data)

# 数据分割
X_train, X_test, y_train, y_test = train_test_split(data[:, :-1], data[:, -1], test_size=0.2, random_state=42)
```

## 4.2 特征提取

```python
from scipy.signal import hilbert

# 时域特征
def domain_features(data):
    features = []
    for x in data:
        mean = np.mean(x)
        std = np.std(x)
        features.append([mean, std])
    return np.array(features)

# 频域特征
def frequency_features(data):
    features = []
    for x in data:
        fft = np.fft.fft(x)
        mean = np.mean(np.abs(fft[:10]))
        features.append(mean)
    return np.array(features)

# 时频特征
def time_frequency_features(data):
    features = []
    for x in data:
        hilbert_envelope = hilbert(x)
        mean = np.mean(hilbert_envelope)
        std = np.std(hilbert_envelope)
        features.append([mean, std])
    return np.array(features)

# 提取特征
X_train_features = domain_features(X_train)
X_test_features = domain_features(X_test)
```

## 4.3 模型训练

```python
from sklearn.ensemble import RandomForestRegressor

# 选择模型
model = RandomForestRegressor(n_estimators=100, random_state=42)

# 参数优化
model.fit(X_train_features, y_train)

# 模型评估
y_pred = model.predict(X_test_features)
from sklearn.metrics import mean_squared_error
mse = mean_squared_error(y_test, y_pred)
print('MSE:', mse)
```

## 4.4 预测与推理

```python
# 预测心率
def predict_heart_rate(data):
    features = time_frequency_features(data)
    pred = model.predict(features)
    return pred

# 推理结果应用
heart_rate = predict_heart_rate(X_test)
print('Predicted Heart Rate:', heart_rate)
```
