def health_knowledge_promotion(audience, knowledge_increase_rate, promotion_method):
    for audience_member in audience:
        knowledge_before = audience_member.knowledge
        knowledge_after = knowledge_before + knowledge_increase_rate
        audience_member.knowledge = knowledge_after
        method_adopted = random.random() < promotion_method
        if method_adopted:
            audience_member.method_adopted = True

audience = [AudienceMember(knowledge=100)]
health_knowledge_promotion(audience, knowledge_increase_rate=10, promotion_method=0.5)
