]

def calculate_score(grades, similarity, rules):
    total_score = 0
    for rule in rules:
        if_part, weight = rule
        conditions = [grades[i][j] for i, j in enumerate(np.argmax(similarity, axis=0))]
        then_part = rule[0].split()[-1]
        score = np.mean(conditions) * weight
        total_score += score
