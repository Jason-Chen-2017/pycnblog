### 4.1 数字孪生代码实例

以下是一个简单的数字孪生代码实例，用于实现温度传感器的数字模型。

```python
import time
import random

class TemperatureSensor:
    def __init__(self, id, interval):
        self.id = id
        self.interval = interval
        self.data = []

    def get_temperature(self):
        return random.uniform(20, 30)

    def start(self):
        while True:
            temperature = self.get_temperature()
            self.data.append(temperature)
            print(f"Sensor {self.id}: {temperature}")
            time.sleep(self.interval)

if __name__ == "__main__":
    sensor1 = TemperatureSensor(1, 1)
    sensor2 = TemperatureSensor(2, 1)
    sensor1.start()
    sensor2.start()
```

### 4.2 人工智能代码实例

以下是一个简单的人工智能代码实例，用于实现线性回归算法。

```python
import numpy as np

class LinearRegression:
    def __init__(self, learning_rate, iterations):
        self.learning_rate = learning_rate
        self.iterations = iterations
        self.weights = None
        self.bias = None

    def fit(self, X, y):
        self.weights = np.zeros(X.shape[1])
        self.bias = 0
        for _ in range(self.iterations):
            y_pred = np.dot(X, self.weights) + self.bias
            gradient_weights = (1 / X.shape[0]) * np.dot(X.transpose(), (y_pred - y))
            gradient_bias = (1 / X.shape[0]) * np.sum(y_pred - y)
            self.weights -= self.learning_rate * gradient_weights
            self.bias -= self.learning_rate * gradient_bias

    def predict(self, X):
        return np.dot(X, self.weights) + self.bias

if __name__ == "__main__":
    X = np.array([[1], [2], [3], [4]])
    y = np.array([2, 3, 4, 5])
    model = LinearRegression(learning_rate=0.01, iterations=1000)
    model.fit(X, y)
    print(model.predict(X))
```
