# 4.具体代码实例和详细解释说明

```python
import numpy as np
import matplotlib.pyplot as plt

def car_arrival(lambda_val):
    return np.random.poisson(lambda_val)

def service_time(mu_val):
    return np.random.exponential(mu_val)

def red_green_light_control(lambda_val, mu_val, num_of_lanes, num_of_cars, num_of_steps):
    cars_in_queue = 0
    cars_served = 0
    time_step = 0
    while cars_served < num_of_cars:
        if time_step % num_of_lanes == 0:
            cars_in_queue += car_arrival(lambda_val)
        if np.random.rand() < 1 / num_of_lanes:
            cars_in_queue -= 1
        if cars_in_queue > 0:
            cars_served += 1
            time_step += service_time(mu_val)
        else:
            time_step += 1
        if time_step >= num_of_steps:
            break
    return cars_served, time_step

num_of_cars = 100
lambda_val = 1
mu_val = 1
num_of_steps = 1000
cars_served, time_step = red_green_light_control(lambda_val, mu_val, 1, num_of_cars, num_of_steps)
print("Cars served: ", cars_served)
print("Time step: ", time_step)
```
