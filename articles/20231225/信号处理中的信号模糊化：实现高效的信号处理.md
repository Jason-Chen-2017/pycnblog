import matplotlib.pyplot as plt

def gaussian_fuzziness(x, mu, sigma):
    return np.exp(-(x - mu)**2 / (2 * sigma**2))

def gaussian_fuzzification(x, mu, sigma):
    return np.exp(-(x - mu)**2 / (2 * sigma**2))

def gaussian_fuzziness_adjustment(x, mu, sigma):
    return np.exp(-(x - mu)**2 / (2 * sigma**2))

def main():
    x = np.linspace(-10, 10, 100)
    mu = 0
    sigma = 1

    fuzziness = gaussian_fuzziness(x, mu, sigma)
    fuzzification = gaussian_fuzzification(x, mu, sigma)
    fuzziness_adjustment = gaussian_fuzziness_adjustment(x, mu, sigma)

    plt.figure()
    plt.plot(x, fuzziness, label='Fuzziness')
    plt.plot(x, fuzzification, label='Fuzzification')
    plt.plot(x, fuzziness_adjustment, label='Fuzziness Adjustment')
    plt.legend()
    plt.show()

if __name__ == '__main__':
    main()
