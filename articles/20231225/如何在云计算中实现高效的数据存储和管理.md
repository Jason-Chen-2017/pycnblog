    id INT PRIMARY KEY,
    name VARCHAR(255),
    age INT,
    salary DECIMAL(10, 2)
);

# 插入数据
INSERT INTO employees (id, name, age, salary) VALUES (1, 'John Doe', 30, 5000.00);

# 查询数据

UPDATE employees SET salary = 5500.00 WHERE id = 1;
