```python
def generate_harmony(notes):
    harmonies = []
    for i in range(len(notes)):
        for j in range(i + 1, len(notes)):
            harmony = notes[i] + notes[j]
            harmonies.append(harmony)
    return harmonies

notes = ["C4", "E4", "G4"]  # 音高列表
harmonies = generate_harmony(notes)
print("和声：", harmonies)
```
