```python
import numpy as np

def ecological_impact(data):
    # 计算生物多样性指数
    biodiversity_index = np.mean(data['species_richness'])
    
    # 计算生态系统健康指数
    ecosystem_health_index = np.mean(data['ecosystem_health'])
    
    # 计算环境影响指数
    impact_index = 1 / (biodiversity_index + ecosystem_health_index)
    
    return impact_index

impact = ecological_impact(D)
```
