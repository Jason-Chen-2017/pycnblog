import matplotlib.pyplot as plt

# 定义大陆壳形成模型的参数

    'time': 10000,  # 模型的时间长度，单位为年
    'continent_area': 140000000,  # 大陆壳的面积，单位为平方公里
    'ocean_area': 130000000,  # 海洋壳的面积，单位为平方公里
    'continent_density': 2.5,  # 大陆壳的密度，单位为千克/立方公里
    'ocean_density': 1.0,  # 海洋壳的密度，单位为千克/立方公里
}

def continent_formation_model(time, continent_area, ocean_area, continent_density, ocean_density):

    continent_mass = continent_area * continent_density
    # 计算海洋壳的质量
    ocean_mass = ocean_area * ocean_density

    earth_mass = continent_mass + ocean_mass
    # 计算大地壳的平均密度
    earth_density = earth_mass / (continent_area + ocean_area)
    # 返回大地壳的平均密度
    return earth_density
