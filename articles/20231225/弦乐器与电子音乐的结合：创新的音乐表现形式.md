```python
import matplotlib.pyplot as plt
from scipy.signal import find_peaks
from scipy.fft import fft

# 加载弦乐器音频文件
guitar_audio = np.load('guitar.wav')

# 计算音频的频谱
guitar_spectrum = fft(guitar_audio)

# 绘制频谱图
plt.plot(guitar_spectrum)
plt.xlabel('Frequency (Hz)')
plt.ylabel('Amplitude')
plt.title('Guitar Spectrum')
plt.show()

# 提取音频的节奏特征
sampling_rate = guitar_audio.shape[0] / guitar_audio.shape[1]
guitar_rms = np.sqrt(np.mean(guitar_audio**2))
guitar_rms_spectrum = guitar_rms / sampling_rate

# 绘制节奏特征图
plt.plot(guitar_rms_spectrum)
plt.xlabel('Time (s)')
plt.ylabel('RMS Amplitude')
plt.title('Guitar RMS Spectrum')
plt.show()
```
