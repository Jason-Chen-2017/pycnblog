import numpy as np

# 初始化粒子
def initialize_particles(num_particles, state_dim, input_dim):
    particles = np.zeros((num_particles, state_dim))
    return particles

# 预测粒子位置
def predict_particles(particles, state_transition_model, time_step):
    updated_particles = np.zeros((particles.shape))
    for i in range(particles.shape[0]):
        updated_particles[i, :] = state_transition_model(particles[i, :], time_step)
    return updated_particles

# 更新粒子权重
def update_particles_weights(particles, likelihood, time_step):
    weights = np.array([likelihood(particle, time_step) for particle in particles])
    normalized_weights = weights / np.sum(weights)
    return normalized_weights

# 重新采样粒子
def resample_particles(particles, weights):
    num_particles = particles.shape[0]
    indices = np.random.choice(num_particles, num_particles, p=weights)
    resampled_particles = particles[indices, :]
    return resampled_particles

# 迭代粒子滤波
def particle_filter(num_particles, state_dim, input_dim, state_transition_model, likelihood, time_steps):
    particles = initialize_particles(num_particles, state_dim, input_dim)
    weights = np.ones(num_particles) / num_particles
    for time_step in time_steps:
        particles = predict_particles(particles, state_transition_model, time_step)
        weights = update_particles_weights(particles, likelihood, time_step)
        particles = resample_particles(particles, weights)
    return particles, weights
