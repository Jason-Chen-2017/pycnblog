# 2.核心概念与联系
# 2.1引力波的基本概念

# 2.2遥感技术的基本概念
遥感技术是一种用于测量地球表面和大气层的远程感知技术。它可以用来观测大气、地表和海洋的物理、化学和生物特性。遥感技术的主要应用包括地质资源探测、气候变化监测、生态环境保护等。

def generate_gravity_wave_signal(amplitude, frequency, duration):
    t = np.linspace(0, duration, int(duration * 1e4))
    signal = amplitude * np.sin(2 * np.pi * frequency * t)
    return signal

# 傅里叶变换
def fft(signal):
    return np.fft.fft(signal)

# 信号处理
def process_signal(signal, sampling_rate):
    fft_signal = fft(signal)
    frequencies = np.fft.fftfreq(len(signal), d=1/sampling_rate)
    return fft_signal, frequencies

# 主函数
