import numpy as np
import random

def firefly_algorithm(f, bounds, n_fireflies, max_iterations, alpha, beta, gamma):
    fireflies = [np.random.uniform(bounds[0], bounds[1]) for _ in range(n_fireflies)]
    fitness = [f(x) for x in fireflies]

    for _ in range(max_iterations):
        for i in range(n_fireflies):
            for j in range(n_fireflies):
                if fitness[j] < fitness[i]:
                    r = np.linalg.norm(fireflies[i] - fireflies[j])
                    beta = beta * np.exp(-gamma * r ** 2)
                    fireflies[i] = fireflies[i] + alpha * beta * (fireflies[j] - fireflies[i]) + np.random.normal(scale=1, size=len(fireflies[i]))
                    fitness[i] = f(fireflies[i])

    best_firefly = fireflies[np.argmin(fitness)]
    return best_firefly, fitness[np.argmin(fitness)]

def rosenbrock_function(x):
    return (1 - x[0]) ** 2 + 100 * (x[1] - x[0] ** 2) ** 2

bounds = ((-2.0, 2.0), (-1.0, 3.0))
n_fireflies = 10
max_iterations = 100
alpha = 0.5
beta = 1.0
gamma = 0.5

best_solution, best_fitness = firefly_algorithm(rosenbrock_function, bounds, n_fireflies, max_iterations, alpha, beta, gamma)
print("Best solution:", best_solution)
print("Best fitness:", best_fitness)
