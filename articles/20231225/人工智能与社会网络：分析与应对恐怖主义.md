### 4.1.2 Apriori算法

```python
from apyori import apriori

# 数据集
transactions = [
    ['Milk', 'Bread', 'Eggs'],
    ['Bread', 'Eggs', 'Cheese'],
    ['Bread', 'Cheese', 'Milk'],
    ['Bread', 'Cheese', 'Eggs'],
    ['Bread', 'Milk', 'Eggs'],
    ['Milk', 'Cheese', 'Eggs']
]

# 使用Apriori算法
association_rules = apriori(transactions, min_support=0.5, min_confidence=0.7)

# 输出关联规则
for rule in association_rules:
    print(rule)
```

### 4.3.3 深度学习

```python
import tensorflow as tf
from tensorflow.keras.preprocessing.text import Tokenizer
from tensorflow.keras.preprocessing.sequence import pad_sequences

# 文本
texts = ["I love this product!", "This is a terrible product!"]

# 使用Tokenizer
tokenizer = Tokenizer(num_words=1000)
tokenizer.fit_on_texts(texts)
sequences = tokenizer.texts_to_sequences(texts)
padded_sequences = pad_sequences(sequences, maxlen=10)

# 使用深度学习
model = tf.keras.Sequential([
    tf.keras.layers.Embedding(1000, 64, input_length=10),
    tf.keras.layers.GlobalAveragePooling1D(),
    tf.keras.layers.Dense(1, activation='sigmoid')
])

model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])

# 训练模型
model.fit(padded_sequences, [1, 0], epochs=10)

# 预测情感
predictions = model.predict(padded_sequences)
print(predictions)
```

### 4.4.4 神经网络

```python
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler

# 数据集
data = pd.read_csv("terrorism_data.csv")

# 特征和标签
X = data.drop("outcome", axis=1)
y = data["outcome"]

# 标准化特征
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y, test_size=0.2, random_state=0)

# 使用神经网络
model = Sequential([
    Dense(64, activation='relu', input_shape=(X_train.shape[1],)),
    Dense(32, activation='relu'),
    Dense(1, activation='sigmoid')
])

model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])

# 训练模型
model.fit(X_train, y_train, epochs=10)

# 预测标签
y_pred = model.predict(X_test)

# 输出准确度
print(accuracy_score(y_test, y_pred))
```
