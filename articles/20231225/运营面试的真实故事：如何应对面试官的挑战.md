import numpy as np

def func(x):
    return x**2

def grad(x):
    return 2*x

def gradient_descent(start, learning_rate, iterations):
    x = start
    for i in range(iterations):
        grad_x = grad(x)
        x = x - learning_rate * grad_x
    return x
