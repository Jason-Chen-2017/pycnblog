   ```python
   import numpy as np
   from scipy.optimize import linprog

   c = np.array([1, 2, 3])  # 成本向量
   A = np.array([[1, 1, 1], [2, 2, 2]])  # 资源矩阵
   b = np.array([10, 20])  # 资源需求向量

   res = linprog(-c, A_ub=A, b_ub=b, bounds=(0, None))
   print(res)
   ```

   b. 动态规划：

   ```python
   def dynamic_programming(n, m, c, w):
       dp = [[0] * (m + 1) for _ in range(n + 1)]
       for i in range(1, n + 1):
           for j in range(1, m + 1):
               dp[i][j] = min(dp[i - 1][j], dp[i][j - 1] + c[i - 1][j - 1])
       return dp[n][m]

   n = 3
   m = 3
   c = [[1, 2], [2, 3], [3, 4]]
   w = 10
   print(dynamic_programming(n, m, c, w))
   ```

   c. 贪婪算法：

   ```python
   def greedy_algorithm(n, c):
       dp = [0] * n
       for i in range(n):
           dp[i] = c[i]
           for j in range(i):
               if c[j] < c[i]:
                   dp[i] = c[j]
       return dp

   n = 3
   c = [1, 2, 3]
   print(greedy_algorithm(n, c))
   ```
