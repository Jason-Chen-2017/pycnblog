```python
import socket

def receive_data():
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.bind(('localhost', 8080))
    s.listen(5)
    while True:
        conn, addr = s.accept()
        data = conn.recv(1024)
        print(data.decode())
        conn.close()
```
