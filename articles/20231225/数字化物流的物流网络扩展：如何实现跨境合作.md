def dp(n, W):
    dp = [[0] * (W + 1) for _ in range(n + 1)]
    for i in range(n + 1):
        for w in range(W + 1):
            if i == 0 or w == 0:
                dp[i][w] = 0
            elif W - w >= 0:
                dp[i][w] = max(dp[i - 1][w], dp[i - 1][w - W] + 1)
            else:
                dp[i][w] = dp[i - 1][w]
    return dp[n][W]

### 5.1 未来发展趋势
