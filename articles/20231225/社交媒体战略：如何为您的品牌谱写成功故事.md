```python
import numpy as np
from scipy.sparse.linalg import spsolve
from scipy.sparse import csc_matrix

def sim_matrix(ratings):
    similarity = np.zeros((len(ratings), len(ratings)))
    for i, row in enumerate(ratings):
        for j, rating in enumerate(row):
            if rating != 0:
                similarity[i, j] = 1
            if i != j:
                similarity[i, j] += 1
    similarity = similarity / np.sum(similarity, axis=1)[:, None]
    return similarity

def recommend(ratings, user, num_recommendations):
    similarity = sim_matrix(ratings)
    similarity = np.nan_to_num(similarity)
    movie_indices = np.where(ratings[user] == 0)[0]
    if len(movie_indices) == 0:
        return ratings[user]
    similarity_user = similarity[user]
    similarity_user = similarity_user[np.delete(np.arange(len(similarity_user)), user)]
    scores = np.dot(similarity_user, ratings.T)
    recommendations = np.argsort(-scores)[:num_recommendations]
    return np.array([ratings[user][i] if i != user else 0 for i in movie_indices[recommendations]])

ratings = np.array([
    [4, 0, 0, 0, 3],
    [0, 2, 0, 0, 1],
    [0, 0, 3, 0, 2],
    [1, 0, 0, 2, 0],
    [0, 3, 0, 1, 0]
])

user = 0
num_recommendations = 2
recommended_movies = recommend(ratings, user, num_recommendations)
print(recommended_movies)
```
