import numpy as np
from sklearn.kernel_approximation import Nystroem
from sklearn.decomposition import PCA
from sklearn.metrics.pairwise import rbf_kernel

# 定义数据
X = np.array([[1, 2], [3, 4], [5, 6], [7, 8]])

# 定义核函数
def higher_order_kernel(X, degree=2):
    n_samples, n_features = X.shape
    K = np.zeros((n_samples, n_samples))
    for i in range(n_samples):
        for j in range(i + 1, n_samples):
            K[i, j] = rbf_kernel(X[i:i + 1, :], X[j:j + 1, :], degree)
    return K

# 计算核矩阵
K = higher_order_kernel(X, degree=2)

# 使用 Nystroem 算法进行核Approximation
n_components = 100
n_iters = 100
n_samples = X.shape[0]
n_indices = np.random.randint(0, n_samples, size=n_components)

nystroem = Nystroem(kernel=K, algorithm='randomized', n_components=n_components, n_iters=n_iters)
X_approx = nystroem.fit_transform(X)

# 使用 PCA 进行降维
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_approx)

# 绘制数据
import matplotlib.pyplot as plt
plt.scatter(X_pca[:, 0], X_pca[:, 1])
plt.show()
