```python
import numpy as np

class Nanorobot:
    def __init__(self, position, velocity, acceleration):
        self.position = position
        self.velocity = velocity
        self.acceleration = acceleration

    def move(self, time):
        self.position = self.position + self.velocity * time + 0.5 * self.acceleration * time**2
        self.velocity = self.velocity + self.acceleration * time

robot = Nanorobot(np.array([0, 0]), np.array([1, 0]), np.array([0, -9.8]))
robot.move(2)
print(robot.position)
```
