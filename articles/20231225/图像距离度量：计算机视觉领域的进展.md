```python
import cv2

def gray_level_statistics(image):
    gray_level_count = np.zeros(256)
    for pixel in image.flatten():
        gray_level_count[pixel] += 1

```

### 4.4图像相似性计算

```python
import numpy as np
import cv2

def image_similarity(image1, image2):
    gray_level_count1 = gray_level_statistics(image1)
    gray_level_count2 = gray_level_statistics(image2)
    similarity = 0
    for i in range(256):
        similarity += min(gray_level_count1[i], gray_level_count2[i])
    return similarity / 256
```
