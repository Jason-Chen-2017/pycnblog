```python
import requests
from bs4 import BeautifulSoup

url = 'https://www.lagou.com/jobs/list_'
headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/58.0.3029.110 Safari/537.3'}

def get_job_list(page):
    response = requests.get(url + page, headers=headers)
    soup = BeautifulSoup(response.text, 'html.parser')
    job_list = soup.select('.job-list')
    return job_list

def parse_job(job):
    title = job.select_one('.title').text
    company = job.select_one('.company').text
    salary = job.select_one('.salary').text
    return {'title': title, 'company': company, 'salary': salary}

page = 1
job_list = get_job_list(page)
jobs = [parse_job(job) for job in job_list]
```

```python
import random

def generate_training_plan():
    training_plan = {}
    training_plan['name'] = ' '.join(random.sample(['Python', 'Java', 'C++', 'JavaScript', '数据库', '算法', '网络'], 6))
    training_plan['duration'] = random.randint(1, 6)
    training_plan['format'] = random.choice(['面授', '在线', '混合'])
    training_plan['cost'] = random.randint(1000, 10000)
    return training_plan

training_plan = generate_training_plan()
print(training_plan)
```
