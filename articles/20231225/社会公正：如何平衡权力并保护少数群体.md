```python
import numpy as np

# 预处理数据
scaler = StandardScaler()

# 计算权重
pca = PCA(n_components=2)
data_pca = pca.fit_transform(data_scaled)
weights = np.mean(data_pca, axis=0)

# 计算影响力
linear_model = LinearRegression()
linear_model.fit(data_scaled, weights)
impact_factors = linear_model.predict(data_scaled)

# 调整算法
adjusted_impact_factors = np.where(data['minority'], impact_factors * 1.5, impact_factors)
```
