                'transactions': []}

        return block

        transaction = {'sender': sender,
                      'recipient': recipient,
                      'amount': amount}
        self.chain[len(self.chain) - 1]['transactions'].append(transaction)
        return self.last_block['hash']

    def new_block(self, proof, previous_hash):
        previous_blocks = self.chain[len(self.chain) - 5:]
        if previous_blocks:
            if previous_blocks[-1]['hash'] == previous_hash:
                self.chain.append(self.create_block(proof, previous_hash))
                return True
        return False

    @staticmethod
    def hash(block):
        block_string = json.dumps(block, sort_keys = True).encode()
        return hashlib.sha256(block_string).hexdigest()

    @property
    def last_block(self):
        return self.chain[-1]
