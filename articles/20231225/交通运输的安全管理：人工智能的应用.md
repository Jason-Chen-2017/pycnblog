## 4.3 交通事故预测代码实例
```python
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression

# 加载数据
data = pd.read_csv("accident_data.csv")

# 预处理数据
data = data.dropna()
data = data[['accident_rate', 'traffic_volume', 'road_length', 'weather_condition']]

# 分割数据
X_train, X_test, y_train, y_test = train_test_split(data, data['accident_rate'], test_size=0.2, random_state=42)

# 构建模型
model = LinearRegression()
model.fit(X_train, y_train)

# 评估模型
accuracy = model.score(X_test, y_test)
print("Accuracy:", accuracy)
```
