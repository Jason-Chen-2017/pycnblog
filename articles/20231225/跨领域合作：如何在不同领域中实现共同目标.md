```python
from sklearn.preprocessing import StandardScaler

# 加载数据
climate_data = pd.read_csv('climate_data.csv')
geomorphology_data = pd.read_csv('geomorphology_data.csv')
ecology_data = pd.read_csv('ecology_data.csv')

# 数据清洗
climate_data = climate_data.dropna()
geomorphology_data = geomorphology_data.dropna()
ecology_data = ecology_data.dropna()

# 数据转换
climate_data = pd.get_dummies(climate_data)
geomorphology_data = pd.get_dummies(geomorphology_data)
ecology_data = pd.get_dummies(ecology_data)

# 数据融合
data = pd.concat([climate_data, geomorphology_data, ecology_data], axis=1)

# 数据标准化
scaler = StandardScaler()
data = scaler.fit_transform(data)
```
