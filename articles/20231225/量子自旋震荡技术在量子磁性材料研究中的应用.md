import matplotlib.pyplot as plt

    t2 = np.arange(0, k*2*np.pi/T1, 2*np.pi/T1, dtype=np.float64)
    t = np.sqrt(t1**2 + t2**2)
    echo = np.exp(-t**2/(2*T2)) * np.cos(t**2/(2*T1))
    plt.plot(t1, echo)
    plt.xlabel('t1')
    plt.ylabel('echo')
    plt.show()

k = 10
spin_echo(T1, T2, k)
