```python
import random

def wiggle_method(probabilities):
    """
    摇动法的随机数生成器
    :param probabilities: 概率分布
    :return: 随机数
    """
    while True:
        x = random.choices(['A', 'B', 'C'], probabilities)
        if x == 'B':
            return x
```
