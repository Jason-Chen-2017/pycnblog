class ArtificialMuscle:
    def __init__(self, k, n, length):
        self.k = k
        self.n = n
        self.length = length

    def force(self, velocity):
        return self.k * (velocity ** self.n)

    def energy(self, distance):
        return self.force(1) * distance

    def heat(self, distance):
        return self.energy(distance) - self.work(distance)

    def work(self, distance):
        return self.force(1) * distance

# 初始化人工肌肉材料参数

heat = am.heat(1)

print("Energy:", energy)
