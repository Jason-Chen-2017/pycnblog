## 4.1代码实例

```python
import numpy as np
import cv2
import matplotlib.pyplot as plt

def high_order_nonlinear_kernel(images, order, kernel_size):
    # 读取原始图像信息
    img1 = cv2.imread(images[0], cv2.IMREAD_GRAYSCALE)
    img2 = cv2.imread(images[1], cv2.IMREAD_GRAYSCALE)

    # 对原始图像信息进行高阶采样，得到高阶特征信息
    for n in range(2, order + 1):
        img1 = cv2.resize(img1, (img1.shape[1] // n, img1.shape[0] // n))
        img2 = cv2.resize(img2, (img2.shape[1] // n, img2.shape[0] // n))

    # 对高阶特征信息进行非线性融合，得到融合后的高阶特征信息
    fused_img1 = cv2.filter2D(img1, -1, np.ones((kernel_size, kernel_size)))
    fused_img2 = cv2.filter2D(img2, -1, np.ones((kernel_size, kernel_size)))

    # 通过逆向处理，得到融合后的图像信息
    fused_img = cv2.resize(fused_img1, (img1.shape[1], img1.shape[0]))
    fused_img = cv2.resize(fused_img, (img2.shape[1], img2.shape[0]))

    # 保存融合后的图像信息

    # 显示融合后的图像信息
    plt.subplot(121), plt.imshow(img1, cmap='gray')
    plt.title('Original Image 1'), plt.xticks([]), plt.yticks([])
    plt.subplot(122), plt.imshow(img2, cmap='gray')
    plt.title('Original Image 2'), plt.xticks([]), plt.yticks([])
    plt.subplot(123), plt.imshow(fused_img, cmap='gray')
    plt.title('Fused Image'), plt.xticks([]), plt.yticks([])
    plt.show()

# 调用高阶非线性核在图像融合中的函数
```
