## 1.4 具体代码实例和详细解释说明

```python
import numpy as np

def distance(x1, y1, x2, y2):
    return np.sqrt((x1 - x2)**2 + (y1 - y2)**2)

def competition(d):
    return 1 / d**2 - 1 / d**4

def attraction(d):
    return 1 / d**2

def update_position(x, v, a, rand):
    return x + v * 0.1 + a * 0.1 * rand * (x - x) / np.linalg.norm(x - x)

# 初始化鱼群
fish = [{'x': 0, 'y': 0, 'v': 0, 'a': 0} for _ in range(100)]

# 主循环
while True:
    for i in range(len(fish)):
        # 计算竞争力
        for j in range(len(fish)):
            if i != j:
                d = distance(fish[i]['x'], fish[i]['y'], fish[j]['x'], fish[j]['y'])
                fish[i]['a'] += competition(d)

        # 计算吸引力
        for j in range(len(fish)):
            if i != j:
                d = distance(fish[i]['x'], fish[i]['y'], fish[j]['x'], fish[j]['y'])
                fish[i]['a'] += attraction(d)

        # 更新鱼群
        fish[i]['x'] = update_position(fish[i]['x'], fish[i]['v'], fish[i]['a'], np.random.random())

    # 判断终止条件
    if all(abs(fish[i]['x'] - fish[j]['x']) > 100 for i in range(len(fish)) for j in range(i + 1, len(fish))):
        break
```
