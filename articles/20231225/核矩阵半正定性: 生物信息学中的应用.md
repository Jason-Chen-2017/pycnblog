import numpy as np
import scipy.integrate as spi

def NMHL(t, lambda_):
    def integrand(x):
        return 1 / t * np.exp(-lambda_ * x)
    result, error = spi.quad(integrand, 0, t)
    return result

lambda_ = 0.01  # 核反应率
