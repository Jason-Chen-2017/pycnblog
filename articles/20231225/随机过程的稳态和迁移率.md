## 4.1 代码实例

```python
import numpy as np

# 状态转移矩阵
P = np.array([[0.5, 0.5], [0.3, 0.7]])

# 求稳态概率分布
def steady_state_distribution(P):
    P_prev = P
    P_current = P_prev
    while not np.allclose(P_prev, P_current):
        P_prev = P_current
        P_current = np.dot(P_prev, P)
    return P_current

# 求迁移率
def transition_matrix(P):
    return np.linalg.inv(P)

# 求稳态概率分布和迁移率
P_steady = steady_state_distribution(P)
P_transition = transition_matrix(P)

print("稳态概率分布：", P_steady)
print("迁移率：", P_transition)
```
