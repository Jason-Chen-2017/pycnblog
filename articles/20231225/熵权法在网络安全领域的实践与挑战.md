## 4.1 熵权法的Python实现

import numpy as np

def entropy(data):
    # 计算熵值
    hist, bins = np.histogram(data, bins=10)
    prob = hist / hist.sum()
    return -np.sum(prob * np.log2(prob))

def security(weights, entropies):
    # 计算系统安全性
    return np.sum(weights * entropies)

# 示例数据
data = np.random.randint(0, 100, size=100)
weights = np.array([1, 1, 1, 1])

# 计算熵权法指标
entropy_value = entropy(data)
security_value = security(weights, [entropy_value])

print("熵值: ", entropy_value)
print("系统安全性: ", security_value)
