```python
import numpy as np

def f(x, y):
    return x**2 + y**2 - 1

def jacobian(x, y):
    return np.array([[2*x, 2*y], [2*y, 2*x]])

def solve_root(x0, y0, tol=1e-6):
    x, y = x0, y0
    while True:
        j = jacobian(x, y)
        if np.linalg.det(j) == 0:
            break
        j_inv = np.linalg.inv(j)
        x_new = x - j_inv @ np.array([f(x, y), f(x, y)])
        y_new = y - j_inv @ np.array([f(x, y), f(x, y)])
        if np.linalg.norm(np.array([x - x_new, y - y_new])) < tol:
            break
        x, y = x_new, y_new
    return x, y

x0, y0 = 1, 1
root = solve_root(x0, y0)
print("方程的根为：", root)
```
