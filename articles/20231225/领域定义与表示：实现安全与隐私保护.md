    """
    实现哈希函数的映射算法
    """
    # 使用SHA256算法对数据进行哈希
    hash_data = hashlib.sha256(data.encode()).hexdigest()
    # 对哈希值进行取模操作
    hash_value = int(hash_data, 16) % 1000000
    return hash_value

# 示例数据
