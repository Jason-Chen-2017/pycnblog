```python
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///electronic_medical_records.db'
db = SQLAlchemy(app)

class Patient(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(50))
    age = db.Column(db.Integer)
    diagnosis = db.Column(db.String(100))
    treatment = db.Column(db.String(100))

@app.route('/patient', methods=['GET', 'POST'])
def patient():
    if request.method == 'GET':
        patients = Patient.query.all()
        return jsonify([{'id': p.id, 'name': p.name, 'age': p.age, 'diagnosis': p.diagnosis, 'treatment': p.treatment} for p in patients])
    elif request.method == 'POST':
        data = request.get_json()
        patient = Patient(name=data['name'], age=data['age'], diagnosis=data['diagnosis'], treatment=data['treatment'])
        db.session.add(patient)
        db.session.commit()
        return jsonify({'id': patient.id}), 201

if __name__ == '__main__':
    app.run()
```

```python
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import MinMaxScaler

# 加载健康数据
def load_health_data(file_path):
    data = pd.read_csv(file_path)
    return data

# 数据预处理
def preprocess_data(data):
    data['temperature'] = (data['temperature'] - np.mean(data['temperature'])) / np.std(data['temperature'])
    data['heart_rate'] = (data['heart_rate'] - np.mean(data['heart_rate'])) / np.std(data['heart_rate'])
    return data

# 训练线性回归模型
def train_linear_regression_model(data):
    X = data[['temperature', 'heart_rate']]
    y = data['blood_pressure']
    model = LinearRegression()
    model.fit(X, y)
    return model

# 预测血压
def predict_blood_pressure(model, temperature, heart_rate):
    X = np.array([[temperature, heart_rate]])
    y = model.predict(X)
    return y

if __name__ == '__main__':
    # 加载健康数据
    file_path = 'path/to/health_data.csv'
    data = load_health_data(file_path)

    # 数据预处理
    data = preprocess_data(data)

    # 训练线性回归模型
    model = train_linear_regression_model(data)

    # 预测血压
    temperature = 36.5
    heart_rate = 75
    predicted_blood_pressure = predict_blood_pressure(model, temperature, heart_rate)
    print('Predicted Blood Pressure:', predicted_blood_pressure)
```
