import numpy as np

def detect_freshness(image):
    # 使用边缘检测算法
    edges = cv2.Canny(image, 100, 200)
    # 使用对象识别算法
    objects = cv2.simpleBlobDetector(edges)
    # 判断食材的生鲜度
    if objects.size > 0:
        return '生鲜'
    else:
        return '过期'

## 4.3 智能配方
我们可以使用Python编程语言，以及NLTK库，来实现智能配方的查找和推荐。具体代码如下：
```python
import nltk
from nltk.corpus import recipe

def find_recipe(query):
    # 分析查询语句
    words = nltk.word_tokenize(query)
    # 提取关键词
    keywords = nltk.FreqDist(words).most_common(3)
    # 查找配方
    recipes = recipe.recipes()
    result = []
    for recipe in recipes:
        for keyword in keywords:
            if keyword[0] in recipe:
                result.append(recipe)
                break
    return result

# 查询配方
query = '烤鸡腿'
recipes = find_recipe(query)
for recipe in recipes:
    print(recipe)
```
