import numpy as np
import matplotlib.pyplot as plt

# 生物肌肉力-长度关系模拟
def muscle_force(length):
    k = 100
    n = 4
    force = k * (length ** n)
    return force

# 电信号传导模拟
def electrical_signal(time, amplitude, frequency, phase):
    voltage = amplitude * np.sin(2 * np.pi * frequency * time + phase)
    return voltage

# 控制系统整合模拟
def control_system(input_signal):
    gain = 0.5
    output = gain * input_signal
    return output

# 主程序
if __name__ == "__main__":
    # 生物肌肉力-长度关系模拟
    lengths = np.linspace(0, 1, 100)
    forces = [muscle_force(length) for length in lengths]
    plt.plot(lengths, forces)
    plt.xlabel('Length')
    plt.ylabel('Force')
    plt.title('Muscle Force-Length Relation')
    plt.show()

    # 电信号传导模拟
    time = np.linspace(0, 2 * np.pi, 100)
    voltages = [electrical_signal(t, 1, 1, 0) for t in time]
    plt.plot(time, voltages)
    plt.xlabel('Time')
    plt.ylabel('Voltage')
    plt.title('Electrical Signal')
    plt.show()

    # 控制系统整合模拟
    input_signal = np.sin(2 * np.pi * 10 * time)
    output = control_system(input_signal)
    plt.plot(time, output)
    plt.xlabel('Time')
    plt.ylabel('Output')
    plt.title('Control System Integration')
    plt.show()
