    log_image = cv2.log(1 + image)
    # 乘以因子
    result = log_image * factor
    # 取对数逆变换
    result = np.exp(result) - 1
    # 归一化
    result = cv2.normalize(result, None, 0, 1, cv2.NORM_MINMAX)
    return result

# 读取原始影像
# 对影像进行对比度增强
enhanced_image = lighten_image(image)
# 显示增强后的影像
