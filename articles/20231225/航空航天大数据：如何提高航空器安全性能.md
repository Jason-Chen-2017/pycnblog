import pandas as pd
import numpy as np

# 读取数据

# 处理噪声

import pandas as pd

data1 = pd.read_csv("data1.csv")

# 数据预处理
data1 = data1.dropna()

# 数据融合
data_fused = pd.merge(data1, data2, on="column_name")

# 数据后处理
data_fused["column_name"] = data_fused["column_name"].astype("float64")
```
