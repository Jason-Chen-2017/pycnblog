class OrderedSetPolynomialVectorSpace:
    def __init__(self):
        self.vectors = []

    def add_vector(self, vector):
        self.vectors.append(vector)

    def similarity(self, u_i, u_j):
        similarity = 0
        for v_i, w_i in u_i.items():
            for v_j, w_j in u_j.items():
                similarity += w_i * w_j * v_i * v_j
        return similarity / (self.norm(u_i) * self.norm(u_j))

    def norm(self, u):
        norm = 0
        for v, w in u.items():
            norm += w * v**2
        return norm**0.5
