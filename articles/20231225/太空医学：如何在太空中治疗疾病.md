import numpy as np

def calculate_bone_loss(rho, V, delta_g):
    bone_loss = rho * V * delta_g
    return bone_loss

rho = 1.0 # g/cm^3
V = 1000.0 # cm^3
delta_g = 0.01 # g

print("Bone loss: ", bone_loss)
