    if water_quality < 100:
        return 'clean'
    elif water_quality < 200:
        return 'moderate'
    else:
        return 'heavy'

def water_processing(water_quality):
    if water_test(water_quality) == 'clean':
        return water_quality
    elif water_test(water_quality) == 'moderate':
        return water_quality * 0.8
    else:
