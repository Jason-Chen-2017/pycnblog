```python
class SmartControlSystem:
    def __init__(self, temperature_setpoint):
        self.temperature_setpoint = temperature_setpoint

    def control_heating(self, current_temperature, time_of_day):
        if current_temperature < self.temperature_setpoint:
            if time_of_day == "morning" or time_of_day == "evening":
                print("启动加热设备")
            else:
                print("不启动加热设备")
        else:
            print("不启动加热设备")

    def control_cooling(self, current_temperature, time_of_day):
        if current_temperature > self.temperature_setpoint:
            if time_of_day == "morning" or time_of_day == "evening":
                print("启动冷气设备")
            else:
                print("不启动冷气设备")
        else:
            print("不启动冷气设备")

temperature_setpoint = 25  # 设定温度为 25 度
smart_control_system = SmartControlSystem(temperature_setpoint)
current_temperature = 20  # 当前温度为 20 度
time_of_day = "morning"
smart_control_system.control_heating(current_temperature, time_of_day)
smart_control_system.control_cooling(current_temperature, time_of_day)
```

不启动冷气设备
