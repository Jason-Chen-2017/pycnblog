from qiskit import QuantumCircuit, Aer, transpile, assemble

# 创建量子电路

# 初始化量子比特
qc.h(0)

# 设置量子门
qc.pow(2, 0)

# 获取量子电路的二进制代码
qasm_code = qc.qasm()

# 将量子电路编译为可执行的量子电路

# 使用量子回声模拟器执行量子电路

job = simulator.run(executable)

# 获取量子电路的结果
result = job.result()

# 绘制结果的直方图
plot_histogram(result.get_counts())
