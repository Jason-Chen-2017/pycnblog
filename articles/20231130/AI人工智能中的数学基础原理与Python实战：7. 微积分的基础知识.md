# 4.具体代码实例和详细解释说明

```python
import numpy as np

# 定义模型参数
theta = np.random.rand(1, 1)

# 定义损失函数
def loss_function(x, y, theta):
    return np.sum((x @ theta - y) ** 2)

# 定义梯度
def gradient(x, y, theta):
    return (x.T @ (x @ theta - y)) / len(x)

# 定义学习率
alpha = 0.01

# 定义迭代次数
iterations = 1000

# 训练模型
for i in range(iterations):
    x = np.array([[1], [2], [3]])
    y = np.array([[2], [4], [6]])
    grad = gradient(x, y, theta)
    theta = theta - alpha * grad

# 输出结果
print("theta:", theta)
```
