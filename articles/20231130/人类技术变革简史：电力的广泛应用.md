# 2.核心概念与联系

电力是指电子、电磁场的能量流动，它是人类生活和工业生产中最重要的能源之一。电磁力是电场和磁场之间的相互作用力，它是电力技术的基础。电流是电子在导体中流动的速度，电压是电子在导体中的能量密度，电能是电子在电路中的总能量，电阻是电子在导体中的阻碍力，电容是电子在电容器中的存储能量，电感是电子在电感器中的能量转换。

```python
# 电路模拟代码实例
import spice
import numpy as np

# 定义电路网络
R1 = spice.R(1, 1000)
R2 = spice.R(2, 1000)
C1 = spice.C(1, 10e-6)
Vs = spice.Vs(1, 0, 10, dc=True)

# 建立电路网络
net = spice.Net()
net.add(R1)
net.add(R2)
net.add(C1)
net.add(Vs)
net.connect(Vs.out, R1.in, R2.in, C1.in)

# 求解电路网络
sol = spice.solve(net)

# 输出电路网络的电压和电流
V1 = sol.get_v(Vs.out)
I1 = sol.get_i(R1.in)
print("V1 =", V1)
print("I1 =", I1)
```

```python
# 电磁场计算代码实例
import comsol
import numpy as np

# 定义电磁场问题
domain = comsol.Domain(type="rectangle", size=(1, 1, 1), material="air")
source = comsol.Source(type="electric_current", position=(0.5, 0.5, 0), strength=1)
boundary = comsol.Boundary(type="perfect_electric_conductor", position=(0, 0, 0), size=(1, 1, 1))

# 建立电磁场问题
problem = comsol.Problem(domain=domain, source=source, boundary=boundary)

# 求解电磁场问题
solution = comsol.solve(problem)

# 输出电磁场的电场强度和磁场强度
E = solution.get_e()
H = solution.get_h()
print("E =", E)
print("H =", H)
```
