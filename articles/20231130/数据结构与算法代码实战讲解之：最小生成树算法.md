def kruskal(graph):
    edges = sorted(graph.edges(), key=lambda x: x[2])
    result = []
    parent = [i for i in range(len(graph.nodes()))]

    def find(x):
        if parent[x] != x:
            parent[x] = find(parent[x])
        return parent[x]

    def union(x, y):
        x_root = find(x)
        y_root = find(y)
        if x_root != y_root:
            parent[x_root] = y_root

    for edge in edges:
        x, y, weight = edge
