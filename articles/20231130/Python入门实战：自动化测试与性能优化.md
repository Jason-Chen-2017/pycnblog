```python
import time

def calculate_sum(numbers):
    start_time = time.time()
    total = 0
    for number in numbers:
        total += number
    end_time = time.time()
    return total, end_time - start_time

numbers = [i for i in range(1000000)]
total, time_taken = calculate_sum(numbers)
print(f'Total: {total}, Time taken: {time_taken} seconds')
```
