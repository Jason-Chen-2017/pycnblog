- TCP/IP协议：TCP/IP协议是一种面向连接的、可靠的、基于字节流的传输层通信协议。下面我们来详细讲解TCP/IP协议的具体代码实例和详细解释说明。

```python
import socket

def tcp_client():
    # 创建socket对象
    client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

    # 连接服务器
    server_address = ('127.0.0.1', 12345)
    client_socket.connect(server_address)

    # 发送数据
    data = b'Hello, World!'
    client_socket.send(data)

    # 接收数据
    data = client_socket.recv(1024)
    print(data)

    # 关闭socket
    client_socket.close()

if __name__ == '__main__':
    tcp_client()
```

def data_compress():
