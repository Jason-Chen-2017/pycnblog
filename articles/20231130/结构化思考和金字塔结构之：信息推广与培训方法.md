   PR(A) = (1-d) + d * Σ(PR(t) / C(t))

   其中，PR(A) 表示页面 A 的权重，d 是拓扑下降因子（通常为0.85），PR(t) 表示页面 t 的权重，C(t) 表示页面 t 出链的页面数量。

from bs4 import BeautifulSoup

def get_page_rank(url):
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    links = soup.find_all('a')
    return calculate_page_rank(links)

def calculate_page_rank(links):
    total_links = len(links)
    page_rank = {}
    for link in links:
        href = link.get('href')
        if href not in page_rank:
            page_rank[href] = 1 / total_links
        else:
            page_rank[href] += 1 / total_links
    return page_rank
```

import numpy as np

def k_means(data, k):
    centroids = initialize_centroids(data, k)
    while True:
        distances = calculate_distances(data, centroids)
        new_centroids = calculate_new_centroids(data, distances, centroids)
        if np.all(distances == calculate_distances(data, new_centroids)):
            break
        centroids = new_centroids
    return centroids

def initialize_centroids(data, k):
    n_samples, n_features = data.shape
    centroids = data[np.random.choice(n_samples, k, replace=False)]
    return centroids

def calculate_distances(data, centroids):
    n_samples, n_features = data.shape
    distances = np.zeros((n_samples, k))
    for i in range(n_samples):
        for j in range(k):
            distance = np.linalg.norm(data[i] - centroids[j])
            distances[i, j] = distance
    return distances

def calculate_new_centroids(data, distances, centroids):
    n_samples, n_features = data.shape
    new_centroids = np.zeros((k, n_features))
    for j in range(k):
        closest_samples = np.argmin(distances, axis=1) == j
        new_centroid = np.mean(data[closest_samples], axis=0)
        new_centroids[j] = new_centroid
    return new_centroids
```
