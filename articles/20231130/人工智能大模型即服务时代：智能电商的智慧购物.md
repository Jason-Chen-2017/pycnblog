```python
user_buy_record = [
    [1, 2, 3],
    [2, 3, 4],
    [3, 4, 5],
    [4, 5, 6],
    [5, 6, 7]
]

# 商品特征
product_features = [
    [1, 2, 3],
    [2, 3, 4],
    [3, 4, 5],
    [4, 5, 6],
    [5, 6, 7]
]

# 计算用户购买记录的相似度
user_similarity = cosine_similarity(user_buy_record)

# 计算商品特征的相似度
product_similarity = cosine_similarity(product_features)

# 推荐商品
def recommend_products(user_id, user_similarity, product_similarity):
    # 获取用户购买记录
    user_buy_record = user_buy_record[user_id]

    # 计算用户购买记录与商品特征的相似度
    product_similarity_with_user_buy_record = cosine_similarity(user_buy_record.reshape(-1, 1), product_features)

    # 获取与用户购买记录最相似的商品
    recommended_products = product_features[product_similarity_with_user_buy_record.argsort()[-5:]]

    # 返回推荐商品
    return recommended_products

# 推荐商品
user_id = 0
recommended_products = recommend_products(user_id, user_similarity, product_similarity)
print(recommended_products)
```
