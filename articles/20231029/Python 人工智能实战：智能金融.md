from nltk import sent_tokenize, word_tokenize
from nltk.corpus import stopwords
from spaCy.lang.en import EnglishLanguageModel

nltk.download('punkt')
nltk.download('stopwords')

text = "央行降准对于股市影响很大"
doc = EnglishLanguageModel()
stop_words = set(stopwords.words('chinese'))

sentences = sent_tokenize(text)
tokens = [word_tokenize(sentence) for sentence in sentences if sentence.lower() not in stop_words]

def get_feature(tokens):
    return dict([(word, True) for word in tokens])

feats = [[get_feature(sentence)] for sentence in tokens]
