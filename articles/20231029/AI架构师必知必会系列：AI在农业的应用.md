# 遍历所有轮廓线，计算面积和面积比

aspect_ratios = []
for contour in contours:
    area = cv2.contourArea(contour)
    perimeter = cv2.arcLength(contour, True)
    aspect_ratio = float(area)/perimeter
    areas.append(area)
    aspect_ratios.append(aspect_ratio)

# 对面积和面积比进行分析，判断是否存在特定的农田对象
if len(areas) > 1 and sum(areas)/len(areas) > 0.8:
