import numpy as np

def calculate_cooperation_degree(matrix):
    n = matrix.shape[0]
    cooperation_degree = np.zeros(n)
    for i in range(n):
        for j in range(n):
            cooperation_degree[i] += matrix[i, j]
    cooperation_degree = cooperation_degree / (n - 1)
    return cooperation_degree

def calculate_coordination_degree(matrix):
    n = matrix.shape[0]
    coordination_degree = np.zeros(n)
    for i in range(n):
        for j in range(n):
            coordination_degree[i] += matrix[i, j]
    coordination_degree = coordination_degree / (n - 1)
    return coordination_degree

def calculate_collaboration_degree(matrix):
    n = matrix.shape[0]
    collaboration_degree = np.zeros(n)
    for i in range(n):
        for j in range(n):
            collaboration_degree[i] += matrix[i, j]
    collaboration_degree = collaboration_degree / (n - 1)
    return collaboration_degree

def music_cooperation(matrix):
    cooperation_degree = calculate_cooperation_degree(matrix)
    coordination_degree = calculate_coordination_degree(matrix)
    collaboration_degree = calculate_collaboration_degree(matrix)
    return cooperation_degree, coordination_degree, collaboration_degree

print("协作度:", collaboration_degree)
