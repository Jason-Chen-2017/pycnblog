```python
    {
        "source_ip": "192.168.1.0/24",
        "destination_ip": "0.0.0.0/0",
        "protocol": "tcp",
        "source_port": "1-65535",
        "destination_port": "80",
        "action": "accept"
    },
    {

        "destination_port": "80",
        "action": "reject"
    }

```
