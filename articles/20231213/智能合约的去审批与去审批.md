```solidity

contract SimpleContract {
    address public owner;
    uint public balance;

    constructor() public {
        owner = msg.sender;
        balance = 100 ether;
    }

    function withdraw(uint _amount) public {
        require(msg.sender == owner);
        require(_amount <= balance);
        balance -= _amount;
        msg.sender.transfer(_amount);
    }
}
```
