def f(x):
    return x**2 + 5*x + 4

def update_velocity(w, c1, c2, r1, r2, x_best, x_gbest, v):
    return w * v + c1 * r1 * (x_best - x_gbest) + c2 * r2 * (x_gbest - x_gbest)

def update_position(x, v):
    return x + v

def main():
    num_particles = 50
    max_iterations = 100
    w = 0.7
    c1 = 1.5
    c2 = 1.5

    x = np.random.uniform(-5, 5, num_particles)
    v = np.random.uniform(-5, 5, num_particles)

    for _ in range(max_iterations):
        r1 = np.random.rand()
        r2 = np.random.rand()

        x_best = np.argmin(f(x))
        x_gbest = np.argmin(f(x))

        v = update_velocity(w, c1, c2, r1, r2, x_best, x_gbest, v)
        x = update_position(x, v)

    print("Global best solution:", x_gbest)

if __name__ == "__main__":
    main()
