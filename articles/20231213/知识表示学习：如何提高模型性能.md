```python
import networkx as nx
import numpy as np

# 创建一个图
G = nx.Graph()

# 添加节点
G.add_node(1)
G.add_node(2)
G.add_node(3)

# 添加边
G.add_edge(1, 2)
G.add_edge(2, 3)

# 获取邻接矩阵
adjacency_matrix = nx.adjacency_matrix(G).todense()

# 获取邻接表
adjacency_list = nx.adjacency_list(G)

# 获取图的度
degrees = nx.degree(G)

# 获取图的最短路径
shortest_path = nx.shortest_path(G, source=1, target=3)

# 获取图的最大秩
max_rank = nx.graph_max_rank(G)

# 获取图的最小秩
min_rank = nx.graph_min_rank(G)

# 获取图的中心性
centrality = nx.degree_centrality(G)

# 获取图的聚类系数
clustering_coefficient = nx.transitivity(G)

# 获取图的平均路径长度
average_path_length = nx.average_shortest_path_length(G)

# 获取图的平均度
average_degree = nx.average_degree(G)

# 获取图的平均克隆度
average_clustering = nx.average_clustering(G)
```
