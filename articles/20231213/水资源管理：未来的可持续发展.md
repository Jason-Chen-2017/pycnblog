```python
import numpy as np
from scipy.optimize import linprog

# 水资源的供应和需求数据
supply = [100, 200, 300]
demand = [50, 150, 250]

# 建立优化模型
c = [-1] * len(supply)  # 目标函数系数
A = np.array([[1, 1, 1]])  # 约束条件矩阵
b = np.array([sum(supply)])  # 约束条件向量

# 求解最优解
x = linprog(c, A_ub=A, b_ub=b, bounds=[(0, None)] * len(supply))

# 输出结果

```
