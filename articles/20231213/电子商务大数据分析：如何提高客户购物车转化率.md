```python
import pandas as pd
import numpy as np

# 数据清洗
def clean_data(data):
    data = data.dropna()
    return data

# 数据转换
def transform_data(data):
    data = pd.get_dummies(data)
    return data

# 数据归一化
def normalize_data(data):
    data = (data - data.mean()) / data.std()
    return data

# 数据预处理
def preprocess_data(data):
    data = clean_data(data)
    data = transform_data(data)
    data = normalize_data(data)
    return data
```
