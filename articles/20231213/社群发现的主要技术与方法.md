# 3. 社群发现的具体代码实例和详细解释说明

在本节中，我们将提供一个基于距离的社群发现算法的具体代码实例和详细解释说明。

```python
import networkx as nx
import numpy as np

def social_network_distance(G, u, v):
    # 计算人们u和v之间的距离
    dist = np.linalg.norm(G.nodes[u]['coordinates'] - G.nodes[v]['coordinates'])
    return dist

def social_network_community_detection(G, dist_threshold):
    # 初始化社群集合
    communities = {i: [] for i in G.nodes()}
    # 将每个节点放入初始社群
    for node in G.nodes():
        communities[node].append(node)
    # 计算每个节点之间的距离
    distances = {(u, v): social_network_distance(G, u, v) for u, v in G.edges()}
    # 迭代更新社群
    while len(distances) > 0:
        # 选择最短距离的边
        min_distance, min_edge = min(distances.items(), key=lambda x: x[1])
        # 将边的两个节点合并到同一社群
        u, v = min_edge
        communities[u].extend(communities[v])
        communities[v] = []
        # 更新距离
        for node in G.nodes():
            if node in communities[u]:
                for other_node in communities[v]:
                    distances[(node, other_node)] = min(distances.get((node, other_node), float('inf')), dist_threshold)
            elif node in communities[v]:
                for other_node in communities[u]:
                    distances[(node, other_node)] = min(distances.get((node, other_node), float('inf')), dist_threshold)
        # 移除已经合并过的边
        distances.pop(min_edge, None)
    # 返回社群集合
    return communities
```
