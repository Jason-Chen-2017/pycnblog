import pandas as pd
import matplotlib.pyplot as plt

# 数据收集
data = pd.read_csv('data.csv')

# 数据预处理
data['X_norm'] = (data['X'] - data['X'].min()) / (data['X'].max() - data['X'].min())

# 数据分析
data['Y'] = data['X'].mean()

# 数据挖掘
data['Z'] = (data['X'] - data['Y']).var()

# 数据可视化
plt.hist(data['X'], bins=20, density=True)
plt.xlabel('X')
plt.ylabel('P(X)')
plt.show()
