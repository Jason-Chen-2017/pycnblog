```python

import shutil

def copy_file(src_file, dst_file):
    try:
        shutil.copy2(src_file, dst_file)
    except Exception as e:
        print("Copy file failed: ", e)
        # 错误处理
        if os.path.exists(dst_file):
            os.remove(dst_file)
        # 错误恢复
        copy_file(src_file, dst_file)

copy_file("src_file", "dst_file")
```
