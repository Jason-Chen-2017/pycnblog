```python
import numpy as np
import matplotlib.pyplot as plt

def queue_model(lambda_, mu_, t_max):
    L = np.zeros(t_max + 1)
    W = np.zeros(t_max + 1)
    N = np.zeros(t_max + 1)

    L[0] = lambda_
    N[0] = 1

    for t in range(1, t_max + 1):
        N[t] = N[t - 1] + lambda_ - mu_
        L[t] = L[t - 1] + N[t]
        W[t] = L[t] / lambda_

    return L, W, N

lambda_ = 10
mu_ = 5
t_max = 100

L, W, N = queue_model(lambda_, mu_, t_max)

plt.plot(L)
plt.plot(W)
plt.plot(N)
plt.show()
```
