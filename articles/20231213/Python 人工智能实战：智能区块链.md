        return self.calculate_hash() == hashlib.sha256(signature.encode()).hexdigest()

    def execute(self, public_key, private_key):

private_key = "Alice's private key"
public_key = "Alice's public key"
signature = smart_contract.sign(private_key)

smart_contract.execute(public_key, signature)
