```python
import math

def distance(x1, y1, x2, y2):
    # 计算两点之间的距离公式
    return math.sqrt((x2 - x1)**2 + (y2 - y1)**2)

# 测试

x2 = 1
y2 = 1
print(distance(x1, y1, x2, y2)) # 输出：1.0
```
