5. 判断是否到达最优解：如果蜻蜓群中的某个蜻蜓找到了最优解，则停止算法。否则，继续步骤2-4，直到满足终止条件。

# 4.具体代码实例和详细解释说明
以下是一个简单的蜻蜓优化算法的Python代码实例：

```python
import random
import numpy as np

class Firefly:
    def __init__(self, x, y, brightness):
        self.x = x
        self.y = y
        self.brightness = brightness

def initialize_fireflies(num_fireflies, lower_bound, upper_bound):
    fireflies = []
    for _ in range(num_fireflies):
        x = random.uniform(lower_bound, upper_bound)
        y = random.uniform(lower_bound, upper_bound)
        brightness = random.uniform(0, 1)
        fireflies.append(Firefly(x, y, brightness))
    return fireflies

def calculate_distance(firefly1, firefly2):
    return np.sqrt((firefly1.x - firefly2.x)**2 + (firefly1.y - firefly2.y)**2)

def update_brightness(fireflies, distance_threshold):
    for i in range(len(fireflies)):
        for j in range(i+1, len(fireflies)):
            distance = calculate_distance(fireflies[i], fireflies[j])
            if distance <= distance_threshold:
                fireflies[i].brightness += fireflies[j].brightness

def update_position(fireflies, alpha, beta, gamma):
    for firefly in fireflies:
        firefly.x += (alpha * random.random() - beta) * firefly.brightness + gamma * random.random()
        firefly.y += (alpha * random.random() - beta) * firefly.brightness + gamma * random.random()

def firefly_algorithm(num_fireflies, lower_bound, upper_bound, distance_threshold, max_iterations):
    fireflies = initialize_fireflies(num_fireflies, lower_bound, upper_bound)
    for _ in range(max_iterations):
        update_brightness(fireflies, distance_threshold)
        update_position(fireflies, alpha, beta, gamma)
    return fireflies

# 使用示例
num_fireflies = 50
lower_bound = -10
upper_bound = 10
distance_threshold = 1
max_iterations = 100

fireflies = firefly_algorithm(num_fireflies, lower_bound, upper_bound, distance_threshold, max_iterations)
best_firefly = min(fireflies, key=lambda firefly: firefly.brightness)
print("最优解：", best_firefly.x, best_firefly.y)
```
