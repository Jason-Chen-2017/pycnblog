        total_score += item['score']
    return total_score

def constraint_function(items):
    total_weight = 0
    for item in items:
        total_weight += item['weight']
    return total_weight <= 10

def selection_function(items):
    return max(items, key=lambda item: item['score'] / item['weight'])

def greedy_algorithm(items):
    selected_items = []
    while items:
        selected_item = selection_function(items)
        selected_items.append(selected_item)
        items.remove(selected_item)
    return selected_items

]

selected_items = greedy_algorithm(items)
