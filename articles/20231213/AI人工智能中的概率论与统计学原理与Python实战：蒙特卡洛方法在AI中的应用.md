import random

def monte_carlo(n, p):
    # 初始化变量
    success_count = 0
    failure_count = 0

    # 生成n个随机数
    for _ in range(n):
        if random.random() < p:
            success_count += 1
        else:
            failure_count += 1

    # 计算估计值
    estimate = success_count / n

    return estimate
