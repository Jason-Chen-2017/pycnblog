class RuleEngine:
    def __init__(self):
        self.rules = []

    def add_rule(self, condition, action):
        self.rules.append((condition, action))

    def execute(self, data):
        for condition, action in self.rules:
            if condition(data):
                action(data)

# 示例规则
def check_temperature(data):
    return data['temperature'] > 30

def turn_on_air_conditioner(data):
    data['air_conditioner'] = True

# 创建规则引擎

# 执行规则
rule_engine.execute(data)
