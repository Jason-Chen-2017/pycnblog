```python
import heapq

def resource_scheduler(tasks, resources):
    task_queue = []
    resource_queue = []

    for task in tasks:
        heapq.heappush(task_queue, (task.priority, task))

    for resource in resources:
        heapq.heappush(resource_queue, (resource.capacity, resource))

    while task_queue:
        task = heapq.heappop(task_queue)
        resource = heapq.heappop(resource_queue)

        if task.capacity <= resource.capacity:
            resource.assign_task(task)
        else:
            heapq.heappush(resource_queue, (resource.capacity, resource))

    return resources
```
