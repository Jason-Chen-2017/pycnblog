data = data.dropna()
data['temperature'] = (data['temperature'] - data['temperature'].mean()) / data['temperature'].std()
data['humidity'] = (data['humidity'] - data['humidity'].mean()) / data['humidity'].std()

# 环境参数的分析与预测
X = data[['temperature', 'humidity']]
y = data['air_quality']
model = LinearRegression()
model.fit(X, y)

# 环境参数的调整与优化
temperature_threshold = 0.5
humidity_threshold = 0.6
if model.predict([[temperature, humidity]]) < temperature_threshold:
    print('调整温度参数')
if model.predict([[temperature, humidity]]) < humidity_threshold:
    print('调整湿度参数')

# 环境感知技术的实现
def environment_perception(temperature, humidity):
    if temperature < temperature_threshold:
        print('调整温度参数')
    if humidity < humidity_threshold:
        print('调整湿度参数')

environment_perception(temperature, humidity)
