```python
from scipy.integrate import quad

# 定义粒子的波函数
def wave_function(x):
    return np.exp(-x**2)

# 计算粒子之间的相互作用力
def interaction_force(x):
    integral = quad(lambda x: x * wave_function(x), 0, x)
    return integral[0]

# 使用数学模型公式来描述粒子之间的相互作用力
def strong_interaction(x):
    return interaction_force(x)

# 主程序
x = np.linspace(-10, 10, 1000)
y = strong_interaction(x)

# 绘制图像
import matplotlib.pyplot as plt
plt.plot(x, y)
plt.xlabel('x')
plt.ylabel('y')
plt.title('Strong Interaction')
plt.show()
```
