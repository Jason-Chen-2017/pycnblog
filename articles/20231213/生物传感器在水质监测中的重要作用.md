import numpy as np
import matplotlib.pyplot as plt

# 生物传感器的敏感度
def sensitivity(OD, C):
    return (OD - OD_0) / C

# 生物传感器的线性范围
def linear_range(OD, OD_0, m):
    return (OD - OD_0) / m

# 生物传感器的响应时间
def response_time(C_0, V, n, F):
    return (C_0 * V) / (n * F)

# 生物传感器的应用
def water_quality_monitoring(C, OD):
    return C, OD

# 生物传感器的参数

m = 0.1
