from sensor_msgs.msg import LaserScan
from geometry_msgs.msg import Twist

class SmartMedicalDevice:
    def __init__(self):
        rospy.init_node('smart_medical_device', anonymous=True)
        self.scan_sub = rospy.Subscriber('/scan', LaserScan, self.scan_callback)
        self.cmd_vel_pub = rospy.Publisher('/cmd_vel', Twist, queue_size=10)

    def scan_callback(self, scan):
        # 获取扫描数据
        min_distance = scan.ranges[0]
        # 计算旋转角度
        angle = scan.angle_min + (scan.angle_increment * (scan.ranges.index(min_distance)))
        # 设置旋转速度
        twist = Twist()
        twist.angular.z = 0.5
        self.cmd_vel_pub.publish(twist)

if __name__ == '__main__':
    try:
        smart_medical_device = SmartMedicalDevice()
        rospy.spin()
    except rospy.ROSInterruptException:
        pass
