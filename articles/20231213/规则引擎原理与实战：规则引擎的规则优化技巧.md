# 创建知识库

# 加载规则文件

# 创建工作内存

# 创建规则会话
session = RuleSession(kb)

# 启动规则引擎
session.start()

# 添加事实数据
wm.insert(Fact(name="John"))

# 执行规则

# 获取结果

print(result)
