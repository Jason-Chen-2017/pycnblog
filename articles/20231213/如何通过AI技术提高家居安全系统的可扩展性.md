```python
    if confidence > 0.5:
        class_id = int(scores[i][1])
```

```python
import pyttsx3

# 创建语音识别和语音合成对象
recognizer = sr.Recognizer()

# 读取音频文件

    audio = recognizer.record(source)

# 进行语音识别

    text = recognizer.recognize_google(audio)
    print('You said:', text)

    # 根据语音命令执行控制
    if text.lower() == 'open door':
        engine.say('Opening door')
        engine.runAndWait()
    elif text.lower() == 'close door':
        engine.say('Closing door')
        engine.runAndWait()
    else:
        engine.say('I don\'t understand')
        engine.runAndWait()
except sr.UnknownValueError:

    print('Could not request results from Google Speech Recognition service; check your online connection')
```
