```python
import numpy as np

def calculate_gravity(m1, m2, distance):
    G = 6.67430e-11
    return G * m1 * m2 / (distance ** 2)

def calculate_acceleration(mass, force):
    return force / mass

def calculate_trajectory(initial_velocity, time, gravity):
    return initial_velocity * time + 0.5 * gravity * time ** 2

def main():
    m1 = 5.972e24  # mass of Earth
    m2 = 1.989e30  # mass of Sun
    distance = 1.496e11  # distance between Earth and Sun in meters
    initial_velocity = 29.5e3  # initial velocity of Earth in m/s
    time = 365.25 * 24 * 3600  # time in seconds

    gravity = calculate_gravity(m1, m2, distance)
    acceleration = calculate_acceleration(m1, gravity)
    trajectory = calculate_trajectory(initial_velocity, time, gravity)

    print("Gravity:", gravity)
    print("Acceleration:", acceleration)
    print("Trajectory:", trajectory)

if __name__ == "__main__":
    main()
```
