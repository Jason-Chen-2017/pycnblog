    c = np.random.rand(n+1)

    return c, phi

# 光子的相互作用
def interaction(V, a, a_dag):
    H = np.zeros((n+1, n+1))
    for i in range(n+1):
        for j in range(n+1):
            H[i, j] = V[i, j] * a[i].dot(a_dag[j].conj())
    return H

# 光子的干涉
def interference(psi, x):
    return np.abs(psi)**2

# 光子的相位相关性

    return np.inner(psi, psi[delta_x, :]) / np.inner(psi, psi)
