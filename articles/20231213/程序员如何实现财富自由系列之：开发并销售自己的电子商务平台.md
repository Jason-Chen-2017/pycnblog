# 4.具体代码实例和详细解释说明

在开发电子商务平台时，我们可以选择使用Python、Java、PHP等编程语言。以下是一个简单的Python代码实例，用于实现商品信息管理功能：

```python
import sqlite3

# 连接数据库
conn = sqlite3.connect('electronic_commerce.db')
cursor = conn.cursor()

# 创建商品信息表
cursor.execute('''
CREATE TABLE IF NOT EXISTS goods_info (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    price REAL NOT NULL,
    stock INTEGER NOT NULL
)
''')

# 添加商品信息
def add_goods_info(name, price, stock):
    cursor.execute('''
    INSERT INTO goods_info (name, price, stock) VALUES (?, ?, ?)
    ''', (name, price, stock))
    conn.commit()

# 修改商品信息
def update_goods_info(id, name=None, price=None, stock=None):
    if name is not None:
        cursor.execute('''
        UPDATE goods_info SET name=? WHERE id=?
        ''', (name, id))
    if price is not None:
        cursor.execute('''
        UPDATE goods_info SET price=? WHERE id=?
        ''', (price, id))
    if stock is not None:
        cursor.execute('''
        UPDATE goods_info SET stock=? WHERE id=?
        ''', (stock, id))
    conn.commit()

# 删除商品信息
def delete_goods_info(id):
    cursor.execute('''
    DELETE FROM goods_info WHERE id=?
    ''', (id,))
    conn.commit()

# 查询商品信息
def query_goods_info(id=None, name=None, price=None, stock=None):
    query = 'SELECT * FROM goods_info'
    params = []
    if id is not None:
        query += ' WHERE id=?'
        params.append(id)
    if name is not None:
        query += ' AND name=?'
        params.append(name)
    if price is not None:
        query += ' AND price=?'
        params.append(price)
    if stock is not None:
        query += ' AND stock=?'
        params.append(stock)
    cursor.execute(query, params)
    return cursor.fetchall()

# 关闭数据库连接
conn.close()
```
