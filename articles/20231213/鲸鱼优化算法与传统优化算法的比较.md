def update(whales, p, v):
    new_whales = []
    for whale in whales:
        new_whale = leap(whale, p, v)
        new_whales.append(new_whale)
    return new_whales

# 主函数
def main():
    pop_size = 50
    search_space = (-10, 10)
    max_iter = 100

    # 初始化鲸鱼群
    whales = init_whale(pop_size, search_space)

    # 主循环
    for _ in range(max_iter):
        # 计算适应度
        fitness_values = np.array([fitness(whale) for whale in whales])
