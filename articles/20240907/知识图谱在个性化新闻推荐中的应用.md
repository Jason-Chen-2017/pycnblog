                if token.dep_ == 'pobj':
                    relations.append({
                        "entity": ent.text,
                        "relation": token.head.text,
                        "relation_entity": token.text
                    })

    return entities, relations

def build_knowledge_graph(news_data):
    knowledge_graph = []

    for text in news_data:
        entities, relations = extract_entities_and_relations(text)

        for entity in entities:
            existing_entity = next((e for e in knowledge_graph if e['entity'] == entity), None)

            if existing_entity:
                existing_entity['relations'].extend(relations)
            else:
                knowledge_graph.append({
                    "entity": entity,
                    "relations": relations
                })

    return knowledge_graph

def sparse_matrix_factorization(R, k, num_iterations=1000, learning_rate=0.01):
    num_entities = R.shape[0]
    num_relations = R.shape[1]

    P = np.random.rand(num_entities, k)
    Q = np.random.rand(num_relations, k)

        for i in range(num_entities):
            for j in range(num_relations):
