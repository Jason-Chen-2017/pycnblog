```python
import tensorflow as tf

from tensorflow.keras.layers import Embedding, LSTM, Dense, Concatenate

# 假设 X 是输入矩阵，y 是标签向量
input_text = Input(shape=(sequence_length,))
input_entity = Input(shape=(entity_size,))

# 文本嵌入
embedding = Embedding(vocab_size, embedding_dim)(input_text)

# 循环神经网络
lstm_output = LSTM(units=50)(embedding)

# 实体嵌入
entity_embedding = Embedding(entity_size, entity_dim)(input_entity)

# 合并文本和实体嵌入
merged = Concatenate()([lstm_output, entity_embedding])

model = Model(inputs=[input_text, input_entity], outputs=output)
