def calculate_total_income(user_id, purchase_records):

        elif record['type'] == 'membership':
            total_income += record['price']
        elif record['type'] == 'ebook':

def recommend_courses(user_id, purchase_records, browse_records, click_records):
    course_interest_matrix = {}
    for record in purchase_records:
        course_id = record['course_id']
        course_interest_matrix[course_id] = course_interest_matrix.get(course_id, 0) + 1

    for record in browse_records:
        course_id = record['course_id']
        course_interest_matrix[course_id] = course_interest_matrix.get(course_id, 0) + 0.5

    for record in click_records:
        course_id = record['course_id']
        course_interest_matrix[course_id] = course_interest_matrix.get(course_id, 0) + 0.1

    sorted_courses = sorted(course_interest_matrix.items(), key=lambda x: x[1], reverse=True)
    recommended_courses = [course_id for course_id, _ in sorted_courses]

    return recommended_courses

def predict_churn(user_id, activity_data, purchase_data, usage_data):
    # 数据预处理
    activity_df = pd.DataFrame(activity_data)
    purchase_df = pd.DataFrame(purchase_data)
    usage_df = pd.DataFrame(usage_data)

    # 计算平均值
    avg_activity = activity_df.describe().loc['mean']
    avg_purchase = purchase_df.describe().loc['mean']

    model.fit(features, [1])  # 假设所有用户都流失

    # 预测流失概率
    probability = model.predict_proba(features)[0][1]
