#### 题目：实现一个简单的神经网络
**答案：**
```python
import numpy as np

def forward(x, weights):
    z = np.dot(x, weights)
    return z

def backward(z, dZ):
    dW = np.dot(dZ.T, x)
    dx = np.dot(dZ, weights.T)
    return dW, dx

x = np.array([1, 2, 3])
weights = np.array([[0.1, 0.2], [0.3, 0.4]])

z = forward(x, weights)
print("Z:", z)

dZ = np.array([[1, 1], [1, 1]])
dW, dx = backward(z, dZ)
print("dW:", dW)
print("dx:", dx)
```

#### 题目：实现一个简单的决策树分类器
**答案：**
```python
import numpy as np

def entropy(y):
    hist = np.bincount(y)
    ps = hist / len(y)
    return -np.sum(np.log2(ps))

def gini(y):
    hist = np.bincount(y)
    ps = hist / len(y)
    return 1 - np.sum(ps ** 2)

def best_split(X, y):
    best_idx, best_val, best_score = None, None, -1
    for idx in range(X.shape[1]):
        val = X[:, idx]
        unique_vals = np.unique(val)
        for v in unique_vals:
            left_idxs = val[val < v]
            right_idxs = val[val >= v]
            if len(left_idxs) == 0 or len(right_idxs) == 0:
                continue
            left_y = y[left_idxs]
            right_y = y[right_idxs]
            score = 0.5 * entropy(left_y) + 0.5 * entropy(right_y)
            if score > best_score:
                best_score = score
                best_idx = idx
                best_val = v
    return best_idx, best_val

X = np.array([[1, 1], [1, 0], [0, 1], [0, 0]])
y = np.array([1, 1, 0, 0])

best_idx, best_val = best_split(X, y)
print("Best split index:", best_idx)
print("Best split value:", best_val)
```
