    blob = TextBlob(text)

**3. 推荐系统示例**

```python
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

# 用户画像（示例）
user_profile = np.array([0.2, 0.3, 0.5])

# 推荐内容（示例）
content = np.array([
    [0.1, 0.4, 0.5],
    [0.3, 0.2, 0.4],
    [0.5, 0.1, 0.3]
])

# 推荐算法（示例）
def recommend_content(user_profile, content):
    similarity_scores = cosine_similarity(user_profile.reshape(1, -1), content)
    recommended_indices = np.argsort(similarity_scores)[0][-3:]
    return recommended_indices

# 推荐内容调用示例
recommended_indices = recommend_content(user_profile, content)
print("推荐内容索引：", recommended_indices)
```
