```python
def chatbot():
    while True:
        user_input = input("请输入您的问题：")
        if user_input == "退出":
            break
        if user_input == "你好":
            print("你好！有什么可以帮你的吗？")
        else:
            print("抱歉，我不理解您的问题。")

chatbot()
```
