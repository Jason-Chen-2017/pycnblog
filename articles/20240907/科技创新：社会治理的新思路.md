**代码示例：**
```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract ElectronicContract {
    // 合同结构体
    struct Contract {
        address sender;
        address receiver;
        string content;
        bool isSigned;
    }

    // 存储合同映射
    mapping(bytes32 => Contract) public contracts;

    // 签订合同
    function signContract(bytes32 contractId, string memory content) public {
        require(!contracts[contractId].isSigned, "Contract already signed");
        contracts[contractId] = Contract(msg.sender, address(0), content, false);
    }

    // 确认合同
    function confirmContract(bytes32 contractId) public {
        require(contracts[contractId].receiver == msg.sender, "Not the receiver");
        contracts[contractId].isSigned = true;
    }
}
```
