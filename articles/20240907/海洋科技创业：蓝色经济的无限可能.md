**源代码实例：**

```python
import networkx as nx

# 创建图
G = nx.Graph()

# 添加节点和边
G.add_nodes_from(['A', 'B', 'C', 'D'])
G.add_edges_from([('A', 'B', {'weight': 2}),
                  ('B', 'C', {'weight': 3}),
                  ('C', 'D', {'weight': 2}),
                  ('A', 'D', {'weight': 4})])

# 使用Dijkstra算法计算最短路径
shortest_path = nx.shortest_path(G, source='A', target='D', weight='weight')

# 输出最短路径
print(shortest_path)
```

```python
import pandas as pd

# 评估技术风险
def assess_technical_risk(technology_score):
    if technology_score > 8:
        return "低风险"
    elif technology_score > 5:
        return "中等风险"
    else:
        return "高风险"

# 评估市场风险
def assess_market_risk(market_score):
    if market_score > 8:
        return "低风险"
    elif market_score > 5:
        return "中等风险"
    else:
        return "高风险"

# 评估财务风险
def assess_financial_risk(financial_score):
    if financial_score > 8:
        return "低风险"
    elif financial_score > 5:
        return "中等风险"
    else:
        return "高风险"

# 构建风险评估表
assessment = pd.DataFrame({
    'Technology Score': [7],
    'Market Score': [6],
    'Financial Score': [8]
})

assessment['Technical Risk'] = assessment['Technology Score'].apply(assess_technical_risk)
assessment['Market Risk'] = assessment['Market Score'].apply(assess_market_risk)
assessment['Financial Risk'] = assessment['Financial Score'].apply(assess_financial_risk)

print(assessment)
```
