class InferenceMachine:
    def __init__(self, rules):
        self.rules = rules

    def infer(self, premises, conclusion):
        inferred = set()
        inferred.add(conclusion)
        
        while inferred:
            new_inferred = set()
            for rule in self.rules:
                if rule.requires().issubset(inferred):
                    inferred.add(rule.conclusion)
                    new_inferred.add(rule.conclusion)
            
            inferred = new_inferred
        
        return inferred

rules = [
    Rule("A", {"P": True, "Q": True}),
    Rule("B", {"P": True, "R": True}),

    Rule("D", {"A": True, "B": True, "C": True}),
    Rule("E", {"D": True, "P": True}),

premises = {"P": True}

inferred = inference_machine.infer(premises, conclusion)
print(inferred)  # 输出：{'A', 'B', 'C', 'D', 'E'}
