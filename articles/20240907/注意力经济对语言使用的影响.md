def high_frequency_words(text, n=10):
    words = re.findall(r'\b\w+\b', text.lower())
    word_counts = Counter(words)
    most_common = word_counts.most_common(n)
    return most_common

text = "大家好！💕今天是个好日子，我们一起出去玩吧！🎉😄"
