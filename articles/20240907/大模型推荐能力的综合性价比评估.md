```python
import numpy as np

def user_based_collaborative_filter(ratings_matrix, user_id, top_n=10):
    # 计算用户之间的相似度矩阵
    similarity_matrix = np.dot(ratings_matrix, ratings_matrix.T) / np.linalg.norm(ratings_matrix, axis=1)[:, np.newaxis]

    # 计算用户与所有其他用户的相似度
    similarity_scores = similarity_matrix[user_id]

    # 选择相似度最高的 top_n 个用户
    top_n_users = np.argsort(similarity_scores)[-top_n:]

    # 获取 top_n 个用户的推荐商品
    recommended_items = np.mean(ratings_matrix[top_n_users], axis=0)

    return recommended_items
```

def item_based_collaborative_filter(ratings_matrix, item_id, top_n=10):

    similarity_scores = similarity_matrix[item_id]

    # 获取 top_n 个物品的推荐用户
    recommended_users = np.mean(ratings_matrix[:, top_n_items], axis=1)
