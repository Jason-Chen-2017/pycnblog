```go
package time_management

import (
    "fmt"
    "time"
)

// 定义一个任务结构体
type Task struct {
    Name       string
    StartTime   time.Time
    EndTime    time.Time
}

// 添加任务
func AddTask(tasks []Task, task Task) []Task {
    tasks = append(tasks, task)
    return tasks
}

// 打印任务列表
func PrintTasks(tasks []Task) {
    for _, task := range tasks {
        fmt.Printf("%s: %s - %s\n", task.Name, task.StartTime.Format("2006-01-02 15:04:05"), task.EndTime.Format("2006-01-02 15:04:05"))
    }
}

func main() {
    tasks := []Task{
        {"任务1", time.Now().Add(-2 * time.Hour), time.Now().Add(-1 * time.Hour)},
        {"任务2", time.Now().Add(-1 * time.Hour), time.Now().Add(1 * time.Hour)},
        {"任务3", time.Now().Add(1 * time.Hour), time.Now().Add(2 * time.Hour)},
    }

    PrintTasks(tasks)
}
```

```java
        this.tasks = new HashMap<>();
    }

    public void addTask(String teamMember, String task) {

        manager.addTask("Alice", "整理数据");
```

```java
import java.util.*;

class Project {
    private String name;

        this.name = name;
        this.tasks = new ArrayList<>();
        this.taskAssignees = new HashMap<>();

        tasks.add(task);
    }

        taskAssignees.put(task, assignee);

        for (String task : tasks) {
            System.out.println("- " + task);
        }
        System.out.println("任务分配：");
        for (Map.Entry<String, String> entry : taskAssignees.entrySet()) {
            System.out.println("- " + entry.getKey() + "：由 " + entry.getValue() + " 负责");
        }
    }
}

public class ProjectManagement {
    public static void main(String[] args) {

        project.addTask("系统设计");
        project.addTask("开发");

        project.assignTask("需求分析", "Alice");
        project.assignTask("系统设计", "Bob");
        project.assignTask("开发", "Charlie");

        project.printTasks();

}
```
