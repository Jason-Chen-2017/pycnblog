```python
import os
import numpy as np

def calculate_color(file_path):
    file_name = os.path.basename(file_path)
    r, g, b = int(file_name[0]), int(file_name[1]), int(file_name[2])
    return np.array([r, g, b])

def calculate_position(color):
    return np.mean(color)

def hue_files(file_paths):
    colors = [calculate_color(file_path) for file_path in file_paths]
    positions = [calculate_position(color) for color in colors]
    return positions

file_paths = ['file1.jpg', 'file2.jpg', 'file3.jpg']
positions = hue_files(file_paths)
print(positions)
```
