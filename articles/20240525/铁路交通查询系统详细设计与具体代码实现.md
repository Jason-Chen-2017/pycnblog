```python
from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

@app.route('/query', methods=['GET'])
def query():
    station = request.args.get('station')
    response = requests.get(f'https://railway.api/{station}')
    data = response.json()
    return jsonify(data)

if __name__ == '__main__':
    app.run()
```
