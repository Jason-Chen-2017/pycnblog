4. 查询：查询是指根据一定条件搜索图书的过程。

        else:
            self.books.append(book)

    def remove_book(self, book_id):
        for b in self.books:
            if b.id == book_id:
                self.books.remove(b)
                print("Book removed.")
                return
        print("Book not found.")

    def search_books(self, search_str):
        results = [b for b in self.books if search_str in b.name or search_str in b.author]
        return results

    def statistics(self):
        total_books = len(self.books)
        author_stats = {}
        for b in self.books:
            if b.author in author_stats:
                author_stats[b.author] += 1
            else:
                author_stats[b.author] = 1
        return total_books, author_stats
