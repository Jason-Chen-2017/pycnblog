import numpy as np
from sklearn.cluster import KMeans

# 数据收集
data = pd.read_csv('user_behavior_data.csv')

# 数据清洗
data = data.dropna()
data = data.drop_duplicates()

# 数据分析

kmeans.fit(data)
labels = kmeans.predict(data)
data['cluster'] = labels
