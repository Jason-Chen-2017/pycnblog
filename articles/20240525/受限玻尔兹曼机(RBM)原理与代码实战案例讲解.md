```python
def sigmoid(x):
    return 1 / (1 + np.exp(-x))

# 定义正向传播函数
def forward(x):
    h = sigmoid(np.dot(W, x) + b)
    return h

# 定义反向传播函数
def backward(x, h, y):
    error = y - h
    dW = np.dot(x, error.T) * sigmoid(-np.dot(W, x) - b) * (1 - sigmoid(np.dot(W, x) + b))
    db = error * sigmoid(-np.dot(W, x) - b) * (1 - sigmoid(np.dot(W, x) + b))
    return dW, db

# 训练模型
x = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y = np.array([[0], [1], [1], [0]])
for epoch in range(1000):
    h = forward(x)
    dW, db = backward(x, h, y)
    W -= dW * 0.1
    b -= db * 0.1
```
