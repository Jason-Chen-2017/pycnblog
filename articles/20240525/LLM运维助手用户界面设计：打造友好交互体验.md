```python
import nltk
from nltk.chat.util import Chat, reflections

pairs = [
    [
        r"hi|hello|hey",
        ["Hello!", "Hey there!", "Hi there!"]
    ],
    [
        r"how are you?",
        ["I'm doing well, how about you?", "I'm fine, thanks for asking."]
    ],
    [
        r"what is your name?",
        ["I am an LLM Operative Assistant."]
    ]
]

def chatbot():
    print("Hi, I'm your LLM Operative Assistant. How can I help you?")
    chat = Chat(pairs, reflections)
    chat.converse()

if __name__ == "__main__":
    chatbot()
```
