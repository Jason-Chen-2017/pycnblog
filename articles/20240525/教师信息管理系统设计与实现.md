## 4. 项目实践：代码实例和详细解释说明

下面是一个简单的教师信息管理系统的代码示例，使用Python和Flask框架实现。

```python
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'mysql://username:password@localhost/teachers_db'
db = SQLAlchemy(app)

class Teacher(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(255), nullable=False)
    job_title = db.Column(db.String(255))
    department = db.Column(db.String(255))
    phone = db.Column(db.String(20))
    email = db.Column(db.String(255))

@app.route('/teachers', methods=['GET', 'POST'])
def teachers():
    if request.method == 'POST':
        teacher = Teacher(
            name=request.form['name'],
            job_title=request.form['job_title'],
            department=request.form['department'],
            phone=request.form['phone'],
            email=request.form['email']
        )
        db.session.add(teacher)
        db.session.commit()
        return jsonify({'message': 'Teacher added successfully'}), 201
    elif request.method == 'GET':
        teachers = Teacher.query.all()
        return jsonify([{'id': teacher.id, 'name': teacher.name, 'job_title': teacher.job_title, 'department': teacher.department, 'phone': teacher.phone, 'email': teacher.email} for teacher in teachers])

if __name__ == '__main__':
    app.run(debug=True)
```
