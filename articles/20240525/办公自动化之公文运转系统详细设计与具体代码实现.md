```python
    def get_status(self):

class ApprovalFlow:
    def __init__(self, documents):
        self.documents = documents

    def process(self, user, role):
        for document in self.documents:
            if document.get_status() == "created":
                document.update_status("in_progress")
                print(f"Document {document.id} has been processed by {role}")
            elif document.get_status() == "in_progress":
                document.update_status("approved")
                print(f"Document {document.id} has been approved by {role}")
```
