```python
import matplotlib.pyplot as plt

def generate_gantt(tasks):
    fig, ax = plt.subplots()
    for task in tasks:
        ax.barh(task['start'] - 0.2, task['duration'], left=task['start'], height=0.4, label=task['name'])
    ax.set_yticks(range(len(tasks)))
    ax.set_yticklabels([task['name'] for task in tasks])
    ax.invert_yaxis()
    ax.set_xlabel('Time')
    ax.set_title('Gantt Chart')
    plt.legend()
    plt.show()

tasks = [
    {'name': 'Task 1', 'start': 0, 'duration': 5},
    {'name': 'Task 2', 'start': 3, 'duration': 3},
    {'name': 'Task 3', 'start': 5, 'duration': 2},
]

generate_gantt(tasks)
```

1. 任务分配

以下是一个使用Python编写的最短作业优先（SJF）算法的代码示例：

```python
import heapq

def sjf(tasks):
    heap = []
    for task in tasks:
        heapq.heappush(heap, (task['duration'], task))
    schedule = []
    while heap:
        duration, task = heapq.heappop(heap)
        schedule.append(task)
    return schedule

tasks = [
    {'name': 'Task 1', 'duration': 5},
    {'name': 'Task 2', 'duration': 3},
    {'name': 'Task 3', 'duration': 2},
]

schedule = sjf(tasks)
print(schedule)
```
