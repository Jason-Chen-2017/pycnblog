```python
from pulp import *

# 目标函数
objective = LpMaximize(LpSum([lp.Variable(i) * c[i] for i in range(n)]))

# 约束条件
constraints = [LpSum([lp.Variable(i) for i in range(n)]) == P]

# 创建LP问题
prob = LpProblem("Optimize_Inventory", objective)

# 添加变量和约束
for i in range(n):
    prob += lp.Variable(i)
    prob += constraints[0]

# 求解问题

# 打印结果
for v in prob.variables():
    print(v.name, "=", v.varValue)
```
