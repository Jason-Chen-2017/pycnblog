```python
import hashlib

class IDCardManager:
    def __init__(self):
        self.db = {}

    def create(self, idcard_number, name, sex, birthdate):
        hash_value = hashlib.sha256(idcard_number.encode()).hexdigest()
        if hash_value in self.db:
            raise Exception("Identity card number already exists.")
        self.db[hash_value] = {
            "name": name,
            "sex": sex,
            "birthdate": birthdate
        }

    def query(self, idcard_number):
        hash_value = hashlib.sha256(idcard_number.encode()).hexdigest()
        if hash_value not in self.db:
            raise Exception("Identity card number not found.")
        return self.db[hash_value]

    def update(self, idcard_number, name=None, sex=None, birthdate=None):
        hash_value = hashlib.sha256(idcard_number.encode()).hexdigest()
        if hash_value not in self.db:
            raise Exception("Identity card number not found.")
        if name:
            self.db[hash_value]["name"] = name
        if sex:
            self.db[hash_value]["sex"] = sex
        if birthdate:
            self.db[hash_value]["birthdate"] = birthdate

    def delete(self, idcard_number):
        hash_value = hashlib.sha256(idcard_number.encode()).hexdigest()
        if hash_value not in self.db:
            raise Exception("Identity card number not found.")
        del self.db[hash_value]

```
