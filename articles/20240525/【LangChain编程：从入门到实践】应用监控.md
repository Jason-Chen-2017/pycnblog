```python
import langchain as lc
import numpy as np

# 定义度量指标
def measure_response_time(requests):
    response_times = []
    for request in requests:
        start_time = time.time()
        response = lc.request(request)
        end_time = time.time()
        response_times.append(end_time - start_time)
    return np.mean(response_times)

# 收集数据
requests = lc.generate_requests(1000)
response_times = measure_response_time(requests)

# 分析结果
response_time_distribution = np.histogram(response_times, bins=50)
print("Response Time Distribution:", response_time_distribution)

# 可视化展示
lc.plot(response_time_distribution)
```
