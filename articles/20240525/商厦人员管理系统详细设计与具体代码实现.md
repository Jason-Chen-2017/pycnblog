import datetime

class Employee:
    def __init__(self, name, sex, age, position, department):
        self.name = name
        self.sex = sex
        self.age = age
        self.position = position
        self.department = department

    def work(self, hours):
        self.hours = hours

    def take_leave(self, days, reason):
        self.days = days
        self.reason = reason

    def calculate_salary(self):
        basic_salary = 5000
        overtime = self.hours - 8
        if overtime > 0:
            overtime_fee = overtime * 200
        else:
            overtime_fee = 0
        total_salary = basic_salary + overtime_fee
        return total_salary

class Manager:
    def __init__(self):
        self.employees = []

    def add_employee(self, employee):
        self.employees.append(employee)

    def calculate_salaries(self):
        total_salaries = 0
        for employee in self.employees:
            total_salaries += employee.calculate_salary()
        return total_salaries
