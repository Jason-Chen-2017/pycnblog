### 4.1. 运动员注册

我们可以使用Python和Flask框架来实现运动员注册功能。以下是一个简单的示例：

```python
from flask import Flask, render_template, request
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///athletics.db'
db = SQLAlchemy(app)

class Athlete(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(50))
    age = db.Column(db.Integer)
    gender = db.Column(db.String(10))

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/register', methods=['GET', 'POST'])
def register():
    if request.method == 'POST':
        name = request.form['name']
        age = request.form['age']
        gender = request.form['gender']
        athlete = Athlete(name=name, age=age, gender=gender)
        db.session.add(athlete)
        db.session.commit()
    return render_template('register.html')

if __name__ == '__main__':
    app.run(debug=True)
```

### 4.2. 比赛安排

我们可以使用Python和Flask框架来实现比赛安排功能。以下是一个简单的示例：

```python
from flask import Flask, render_template, request
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///athletics.db'
db = SQLAlchemy(app)

class Competition(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    event = db.Column(db.String(50))
    time = db.Column(db.String(50))
    location = db.Column(db.String(50))

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/schedule', methods=['GET', 'POST'])
def schedule():
    if request.method == 'POST':
        event = request.form['event']
        time = request.form['time']
        location = request.form['location']
        competition = Competition(event=event, time=time, location=location)
        db.session.add(competition)
        db.session.commit()
    return render_template('schedule.html')

if __name__ == '__main__':
    app.run(debug=True)
```
