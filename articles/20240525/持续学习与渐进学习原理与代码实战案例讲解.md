```python
# 计算器项目的代码实例
class Calculator:
    def add(self, x, y):
        return x + y

    def sub(self, x, y):
        return x - y

    def mul(self, x, y):
        return x * y

    def div(self, x, y):
        if y == 0:
            raise ValueError("Cannot divide by zero")
        return x / y

# 实例化计算器并使用
calc = Calculator()
print(calc.add(1, 2))  # 输出 3
print(calc.sub(5, 3))  # 输出 2
print(calc.mul(4, 3))  # 输出 12
print(calc.div(9, 3))  # 输出 3
```
