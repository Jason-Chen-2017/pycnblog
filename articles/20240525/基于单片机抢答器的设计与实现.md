## 5. 项目实践：代码实例和详细解释说明

在本节中，我们将通过一个具体的项目实例，详细讲解如何实现基于单片机的抢答器。

```python
import random

class QuizMachine:
    def __init__(self):
        self.questions = [
            {"question": "What is the capital of France?", "options": ["Paris", "London", "Berlin"], "answer": "Paris"},
            {"question": "What is 2 + 2?", "options": ["3", "4", "5"], "answer": "4"},
        ]

    def ask_question(self):
        question = random.choice(self.questions)
        print(question["question"])
        for option in question["options"]:
            print(option)
        user_answer = input("Your answer: ")
        return user_answer

    def check_answer(self, user_answer, correct_answer):
        return user_answer.lower() == correct_answer.lower()

    def run(self):
        correct = 0
        total = len(self.questions)
        for question in self.questions:
            user_answer = self.ask_question()
            if self.check_answer(user_answer, question["answer"]):
                print("Correct!")
                correct += 1
            else:
                print(f"Wrong! The correct answer is {question['answer']}.")
        print(f"You got {correct} out of {total} questions correct.")

if __name__ == "__main__":
    machine = QuizMachine()
    machine.run()
```
