```python
import numpy as np
import matplotlib.pyplot as plt

def simulated_annealing(func, x0, T0, alpha, bounds):
    T = T0
    x = x0
    history = [x]

    while T > 1e-6:
        x_new = x + alpha * np.random.normal(size=x.shape)
        x_new = np.clip(x_new, bounds[:, 0], bounds[:, 1])

        if func(x_new) < func(x):
            x = x_new
            history.append(x)

        T *= np.exp(-alpha / T)

    return np.array(history)

func = lambda x: np.sum(x**2)
x0 = np.array([10, 10])
T0 = 1000
alpha = 0.1
bounds = np.array([[-10, -10], [10, 10]])

history = simulated_annealing(func, x0, T0, alpha, bounds)
plt.plot(history[:, 0], history[:, 1])
plt.show()
```
