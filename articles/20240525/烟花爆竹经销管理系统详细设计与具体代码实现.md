## 5. 项目实践：代码实例和详细解释说明

在实际项目中，我们需要编写代码来实现上述设计。以下是一个简单的代码示例：

```python
# 商品管理类
class GoodsManager:
    def __init__(self):
        self.goods = {}

    def add_goods(self, name, price, stock):
        self.goods[name] = {'price': price, 'stock': stock}

    def sell_goods(self, name, quantity):
        if name in self.goods and self.goods[name]['stock'] >= quantity:
            self.goods[name]['stock'] -= quantity
            return True
        return False

# 订单管理类
class OrderManager:
    def __init__(self):
        self.orders = []

    def create_order(self, goods_name, quantity, customer_id):
        order = {'goods_name': goods_name, 'quantity': quantity, 'customer_id': customer_id}
        self.orders.append(order)
        return order

# 客户管理类
class CustomerManager:
    def __init__(self):
        self.customers = {}

    def add_customer(self, name, contact):
        self.customers[name] = {'contact': contact}

    def get_customer(self, name):
        return self.customers[name]

# 财务管理类
class FinanceManager:
    def __init__(self):
        self.sales = {}
        self.profits = {}

    def record_sale(self, goods_name, quantity, price):
        if goods_name not in self.sales:
            self.sales[goods_name] = 0
        self.sales[goods_name] += quantity * price

    def calculate_profit(self):
        for goods_name, total_sale in self.sales.items():
            if goods_name not in self.profits:
                self.profits[goods_name] = 0
            self.profits[goods_name] = total_sale - self.profits[goods_name]
```
