## 4. 项目实践：代码实例和详细解释说明

在本节中，我们将使用Python和Flask框架来实现鲜花销售管理系统的前端和后端。以下是一个简化的代码示例，用于创建客户、订单和销售额表，并执行一些基本操作。

```python
import flask
from flask_sqlalchemy import SQLAlchemy

app = flask.Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///flower_shop.db'
db = SQLAlchemy(app)

class Customer(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(50), nullable=False)
    contact = db.Column(db.String(50), nullable=False)
    address = db.Column(db.String(100), nullable=False)

class Order(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    customer_id = db.Column(db.Integer, nullable=False)
    date = db.Column(db.Date, nullable=False)
    total = db.Column(db.Float, nullable=False)
    paid = db.Column(db.Boolean, nullable=False)

class Sale(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    order_id = db.Column(db.Integer, nullable=False)
    date = db.Column(db.Date, nullable=False)
    amount = db.Column(db.Float, nullable=False)

@app.route('/customers', methods=['GET', 'POST'])
def customers():
    if flask.request.method == 'POST':
        customer = Customer(name=flask.request.form['name'], contact=flask.request.form['contact'], address=flask.request.form['address'])
        db.session.add(customer)
        db.session.commit()
        return flask.redirect('/customers')
    customers = Customer.query.all()
    return flask.render_template('customers.html', customers=customers)

@app.route('/orders', methods=['GET', 'POST'])
def orders():
    if flask.request.method == 'POST':
        order = Order(customer_id=flask.request.form['customer_id'], date=flask.request.form['date'], total=flask.request.form['total'], paid=flask.request.form['paid'] == 'true')
        db.session.add(order)
        db.session.commit()
        return flask.redirect('/orders')
    orders = Order.query.all()
    return flask.render_template('orders.html', orders=orders)

@app.route('/sales', methods=['GET', 'POST'])
def sales():
    if flask.request.method == 'POST':
        sale = Sale(order_id=flask.request.form['order_id'], date=flask.request.form['date'], amount=flask.request.form['amount'])
        db.session.add(sale)
        db.session.commit()
        return flask.redirect('/sales')
    sales = Sale.query.all()
    return flask.render_template('sales.html', sales=sales)

if __name__ == '__main__':
    db.create_all()
    app.run(debug=True)
```
