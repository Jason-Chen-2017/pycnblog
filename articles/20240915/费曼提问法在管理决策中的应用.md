        user_experience_demand = ask_question("请描述用户体验需求：")

    else:
        return None, None

    """方案评估"""

        print(f"{i+1}. {scheme}")
    
    scheme_choice = ask_question("请选择方案（1/2/3...）：")

    """风险控制"""
