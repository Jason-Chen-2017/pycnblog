```python
import numpy as np
import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

# 加载用户行为数据
data = pd.read_csv('user行为数据.csv')
users = data['用户ID'].unique()
items = data['课程ID'].unique()

# 构建用户-物品评分矩阵
ratings = pd.pivot_table(data, index='用户ID', columns='课程ID', values='评分').fillna(0)

# 计算用户-物品相似度
similarity_matrix = cosine_similarity(ratings.values)

# 为用户推荐课程
def recommend_courses(user_id, similarity_matrix, ratings, top_n=5):
    user_ratings = ratings.loc[user_id]
    user_similarity = similarity_matrix[user_id]
    course_similarity = user_similarity.dot(user_ratings)
    recommended_courses = course_similarity.argsort()[::-1]
    return recommended_courses[top_n+1:]

# 测试推荐效果
user_id = 1001
recommended_courses = recommend_courses(user_id, similarity_matrix, ratings)
print("推荐的课程ID：", recommended_courses)

以下是一个简单的协同过滤推荐算法的实现示例：

```python
import numpy as np
import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

# 加载用户行为数据
data = pd.read_csv('user行为数据.csv')
users = data['用户ID'].unique()
items = data['课程ID'].unique()

# 构建用户-物品评分矩阵
ratings = pd.pivot_table(data, index='用户ID', columns='课程ID', values='评分').fillna(0)

# 计算用户-物品相似度
similarity_matrix = cosine_similarity(ratings.values)

# 为用户推荐课程
def recommend_courses(user_id, similarity_matrix, ratings, top_n=5):
    user_ratings = ratings.loc[user_id]
    user_similarity = similarity_matrix[user_id]
    course_similarity = user_similarity.dot(user_ratings)
    recommended_courses = course_similarity.argsort()[::-1]
    return recommended_courses[top_n+1:]

# 测试推荐效果
user_id = 1001
recommended_courses = recommend_courses(user_id, similarity_matrix, ratings)
print("推荐的课程ID：", recommended_courses)
```
