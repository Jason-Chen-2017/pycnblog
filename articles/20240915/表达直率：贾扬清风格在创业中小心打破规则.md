```python
import numpy as np
import matplotlib.pyplot as plt

# 定义市场洞察、技术创新、团队协作、快速行动和传统规则限制的权重
weights = {'市场洞察': 0.3, '技术创新': 0.3, '团队协作': 0.2, '快速行动': 0.2, '传统规则限制': 0.2}

# 定义创新函数
def innovation(situation):
    market_insight = situation['市场洞察']
    technological_innovation = situation['技术创新']
    team_collaboration = situation['团队协作']
    rapid_action = situation['快速行动']
    traditional_rule_limitation = situation['传统规则限制']
    
    # 计算创新值
    innovation_value = (market_insight + technological_innovation + team_collaboration + rapid_action) / traditional_rule_limitation
    
    return innovation_value

# 示例数据
situation = {
    '市场洞察': 0.8,
    '技术创新': 0.9,
    '团队协作': 0.7,
    '快速行动': 0.8,
    '传统规则限制': 0.5
}

# 计算创新值
innovation_value = innovation(situation)
print(f"创新值：{innovation_value}")

# 绘制创新曲线
x = np.linspace(0, 1, 100)
y = innovation({k: v * weights[k] for k, v in situation.items()})
plt.plot(x, y)
plt.xlabel('因素值')
plt.ylabel('创新值')
plt.title('创新曲线')
plt.show()
```

```python
import numpy as np
import matplotlib.pyplot as plt

# 定义市场洞察、技术创新、团队协作、快速行动和传统规则限制的权重
weights = {'市场洞察': 0.3, '技术创新': 0.3, '团队协作': 0.2, '快速行动': 0.2, '传统规则限制': 0.2}

# 定义创新函数
def innovation(situation):
    market_insight = situation['市场洞察']
    technological_innovation = situation['技术创新']
    team_collaboration = situation['团队协作']
    rapid_action = situation['快速行动']
    traditional_rule_limitation = situation['传统规则限制']
    
    # 计算创新值
    innovation_value = (market_insight + technological_innovation + team_collaboration + rapid_action) / traditional_rule_limitation
    
    return innovation_value

# 示例数据
situation = {
    '市场洞察': 0.8,
    '技术创新': 0.9,
    '团队协作': 0.7,
    '快速行动': 0.8,
    '传统规则限制': 0.5
}

# 计算创新值
innovation_value = innovation(situation)
print(f"创新值：{innovation_value}")

# 绘制创新曲线
x = np.linspace(0, 1, 100)
y = innovation({k: v * weights[k] for k, v in situation.items()})
plt.plot(x, y)
plt.xlabel('因素值')
plt.ylabel('创新值')
plt.title('创新曲线')
plt.show()
```
