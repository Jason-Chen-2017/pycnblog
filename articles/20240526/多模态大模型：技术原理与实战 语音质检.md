        self.dense2 = tf.keras.layers.Dense(64, activation='relu')
        self.dense3 = tf.keras.layers.Dense(1, activation='sigmoid')

    def call(self, inputs):

        x = tf.keras.layers.Concatenate()([x1, x2, x3])
        x = self.dense1(x)
        x = self.dense2(x)
        return self.dense3(x)
