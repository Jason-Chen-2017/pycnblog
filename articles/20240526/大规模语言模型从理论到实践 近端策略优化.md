# 定义梯度下降法
def gradient_descent(X, y, theta, alpha, iterations):
    m = len(y)
    for i in range(iterations):
        predictions = np.dot(X, theta)
        errors = predictions - y
        gradients = np.dot(X.T, errors) / m
        theta = theta - alpha * gradients
    return theta
