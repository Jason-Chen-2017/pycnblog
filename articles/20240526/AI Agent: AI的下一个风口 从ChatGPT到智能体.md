```python
import numpy as np

def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def forward(x, W, b):
    y = np.dot(W, x) + b
    return sigmoid(y)

W = np.random.rand(2, 1)
b = np.random.rand(1)
x = np.array([[0.5, 0.5]])
y = forward(x, W, b)
print(y)
```
