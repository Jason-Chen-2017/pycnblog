import requests
from bs4 import BeautifulSoup
import json
import mysql.connector

def get_player_data(player_id):
    url = f"https://example.com/api/player/{player_id}"
    response = requests.get(url)

    if response.status_code == 200:
        data = json.loads(response.text)
        return data
    else:
        return None

def create_player(player_data):
    db = mysql.connector.connect(
        host="localhost",
        user="username",
        password="password",
        database="database"
    )
    cursor = db.cursor()

    sql = "INSERT INTO players (name, position, team_id, team_name, contract_year, salary) VALUES (%s, %s, %s, %s, %s, %s)"
    cursor.execute(sql, (player_data["name"], player_data["position"], player_data["team_id"], player_data["team_name"], player_data["contract_year"], player_data["salary"]))

    db.commit()
    cursor.close()
    db.close()


# 示例：利用数字化技术提升球员的商业价值

player_id = 100
player_data = {
    "name": "李华",
    "position": "前锋",
    "team_id": 2,
    "team_name": "ABC篮球队",

}

response = requests.post(

        "team_id": player_data["team_id"],

        "salary": player_data["salary"]

)

data = response.json()

if data:
    print(f"成功添加球员信息：{data['name']}，{data['position']}，{data['team_id']}，{data['team_name']}，{data['contract_year']}，{data['salary']}")
else:
    print("添加球员信息失败")
else:
    print("保存球员信息成功")
