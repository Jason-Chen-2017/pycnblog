4.3. 核心代码实现

```python
# 定义最小生成树函数
def generate_tree(graph):
    # 初始化一个空生成树
    tree = [0] * 5000
    # 遍历所有的顶点
    for i in range(5000):
        # 找到距离当前顶点最近的边
        neighbor = find_neighbor(i, graph)
        # 更新当前顶点的权值
        tree[i] = min(tree[i], neighbor)
    return tree

# 定义查找边函数
def find_neighbor(i, graph):
    # 遍历所有的边
    for edge in graph:
        # 计算当前顶点与边之间的权值
        weight = edge[i]
        # 更新当前顶点的权值
        if edge[i] == weight:
            return edge[i]
    # 返回距离当前顶点最近的边
    return weight

# 实现最小生成树算法
tree = generate_tree(graph)

# 打印结果

print(tree)
```
