```python
import random

def turn_device(device, state):
    if state:
        # 打开设备
        device.open()
    else:
        # 关闭设备
        device.close()

def main():
    # 初始化设备
    device1 = random.choice(['device1', 'device2'])
    device2 = random.choice(['device1', 'device2'])
    device = {'device1': turn_device(device1, True), 'device2': turn_device(device2, False)}
    
    # 循环接收用户语音命令
    while True:
        user_input = input('请告诉我你想要控制的设备：')
        
        # 根据用户输入调整设备状态
        if user_input.lower() == 'up':
            turn_device(device1, True)
        elif user_input.lower() == 'down':
            turn_device(device1, False)
        elif user_input.lower() == 'left':
            turn_device(device2, True)
        elif user_input.lower() == 'right':
            turn_device(device2, False)
        else:
            print('输入有误，请重新输入！')
            continue
        
        # 更新设备状态
        turn_device(device1, False)
        turn_device(device2, True)
        
    # 关闭设备
    device1.close()
    device2.close()

if __name__ == '__main__':
    main()
```
