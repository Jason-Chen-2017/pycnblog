首先，我们将用户行为数据和商品属性数据进行稀疏表示，以便于后续的卷积操作。然后，我们使用半监督图卷积网络构建商品推荐系统。

```
import numpy as np
import tensorflow as tf
from tensorflow.keras.layers import Input, Conv2D, MaxPooling2D, Flatten, Dense, Dropout

# 定义商品特征
class ProductFeature(Input):
    def __init__(self, input_shape):
        super(ProductFeature, self).__init__(input_shape)

    def call(self, inputs):
        return Conv2D(32, kernel_size=3, padding='same', activation='relu')(inputs)

# 定义卷积层
class Conv2DLayer(Layer):
    def __init__(self, input_shape, num_classes):
        super(Conv2DLayer, self).__init__()
        self.conv = Conv2D(input_shape[1], kernel_size=3, padding='same', activation='relu')
        self.conv_dropout = Dropout(0.25)
        self.conv_norm = Flatten()
        self.conv_out = self.conv(self.conv_dropout)
        self.conv_out = self.conv_norm(self.conv_out)
        self.output = self.conv_out

# 定义池化层
class MaxPooling2D(Layer):
    def __init__(self, pool_size):
        super(MaxPooling2D, self).__init__()
        self.pool = MaxPooling2D(pool_size=pool_size, kernel_size=2, padding='same')

    def call(self, inputs):
        return self.pool(inputs)

# 定义全连接层
class Flatten(Layer):
    def __init__(self):
        super(Flatten, self).__init__()

    def call(self, inputs):
        return inputs.flatten()

# 定义模型
model = tf.keras.Sequential([
    ProductFeature(input_shape=(28, 28)),
    Conv2DLayer(32, num_classes=10),
    MaxPooling2D(pool_size=2),
    Flatten(),
    Dense(128, activation='relu'),
    Dropout(0.5),
    Dense(10)
])

# 编译模型
model.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])

# 训练模型
model.fit(x=[{'train': [1, 2, 3], 'test': [4, 5, 6]},
          y=[{'train': [7, 8, 9], 'test': [10, 11, 12]},
          {'train': [13, 14, 15], 'test': [16, 17, 18]},
          {'train': [19, 20, 21], 'test': [22, 23, 24]}}
          ]
        )

# 评估模型
score = model.evaluate(x=[{'train': [25, 26, 27], 'test': [29, 30, 31]},
                   {'train': [24, 28, 32], 'test': [26, 27, 29]},
                   {'train': [23, 25, 33], 'test': [29, 30, 31]}}
                  ])

# 打印评估结果
print('Accuracy:', score)
```
