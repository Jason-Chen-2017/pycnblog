4.3 核心代码实现
---------------------

这里以一个实时监测心率的智能健康监测设备为例，给出一个核心代码实现。

```python
import numpy as np
import os
import time

class HeartRateSensor:
    def __init__(self, device):
        self.device = device

    def read_heart_rate(self):
        return self.device.read_heart_rate()

    def save_heart_rate(self, heart_rate):
        self.device.write_heart_rate(heart_rate)

    def start_monitoring(self):
        self.device.start_monitoring()

    def stop_monitoring(self):
        self.device.stop_monitoring()

if __name__ == '__main__':
    # 初始化智能健康监测设备
    heart_rate_sensor = HeartRateSensor('smart_heart_rate_device')

    # 实时监测心率
    while True:
        heart_rate = heart_rate_sensor.read_heart_rate()
        print('当前心率：', heart_rate)

        #  save_heart_rate 将实时监测的心率保存到文件
        heart_rate_file = open('heart_rate.txt', 'a')
        heart_rate_file.write(str(heart_rate))
        heart_rate_file.close()

        # 暂停 1 秒，减少监测间隔
        time.sleep(1)
```
