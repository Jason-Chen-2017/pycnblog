```
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.naive_bayes import MultinomialNB

unlabeled_data = load_data("unlabeled_data.txt")

X_train, X_test, y_train, y_test = train_test_split(unlabeled_data, test_size=0.2, random_state=0)

# 创建一个情感分析模型

# 使用训练集来训练模型
clf.fit(X_train, y_train)

# 使用测试集来评估模型性能
accuracy = clf.score(X_test, y_test)

print("Accuracy:", accuracy)
```
