4.3. 核心代码实现
```python
# 导入所需模块
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import linear_model
from sklearn.metrics import mean_squared_error

# 读取原始数据
data = pd.read_csv('energy_data.csv')

# 数据预处理
# 假设数据为3行，分别是能源消耗量、设备运行时间
# 实际数据应根据实际情况进行预处理

# 将数据分为训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(data.drop(['能源消耗量'], axis=1), data['能源消耗量'], test_size=0.2, random_state=0)

# 构建线性回归模型
model = linear_model.lm(y_train, X_train)

# 预测能源消耗量
y_pred = model.predict(X_test)

# 计算模型的均方误差
mse = mean_squared_error(y_test, y_pred)
print(f'Mean Squared Error: {mse}')

# 进行优化管理
```
