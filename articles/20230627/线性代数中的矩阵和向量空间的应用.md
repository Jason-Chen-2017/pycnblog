--------------------

2.1 基本概念解释
-----------------------

矩阵和向量空间都是线性代数中的基本概念。矩阵是一种特殊的向量，用于表示线性变换。向量是一种基本的物理量，用于表示空间中的点。矩阵的行和列表示向量空间中点的数量，而元素则表示这些点在空间中的位置。

2.2 技术原理介绍：算法原理，操作步骤，数学公式等
-------------------------------------------------------

线性代数中的矩阵算法非常丰富，包括矩阵的加法、减法、乘法等操作。矩阵的加法和减法可以通过一行或多行的方式实现，而矩阵的乘法则需要按照列式相乘，行式相加的方式进行。矩阵的乘法满足结合律、交换律和单位元等性质，是线性代数中非常基础的算法。

    for (int i = 0; i < data.size(); i++) {

// 矩阵加法
void matrixAdd(vector<vector<double>>& A, vector<vector<double>>& B, vector<vector<double>>& result) {
    for (int i = 0; i < A.size(); i++) {
        for (int j = 0; j < B.size(); j++) {
            for (int k = 0; k < result.size(); k++) {
                result[i][j] = result[i][j] + A[i][k] + B[k][j];
            }
        }
    }
}

// 矩阵减法
void matrixSub(vector<vector<double>>& A, vector<vector<double>>& B, vector<vector<double>>& result) {
    for (int i = 0; i < A.size(); i++) {
        for (int j = 0; j < B.size(); j++) {
            for (int k = 0; k < result.size(); k++) {
                result[i][j] = result[i][j] - B[i][k];
            }
        }
    }
}

// 矩阵乘法
void matrixMult(vector<vector<double>>& A, vector<vector<double>>& B, vector<vector<double>>& result) {
    int n = A[0].size();
    int m = B.size();
    for (int i = 0; i < result.size(); i++) {
        for (int j = 0; j < n; j++) {
            for (int k = 0; k < m; k++) {
                int row = j * n + k;
                int col = i * m + k;
                result[i][j] = result[i][j] + A[i][j] * B[col][k];
            }
        }
    }
}

int main() {
    // 读入数据
    vector<vector<double>> data(3, vector<double>(4));
    for (int i = 0; i < data.size(); i++) {
        for (int j = 0; j < data[i].size(); j++) {
            data[i][j] = data[i][j] + 2 * i + j;
        }
    }

    // 计算线性变换
    vector<vector<double>> result(data.size(), vector<double>(4));
    matrixAdd(data, result, result);
    matrixSub(data, result, result);
    matrixMult(data, result, result);

    // 输出结果
    for (int i = 0; i < result.size(); i++) {
        for (int j = 0; j < result[i].size(); j++) {
            cout << result[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
