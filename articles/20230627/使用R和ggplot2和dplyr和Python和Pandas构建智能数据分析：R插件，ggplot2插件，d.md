```R
# 安装dplyr和ggplot2
install.packages(c("dplyr", "ggplot2"))

# 导入需要的库
library(dplyr)
library(ggplot2)

# 读取数据
data <- read.csv("data.csv")

# 合并数据
df <- merge(data.frame(x = c(1, 2, 3), y = c(4, 5, 6)), by = "x")

# 数据可视化
ggplot(df, aes(x = x)) + 
  geom_line()
```
