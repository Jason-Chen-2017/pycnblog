```python
import requests
from bs4 import BeautifulSoup

import numpy as np
import nltk
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.pipeline import make_pipeline
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split

nltk.download('punkt')
nltk.download('wordnet')

def read_data(file_path):
    with open(file_path, 'r', encoding='utf-8') as f:
        return f.read()

def nltk_tokenize(text):
    return nltk.word_tokenize(text.lower())

def preprocess_text(text):
    words = nltk_tokenize(text.lower())
    filtered_words = [word for word in words if word not in nltk.word_v frequency(words)]
    return''.join(filtered_words)

def create_ word_vector_matrix(texts, tokenizer):
    vectorizer = CountVectorizer()
    input_doc = vectorizer.fit_transform(texts)
    return input_doc.toarray()

def extract_features(text, word_vector_matrix):
    features = []
    for word, index in nltk.factorize(text):
        feature = word
        start = index
        end = start + len(word)
        if start + end > len(text):
            feature += '<STOPWORDS>'
        else:
            feature += word
            feature += '</STOPWORDS>'
        features.append(feature)
    return features

def train_model(data_path, model_path):
    # Load data
    data = read_data(data_path)
    texts = []
    for line in data.split('
'):
        texts.append(line.strip())

    # Preprocess text
    for text in texts:
        text = preprocess_text(text)
        texts.append(text)

    # Create word vector matrix
    input_docs = []
    for text in texts:
        input_docs.append(create_word_vector_matrix(text, word_vector_matrix))

    # Train model
    model = MultinomialNB()
    model.fit(input_docs, texts)

    # Save model
    model.save(model_path)

# Example usage
data_path = 'data.txt'
model_path ='model.pkl'

train_model(data_path, model_path)
```
