```php
// 人脸识别
class FaceRecognizer {
    constructor() {
        this.faceCascade = loadFaceCascade();
    }
    
    update(image) {
        let face = extractFace(image);
        
        if (this.faceCascade.check(face)) {
            this.knownPeople.push(face);
        }
    }
    
    knownPeople = [];
    
    // 行为识别
    class BehaviorRecognizer {
        constructor() {
            this.行為识别模型 = loadModel('behavior_recognizer.h');
        }
        
        update(image) {
            let feature = extractFeature(image);
            
            if (this.行為识别模型.canClassify(feature)) {
                let className = this.行為识别模型.predictClass(feature)[0];
                
                if (className === 'normal') {
                    this.knownPeople.push(feature);
                }
            }
        }
        
        knownPeople = [];
    }
}
```
