4.3. 核心代码实现
```python
import numpy as np
import pandas as pd
import mysql.connector
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import accuracy_score
import tensorflow as tf

# 数据预处理
def preprocess_data(data):
    # 清洗
    data = data.dropna()
    data = data.dropna(axis=1)
    # 格式转换
    data['ip_address'] = data['ip_address'].astype(str)
    data['port'] = data['port'].astype(int)
    # 去重
    data = data.drop_duplicates()
    return data

# 特征提取
def extract_features(data):
    # 网络信息
    data['os_type'] = data['os_type'].astype(str)
    data['browser'] = data['browser'].astype(str)
    data['device'] = data['device'].astype(str)
    # 特征值计算
    data['user_agent'] = data['user_agent'].astype(str)
    data['pages'] = data['pages'].astype(int)
    data['referrer'] = data['referrer'].astype(str)
    # 特征选择
    data = data[['browser', 'device', 'ip_address', 'os_type', 'pages','referrer']]
    return data

# 机器学习算法
def train_model(X_train, y_train):
    # 数据划分
    X = extract_features(X_train)
    y = y_train
    # 划分训练集和测试集
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)
    # 创建模型
    model = LinearRegression()
    # 训练模型
    model.fit(X_train, y_train)
    # 预测
    y_pred = model.predict(X_test)
    # 评估模型
    acc = accuracy_score(y_test, y_pred)
    print('Accuracy:', acc)

# 模型评估
def evaluate_model(X, y, model):
    # 预测
    y_pred = model.predict(X)
    # 计算误差
    error = np.sum(y - y_pred) / len(y)
    return error

# 系统集成
def integrate_system(data):
    # 创建数据库连接
    cnx = mysql.connector.connect(user='username', password='password', host='host', database='database')
    # 创建数据表
     tables = ['features', 'ip_addresses', 'os_types', 'pages','referrers']
    with cnx.cursor() as cursor:
        for table in tables:
            cursor.execute('CREATE TABLE {} (id INT, {} {} {} {} {}?, {}?, {}?)'.format(table, 'id', '特征值', '网络安全风险', '安全事件发生概率', '模型的准确性'))
    # 整合数据
    data = data.join(['table_' + table for table in tables], ['id', '特征值', '网络安全风险', '安全事件发生概率', '模型的准确性'])
    # 存储数据
    cursor.execute('INSERT INTO {} (table, data) VALUES (?,?)'.format('table_' + table, data))
    cnx.commit()
    cursor.close()
    cnx.close()

# 应用示例
data = preprocess_data(data)
X_train = extract_features(data)
y_train = train_model(X_train, y_train)
X_test = extract_features(data)
y_test = y_train
integrate_system(data)
```
