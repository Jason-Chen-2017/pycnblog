def target_function(x):
    return -(x**2)

def energy_function(x):
    return target_function(x)

def simulated_annealing(initial_temperature, cooling_rate, search_space, max_iterations):
    current_temperature = initial_temperature
    current_state = np.random.uniform(search_space[0], search_space[1])
    best_state = current_state
    best_energy = energy_function(current_state)

    for iteration in range(max_iterations):
        if np.random.rand() < cooling_rate:
            current_temperature *= cooling_rate

        candidate_state = current_state + np.random.uniform(-1, 1)
        candidate_energy = energy_function(candidate_state)

        if candidate_energy < best_energy or np.random.rand() < np.exp(-(candidate_energy - best_energy) / current_temperature):
            current_state = candidate_state
            best_state = current_state
            best_energy = candidate_energy

    return best_state, best_energy

initial_temperature = 100
cooling_rate = 0.99
search_space = (-10, 10)
max_iterations = 1000

best_state, best_energy = simulated_annealing(initial_temperature, cooling_rate, search_space, max_iterations)

print("Best energy:", best_energy)
