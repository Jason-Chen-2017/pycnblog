# 4.1 数据整合的代码实例和详细解释说明
```python
import pandas as pd
import pymysql

# 连接数据库
conn = pymysql.connect(host='localhost', user='root', password='123456', database='test')

# 提取数据
sql = 'SELECT * FROM employee'
df = pd.read_sql(sql, conn)

# 转换数据
df['age'] = df['age'].astype(int)

# 集成数据
df.to_csv('employee.csv', index=False)

# 数据清洗
df['age'].fillna(df['age'].mean(), inplace=True)
df.drop_duplicates(inplace=True)

# 数据标准化
df['age'] = df['age'] * 1.0
df.rename(columns={'age': 'age_standard'}, inplace=True)

# 数据共享
df.to_csv('employee_standard.csv', index=False)

# 数据应用
df.groupby('department').mean().to_csv('employee_statistics.csv', index=False)
```
