## 4.1 简单规则引擎的Python实现

以下是一个简单的规则引擎的Python实现：

```python
import re

class RuleEngine:
    def __init__(self):
        self.facts = {}
        self.rules = []
        self.results = []

    def load_facts(self, facts_file):
        with open(facts_file, 'r') as f:
            for line in f:
                key, value = line.strip().split(':')
                self.facts[key] = value

    def load_rules(self, rules_file):
        with open(rules_file, 'r') as f:
            for line in f:
                rule = Rule(line.strip())
                self.rules.append(rule)

    def execute(self):
        for rule in self.rules:
            if rule.is_match(self.facts):
                result = rule.execute(self.facts)
                self.results.append(result)

    def save_results(self, results_file):
        with open(results_file, 'w') as f:
            for result in self.results:
                f.write(result + '\n')

class Rule:
    def __init__(self, rule_str):
        self.condition = re.compile(rule_str.split('=>')[0])
        self.action = rule_str.split('=>')[1]

    def is_match(self, facts):
        return self.condition.match(facts)

    def execute(self, facts):
        return self.action.format(**facts)

if __name__ == '__main__':
    rule_engine = RuleEngine()
    rule_engine.load_facts('facts.txt')
    rule_engine.load_rules('rules.txt')
    rule_engine.execute()
    rule_engine.save_results('results.txt')
```
