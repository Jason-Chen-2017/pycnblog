### 4.2.2MIP

```python
from ortools.linear_solver import pywraplp
import numpy as np

# 定义能源供应和需求数据
supply = np.array([100, 200, 300])
demand = np.array([150, 250, 350])

# 构建混合整数规划模型
solver = pywraplp.Solver('energy_allocation', pywraplp.Solver.GLOP_LINEAR_PROGRAMMING)

# 定义变量
x1 = solver.NumVar(0, None, 'x1')
x2 = solver.NumVar(0, None, 'x2')
x3 = solver.NumVar(0, None, 'x3')

# 构建线性目标函数
objective = solver.NumVar(0, None, 'objective')
solver.Minimize(objective)

# 构建混合整数约束条件
solver.Add(supply[0] * x1 + supply[1] * x2 + supply[2] * x3 == demand[0])
solver.Add(demand[1] * x1 + demand[2] * x2 == supply[1])
solver.Add(x1 + x2 + x3 == supply[2])

# 求解最优解
status = solver.Solve()
print(solver.SolutionValue(objective))
```

### 4.2.3GA

```python
import numpy as np
import random

# 定义能源供应和需求数据
supply = np.array([100, 200, 300])
demand = np.array([150, 250, 350])

# 创建初始种群
population_size = 100
population = [random.randint(0, 400) for _ in range(population_size)]

# 评估种群适应度
def fitness(x):
    supply_cost = 0
    demand_cost = 0
    for i in range(len(x)):
        supply_cost += supply[i] * x[i]
        demand_cost += demand[i] * x[i]
    return supply_cost + demand_cost

fitness_values = [fitness(x) for x in population]

# 选择Parents
parents = sorted(range(population_size), key=lambda i: fitness_values[i])[:5]

# 进行交叉和变异
def crossover(parent1, parent2):
    child = (parent1 + parent2) // 2
    return child

def mutation(child):
    mutation_rate = 0.1
    for i in range(len(child)):
        if random.random() < mutation_rate:
            child[i] = random.randint(0, 400)
    return child

offspring = [crossover(parents[i], parents[(i+1)%len(parents)]) for i in range(len(parents))]
offspring = [mutation(child) for child in offspring]

# 创建新的种群
new_population = population[:len(population)//2] + offspring

# 重复步骤，直到达到终止条件
max_generations = 100
for generation in range(max_generations):
    fitness_values = [fitness(x) for x in new_population]
    parents = sorted(range(len(new_population)), key=lambda i: fitness_values[i])[:5]
    offspring = [crossover(parents[i], parents[(i+1)%len(parents)]) for i in range(len(parents))]
    offspring = [mutation(child) for child in offspring]
    new_population = new_population[:len(new_population)//2] + offspring

    if generation % 10 == 0:
        print(f'Generation {generation}, best fitness: {min(fitness_values)}')

# 找到最优解
best_index = parents[0]
best_solution = population[best_index]
print(f'Best solution: {best_solution}, best fitness: {fitness(best_solution)}')
```
