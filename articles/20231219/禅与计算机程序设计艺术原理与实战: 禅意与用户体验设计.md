import tkinter as tk

class Calculator:
    def __init__(self, master):
        self.master = master
        self.master.title("计算器")
        self.result = tk.Entry(self.master, width=20)
        self.result.pack(fill=tk.X, padx=10, pady=10)

        self.button_1 = tk.Button(self.master, text="1", width=10, command=lambda: self.button_click(1))
        self.button_1.pack(side=tk.LEFT, padx=10, pady=10)

        self.button_2 = tk.Button(self.master, text="2", width=10, command=lambda: self.button_click(2))
        self.button_2.pack(side=tk.LEFT, padx=10, pady=10)

        self.button_3 = tk.Button(self.master, text="3", width=10, command=lambda: self.button_click(3))
        self.button_3.pack(side=tk.LEFT, padx=10, pady=10)

        self.button_plus = tk.Button(self.master, text="+", width=10, command=lambda: self.button_click("+"))
        self.button_plus.pack(side=tk.LEFT, padx=10, pady=10)

        self.button_equal = tk.Button(self.master, text="=", width=10, command=self.button_equal)
        self.button_equal.pack(side=tk.RIGHT, padx=10, pady=10)

    def button_click(self, number):
        current = self.result.get()
        self.result.delete(0, tk.END)
        self.result.insert(0, current + str(number))

    def button_equal(self):
        current = self.result.get()
        result = eval(current)
        self.result.delete(0, tk.END)
        self.result.insert(0, result)

if __name__ == "__main__":
    root = tk.Tk()
    calculator = Calculator(root)
    root.mainloop()
```
