# 4.具体代码实例和详细解释说明

以下是一个简单的Python代码实例，用于在线课程平台的课程发布和推广：

```python
import requests

# 发布课程
def publish_course(course_title, course_content):
    url = "https://api.onlinecourse.com/publish"
    data = {
        "course_title": course_title,
        "course_content": course_content
    }
    response = requests.post(url, data=data)
    if response.status_code == 200:
        print("课程发布成功")
    else:
        print("课程发布失败")

# 推广课程
def promote_course(course_id):
    url = f"https://api.onlinecourse.com/promote/{course_id}"
    response = requests.get(url)
    if response.status_code == 200:
        print("课程推广成功")
    else:
        print("课程推广失败")

# 主函数
def main():
    course_title = "Python编程基础"
    course_content = "Python编程基础的课程内容"
    course_id = publish_course(course_title, course_content)
    promote_course(course_id)

if __name__ == "__main__":
    main()
```
