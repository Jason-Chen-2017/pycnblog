def topological_state(L, N, phi):
    k = 2 * np.pi / L
    psi = np.zeros(N)
    for m in range(-N//2, N//2):
        psi[m] = np.exp(1j * m * k * phi)
    return psi

N = 100
phi = np.pi / 2  # radian

result = topological_state(L, N, phi)
print("Topological State: ", result)
