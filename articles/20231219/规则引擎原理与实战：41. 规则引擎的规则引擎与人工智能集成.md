```python
from django.core.management.commands import runserver

class RuleEngine:
    def __init__(self):
        self.rules = []

    def add_rule(self, condition, action):
        self.rules.append((condition, action))

    def execute(self, input_data):
        for condition, action in self.rules:
            if condition(input_data):
                action(input_data)

class AgeRule:
    def __init__(self, age, message):
        self.age = age
        self.message = message

    def condition(self, input_data):
        return input_data['age'] == self.age

    def action(self, input_data):
        print(self.message)

engine = RuleEngine()
engine.add_rule(AgeRule(20, '你是年轻人'))
engine.add_rule(AgeRule(30, '你是中年人'))
engine.add_rule(AgeRule(40, '你是老年人'))
engine.execute({'age': 25})
```

## 4.2 人工智能代码实例

以下是一个简单的自然语言处理代码实例，它可以根据用户输入的文本判断用户是否感到困惑：

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

class ConfusionDetector:
    def __init__(self, text_data):
        self.vectorizer = TfidfVectorizer()
        self.text_data = text_data

    def train(self):
        X = self.vectorizer.fit_transform(self.text_data)
        self.similarity = cosine_similarity(X)

    def detect(self, input_text):
        input_vector = self.vectorizer.transform([input_text])
        similarity = self.similarity[input_vector.toarray()]
        return similarity.max()

confusion_data = ['我不知道怎么做', '我不明白这个问题', '我不懂这个概念']
detector = ConfusionDetector(confusion_data)
detector.train()
print(detector.detect('我不知道怎么做'))
```
