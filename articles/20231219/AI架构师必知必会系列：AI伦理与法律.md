### 4.2 数据使用权算法代码实例

```python
from flask import Flask, request, jsonify
from functools import wraps

app = Flask(__name__)

def authorize(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'Authorization' not in request.headers:
            return jsonify({'error': 'Missing Authorization Header'}), 401
        return f(*args, **kwargs)
    return decorated_function

@app.route('/data', methods=['GET'])
@authorize
def get_data():
    # Retrieve data from database
    data = {'name': 'John', 'age': 30, 'email': 'john@example.com'}
    return jsonify(data)

if __name__ == '__main__':
    app.run()
```

from sklearn.tree import DecisionTreeClassifier
from sklearn.datasets import load_iris
from sklearn.tree import export_graphviz

# Load dataset
data = load_iris()
X, y = data.data, data.target

# Train model
model = DecisionTreeClassifier()
model.fit(X, y)

# Export decision tree as DOT format
dot_data = export_graphviz(model, out_file=None, feature_names=data.feature_names, class_names=data.target_names, filled=True, rounded=True, special_characters=True)

# Save decision tree as PNG image
