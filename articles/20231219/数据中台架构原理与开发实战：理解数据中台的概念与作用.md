```python
    'age': (1, 100),
    'weight': (1, 500),
    'height': (1, 300)
}

# 数据质量检查
def check_data_quality(df, standards):
    for column, (min_value, max_value) in standards.items():
        if df[column].min() < min_value or df[column].max() > max_value:
            return False
    return True

# 数据质量问题处理
if not check_data_quality(df, data_quality_standards):
    df['age'] = df['age'].clip(lower=1, upper=100)

# 数据质量报告
print('数据质量检查结果：', check_data_quality(df, data_quality_standards))
```
