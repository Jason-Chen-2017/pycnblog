    'A': {'活跃度': '高', '专业性': '一般', '礼貌度': '低'},

# 添加节点和边
for node, attributes in nodes.items():

    for other_node, other_attributes in nodes.items():
        if node != other_node:
            G.add_edge(node, other_node, weight=random.uniform(0.1, 1.0))

plt.show()
