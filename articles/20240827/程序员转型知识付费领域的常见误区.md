### 5.2 源代码详细实现

以下是一个基于 Python 的用户兴趣度预测和内容推荐模型的源代码实现：

```python
# 用户兴趣度预测模型
def user_interest_prediction(ratings, user_id, content_id):
    user_rating_vector = ratings[user_id]
    content_rating_vector = ratings[content_id]
    interest_score = np.dot(user_rating_vector, content_rating_vector)
    return np.sign(interest_score)

# 内容推荐模型
def content_recommendation(ratings, user_id, top_n=5):
    user_rating_vector = ratings[user_id]
    content_rating_vectors = ratings.values()
    content_recommendation_scores = np.dot(content_rating_vectors, user_rating_vector)
    sorted_recommendation_scores = np.argsort(-content_recommendation_scores)
    return sorted_recommendation_scores[:top_n]

# 数据集加载与预处理
ratings = pd.read_csv('knowledge付费数据集.csv')
ratings = ratings.set_index('user_id')
ratings = ratings.T.values

# 用户兴趣度预测
user_interest_prediction_result = user_interest_prediction(ratings, user_id=1, content_id=101)

# 内容推荐
content_recommendation_result = content_recommendation(ratings, user_id=1, top_n=5)

# 输出结果
print("用户兴趣度预测结果：", user_interest_prediction_result)
print("内容推荐结果：", content_recommendation_result)
```
