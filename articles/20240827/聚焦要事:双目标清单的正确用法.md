```python
import pandas as pd

class DoubleGoalList:
    def __init__(self):
        self.tasks = pd.DataFrame(columns=['task', 'short_term_goal', 'long_term_goal'])

    def add_task(self, task, short_term_goal, long_term_goal):
        new_task = {'task': task, 'short_term_goal': short_term_goal, 'long_term_goal': long_term_goal}
        self.tasks = self.tasks.append(new_task, ignore_index=True)

    def sort_tasks(self):
        self.tasks.sort_values(by='short_term_goal', ascending=False, inplace=True)

    def calculate_goal_achievement(self):
        short_term_goals = self.tasks['short_term_goal'].sum()
        long_term_goals = self.tasks['long_term_goal'].sum()
        total_goals = short_term_goals + long_term_goals
        goal_achievement = (short_term_goals + long_term_goals) / total_goals
        return goal_achievement

    def display_tasks(self):
        print(self.tasks)

if __name__ == '__main__':
    dgl = DoubleGoalList()
    dgl.add_task('写博客', True, False)
    dgl.add_task('做市场调研', False, True)
    dgl.sort_tasks()
    dgl.display_tasks()
    print('目标实现度：', dgl.calculate_goal_achievement())
```
