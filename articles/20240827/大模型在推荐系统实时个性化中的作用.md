### 5.2 源代码详细实现

以下是一个简单的基于TensorFlow的深度学习模型实现，用于生成用户兴趣向量：

```python
import tensorflow as tf
import numpy as np
import pandas as pd
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, LSTM, Embedding
from tensorflow.keras.optimizers import Adam

# 加载数据集
user_behavior = pd.read_csv('user_behavior.csv')
content_feature = pd.read_csv('content_feature.csv')

# 数据预处理
user_behavior = user_behavior.values
content_feature = content_feature.values

# 构建模型
model = Sequential()
model.add(Embedding(input_dim=user_behavior.shape[1], output_dim=128))
model.add(LSTM(128, return_sequences=True))
model.add(Dense(1, activation='sigmoid'))

# 编译模型
model.compile(optimizer=Adam(learning_rate=0.001), loss='binary_crossentropy', metrics=['accuracy'])

# 训练模型
model.fit(user_behavior, content_feature, epochs=10, batch_size=32)

# 生成用户兴趣向量
user_interest = model.predict(user_behavior)

# 可视化
import matplotlib.pyplot as plt

plt.scatter(user_interest[:, 0], user_interest[:, 1])
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.title('User Interest Vector')
plt.show()
```
