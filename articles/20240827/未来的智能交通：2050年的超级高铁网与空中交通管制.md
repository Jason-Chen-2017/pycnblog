import numpy as np
import matplotlib.pyplot as plt

# Dijkstra算法
def dijkstra(G, s, t):
    dist = [float('inf')] * len(G)
    dist[s] = 0
    visited = [False] * len(G)
    for _ in range(len(G)):
        u = min_idx(dist, visited)
        visited[u] = True
        for v in G[u]:
            if not visited[v]:
                alt = dist[u] + G[u][v]
                if alt < dist[v]:
                    dist[v] = alt
    return dist

# 最小值索引
def min_idx(arr, visited):
    min_val = float('inf')
    min_idx = -1
    for i in range(len(arr)):
        if not visited[i] and arr[i] < min_val:
            min_val = arr[i]
            min_idx = i
    return min_idx

# 资源调度
def resource_allocation(G, u, v):
    c = [0] * len(G)
    p = [None] * len(G)
    for _ in range(len(G)):
        u = min_idx(c, p)
        p[u] = v
        c[u] = c[u] + 1
    return p

# 数据
G = [
    [0, 1, 2],
    [1, 0, 3],
    [2, 3, 0]
]

# 路径规划
dist = dijkstra(G, 0, 2)
print(dist)

# 资源调度
p = resource_allocation(G, 0, 2)
print(p)
