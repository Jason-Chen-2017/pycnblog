```python
def average_donation(participants):
    total_donation = 0
    for participant in participants:
        total_donation += participant['donation']
    return total_donation / len(participants)
```

```python
from collections import defaultdict

def influence_network(participants):
    graph = defaultdict(list)
    for participant in participants:
        for influencer in participant['influenced_by']:
            graph[influencer].append(participant['name'])
    return graph
```
