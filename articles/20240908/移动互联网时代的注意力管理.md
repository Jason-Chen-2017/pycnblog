```go
func recommendProducts(userBehavior map[string]int) []string {
    // 假设productMap是一个包含商品和其相关度的映射
    productMap := map[string]int{
        "智能手机": 90,
        "耳机": 80,
        "平板电脑": 70,
    }
    
    // 根据用户行为为每个商品计算相关度得分
    behaviorScores := make(map[string]int)
    for product, relevance := range productMap {
        if _, exists := userBehavior[product]; exists {
            behaviorScores[product] = relevance * userBehavior[product]
        }
    }
    
    // 根据得分排序推荐商品
    sortedProducts := sortProductsByScore(behaviorScores)
    
    return sortedProducts
}

func sortProductsByScore(scores map[string]int) []string {
    // 将映射转换为切片并排序
    items := make([][]string, 0, len(scores))
    for product, score := range scores {
        items = append(items, []string{product, strconv.Itoa(score)})
    }
    sort.Slice(items, func(i, j int) bool {
        return items[i][1] > items[j][1]
    })
    
    // 提取排序后的商品名称
    sortedProducts := make([]string, len(items))
    for i, item := range items {
        sortedProducts[i] = item[0]
    }
    
    return sortedProducts
}
```

```go
func downloadFile(url string, ch chan<- string) {
    resp, err := http.Get(url)
    if err != nil {
        ch <- err.Error()
        return
    }
    defer resp.Body.Close()

    // 这里只是简单地打印文件内容，实际中可以进行文件写入等操作
    buffer := new(bytes.Buffer)
    _, err = buffer.ReadFrom(resp.Body)
    if err != nil {
        ch <- err.Error()
        return
    }

    ch <- buffer.String()
}

func main() {
    urls := []string{
        "http://example.com/file1.txt",
        "http://example.com/file2.txt",
        "http://example.com/file3.txt",
    }
    var wg sync.WaitGroup
    ch := make(chan string, len(urls))

    for _, url := range urls {
        wg.Add(1)
        go func(url string) {
            defer wg.Done()
            downloadFile(url, ch)
        }(url)
    }

    go func() {
        wg.Wait()
        close(ch)
    }()

    for file := range ch {
        fmt.Println(file)
    }
}
```

**解析：** 在这个例子中，我们使用 goroutines 来并发下载多个文件，并通过 channels 将下载结果发送回主线程。使用缓冲通道可以避免频繁的阻塞，提高程序的并发性能。
