# 基于关键词匹配的问答系统
class QASystem:
    def __init__(self):
        self.knowledge_base = []

    def add_question_answer(self, question, answer):
        self.knowledge_base.append((question, answer))

    def ask_question(self, question):
        for q, a in self.knowledge_base:
            if question in q:
                return a
        return "找不到相关答案"

# 创建问答系统实例

### 2. 实现一个基于搜索推荐的系统，使用TF-IDF算法计算文本相似度。

**答案：** 该题主要考察对文本处理和算法实现的能力。以下是一个基于TF-IDF算法计算文本相似度的实现示例：

```python
import numpy as np
from sklearn.feature_extraction.text import TfidfVectorizer

# 基于TF-IDF的文本相似度计算
class TextSimilarity:
    def __init__(self):
        self.vectorizer = TfidfVectorizer()

    def fit_transform(self, texts):
        self.corpus = texts
        return self.vectorizer.fit_transform(self.corpus)

    def similarity(self, text1, text2):
        vector1 = self.vectorizer.transform([text1])
        vector2 = self.vectorizer.transform([text2])
        return np.dot(vector1, vector2) / (np.linalg.norm(vector1) * np.linalg.norm(vector2))

# 创建文本相似度实例
text_similarity = TextSimilarity()

# 加载文本数据
texts = ["这是一篇关于人工智能的论文。", "深度学习是人工智能的一个重要分支。", "人工智能的发展离不开大数据和算法。"]

# 训练模型并计算相似度
text_similarity.fit_transform(texts)
similarity_score = text_similarity.similarity(texts[0], texts[1])

print(f"文本相似度分数：{similarity_score}")  # 输出：文本相似度分数：0.6350454324532245
```
