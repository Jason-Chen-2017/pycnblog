以下是一个简单的课程购买功能实现示例（使用 Python 的 Flask 框架）：

```python
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///course.db'
db = SQLAlchemy(app)

class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(80), unique=True, nullable=False)
    password = db.Column(db.String(120), nullable=False)
    balance = db.Column(db.Float, nullable=False)

class Course(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(120), nullable=False)
    price = db.Column(db.Float, nullable=False)
    description = db.Column(db.Text, nullable=True)

@app.route('/buy_course', methods=['POST'])
def buy_course():
    user_id = request.form['user_id']
    course_id = request.form['course_id']
    user = User.query.get(user_id)
    course = Course.query.get(course_id)

    if user.balance >= course.price:
        user.balance -= course.price
        db.session.commit()
        return jsonify({'status': 'success', 'message': '购买成功'})
    else:
        return jsonify({'status': 'failure', 'message': '余额不足'})

if __name__ == '__main__':
    db.create_all()
    app.run()
```

以下是一个简单的基于内容的推荐功能实现示例（使用 Python 的 Flask 框架）：

```python
from flask import Flask, request, jsonify
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

app = Flask(__name__)

courses = [
    {"id": 1, "name": "Python 基础", "content": "Python 是一种广泛应用的编程语言，具有简洁、易学、易用的特点。"},
    {"id": 2, "name": "Python 进阶", "content": "Python 进阶内容包括面向对象编程、网络编程、多线程等。"},
    {"id": 3, "name": "数据结构与算法", "content": "数据结构与算法是计算机科学的核心内容，包括数组、链表、树、图等。"},
    {"id": 4, "name": "机器学习基础", "content": "机器学习是人工智能的一个重要分支，包括监督学习、无监督学习等。"},
]

@app.route('/recommend_courses', methods=['POST'])
def recommend_courses():
    user_content = request.form['content']
    vectorizer = TfidfVectorizer()
    tfidf_matrix = vectorizer.fit_transform([course['content'] for course in courses])
    user_vector = vectorizer.transform([user_content])

    similarity = cosine_similarity(user_vector, tfidf_matrix)
    recommended_courses = [course['id'] for course in courses if similarity[0][courses.index(course)] > 0.5]

    return jsonify({'recommended_courses': recommended_courses})

if __name__ == '__main__':
    app.run()
```
