```go
)

}

    fmt.Println(p.Name, "is walking.")
}

    p.Status = "Running"
    fmt.Println(p.Name, "is running.")

    fmt.Println(p.Name, "is sleeping.")

    pet.Run()

```

#### 2. 如何实现宠物间的互动？

**题目：** 编写一个程序，实现两只宠物之间的互动，如打招呼、玩耍等。

**答案：**

```go
package main

import (
    "fmt"
)

type Pet struct {
    Name   string
    Status string
}

func (p *Pet) Greet(other *Pet) {
    fmt.Printf("%s greets %s!\n", p.Name, other.Name)
}

func (p *Pet) Play(other *Pet) {
    fmt.Printf("%s is playing with %s!\n", p.Name, other.Name)
}

func main() {
    pet1 := Pet{Name: "Kitty", Status: ""}
    pet2 := Pet{Name: "Doggy", Status: ""}

    pet1.Greet(&pet2)
    pet1.Play(&pet2)
}
```

**解析：** 该程序定义了一个 `Pet` 结构体，包含 `Name` 和 `Status` 两个字段。通过定义 `Greet` 和 `Play` 方法，实现宠物之间的互动。
