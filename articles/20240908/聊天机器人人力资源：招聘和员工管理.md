```python
class ChatBot:
    def __init__(self):
        self.responses = {
            "greeting": "Hello! How can I assist you today?",
            "thank_you": "You're welcome! Is there anything else I can help with?"
        }

    def respond(self, message):
        if message.lower() in ["hello", "hi"]:
            return self.responses["greeting"]
        elif message.lower() == "thank you":
            return self.responses["thank_you"]
        else:
            return "I'm not sure how to respond to that. Can you try rephrasing your question?"

bot = ChatBot()
print(bot.respond("Hello!"))  # 输出: Hello! How can I assist you today?
print(bot.respond("Thank you!"))  # 输出: You're welcome! Is there anything else I can help with?
```

        self.history.append({"user_id": user_id, "message": message})

    def delete_entry(self, user_id):
        self.history = [entry for entry in self.history if entry["user_id"] != user_id]

        return [entry["message"] for entry in self.history if entry["user_id"] == user_id]

history = ChatHistory()
