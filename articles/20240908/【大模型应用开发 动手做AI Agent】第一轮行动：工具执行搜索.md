```python
# 倒排索引示例

class InvertedIndex:
    def __init__(self):
        self.index = {}

    def add_document(self, document_id, words):
        for word in words:
            if word not in self.index:
                self.index[word] = []
            self.index[word].append(document_id)

    def search(self, query):
        query_words = query.split()
        result = set(query_words[0])
        for word in query_words[1:]:
            result &= set(self.index.get(word, []))
        return result

ii = InvertedIndex()
ii.add_document(1, ["apple", "banana", "orange"])
ii.add_document(2, ["apple", "mango"])
ii.add_document(3, ["orange", "mango", "banana"])

print(ii.search("apple mango"))  # 输出 {2}
```
