func createChatRoom(user1, user2 string) *ChatRoom {
    // 创建聊天室，初始化成员
    chatRoom := &ChatRoom{
        Members: []string{user1, user2},
        Messages: []string{},
    }
    return chatRoom
}

// 发送语音消息
func sendVoiceMessage(sender string, chatRoom *ChatRoom, message string) {
    // 将语音消息加入聊天室
    chatRoom.Messages = append(chatRoom.Messages, sender+":"+message)
}

// 接收语音消息
func receiveVoiceMessage(receiver string, chatRoom *ChatRoom) string {
    // 查找并返回最新的语音消息
    for _, message := range chatRoom.Messages {
        parts := strings.Split(message, ":")
        if parts[0] == receiver {
            return parts[1]
        }
    }
    return ""
}
