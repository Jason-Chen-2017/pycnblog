```python
# 示例：基于内容的推荐算法
def content_based_recommendation(user_data, courses):
    recommended_courses = []
    for course in courses:
        if course['topic'] == user_data['interested_topic']:
            recommended_courses.append(course)
    return recommended_courses
```
