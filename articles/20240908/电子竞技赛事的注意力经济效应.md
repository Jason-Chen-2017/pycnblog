    {"name": "Viewer C", "start_date": "2021-03-01", "end_date": "2021-03-20", "watch_days": 20}

from datetime import datetime

def calculate_retention(viewer_retention):
    retention_scores = {}
    for retention in viewer_retention:
        start_date = datetime.strptime(retention["start_date"], "%Y-%m-%d")

def calculate_loyalty(viewer_activities):
    loyalty_scores = {}
    for activity in viewer_activities:
        watch_freq = len(activity["watch_dates"])
        purchase_freq = len(activity["purchase_dates"])
        max_days = 0
        for date1, date2 in zip(activity["watch_dates"], activity["purchase_dates"]):
