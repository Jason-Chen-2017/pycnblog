```python
def line_intersection(line1, line2):
    x_diff = (line1[2] - line1[0], line2[2] - line2[0])
    y_diff = (line1[3] - line1[1], line2[3] - line2[1])

    def det(a, b):
        return a[0] * b[1] - a[1] * b[0]

    det = det(x_diff, y_diff)
    if det == 0:
        return False  # 线段平行，无交点

    x = det((line1[3], -line1[0]), (line2[3], -line2[0])) / det(x_diff)
    y = det((line1[2], -line1[0]), (line2[2], -line2[0])) / det(y_diff)

    if 0 <= x <= line1[2] - line1[0] and 0 <= y <= line1[3] - line1[1] and \
        0 <= x <= line2[2] - line2[0] and 0 <= y <= line2[3] - line2[1]:
        return True  # 线段相交
    else:
        return False  # 线段不相交
```
