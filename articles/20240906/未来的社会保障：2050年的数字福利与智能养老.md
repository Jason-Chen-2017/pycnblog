```python
class LoanApplication:
    def __init__(self, applicant_name, loan_amount, interest_rate):
        self.applicant_name = applicant_name
        self.loan_amount = loan_amount
        self.interest_rate = interest_rate

    def calculate_monthly_payment(self, loan_duration):
        monthly_interest = self.loan_amount * self.interest_rate / 12
        total_interest = monthly_interest * loan_duration
        total_payment = self.loan_amount + total_interest
        monthly_payment = total_payment / loan_duration
        return monthly_payment

```

```python
import datetime

class MeetingScheduler:
    def __init__(self):
        self.schedule = {}

    def schedule_meeting(self, meeting_id, start_time, end_time, participants):
        self.schedule[meeting_id] = {
            "start_time": start_time,
            "end_time": end_time,
            "participants": participants
        }

scheduler.list_meetings()
```
