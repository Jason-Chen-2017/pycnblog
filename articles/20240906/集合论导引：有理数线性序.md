## 2. 有理数排序算法

### 题目

from collections import defaultdict

def counting_sort(arr, exp1):
    n = len(arr)
    output = [0] * n
    count = [0] * 10

    # 计算每个数位上的计数
    for i in range(0, n):
        index = int(arr[i] / exp1)
        count[index % 10] += 1

    # 更新计数数组
    for i in range(1, 10):
        count[i] += count[i - 1]

    # 创建输出数组
    i = n - 1
    while i >= 0:
        index = int(arr[i] / exp1)
        output[count[index % 10] - 1] = arr[i]
        count[index % 10] -= 1
        i -= 1

    # 复制输出数组到输入数组
    for i in range(0, len(arr)):
        arr[i] = output[i]

# 主函数
def radix_sort(arr):
    max1 = max(arr)
    exp1 = 1
    while max1 / exp1 > 0:
        counting_sort(arr, exp1)
        exp1 *= 10
