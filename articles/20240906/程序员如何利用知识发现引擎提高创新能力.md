```python
# Python 实现简单的知识发现引擎

from mlxtend.frequent_patterns import apriori
from mlxtend.preprocessing import TransactionEncoder

# 数据预处理
te = TransactionEncoder()
te.fit(data)
data_encoded = te.transform(data)

# 频繁项集挖掘
frequent_itemsets = apriori(data_encoded, min_support=0.05, use_colnames=True)

# 结果可视化
from mlxtend.plotting import plot_frequent_itemsets
plot_frequent_itemsets(frequent_itemsets, size=15, max_itemsets=10)
```
