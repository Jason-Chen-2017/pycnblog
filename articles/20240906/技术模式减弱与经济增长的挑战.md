**题目 1：** 最长公共子序列

**问题描述：** 给定两个字符串 `s1` 和 `s2`，找到它们的最长公共子序列。

**示例：** 

```

s2 = "BDCAB"
```

**答案：**

```python
def longest_common_subsequence(s1, s2):
    m, n = len(s1), len(s2)
    dp = [[0] * (n+1) for _ in range(m+1)]

    for i in range(1, m+1):
        for j in range(1, n+1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])

    return dp[m][n]

print(longest_common_subsequence("ABCBDAB", "BDCAB"))  # 输出 4
```

**题目 2：** 最小生成树

**问题描述：** 给定一个无向图和图的边权，求该图的最小生成树。

**示例：**

```
edges = [[0, 1, 10], [0, 2, 6], [0, 3, 5], [1, 3, 15], [2, 3, 4]]
```

**答案：**

```python
import heapq

def prim(graph):
    n = len(graph)
    mst = []
    visited = [False] * n
    start = 0
    visited[start] = True
    pq = [(0, start)]

    while pq:
        weight, vertex = heapq.heappop(pq)
        mst.append((vertex, weight))

        for next, edge in enumerate(graph[vertex]):
            if not visited[next]:
                visited[next] = True
                heapq.heappush(pq, (edge, next))

    return sum(weight for vertex, weight in mst)

print(prim([[0, 1, 10], [0, 2, 6], [0, 3, 5], [1, 3, 15], [2, 3, 4]]))  # 输出 20
```

**题目 4：** 背包问题

**问题描述：** 给定一个价值数组和一个重量数组以及一个背包的容量，找出能够装入背包的最大价值。

**示例：**

```
values = [1, 6, 10, 16]
weights = [1, 2, 3, 5]
W = 7
```

**答案：**

```python
def knapsack(values, weights, W):
    n = len(values)
    dp = [[0] * (W + 1) for _ in range(n + 1)]

    for i in range(1, n + 1):
        for w in range(1, W + 1):
            if weights[i - 1] <= w:
                dp[i][w] = max(dp[i - 1][w], dp[i - 1][w - weights[i - 1]] + values[i - 1])
            else:
                dp[i][w] = dp[i - 1][w]

    return dp[n][W]

print(knapsack([1, 6, 10, 16], [1, 2, 3, 5], 7))  # 输出 22
```

```python
    max_ending_here = max_so_far = nums[0]

    for x in nums[1:]:

        max_so_far = max(max_so_far, max_ending_here)
```

```python
def binary_search(nums, target):
    left, right = 0, len(nums) - 1

    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
            return mid
        elif nums[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return -1

print(binary_search([1, 3, 5, 6], 5))  # 输出 2

```

```python
def climb_stairs(n):
    if n == 1:
        return 1
    if n == 2:
        return 2
    a, b = 1, 2
    for i in range(2, n):
        a, b = b, a + b
    return b

print(climb_stairs(3))  # 输出 3
```

**问题描述：** 给定一个带权重的无向图和起点、终点，求从起点到终点的最短路径。

**示例：**


```python
import heapq

def shortest_path(graph, start, end):
    n = len(graph)
    distances = [float('inf')] * n
    distances[start] = 0
    priority_queue = [(0, start)]

    while priority_queue:
        distance, vertex = heapq.heappop(priority_queue)

        if vertex == end:
            return distance

        for next, edge in enumerate(graph[vertex]):
            new_distance = distance + edge
            if new_distance < distances[next]:
                distances[next] = new_distance
                heapq.heappush(priority_queue, (new_distance, next))

    return -1
```

```python
import heapq

def single_source_shortest_path(graph, start):
    n = len(graph)
    distances = [float('inf')] * n
    distances[start] = 0
    priority_queue = [(0, start)]

    while priority_queue:
        distance, vertex = heapq.heappop(priority_queue)

        if distance != distances[vertex]:
            continue

            new_distance = distance + edge
            if new_distance < distances[next]:
                distances[next] = new_distance
                heapq.heappush(priority_queue, (new_distance, next))

    return distances

print(single_source_shortest_path([[0, 4, 0, 0, 0], [4, 0, 3, 2, 0], [0, 3, 0, 1, 4], [0, 2, 1, 0, 3], [0, 0, 4, 3, 0]], 0))  # 输出 [0, 4, 3, 2, 1]
```

```python
def binary_search(nums, target):
    left, right = 0, len(nums) - 1

    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
            return mid
        elif nums[mid] < target:
```

```python
    return "" if min_len == float('inf') else s[min_l: min_r]
```

```python
        return []

    result = []
    current_width = 0

        if i > 0:
```

**问题描述：** 给定两个字符串 `s1` 和 `s2`，求它们的最长公共子串。

```python
def longest_common_substring(s1, s2):
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    max_len = 0
    end_pos = 0

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i - 1] == s2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
                if dp[i][j] > max_len:
                    max_len = dp[i][j]
                    end_pos = i
            else:
                dp[i][j] = 0

    return s1[end_pos - max_len: end_pos]
```
