```python
def inventory_allocation(sales_data, inventory_data):
    # 初始化动态规划表格
    dp = [[0] * (len(inventory_data) + 1) for _ in range(len(sales_data) + 1)]
    
    # 填充动态规划表格
    for i, sales in enumerate(sales_data):
        for j, inventory in enumerate(inventory_data):
            if sales <= inventory:
                dp[i+1][j+1] = max(dp[i][j], dp[i][j+1] + sales)
            else:
                dp[i+1][j+1] = dp[i][j]
    
    # 返回最优解
    return dp[-1][-1]
```
