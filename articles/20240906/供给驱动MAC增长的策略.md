def simulate_growth_strategy(initial_mac, growth_rate, demand, periods):

**答案：**

```python
import random

def elastic_extension(initial_mac, growth_rate, min_demand, max_demand, periods):
    mac = initial_mac
    benefits = []

    for _ in range(periods):
        demand = random.uniform(min_demand, max_demand)
        if demand < mac:
            mac = max(mac - mac * 0.1, min_demand)  # 需求小于MAC时，弹性缩减MAC
        elif demand > mac:
            mac = mac + mac * growth_rate  # 需求大于MAC时，弹性扩展MAC
        revenue = min(mac, demand) * 100
        cost = mac * 10
        net_benefit = revenue - cost
        benefits.append(net_benefit)

    return benefits

initial_mac = 100

max_demand = 1000
periods = 10

```

**解析：** 该程序实现了弹性扩展算法，根据需求与MAC值的关系，动态调整MAC值，以适应需求波动，并计算每期的净收益。

import numpy as np

def predictive_maintenance(initial_mac, growth_rate, max_usage, min_usage, max_maintenance_time, min_maintenance_time, periods):

        usage = random.uniform(min_usage, max_usage)
        if usage > max_usage * 0.8:  # 当使用率超过80%时，需要维护
            maintenance_time = random.uniform(min_maintenance_time, max_maintenance_time)
            maintenance_times.append(maintenance_time)

        else:
            maintenance_time = 0

        revenue = min(mac, demand) * 100
        cost = mac * 10
        net_benefit = revenue - cost - maintenance_time
        benefits.append(net_benefit)

    return benefits, maintenance_times
