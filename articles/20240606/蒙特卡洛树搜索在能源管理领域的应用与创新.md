### 5.1 蒙特卡洛树搜索的代码实例

以下是蒙特卡洛树搜索的代码实例：

```python
class Node:
    def __init__(self, state):
        self.state = state
        self.children = []
        self.visits = 0
        self.value = 0

    def expand(self):
        actions = self.state.get_actions()
        for action in actions:
            next_state = self.state.get_next_state(action)
            child = Node(next_state)
            self.children.append(child)

    def select(self):
        best_child = None
        best_value = float('-inf')
        for child in self.children:
            value = child.value / child.visits + math.sqrt(2 * math.log(self.visits) / child.visits)
            if value > best_value:
                best_child = child
                best_value = value
        return best_child

    def simulate(self):
        state = self.state
        while not state.is_terminal():
            action = state.get_random_action()
            state = state.get_next_state(action)
        return state.get_reward()

    def update(self, value):
        self.visits += 1
        self.value += value

def monte_carlo_tree_search(root, iterations):
    for i in range(iterations):
        node = root
        while node.children:
            node = node.select()
        if node.visits == 0:
            node.expand()
        value = node.simulate()
        node.update(value)
    return max(root.children, key=lambda child: child.visits)

```
