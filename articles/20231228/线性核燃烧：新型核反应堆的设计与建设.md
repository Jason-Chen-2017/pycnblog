```python
import numpy as np
import matplotlib.pyplot as plt

def carbon_hydrogen_burning(X0, Y0, t_end):
    dt = 0.1
    t = np.arange(0, t_end, dt)
    X = np.zeros(len(t))
    Y = np.zeros(len(t))
    Z = np.zeros(len(t))

    Rf = 0.1
    Rc = 0.2
    Ra = 0.05

    X[0] = X0
    Y[0] = Y0
    Z[0] = 0

    for i in range(1, len(t)):
        X[i] = X[i-1] - Rf * X[i-1] * dt
        Y[i] = Y[i-1] - (Rf * X[i-1] + Rc * Y[i-1]) * dt
        Z[i] = Z[i-1] + (Rc * Y[i-1] - Ra * Z[i-1]) * dt

    return X, Y, Z

X0 = 100
Y0 = 100
t_end = 100

X, Y, Z = carbon_hydrogen_burning(X0, Y0, t_end)

plt.plot(t, X, label='Carbon')
plt.plot(t, Y, label='Hydrogen')
plt.plot(t, Z, label='Radioactive Material')
plt.xlabel('Time (s)')
plt.ylabel('Concentration')
plt.legend()
plt.show()
```
