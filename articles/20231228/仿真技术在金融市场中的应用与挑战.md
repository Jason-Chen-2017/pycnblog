## 4.具体代码实例和详细解释说明

### 4.1 Euler方法的Python实现

```python
import numpy as np

def euler_method(f, x0, T, dt):
    t = np.arange(0, T + dt, dt)
    x = np.zeros(len(t))
    x[0] = x0

    for i in range(1, len(t)):
        x[i] = x[i - 1] + f(x[i - 1], t[i - 1]) * dt

    return t, x
```

### 4.2 Runge-Kutta方法的Python实现

```python
import numpy as np

def runge_kutta_method(f, x0, T, dt):
    t = np.arange(0, T + dt, dt)
    x = np.zeros(len(t))
    x[0] = x0

    for i in range(1, len(t)):
        k1 = f(x[i - 1], t[i - 1])
        k2 = f(x[i - 1] + k1 * dt / 2, t[i - 1] + dt / 2)
        k3 = f(x[i - 1] + k2 * dt / 2, t[i - 1] + dt / 2)
        k4 = f(x[i - 1] + k3 * dt, t[i - 1] + dt)

        x[i] = x[i - 1] + (k1 + 2 * k2 + 2 * k3 + k4) * dt / 6

    return t, x
```

import numpy as np

def langevin_equation(mu, sigma, dt, T):
    t = np.arange(0, T + dt, dt)
    x = np.zeros(len(t))
    x[0] = 0

    for i in range(1, len(t)):
        w = np.random.normal(0, 1, dt)
        x[i] = x[i - 1] + (-mu * x[i - 1] + sigma * w) * dt

    return t, x
