        if char in frequency:
            frequency[char] += 1
        else:
            frequency[char] = 1

        lo = heapq.heappop(heap)
        hi = heapq.heappop(heap)

            pair[1] = '0' + pair[1]

            pair[1] = '1' + pair[1]
