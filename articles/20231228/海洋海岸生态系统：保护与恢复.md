```python
data = pd.read_csv('生态环境监测数据.csv')

# 计算生态环境指标
def ecological_environment_index(data):
    # 计算水质参数指标
    water_quality_index = data['水质参数'].mean()
    # 计算土壤参数指标
    soil_index = data['土壤参数'].mean()
    # 计算气候参数指标
    climate_index = data['气候参数'].mean()
    # 计算生态环境指标
    ecological_environment_index = (water_quality_index + soil_index + climate_index) / 3
    return ecological_environment_index

# 执行生态环境监测
ecological_environment_index = ecological_environment_index(data)
print('生态环境指标：', ecological_environment_index)
```
