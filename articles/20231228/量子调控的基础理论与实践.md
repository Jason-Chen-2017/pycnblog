# 创建一个量子电路
qc = QuantumCircuit(2)

# 将量子态设置为初始状态
qc.initialize([[1, 0], [0, 1]], range(2))

# 将量子门应用于量子态上

qc.measure([0], [0])

# 执行量子电路
