2. 使用Java编程语言，实现一个RSA密钥对生成和加密解密示例，如下所示：

```java
import java.security.KeyPair;
import java.security.KeyPairGenerator;
import java.security.PrivateKey;
import java.security.PublicKey;
import javax.crypto.Cipher;

public class RSAExample {
    public static void main(String[] args) throws Exception {
        KeyPairGenerator keyPairGenerator = KeyPairGenerator.getInstance("RSA");
        keyPairGenerator.initialize(2048);
        KeyPair keyPair = keyPairGenerator.generateKeyPair();
        PrivateKey privateKey = keyPair.getPrivate();
        PublicKey publicKey = keyPair.getPublic();

        String data = "支付信息";
        Cipher cipher = Cipher.getInstance("RSA");
        cipher.init(Cipher.ENCRYPT_MODE, publicKey);
        byte[] encryptedData = cipher.doFinal(data.getBytes());

        cipher.init(Cipher.DECRYPT_MODE, privateKey);
        byte[] decryptedData = cipher.doFinal(encryptedData);

        System.out.println("原始数据：" + data);
        System.out.println("加密后数据：" + new String(encryptedData));
        System.out.println("解密后数据：" + new String(decryptedData));
    }
}
```

```java
import java.net.URI;

import org.apache.http.HttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.HttpClientBuilder;
import org.apache.http.util.EntityUtils;

public class OAuth2Example {
    public static void main(String[] args) throws Exception {
        String authorizeURL = "https://example.com/oauth/authorize";
        String clientID = "your_client_id";
        String redirectURI = "https://example.com/callback";
        String code = "your_authorization_code";

        String tokenURL = "https://example.com/oauth/token";
        HttpGet tokenRequest = new HttpGet(tokenURL);
        tokenRequest.setParameter("grant_type", "authorization_code");
        tokenRequest.setParameter("client_id", clientID);
        tokenRequest.setParameter("client_secret", "your_client_secret");
        tokenRequest.setParameter("redirect_uri", redirectURI);
        tokenRequest.setParameter("code", code);

        HttpResponse tokenResponse = HttpClientBuilder.create().build().execute(tokenRequest);
        String tokenResponseBody = EntityUtils.toString(tokenResponse.getEntity());
        System.out.println("tokenResponseBody：" + tokenResponseBody);

        String accessToken = "your_access_token";
        URI resourceURL = new URI("https://example.com/api/resource");

        HttpGet resourceRequest = new HttpGet(resourceURL);
        resourceRequest.setHeader("Authorization", "Bearer " + accessToken);

        HttpResponse resourceResponse = HttpClientBuilder.create().build().execute(resourceRequest);
        String resourceResponseBody = EntityUtils.toString(resourceResponse.getEntity());
        System.out.println("resourceResponseBody：" + resourceResponseBody);
    }
}
```
