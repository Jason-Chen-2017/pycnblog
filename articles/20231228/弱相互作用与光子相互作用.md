def weak_interaction(m1, m2, r, MW):
    G_F = 1.16639e-5 * np.power(np.sqrt(2), 2)
    F = (G_F**2 / np.power(r, 4)) * (m1 * m2 / MW**2)
    return F

def photon_interaction(e, r):
    h = 6.62607e-34
    I = (e**2 / (2 * np.pi**2 * h**2 * np.power(np.pi*epsilon_0, 2))) * (1 / np.power(r, 2))
    return I

m1 = 1e-27
m2 = 1e-27
r = 1e-15
MW = 8.8e-28
e = 1.60218e-19
epsilon_0 = 8.854e-12

F_weak = weak_interaction(m1, m2, r, MW)
I_photon = photon_interaction(e, r)
