```python
import numpy as np

def calculate_f_score(answers, question_type):
    if question_type == 'math':
        mean_answer = np.mean(answers)
        return np.sum((answers - mean_answer) ** 2) / len(answers)
    elif question_type == 'lang':
        mean_answer = np.mean(answers)
        return np.sum((answers - mean_answer) ** 2) / len(answers)
    elif question_type == 'logic':
        mean_answer = np.mean(answers)
        return np.sum((answers - mean_answer) ** 2) / len(answers)
    else:
        raise ValueError('Invalid question type')

math_answers = [4, 5, 6, 7, 8]
lang_answers = [9, 10, 11, 12, 13]
logic_answers = [14, 15, 16, 17, 18]

math_f_score = calculate_f_score(math_answers, 'math')
lang_f_score = calculate_f_score(lang_answers, 'lang')
logic_f_score = calculate_f_score(logic_answers, 'logic')

print('Math F score:', math_f_score)
print('Lang F score:', lang_f_score)
print('Logic F score:', logic_f_score)
```
