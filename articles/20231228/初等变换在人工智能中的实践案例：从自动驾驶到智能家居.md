import numpy as np
import matplotlib.pyplot as plt

# 自动驾驶中的初等变换应用
def auto_driving_transformation(data):
    # 数据清洗
    data = np.nan_to_num(data)
    data = data.astype(np.float32)

    # 数据过滤
    data = data[data > 0]

    # 数据标准化
    data = (data - np.mean(data)) / np.std(data)

    return data

# 智能家居中的初等变换应用
def smart_home_transformation(data):
    # 数据清洗
    data = np.nan_to_num(data)
    data = data.astype(np.float32)

    # 数据过滤
    data = data[data > 0]

    # 数据标准化
    data = (data - np.mean(data)) / np.std(data)

    return data

# 测试数据
data = np.random.rand(100)

# 自动驾驶中的初等变换应用
auto_driving_data = auto_driving_transformation(data)

# 智能家居中的初等变换应用
smart_home_data = smart_home_transformation(data)

# 可视化

plt.subplot(211)
plt.plot(data)
plt.title('原始数据')
plt.subplot(212)

plt.plot(smart_home_data, label='智能家居')
plt.title('初等变换后的数据')
plt.legend()
plt.show()
