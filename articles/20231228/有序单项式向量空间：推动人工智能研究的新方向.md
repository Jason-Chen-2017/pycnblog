```python
class PolynomialVector:
    def __init__(self, coefficients):
        self.coefficients = coefficients

    def __add__(self, other):
        return PolynomialVector([c1 + c2 for c1, c2 in zip(self.coefficients, other.coefficients)])

    def __mul__(self, other):
        return PolynomialVector([c1 * c2 for c1, c2 in zip(self.coefficients, other.coefficients)])

    def __le__(self, other):
        for i in range(max(len(self.coefficients), len(other.coefficients))):
            if self.coefficients[i] < other.coefficients[i]:
                return True
            elif self.coefficients[i] > other.coefficients[i]:
                return False
        return False


v1 = PolynomialVector([1, 2, 3])
v2 = PolynomialVector([1, 1, 1])
v3 = PolynomialVector([1, 0, 1])

print(v1 <= v2)  # True
print(v1 <= v3)  # False

v4 = v1 + v2
print(v4.coefficients)  # [2, 3, 4]

v5 = v1 * v2
print(v5.coefficients)  # [1, 2, 3]
```
