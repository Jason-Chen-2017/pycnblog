# 4.1 最近点对问题代码实例
```python
import numpy as np
from scipy.spatial.distance import cdist

def nearest_pair(points):
    n = len(points)
    min_dist = np.inf
    min_pair = None
    for i in range(n):
        for j in range(i+1, n):
            dist = cdist([points[i]], [points[j]], 'euclidean')
            if dist < min_dist:
                min_dist = dist
                min_pair = (points[i], points[j])
    return min_pair

points = np.random.rand(10, 2)
print(nearest_pair(points))
```
