## 4.1 编写一个简单的区块链程序

以下是一个简单的区块链程序的代码示例：

```python
import hashlib
import time

class Blockchain:
    def __init__(self):
        self.chain = []
        self.create_block(proof=1, previous_hash='0')

    def create_block(self, proof, previous_hash):
        block = {'index': len(self.chain) + 1,
                 'timestamp': time.time(),
                 'proof': proof,
                 'previous_hash': previous_hash}
        self.chain.append(block)
        return block

    def get_last_block(self):
        return self.chain[-1]

    def is_valid(self, index):
        block = self.chain[index]
        if index == 0:
            return True
        else:
            return hashlib.sha256(str(block['index'] + block['proof'] + block['previous_hash']).encode()).hexdigest() == block['previous_hash']

    def proof_of_work(self, last_proof):
        proof = 0
        while True:
            hash_operation = hashlib.sha256(str(proof + last_proof).encode()).hexdigest()
            if hash_operation[:4] == '0000':
                return proof
            proof += 1

    def add_block(self, proof, previous_hash):
        self.create_block(proof, previous_hash)
        self.chain.append(self.create_block(proof, previous_hash))
        self.chain = self.chain[::-1]
        return self.chain
```

## 4.3 编写一个简单的交易平台程序

以下是一个简单的交易平台程序的代码示例：

```python
from flask import Flask, request, jsonify

app = Flask(__name__)

blockchain = Blockchain()

@app.route('/mine', methods=['GET'])
def mine():
    last_block = blockchain.get_last_block()
    proof = blockchain.proof_of_work(last_block['proof'])

    blockchain.add_block(proof, last_block['hash'])

    return jsonify({'last_block': blockchain.get_last_block()})

@app.route('/transactions/new', methods=['POST'])
def new_transaction():
    values = request.get_json()

    required = ['sender', 'recipient', 'amount']
    if not all(k in values for k in required):
        return jsonify({'message': 'Missing values'}), 400

    index = len(blockchain.chain) + 1
    blockchain.add_block(0, '0')

    blockchain.chain[index]['index'] = index
    blockchain.chain[index]['timestamp'] = time.time()
    blockchain.chain[index]['transactions'] = [values]
    blockchain.chain[index]['nonce'] = 0

    return jsonify({'message': 'Transaction created'}), 201

@app.route('/chain', methods=['GET'])
def full_chain():
    response = {
        'chain': blockchain.chain,
        'length': len(blockchain.chain)
    }
    return jsonify(response), 200

if __name__ == '__main__':
    app.run(port=5000)
```
