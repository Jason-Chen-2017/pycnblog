3. 访问控制：Python语言中的基于角色的访问控制（RBAC）代码实例如下：

```python
class Role:
    def __init__(self, name):
        self.name = name
        self.permissions = []

    def add_permission(self, permission):
        self.permissions.append(permission)

class Permission:
    def __init__(self, name):
        self.name = name

class User:
    def __init__(self, name):
        self.name = name
        self.roles = []

    def add_role(self, role):
        self.roles.append(role)

# 创建角色
admin_role = Role('Admin')
user_role = Role('User')

# 创建权限
view_permission = Permission('View')
edit_permission = Permission('Edit')

# 为角色添加权限
admin_role.add_permission(view_permission)
admin_role.add_permission(edit_permission)
user_role.add_permission(view_permission)

# 为用户添加角色
user = User('John')
user.add_role(admin_role)

# 判断用户是否具有某权限
def has_permission(user, permission):
    for role in user.roles:
        for p in role.permissions:
            if p.name == permission.name:
                return True
    return False

# 使用示例
print(has_permission(user, view_permission))  # True
print(has_permission(user, edit_permission))  # True
```
