def super_conducting_power_transmission(current, resistance):

    for component in system_components:

            power = transformer_efficiency(power)
        elif component == 'transmission_line':
            power = transmission_line_efficiency(power)
        elif component == 'substation':
            power = substation_efficiency(power)
    return power

    return power * 0.99
