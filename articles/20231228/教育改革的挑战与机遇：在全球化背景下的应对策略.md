```python
import requests
from elasticsearch import Elasticsearch

# 资源收集
def collect_resources(url):
    response = requests.get(url)
    resources = response.json()
    return resources

# 资源存储
def store_resources(resources):
    es = Elasticsearch()
    for resource in resources:
        es.index(index='education', doc_type='resource', body=resource)

# 资源索引
def index_resources(es):
    query = {
        "query": {
            "match_all": {}
        }
    }
    es.search(index='education', body=query)

# 资源推荐
def recommend_resources(es, user_interest):
    query = {
        "query": {
            "match": {
                "interest": user_interest
            }
        }
    }
    response = es.search(index='education', body=query)
    return response['hits']['hits']
```

详细解释说明：

1. 资源收集：通过发送 HTTP 请求，从教育资源提供者处收集教育资源，并将其存储为 JSON 格式。
