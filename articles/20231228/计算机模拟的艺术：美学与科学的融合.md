```python

import numpy as np

# 生成随机点
def generate_random_points(num_points, point_radius):
    x = np.random.uniform(-point_radius, point_radius, num_points)
    y = np.random.uniform(-point_radius, point_radius, num_points)
    return x, y

# 绘制随机点
def draw_random_points(x, y):
    plt.scatter(x, y)
    plt.show()

# 主函数
if __name__ == '__main__':
    num_points = 100
    point_radius = 10
    x, y = generate_random_points(num_points, point_radius)
    draw_random_points(x, y)
```
