```python
import matplotlib.pyplot as plt
from scipy.signal import butter, filtfilt

# 生成随机引力波信号
t = np.linspace(0, 1, 1000)
h = np.sin(2 * np.pi * 100 * t) + 0.1 * np.random.randn(1000)

# 定义低通滤波器
fs, fc, b, a = butter(2, 10, btype='low', analog=False)

# 对引力波信号进行滤波处理
y = filtfilt(b, a, h)

# 绘制原始引力波信号和滤波后的信号
plt.figure()
plt.subplot(2, 1, 1)
plt.plot(t, h)
plt.title('Original Gravitational Wave Signal')
plt.subplot(2, 1, 2)
plt.plot(t, y)
plt.title('Filtered Gravitational Wave Signal')
plt.show()
```
