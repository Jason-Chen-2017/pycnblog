# 定义两个文本序列

text2 = ["I love machine learning", "I love artificial intelligence"]

# 使用CountVectorizer将文本序列转换为词频向量

# 使用sparse_pairwise_distances计算斯皮尔曼距离
distances = sparse_pairwise_distances(X, X, metric='jaccard')

# 打印斯皮尔曼距离
print(distances)
