# 4.具体代码实例和详细解释说明

在本节中，我们将通过一个具体的代码实例来说明蜻蜓优化算法的应用。我们将使用蜻蜓优化算法来解决一维最小化问题。

```python
import numpy as np
import random

def f(x):
    return x**2

def update_position(x, beta0, gamma, r, xi):
    return x + beta0 * np.exp(-gamma * r**2) * r * xi

def firefly_algorithm(n, max_iter):
    x = [random.uniform(-10, 10) for _ in range(n)]
    for _ in range(max_iter):
        for i in range(n):
            best_x = x[np.argmin([f(x_i) for x_i in x])]
            r = np.linalg.norm(x[i] - best_x)
            x[i] = update_position(x[i], beta0=0.5, gamma=0.2, r=r, xi=random.uniform(-1, 1))
    return x[np.argmin([f(x_i) for x_i in x])]

x_opt = firefly_algorithm(n=10, max_iter=100)
print("最优解:", x_opt)
```
