import numpy as np
import matplotlib.pyplot as plt

# 获取音频信号
audio_signal = np.sin(2 * np.pi * 440 * np.linspace(0, 1, 1000))

# 应用傅里叶变换
fft_result = np.fft.fft(audio_signal)

# 可视化频谱分析结果
plt.plot(fft_result)
plt.xlabel('Frequency (Hz)')
plt.ylabel('Amplitude')
plt.title('Frequency Spectrum Analysis')
plt.show()

# 应用低通滤波器
low_pass_filtered_signal = np.fft.ifft(np.multiply(fft_result, np.ones(len(fft_result)) - 0.5))

# 可视化谐音消除结果
plt.plot(low_pass_filtered_signal)
plt.xlabel('Time (s)')
plt.ylabel('Amplitude')
plt.title('Harmonic Suppression')
plt.show()

# 应用自动检测算法
audio_signal = np.sin(2 * np.pi * 440 * np.linspace(0, 1, 1000)) + np.sin(2 * np.pi * 880 * np.linspace(0, 1, 1000))
audio_signal = np.hstack((np.zeros(10), audio_signal, np.zeros(10)))

fft_result = np.fft.fft(audio_signal)
pitch_frequency = np.argmax(np.abs(fft_result))
pitch = pitch_frequency / len(audio_signal) / 2

print('Detected pitch:', pitch)
