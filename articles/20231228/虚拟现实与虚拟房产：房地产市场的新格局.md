```
def create_virtual_property(name, position, scale, rotation):
    bpy.ops.object.select_all(action='DESELECT')
    bpy.ops.mesh.primitive_cube_add(location=position, scale=scale, rotation=rotation)
    bpy.context.active_object.name = name
    bpy.context.view_layer.objects.active = bpy.context.active_object

create_virtual_property("virtual_house", (0, 0, 0), (5, 5, 5), (0, 0, 0))
```
