# 4.具体代码实例和详细解释说明
## 4.1 医疗资源的合理分配
```python
import json
from eth_account import Account
from web3 import Web3

# 初始化Web3对象
w3 = Web3(Web3.HTTPProvider("http://localhost:8545"))

# 获取账户对象
account = Account.create("your_private_key")

# 获取合约对象
contract = w3.eth.contract(address="your_contract_address", abi=your_abi)

# 调用合约方法
def allocate_resources(resource_id, quantity):
    tx = contract.functions.allocate_resources(resource_id, quantity).buildTransaction({
        "from": account.address,
        "gas": 1000000,
    })
    signed_tx = account.signTransaction(tx)
    tx_hash = w3.eth.sendRawTransaction(signed_tx.rawTransaction)
    tx_receipt = w3.eth.waitForTransactionReceipt(tx_hash)
    return tx_receipt

# 调用示例
allocate_resources(1, 100)
```
