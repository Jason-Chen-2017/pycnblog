# 初始化高斯过程状态空间模型
gp_state_space = GaussianProcess(kernel='rbf', alpha=1e-4)

# 初始化高斯过程动态模型
gp_dynamic = GaussianProcess(kernel='rbf', alpha=1e-4)

# 初始化高斯过程观测模型
gp_observation = GaussianProcess(kernel='rbf', alpha=1e-4)

# 生成随机数据
np.random.seed(0)
X = np.random.rand(100, 2)
y = np.sin(X[:, 0]) + np.cos(X[:, 1]) + np.random.randn(100)

# 初始化粒子滤波
pf = ParticleFilter(gp_state_space, gp_dynamic, gp_observation)
pf.initialize(X, y)

# 进行粒子滤波预测
for t in range(1, len(X)):
    pf.predict(X[t])
    pf.update(y[t])

# 输出粒子滤波结果
pf.output()
