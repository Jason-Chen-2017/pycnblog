```python
import numpy as np
import matplotlib.pyplot as plt

# 定义一元函数
def f(x):
    return x**2 + 3*x + 2

# 生成x的数组
x = np.linspace(-10, 10, 100)

# 计算y的数组
y = f(x)

# 绘制函数图像
plt.plot(x, y)
plt.xlabel('x')
plt.ylabel('y')
plt.title('一元函数的图像')
plt.grid(True)
plt.show()
```
