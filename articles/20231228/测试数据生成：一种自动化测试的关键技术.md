import random

def generate_random_data(n):
    data = []
    for _ in range(n):
        data.append({
            'name': random.choice(['Alice', 'Bob', 'Charlie']),
            'age': random.randint(18, 65),
            'gender': random.choice(['male', 'female'])
        })
    return data
