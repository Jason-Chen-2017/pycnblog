```python
# 定义商品维度
class Product:
    def __init__(self, id, name, price, category, sales):
        self.id = id
        self.name = name
        self.price = price
        self.category = category
        self.sales = sales

# 定义用户维度
class User:
    def __init__(self, id, name, address, phone):
        self.id = id
        self.name = name
        self.address = address
        self.phone = phone

# 定义订单维度
class Order:
    def __init__(self, id, product_id, user_id, purchase_time):
        self.id = id
        self.product_id = product_id
        self.user_id = user_id
        self.purchase_time = purchase_time

# 定义商品与分类之间的关系
class Category:
    def __init__(self, id, name):
        self.id = id
        self.name = name

# 创建商品与分类之间的关系
category1 = Category(1, '电子产品')
category2 = Category(2, '服装')
product1 = Product(1, '手机', 1000, category1, 100)
product2 = Product(2, '衬衫', 200, category2, 50)

# 创建订单
user1 = User(1, '张三', '北京', '13812345678')
order1 = Order(1, product1.id, user1.id, '2021-01-01 10:00:00')

# 打印订单信息
print(order1.product_id, order1.user_id, order1.purchase_time)
```
