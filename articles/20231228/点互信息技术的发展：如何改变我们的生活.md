## 4.1 信息传输代码实例

以下是一个简单的信息传输代码实例，它使用Python编程语言实现了信息的编码、传输和解码过程：

```python
import socket
import base64

# 信息编码
def encode(message):
    return base64.b64encode(message.encode())

# 信息解码
def decode(encoded_message):
    return base64.b64decode(encoded_message).decode()

# 信息传输
def transfer(sender_host, sender_port, receiver_host, receiver_port, message):
    encoded_message = encode(message)
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.connect((receiver_host, receiver_port))
        s.sendall(encoded_message)
        received_message = s.recv(1024)
        return decode(received_message)

# 测试
message = "Hello, World!"
print(transfer("localhost", 12345, "localhost", 12346, message))
```

## 4.3 人脉网络代码实例

以下是一个简单的人脉网络代码实例，它使用Python编程语言实现了人脉网络的建立、传递和交流过程：

```python
import networkx as nx

# 人脉网络建立
def create_network():
    G = nx.Graph()
    nodes = ["Alice", "Bob", "Charlie", "David"]
    edges = [("Alice", "Bob"), ("Alice", "Charlie"), ("Bob", "Charlie"), ("Charlie", "David")]
    G.add_nodes_from(nodes)
    G.add_edges_from(edges)
    return G

# 信息传递
def propagate_info(G, source_node, target_node, message):
    G.nodes[source_node]['message'] = message
    for neighbor in G.neighbors(source_node):
        G.nodes[neighbor]['message'] = G.nodes[neighbor]['message'] + G.nodes[source_node]['message']
    return G.nodes[target_node]['message']

# 测试
G = create_network()
message = "Hello, friends!"
print(propagate_info(G, "Alice", "David", message))
```
