```
  "name": "Simple Business Process",
  "components": [
    {
      "type": "task",
      "name": "Complete Task",
      "description": "Complete a task"
    },
    {
      "type": "condition",
      "name": "Check Condition",
      "description": "Check a condition"
    }
  ],
  "relations": [
    {
      "source": "task",
      "target": "condition"
    }
  ]
}
```

```
    {
      "type": "condition",
      "name": "Check Condition",
      "description": "Check a condition",
      "attributes": {

  "relations": [
    {

  ]
```
