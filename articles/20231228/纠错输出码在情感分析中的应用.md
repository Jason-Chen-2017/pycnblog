def encode(bitstream):

    k = 5
    C = np.zeros(n + k)
    for i in range(n):
        C[i] = bitstream[i]
    for i in range(k):
        C[n + i] = 0
    return C

def decode(C):
    n = len(C) - 1
    d = 0
    for i in range(n):
        if C[i] != C[i + 1]:
            d += 1
    return d

d = decode(C)
