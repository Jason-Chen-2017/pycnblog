import matplotlib.pyplot as plt
from scipy.signal import find_peaks

# 生成一个简单的音频信号
fs = 1000  # 采样率
t = np.arange(0, 1, 1/fs)  # 时间域样本
f = 5  # 频率
signal = np.sin(2*np.pi*f*t)  # 信号

# 傅里叶变换
frequencies, amplitudes = signal.fftfreq(len(signal))[0:100], np.abs(np.fft.fft(signal))
plt.plot(frequencies, amplitudes)
plt.xlabel('Frequency (Hz)')
plt.ylabel('Amplitude')
plt.title('Fourier Transform')
plt.show()

# 波形匹配
template = np.sin(2*np.pi*10*t)  # 模板信号
matched, _ = find_peaks(np.correlate(signal, template, mode='valid'))  # 匹配结果
plt.plot(t, signal)
plt.plot(t[matched], template, 'r')
plt.title('Waveform Matching')
plt.show()
