## 4.1 简单短视频平台推荐系统的代码实例

以下是一个简单的短视频平台推荐系统的代码实例，使用 Python 编程语言编写：

```python
import numpy as np

def cosine_similarity(x, y):
    dot_product = np.dot(x, y)
    norm_x = np.linalg.norm(x)
    norm_y = np.linalg.norm(y)
    return dot_product / (norm_x * norm_y)

def recommend_videos(user_id, video_data):
    user_preferences = video_data[user_id]
    similarities = []
    for video_id, video in video_data.items():
        if video_id != user_id:
            similarity = cosine_similarity(user_preferences, video)
            similarities.append((video_id, similarity))
    similarities.sort(key=lambda x: x[1], reverse=True)
    return similarities

video_data = {
    1: np.array([1, 2, 3]),
    2: np.array([2, 3, 4]),
    3: np.array([3, 4, 5]),
    4: np.array([4, 5, 6]),
}

user_id = 1
recommended_videos = recommend_videos(user_id, video_data)
print(recommended_videos)
```

4. 定义了一个 `video_data` 字典，用于存储所有视频的观看记录。
