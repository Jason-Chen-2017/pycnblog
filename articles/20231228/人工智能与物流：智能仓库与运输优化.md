```python
def distance(pos1, pos2):
    return ((pos1[0] - pos2[0]) ** 2 + (pos1[1] - pos2[1]) ** 2) ** 0.5

def assign_position(goods, positions):
    for good in goods:
        min_dist = float('inf')
        min_pos = None
        for pos in positions:
            dist = distance(good.pos, pos)
            if dist < min_dist:

                min_pos = pos
```

```python
def first_come_first_served(orders, goods):
    for order in orders:
        good = None
        for g in goods:
            if g.type == order.type and g.quantity >= order.quantity:
                good = g
                break
        if good is None:
            print("No such good")
            continue
        good.quantity -= order.quantity
        order.good = good
```

```python
def shortest_path(order, goods):
    min_dist = float('inf')
    min_good = None
    for good in goods:
        dist = distance(order.pos, good.pos)
        if dist < min_dist:
            min_dist = dist
            min_good = good
    return min_good

def shortest_path_pickup(orders, goods):
```
