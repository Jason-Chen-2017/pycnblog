# 加载用户行为数据

from scipy.optimize import minimize

    memory_usage = x[1]
    gpu_usage = x[2]

import networkx as nx
import matplotlib.pyplot as plt

# 创建社交网络图
G = nx.Graph()

# 加载用户关系数据
data = pd.read_csv('social_network.csv')

# 添加用户节点和关系边
for index, row in data.iterrows():
