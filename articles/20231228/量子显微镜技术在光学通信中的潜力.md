```python
import numpy as np

def quantum_tomography(rho):
    d = rho.shape[0]
    U = np.zeros((d, d, d), dtype=np.complex)
    for i in range(d):
        for j in range(d):
            U[i][j] = np.trace(rho @ np.kron(np.eye(d), np.identity(d) @ np.kron(np.eye(d), eps[i, j])))
    return U

def quantum_state_reconstruction(U):
    d = U.shape[0]
    rho = np.zeros((d, d), dtype=np.complex)
    for i in range(d):
        for j in range(d):
            rho[i][j] = np.trace(U @ np.kron(np.eye(d), np.identity(d) @ np.kron(np.eye(d), eps[i, j].conj())))
    return rho

eps = np.array([[0, 0], [0, 1]])
rho = np.random.rand(2, 2)
U = quantum_tomography(rho)
rho_recon = quantum_state_reconstruction(U)
print("Original quantum state:")
print(rho)
print("\nReconstructed quantum state:")
print(rho_recon)
```
