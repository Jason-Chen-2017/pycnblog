## 4.1 极值分布的估计

### 4.1.1 极大值法

```python
import numpy as np

def max_value_estimate(data):
    data = np.sort(data)
    max_value = data[-1]
    return max_value

data = np.random.normal(0, 1, 1000)
max_value = max_value_estimate(data)
print("极大值:", max_value)
```

```python
def min_value_estimate(data):

```

```python
def mixed_value_estimate(data):
```

### 4.2.1 极大值分布

```python
import scipy.stats as stats

def max_value_distribution(data, mu, sigma, beta):
    dist = stats.genextreme_r(mu=mu, loc=mu, scale=sigma, shape=beta)
    max_value = dist.rvs(size=1)[0]
    return max_value

mu = 0
sigma = 1
beta = 2
max_value = max_value_distribution(data, mu, sigma, beta)
print("极大值:", max_value)
```

### 4.2.2 极小值分布

```python
def min_value_distribution(data, mu, sigma, beta):
    dist = stats.genextreme_l(mu=mu, loc=mu, scale=sigma, shape=beta)
    min_value = dist.rvs(size=1)[0]
    return min_value

min_value = min_value_distribution(data, mu, sigma, beta)

```

### 4.2.3 混合极值分布

```python
def mixed_value_distribution(data, mu1, sigma1, beta1, mu2, sigma2, beta2):
    p = 0.5
    dist1 = stats.genextreme_r(mu=mu1, loc=mu1, scale=sigma1, shape=beta1)
    dist2 = stats.genextreme_l(mu=mu2, loc=mu2, scale=sigma2, shape=beta2)
    mixed_value = p * dist1.rvs(size=1)[0] + (1 - p) * dist2.rvs(size=1)[0]
    return mixed_value
```
