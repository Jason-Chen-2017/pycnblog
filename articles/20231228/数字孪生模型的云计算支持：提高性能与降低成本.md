from google.cloud import pubsub_v1

project_id = 'my_project'
topic_id = 'sensor_data'

subscriber = pubsub_v1.SubscriberClient()
subscription_path = subscriber.subscription_path(project_id, topic_id)

def callback(message):
    print(f'Received {message}')
    message.ack()

subscriber.subscribe(subscription_path, callback=callback)
```

### 4.2 数据存储与管理

from google.cloud import bigquery

project_id = 'my_project'
dataset_id = 'sensor_data'
table_id = 'sensor_data_table'

client = bigquery.Client(project=project_id)
dataset_ref = client.dataset(dataset_id)
table_ref = dataset_ref.table(table_id)

job_config = bigquery.LoadJobConfig()
job_config.autodetect = True
job_config.source_format = bigquery.SourceFormat.NEWLINE_DELIMITED_JSON

with open('sensor_data.json', 'w') as f:
    f.write('{"timestamp": "2021-01-01T00:00:00", "value": 100}\n')

job = client.load_table_from_file('sensor_data.json', job_config=job_config)
job.result()
```

### 4.3 数据处理与分析

然后，我们需要对收集到的数据进行处理和分析。我们可以使用GCP的Dataflow服务来实现数据的处理和分析。

```python
from apache_beam import Pipeline
from apache_beam.options.pipeline_options import PipelineOptions
from apache_beam.io import ReadFromBigQuery, WriteToBigQuery

pipeline_options = PipelineOptions([
    '--project=my_project',
    '--runner=DataflowRunner',
    '--temp_location=gs://my_bucket/temp',
    '--region=us-central1',
])

pipeline = Pipeline(options=pipeline_options)

sensor_data = (
    pipeline
    | 'ReadSensorData' >> ReadFromBigQuery(
        table='my_project:my_dataset.my_table',
        use_standard_sql=True,
    )
    | 'FilterValidData' >> BeamPyDoFn(filter_valid_data)
    | 'CalculateStatistics' >> BeamPyDoFn(calculate_statistics)
    | 'WriteStatistics' >> WriteToBigQuery(
        'my_project:my_dataset.my_table_statistics',
        schema='timestamp:TIMESTAMP, value:INTEGER, min_value:INTEGER, max_value:INTEGER, avg_value:FLOAT',
        create_disposition=BeamPyDoFn.CreateDisposition.CREATE_IF_NEEDED,
        write_disposition=BeamPyDoFn.WriteDisposition.WRITE_APPEND,
    )
)

result = pipeline.run()
result.wait_until_finish()
```

### 4.4 模型构建与更新

接下来，我们需要构建数字模型并实现模型的更新。我们可以使用GCP的AI Platform服务来实现模型的训练和预测。

```python
from google.cloud import aiplatform

project_id = 'my_project'
model_name = 'sensor_data_model'

client = aiplatform.ProjectClient(project=project_id)

# 创建模型
model = aiplatform.Model(
    display_name=model_name,
    description='A model for predicting sensor data',
    base_model_type='ONNX',
    model_type='ONNX',
)

# 训练模型
model.train(
    scaler=aiplatform.Scaler(
        display_name='sensor_data_scaler',
        model_path='gs://my_bucket/sensor_data_model',
    ),
    model_path='gs://my_bucket/sensor_data_model',
)

# 预测
predictor = aiplatform.Interpreter(
    display_name='sensor_data_predictor',
    model_path='gs://my_bucket/sensor_data_model',
)

input_data = {
    'timestamp': '2021-01-01T00:00:00',
    'value': 100,
}

result = predictor.predict(input_data)
print(result)
```

### 4.5 模型与物理实体的同步与对应

最后，我们需要实现数字模型与物理实体之间的实时同步和互相对应关系。我们可以使用GCP的Cloud Functions服务来实现实时同步。

```python
from google.cloud import functions

project_id = 'my_project'
region = 'us-central1'
trigger_name = 'sensor_data_trigger'

deploy_context = functions.DeployContext(
    project_id=project_id,
    region=region,
    trigger_name=trigger_name,
)

def sensor_data_trigger(event, context):
    sensor_data = event['data']
    timestamp = event['timestamp']
    value = sensor_data['value']

    # 更新数字模型
    model.update(
        scaler=aiplatform.Scaler(
            display_name='sensor_data_scaler',
            model_path='gs://my_bucket/sensor_data_model',
        ),
        model_path='gs://my_bucket/sensor_data_model',
        input_data={'timestamp': timestamp, 'value': value},
    )

    # 更新物理实体
    client.update_physical_entity(timestamp, value)

deploy_context.deploy()
```
