# 图像读取

# 梯度计算
grad_x = cv2.Sobel(img_preprocessed, cv2.CV_64F, 1, 0, ksize=3)
grad_y = cv2.Sobel(img_preprocessed, cv2.CV_64F, 0, 1, ksize=3)
grad = np.hypot(grad_x, grad_y)

# 阈值分割
ret, binary = cv2.threshold(grad, 0.1 * np.max(grad), 255, cv2.THRESH_BINARY)

# 边缘连通域分析
contours, hierarchy = cv2.findContours(binary, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)

# 绘制边缘
img_edges = np.zeros_like(img)
cv2.drawContours(img_edges, contours, -1, (255, 0, 0), 2)

# 图像显示
plt.subplot(121), plt.imshow(img, cmap='gray')
plt.title('Original Image'), plt.xticks([]), plt.yticks([])

plt.title('Edge Detection'), plt.xticks([]), plt.yticks([])
