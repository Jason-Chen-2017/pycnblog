```python
# 生成一个密钥
key = Fernet.generate_key()

# 初始化一个加密器
cipher_suite = Fernet(key)

# 加密一个消息
cipher_text = cipher_suite.encrypt(b"Hello, World!")

# 解密一个消息
plain_text = cipher_suite.decrypt(cipher_text)

print(plain_text.decode())
```
