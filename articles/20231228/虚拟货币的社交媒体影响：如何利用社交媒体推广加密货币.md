```python
import hashlib
import hmac
import json
import requests

# 设置API密钥和密钥
api_key = 'your_api_key'
secret_key = 'your_secret_key'

# 设置请求头和参数
headers = {'Content-Type': 'application/json', 'API-Key': api_key}
params = {'symbol': 'BTC/USD', 'amount': 0.1}

# 生成签名
signature = hmac.new(secret_key.encode(), json.dumps(params).encode(), hashlib.sha384).digest()

# 发起请求
response = requests.get('https://api.example.com/v1/ticker', headers=headers, params=params, auth=('your_username', signature.hex()))

# 解析响应
data = response.json()
print(data)
```
