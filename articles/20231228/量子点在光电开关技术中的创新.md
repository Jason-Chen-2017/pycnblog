import matplotlib.pyplot as plt

def schrodinger_equation(psi, dt, dx, V):

    k_squared = k**2
    psi_new = (1 - dt**2 * k_squared / (2 * np.abs(V)**2)) * psi + dt * (1 / (2 * np.abs(V))) * (np.fft.fft(psi) - V * psi)
    return psi_new

def quantum_dot_switch(light_intensity, dt, dx, V):
    N = int(light_intensity / dt)
    psi = np.zeros(N, dtype=complex)
    psi[0] = 1
    for n in range(1, N):
        psi[n] = schrodinger_equation(psi[n-1], dt, dx, V)
    plt.plot(psi)
    plt.xlabel('Quantum State')
    plt.ylabel('Amplitude')
    plt.title('Quantum Dot Switch')
    plt.show()
