```python
import numpy as np

# 使用Haar特征检测器检测门匠工具

doors = haar_cascade.detectMultiScale(img, scaleFactor=1.1, minNeighbors=5)

# 绘制检测结果
for (x, y, w, h) in doors:
    cv2.rectangle(img, (x, y), (x+w, y+h), (255, 0, 0), 2)

# 显示结果
cv2.imshow('Door Detection', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
