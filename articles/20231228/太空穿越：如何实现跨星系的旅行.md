class Starship:
    def __init__(self, energy_source, propulsion_system, life_support_system):
        self.energy_source = energy_source
        self.propulsion_system = propulsion_system
        self.life_support_system = life_support_system

    def fly(self, distance):
        energy = self.energy_source.provide_energy(distance)
        velocity = self.propulsion_system.accelerate(energy)
        self.life_support_system.maintain(distance)
        return velocity

class NuclearEnergy:
    def provide_energy(self, distance):
        return distance * 1e12  # Joule

class IonPropulsion:
    def accelerate(self, energy):
        return energy / 1000  # m/s

class BiologicalLifeSupport:
    def maintain(self, distance):
        return "Maintain life support system at distance {}".format(distance)

starship = Starship(NuclearEnergy(), IonPropulsion(), BiologicalLifeSupport())
distance = 10  # AU
velocity = starship.fly(distance)
print("Velocity: {} m/s".format(velocity))
