    t = np.linspace(-10, 10, 1000)
    p = (1 / np.sqrt(2 * np.pi * sigma**2)) * np.exp(-(t - mu)**2 / (2 * sigma**2))
    return t, p

plt.plot(t, p)
plt.xlabel('Time')

plt.title('Time Arrow Algorithm')
plt.show()
