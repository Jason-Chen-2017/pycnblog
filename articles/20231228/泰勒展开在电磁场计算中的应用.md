```python
import numpy as np

def taylor_expansion(f, a, n):
    f_prime = lambda x: f(x+1) - f(x)
    taylor_coefficients = [f(a)]
    for k in range(1, n+1):
        f_prime_k = f_prime(a)
        taylor_coefficients.append(f_prime_k / k)
        a += 1
    return taylor_coefficients

def taylor_approximation(f, a, coefficients):
    return sum(coefficients[i] * (x-a)**i for i in range(len(coefficients)))

def main():
    f = lambda x: np.exp(x)
    a = 0
    n = 5
    coefficients = taylor_expansion(f, a, n)
    x = np.linspace(-1, 1, 100)
    y = taylor_approximation(f, a, coefficients)
    plt.plot(x, y, label='Taylor approximation')
    plt.plot(x, np.exp(x), label='Exact solution')
    plt.legend()
    plt.show()

if __name__ == '__main__':
    main()
```
