from sklearn.metrics import jaccard_similarity_score

    similarity = jaccard_similarity_score(culture1, culture2)
    return similarity

def cultural_fusion(culture1, culture2):
    fusion = np.mean([culture1, culture2])
    return fusion

def cultural_exchange(cultures):
    similarities = []
    fusions = []
    for i in range(len(cultures)):
        for j in range(i+1, len(cultures)):
            similarity = cultural_similarity(cultures[i], cultures[j])
            fusion = cultural_fusion(cultures[i], cultures[j])
            similarities.append(similarity)
            fusions.append(fusion)
    return similarities, fusions

cultures = [culture1, culture2, culture3, culture4]
similarities, fusions = cultural_exchange(cultures)
