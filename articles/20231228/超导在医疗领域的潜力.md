## 4.1 超导磁共振成像（MRI）的代码实例

```python

import scipy.fftpack as fftpack

def MRI_pulse(B0, B1, t1, t2, TR, TE, flip_angle):
    # 定义体内水分质的脉冲磁共振测量函数
    def pulse_sequence(t):
        # 生成脉冲磁场
        B1_pulse = B1 * np.exp(-(t - t1/2)**2 / (2 * t2**2)) * np.sin(2 * np.pi * 64 * t)
        # 生成磁场变化的信号
        signal = B0 * (1 - np.exp(-t / T1)) * np.exp(-t / T2) + B1_pulse * np.exp(-t / T1)
        return signal

    # 生成时间序列
    t = np.linspace(0, TR, int(TR * 1e-3))
    # 生成脉冲磁共振测量信号
    signal = pulse_sequence(t)
    # 对信号进行傅里叶变换
    f = fftpack.fft(signal)
    # 获取脉冲磁共振信号的频率和强度
    frequency = np.fft.fftfreq(len(signal))
    intensity = np.abs(f)**2
    return frequency, intensity

# 设置参数
B0 = 1.5  # 静态磁场（T）
B1 = 1000  # 脉冲磁场（A/m）
t1 = 0.5  # T1时间（ms）
t2 = 0.1  # T2时间（ms）
TR = 2  # TR时间（s）
TE = 1  # TE时间（ms）
flip_angle = np.pi / 2  # 翻转角度（弧度）

# 计算脉冲磁共振测量的频率和强度

# 打印结果

print(frequency)
print(intensity)
```
