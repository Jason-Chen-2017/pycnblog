def calculate_resistance(length, area, resistivity):
    return length / (area * resistivity)

def calculate_capacitance(area, distance, permittivity):
    return permittivity * area / distance

capacitance = calculate_capacitance(area, distance, permittivity)

print("电阻值：{:.6f} 欧姆".format(resistance))
