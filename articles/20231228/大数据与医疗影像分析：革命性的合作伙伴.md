    # 数据纠正

    # 数据过滤
    data = pd.DataFrame(data.dropna())
    return data

def data_fusion(data1, data2):
    # 数据转换
    data1 = pd.DataFrame(data1.apply(lambda x: x.map(lambda y: y.replace('A', '1').replace('B', '2').replace('C', '3'))))
    data2 = pd.DataFrame(data2.apply(lambda x: x.map(lambda y: y.replace('A', '1').replace('B', '2').replace('C', '3'))))
    # 数据重映射
    data1 = pd.DataFrame(data1.apply(lambda x: x.map(lambda y: y.replace('1', 'A').replace('2', 'B').replace('3', 'C'))))
    data2 = pd.DataFrame(data2.apply(lambda x: x.map(lambda y: y.replace('1', 'A').replace('2', 'B').replace('3', 'C'))))
    # 数据融合
    data = pd.concat([data1, data2], axis=1)
    return data

# 数据减噪
def data_denoising(data, method='median'):
    if method == 'median':
        data = pd.DataFrame(data.apply(lambda x: x.map(lambda y: y.median())))
    elif method == 'mean':
        data = pd.DataFrame(data.apply(lambda x: x.map(lambda y: y.mean())))
