```python
import numpy as np

# 运动选择算法
def motion_selection(interest, ability, projects):
    suitable_projects = []
    for project in projects:
        if project['interest_level'] >= interest and project['ability_level'] <= ability:
            suitable_projects.append(project)
    ranked_projects = sorted(suitable_projects, key=lambda x: x['availability'] + x['student_demand'], reverse=True)
    selected_project = ranked_projects[0]
    return selected_project

# 运动评估算法
def motion_evaluation(performance, level):
    level_score = {
        'beginner': 1,
        'intermediate': 2,
        'advanced': 3
    }
    progress_score = np.mean(performance)
    total_score = level_score[level] + progress_score
    grade = 'A' if total_score >= 90 else 'B' if total_score >= 80 else 'C'
    return grade

# 运动安排算法
def motion_scheduling(time, location, frequency, activities):
    suitable_activities = []
    for activity in activities:
        if activity['time_range'] == time and activity['location'] == location and activity['frequency'] == frequency:
            suitable_activities.append(activity)
    ranked_activities = sorted(suitable_activities, key=lambda x: x['availability'] + x['student_demand'], reverse=True)
    selected_activity = ranked_activities[0]
    return selected_activity
```
