# 读取数据
data = pd.read_csv('data.csv')

# 数据清洗
data['column1'] = data['column1'].str.strip()

### 4.2 数据清洗的代码实例

```python
import pandas as pd

# 读取数据
data = pd.read_csv('data.csv')

# 数据整理
data = data.dropna()

# 数据预处理
data['column1'] = data['column1'].str.strip()
data['column2'] = data['column2'].str.replace(r'[^\w]', '', regex=True)

# 数据纠正
data['column3'] = data['column3'].apply(lambda x: x if x > 0 else None)

# 数据填充
data['column4'] = data['column4'].fillna(method='bfill')

# 数据去噪
data = data.dropna()
```
