```python
import pandas as pd

# 加载数据集
data = pd.read_csv('customer_data.csv')

# 删除重复数据
data = data.drop_duplicates()

# 填充缺失值

# 转换数据类型
data['age'] = data['age'].astype(int)
```

```python

    min_value = data[column].min()

# 检查数据一致性
```
