def analyze_fault(fault_probability):
    fault_count = 0
    for i in range(1000):
        if np.random.rand() < fault_probability:
            fault_count += 1
    return fault_count / 1000

def main():
    fault_probability = 0.01
    fault_count = analyze_fault(fault_probability)
    print("Fault count: ", fault_count)

if __name__ == "__main__":
    main()
