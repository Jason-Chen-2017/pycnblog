```python
import numpy as np
import qiskit
from qiskit import QuantumCircuit, Aer, transpile, assemble
from qiskit.visualization import plot_histogram

# 创建一个量子电路
qc = QuantumCircuit(2, 2)

# 初始化量子粒子的自旋状态
qc.h(0)
qc.h(1)

# 应用外部磁场对量子粒子的自旋状态进行操作
qc.x(0)
qc.x(1)

# 对量子粒子的自旋状态进行测量
qc.measure([0, 1], [0, 1])

# 将量子电路编译为可执行的代码
qc = transpile(qc, basis_gates=['u1', 'u2', 'u3', 'cx', 'ch'])

# 使用QASM模拟器执行量子电路
simulator = Aer.get_backend('qasm_simulator')
job = simulator.run(assemble(qc))

# 解码测量结果
counts = job.result().get_counts()
print(counts)
```
