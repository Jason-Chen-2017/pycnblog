### 1.4.1 有向图的拓扑相变检测算法实例

```python
import networkx as nx

def detect_topological_transition(G, state1, state2):
    # 构建有向图
    G1 = nx.DiGraph()
    G2 = nx.DiGraph()
    for node, neighbors in G.nodes(data=True):
        if state1[node] == state2[node]:
            G1.add_node(node, **state1[node])
            G2.add_node(node, **state2[node])
        elif state1[node] == 1:
            G1.add_node(node, **state1[node])
            G2.add_node(node, **state2[node])
        elif state2[node] == 1:
            G1.add_node(node, **state1[node])
            G2.add_node(node, **state2[node])
        else:
            G1.add_node(node, **state1[node])
            G2.add_node(node, **state2[node])
    # 检测拓扑相变
    if nx.is_isomorphic(G1, G2):
        return False
    else:
        return True
```
