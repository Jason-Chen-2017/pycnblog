### 1.4.1 双极子放大器示例

双极子放大器是非线性模拟电路的典型例子，其主要组成部分包括：输入双极子、输出双极子和电容器。双极子放大器的工作原理是利用双极子的输入输出关系实现信号放大。以下是双极子放大器的具体代码实例和详细解释说明：

```python
import numpy as np
import matplotlib.pyplot as plt

# 双极子参数
R1 = 1000
R2 = 100
C = 1e-6
Vt = 0.5

# 输入信号
t = np.linspace(0, 1, 1000)
v_in = np.sin(2 * np.pi * 1000 * t)

# 双极子放大器模型
def amplifier_model(v_in, R1, R2, C, Vt):
    i_in = v_in / R1
    v_be = i_in * R2
    i_c = v_be / (1 / (1 + 1 / (Vt / C)))
    v_out = i_c * R1
    return v_out

# 求解输出波形
v_out = amplifier_model(v_in, R1, R2, C, Vt)

# 绘制输入输出波形
plt.figure()
plt.subplot(211)
plt.plot(t, v_in, label='Input')
plt.plot(t, v_out, label='Output')
plt.legend()
plt.xlabel('Time [s]')
plt.ylabel('Voltage [V]')
plt.title('BJT Amplifier')

# 绘制频谱分析
plt.subplot(212)
plt.specgram(v_out, Fs=1000)
plt.xlabel('Time [s]')
plt.ylabel('Frequency [Hz]')
plt.title('Spectrogram')

plt.show()
```

电容器充电滤波器是非线性模拟电路的另一个典型例子，其主要组成部分包括：电容器、电阻和输入信号源。电容器充电滤波器的工作原理是利用电容器的充电特性实现低频信号传递，高频信号阻断。以下是电容器充电滤波器的具体代码实例和详细解释说明：

```python
import numpy as np
import matplotlib.pyplot as plt

# 电容器充电滤波器参数
C1 = 1e-6
C2 = 1e-6
R = 1000
f_in = 1000
f_out = 10

# 输入信号
t = np.linspace(0, 1, 1000)
v_in = np.sin(2 * np.pi * f_in * t)

# 电容器充电滤波器模型
def RC_filter_model(v_in, C1, C2, R, f_in, f_out):
    w_in = 2 * np.pi * f_in
    w_out = 2 * np.pi * f_out
    H = w_out / w_in
    alpha = 1 - H**2
    beta = 2 * H
    gamma = H**2
    i_c = v_in / R
    v_c1 = i_c * (1 / (1 / C1 + 1 / C2))
    v_c2 = v_c1 * alpha + gamma * np.integrate(v_c1, 0, t)
    v_out = v_c2 / (1 / C2 + R)
    return v_out

# 求解输出波形
v_out = RC_filter_model(v_in, C1, C2, R, f_in, f_out)

# 绘制输入输出波形
plt.figure()
plt.subplot(211)
plt.plot(t, v_in, label='Input')
plt.plot(t, v_out, label='Output')
plt.legend()
plt.xlabel('Time [s]')
plt.ylabel('Voltage [V]')
plt.title('RC Filter')

# 绘制频谱分析
plt.subplot(212)

plt.xlabel('Time [s]')
plt.ylabel('Frequency [Hz]')
plt.title('Spectrogram')
```
