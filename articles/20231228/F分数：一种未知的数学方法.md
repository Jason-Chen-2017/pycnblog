$$

def euclidean_distance(a, b):
    return np.sqrt(np.sum((a - b) ** 2))

def f_score(a, b, alpha=1.0, beta=0.5):
    distance = euclidean_distance(a, b)
    f_score = 1 / (1 + np.exp(-alpha * (distance - beta)))
    return f_score

# 示例向量

vector_b = np.array([4, 5, 6])

f_score = f_score(vector_a, vector_b)
print(f"F 分数: {f_score}")
