import random

def initialize_fireflies(n, lb, ub):
    return np.random.uniform(lb, ub, (n, len(lb)))

def calculate_luminance(x, f):
    return f(x)

def update_position(x, y, c, beta0, beta1, r):
    return x + c * beta0 * np.exp(-beta1 * r**2) * (y - x)

def firefly_algorithm(f, lb, ub, n, max_iter):
    fireflies = initialize_fireflies(n, lb, ub)
    luminances = np.array([calculate_luminance(firefly, f) for firefly in fireflies])

    for _ in range(max_iter):
        for i in range(n):
            for j in range(n):
                if luminances[j] > luminances[i]:

                    fireflies[i] = update_position(fireflies[i], fireflies[j], random.random(), beta0, beta1, r)
                    luminances[i] = calculate_luminance(fireflies[i], f)

    return fireflies, luminances
