```python
import math

class Spacecraft:
    def __init__(self, mass, acceleration):
        self.mass = mass
        self.acceleration = acceleration

    def calculate_force(self):
        return self.mass * self.acceleration

    def calculate_velocity(self, time):
        return self.acceleration * time

    def calculate_distance(self, time):
        return self.calculate_velocity(time) * time + 0.5 * self.acceleration * time**2

spacecraft = Spacecraft(1000, 9.8)

velocity = spacecraft.calculate_velocity(time)

print("Distance:", distance)
```

```python
class CommunicationSystem:
    def __init__(self, power, resistance, capacitance, inductance):
        self.power = power
        self.resistance = resistance
        self.capacitance = capacitance
        self.inductance = inductance

    def calculate_bandwidth(self):
        return 2 * self.power / (self.resistance + (self.capacitance * self.inductance)**2)

    def calculate_noise_margin(self):
        return self.power / (self.resistance + self.capacitance + self.inductance)

communication_system = CommunicationSystem(100, 1, 1e-9, 1e-6)
bandwidth = communication_system.calculate_bandwidth()
noise_margin = communication_system.calculate_noise_margin()

print("Bandwidth:", bandwidth)
print("Noise Margin:", noise_margin)
```
