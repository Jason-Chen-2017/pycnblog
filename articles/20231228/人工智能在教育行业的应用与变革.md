```python
import numpy as np
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 加载数据
data = pd.read_csv('resource_data.csv')

# 数据预处理
X = data['content']

# 特征选择
vectorizer = TfidfVectorizer()
X = vectorizer.fit_transform(X)

# 模型选择
model = cosine_similarity(X)

# 资源推荐
def recommend_resource(resource_id, similarity_threshold=0.8):
    similarity_matrix = model[resource_id]
    recommended_resources = []
    for i, similarity in enumerate(similarity_matrix):
        if similarity > similarity_threshold:
            recommended_resources.append(i)
    return recommended_resources

# 测试
print(recommend_resource(0))
```
