import numpy as np
import matplotlib.pyplot as plt

# 定义思考能力函数
def thinking_ability(x, k=1, c=0):
    return np.sqrt(1 + np.exp(-k * (x - c)))

# 计算不同理解程度下的思考能力

T = thinking_ability(x)

# 绘制思考能力曲线

plt.show()
