import matplotlib.pyplot as plt

# 目标函数
def objective_function(x):
    return 0.5 * np.sum((x - 0.5)**2)

# 约束条件
def constraint_1(x):
    return np.sum(x - 0.5) - 0.8

def constraint_2(x):
    return np.sum(x - 0.5) - 0.8

def constraint_3(x):
    return np.sum(x - 0.5) - 0.8

# 求解优化问题
x_optimal = np.optimize.minimize(objective_function, x0=[0.5], constraints=[{'type': 'ineq', 'fun': constraint_1}, {'type': 'ineq', 'fun': constraint_2}, {'type': 'eq', 'fun': constraint_3}]).x

plt.xlabel('Adaptation Level')
plt.ylabel('Objective Value')
plt.legend()
plt.show()
