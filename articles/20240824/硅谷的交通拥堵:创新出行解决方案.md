### 5.2 源代码详细实现

以下是一个简单的交通流量预测和路径优化的 Python 代码实例：

```python
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error
import matplotlib.pyplot as plt

# 数据收集与预处理
data = pd.read_csv('traffic_data.csv')
data = data[['time', 'traffic_light', 'weather', 'road_condition', 'traffic_flow']]
data = data[data['time'] == '10:00']

# 交通流量预测模型
X = data[['traffic_light', 'weather', 'road_condition']]
y = data['traffic_flow']
model = LinearRegression()
model.fit(X, y)
y_pred = model.predict(X)

# 路径优化模型
def dijkstra(graph, start, end):
    distances = {node: float('infinity') for node in graph}
    distances[start] = 0
    visited = set()

    while True:
        next_node = min((dist, node) for node, dist in distances.items() if node not in visited)
        if next_node is None:
            break
        visited.add(next_node[1])

        for neighbor, weight in graph[next_node[1]].items():
            old_distance = distances[neighbor]
            new_distance = distances[next_node[1]] + weight
            if new_distance < old_distance:
                distances[neighbor] = new_distance

    return distances[end]

# 运行结果展示
plt.scatter(data['traffic_flow'], y_pred)
plt.xlabel('Actual Traffic Flow')
plt.ylabel('Predicted Traffic Flow')
plt.title('Traffic Flow Prediction')
plt.show()

print('Optimal Path:', dijkstra(graph, 'A', 'Z'))
```
