```python
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import MinMaxScaler
from sklearn.metrics.pairwise import cosine_similarity
from sklearn.metrics import mean_squared_error

# 数据预处理
data = pd.read_csv('user_behavior.csv')
X = data[['course_id', 'click', 'buy', 'review']]
y = data['rating']

# 数据标准化
scaler = MinMaxScaler()
X_scaled = scaler.fit_transform(X)

# 划分训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y, test_size=0.2, random_state=42)

# 计算用户和课程的余弦相似度
user_similarity = cosine_similarity(X_train, X_train)
course_similarity = cosine_similarity(X_test, X_test)

# 为用户推荐课程
def recommend_courses(user_id, course_id, similarity_matrix, top_n=5):
    user_similarity = similarity_matrix[user_id]
    scores = {}
    for i, score in enumerate(user_similarity):
        if i != user_id and i != course_id:
            scores[i] = score
    sorted_scores = sorted(scores.items(), key=lambda x: x[1], reverse=True)
    return sorted_scores[:top_n]

# 计算推荐结果的均方误差
predicted_ratings = [recommend_courses(i, course_id, course_similarity) for i, course_id in enumerate(X_test)]
mse = mean_squared_error(y_test, predicted_ratings)
print('MSE:', mse)
```

- **计算用户和课程的余弦相似度**：利用余弦相似度计算用户和课程之间的相似度。
- **为用户推荐课程**：根据用户和课程的相似度，为用户推荐课程。
