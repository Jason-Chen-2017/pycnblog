    image = tf.image.resize(image, [224, 224])

    return image

def build_model():
    input_layer = tf.keras.Input(shape=(224, 224, 3))
    x = Conv2D(32, (3, 3), activation='relu')(input_layer)

    x = Flatten()(x)
