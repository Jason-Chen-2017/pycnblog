```python
import numpy as np
import matplotlib.pyplot as plt

# 定义线性规划模型
def linear_programming(x, y):
    c = np.array([2, 3])  # 目标函数系数
    A = np.array([[1, 1], [0, 1]])  # 约束条件系数
    b = np.array([10, 10])  # 约束条件常数

    # 求解最优解
    from scipy.optimize import linprog
    result = linprog(c, A_ub=A, b_ub=b, bounds=(0, None), method='highs')

    return result.x

# 测试代码
if __name__ == "__main__":
    x, y = linear_programming(1, 1)
    print("最优解：x =", x, "y =", y)
    plt.plot([0, 10], [10 - x, 0], label="约束条件1")
    plt.plot([0, y], [x, 0], label="约束条件2")
    plt.xlabel("x轴（产品1销量）")
    plt.ylabel("y轴（产品2销量）")
    plt.legend()
    plt.show()
```
