contract AttentionCoin {
    mapping(address => uint256) public balances;
    uint256 public totalSupply;
    uint256 public totalAttention;

    event Transfer(address sender, address receiver, uint256 amount);

    constructor() {
        totalSupply = 1000000;
        balances[msg.sender] = totalSupply;
    }

    function generateAttention(uint256 attention) public {
        require(msg.sender == tx.origin, "No contract calls allowed");
        totalAttention += attention;

        emit Transfer(address(0), msg.sender, attention);

    function transfer(address receiver, uint256 amount) public {
        require(balances[msg.sender] >= amount, "Insufficient balance");
        require(receiver != address(0), "Invalid receiver");
        balances[msg.sender] -= amount;
        balances[receiver] += amount;
        emit Transfer(msg.sender, receiver, amount);
    }

    function getAttentionValue(uint256 attention) public view returns (uint256) {
        return totalSupply * attention / totalAttention;
    }
}
