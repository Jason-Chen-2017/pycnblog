def gradient_descent(x, learning_rate, max_iter):
    for i in range(max_iter):
        grad = -1 / np.sqrt(x)
        x = x - learning_rate * grad
        if i % 100 == 0:
            print(f"Iteration {i}: x = {x}, f(x) = {np.sum(-1 / np.sqrt(x))}")
    return x

print(f"Optimal x: {optimal_x}, f(x): {np.sum(-1 / np.sqrt(optimal_x))}")
