import numpy as np
import matplotlib.pyplot as plt

# 初始化纳米机器人参数
m = 1.0  # 纳米机器人质量
F = 1.0  # 受力大小
t = 0.1  # 时间步长

# 初始化位置和速度
x = 0.0
v = 0.0

# 模拟纳米机器人的运动
for _ in range(100):
    a = F / m
    v += a * t
    x += v * t
    
    # 更新位置和速度
    x_new = x + v * t
    v_new = v + a * t

# 绘制纳米机器人的运动轨迹

plt.xlabel('Position')
plt.ylabel('Velocity')
plt.title('Nanorobot Motion')
plt.show()
