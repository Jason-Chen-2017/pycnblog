from bs4 import BeautifulSoup
import pandas as pd

def get_blog_data(url):

    ip_matrix = blog_data['Score'].sum()

    ip_matrix = calculate_ip_matrix(data)

    blog_data = pd.DataFrame(data, columns=['Title', 'Content'])
    blog_data['Score'] = blog_data.apply(lambda x: len(x['Content'].split()), axis=1)
    ip_matrix = blog_data['Score'].sum()
    return ip_matrix

if __name__ == '__main__':
    urls = [
        'https://blog.csdn.net/your_username/article/details/12345678',
        'https://www.cnblogs.com/your_username/p/12345678.html',
        # 更多博客链接
    ]
    data = [get_blog_data(url) for url in urls]
    ip_matrix = calculate_ip_matrix(data)
    print(f'Personal IP Matrix Score: {ip_matrix}')
```
