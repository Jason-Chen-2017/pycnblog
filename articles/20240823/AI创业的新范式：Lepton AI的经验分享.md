$$
z^{[l]} = W^{[l]} \cdot a^{[l-1]} + b^{[l]}

$$
a^{[l]} = \sigma(z^{[l]})
$$

$$
\delta^{[l]} = \frac{\partial J}{\partial z^{[l]}}
$$

$$
\delta^{[l-1]} = (\sigma^{'}(z^{[l-1]})) \cdot (W^{[l]} \cdot \delta^{[l]})
