```python
import datetime

class Goal:
    def __init__(self, name, target, start_date, end_date):
        self.name = name
        self.target = target
        self.start_date = start_date
        self.end_date = end_date
        self.current_progress = 0

    def set_progress(self, progress):
        self.current_progress = progress

    def get_progress(self):
        return self.current_progress

    def calculate_progress_percentage(self):
        return (self.current_progress / self.target) * 100

def set_goals(goals):
    for goal in goals:
        print(f"设定目标：{goal.name}，目标值：{goal.target}，起始日期：{goal.start_date}，结束日期：{goal.end_date}")

def monitor_goals(goals):
    for goal in goals:
        progress_percentage = goal.calculate_progress_percentage()
        print(f"目标：{goal.name}，当前进度：{progress_percentage}％")

def main():
    goals = [
        Goal("完成项目A", 100, datetime.date(2023, 1, 1), datetime.date(2023, 6, 30)),
        Goal("提高品牌知名度", 200, datetime.date(2023, 1, 1), datetime.date(2023, 6, 30))
    ]

    set_goals(goals)

    # 模拟项目进展
    goals[0].set_progress(50)
    goals[1].set_progress(75)

    monitor_goals(goals)

if __name__ == "__main__":
    main()
```

函数`set_goals`用于设置目标，函数`monitor_goals`用于监控目标进度。

在主函数`main`中，我们创建了一个包含两个目标的列表`goals`，并分别设置了这两个目标的起始日期和结束日期。然后，我们通过模拟项目进展，调用`set_progress`方法设置目标的当前进度，并调用`monitor_goals`函数显示目标的当前进度。
