```javascript
// 智能合约 - 注意力时间银行
pragma solidity ^0.8.0;

contract TimeBank {
    mapping(address => uint256) public balances;
    mapping(address => uint256) public skills;

    function contributeTime(uint256 hours) public {
        balances[msg.sender] += hours;
    }

    function contributeSkill(uint256 hours) public {
        skills[msg.sender] += hours;
    }

    function calculateBalance() public view returns (uint256) {
        return balances[msg.sender] + skills[msg.sender];
    }

    function transfer(address to, uint256 amount) public {
        require(balances[msg.sender] >= amount, "Insufficient balance");
        balances[msg.sender] -= amount;
        balances[to] += amount;
    }
}
```
