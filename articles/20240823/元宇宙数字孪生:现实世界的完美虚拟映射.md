    b = 0.2

    w = 0.1
    b = 0.2
    c = 0.3
    return 1 / (1 + np.exp(w * x + b * y + c * z))

# 定义数据模型
def data_model(x, y, z):
    w = 0.1
    b = 0.2
    c = 0.3

x = np.random.rand(100)
