### 5.2 源代码详细实现

以下是实现注意力分散控制算法的 Python 代码：

```python
import tensorflow as tf
import numpy as np

# 定义注意力控制模型
class AttentionControlModel(tf.keras.Model):
    def __init__(self, learning_rate=0.001):
        super(AttentionControlModel, self).__init__()
        self.learning_rate = learning_rate
        self.model = tf.keras.Sequential([
            tf.keras.layers.Dense(64, activation='relu', input_shape=(2,)),
            tf.keras.layers.Dense(1)
        ])

    @tf.function
    def train_step(self, data):
        x, y = data
        with tf.GradientTape(persistent=True) as tape:
            y_pred = self.model(x)
            loss = tf.reduce_mean(tf.square(y - y_pred))
        grads = tape.gradient(loss, self.model.trainable_variables)
        self.model.optimizer.apply_gradients(zip(grads, self.model.trainable_variables))
        return loss

    def train(self, x, y, epochs=10):
        for epoch in range(epochs):
            loss = self.train_step((x, y))
            print(f"Epoch {epoch+1}, Loss: {loss.numpy()}")

# 数据预处理
def preprocess_data(data):
    x = np.array([d[0] for d in data])
    y = np.array([d[1] for d in data])
    return x, y

# 训练模型
model = AttentionControlModel(learning_rate=0.001)
x, y = preprocess_data(data)
model.train((x, y), epochs=10)

# 测试模型
test_data = [[10, 50], [20, 40], [30, 30], [40, 20], [50, 10]]
for d in test_data:
    print(f"Input: {d}, Output: {model.model(np.array([d[0], d[1]])).numpy()[0]}")
```
