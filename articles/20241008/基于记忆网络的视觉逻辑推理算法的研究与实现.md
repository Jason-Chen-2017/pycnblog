# 写入特征向量到记忆网络
def write_to_memory(feature_vector):
    global write_weight
    memory_state = np.dot(write_weight, feature_vector)
    memory_state = np.tanh(memory_state)
    return memory_state

def retrieve_from_memory():
    global read_weight
    return np.dot(read_weight, memory_state)
