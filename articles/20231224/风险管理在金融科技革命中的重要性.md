2.OAuth2.0授权码流程的Python实现：

```python
from flask import Flask, request, redirect
from flask_oauthlib.client import OAuth

app = Flask(__name__)
oauth = OAuth(app)

google = oauth.remote_app(
    'google',
    consumer_key='your-consumer-key',
    consumer_secret='your-consumer-secret',
    request_token_params={
        'scope': 'https://www.googleapis.com/auth/userinfo.email'
    },
    base_url='https://www.googleapis.com/oauth2/v1/',
    request_token_url=None,
    access_token_method='POST',
    access_token_url='https://accounts.google.com/o/oauth2/token',
    authorize_url='https://accounts.google.com/o/oauth2/auth',
)

@app.route('/login')
def login():
    return google.authorize(callback=url_for('authorized', _external=True))

@app.route('/authorized')
def authorized():
    resp = google.authorized_response()
    if resp is None or resp.get('access_token') is None:
        # 授权失败
        return 'Access denied: reason={} error={}'.format(
            request.args['error_reason'],
            request.args['error_description']
        )

    # 授权成功，获取用户信息
    get_user_info_url = 'https://www.googleapis.com/oauth2/v1/userinfo?access_token={}'
    res = google.get(get_user_info_url.format(resp['access_token']))
    user_info = res.data

    return 'User info: {}'.format(user_info)
```

3.负责任借贷算法的Python实现：

```python
import numpy as np

def calculate_credit_score(age, income, debt):
    credit_score = 0.5 * age + 0.3 * income + 0.2 * debt
    return credit_score

def is_responsible_loan(credit_score, threshold=600):
    return credit_score >= threshold

# 示例
age = 30
income = 50000
debt = 10000
credit_score = calculate_credit_score(age, income, debt)
is_responsible_loan = is_responsible_loan(credit_score)

print('Credit score:', credit_score)
print('Is responsible loan:', is_responsible_loan)
```
