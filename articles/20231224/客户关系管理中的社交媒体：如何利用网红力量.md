```python
import requests
from bs4 import BeautifulSoup

def collect_red_info(url):
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    red_info = []
    for red in soup.find_all('div', class_='red-info'):
        name = red.find('span', class_='name').text
        fans_num = int(red.find('span', class_='fans-num').text.replace('万', ''))
        trust_degree = float(red.find('span', class_='trust-degree').text.replace('%', ''))
        creativity_degree = float(red.find('span', class_='creativity-degree').text.replace('%', ''))
        red_info.append({
            'name': name,
            'fans_num': fans_num,
            'trust_degree': trust_degree,
            'creativity_degree': creativity_degree
        })
    return red_info
```

```python
def filter_red(red_info, weight1=1, weight2=1, weight3=1):
    filtered_red = []
    for red in red_info:
        score = weight1 * red['fans_num'] + weight2 * red['trust_degree'] + weight3 * red['creativity_degree']
```

# 6.附录常见问题与解答

## 6.1 如何选择合适的网红？

选择合适的网红需要考虑以下几个方面：

1. 网红的粉丝数和信任度：选择粉丝数较多且信任度较高的网红，可以帮助企业扩大曝光范围和提高品牌知名度。
2. 网红的创意度和专业背景：选择具有较高创意度和专业背景的网红，可以帮助企业创作出更有吸引力的内容。
3. 网红与企业的相似度：选择与企业业务相关的网红，可以帮助企业更好地推广产品和服务。

## 6.2 如何评估推广效果？

推广效果可以通过以下几个指标进行评估：

1. 曝光度：包括点赞数、转发数等指标，表示内容在社交媒体上的曝光程度。
2. 转化率：表示曝光后的用户转化为购买客户的比例。
3. 销售额：表示推广活动后的实际销售额。

## 6.3 如何保护数据隐私和安全？

1. 遵守相关法律法规和规定：企业需要严格遵守相关的法律法规和规定，确保在社交媒体上的活动符合法律要求。
2. 加密处理用户数据：企业需要对用户数据进行加密处理，以保护用户数据的安全。
3. 限制数据泄露风险：企业需要对数据泄露风险进行评估，并采取相应的防范措施，如数据加密、数据隔离等。
