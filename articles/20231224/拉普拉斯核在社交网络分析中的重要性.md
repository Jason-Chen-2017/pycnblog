from scipy.sparse import csr_matrix
from sklearn.metrics.pairwise import rbf_kernel

# 构建拉普拉斯矩阵
def build_laplacian_matrix(graph):
    adjacency_matrix = np.zeros((len(graph.nodes()), len(graph.nodes())))
    degree_matrix = np.zeros((len(graph.nodes()), len(graph.nodes())))

    for node, neighbors in graph.nodes(data=True):
        adjacency_matrix[node, neighbors] = 1
        degree_matrix[node, node] = len(neighbors)

    laplacian_matrix = degree_matrix - np.diag(np.sum(adjacency_matrix, axis=1))
    laplacian_matrix = csr_matrix(laplacian_matrix)

    return laplacian_matrix
