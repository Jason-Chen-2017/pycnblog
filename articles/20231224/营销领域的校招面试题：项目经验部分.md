```python
import pandas as pd

# 读取数据
data = pd.read_csv('online_marketing_data.csv')

# 计算ROI
roi = (data['revenue'] - data['cost']) / data['cost']

cpa = data['cost'] / data['conversions']

# 计算CPC
cpc = data['cost'] / data['clicks']

# 输出结果
print(roi)
print(cpa)
print(cpc)
```
