## 4.1 电子门匠解锁的具体代码实例

以下是一个简单的电子门匠解锁的具体代码实例：

```python
import hashlib

class ElectronicLock:
    def __init__(self):
        self.password = ""
        self.lock_status = False

    def set_password(self, password):
        self.password = hashlib.sha256(password.encode()).hexdigest()

    def verify_password(self, input_password):
        return self.password == hashlib.sha256(input_password.encode()).hexdigest()

    def lock(self):
        self.lock_status = True

    def unlock(self):
        self.lock_status = False

lock = ElectronicLock()
lock.set_password("1234")
if lock.verify_password("1234"):
    lock.unlock()
else:
    print("验证失败")
```

## 4.2 智能安防系统的具体代码实例

以下是一个简单的智能安防系统的具体代码实例：

```python
import numpy as np
from sklearn.cluster import KMeans

class SmartSecuritySystem:
    def __init__(self):
        self.data = []
        self.clusters = []

    def add_data(self, data):
        self.data.append(data)

    def cluster(self, n_clusters=2):
        kmeans = KMeans(n_clusters=n_clusters)
        kmeans.fit(np.array(self.data))
        self.clusters = kmeans.labels_

    def detect_anomaly(self):
        for data, label in zip(self.data, self.clusters):
            if label == 0:
                print(f"发现异常数据：{data}")

security_system = SmartSecuritySystem()
security_system.add_data([1, 2, 3])
security_system.add_data([4, 5, 6])
security_system.cluster()
security_system.detect_anomaly()
```
