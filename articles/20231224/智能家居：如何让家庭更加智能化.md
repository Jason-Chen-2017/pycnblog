```python
# 智能门锁的智能化控制
class SmartLock:
    def __init__(self):
        self.is_locked = True

    def unlock(self):
        self.is_locked = False
        print("门锁已解锁")

    def lock(self):
        self.is_locked = True
        print("门锁已锁定")

    def remote_control(self, code):
        if code == "1234":
            if self.is_locked:
                self.unlock()
            else:
                print("门锁已解锁")

            print("错误的解锁密码")

    def timed_control(self, time):
        if time == "morning":
            self.lock()
        elif time == "evening":
            self.unlock()
        else:
            print("无效的时间")
```
