def gravitational_force(m1, m2, r):

def update_velocity(v, force, mass):

    v += a * dt
    return v

def update_position(r, v):
    r += v * dt
    return r

# 初始星系位置和速度

velocities = np.array([[0, 0], [0, 1]])

# 星系弱相互作用迭代
