## 4.1 ArangoDB 插件开发示例

### 4.1.1 数据导入/导出插件

我们将创建一个简单的数据导入/导出插件，它可以将 CSV 文件导入到 ArangoDB 中，并将 ArangoDB 数据导出到 CSV 文件。

```javascript
const fs = require('fs');
const path = require('path');
const csv = require('csv-parser');
const { ArangoClient } = require('@arangodb/arango-client');

const client = new ArangoClient({ url: 'http://localhost:8529' });
const db = client.db('test');
const collection = db.collection('users');

async function importCSV(filePath) {
  const inputStream = fs.createReadStream(filePath);
  const outputStream = fs.createWriteStream(path.join(__dirname, 'output.csv'));

  inputStream.pipe(csv())
    .on('data', async (row) => {
      await collection.insert(row);
    })
    .on('end', () => {
      console.log('CSV file successfully imported');
    });
}

async function exportCSV(collectionName) {
  const collection = db.collection(collectionName);
  const data = await collection.toArray();

  const outputStream = fs.createWriteStream(path.join(__dirname, 'output.csv'));
  outputStream.write('id,name,age\n');

  data.forEach((doc) => {
    outputStream.write(`${doc._id},${doc.name},${doc.age}\n`);
  });

  outputStream.end();
  console.log('CSV file successfully exported');
}

// Usage
importCSV('input.csv');
exportCSV('users');
```
