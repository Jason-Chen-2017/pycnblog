```python
import cv2
import numpy as np

def histeq(image):
    # 计算图像的直方图
    hist, bins = np.histogram(image.ravel(), 256, [0, 256])
    # 计算直方图的累积和
    cdf = hist.cumsum()
    # 计算累积直方图的归一化
    cdf_normalized = (cdf * hist.max()) / cdf.max()
    # 计算累积直方图的映射
    for i in range(256):
        cdf_normalized[i] = (cdf_normalized[i] + cdf_normalized[i - 1])
    # 映射原图像
    m = cv2.normalize(image, None, alpha=cdf_normalized[0], beta=cdf_normalized[-1], norm_type=cv2.NORM_MINMAX, dtype=cv2.CV_32F)
    # 将映射后的图像转换为8位整数类型
    m = np.uint8(m * 255)
    return m

output_image = histeq(image)
cv2.imshow('output_image', output_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

```python
import cv2
import numpy as np

def image_stitching(image1, image2):
    # 获取两个图像的尺寸
    rows1, cols1 = image1.shape
    rows2, cols2 = image2.shape
    # 计算两个图像的重叠区域
    overlap_rows = max(0, min(rows1, rows2) - max(0, rows1 - rows2))
    overlap_cols = max(0, min(cols1, cols2) - max(0, cols1 - cols2))
    # 计算两个图像的拼接区域
    stitched_rows = max(rows1, rows2)
    stitched_cols = max(cols1, cols2)
    # 创建一个大图像用于拼接
    stitched_image = np.zeros((stitched_rows, stitched_cols, 3), dtype=np.uint8)
    # 复制第一个图像到大图像
    stitched_image[:rows1, :cols1] = image1
    # 复制第二个图像到大图像
    stitched_image[:rows2, overlap_cols:cols2 + overlap_cols] = image2
    return stitched_image

output_image = image_stitching(image1, image2)
cv2.imshow('output_image', output_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

```python
import cv2
import numpy as np

def template_matching(image, template):
    # 计算模板的尺寸
    rows, cols = template.shape
    # 创建一个用于存储匹配结果的图像
    match_image = np.zeros(image.shape, dtype=np.uint8)
    # 遍历图像中的每个位置
    for i in range(image.shape[0] - rows + 1):
        for j in range(image.shape[1] - cols + 1):
            # 获取当前位置的子图像
            sub_image = image[i:i + rows, j:j + cols]
            # 计算子图像与模板的相似度
            similarity = cv2.matchTemplate(sub_image, template, cv2.TM_CCOEFF_NORMED)
            # 获取相似度最大的位置
            max_val, max_pos = np.max(similarity)
```

```python
import cv2
import numpy as np

def image_registration(image1, image2, method='similarity'):
    # 获取两个图像的尺寸
    rows1, cols1 = image1.shape
    rows2, cols2 = image2.shape
    # 创建一个用于存储注册结果的图像
    registered_image = np.zeros((max(rows1, rows2), max(cols1, cols2), 3), dtype=np.uint8)
    # 根据不同的注册方法进行注册
    if method == 'similarity':
        H = cv2.findTransformECC(image1, image2)
        if len(H) == 2:
            H = H[1]
        registered_image = cv2.warpAffine(image1, H, (max(cols1, cols2), max(rows1, rows2)))
    elif method == 'rigid':
        H = cv2.findHomography(image1, image2)
        registered_image = cv2.warpPerspective(image1, H, (max(cols1, cols2), max(rows1, rows2)))
    return registered_image

output_image = image_registration(image1, image2, method='similarity')
cv2.imshow('output_image', output_image)
```
