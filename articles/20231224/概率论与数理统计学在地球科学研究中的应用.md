```python
# 一元统计检验
def one_sample_t_test(sample, value):
    t_statistic = (value - np.mean(sample)) / (np.std(sample) / np.sqrt(len(sample)))
    p_value = 2 * (1 - stats.t.cdf(abs(t_statistic), df=len(sample) - 1))
    return t_statistic, p_value

# 多元统计检验
def two_sample_t_test(sample1, sample2):
    t_statistic = (np.mean(sample1) - np.mean(sample2)) / np.sqrt((np.var(sample1) / len(sample1)) + (np.var(sample2) / len(sample2)))
    p_value = 2 * (1 - stats.t.cdf(abs(t_statistic), df=len(sample1) + len(sample2) - 2))
    return t_statistic, p_value
```

### 4.2.4 模型选择

# AIC信息Criterion
def AIC(sample, k):
    return 2 * k - len(sample) + np.sum((np.array(sample) - np.mean(sample)) ** 2) / len(sample)

# BIC信息Criterion
def BIC(sample, k):
    return np.log(len(sample)) * k + np.sum((np.array(sample) - np.mean(sample)) ** 2) / len(sample)
