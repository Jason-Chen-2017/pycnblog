import numpy as np

def linear_regression(X, y, learning_rate, iterations):
    m, n = X.shape
    weights = np.zeros(n)
    for _ in range(iterations):
        predictions = X.dot(weights)
        errors = predictions - y
        gradient = X.T.dot(errors)
        weights -= learning_rate * gradient
    return weights
```
