from scipy.optimize import minimize

# 定义团队成员的评分函数
def team_score(members):
    score = 0
    for i in range(len(members)):
        for j in range(i+1, len(members)):
            score += np.corrcoef(members[i], members[j])[0, 1]
    return -score

# 定义团队成员的选择函数
def select_team(n, members):
    constraints = ({'type': 'eq', 'fun': lambda x: n - len(x)}, None)
    result = minimize(team_score, members, method='SLSQP', bounds=members_bounds, constraints=constraints)
    return result.x

# 定义团队成员的范围
members_bounds = [(0, 100) for _ in range(5)]

# 获取团队成员的评分
team_members = np.random.rand(5, 10)
team_scores = [select_team(n, members) for members in team_members]
