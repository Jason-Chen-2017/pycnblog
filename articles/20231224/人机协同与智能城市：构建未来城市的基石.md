&\xi _{i}\geq 0,i=1,2, \ldots, n
\end{aligned}
$$

def kernel_matrix(X, kernel):
    m, n = X.shape
    K = np.zeros((m, m))
    if kernel == 'linear':
        for i in range(m):
            for j in range(m):
                K[i, j] = np.dot(X[i, :], X[j, :])
    elif kernel == 'rbf':
