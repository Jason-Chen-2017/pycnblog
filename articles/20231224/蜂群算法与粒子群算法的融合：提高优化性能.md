## 4.1 蜂群算法代码实例
```python
import numpy as np

def fitness_function(x):
    # 目标函数，例如最小化x的平方
    return x**2

def update_position(x, best_x):
    c1 = 0.5
    c2 = 0.5
    rand1 = np.random.rand()
    rand2 = np.random.rand()
    return x + c1 * rand1 * (best_x - x) + c2 * rand2 * (best_x - x)

def bees_algorithm(n, n_iterations):
    bees = np.random.rand(n)
    best_x = min(bees, key=fitness_function)
    for _ in range(n_iterations):
        for i in range(n):
            bees[i] = update_position(bees[i], best_x)
        new_best_x = min(bees, key=fitness_function)
        if fitness_function(new_best_x) < fitness_function(best_x):
            best_x = new_best_x
    return best_x

n = 100
n_iterations = 1000
best_x = bees_algorithm(n, n_iterations)
print("最佳解:", best_x)
```

    particles = np.random.rand(n)
    best_x = min(particles, key=fitness_function)
    best_global_x = best_x
    for _ in range(n_iterations):
        for i in range(n):
            particles[i] = update_position(particles[i], best_x)
        if fitness_function(particles[i]) < fitness_function(best_x):

            best_global_x = best_x
