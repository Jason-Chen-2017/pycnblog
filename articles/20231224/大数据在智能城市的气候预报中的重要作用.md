## 4.2 气候模型的构建和优化
### 4.2.1 气候模型的构建
```python
import numpy as np
import matplotlib.pyplot as plt

def build_climate_model(x, y):
    x_train = x[:int(len(x)*0.8)]
    y_train = y[:int(len(y)*0.8)]
    x_test = x[int(len(x)*0.8):]
    y_test = y[int(len(y)*0.8):]

    model = np.polyfit(x_train, y_train, 1)
    y_pred = np.polyval(model, x_test)

    plt.scatter(x_test, y_test, color='blue')
    plt.plot(x_test, y_pred, color='red')
    plt.show()

x = np.array([1, 2, 3, 4, 5])
y = np.array([2, 4, 6, 8, 10])
build_climate_model(x, y)
```
### 4.2.2 气候模型的优化
```python
from scipy.optimize import minimize

def objective_function(model):
    x_train = x[:int(len(x)*0.8)]
    y_train = y[:int(len(y)*0.8)]
    y_pred = np.polyval(model, x_train)
    return np.sum((y_pred - y_train)**2)

x = np.array([1, 2, 3, 4, 5])
y = np.array([2, 4, 6, 8, 10])
initial_model = [0, 1]
result = minimize(objective_function, initial_model)
result.x
```
