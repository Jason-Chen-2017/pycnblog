## 4.2 社交网络算法的具体代码实例

```python
import networkx as nx

# 构建社交网络
G = nx.Graph()
G.add_edges_from([(1, 2), (1, 3), (2, 3), (2, 4), (3, 4), (3, 5)])

# 内容传播模型
def viral_propagation(G, seed_nodes, num_iterations):
    viral_score = {node: 0 for node in G.nodes()}
    viral_score[seed_nodes[0]] = 1

    for _ in range(num_iterations):
        for node in G.nodes():
            if node in seed_nodes:
                continue
            viral_score[node] = sum(viral_score[neighbor] for neighbor in G.neighbors(node))

    return viral_score

# 内容推荐
recommend_nodes = viral_propagation(G, [1, 2], 3)
recommend_scores = sorted(recommend_nodes.items(), key=lambda x: x[1], reverse=True)

print(recommend_scores)
```

## 4.3 搜索引擎算法的具体代码实例

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 影视内容数据
movie_data = ['电影A是一个科幻电影', '电影B是一个动作电影', '电影C是一个悬疑电影']

# 内容索引
vectorizer = TfidfVectorizer()
tfidf_matrix = vectorizer.fit_transform(movie_data)

# 搜索引擎构建
def search_engine(query, tfidf_matrix):
    query_vector = vectorizer.transform([query])
    similarity_scores = cosine_similarity(query_vector, tfidf_matrix)
    return np.argsort(-similarity_scores.flatten())

# 搜索结果排序
query = '科幻电影'
search_results = search_engine(query, tfidf_matrix)

print(search_results)
```
