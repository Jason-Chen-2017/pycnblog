## 4.1 数据清洗和预处理

```python
import pandas as pd
import numpy as np

# 加载数据
data = pd.read_csv('data.csv')

# 数据检查
print(data.info())

# 数据清洗
data['column'] = data['column'].fillna(0)

# 数据转换
data['column'] = data['column'].astype(float)

# 数据归一化
data['column'] = (data['column'] - np.mean(data['column'])) / np.std(data['column'])

# 数据分割
train_data = data[:int(len(data)*0.8)]
test_data = data[int(len(data)*0.8):]
valid_data = data[int(len(data)*0.6):int(len(data)*0.8)]
```

## 4.2 特征选择和提取

```python
from sklearn.feature_selection import SelectKBest
from sklearn.feature_selection import chi2

# 特征筛选
selector = SelectKBest(chi2, k=10)
selector.fit(train_data.drop('target', axis=1), train_data['target'])
selected_features = selector.transform(train_data.drop('target', axis=1))

# 特征提取
from sklearn.decomposition import PCA

pca = PCA(n_components=2)
pca.fit(train_data.drop('target', axis=1))
reduced_data = pca.transform(train_data.drop('target', axis=1))
```

## 4.3 模型评估和优化

```python
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score

# 模型评估
X_train, X_valid, y_train, y_valid = train_test_split(selected_features, train_data['target'], test_size=0.2, random_state=42)
model = RandomForestClassifier()
model.fit(X_train, y_train)
valid_pred = model.predict(X_valid)

# 模型优化
from sklearn.model_selection import GridSearchCV

param_grid = {'n_estimators': [100, 200, 300], 'max_depth': [5, 10, 15]}
grid_search = GridSearchCV(model, param_grid, cv=5)
grid_search.fit(X_train, y_train)
best_model = grid_search.best_estimator_

# 模型评估
test_pred = best_model.predict(test_data.drop('target', axis=1))
print('Accuracy:', accuracy_score(test_data['target'], test_pred))
print('Precision:', precision_score(test_data['target'], test_pred, average='weighted'))
print('Recall:', recall_score(test_data['target'], test_pred, average='weighted'))
print('F1 Score:', f1_score(test_data['target'], test_pred, average='weighted'))
```
