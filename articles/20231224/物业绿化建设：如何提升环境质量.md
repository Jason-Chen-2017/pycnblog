import matplotlib.pyplot as plt

# 绿色建筑
def green_building(energy_consumption, water_consumption, electricity_consumption, material_consumption, waste_production):
    energy_saving = energy_consumption * 0.8
    water_saving = water_consumption * 0.9
    electricity_saving = electricity_consumption * 0.7
    material_saving = material_consumption * 0.6
    waste_reduction = waste_production * 0.5
    return energy_saving, water_saving, electricity_saving, material_saving, waste_reduction

    return energy_saving, water_saving, electricity_saving, material_saving, waste_reduction

    material_saving = material_consumption * 0.4
    waste_reduction = waste_production * 0.3

Q1：绿色建筑和绿色设计有什么区别？

A1：绿色建筑是指在建筑过程中采用环保材料、节能设计、绿化装饰等方式，以减少对环境的影响。绿色设计是指在室内设计过程中采用环保材料、节能设计、绿化装饰等方式，以提高室内环境质量。绿色建筑和绿色设计的区别在于，绿色建筑关注的是建筑的整体环保设计，而绿色设计关注的是室内环境的环保设计。

Q2：绿色管理和绿色设计有什么区别？
