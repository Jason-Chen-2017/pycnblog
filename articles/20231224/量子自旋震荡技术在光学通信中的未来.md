```python
import numpy as np

def cnot(u, v):
    return np.kron(u, v)

def hadamard(u):
    return (np.kron(u, np.eye(2)) + np.kron(np.eye(2), u)) / 2

u = np.array([[1, 0], [0, 0]])
v = np.array([[0, 0], [0, 1]])

u_hadamard = hadamard(u)
v_hadamard = hadamard(v)

result = cnot(u_hadamard, v_hadamard)

print(result)
```
