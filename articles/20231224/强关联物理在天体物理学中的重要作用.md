def solve_kepler(m, M, r0, t0, dt):
    r = np.array([r0])
    v = np.array([0.0])
    t = np.array([t0])

    while r[-1] > 0:
        a = acceleration(m, M, r[-1])
        r_next = r[-1] + v[-1] * dt + 0.5 * a * dt**2
        v_next = v[-1] + a * dt / 2.0
        r = np.append(r, r_next)
        v = np.append(v, v_next)

    return r, t
