1. 支持向量机（SVM）算法实现：

```python
from sklearn import datasets
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.svm import SVC

# 加载数据
iris = datasets.load_iris()
X = iris.data
y = iris.target

# 数据预处理
sc = StandardScaler()
X = sc.fit_transform(X)

# 数据分割
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# 模型训练
clf = SVC(kernel='linear', C=1).fit(X_train, y_train)

# 模型评估
accuracy = clf.score(X_test, y_test)
print('Accuracy: %.2f' % (accuracy * 100.0))
```

3. 基于GPS的定位算法实现：

```python
import gps
import time

# GPS定位
def gps_update(gps_data):
    if 'fix' in gps_data and gps_data['fix'] == True:
        print('GPS定位成功')
        print('纬度: {:.6f}'.format(gps_data['latitude']))
        print('经度: {:.6f}'.format(gps_data['longitude']))

# 初始化GPS
gps_device = gps.gps()
gps_device.start()

# 定位循环
while True:
    time.sleep(1)
    gps_data = gps_device.next()
    gps_update(gps_data)
```
