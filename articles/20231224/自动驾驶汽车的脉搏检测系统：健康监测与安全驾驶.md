```python
import numpy as np

def process_light_signal(signal):
    # 计算光信号变化的周期数
    T = np.sum(np.diff(signal))
    # 计算脉搏率
    heart_rate = 60 / T
    return heart_rate
```

import socket

def send_heart_rate_data(driver_heart_rate, host, port):
