import scipy.optimize as opt

c = np.array([1, 2, 3])

              [1, 0, 1],
              [0, 1, 1]])
b = np.array([3, 3, 3])

x, success = opt.linprog(c, A_ub=A, b_ub=b, bounds=[(0, None), (0, None), (0, None)], method='highs')
