```python
import aerospike

# 连接到数据库和节点
client = aerospike.client()
client.connect((127, 0, 0, 10001))

# 创建命名空间、集群和设备
ns = client.namespace("test")
set = ns["test"]

# 创建键
key = ("test", "test")

# 写入数据
data = {"name": "John", "age": 30}
client.put(set, key, data)

# 读取数据
ret = client.get(set, key)
print(ret)

# 关闭连接
client.close()
```
