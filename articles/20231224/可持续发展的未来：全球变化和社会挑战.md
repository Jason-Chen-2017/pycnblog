## 4.1 系统动态模型代码实例
我们可以使用Python编程语言来编写系统动态模型的代码实例。例如，我们可以使用NumPy库来实现一个简单的差分方程模型。

```python
import numpy as np

def diff_eq(t, y):
    y0, y1 = y
    dy0dt = y0 * 0.1
    dy1dt = y1 * 0.2 + y0 * 0.3
    return [dy0dt, dy1dt]

t = np.linspace(0, 10, 100)
y0 = 1
y1 = 0
y = np.linalg.solve(np.array([[0.1, 0.3], [0.2, 0]]), [y0, y1])
sol = np.vectorize(diff_eq)
sol = np.fromfunction(sol, (len(t), 2))

import matplotlib.pyplot as plt
plt.plot(t, sol[:, 0], label='y0')
plt.plot(t, sol[:, 1], label='y1')
plt.xlabel('时间')
plt.ylabel('变量值')
plt.legend()
plt.show()
```

def cost_function(x):
    return (x - 2) ** 2 + 3 * (x - 4) ** 2

x = np.linspace(-10, 10, 100)
y = np.vectorize(cost_function)
y = np.fromfunction(y, (len(x), 1))

import matplotlib.pyplot as plt
plt.plot(x, y[:, 0], label='代价函数')
plt.xlabel('变量')
plt.ylabel('代价值')
plt.legend()
plt.show()
```

## 4.3 优化模型代码实例

from scipy.optimize import linprog

# 目标函数
c = [-1, -2]

# 约束条件
A = [[1, 1], [1, 0]]
b = [2, 1]

# 变量
x0 = linprog(c, A_ub=A, b_ub=b, bounds=(0, None))

```
