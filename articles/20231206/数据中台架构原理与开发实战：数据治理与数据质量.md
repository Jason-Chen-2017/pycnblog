# 2.核心概念与联系

# 4.具体代码实例和详细解释说明

## 4.1数据清洗代码实例

以下是一个数据清洗的代码实例，主要包括以下几个步骤：

- 数据校验：对数据进行格式和规则检查。
- 数据填充：对缺失的数据进行填充。
- 数据转换：对数据进行类型和格式转换。
- 数据过滤：对不符合预期的数据进行过滤。

```python
import pandas as pd

# 数据校验
def check_data(data):
    # 格式校验
    for col in data.columns:
        if not data[col].apply(lambda x: isinstance(x, str)).all():
            print(f"{col} 列中存在非字符串数据")
        if not data[col].apply(lambda x: isinstance(x, int)).all():
            print(f"{col} 列中存在非整数数据")
        if not data[col].apply(lambda x: isinstance(x, float)).all():
            print(f"{col} 列中存在非浮点数数据")
        if not data[col].apply(lambda x: isinstance(x, datetime.datetime)).all():
            print(f"{col} 列中存在非日期数据")

    # 规则校验
    for col in data.columns:
        if not data[col].apply(lambda x: x >= 0).all():
            print(f"{col} 列中存在负数数据")
        if not data[col].apply(lambda x: x <= 100).all():
            print(f"{col} 列中存在超过100的数据")

# 数据填充
def fill_data(data):
    for col in data.columns:
        if data[col].isnull().all():
            data[col] = data[col].mean()

# 数据转换
def transform_data(data):
    # 类型转换
    data['age'] = data['age'].astype(int)
    data['salary'] = data['salary'].astype(float)

    # 格式转换
    data['birthday'] = data['birthday'].apply(lambda x: x.strftime('%Y-%m-%d'))

# 数据过滤
def filter_data(data):
    data = data[data['age'] >= 18]
    data = data[data['salary'] >= 0]

# 数据清洗
def clean_data(data):
    check_data(data)
    fill_data(data)
    transform_data(data)
    filter_data(data)

# 数据清洗示例
data = pd.read_csv('data.csv')
clean_data(data)
```
