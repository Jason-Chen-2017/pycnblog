基于文本内容的推荐算法的核心步骤如下：

1. 文本预处理：对商品或内容的文本进行预处理，包括去除停用词、词干提取、词汇拆分等。

2. 词汇建模：将预处理后的词汇建模为向量，例如TF-IDF向量或Word2Vec向量。

3. 用户兴趣建模：根据用户的历史行为，为用户建模为向量，例如用户行为向量或用户兴趣向量。

4. 推荐计算：根据用户兴趣向量和商品或内容向量的相似度，计算商品或内容的推荐得分。

2. 用户相似度计算：根据用户的历史行为数据，计算用户之间的相似度，例如欧氏距离或皮尔逊相关系数。

3. 推荐计算：根据用户的兴趣向量和商品或内容向量的相似度，计算商品或内容的推荐得分。

```python
import numpy as np
from scipy.spatial.distance import cosine

# 用户行为数据
user_behavior_data = {
    'user1': ['item1', 'item2', 'item3'],
    'user2': ['item3', 'item4', 'item5'],
    'user3': ['item1', 'item5', 'item6']
}

# 商品或内容特征
item_features = {
    'item1': {'category': '电子产品', 'brand': '品牌1', 'price': 100},
    'item2': {'category': '电子产品', 'brand': '品牌2', 'price': 200},
    'item3': {'category': '服装', 'brand': '品牌3', 'price': 300},
    'item4': {'category': '服装', 'brand': '品牌4', 'price': 400},
    'item5': {'category': '服装', 'brand': '品牌5', 'price': 500},
    'item6': {'category': '服装', 'brand': '品牌6', 'price': 600}
}

# 用户兴趣建模
def user_interest_model(user_behavior_data):
    user_interest_vector = {}
    for user, items in user_behavior_data.items():
        user_interest_vector[user] = np.array([item_features[item]['category'] for item in items])
    return user_interest_vector

# 商品或内容特征建模
def item_feature_model(item_features):
    item_feature_vector = {}
    for item, feature in item_features.items():
        item_feature_vector[item] = np.array([feature['category'], feature['brand'], feature['price']])
    return item_feature_vector

# 计算用户相似度
def user_similarity(user_interest_vector, item_feature_vector, user_behavior_data):
    user_similarity_matrix = {}
    for user, interest_vector in user_interest_vector.items():
        user_similarity_matrix[user] = {}
        for other_user, other_interest_vector in user_interest_vector.items():
            if user != other_user:
                similarity = cosine(interest_vector, other_interest_vector)
                user_similarity_matrix[user][other_user] = similarity
    return user_similarity_matrix

# 推荐计算
def recommendation(user_interest_vector, item_feature_vector, user_similarity_matrix, user_behavior_data):
    recommendation_matrix = {}
    for user, interest_vector in user_interest_vector.items():
        recommendation_matrix[user] = {}
        for item, feature_vector in item_feature_vector.items():
            similarity = cosine(interest_vector, feature_vector)
            for other_user, other_similarity in user_similarity_matrix[user].items():
                if other_user in user_behavior_data and item in user_behavior_data[other_user]:
                    similarity -= other_similarity
            recommendation_matrix[user][item] = similarity
    return recommendation_matrix

# 推荐排序
def recommendation_sort(recommendation_matrix):
    sorted_recommendation_matrix = {}
    for user, recommendation in recommendation_matrix.items():
        sorted_recommendation_matrix[user] = sorted(recommendation.items(), key=lambda x: x[1], reverse=True)
    return sorted_recommendation_matrix

# 主函数
def main():
    user_interest_vector = user_interest_model(user_behavior_data)
    item_feature_vector = item_feature_model(item_features)
    user_similarity_matrix = user_similarity(user_interest_vector, item_feature_vector, user_behavior_data)
    recommendation_matrix = recommendation(user_interest_vector, item_feature_vector, user_similarity_matrix, user_behavior_data)
    sorted_recommendation_matrix = recommendation_sort(recommendation_matrix)
    print(sorted_recommendation_matrix)

if __name__ == '__main__':
    main()
```
