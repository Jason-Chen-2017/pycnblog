def random_point():
    x = random.uniform(-1, 1)
    y = random.uniform(-1, 1)
    return (x, y)

def is_in_circle(point):
    x, y = point
    return x*x + y*y <= 1

def monte_carlo_pi(n):
    inside_circle = 0
    for _ in range(n):
        point = random_point()
        if is_in_circle(point):
            inside_circle += 1
    pi = 4 * inside_circle / n
    return pi

n = 1000000
pi = monte_carlo_pi(n)
print("π 的估计值为：", pi)
