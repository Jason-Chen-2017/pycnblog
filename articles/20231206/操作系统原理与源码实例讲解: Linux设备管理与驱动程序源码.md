```c
#include <linux/kernel.h>
#include <linux/init.h>
#include <linux/device.h>

static struct device *my_device;

static int __init my_init(void)
{
    my_device = device_create(my_driver, NULL, 0, NULL, "my_device");
    if (IS_ERR(my_device)) {
        printk(KERN_ERR "Failed to create device\n");
        return PTR_ERR(my_device);
    }

    printk(KERN_INFO "My device is created\n");
    return 0;
}

static void __exit my_exit(void)
{
    device_destroy(my_driver, my_device->devt);
    printk(KERN_INFO "My device is destroyed\n");
}

module_init(my_init);
module_exit(my_exit);

MODULE_AUTHOR("Your Name");
MODULE_DESCRIPTION("My Device Driver");
MODULE_LICENSE("GPL");
```

在这个代码实例中，我们创建了一个简单的设备驱动程序，用于创建和销毁设备。当设备创建时，会输出“My device is created”的信息；当设备销毁时，会输出“My device is destroyed”的信息。
