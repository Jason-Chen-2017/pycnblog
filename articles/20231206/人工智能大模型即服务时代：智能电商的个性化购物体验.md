import numpy as np
from scipy.sparse.linalg import svds

def compute_similarity(R):
    n_users = R.shape[0]
    n_items = R.shape[1]
    similarity = np.zeros((n_users, n_users))

    for u in range(n_users):
        for v in range(u + 1, n_users):
            similarity[u, v] = similarity[v, u] = compute_cosine_similarity(R[u, :], R[v, :])

    return similarity

def compute_cosine_similarity(u, v):
    dot_product = np.dot(u, v)
    norm_u = np.linalg.norm(u)
    norm_v = np.linalg.norm(v)
    return dot_product / (norm_u * norm_v)

def collaborative_filtering(R, k, n_iterations=10):
    n_users = R.shape[0]
    n_items = R.shape[1]
    similarity = compute_similarity(R)

    for _ in range(n_iterations):
        for u in range(n_users):
            for i in range(n_items):
                if R[u, i] == 0:
                    R[u, i] = np.dot(R[u, :], similarity[u, :] * R[:, i]) / np.linalg.norm(R[u, :])

    return R

# 使用基于协同过滤的推荐算法进行推荐
R = np.array([[4, 0, 0, 0, 0],
              [0, 0, 5, 0, 0],
              [0, 0, 0, 3, 0],
              [0, 0, 0, 0, 4]])
k = 2
n_iterations = 10

R_hat = collaborative_filtering(R, k, n_iterations)

# 生成推荐列表
recommended_items = np.argwhere(R_hat > 0)
print(recommended_items)
