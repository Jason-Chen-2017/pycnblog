import numpy as np
from sklearn.preprocessing import LabelEncoder

# 数据清洗

data['gender'] = LabelEncoder().fit_transform(data['gender'])
data['income'] = data['income'].fillna(data['income'].mean())
data['age'] = data['age'].apply(lambda x: x if x >= 18 and x <= 65 else np.nan)

# 数据整合
data_integrated = pd.read_csv('data_integrated.csv')
data = pd.concat([data, data_integrated], axis=1)
data.drop(['id'], axis=1, inplace=True)

# 数据安全
key = b'your_key_here'
cipher_suite = Fernet(key)
data_encrypted = cipher_suite.encrypt(data.values)
data_decrypted = cipher_suite.decrypt(data_encrypted)

# 数据隐私保护
data['age'] = data['age'].apply(lambda x: '***' if x < 18 or x > 65 else x)
data['income'] = data['income'].apply(lambda x: '***' if x < 10000 or x > 100000 else x)

# 数据质量指标
y_true = data['label']
y_pred = data['pred']
accuracy = accuracy_score(y_true, y_pred)
recall = recall_score(y_true, y_pred, average='weighted')
