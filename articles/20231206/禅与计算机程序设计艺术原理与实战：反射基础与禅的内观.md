import types

# 创建反射对象
class MyClass:
    def my_method(self):
        return "Hello, World!"

# 访问反射对象
ref = types.MethodType(MyClass.my_method, None)
print(ref())  # 输出：Hello, World!

# 修改反射对象
def new_method(self):
    return "Hello, New World!"

ref.__set__('my_method', new_method)
print(ref())  # 输出：Hello, New World!

# 删除反射对象
del ref.__dict__['my_method']
print(ref())  # 输出：AttributeError: 'NoneType' object has no attribute 'my_method'
