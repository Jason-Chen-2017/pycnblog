```python
# 定义线性回归模型
class LinearRegression:
    def __init__(self):
```

```python
# 定义逻辑回归模型
class LogisticRegression:
    def __init__(self):
        self.w = None

    def fit(self, X, y):
        # 计算权重
        X = np.hstack((np.ones((X.shape[0], 1)), X))
        self.w = np.linalg.inv(X.T @ X) @ X.T @ y

    def predict(self, X):
        # 预测结果
        p = 1 / (1 + np.exp(-(X @ self.w)))
        return p > 0.5

# 生成数据
X = np.random.rand(100, 2)
y = np.where(3 * X[:, 0] + 2 * X[:, 1] > 0, 1, 0)

# 训练逻辑回归模型
model = LogisticRegression()
model.fit(X, y)

# 预测结果
y_pred = model.predict(X)
```

## 4.5 梯度下降

梯度下降是一种监督学习技术，它让计算机能够从标注的数据中学习。梯度下降的目标是让计算机能够理解自然语言、学习、推理、解决问题、理解环境、自主决策、学习和适应等。梯度下降的数学模型公式为：

$$
w_{t+1} = w_t - \eta \nabla J(w_t)
$$

梯度下降的具体代码实现如下：

```python
import numpy as np

# 定义线性回归模型
class LinearRegression:
    def __init__(self):
        self.w = None

    def fit(self, X, y):
        # 计算权重
        self.w = np.linalg.inv(X.T @ X) @ X.T @ y

    def predict(self, X):
        return X @ self.w

    def grad(self, X, y):
        return 2 * (X.T @ (X @ self.w - y))

# 生成数据
X = np.random.rand(100, 2)
y = 3 * X[:, 0] + 2 * X[:, 1] + np.random.rand(100, 1)

# 训练线性回归模型
model = LinearRegression()

# 梯度下降
learning_rate = 0.01
num_iterations = 1000

for _ in range(num_iterations):
    grad = model.grad(X, y)
    model.w -= learning_rate * grad

# 预测结果
y_pred = model.predict(X)
```
