    {
        "condition": "age < 18 or income < 30000",
        "action": "reject_loan"
    }
]

# 获取用户信息

    "age": 20,
    "income": 25000
}

# 执行规则
for rule in rules:
    if eval(rule["condition"]):
        rule["action"](user_info)
        break

# 处理结果
if "grant_loan" in user_info:
    print("用户贷款被批准")
else:
    print("用户贷款被拒绝")
