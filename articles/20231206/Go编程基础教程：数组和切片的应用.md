## 4.具体代码实例和详细解释说明

### 4.1 数组的具体代码实例

```go
package main

import "fmt"

func main() {
    // 数组的初始化
    var arr [3]int
    fmt.Println(arr) // [0 0 0]
    fmt.Println(len(arr)) // 3
    fmt.Println(cap(arr)) // 3

    // 数组的访问
    arr[0] = 1
    arr[1] = 2
    arr[2] = 3
    fmt.Println(arr) // [1 2 3]

    // 数组的遍历
    for i := 0; i < len(arr); i++ {
        fmt.Println(arr[i]) // 1 2 3
    }
}
```

    // 切片的访问

    // 切片的遍历

    }
