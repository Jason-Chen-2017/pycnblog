# 4.具体代码实例和详细解释说明

在这里，我们将提供一个具体的代码实例，以及对其的详细解释说明。

```go
package main

import (
	"fmt"
	"sync"
)

type Task struct {
	id       int
	priority int
	depends  []int
}

type Project struct {
	tasks  []Task
	wg     sync.WaitGroup
	mu     sync.Mutex
	status string
}

func (p *Project) AddTask(task Task) {
	p.mu.Lock()
	p.tasks = append(p.tasks, task)
	p.mu.Unlock()
}

func (p *Project) Start() {
	p.wg.Add(len(p.tasks))
	for _, task := range p.tasks {
		go func(task Task) {
			p.wg.Done()
			fmt.Printf("Task %d is done\n", task.id)
		}(task)
	}
	p.wg.Wait()
	fmt.Printf("Project %s is done\n", p.status)
}

func main() {
	project := &Project{
		status: "test",
	}

	task1 := Task{
		id:       1,
		priority: 1,
		depends:  []int{},
	}
	task2 := Task{
		id:       2,
		priority: 2,
		depends:  []int{task1.id},
	}
	task3 := Task{
		id:       3,
		priority: 3,
		depends:  []int{task2.id},
	}

	project.AddTask(task1)
	project.AddTask(task2)
	project.AddTask(task3)

	project.Start()
}
```

在这个代码实例中，我们创建了一个项目管理系统，它可以让我们添加任务、开始项目并等待任务完成。我们使用了Go语言中的`sync`包来实现互斥锁和读写锁，以及`fmt`包来实现输出。
