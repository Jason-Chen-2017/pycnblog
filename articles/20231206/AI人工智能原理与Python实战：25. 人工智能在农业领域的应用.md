# 4.1.农业生产的自动化设备
```python
import time
import RPi.GPIO as GPIO

# 设置GPIO口
GPIO.setmode(GPIO.BCM)
GPIO.setup(17, GPIO.OUT)

# 设置自动化设备
def auto_device():
    GPIO.output(17, GPIO.HIGH)
    time.sleep(1)
    GPIO.output(17, GPIO.LOW)
    time.sleep(1)

# 主程序
if __name__ == '__main__':
    auto_device()
```

# 4.2.农业生产的智能控制
```python
import time
import RPi.GPIO as GPIO

# 设置GPIO口
GPIO.setmode(GPIO.BCM)
GPIO.setup(17, GPIO.IN)

# 设置智能控制
def smart_control():
    if GPIO.input(17) == GPIO.HIGH:
        print("智能控制：开启")
    else:
        print("智能控制：关闭")

# 主程序
if __name__ == '__main__':
    smart_control()
```
