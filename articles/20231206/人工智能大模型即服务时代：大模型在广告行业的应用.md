```python
def generate_ad_creative(ad_creative_template, user_preferences):
    with tf.Session() as sess:
        model = tf.keras.Sequential([
            tf.keras.layers.Embedding(input_dim=len(user_preferences), output_dim=10),
            tf.keras.layers.LSTM(100),
            tf.keras.layers.Dense(1, activation='sigmoid')
        ])

        model.fit(user_preferences, ad_creative_template, epochs=10)
        generated_ad_creative = model.predict(user_preferences)
    return generated_ad_creative
```
