```python
    next = get_next(pattern)
    i = 0
    j = 0
    while i < len(text) and j < len(pattern):

            i += 1

def get_next(pattern):
    next = [0] * len(pattern)
    j = 0
    for i in range(1, len(pattern)):
        while j > 0 and pattern[j] != pattern[i]:
            j = next[j - 1]
        if pattern[j] == pattern[i]:
            j += 1
        next[i] = j
    return next
```
