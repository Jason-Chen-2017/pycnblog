```python
import tensorflow as tf

class RNN(tf.keras.Model):
    def __init__(self, vocab_size, embedding_dim, rnn_units, batch_size, seq_length, rnn_dropout):
        super(RNN, self).__init__()
        self.embedding = tf.keras.layers.Embedding(vocab_size, embedding_dim)
        self.rnn = tf.keras.layers.SimpleRNN(rnn_units, return_sequences=True, recurrent_dropout=rnn_dropout)
        self.dense = tf.keras.layers.Dense(rnn_units, activation='relu')
        self.output = tf.keras.layers.Dense(rnn_units, activation='softmax')

    def call(self, x):
        x = self.embedding(x)
        x = self.rnn(x)
        x = self.dense(x)
        return self.output(x)

model = RNN(vocab_size, embedding_dim, rnn_units, batch_size, seq_length, rnn_dropout)
model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
```
