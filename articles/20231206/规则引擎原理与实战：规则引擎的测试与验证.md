```python
# 定义规则
rule1 = {
    "condition": "age >= 18",
    "action": "grant_access"
}

rule2 = {
    "condition": "age < 18",
    "action": "deny_access"
}

# 创建规则库
rule_library = []

# 添加规则到规则库
rule_library.append(rule1)
rule_library.append(rule2)

# 执行规则引擎
def execute_rule_engine(data):
    for rule in rule_library:
        if eval(rule["condition"]):
            rule["action"](data)
            break

# 示例数据
data = {
    "age": 20
}

# 执行规则引擎
execute_rule_engine(data)
```
