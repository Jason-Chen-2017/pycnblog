3. 选择：根据问题的适应度来选择种群中的适应度较高的个体，以便进行交叉和变异操作。

4. 交叉：通过交叉操作来产生新的种群，以便进行变异操作。

5. 变异：通过变异操作来产生新的种群，以便进行选择操作。

6. 评估：根据问题的适应度来评估新生成的种群，以便进行选择操作。

7. 终止条件：当满足终止条件时，则停止遗传算法的运行。

$$
P = \frac{f(x_i)}{\sum_{i=1}^{N} f(x_i)}

$$

# 初始化参数
population_size = 100
mutation_rate = 0.1

# 编码
def encode(x):
    return np.array([x])

# 适应度函数
def fitness(x):
    return -x**2

# 选择操作
def selection(population):
    fitness_values = np.array([fitness(x) for x in population])
    probabilities = fitness_values / np.sum(fitness_values)
    selected_indices = np.random.choice(np.arange(len(population)), size=population_size, p=probabilities)
    return [population[i] for i in selected_indices]

# 交叉操作
def crossover(parent1, parent2):
    crossover_point = np.random.randint(0, len(parent1))
    child1 = np.concatenate((parent1[:crossover_point], parent2[crossover_point:]))
    child2 = np.concatenate((parent2[:crossover_point], parent1[crossover_point:]))
    return child1, child2

# 变异操作
def mutation(x, mutation_rate):
    if np.random.rand() < mutation_rate:
        x = x + np.random.uniform(-1, 1)
    return x

# 评估操作
def evaluate(x):
    return fitness(x)

# 遗传算法主体
def genetic_algorithm(population, mutation_rate, generation_num):
    for _ in range(generation_num):
        population = selection(population)
        new_population = []
        for i in range(0, len(population), 2):
            if i + 1 < len(population):
                child1, child2 = crossover(population[i], population[i+1])
                child1 = mutation(child1, mutation_rate)
                child2 = mutation(child2, mutation_rate)
                new_population.extend([child1, child2])
            else:
                new_population.append(population[i])
        population = new_population
    return population

# 主函数
if __name__ == '__main__':
    population = np.random.uniform(-10, 10, population_size)
    mutation_rate = 0.1
    generation_num = 100
    result = genetic_algorithm(population, mutation_rate, generation_num)
    print(result)
