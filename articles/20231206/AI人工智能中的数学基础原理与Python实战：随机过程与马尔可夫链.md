variance = np.sum((np.arange(transition_prob.shape[0]) - mean) ** 2 * p)

# 状态转移
next_state = np.random.choice(np.arange(transition_prob.shape[0]), p=transition_prob[random_variable, :])

# 输出结果

print("下一个状态的值:", next_state)
print("期望:", mean)
print("方差:", variance)
