```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm
import muon

# 生成随机数样本
np.random.seed(42)
n = 1000
x = np.random.randn(n)

# 生成正态分布样本
x_norm = norm.rvs(np.mean(x), np.std(x))

# 计算正态分布样本的均值和标准差
mu = x_norm.mean()
sigma = x_norm.std()

# 生成模拟样本
x_模拟 = mu + sigma * np.random.normal(size=n)

# 计算模拟样本的期望值
np.mean(x_模拟)

# 输出模拟结果
plt.plot(x_norm, np.mean(x_norm), label='均值')
plt.plot(x_模拟， np.mean(x_模拟), label='期望值')
plt.xlabel('均值')
plt.ylabel('期望值')
plt.legend()
plt.show()
```
