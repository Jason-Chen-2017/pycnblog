```
import numpy as np
import tensorflow as tf

# 读取数据
data = pd.read_csv('data.csv')
X = data[['x1', 'x2', 'x3']]
y = data['y']

# 特征提取
X_train = X[:, :1]
X_test = X[:, -1:]

# 特征建模
Y_train = y.copy()
X_train_m = X_train.copy()
X_train_m = X_train_m[:, np.newaxis, np.newaxis]
X_test_m = X_test.copy()
X_test_m = X_test_m[:, np.newaxis, np.newaxis]

# 训练模型
m = tf.keras.models.Sequential([
    tf.keras.layers.Dense(128, activation='relu', input_shape=(X_train_m.shape[1],)),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dense(1)
])
model = m.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy'])

# 训练模型
model.fit(X_train_m, y_train, epochs=5, validation_data=(X_test_m, y_test))
```
