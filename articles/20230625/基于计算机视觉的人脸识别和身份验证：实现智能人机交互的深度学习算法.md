#include <iostream>
using namespace std;

int main()
{
    // 定义模型参数
    int32_t W = 32;
    int32_t C = 8;
    int32_t K = 3;
    int32_t B = 2;
    
    // 定义训练数据
    float* data;
    float* labels;
    
    // 读取训练数据
    data = new float[W * C * B];
    labels = new float[W * C * B];
    for (int i = 0; i < W * C * B; i++)
    {
        data[i] = 0;
        labels[i] = 0;
    }
    for (int i = 0; i < W * C * B; i++)
    {
        data[i] = 1;
        labels[i] = 1;
    }
    
    // 模型训练
    for (int i = 0; i < C * B; i++)
    {
        for (int j = 0; j < B; j++)
        {
            int k = i * K + j;
            if (data[k] == 1)
            {
                for (int l = 0; l < C; l++)
                {
                    if (labels[l] == 1)
                    {
                        data[k]--;
                        labels[l]--;
                    }
                }
            }
        }
    }
    
    // 模型测试
    float accuracy = 0;
    for (int i = 0; i < W * C * B; i++)
    {
        if (data[i] == labels[i])
        {
            accuracy++;
        }
    }
    
    cout << "Accuracy: " << accuracy << endl;
    
    return 0;
}
