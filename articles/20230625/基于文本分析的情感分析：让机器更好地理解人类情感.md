```python
import re
from collections import defaultdict
from bs4 import BeautifulSoup

def sentiment_analysis(text):
    """情感分析函数"""
    # 正则表达式提取文本中的情感信息
    sentiment_pattern = r'\b(<strong>.*</strong>).*\b'
    if sentiment_pattern.ismatch(text):
        # 提取情感强度
        sentiment_score = float(re.search(sentiment_pattern, text).group(1))
        # 提取情感倾向
        sentiment_type = re.search(sentiment_pattern, text).group(2)
        # 将情感信息存储到字典中
        sentiment_dict = defaultdict(int)
        # 遍历字典中的元素
        for key, value in sentiment_dict.items():
            # 赋予值一个概率值
            sentiment_dict[key] += sentiment_score
            # 将概率值转化为情感强度值
            sentiment_score += 0.5
            # 将情感倾向设置为积极(1)或消极(-1)
            if sentiment_type in ['-1', '0']:
                sentiment_type = '积极'
            else:
                sentiment_type = '消极'
            # 将情感强度值转化为情感强度值和倾向值
            sentiment_dict[key] = (sentiment_score, sentiment_type)
        return sentiment_dict
```
