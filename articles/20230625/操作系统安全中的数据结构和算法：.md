```
class Stack:
    def __init__(self):
        self.data = []
        self.size = 0
        
    def push(self, value):
        self.data.append(value)
        self.size += 1
        
    def pop(self):
        return self.data[-1]
        
    def peek(self):
        return self.data[0]
        
    def size(self):
        return self.size
```
