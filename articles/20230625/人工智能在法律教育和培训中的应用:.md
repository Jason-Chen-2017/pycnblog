from nltk.stem import WordNetLemmatizer
from nltk.stem import WordNetLemmatizer

lemmatizer.lemmatize('法律案例分析')
data.append({'lemmatizer':lemmatizer, 'text':stop_words.intersection(lemmatizer.lemmatize('法律案例分析')), 'filename':filename})

# 模型训练
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.model_selection import train_test_split

X_train, X_test, y_train, y_test = train_test_split(data['content'], data['filename'], test_size=0.2)
vectorizer = CountVectorizer(stop_words=stop_words)
X_train = vectorizer.fit_transform(X_train)
X_test = vectorizer.transform(X_test)

# 模型测试
def accuracy(y_true, y_pred):
    total = sum(y_true == y_pred)
    return round(total, 2)

y_pred = vectorizer.predict(X_test)
accuracy = accuracy(y_test, y_pred)
print('Accuracy:', accuracy)
