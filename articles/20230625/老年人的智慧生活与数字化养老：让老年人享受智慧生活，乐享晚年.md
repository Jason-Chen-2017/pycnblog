```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

class elderly_monitor:
    def __init__(self, sensor, threshold):
        self.sensor = sensor
        self.threshold = threshold

    def read_sensor(self):
        return self.sensor.read()

    def check_alarming(self):
        if self.read_sensor() > self.threshold:
            return True
        else:
            return False

class quality_of_life:
    def __init__(self, sensor, threshold):
        self.sensor = sensor
        self.threshold = threshold

    def analyze_qol(self):
        data = []
        for i in range(100):
            sensor_value = self.sensor.read()
            alarm_status = elderly_monitor.check_alarming(sensor_value)
            data.append(alarm_status)
        qol_score = sum(data) / 100
        return qol_score

```
