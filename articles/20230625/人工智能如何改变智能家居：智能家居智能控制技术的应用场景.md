class Smart音箱:
    def __init__(self, audio_source):
        self.recognizer = sr.Recognizer()
        self.audio_source = audio_source
        self.audio = []

    def speak(self, text):
        self.audio.append(self.recognizer.listen(self.audio_source)[0])

    def play(self, index):
        self.recognizer.recognize_sphinx(self.audio[index], self.audio_source, None, 0.5)

    def stop(self):
        self.recognizer.stop_recognizing()
        self.audio.append(None)
