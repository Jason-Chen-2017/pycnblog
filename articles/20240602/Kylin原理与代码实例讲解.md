def main():
    # 创建Kylin实例

    # 设置数据清洗规则
    kylin.set_data_cleaning_rule("data_cleaning.conf")

    # 设置数据类型转换规则
    kylin.set_data_type_conversion_rule("data_type_conversion.conf")

    # 设置数据分片策略
    kylin.set_data_sharding_rule("data_sharding.conf")

    # 设置数据压缩算法
    kylin.set_data_compression_algorithm("gzip")

    # 执行数据处理任务
    kylin.execute()

if __name__ == "__main__":
    main()
