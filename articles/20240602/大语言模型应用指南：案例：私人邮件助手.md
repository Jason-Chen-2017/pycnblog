```
#coding=utf-8
import smtplib
from email.mime.text import MIMEText

def send_email(subject, content, to):
    # 设置邮件服务器
    server = smtplib.SMTP('smtp.example.com', 25)
    # 设置邮件发送者
    server.login('your_email@example.com', 'your_password')
    # 设置邮件接收者
    msg = MIMEText(content)
    msg['Subject'] = subject
    msg['From'] = 'your_email@example.com'
    msg['To'] = to
    # 发送邮件
    server.sendmail('your_email@example.com', [to], msg.as_string())
    # 退出邮件服务器
    server.quit()

def email_reply(user_email, subject, content):
    # 发送邮件
    send_email(subject, content, user_email)
    print("Email sent to %s" % user_email)

if __name__ == '__main__':
    user_email = 'user@example.com'
    subject = 'Auto Reply'
    content = 'Hello, this is a test email from your Personal Email Assistant!'
    email_reply(user_email, subject, content)
```
