```python
import numpy as np
import tensorflow as tf
from tensorflow.keras.layers import Embedding

class PositionalEncoding(tf.keras.layers.Layer):
    def __init__(self, d_model, num_heads, num_positional_encoding):
        super(PositionalEncoding, self).__init__()
        self.pos_encoding = self.positional_encoding(num_positional_encoding)

    def get_angles(self, position):
        angles = 1 / np.power(10000., (np.arange(self.num_heads) * -self.d_model) / self.num_heads)
        return position * angles

    def positional_encoding(self, position):
        seq_len = tf.shape(self.input)[1]
        angles = self.get_angles(tf.range(tf.cast(seq_len, tf.int64), dtype=tf.float32))
        angles = np.arange(position)[:, None] * angles[None, :]
        return tf.expand_dims(tf.concat([tf.math.sin(angles[:, :, 0]), tf.math.cos(angles[:, :, 1])], axis=-1), axis=-1)

    def call(self, inputs):
        return inputs + self.pos_encoding
```
