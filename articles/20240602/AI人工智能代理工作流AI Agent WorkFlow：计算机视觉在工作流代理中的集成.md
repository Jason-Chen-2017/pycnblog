## 项目实践：代码实例和详细解释说明

以下是一个计算机视觉在工作流代理中的集成代码示例：

```python
import cv2
import numpy as np

def gray_transform(image):
    gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    return gray_image

def sobel_edge_detect(image):
    sobel_x = cv2.Sobel(image, cv2.CV_64F, 1, 0, ksize=3)
    sobel_y = cv2.Sobel(image, cv2.CV_64F, 0, 1, ksize=3)
    sobel_edge = cv2.addWeighted(sobel_x, 0.5, sobel_y, 0.5, 0)
    return sobel_edge

def main():
    image = cv2.imread('example.jpg')
    gray_image = gray_transform(image)
    sobel_edge = sobel_edge_detect(gray_image)
    cv2.imshow('Sobel Edge', sobel_edge)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

if __name__ == '__main__':
    main()
```
