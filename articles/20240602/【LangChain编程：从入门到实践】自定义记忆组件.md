        super().__init__()
        self.model = model
        self.tokenizer = tokenizer

    def extract(self, text):
        tokens = self.tokenizer.tokenize(text)
        return tokens

    def update(self, tokens, label):
        self.model.train(tokens, label)

        return self.model.predict(tokens)
