```python
from mesos import Mesos

def main():
    mesos = Mesos("zk://localhost:2181/mesos", "my-slave")
    mesos.start()
    while True:
        offers = mesos.fetch_offers()
        for offer in offers:
            for task in offer.get_tasks():
                if task.get_state() == "RUNNING":
                    mesos.launch_task(task)

if __name__ == "__main__":
    main()
```
