```python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.distributions import Categorical

class MetaRL(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(MetaRL, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_size)
        self.fc2 = nn.Linear(hidden_size, output_size)
        self.log_std = nn.Parameter(torch.zeros(output_size))

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        mean = self.fc2(x)
        std = torch.exp(self.log_std)
        dist = Categorical(torch.distributions.Normal(mean, std))
        return dist

    def meta_optimize(self, task_optimizer, task_data, task_target):
        # optimize on the task
        task_optimizer.zero_grad()
        loss = -self(task_data).log_prob(task_target).mean()
        loss.backward()
        task_optimizer.step()

        # update meta-parameters
        meta_optimizer.zero_grad()
        meta_loss = loss
        meta_loss.backward()
        meta_optimizer.step()

    def train(self, meta_optimizer, task_optimizer, task_data, task_target):
        for _ in range(100):
            self.meta_optimize(task_optimizer, task_data, task_target)
```

在这个代码示例中，我们使用了一个简单的元学习模型。这个模型包括两个全连接层和一个softmax层。我们使用Categorical分布来生成策略。通过运行`train`方法，我们可以训练这个模型来优化策略。
