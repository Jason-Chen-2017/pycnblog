```python
class ShoppingCart:
    def __init__(self):
        self.items = {}

    def add_item(self, id, price):
        if id in self.items:
            self.items[id]['quantity'] += 1
        else:
            self.items[id] = {'price': price, 'quantity': 1}

    def remove_item(self, id):
        if id in self.items:
            del self.items[id]
```
