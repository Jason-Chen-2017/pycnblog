# 定义卷积神经网络
def conv_net(x, shape):
    x = tf.reshape(x, [-1, shape[0], shape[1], 1])
    conv1 = tf.layers.conv2d(x, 32, (3, 3), activation=tf.nn.relu)
    pool1 = tf.layers.max_pooling2d(conv1, (2, 2), strides=(2, 2))
    conv2 = tf.layers.conv2d(pool1, 64, (3, 3), activation=tf.nn.relu)
    pool2 = tf.layers.max_pooling2d(conv2, (2, 2), strides=(2, 2))
    flatten = tf.layers.flatten(pool2)
    dense1 = tf.layers.dense(flatten, 128, activation=tf.nn.relu)
    return dense1

# 定义循环神经网络
def rnn_net(x, shape):
    x = tf.reshape(x, [-1, shape[0], 1])
    lstm = tf.keras.layers.LSTM(128)
    output, state = lstm(x)
    return output

# 定义多模态大模型
def mmbigmodel(x, y):
    conv_shape = (28, 28, 1)
    rnn_shape = (28,)
    conv_out = conv_net(x, conv_shape)
    rnn_out = rnn_net(conv_out, rnn_shape)
    dense1 = tf.layers.dense(rnn_out, 128, activation=tf.nn.relu)
    logits = tf.layers.dense(dense1, 10)
    return logits

# 定义损失函数
def loss_fn(logits, labels):
    loss = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits_v2(logits=logits, labels=labels))
    return loss

# 定义优化算法
def train_op(loss):
    optimizer = tf.train.AdamOptimizer(0.001)
    train_op = optimizer.minimize(loss)
    return train_op
