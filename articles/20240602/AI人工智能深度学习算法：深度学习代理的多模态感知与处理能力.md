```python
import tensorflow as tf
from tensorflow.keras.models import Model
from tensorflow.keras.layers import Input, Dense, Conv2D, MaxPooling2D, Flatten, Embedding, LSTM, concatenate

# 图像数据处理
def image_input(image_shape):
    image = Input(shape=image_shape)
    x = Conv2D(32, (3, 3), activation='relu')(image)
    x = MaxPooling2D((2, 2))(x)
    x = Flatten()(x)
    return image, x

# 文本数据处理
def text_input(vocab_size, max_length):
    text = Input(shape=(max_length,))
    x = Embedding(vocab_size, 128)(text)
    x = LSTM(64)(x)
    return text, x

# 融合
def fusion(image_output, text_output):
    x = concatenate([image_output, text_output])
    x = Dense(64, activation='relu')(x)
    x = Dense(1, activation='sigmoid')(x)
    return x

# 模型构建
image_shape = (28, 28, 1)  # 图像数据形状
vocab_size = 10000  # 词汇表大小
max_length = 100  # 文本序列长度

image, image_output = image_input(image_shape)
text, text_output = text_input(vocab_size, max_length)
output = fusion(image_output, text_output)

model = Model(inputs=[image, text], outputs=output)
model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
```
