```
pragma solidity ^0.8.0;

contract Auth {
    address owner;
    mapping(address => bool) public balances;

    constructor() public {
        owner = msg.sender;
        balances[msg.sender] = true;
    }

    function login(address _user) public payable {
        require(msg.sender == _user, "Receiver is not the sender.");
        require(balances[msg.sender], "Sender does not have sufficient balance.");
        owner.transfer(msg.value);
    }

    function logout(address _user) public {
        require(msg.sender == _user, "Receiver is not the sender.");
        balances[msg.sender] = false;
    }

    function isBalanceActive(address _user) public view returns (bool) {
        return balances[msg.sender] > 0;
    }
}
```
