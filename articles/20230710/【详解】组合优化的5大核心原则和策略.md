```
#include <iostream>
using namespace std;

int max(int a, int b)
{
    return a > b? a : b;
}

{
    int x, y;
    cout << "Enter two integers: ";
    cin >> x >> y;
    int result = max(x, y);
    cout << "The maximum number is: " << result << endl;
    return 0;

```
