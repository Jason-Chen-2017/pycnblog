import spacy

def preprocess_text(text):
    # 加载预训练的spaCy模型
    nlp = spacy.load("en_core_web_sm")
    # 对文本进行分词
    tokens = nlp.preprocess(" ".join(text.split()))
    # 对文本进行词性标注
    tagged_tokens = nlp.pos_tag(tokens)
    # 返回处理后的文本
    return " ".join(tagged_tokens)

def convert_text_to_speech(text, language):
    # 加载预录制的音频文件
    recognizer = sr.Recognizer()
    audio = recognizer.read(f"audio_{language}.mp3")
    # 对文本进行自然语言处理
    text = preprocess_text(text)
    # 将文本转换为语音
    voice = recognizer.voice(audio)
    # 返回处理后的音频
    return voice
