### 4.3. 核心代码实现


```
#include <opencv2/opencv.hpp>  // 包含 OpenCV
#include <opencv2/core.hpp>  // 包含 OpenCV 核心库

using namespace cv;
using namespace cv::ml;  // 用于物体检测的 OpenCV 机器学习库

int main(int argc, char** argv)
{
    // 初始化摄像头
    VideoCapture camera(0);  // 从0号摄像头开始
    camera.set(CAP_PROP_FRAME_WIDTH, 640);  // 设置帧宽为640
    camera.set(CAP_PROP_FRAME_HEIGHT, 480);  // 设置帧高为480
    // 循环捕捉每一帧图像
    while(true)
    {
        // 读取图像
        // 使用 OpenCV 库读取图像，并转换为 BGR 颜色空间
        Mat frame;
        if(camera.read(frame)!= -1)
        {
            // 将图像转换为灰度图像
            cvtColor(frame, frame, CV_RGB2GRAY);
            // 使用 YOLO 物体检测模型进行物体检测
            detectMultiScale(frame, objects, scaleFactor, minNeighbors, nPoint);
            // 在检测到物体时，画出矩形框并标注类别
            for(int i = 0; i < objects.size(); i++)
            {
                Rect rect = getRectSubpix(frame, objects[i].x, objects[i].y, objects[i].w, objects[i].h);
                putText(frame, rect.x, rect.y, cv::FONT_HERSHEY_SIMPLEX, 1.0, cv::GRAY(0, 255, 255), 2, cv::LINE_AA);
                // 绘制边框
                rectangle(frame, rect.x, rect.y, cv::GRAY(255, 0, 0), 2);
            }
        }
        // 按 'q' 键退出循环
        if(waitKey(1) & 0xFF == ord('q')) break;
    }
    return 0;
}
```

### 4.4. 代码讲解说明

上述代码实现中，我们使用 OpenCV 和 OpenCV 机器学习库中的 YOLO 物体检测模型进行物体检测。首先，初始化摄像头并设置帧宽和帧高。然后，循环捕捉每一帧图像，读取并转换为 BGR 颜色空间的图像。接着，使用 YOLO 模型检测每一帧图像中的物体，并将检测到的物体画出矩形框并标注类别。最后，循环直到按下 'q' 键退出循环，返回 0。
