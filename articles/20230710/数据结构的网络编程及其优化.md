### 3.2. 核心模块实现

实现网络通信的核心模块主要包括以下几个部分：

```python
import socket
import networkx as nx
import threading


class NetworkCommunication:
    def __init__(self, host, port, socket_type):
        self.host = host
        self.port = port
        self.socket_type = socket_type

        # 创建一个socket对象
        self.socket = socket.socket(socket_type)
        self.socket.connect((host, port))

    def send_message(self, message):
        # 发送消息
        print(f"发送消息: {message}")
        self.socket.sendall(message.encode())

    def receive_message(self):
        # 接收消息
        print(f"接收消息: {self.socket.recv(1024).decode()}")

    def close(self):
        # 关闭socket
        print(f"关闭socket: {self.socket.close()}")


if __name__ == "__main__":
    # 创建一个Communication实例
    communication = NetworkCommunication("127.0.0.1", 8888, "socket")

    # 发送消息
    communication.send_message("Hello, network communication!")

    # 接收消息
    communication.receive_message()

    # 关闭socket
    communication.close()
```
