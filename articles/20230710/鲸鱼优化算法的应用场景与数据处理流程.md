4.2. 应用实例分析

import numpy as np
import random

class Vehicle:
    def __init__(self, position, destination, distance):
        self.position = position
        self.destination = destination
        self.distance = distance

    def move(self, distance):
        self.position += self.distance

    def get_distance_to_destination(self, destination):
        return self.distance


def is_valid(vehicle, destination, distance):
    return (
        vehicle.get_distance_to_destination(destination) <= 1  # 距离不能超过 1
        + vehicle.position - destination.position <= 1  # 当前位置与终点之间的距离不能超过 1
    )


def schedule_vehicle(vehicle_queue, vehicles, destinations, distance_matrix):
    vehicle = random.choice(vehicles)
    destination = random.choice(destinations)
    if is_valid(vehicle, destination, distance_matrix[vehicle, :]):
        vehicle.move(distance_matrix[vehicle, :])
        return vehicle
    else:
        return None


def main():
    vehicles = [Vehicle(1, 2, 1), Vehicle(2, 3, 2), Vehicle(3, 4, 3)]
    destinations = [1, 3, 2]
    distance_matrix = [
