from itertools import permutations
from networkx import Graph

def create_graph(data):
    return Graph()

def add_nodes(graph, nodes):
    for node in nodes:
        graph.add_node(node)

def add_links(graph, edges):
    for edge in edges:
        graph.add_edge(edge)

def create_path(graph, start, end, weight):
    return list(permutations(graph.neighbors(start), weight))

def dijkstra(graph, start, end, weight):
    dist = [float('inf')] * len(graph)
    dist[start] = 0
    path = [None] * len(graph)
    path[start] = (start, 0)

    for _ in range(len(graph) - 1):
        current = list(dijkstra(graph, start + 1, end, dist))
        if current[0] == end:
            break
        dist[current[0]] = current[1]
        path[current[0]] = current[0]

    return path

def main():
    data = []
    nodes = ["A", "B", "C", "D", "E"]
    edges = [
        ("A", "B"),
        ("A", "C"),
        ("A", "D"),
        ("A", "E"),
        ("B", "C"),
        ("B", "D"),
        ("B", "E"),
        ("C", "E")
    ]

    graph = create_graph(data)
    add_nodes(graph, nodes)
    add_links(graph, edges)
