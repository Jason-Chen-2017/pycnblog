//天气应用->使用C++实现一个天气应用

#include <iostream>
#include <fstream>
#include <string>

using namespace std;

string get_weather(string city) {
    //从API获取天气数据
    string api_url = "https://api.openweathermap.org/data/2.5/weather?q=";
    api_url += city;
    api_url += "&appid=YOUR_APP_KEY&units=metric";

    //解析天气数据
    string data = "";
    if (api_url.back() == 'https') {
        data = api_url.substr(api_url.back()+4);
    }

    //提取天气信息
    string temperature = api_url.substr(api_url.back()+8, 6);
    string condition = api_url.substr(api_url.back()+12, 6);

    return temperature + "°C " + condition;
}

void display_weather(string city) {
    //从API获取天气数据
    string weather_data = get_weather(city);

    //解析天气数据
    string temperature = weather_data.substr(0, 10);
    string condition = weather_data.substr(11);

    //显示天气信息
    cout << "温度：" << temperature << "°C，天气：" << condition << endl;
}
