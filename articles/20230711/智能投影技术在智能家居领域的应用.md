# 将图像转化为灰度图像

match_result = []
for box, class_id, score in zip(boxes, classes, scores):

    # 绘制

    # 添加
    match_result.append((score, class_id, box))

# 绘制检测结果
cv2.figure()
cv2.imshow('detect_result', match_result)
