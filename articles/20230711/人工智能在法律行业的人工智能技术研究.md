        words.append(word.strip())

```python
import os
import re
import nltk
from nltk.corpus import stopwords

# 读取数据
cases_folder = 'cases'

# 读取数据
cases = []
for filename in os.listdir(cases_folder):
    with open(os.path.join(cases_folder, filename)) as f:
        cases.append([line.strip() for line in f])

# 分词
words = []
for case in cases:
    for line in case:
        words.append(word.strip())

# 去停用词
stop_words = set(stopwords.words('english'))
words = [word for word in words if word not in stop_words]

# 统计词频
word_counts = {}
for word in words:
    if word in word_counts:
        word_counts[word] += 1
    else:
        word_counts[word] = 1

# 输出数据
for word, count in word_counts.items():
    print(f'{word}: {count}')
```

## Q:

2. 如何去除 NLTK 库中的停用词？

A:

在 NLTK 库中，有一些预定义的停用词，如：the、a、an、to、in等。如果您想移除这些停用词，可以使用 `nltk.corpus.stopwords.words_from_file` 方法，它可以在 NLTK 库中加载停用词列表。
```
import nltk

nltk.download('punkt')

stop_words = nltk.corpus.stopwords.words_from_file('english_stopwords.txt')

text = 'example text here'

words = nltk.word_tokenize(text)
filtered_words = [word for word in words if word not in stop_words]
```
