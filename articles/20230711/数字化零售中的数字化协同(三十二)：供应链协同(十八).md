```python
import requests
import pandas as pd

class DataSourcing:
    def __init__(self):
        self.url1 = "https://api.example.com/v1/data1"
        self.url2 = "https://api.example.com/v1/data2"
        self.df = pd.DataFrame()

    def fetch_data(self):
        response1 = requests.get(self.url1)
        response2 = requests.get(self.url2)
        df = pd.DataFrame(response1.json(), columns=response1.keys())
        df = df[df["column_name"]!= ""]
        df = df[df["column_name"] == ""]
        self.df = df

    def store_data(self, data):
        self.df["data"] = data
        self.df.to_csv("data.csv", index=False)

    def apply_data(self, data):
        self.df["data"] = data
```
