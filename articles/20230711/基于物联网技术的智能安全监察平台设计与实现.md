4.4. 代码讲解说明

4.4.1 传感器数据采集

本案例中，通过继电器和烟雾报警器，可以实时监测烟雾报警器的状态。具体实现方式如下：
```
#include <Wire.h>

// 定义传感器类型
#define SENSOR_TYPE 0

// 定义传感器引脚
#define SENSOR_PIN 2

// 定义传感器状态
typedef enum {
  STATE_ALARM = 0,
  STATE_NORMAL = 1,
  STATE_ERROR = 2
} SensorState;

void setup() {
  // 将引脚设置为输入模式
  pinMode(SENSOR_PIN, INPUT);

  // 初始化串口
  Serial.begin(9600);
}

void loop() {
  // 读取传感器数据
  int state = digitalRead(SENSOR_PIN);

  // 处理传感器数据
  if (state == STATE_ALARM) {
    // 发送报警消息
    Serial.print("ALARM");
    Serial.println(" - ");
  } else {
    // 发送正常消息
    Serial.print("NORMAL");
    Serial.println(" - ");
  }

  // 等待下一轮数据
  delay(100);
}
```
