1. 用户发起查询请求，包括商品名称、购买时间等信息。

### 3.3. 核心代码实现

```
// 智能合约

pragma solidity ^0.8.0;

contract MyContract {
    mapping(address => uint256) public balances;
    mapping(address => mapping(address => uint256)) publicTransactions;

    event Transfer(address indexed from, address indexed to, uint256 value);

    function transfer(address sender, address recipient, uint256 amount) external {
        require(sender.send(amount), "发送失败");
        require(recipient. receive(amount), "接收失败");
        balances[sender] += amount;
        balances[recipient] += amount;
        emit Transfer(sender, recipient, amount);
    }

    function getBalance(address sender) external view returns (uint256) {
        return balances[sender];
    }

    function getTransactions(address sender) external view returns (mapping(address => uint256)) {
        return transactions[sender];
    }

    function addTransactions(address sender, uint256 value) external {
        transactions[sender] = value;
    }
}

// 前端页面

<script>
    let contract = new MyContract();
    let blockchain = new Web3.providers.HttpProvider("wss://timestamp.blockchain.info/");

    contract.connect(blockchain);

    document.querySelector("#query-input").addEventListener("input", event => {
        let query = event.target.value;
        contract.send(`query ${query}`).then(result => {
            let result = result.toString();
            let data = JSON.parse(result);
            let diameter = data.diameter;
            let price = data.price;
            let details = `${diameter} ${price}`;
            document.querySelector("#查詢結果").innerHTML = `<div>${details}</div>`;
        });
    });
</script>
```
