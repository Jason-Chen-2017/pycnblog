Direction dir = Direction::NORTH;

State state = State::IDLE;

    // 初始化机器人的位置和方向
    x = Width / 2;
    y = Height / 2;
    dir = Direction::NORTH;
    state = State::IDLE;
    color = RGB::RED;
    // 设置机器人的速度
    speed = 50;

void update() {
    // 更新机器人的位置和方向
    if (dir == Direction::NORTH) {
        x -= speed;
    }
    if (dir == Direction::SOUTH) {
        y += speed;
    }
    if (dir == Direction::EAST) {
        x += speed;
    }
    if (dir == Direction::WEST) {
        y -= speed;
    }
    // 判断机器人的状态
    if (state == State::IDLE) {
        if (x < 0 || x > Width) {
            state = State::HOVER;
        }
        if (y < 0 || y > Height) {
            state = State::HOVER;
        }
    }
    if (state == State::HOVER) {
        if (dir == Direction::NORTH) {
            x -= 2;
        }
        if (dir == Direction::SOUTH) {
            x += 2;
        }
        if (dir == Direction::EAST) {
            y -= 2;
        }
        if (dir == Direction::WEST) {
            y += 2;
        }
    }
    if (state == State::PICKUP) {
        state = State::PUTDOWN;
    }
    if (state == State::PUTDOWN) {
        state = State::IDLE;
    }
    if (state == State::RED) {
        // 设置机器人的颜色为红色
        rgb = RGB::RED;
    }
    else if (state == State::GREEN) {
        // 设置机器人的颜色为绿色
        rgb = RGB::GREEN;
    }
    else if (state == State::BLUE) {
        // 设置机器人的颜色为蓝色
        rgb = RGB::BLUE;
    }
}

    // 绘制机器人的背景颜色

        for (int x = 0; x < WIDTH; x++) {
            setColor(RGB_STATE);
            // 绘制平面
            rect(x, y, 10, 10);
        }
    }
    // 绘制机器人的轮廓颜色

    TextSize(40);
    TextAlign(CENTER);
    Text(x + 40, y + 50, "Robot");
}
