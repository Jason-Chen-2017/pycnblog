# 读取图像

# 预处理图像
img_gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

# 特征提取
特征 = cv2.Sobel(img_gray, cv2.CV_16S, 0, 1)

# 分类
clf = cv2.CascadeClassifier(0)
thresh = clf.threshold(特征, 0, 255, cv2.THRESH_BINARY)

# 可视化结果
img_thresh = cv2.imshow("canvas", thresh)
cv2.waitKey(0)
cv2.destroyAllWindows()
