#define PIN 18
#define TIP_PIN 23

#define MAX_ATTEMPTS 5

void tryToEnter() {
    int attempts = 0;
    while (attempts < MAX_ATTEMPTS) {
        digitalWrite(PIN, LOW);
        digitalWrite(TIP_PIN, LOW);
        delay(1000);
        digitalWrite(PIN, HIGH);
        digitalWrite(TIP_PIN, HIGH);
        attempts++;
    }
}
