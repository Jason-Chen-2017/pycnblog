## 4.3. 核心代码实现

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import accuracy_score

# 读取原始数据
data = pd.read_csv('data.csv')

# 预处理数据，包括去重、标准化等操作
data = data.drop_duplicates().values
data = (data - 0.5) / 2

# 提取特征
features = []
for col in data:
    features.append(col)

# 将特征添加到模型中
clf = LinearRegression()
clf.fit(features, data)

# 构建检测模型
def detect(threshold):
    models = []
    for i in range(10):
        model = LinearRegression()
        model.fit(features, data)
        models.append(model)
    return models

# 检测网络流量
def detect_network_flow(threshold):
    models = detect([threshold] * 10)
    correct = 0
    for i in range(10):
        model = models[i]
        predicted = model.predict(data)
        if predicted == threshold:
            correct += 1
    return correct / 10.0

# 应用检测模型
detect_flow = detect_network_flow(0.01)
print('检测到的网络流量中，恶意流量所占比例为：', detect_flow)
```
