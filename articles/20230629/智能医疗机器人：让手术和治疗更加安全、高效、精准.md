4.4. 代码讲解说明
这里以一个具体的代码实现为例，展示智能医疗机器人的核心代码实现过程。首先，需要实现机器人的感知模块，包括对手术部位的视觉感知和数据采集。
```python
import cv2
import numpy as np

# 定义手术部位的颜色范围
lower_blue = np.array([50, 50, 200])
upper_blue = np.array([255, 255, 255])

# 定义手术部位的数据采集间隔
interval = 0.1

# 定义手术部位的颜色
color = None

# 循环遍历手术部位
while True:
    # 读取图像数据
    ret, frame = cv2.imread('brain_data.jpg')
    if not ret:
        break
    
    # 调整图像颜色
    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
    hsv[..., 1] = hsv[..., 1] * 0.5
    hsv = cv2.cvtColor(hsv, cv2.COLOR_HSV2BGR)
    
    # 定义数据范围
    lower_red = np.array([0, 0, 0])
    upper_red = np.array([255, 255, 255])
    
    # 循环遍历数据点
    for x in range(0, frame.shape[1], interval):
        for y in range(0, frame.shape[0], interval):
            # 计算颜色差
            diff = cv2.absdiff(hsv[y, x], lower_red, upper_red)
            
            # 调整颜色范围
            lower_blue[0] = max(lower_blue[0], diff[2] - lower_red[2])
            upper_blue[1] = min(upper_blue[1], diff[3] - upper_red[3])
            lower_red[0] = max(lower_red[0], diff[0] - lower_blue[0])
            upper_red[1] = min(upper_red[1], diff[1] - lower_blue[1])
            
            # 统计颜色出现的次数
            count = 0
            
            # 查找最有可能的颜色
            most_probable = None
            most_probable_index = np.argmax(diff)
            
            # 如果颜色出现次数超过阈值
            if count > 10:
                # 计算概率
                prob = most_probable * 100
                count /= prob
                
                # 查找概率最大的颜色
                max_prob = prob
                max_prob_index = np.argmax(prob)
                
                # 计算颜色差异
                delta = diff[max_prob_index] - diff[most_probable_index]
                
                # 如果颜色差异在一定范围内，则认为是同一种颜色
                if delta <= 50:
                    color = (0, 0, 255)
                    break
                
            # 否则，认为是不同颜色
            else:
                color = None
    
    # 保存数据
    if color:
        cv2.imwrite('brain_data_color.jpg', frame)
```
