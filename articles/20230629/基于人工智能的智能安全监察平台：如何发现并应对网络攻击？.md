# 导入所需库
import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras.preprocessing.text import Tokenizer
from tensorflow.keras.preprocessing.sequence import pad_sequences
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Embedding, Dense, Flatten
from tensorflow.keras.optimizers import Adam

# 定义文本特征
def define_text_features(text):
    # 去除标点符号、停用词等
    text = " ".join([word for word in text.split() if word not in stopwords])
    # 对文本进行词向量化处理
    vector = Tokenizer().fit_on_texts(text)
    # 提取特征
    features = vector.transform(pad_sequences(vector.texts_to_sequences(text), maxlen=50))
    # 返回特征
    features = np.array(features)
    return features

# 加载数据集
def load_data(data_path):
    # 读取数据
    data = pd.read_csv(data_path)
    # 去除标点符号、停用词等
    data["text"] = " ".join([word for word in data["text"].split() if word not in stopwords])
    # 对数据进行词向量化处理
    features = define_text_features(data["text"])
    # 返回数据及特征
    return data, features

# 模型训练
def train_model(X_train, y_train):
    # 定义模型
    model = Sequential()
    # 添加嵌入层
    model.add(Embedding(input_dim=X_train.shape[1], output_dim=64, input_length=X_train.shape[2]))
    # 添加一维卷积层
    model.add(Flatten())
    model.add(Dense(32, activation="relu"))
    # 添加输出层
    model.add(Dense(y_train.shape[1], activation="softmax"))
    # 编译模型
    model.compile(optimizer=Adam(lr=0.001), loss="categorical_crossentropy", metrics=["accuracy"])
    # 训练模型
    model.fit(X_train, y_train, epochs=20, batch_size=32)
    # 评估模型
    model.evaluate(X_train, y_train)
    # 返回模型
    return model

# 模型评估
def evaluate_model(model, X_test, y_test):
    # 评估模型
    loss, accuracy = model.evaluate(X_test, y_test)
    # 返回结果
    return loss, accuracy

# 应用示例
if __name__ == "__main__":
    # 读取数据
    data, features = load_data("data.csv")
    # 数据预处理
    X = features
    y = data["label"]
    # 模型训练
    model = train_model(X, y)
    # 模型评估
    loss, accuracy = evaluate_model(model, X_test, y_test)
    # 输出结果
    print("损失:", loss)
    print("准确率:", accuracy)
