4.3. 核心代码实现
这里提供一个简化版的Python代码示例，用于实现一个广告投放系统的核心模块——用户行为分析模块。
```
import jieba
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.pipeline import Pipeline
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split

class UserBehaviorPipeline(Pipeline):
    def __init__(self):
        self.vectorizer = CountVectorizer(stop_words='english', ngram_range=(1, 2))
        self.classifier = MultinomialNB()

    def fit(self, data):
        self.vectorizer.fit(data)
        self.classifier.fit(self.vectorizer.transform(data))

    def transform(self, data):
        return self.classifier.predict(self.vectorizer.transform(data))

# 读取数据
iris = load_iris()
data = iris.data

# 分词
jieba.analyse.jieba_tokenize(data)

# 提取特征
features = self.vectorizer.transform(data)

# 训练模型
p = UserBehaviorPipeline()
p.fit(features)
```
