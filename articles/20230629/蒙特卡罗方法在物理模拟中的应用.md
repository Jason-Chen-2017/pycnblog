```python
import random
import numpy as np

def generate_random_numbers(num_points, num_dims):
    points = []
    for i in range(num_points):
        dim = [0] * num_dims
        for d in range(num_dims):
            dim[d] = random.randint(0, 1)
        points.append(dim)
    return points

def solve_linear_equation(A, b, x):
    return np.linalg.solve(A, b, x)

# 生成随机数
points = generate_random_numbers(100, 2)

# 求解线性方程组
A = np.array([[1, 2], [3, 4]])
b = np.array([[1], [2]])
x = solve_linear_equation(A, b, np.array([points]))

# 分析结果
print("x = ", x)
```
