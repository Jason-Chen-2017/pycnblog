def preprocess_image(image):
    # 1. 缩放图像（将像素值从 0-255 缩放到 0-1 缩放）
    image_small = image / 255.0
    # 2. 图像预处理（将图像从 RGB 模式转换为灰度模式，对像素进行二值化）
    image_gray = np.mean(image_small, axis=2)
    # 3. 图像归一化（将像素值从 0-1 缩放到 0-1）
    image_norm = (image_gray - np.min(image_gray)) / (np.max(image_gray) - np.min(image_gray))
    # 4. 将像素值从 0-1 缩放到 0-1 (与输入图像保持一致)
    image_norm = image_norm * 255.0
    # 5. 将图像从灰度模式转换为 RGB 模式
    image_rgb = (image_norm[:, :, np.newaxis] + 1) / 255.0
    return image_rgb
