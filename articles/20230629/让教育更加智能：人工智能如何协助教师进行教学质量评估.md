- 深度学习（DL）：DL技术可以对复杂数据进行分析和处理，例如对学生的学习过程进行个性化分析。

# 导入所需库
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# 准备数据
data = pd.read_csv('data.csv')

# 提取特征和目标变量
X = data[['score', 'classification']]
y = data['result']

# 将数据分为训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)

# 训练线性回归模型
mse = mean_squared_error(y_test, LinearRegression().fit(X_train, y_train))
rmse = np.sqrt(mse)

# 应用核心代码
def evaluate_teacher(data, X, y):
    # 准备数据
    inner_products = data[['score', 'classification']]
    result = data['result']

    # 训练模型
    model = LinearRegression()
    model.fit(X, y)

    # 进行预测
    result = model.predict(inner_products)

    # 计算评估指标（均方误差）
    rmse = np.sqrt(rmse)

    # 返回评估指标
    return rmse

# 对教师进行教学质量评估
teacher_rmse = evaluate_teacher(data, X, y)
print(f"{teacher_rmse:.2f}")

# 对学生进行教学质量评估
student_rmse = evaluate_teacher(data, X, y)
print(f"{student_rmse:.2f}")
