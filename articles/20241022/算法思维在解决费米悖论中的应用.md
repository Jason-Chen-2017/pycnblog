```python
# 导入必要的库
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.svm import SVC
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score, classification_report

# 加载数据
data = pd.read_csv('exoplanet_data.csv')

# 数据清洗
data.dropna(inplace=True)
data.drop_duplicates(inplace=True)

# 数据整合
X = data[['mass', 'radius', 'temperature', 'atmosphere']]
y = data['habitable']

# 数据标准化
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 划分训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y, test_size=0.2, random_state=42)

# 模型训练
# SVM模型
svm_model = SVC(kernel='linear', C=1.0)
svm_model.fit(X_train, y_train)

# 随机森林模型
rf_model = RandomForestClassifier(n_estimators=100, random_state=42)
rf_model.fit(X_train, y_train)

# 模型评估
# SVM模型
svm_predictions = svm_model.predict(X_test)
print("SVM Accuracy:", accuracy_score(y_test, svm_predictions))
print("SVM Classification Report:")
print(classification_report(y_test, svm_predictions))

# 随机森林模型
rf_predictions = rf_model.predict(X_test)
print("Random Forest Accuracy:", accuracy_score(y_test, rf_predictions))
print("Random Forest Classification Report:")
print(classification_report(y_test, rf_predictions))
```

- **数据整合**：将特征数据X和标签数据y分开，分别用于模型训练和预测。

- **模型训练**：使用SVC和RandomForestClassifier函数分别创建SVM和随机森林模型，并使用fit函数对模型进行训练。
- **模型评估**：使用predict函数对测试集数据进行预测，并使用accuracy_score和classification_report函数评估模型性能，输出准确率和分类报告。
