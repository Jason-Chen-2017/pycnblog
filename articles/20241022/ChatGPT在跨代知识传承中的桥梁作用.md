def generate_text(input_text):
    # 加载预训练模型
    model = load_pretrained_model()

    # 输入文本进行编码
    encoded_input = model.encode(input_text)

    # 生成文本
    generated_text = model.generate(encoded_input, max_length=100)

    return generated_text
