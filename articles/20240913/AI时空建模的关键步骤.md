```python
import numpy as np

def clean_data(data):
    # 去除缺失值
    data = np.nan_to_num(data, nan=np.mean(data))

    # 去除异常值，这里使用三倍标准差法
    mean = np.mean(data)
    std = np.std(data)
    threshold = 3 * std
    data = data[(data > mean - threshold) & (data < mean + threshold)]

    return data
```

```python
import numpy as np

def linear_regression_train(X, y, learning_rate=0.01, num_iterations=1000):
    m, n = X.shape
    w = np.zeros(n)
    
    for i in range(num_iterations):
        predictions = X.dot(w)
        error = predictions - y
        gradient = X.T.dot(error) / m
        
        w -= learning_rate * gradient
    
    return w

def linear_regression_predict(X, w):
    return X.dot(w)
```

#### 答案解析：

在这个函数中，首先初始化模型参数`w`为0。然后，使用梯度下降法训练模型，每次迭代通过计算损失函数的梯度来更新参数。最后，返回训练好的模型参数。在预测阶段，使用训练好的模型参数进行预测。

**答案：** 常见的模型评估指标包括：

1. **准确率（Accuracy）**：预测正确的样本数占总样本数的比例，适用于分类问题；
2. **召回率（Recall）**：预测正确的正样本数占实际正样本数的比例，适用于分类问题；

6. **均方根误差（RMSE）**：均方误差的平方根，用于衡量预测值与真实值之间的差异，适用于回归问题；

```python
from sklearn.metrics import accuracy_score, recall_score, precision_score, f1_score

def model_evaluation(y_true, y_pred):
    accuracy = accuracy_score(y_true, y_pred)
    recall = recall_score(y_true, y_pred, average='weighted')
    precision = precision_score(y_true, y_pred, average='weighted')
    f1 = f1_score(y_true, y_pred, average='weighted')
    
    return accuracy, recall, precision, f1
```
