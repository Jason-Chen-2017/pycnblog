```python
import numpy as np
import matplotlib.pyplot as plt

# 定义场景构建函数
def build_scenario():
    # 生成道路坐标
    road_x = np.linspace(0, 1000, 1000)
    road_y = np.full(1000, 0)

    # 生成障碍物
    obstacle_x = np.random.randint(0, 1000, size=10)
    obstacle_y = np.random.randint(0, 100, size=10)

    # 绘制场景
    plt.plot(road_x, road_y, label="Road")
    plt.scatter(obstacle_x, obstacle_y, label="Obstacles")
    plt.xlabel("X-axis")
    plt.ylabel("Y-axis")
    plt.title("Autonomous Navigation Scenario")
    plt.legend()
    plt.show()

# 调用场景构建函数
build_scenario()
```
