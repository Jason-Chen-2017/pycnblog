class SmartGrid:
    def __init__(self, energy_production, energy_consumption):
        self.energy_production = energy_production
        self.energy_consumption = energy_consumption

    def optimize_energy_distribution(self):
        surplus_energy = self.energy_production - self.energy_consumption
        if surplus_energy > 0:
            print("当前电网有剩余能源：", surplus_energy, "千瓦时")
        else:
            print("当前电网能源不足，请增加生产或减少消耗")

grid = SmartGrid(1200, 1000)
grid.optimize_energy_distribution()
