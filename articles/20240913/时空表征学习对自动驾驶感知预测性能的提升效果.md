```python

import matplotlib.pyplot as plt

# 假设 data 是一个高维时空数据矩阵
tsne = TSNE(n_components=2, random_state=0)
data_2d = tsne.fit_transform(data)

# 可视化降维后的数据
plt.scatter(data_2d[:, 0], data_2d[:, 1])
```
