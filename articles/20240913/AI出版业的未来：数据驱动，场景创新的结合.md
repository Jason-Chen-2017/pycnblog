```python
    {"user_id": 1, "action": "read", "time": 300},
    {"user_id": 1, "action": "like", "time": 10},
    {"user_id": 2, "action": "read", "time": 180},
    # ... 更多数据
]

# 统计每个用户的平均阅读时间
from collections import defaultdict

user_read_time = defaultdict(float)
for action in user_actions:
    user_read_time[action["user_id"]] += action["time"]

avg_read_time = {user_id: time / len(actions) for user_id, time in user_read_time.items()}
print(avg_read_time)
```
