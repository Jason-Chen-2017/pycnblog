```python
import numpy as np
from sklearn.linear_model import LinearRegression

# 假设我们已经有了历史数据X和需求量y
X = np.array([[1, 2], [2, 3], [3, 4], [4, 5]])
y = np.array([2, 3, 4, 5])

# 创建线性回归模型
model = LinearRegression()

# 训练模型
model.fit(X, y)

# 预测下一个月的需求量
next_month = np.array([[5, 6]])
predicted_demand = model.predict(next_month)
print("预测下一个月的需求量：", predicted_demand)
```
