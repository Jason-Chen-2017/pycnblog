### 4.2 学习推荐代码实例

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 加载学习资源数据
resources = pd.read_csv('learning_resources.csv')

# 使用TF-IDF向量化学习资源内容
vectorizer = TfidfVectorizer()
resources['content'] = vectorizer.fit_transform(resources['content'])

# 计算学习资源之间的相似度
similarity = cosine_similarity(resources['content'], resources['content'])

# 根据学生的学习需求和兴趣，为学生推荐合适的学习资源
def recommend(student_interest, resources):
    student_interest_vector = vectorizer.transform([student_interest])
    similarity_scores = student_interest_vector.dot(similarity)
    recommended_resources = resources.iloc[similarity_scores.argsort()[:5]]
    return recommended_resources
```

from flask_cors import CORS

CORS(app)

    student_question = request.args.get('question')
    relevant_resources = resources[resources['category'].isin(['math', 'physics', 'chemistry'])]
    relevant_resources = relevant_resources[relevant_resources['title'].str.contains(student_question, case=False)]
    return jsonify(relevant_resources.to_dict(orient='records'))

    app.run(debug=True)
