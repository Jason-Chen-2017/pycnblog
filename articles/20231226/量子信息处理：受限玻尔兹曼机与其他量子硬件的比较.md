```python
from qiskit import QuantumCircuit, execute, Aer

# 定义量子门
def hadamard_gate(q):
    q.h(0)

# 创建量子电路
qc = QuantumCircuit(1, 1)

# 应用量子门
hadamard_gate(qc)

# 绘制量子电路
print(qc.draw())
```

```python
# 实例化量子计算机
simulator = Aer.get_backend('qasm_simulator')
```
