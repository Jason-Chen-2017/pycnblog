```python
def f(x):
    return np.exp(-x**2)

def g(x):
    return x**2 + 3*x + 2

def h(x):
    return np.sin(x)

def main():
    x = np.linspace(-10, 10, 1000)
    y = f(x)
    z = g(x)
    w = h(x)

    dy_dx = np.gradient(y, x)
    dz_dx = np.gradient(z, x)
    ww = np.gradient(w, x)

    print("dy/dx =", dy_dx)
    print("dz/dx =", dz_dx)
    print("dw/dx =", ww)

if __name__ == "__main__":
    main()
```
