```python
def trade_match(buy_orders, sell_orders):
    matched_orders = []
    for buy_order in buy_orders:
        for sell_order in sell_orders:
            if buy_order[0] == sell_order[0] and buy_order[1] == sell_order[1]:
                matched_orders.append((buy_order, sell_order))
                buy_orders.remove(buy_order)
                sell_orders.remove(sell_order)
    return matched_orders
```
