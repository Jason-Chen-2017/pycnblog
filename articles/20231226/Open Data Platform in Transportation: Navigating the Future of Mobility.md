# 数据转换

# 数据融合

print(model.score(X_test, y_test))

from scipy.optimize import minimize

# 目标函数
def objective(x):
    return x[0]**2 + x[1]**2

# 约束条件
def constraint(x):
    return x[0] + x[1] - 1

# 优化问题
con = {'type': 'eq', 'fun': constraint}
res = minimize(objective, [0, 0], constraints=con)

# 优化结果
print(res.x)
