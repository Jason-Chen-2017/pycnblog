import numpy as np
import matplotlib.pyplot as plt

# 引力波信号模拟
def simulate_grb_signal(frequency, amplitude, duration):
    t = np.linspace(0, duration, int(duration * 1e4))
    signal = amplitude * np.sin(2 * np.pi * frequency * t)
    return signal

# 引力波信号滤波
def filter_grb_signal(signal, filter_cutoff_frequency):
    nyquist_frequency = 0.5 * len(signal)
    filter_coefficients = np.blackman(int(nyquist_frequency / filter_cutoff_frequency))
    filtered_signal = np.convolve(signal, filter_coefficients, mode='valid')
    return filtered_signal

# 引力波信号分析
def analyze_grb_signal(signal):
    freq = np.fft.fft(signal)
    freq = np.abs(freq[1:int(len(freq) / 2)])
    return freq

# 主程序
if __name__ == '__main__':
    frequency = 100  # Hz
    amplitude = 1  # m
    duration = 1  # s
    filter_cutoff_frequency = 10  # Hz

    # 模拟引力波信号
    signal = simulate_grb_signal(frequency, amplitude, duration)

    # 滤波引力波信号
    filtered_signal = filter_grb_signal(signal, filter_cutoff_frequency)

    # 分析引力波信号
    analyzed_signal = analyze_grb_signal(filtered_signal)

    # 绘制信号
    plt.figure()
    plt.plot(signal)
    plt.title('Original Gravitational Wave Signal')
    plt.xlabel('Time [s]')
    plt.ylabel('Amplitude [m]')

    plt.figure()
    plt.plot(filtered_signal)
    plt.title('Filtered Gravitational Wave Signal')
    plt.xlabel('Time [s]')
    plt.ylabel('Amplitude [m]')

    plt.figure()
    plt.plot(analyzed_signal)
    plt.title('Analyzed Gravitational Wave Signal')
    plt.xlabel('Frequency [Hz]')
    plt.ylabel('Amplitude')

    plt.show()
