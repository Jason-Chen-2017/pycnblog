```python
import numpy as np

def calculate_nutrition_ratio(food):
    protein, fat, carbohydrate, vitamin, mineral = food['nutrition']
    total_nutrition = protein + fat + carbohydrate + vitamin + mineral
    nutrition_ratio = [protein / total_nutrition, fat / total_nutrition, carbohydrate / total_nutrition, vitamin / total_nutrition, mineral / total_nutrition]
    return nutrition_ratio

def optimize_food_combination(foods):
    nutrition_ratios = [calculate_nutrition_ratio(food) for food in foods]
    optimized_foods = []
    for food in foods:
        if np.sum(np.abs(nutrition_ratios[foods.index(food)] - nutrition_ratios[optimized_foods[-1]])) < 0.01:
            optimized_foods.append(food)
        else:
            optimized_foods.append(foods[np.argmin(np.abs(nutrition_ratios[foods.index(food)] - nutrition_ratios[optimized_foods[-1]]))])
    return optimized_foods
```
