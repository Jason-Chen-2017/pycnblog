```python
import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler

# 读取数据
data = pd.read_csv('data.csv')

# 数据预处理
data['longitude'] = data['longitude'].apply(lambda x: np.radians(x))
data['latitude'] = data['latitude'].apply(lambda x: np.radians(x))

# 特征提取
features = data[['longitude', 'latitude', 'population']]

# 特征编码
scaler = StandardScaler()
features_encoded = scaler.fit_transform(features)

# 模型训练

model = LinearRegression()
model.fit(features_encoded, data['area'])

# 模型评估

from sklearn.metrics import mean_squared_error

print('MSE:', mse)
```
