```python
def random_graph(n, m):
    G = nx.Graph()
    G.add_nodes(n)
    for i in range(m):
        u, v = random.sample(range(n), 2)
        G.add_edge(u, v)
    return G
```

```python
def random_traffic(n, rate, duration):
    traffic = []
    for i in range(n):
        t = np.random.uniform(0, duration)
        p = np.random.poisson(rate)
        traffic.append((t, p))
    return traffic

rate = 10
duration = 100

```
