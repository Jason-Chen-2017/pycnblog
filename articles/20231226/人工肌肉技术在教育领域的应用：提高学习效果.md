import matplotlib.pyplot as plt

# 模拟人工肌肉技术的工作原理
def simulate_amit(k, l, theta):
    F = k * l * np.cos(theta)
    return F

# 定义人工肌肉技术的参数

l = 10
theta = np.pi / 4

# 计算人工肌肉技术的力

# 绘制人工肌肉技术的力-角度关系曲线

plt.plot(theta_values, F_values)
plt.xlabel('Angle (radians)')
plt.ylabel('Force (N)')
plt.title('Force-Angle Relationship of Artificial Muscle Interface Technology')
