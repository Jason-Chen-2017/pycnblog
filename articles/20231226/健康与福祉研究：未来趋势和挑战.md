# 4. 具体代码实例和详细解释说明

```python
import numpy as np

# 生活期望数据
life_expectancy = [70, 72, 74, 76, 78]

# 健康指数数据

# 福祉指数数据
welfare_index = [0.7, 0.75, 0.8, 0.85, 0.9]

# 权重
weights = [1, 1, 1, 1, 1]

# 计算生活期望
def calculate_life_expectancy(data, weights):
    return np.sum(data * weights) / np.sum(weights)

# 计算健康指数
def calculate_health_index(data, weights):
    return np.sum(data * weights) / np.sum(weights)

# 计算福祉指数
def calculate_welfare_index(data, weights):
    return np.sum(data * weights) / np.sum(weights)

# 计算生活期望
life_expectancy = calculate_life_expectancy(life_expectancy, weights)
print("生活期望：", life_expectancy)

# 计算健康指数
health_index = calculate_health_index(health_index, weights)
print("健康指数：", health_index)

# 计算福祉指数
welfare_index = calculate_welfare_index(welfare_index, weights)
```
