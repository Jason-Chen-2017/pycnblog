## 4.1 人工智能算法实例

### 4.1.1 生产线控制与自动化

```python
import numpy as np
import matplotlib.pyplot as plt

def generate_data():
    t = np.linspace(0, 10, 100)
    x = np.sin(t)
    y = np.cos(t)
    return t, x, y

def control_system(x, y, dt=0.1):
    u = np.arctan2(y, x)
    v = np.array([x, y])
    v_dot = np.array([v[1], -v[0]])
    v_dot += dt * np.array([u, 0])
    return v_dot

t, x, y = generate_data()
v_dot = control_system(x, y)
v = np.cumsum(v_dot, axis=0)

plt.plot(t, x, label='x')
plt.plot(t, y, label='y')
plt.plot(t, v[:, 0], label='v_x')
plt.plot(t, v[:, 1], label='v_y')
plt.legend()
plt.show()
```

### 4.1.2 质量控制与检测

```python
import numpy as np

def generate_data():
    x = np.random.normal(0, 1, 100)
    y = x**2
    return x, y

def regression(x, y, method='linear'):
    if method == 'linear':
        m, c = np.polyfit(x, y, 1)
        return m, c
    else:
        raise ValueError('Unsupported method')

x, y = generate_data()
m, c = regression(x, y)
y_pred = m * x + c

plt.scatter(x, y, label='data')
plt.plot(x, y_pred, label='regression')
plt.legend()
plt.show()
```

### 4.3.1 设备监控与管理

```python
import time
import random

class Device:
    def __init__(self, id, status):
        self.id = id
        self.status = status

    def monitor(self):
        while True:
            status = random.choice(['normal', 'warning', 'error'])
            print(f'Device {self.id}: {status}')
            time.sleep(1)

device1 = Device(1, 'normal')
device2 = Device(2, 'warning')

device1.monitor()
device2.monitor()
```
