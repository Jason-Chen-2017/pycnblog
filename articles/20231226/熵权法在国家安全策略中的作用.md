### 1.4.1 代码实例

以下是一个简单的熵权法代码实例：

```python
import numpy as np

def f1(x):
    return x[0]**2 + x[1]**2

def f2(x):
    return -x[0]**2 - x[1]**2

def entropy_weight(f1, f2):
    w1 = 0.5
    w2 = 1.0
    return w1 * f1 + w2 * f2

def entropy_optimization(f1, f2, w1, w2):
    x = np.array([0.0, 0.0])
    while True:
        x_new = x - 0.1 * np.array([f1(x), f2(x)])
        if f1(x_new) > f1(x):
            x = x_new
        else:
            break
    return x

x_opt = entropy_optimization(f1, f2, w1, w2)
print("优化后的决策变量：", x_opt)
```
