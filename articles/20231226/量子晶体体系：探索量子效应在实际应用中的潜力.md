import scipy.fftpack

# 生成随机光信号
def random_light_signal(t, amplitude, frequency):
    return amplitude * np.cos(2 * np.pi * frequency * t)

# 生成随机晶体体系响应
def random_crystal_response(t, amplitude, frequency):
    return amplitude * np.cos(2 * np.pi * frequency * t + np.random.uniform(-np.pi, np.pi))

# 计算光信号的傅里叶变换
def light_signal_fft(t, amplitude, frequency):
    signal = random_light_signal(t, amplitude, frequency)
    fft_signal = scipy.fftpack.fft(signal)
    return fft_signal

# 计算晶体体系响应的傅里叶变换
def crystal_response_fft(t, amplitude, frequency):
    response = random_crystal_response(t, amplitude, frequency)
    fft_response = scipy.fftpack.fft(response)
    return fft_response

# 计算光信号与晶体体系响应的相关度
def correlation(light_signal_fft, crystal_response_fft):
    return np.sum(light_signal_fft * np.conj(crystal_response_fft)) / (np.linalg.norm(light_signal_fft) * np.linalg.norm(crystal_response_fft))

# 主程序
if __name__ == '__main__':
    t = np.linspace(0, 1, 1000)
    amplitude = 1
    frequency = 10

    light_signal_fft = light_signal_fft(t, amplitude, frequency)
    crystal_response_fft = crystal_response_fft(t, amplitude, frequency)

    correlation_result = correlation(light_signal_fft, crystal_response_fft)
    print("光信号与晶体体系响应的相关度：", correlation_result)
