```python
import numpy as np
import pandas as pd

# 读取基因组数据
gene_data = pd.read_csv('gene_data.csv')

# 定位要修改的基因
target_gene = gene_data.loc[gene_data['gene_name'] == 'target_gene', 'gene_position'].values[0]

# 选择合适的基因编辑器
editor = 'CRISPR/Cas9'

# 设计基因编辑器
editor_design = f'{editor} for {target_gene}'

# 应用基因编辑器
gene_data.loc[gene_data['gene_name'] == 'target_gene', 'gene_position'] = editor_design

# 保存修改后的基因组数据
gene_data.to_csv('modified_gene_data.csv', index=False)
```
