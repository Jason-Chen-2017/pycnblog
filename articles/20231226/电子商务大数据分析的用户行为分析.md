# 2.核心概念与联系

4. **用户行为预测**：通过分析用户行为数据和特征，企业可以预测用户将会进行哪些操作。例如，通过分析用户的购买历史，企业可以预测用户将会购买哪些商品。

import pandas as pd
from scipy.spatial.distance import cosine
from scikit-learn.metrics.pairwise import pairwise_distances

# 读取用户行为数据
data = pd.read_csv('user_behavior.csv')

# 计算用户之间的相似度
similarity = pd.DataFrame(index=data.userId, columns=data.userId)
for i in range(len(data)):
    for j in range(i+1, len(data)):
        similarity.loc[data.userId[i], data.userId[j]] = cosine(data.iloc[i:(i+1)], data.iloc[j:(j+1)])
        similarity.loc[data.userId[j], data.userId[i]] = similarity.loc[data.userId[i], data.userId[j]]

# 找出具有相似行为的用户
threshold = 0.5
similar_users = similarity.loc[(similarity.T > threshold).any(axis=1)]

# 推荐用户喜欢的商品
recommendations = data.groupby('userId').sum().drop('userId', axis=1)
recommended_items = recommendations.loc[similar_users.index].drop(similar_users.index, axis=1)

# 计算推荐结果的准确率
correct_recommendations = 0
for i in range(len(recommended_items.columns)):
    top_n = recommended_items.iloc[:, i].nlargest(10)
    for j in range(len(top_n)):
        if data.iloc[data.userId == recommended_items.index[i]].userAction == 'buy' and data.iloc[data.userId == top_n.index[j]].userAction == 'buy':
            correct_recommendations += 1
accuracy = correct_recommendations / (len(recommended_items.columns) * 10)
print('准确率:', accuracy)
