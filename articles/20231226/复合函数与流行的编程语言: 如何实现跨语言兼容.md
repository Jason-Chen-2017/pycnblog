```python
from py4j.java_gateway import JavaGateway, java_import

# 创建 Java 对象
gateway = JavaGateway()
java_list = java_import("java.util.ArrayList")
java_int = java_import("java.lang.Integer")

# 定义 Python 函数
def python_func(x):
    return x * 2

# 定义 Java 函数
def java_func(x):
    list = java_list()
    for i in range(10):
        list.add(java_int(i * 2))
    return list

# 调用 Java 函数
result = gateway.entry_point.java_func(5)

# 调用 Python 函数
result = gateway.entry_point.python_func(result[0])

# 打印结果
print(result)
```
