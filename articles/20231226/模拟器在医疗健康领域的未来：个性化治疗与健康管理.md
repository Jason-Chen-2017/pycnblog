    # 收集个性化因素

    environment = np.array(environment)
    lifestyle = np.array(lifestyle)
    
    # 建立个性化治疗模型
    model = np.dot(genes, environment) + np.dot(environment, lifestyle)
    
    # 输入参数

    lifestyle = [6, 7]
    
    # 运行模型

    print(result)
