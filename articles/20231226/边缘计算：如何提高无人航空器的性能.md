```python
import cv2

def compress_image(image, quality):
    # 将图像转换为YCrCb色彩空间
    image_ycrcb = cv2.cvtColor(image, cv2.COLOR_BGR2YCrCb)

    # 对Y通道进行压缩
    image_ycrcb[:, :, 0] = cv2.resize(image_ycrcb[:, :, 0], (image.shape[1] // quality, image.shape[0] // quality), interpolation=cv2.INTER_AREA)

    # 将图像转换回BGR色彩空间
    image_compressed = cv2.cvtColor(image_ycrcb, cv2.COLOR_YCrCb2BGR)

    # 返回压缩后的图像
    return image_compressed
```

## 4.3 数据分析

在无人航空器中，数据分析通常使用机器学习算法。例如，我们可以使用Python和scikit-learn库实现目标识别。以下是一个使用Python和scikit-learn库实现的目标识别示例：

```python
from sklearn.cluster import KMeans

def detect_objects(image):
    # 将图像转换为灰度图像
    image_gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # 使用KMeans算法对图像进行分类
    kmeans = KMeans(n_clusters=2)
    labels = kmeans.fit_predict(image_gray.reshape(-1, 2))

    # 绘制分类结果
    image_colored = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    for i in range(image.shape[0]):
        for j in range(image.shape[1]):
            if labels[i, j] == 0:
                image_colored[i, j, :] = [0, 0, 255]
            else:
                image_colored[i, j, :] = [255, 0, 0]

    # 返回分类后的图像
    return image_colored
```
