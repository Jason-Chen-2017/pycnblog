def probability_pca(data, components=50):
    # 对数据点的概率分布进行估计
    mean = np.mean(data, axis=0)

    dist = np.linalg.cholesky(cov)

    # 对估计的概率分布进行PCA
    eigenvalues, eigenvectors = np.linalg.eig(cov)
