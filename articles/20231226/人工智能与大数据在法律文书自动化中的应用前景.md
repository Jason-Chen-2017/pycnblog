## 4.1 文本分类

### 4.1.1 使用朴素贝叶斯分类器

```python
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, f1_score

# 文本数据
texts = ["刑事案件", "民事案件", "行政案件", "经济案件", "税收案件"]

# 文本分类
labels = [0, 1, 2, 3, 4]

# 数据预处理
vectorizer = CountVectorizer()
X = vectorizer.fit_transform(texts)

# 模型训练
clf = MultinomialNB()
clf.fit(X, labels)

# 模型评估
X_test = vectorizer.transform(["刑事案件", "民事案件", "行政案件", "经济案件", "税收案件"])
y_test = [0, 1, 2, 3, 4]

y_pred = clf.predict(X_test)

print("准确率:", accuracy_score(y_test, y_pred))
print("F1分数:", f1_score(y_test, y_pred, average='weighted'))
```

## 4.2 文本摘要

### 4.2.1 使用BERT模型

```python
from transformers import BertTokenizer, BertForSequenceClassification
from transformers import pipeline

# 加载BERT模型和令牌化器
tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
model = BertForSequenceClassification.from_pretrained('bert-base-uncased')

# 文本摘要
def summarize(text):
    input_text = text
    input_ids = tokenizer.encode(input_text, add_special_tokens=True)
    attention_mask = [1] * len(input_ids)

    output = model.generate(input_ids, attention_mask=attention_mask, max_length=100, min_length=30)
    summary = tokenizer.decode(output[0], skip_special_tokens=True)

    return summary

# 测试文本摘要
text = "法律文书自动化是人工智能和大数据技术在法律领域的应用，旨在提高法律文书的生成、审查、分析效率，降低法律文书的成本，提高法律文书的质量。"

summary = summarize(text)
print(summary)
```

def named_entity_recognition(text):
    input_text = text
    input_ids = tokenizer.encode(input_text, add_special_tokens=True)
    attention_mask = [1] * len(input_ids)

    output = model.predict([input_ids, attention_mask])
    predictions = torch.argmax(output[0], dim=2)

    # 解码实体标签
    labels = ["O", "B-PER", "I-PER", "B-ORG", "I-ORG", "B-LOC", "I-LOC", "B-MISC", "I-MISC"]
    entities = []
    for i in range(len(input_ids)):
        label = predictions[i][0]
        if label != 0:
            entity = labels[label]
            start = i
            while i + 1 < len(input_ids) and labels[predictions[i + 1][0]] == entity:
                i += 1
            entities.append((entity, text[start:i + 1]))

    return entities
