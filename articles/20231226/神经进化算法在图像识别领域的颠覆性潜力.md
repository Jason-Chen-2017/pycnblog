```python
import numpy as np

def initialize_weights(shape):
    return np.random.randn(*shape)

def fitness(model, x_train, y_train):
    model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])
    return model.evaluate(x_train, y_train, verbose=0)[1]

def mutate(model):
    weights = model.get_weights()
    for layer in range(len(weights)):
        if np.random.rand() < 0.1:
            weights[layer] += np.random.randn(*weights[layer].shape) * 0.01
    model.set_weights(weights)

def select(models, fitnesses):
    idx = np.argsort(fitnesses)[-2:]
    return models[idx]

# 初始化神经网络的权重和偏置
models = [Sequential() for _ in range(100)]
for model in models:
    model.set_weights(initialize_weights(model.get_weights().shape))

# 计算适应度
fitnesses = [fitness(model, x_train, y_train) for model in models]

# 选择适应度较高的神经网络进行变异
best_model = select(models, fitnesses)

# 对变异后的神经网络进行评估，更新适应度
fitnesses = [fitness(model, x_train, y_train) for model in models]
best_model = select(models, fitnesses)

# 重复步骤2-4，直到达到终止条件
for _ in range(100):
    best_model = select(models, fitnesses)
    mutate(best_model)
    fitnesses = [fitness(model, x_train, y_train) for model in models]
    best_model = select(models, fitnesses)
```
