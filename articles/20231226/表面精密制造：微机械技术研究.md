# 切削制造的力矩模型
import numpy as np

# 切削过程的参数
feed_rate = 0.1  # 喂料速度（mm/min）
spindle_speed = 1000  # 旋转速度（rpm）
cutting_force = 50  # 切削力（N）
cutting_moment = 0  # 切削力矩（N·m）

# 切削机械系统的参数
tool_length = 50  # 工具长度（mm）
tool_diameter = 10  # 工具直径（mm）
tool_mass = 0.5  # 工具质量（kg）
workpiece_mass = 10  # 工件质量（kg）

# 计算切削力矩
def calculate_cutting_moment(feed_rate, spindle_speed, cutting_force, tool_length, tool_diameter, tool_mass, workpiece_mass):
    # 计算切削力矩公式
    cutting_moment = cutting_force * tool_length * tool_diameter / 2 + cutting_force * workpiece_mass / (spindle_speed * tool_mass)
    return cutting_moment

# 主程序
if __name__ == '__main__':
    cutting_moment = calculate_cutting_moment(feed_rate, spindle_speed, cutting_force, tool_length, tool_diameter, tool_mass, workpiece_mass)
    print('切削过程的力矩：', cutting_moment, 'N·m')
