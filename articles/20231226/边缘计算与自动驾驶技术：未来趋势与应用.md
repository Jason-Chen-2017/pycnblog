```python
# 循环获取图像
while True:
    ret, frame = cap.read()
    if not ret:
        break

    # 对图像进行处理
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    edges = cv2.Canny(gray, 50, 150)

    # 显示图像
    cv2.imshow('frame', frame)
    cv2.imshow('edges', edges)

    # 等待键盘输入
    k = cv2.waitKey(1)
    if k == 27:
        break

cap.release()
cv2.destroyAllWindows()
```
