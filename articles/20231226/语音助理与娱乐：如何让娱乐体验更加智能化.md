```python
import tensorflow as tf
import numpy as np
import requests

# 使用基于深度学习的语音识别算法来识别用户的语音指令
def voice_recognition(voice_data):
    model = tf.keras.models.load_model('voice_recognition_model.h5')
    voice_features = model.predict(voice_data)
    return voice_features

# 使用基于规则的语义理解算法来理解用户的语音指令
def semantic_understanding(voice_features):
    rules = ['if movie_genre is action, then recommend action movies']
    for rule in rules:
        if rule in voice_features:
            return rule
    return 'unknown'

# 使用基于内容的推荐系统来搜索和筛选合适的电影
def movie_recommendation(movie_data):
    model = tf.keras.models.load_model('movie_recommendation_model.h5')
    movie_features = model.predict(movie_data)
    return movie_features

# 使用基于规则的推理算法来得出结论
def conclusion(movie_features):
    rules = ['if action_movie_score is high, then recommend action movie']
    for rule in rules:
        if rule in movie_features:
            return rule
    return 'unknown'

# 主程序
if __name__ == '__main__':
    voice_data = np.load('voice_data.npy')
    voice_features = voice_recognition(voice_data)
    rule = semantic_understanding(voice_features)
    if rule == 'unknown':
        print('unknown command')
    else:
        movie_data = np.load('movie_data.npy')
        movie_features = movie_recommendation(movie_data)
        conclusion = conclusion(movie_features)
        print(conclusion)
```
