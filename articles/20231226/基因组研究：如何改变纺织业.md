    if a == b:
        return 0
    else:
        return 1

def dynamic_programming(A, B):
    m, n = len(A), len(B)
    cost = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1):
        for j in range(n + 1):
            if i == 0 or j == 0:
                cost[i][j] = 1
            else:

    score = 0
    for i in range(m):
        for j in range(n):
            score += max(A[i], B[j])
    return score

def greedy_assembly(fragments):
    fragments.sort(key=lambda x: x[2], reverse=True)
    contigs = []
    for fragment in fragments:
        found = False
        for contig in contigs:
            if score(contig, fragment) >= fragment[2]:
                contig += fragment
                found = True
                break
        if not found:
            contigs.append(fragment)
    return contigs
