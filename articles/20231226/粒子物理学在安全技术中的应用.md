```python
import random
import numpy as np
import quantum_toolkit as qt

def generate_quantum_key(n):
    key = []
    for _ in range(n):
        bit = random.randint(0, 1)
        key.append(bit)
    return key

def quantum_key_distribution(key):
    qkey = []
    for bit in key:
        if bit == 0:
            qkey.append(qt.Qubit(0))
        else:
            qkey.append(qt.Qubit(1))
    return qkey

def quantum_key_measurement(qkey):
    measured_key = []
    for q in qkey:
        measurement_result = q.measure()
        measured_key.append(measurement_result)
    return measured_key

def quantum_key_reconstruction(measured_key):
    key = []
    for bit in measured_key:
        if bit == 0:
            key.append(0)
        else:
            key.append(1)
    return key

def main():
    n = 10
    key = generate_quantum_key(n)
    qkey = quantum_key_distribution(key)
    measured_key = quantum_key_measurement(qkey)
    final_key = quantum_key_reconstruction(measured_key)

    print("Original key:", key)
    print("Quantum key:", qkey)
    print("Measured key:", measured_key)
    print("Final key:", final_key)

if __name__ == "__main__":
    main()
```

- `quantum_key_measurement(qkey)`：对量子密钥进行测量，得到测量结果。
- `quantum_key_reconstruction(measured_key)`：将测量结果重构为原始密钥。
