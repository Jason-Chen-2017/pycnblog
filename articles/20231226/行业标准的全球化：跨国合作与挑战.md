```python
import scrapy

class StandardSpider(scrapy.Spider):
    name = 'standard'
    allowed_domains = ['example.com']
    start_urls = ['https://www.example.com/standards']

    def parse(self, response):
        for standard in response.css('div.standard'):
            yield {
                'title': standard.css('h2.title::text').get(),
                'description': standard.css('p.description::text').get(),
                'url': standard.css('a::attr(href)').get(),
            }
```
