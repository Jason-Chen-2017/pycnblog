import matplotlib.pyplot as plt

def generate_light(wave_length, intensity):
    # 生成光子源
    light = np.zeros((512, 512), dtype=np.complex)
    for x in range(512):
        for y in range(512):
            light[x, y] = intensity * np.exp(1j * 2 * np.pi * wave_length * (x + y) / 512)
    return light

def transmit_light(light, transmission_matrix):
    # 光子通道的传输
    transmitted_light = np.dot(light, transmission_matrix)
    return transmitted_light

def collect_light(transmitted_light):
    # 光子传输的收集
    collected_light = np.abs(transmitted_light)**2
    return collected_light

def main():
    # 生成光子源
    wave_length = 500
    intensity = 1
    light = generate_light(wave_length, intensity)

    # 生成光子通道传输矩阵
    transmission_matrix = np.random.rand(512, 512)

    # 光子通道的传输
    transmitted_light = transmit_light(light, transmission_matrix)

    # 光子传输的收集
    collected_light = collect_light(transmitted_light)

    # 绘制光子传输的收集结果
    plt.imshow(collected_light, cmap='gray')
    plt.show()

if __name__ == '__main__':
    main()
