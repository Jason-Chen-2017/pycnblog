import numpy as np

nx = 100

# 创建气候模拟的物理量
T = np.zeros((nx, ny))

# 在每个计算节点上进行计算
if rank == 0:
    for i in range(ny):
        for j in range(nx):
            T[j, i] = 0.5 * (X[j, i] + Y[j, i])
    T[0, :] = 1.0
    T[-1, :] = -1.0
    comm.Send(T, dest=1, tag=1)
elif rank == 1:
    T_recv = comm.Recv(source=0, tag=1)
    print("Process", rank, "received:", T_recv)
else:
    T_recv = comm.Bcast(T, root=0)
    print("Process", rank, "received:", T_recv)

# 将计算结果汇总到一个中心节点上
if rank == 0:
    T_sum = np.sum(T)
    print("Sum of T:", T_sum)
