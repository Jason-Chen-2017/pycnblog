# 添加量子门
qc.h(0)  # 对第0个量子比特进行H门操作

# 绘制量子电路

# 将量子电路编译为可执行的量子代码

# 将量子电路编译为可执行的量子代码

# 使用QASM模拟器执行量子电路

result = simulator.run(qobj).result()

# 绘制结果
