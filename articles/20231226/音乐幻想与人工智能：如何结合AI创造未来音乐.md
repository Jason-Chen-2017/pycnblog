import numpy as np
import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

def load_data():
    data = pd.read_csv('path/to/music/data.csv')
    return data

def compute_similarity(data):
    similarity = pd.DataFrame(index=data.index, columns=data.index)
    for i in range(len(data.index)):
        for j in range(i, len(data.index)):
            similarity.loc[i, j] = cosine_similarity(data.iloc[i], data.iloc[j])
    return similarity

    user_index = data.index.get_loc(user_id)
    similarity = compute_similarity(data)
    similarity_user = similarity.iloc[user_index]
    recommended_indices = similarity_user.nlargest(num_recommendations).index
    return data.loc[recommended_indices]

data = load_data()
