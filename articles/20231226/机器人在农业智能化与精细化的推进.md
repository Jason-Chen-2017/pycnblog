## 4.1 种植机器人代码实例

import time

class PlantRobot:
    def __init__(self, seed_num, plant_interval):
        self.seed_num = seed_num
        self.plant_interval = plant_interval

    def plant(self):
        start_time = time.time()
        while time.time() - start_time < self.plant_interval:
            self.sow_seed()
            self.plant_seed()

    def sow_seed(self):
        print("撒放种子")
        self.seed_num -= 1

    def plant_seed(self):
        print("种植")
        if self.seed_num <= 0:
            print("种子不足")
            self.plant_interval *= 2
