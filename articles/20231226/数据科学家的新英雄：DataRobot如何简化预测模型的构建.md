$$

    name='my_dataset',
    data_source=client.data_sources.create(
        name='my_data_source',
        type='csv',
        file_path='/path/to/my/data.csv'

    name='my_task',
    dataset=data,
    target='target_column',
    type='regression'
)

model = client.models.create(
    task=task,
    algorithm='random_forest'
)

# 评估模型
evaluation = client.evaluations.create(
    model=model,
    test_set=test_set
)

    model=model,

    environment='production'
