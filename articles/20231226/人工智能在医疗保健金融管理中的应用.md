```python
import pandas as pd

# 读取数据
data = pd.read_csv('data.csv')

# 数据清洗
data = data.dropna()
data = data[data['age'] >= 18]
data = data[data['age'] <= 65]

# 数据转换
data['sex'] = data['sex'].map({'male': 0, 'female': 1})

# 数据集成
data = pd.concat([data, pd.get_dummies(data['disease'])], axis=1)
data = data.drop(['disease'], axis=1)
```

### 4.1.2 特征选择

```python
from sklearn.feature_selection import SelectKBest
from sklearn.feature_selection import chi2

# 特征选择
selector = SelectKBest(chi2, k=10)
selector.fit(data.drop('disease', axis=1), data['disease'])
data = selector.transform(data.drop('disease', axis=1))
```

### 4.1.3 模型选择

```python
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score

# 模型选择
X_train, X_test, y_train, y_test = train_test_split(data.drop('disease', axis=1), data['disease'], test_size=0.2, random_state=42)
```

### 4.4.2 投资决策支持

```python
from scipy.optimize import minimize

# 投资决策支持
def portfolio_return(weights, mean, std):
    return np.sum(weights * mean)

def portfolio_volatility(weights, mean, std):
    return np.sqrt(np.sum(np.square(weights * std)))

def portfolio_risk(weights, mean, std, target_return=0.05):
    return abs(portfolio_return(weights, mean, std) - target_return) / portfolio_volatility(weights, mean, std)

def minimize_risk(mean, std, num_assets=10):
    weights = np.ones(num_assets) / num_assets
    return minimize(portfolio_risk, weights, args=(mean, std))

weights = minimize_risk(mean, std)
print('Optimal weights:', weights)
```
