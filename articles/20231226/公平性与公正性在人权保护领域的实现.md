def random_allocation(weight_list):
    weight_sum = sum(weight_list)
    random_num = random.random() * weight_sum
    index = 0
    for weight in weight_list:
        if random_num < weight:
            return index
        random_num -= weight
        index += 1

def fair_allocation(weight_list):
