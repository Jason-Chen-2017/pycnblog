def predict(data, alpha=0.9):
    future = data[-1]
    for x in reversed(data[:-1]):
        future = alpha * x + (1 - alpha) * future

def optimize(production_costs, consumption_costs, supply_limits, demand_limits):
    A = np.hstack([-production_costs, consumption_costs])
    b = np.hstack([-supply_limits, demand_limits])
    x = linprog(-b, A_ub=A, bounds=[(0, None), (0, None)])
    return x.x
