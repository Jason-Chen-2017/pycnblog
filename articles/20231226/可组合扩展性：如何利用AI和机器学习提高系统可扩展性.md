```python

    def __init__(self):
        self.data = []

    def extend(self):
        self.data.append(1)

    def combine(self, other):
        self.data.extend(other.data)

class ModuleB:
    def __init__(self):
        self.data = []

    def extend(self):
        self.data.append(2)

    def combine(self, other):
        self.data.extend(other.data)

    module_b = ModuleB()

    module_a.extend()
    module_b.extend()

    module_a.combine(module_b)

    print(module_a.data)

    main()
```
