- 池化层：这些层用于降低图像的分辨率，以减少计算量和减少过拟合。
- 全连接层：这些层用于将卷积层和池化层的输出转换为高级特征，然后将其用于分类或其他任务。

import tensorflow as tf
import numpy as np
import matplotlib.pyplot as plt

# 加载内容图像和风格图像

# 将图像转换为Tensor
content_image_tensor = tf.keras.preprocessing.image.img_to_array(content_image)
style_image_tensor = tf.keras.preprocessing.image.img_to_array(style_image)

# 将Tensor展平
content_image_flattened = content_image_tensor.reshape(-1,)

# 使用卷积神经网络提取内容特征和风格特征
content_features = cnn.predict(content_image_tensor)
style_features = cnn.predict(style_image_tensor)

# 创建一个新的空白图像
new_image = np.zeros_like(content_image_tensor[0])

# 使用内容特征向量和风格特征向量生成一个新的图像
for i in range(content_image_tensor.shape[0]):
    new_image[i] = content_image_tensor[i] * content_features[0][i] + style_image_tensor[i] * style_features[0][i]

# 使用卷积神经网络来优化生成的图像
optimizer = tf.keras.optimizers.Adam(learning_rate=0.01)
loss_function = tf.keras.losses.MeanSquaredError()

for epoch in range(1000):
    with tf.GradientTape() as tape:
        loss = loss_function(new_image, content_image_tensor)
    gradients = tape.gradient(loss, new_image)
    optimizer.apply_gradients(zip(gradients, new_image))
