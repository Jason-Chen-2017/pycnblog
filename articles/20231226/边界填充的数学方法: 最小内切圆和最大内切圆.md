```python
import numpy as np

def min_circumcircle(points):
    center = np.mean(points, axis=0)
    min_distance = np.min([np.linalg.norm(points - center, axis=1)])
    return center, min_distance

def max_circumcircle(points):
    center = np.mean(points, axis=0)
    max_distance = np.max([np.linalg.norm(points - center, axis=1)])
    return center, max_distance

points = np.array([[1, 2], [3, 4], [5, 6], [7, 8]])
min_circle_center, min_circle_radius = min_circumcircle(points)
max_circle_center, max_circle_radius = max_circumcircle(points)

print("最小内切圆的中心: ", min_circle_center)
print("最小内切圆的半径: ", min_circle_radius)
print("最大内切圆的中心: ", max_circle_center)
print("最大内切圆的半径: ", max_circle_radius)
```
