import hashlib

def get_file_hash(file_path):
    with open(file_path, 'rb') as f:
        file_content = f.read()
        return hashlib.md5(file_content).hexdigest()

def check_signature(file_path, signature):
    file_hash = get_file_hash(file_path)
    return file_hash in signature

signature = {'0123456789abcdef0123456789abcdef': ['file1.exe', 'file2.exe']}
file_path = 'file1.exe'

if check_signature(file_path, signature):
    print('This file is a virus.')
else:
    print('This file is safe.')
