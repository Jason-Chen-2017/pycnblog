```python
import numpy as np
import matplotlib.pyplot as plt

# 神经网络类
class NeuralNetwork:
    def __init__(self):
        # 初始化权重和偏置
        self.weights = np.random.rand(1, 2)
        self.bias = np.random.rand(1, 1)
        self.input = None
        self.output = None

    def forward(self, x):
        # 前向传播
        self.input = x
        self.output = np.dot(self.weights, self.input) + self.bias
        return self.output

    def backward(self, y):
        # 反向传播
        error = y - self.output
        d_weights = np.dot(self.input.T, error)
        d_bias = np.sum(error)
        self.weights += d_weights
        self.bias += d_bias

# 训练神经网络
def train_network(network, x, y, epochs):
    for epoch in range(epochs):
        output = network.forward(x)
        network.backward(y)

# 数据集
x = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y = np.array([[0], [1], [1], [0]])

# 创建神经网络
network = NeuralNetwork()

# 训练神经网络
train_network(network, x, y, 1000)

# 测试神经网络
test_output = network.forward(x)
print("Test output:", test_output)

# 可视化结果
plt.scatter(x[:, 0], x[:, 1], c=test_output[:, 0], cmap=plt.cm.seismic)
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.title('Neural Network Output')
plt.show()
```
