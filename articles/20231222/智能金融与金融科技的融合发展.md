```python
# 用户投资建议（0表示不建议，1表示建议）
y = np.array([0, 1, 1, 0])

# 预测新用户的投资建议
new_user = np.array([[5, 6]])
prediction = model.predict(new_user)
print(prediction)
```

```python
# 用户消费记录数据
data = {
    '用户ID': [1, 2, 3, 4, 5],
    '消费金额': [100, 200, 300, 400, 500],
    '消费日期': ['2021-01-01', '2021-01-02', '2021-01-03', '2021-01-04', '2021-01-05']
}
df = pd.DataFrame(data)

# 对用户消费记录进行分析
df['消费总额'] = df['消费金额'].cumsum()
df['平均消费'] = df['消费总额'] / df['消费日期'].astype('int64').max()

# 保存分析结果到CSV文件
df.to_csv('用户消费分析.csv', index=False)
```
