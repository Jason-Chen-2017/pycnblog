# 4.具体代码实例和详细解释说明

4.1 服务器扩展：

```
#!/bin/bash

# 添加新服务器
sudo virsh net-autostart net0
sudo virsh net-create /etc/libvirt/networks/net0.xml
sudo virsh domadd begin --name server2 --ram 2048 --vcpus 2 --disk path=/dev/vdb,format=qcow2,size=20
sudo virsh domadd begin --name server3 --ram 2048 --vcpus 2 --disk path=/dev/vdc,format=qcow2,size=20

# 配置负载均衡器
sudo iptables -t nat -A PREROUTING -p tcp --dport 80 -j LB_TCP
sudo iptables -t nat -A POSTROUTING -j MASQUERADE
```

4.2 存储扩展：

```
#!/bin/bash

# 添加新存储设备
sudo mdadm --create /dev/md0 --level=1 --raid-devices=2 /dev/sdb /dev/sdc
sudo mkfs.ext4 /dev/md0
sudo mkdir /mnt/md0
sudo mount /dev/md0 /mnt/md0
```
