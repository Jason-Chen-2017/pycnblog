```python
import numpy as np
import matplotlib.pyplot as plt
from qiskit import QuantumCircuit, Aer, transpile, assemble
from qiskit.visualization import plot_histogram

# 创建一个量子电路
qc = QuantumCircuit(2, 2)

# 将第一个量子位置为|0>，第二个量子位置为|1>
qc.initialize([1, 0], 0)
qc.initialize([0, 1], 1)

# 对第一个量子位进行Hadamard门操作
qc.h(0)

# 对第一个量子位和第二个量子位进行CNOT门操作
qc.cx(0, 1)

# 将量子电路编译成可执行的量子电路
qc = transpile(qc, Aer.get_backend('qasm_simulator'))

# 将量子电路编译成量子电路程序
qasm_qc = assemble(qc)

# 使用QASM模拟器执行量子电路程序
result = qasm_qc.run(Aer.get_backend('qasm_simulator')).result()

# 获取量子电路的结果
counts = result.get_counts()

# 绘制结果
plot_histogram(counts)
plt.show()
```
