### 4.2.2 网络处理

```python
import threading

# 创建路由表
route_table = {'10.0.0.1': 8080, '10.0.0.2': 8081}

# 创建交换表
switch_table = {'10.0.0.1': '10.0.0.2', '10.0.0.2': '10.0.0.1'}

# 路由处理
def route_handler(sock):
    while True:
        data = sock.recv(1024)
        ip, port = data.split()
        port = int(port)
        if ip in route_table:
            sock.sendto(b'200 OK', (route_table[ip], port))
        else:
            sock.sendto(b'404 Not Found', (ip, port))

# 交换处理
def switch_handler(sock):
    while True:
        data, ip = sock.recvfrom(1024)
        port = switch_table[ip]
        sock.sendto(data, (port, 8080))

# 创建路由处理线程
route_thread = threading.Thread(target=route_handler, args=(sock,))
route_thread.start()

# 创建交换处理线程
switch_thread = threading.Thread(target=switch_handler, args=(sock,))
switch_thread.start()
```
