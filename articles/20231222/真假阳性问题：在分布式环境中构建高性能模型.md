import numpy as np

def calculate_probability(positive_prob, node_prob_given_positive, node_prob):
    return positive_prob * node_prob_given_positive / node_prob

def main():
    positive_prob = 0.1
    node_prob_given_positive = 0.3
    node_prob = 0.05
    node = 1

    probability = calculate_probability(positive_prob, node_prob_given_positive, node_prob)
    print(f"The probability of node {node} being positive is: {probability}")

if __name__ == "__main__":
    main()
