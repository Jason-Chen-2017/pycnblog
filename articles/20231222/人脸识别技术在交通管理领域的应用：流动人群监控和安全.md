import cv2
import numpy as np
import face_recognition

# 加载人脸数据库
face_images = []
face_labels = []

# 加载人脸图片
for filename in os.listdir("face_images"):
    face_image = face_recognition.load_image_file("face_images/" + filename)
    face_images.append(face_image)
    face_labels.append(filename)

# 人脸检测
face_locations = []
face_encodings = []

for face_image in face_images:
    face_locations = face_recognition.face_locations(face_image)
    face_encodings = face_recognition.face_encodings(face_image, face_locations)

    for face_encoding, face_location in zip(face_encodings, face_locations):
        matches = face_recognition.compare_faces(face_images, face_encoding, tolerance=0.6)

        face_name = "Unknown"
        if True in matches:
            match_index = matches.index(True)
            face_name = face_labels[match_index]

        face_locations.append(face_location)
        face_encodings.append(face_encoding)

        cv2.rectangle(face_image, (face_location[3], face_location[0]), (face_location[1], face_location[2]), (0, 0, 255), 2)
        cv2.putText(face_image, face_name, (face_location[3], face_location[0] - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 255), 2)

cv2.imshow("Face Detection", face_image)
cv2.waitKey(0)
