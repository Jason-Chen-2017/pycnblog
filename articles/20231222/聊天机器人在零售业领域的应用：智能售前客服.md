import numpy as np

def language_model(context, word):
    # 计算条件概率
    probability = np.log(context.count[word]) - np.log(context.total)
    return probability
```

import numpy as np

def sequence_generation(context, embedding, temperature=1.0):
    # 初始化上下文
    current_context = context
    current_word = current_context.start_word
    generated_text = [current_word]

    # 迭代生成单词
    while True:
        # 计算条件概率
        probabilities = np.exp(current_context.language_model(current_word, generated_text)) / temperature
        # 选择最有可能的词
        next_word = np.random.choice(current_context.vocabulary, p=probabilities)
        # 更新上下文
        current_context = current_context.update(generated_text, next_word)
        # 更新生成文本
        generated_text.append(next_word)

        # 检查是否到达最大生成长度
        if len(generated_text) >= current_context.max_length:
            break

    return generated_text
```
