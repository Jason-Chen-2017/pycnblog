```python
import matplotlib.pyplot as plt

# 定义疾病传播模型
def SIR_model(S, I, R, beta, gamma):
    new_infected = beta * S * I
    recovered = gamma * I
    new_recovered = R + recovered
    new_susceptible = new_infected - new_recovered
    return new_susceptible, new_infected, new_recovered

# 初始条件
S0 = 990
I0 = 10
R0 = 0
beta = 0.2
gamma = 0.1

# 时间步长
dt = 1

# 模拟过程
time = 0
S = S0
I = I0
R = R0

while S > 0 or I > 0:
    S, I, R = SIR_model(S, I, R, beta, gamma)
    time += dt

# 绘制图像
plt.plot(time, S, label='Susceptible')
plt.plot(time, I, label='Infected')
plt.plot(time, R, label='Recovered')
plt.legend()
plt.show()
```
