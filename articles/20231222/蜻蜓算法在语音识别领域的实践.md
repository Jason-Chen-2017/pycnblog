import numpy as np

def butterfly_algorithm(f, x0, n, d, max_iter):
    X = x0.reshape(1, -1)
    V = np.zeros((n, d))
    D = np.random.rand(n, d)
    y = f(X)
    best_y = y.copy()
    best_X = X.copy()
    for t in range(max_iter):
        for i in range(n):
            V[i] = D[i] * (y[i] - best_y)
            X[0] += V[i]
            y[i] = f(X)
            if y[i] < best_y:
                best_y = y[i]
                best_X = X.copy()
        X = X - np.mean(V, axis=0)
    return best_X, best_y
