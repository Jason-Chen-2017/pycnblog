import random

class DataPartition:
    def __init__(self, data):
        self.data = data
        self.partitions = []

    def partition(self, dimension, num_partitions):
        for i in range(num_partitions):
            start = i * len(self.data) / num_partitions
            end = (i + 1) * len(self.data) / num_partitions
            partition = self.data[start:end]
            self.partitions.append(partition)

class LoadBalancer:
    def __init__(self, tasks, nodes):
        self.tasks = tasks
        self.nodes = nodes

    def load_balance(self, strategy):
        if strategy == 'random':
            return self._random_load_balance()
        elif strategy == 'round_robin':
            return self._round_robin_load_balance()
        else:
            raise ValueError('Invalid load balancing strategy')

    def _random_load_balance(self):
        assigned_tasks = [[] for _ in range(self.nodes)]
        for task in self.tasks:
            node_id = random.randint(0, self.nodes - 1)
            assigned_tasks[node_id].append(task)
        return assigned_tasks

    def _round_robin_load_balance(self):
        assigned_tasks = [[] for _ in range(self.nodes)]
        node_id = 0
        for task in self.tasks:
            assigned_tasks[node_id].append(task)
            node_id = (node_id + 1) % self.nodes
        return assigned_tasks
