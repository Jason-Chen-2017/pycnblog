```python
import numpy as np

def carbon_fiber_composite(carbon_fiber_density, matrix_density, fiber_volume_fraction):
    fiber_volume = fiber_volume_fraction * matrix_density
    matrix_volume = (1 - fiber_volume_fraction) * matrix_density
    composite_density = fiber_volume + matrix_volume
    return composite_density
```
