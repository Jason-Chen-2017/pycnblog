```python
from tinkerpop.graph import Graph
from tinkerpop.structure import Vertex, Edge

# 创建一个图数据库实例
graph = Graph.open('conf/remote.yaml')

# 定义图数据模型
class Person(Vertex):
    __primary_key__ = 'id'
    name = String()
    age = Integer()

class Friendship(Edge):
    __primary_key__ = 'id'

# 创建节点
person1 = graph.addV(Person)
person1.name = 'Alice'
person1.age = 25

person2 = graph.addV(Person)
person2.name = 'Bob'
person2.age = 30

# 创建边
friendship = graph.addE(Friendship)
friendship.source = person1
friendship.target = person2

# 执行图计算查询
for friendship in graph.V().bothE().has('Friendship').repeat(outE('Friendship')).where(outV().has('name', 'Alice')).values('name', 'age'):
    print(friendship)
```
