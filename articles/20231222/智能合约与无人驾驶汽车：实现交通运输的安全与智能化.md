```solidity
pragma solidity ^0.5.0;

contract CarRental {
    address public renter;
    uint public rentalStartTime;
    uint public rentalEndTime;
    uint public rentalPrice;

    event RentalStarted(address renter, uint rentalStartTime);
    event RentalEnded(address renter, uint rentalEndTime);

    function startRental(address _renter, uint _rentalStartTime, uint _rentalEndTime, uint _rentalPrice) public payable {
        require(msg.value == _rentalPrice, "Incorrect rental price");
        renter = _renter;
        rentalStartTime = _rentalStartTime;
        rentalEndTime = _rentalEndTime;
        rentalPrice = _rentalPrice;
        emit RentalStarted(renter, rentalStartTime);
    }

    function endRental(uint _rentalEndTime) public {
        require(block.timestamp >= _rentalEndTime, "Rental not ended");
        emit RentalEnded(renter, rentalEndTime);
    }
}
```
