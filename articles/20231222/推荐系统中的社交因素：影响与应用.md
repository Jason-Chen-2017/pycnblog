import numpy as np
from scipy.sparse.linalg import svds

def cosine_similarity(matrix):
    return 1 - np.spatial.distance.cdist(matrix.flatten(), matrix.flatten(), 'cosine')

def collaborative_filtering(ratings, similarity):
    similarity_matrix = similarity(ratings)
    similarity_matrix = similarity_matrix.tocsr()
    U, s, Vt = svds(similarity_matrix, k=10)
    return np.dot(np.dot(U, s), Vt)

ratings = np.array([[4, 3, 2, 1], [3, 2, 1, 0], [2, 1, 0, 0]])
similarity = cosine_similarity
predictions = collaborative_filtering(ratings, similarity)
print(predictions)
