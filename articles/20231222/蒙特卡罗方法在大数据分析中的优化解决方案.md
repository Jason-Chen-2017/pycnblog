    return np.exp(-np.linalg.norm(x - y)**2 / (2 * sigma**2))

# 定义蒙特卡罗密度估计器

    n_s = len(S)
    f = np.zeros(X.shape[1])

        f[i] = sum(kernel(X[:, i], s, sigma) for s in S) / n_s
    return f
