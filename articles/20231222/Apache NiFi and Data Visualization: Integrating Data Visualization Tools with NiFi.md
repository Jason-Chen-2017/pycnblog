dataSource.setUrl("http://datavisualizationtool.com/api")

dataProcessor.setInputPorts([dataSource.getOutputPort()])

# 创建数据目的地组件

dataFlow.connect(dataProcessor.getOutputPort(), dataDestination.getInputPort())
