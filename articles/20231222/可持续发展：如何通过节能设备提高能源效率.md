class EnergyEfficiency:
    def __init__(self, output, input, voltage, current, time):
        self.output = output
        self.input = input
        self.voltage = voltage
        self.current = current
        self.time = time

    def calculate_efficiency(self):
        self.efficiency = self.output / self.input
        return self.efficiency

    def calculate_power(self):
        self.power = self.voltage * self.current
        return self.power

    def calculate_energy(self):
        self.energy = self.power * self.time
        return self.energy

    def calculate_efficiency_2(self):
        self.efficiency_2 = self.output / self.energy
        return self.efficiency_2
