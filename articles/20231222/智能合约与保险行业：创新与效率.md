contract Insurance {
    address public owner;
    uint public premium;
    uint public coverage;
    uint public expiration;

    event PolicyIssued(address policyholder, uint premium, uint coverage, uint expiration);

    constructor(uint _premium, uint _coverage, uint _expiration) public {
        owner = msg.sender;
        premium = _premium;
        coverage = _coverage;
        expiration = block.timestamp + _expiration;
        emit PolicyIssued(msg.sender, premium, coverage, expiration);
    }

    function claim() public {
        require(block.timestamp >= expiration, "Claim period not yet open");
        require(block.timestamp <= expiration + 30 days, "Claim period has already closed");
        require(msg.sender == owner, "Only the policyholder can claim");
        uint amount = coverage;
        premium = 0;
        emit ClaimPaid(msg.sender, amount);
    }
}
