```python
class Point:

        self.y = y

class Polygon:
    def __init__(self, points):
        self.points = points

def random_polygon(n, min_x, max_x, min_y, max_y):
    points = []
    for i in range(n):
        x = random.uniform(min_x, max_x)
        y = random.uniform(min_y, max_y)
        points.append(Point(x, y))
    return Polygon(points)

polygon = random_polygon(5, -10, 10, -10, 10)
```
