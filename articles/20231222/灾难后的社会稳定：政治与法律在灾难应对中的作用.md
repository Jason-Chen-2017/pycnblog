```python

from networkx.algorithms.flow import min_cost_flow

# 创建一个有向图
G = nx.DiGraph()

# 添加节点和边
G.add_node(1, capacity=10)
G.add_node(2, capacity=10)
G.add_node(3, capacity=10)
G.add_edge(1, 2, flow=5, cost=1)
G.add_edge(2, 3, flow=5, cost=1)

# 求解最小成本流量
result = min_cost_flow(G, source=1, target=3, flow_limit=10)

# 打印结果
print(result)
```
