```python
import math

class SolarSail:
    def __init__(self, area, solar_irradiance, efficiency):
        self.area = area
        self.solar_irradiance = solar_irradiance
        self.efficiency = efficiency

    def calculate_power(self):
        return self.area * self.solar_irradiance * self.efficiency

print("太阳能伞的输出功率：", solar_sail.calculate_power())
```
