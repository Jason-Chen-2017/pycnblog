```python
import numpy as np

def shannon_diversity_index(species_counts):
    p = np.array(species_counts) / np.sum(species_counts)
    return -np.sum(p * np.log2(p))

species_counts = [10, 20, 30, 40]
shannon_diversity_index = shannon_diversity_index(species_counts)
print(shannon_diversity_index)
```

```python
def needleman_wunsch(seq1, seq2, match_score, mismatch_score, gap_penalty):
    m, n = len(seq1), len(seq2)
    score = np.zeros((m + 1, n + 1))
    traceback = np.zeros((m + 1, n + 1), dtype=str)

    for i in range(m + 1):
        for j in range(n + 1):
            if i == 0 and j == 0:
                score[i][j] = 0
            elif i == 0:
                score[i][j] = score[i][j - 1] - gap_penalty
                traceback[i][j] = 'up'
            elif j == 0:
                score[i][j] = score[i - 1][j] - gap_penalty
                traceback[i][j] = 'left'
            else:
                match = seq1[i - 1] == seq2[j - 1]
                match_score = match_score if match else mismatch_score
                score[i][j] = max(score[i - 1][j] - gap_penalty,
                                  score[i][j - 1] - gap_penalty,
                                  score[i - 1][j - 1] + match_score)
                traceback[i][j] = 'up' if score[i][j] == score[i][j - 1] - gap_penalty else \
                                   'left' if score[i][j] == score[i - 1][j] - gap_penalty else \
                                   'diagonal' if score[i][j] == score[i - 1][j - 1] + match_score else ''

    i, j = m, n
    while traceback[i][j] != '':
        if traceback[i][j] == 'up':
            i -= 1
        elif traceback[i][j] == 'left':
            j -= 1
        elif traceback[i][j] == 'diagonal':
            i -= 1
            j -= 1
            seq1_char = seq1[i]
            seq2_char = seq2[j]
            print(seq1_char, seq2_char)
        else:
            break

seq1 = 'ATGC'
seq2 = 'ATAC'
```
