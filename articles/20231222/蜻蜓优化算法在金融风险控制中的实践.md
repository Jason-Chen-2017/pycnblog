```python
    return np.sum(weights**2)

# 初始化蜻蜓群
n = 50
weights = np.random.rand(n, 1)

# 设置参数
beta_0 = 1
beta_1 = 1
alpha_0 = 1
alpha_1 = 1

# 设置终止条件

# 开始优化
for t in range(max_iter):
    # 评估 FITNESS
    fitness = np.array([risk_function(weight) for weight in weights])

    # 更新位置
    for i in range(n):
        weights[i] = weights[i] + beta_0 * np.exp(beta_1 * fitness[i]) * np.random.randn(1, 1)

    # 判断是否满足终止条件
    if np.max(fitness) < 1e-6:
        break

# 输出最优投资组合
print("最优投资组合:", weights[np.argmin(fitness)])
```
