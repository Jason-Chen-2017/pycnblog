### 4.2 数据库管理

数据库管理的具体代码实例如下：

```python
import sqlite3

# 创建表
def create_table(conn, table_name, columns):
    columns_str = ", ".join([f"{column} TEXT" for column in columns])
    conn.execute(f"CREATE TABLE {table_name} ({columns_str})")

# 插入数据
def insert_data(conn, table_name, data):
    values_str = ", ".join([f"'{value}'" for value in data.values()])
    conn.execute(f"INSERT INTO {table_name} VALUES ({values_str})")

# 删除数据
def delete_data(conn, table_name, condition):
    conn.execute(f"DELETE FROM {table_name} WHERE {condition}")

# 修改数据
def update_data(conn, table_name, condition, new_values):
    set_str = ", ".join([f"{column} = '{new_value}'" for column, new_value in new_values.items()])
    conn.execute(f"UPDATE {table_name} SET {set_str} WHERE {condition}")

# 查询数据
def select_data(conn, table_name, condition):
    conn.execute(f"SELECT * FROM {table_name} WHERE {condition}")
    return conn.fetchall()

# 示例数据库
conn = sqlite3.connect("example.db")

# 创建表
create_table(conn, "users", ["name", "age", "gender"])

# 插入数据
insert_data(conn, "users", {"name": "John", "age": 30, "gender": "male"})

# 删除数据
delete_data(conn, "users", "age < 30")

# 修改数据
update_data(conn, "users", "name = 'John'", {"age": 31, "gender": "female"})

# 查询数据
select_data(conn, "users", "age > 25")

conn.close()
```
