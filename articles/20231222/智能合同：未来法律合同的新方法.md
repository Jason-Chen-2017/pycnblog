contract Trade {
    address public buyer;
    address public seller;
    uint public price = 100;
    bool public payment = false;
    bool public refund = false;
    bool public delivery = false;

    event PaymentMade(bool payment, bool refund);
    event DeliveryMade(bool delivery, bool refund);

    function trade(address _seller) public payable {
        seller = _seller;
        buyer = msg.sender;
        payment = msg.value >= price;
        emit PaymentMade(payment, refund);
    }

    function deliverGoods() public {
        delivery = true;
        emit DeliveryMade(delivery, refund);
    }

    function requestRefund() public {
        refund = true;
    }
}
