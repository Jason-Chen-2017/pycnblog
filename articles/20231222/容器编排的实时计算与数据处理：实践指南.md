```python
from kubernetes import client, config

# 加载 kubernetes 配置
config.load_kube_config()

# 创建实时计算和数据处理系统
def create_real_time_data_processing_system():
    # 创建一个名为 "real-time-data-processing" 的 Namespace
    namespace = client.V1Namespace(metadata=client.V1ObjectMeta(name="real-time-data-processing"))
    client.CoreV1Api().create_namespace(namespace)

    # 创建一个名为 "real-time-data-processing-controller" 的 Deployment
    deployment = client.V1Deployment(
        api_version="apps/v1",
        kind="Deployment",
        metadata=client.V1ObjectMeta(name="real-time-data-processing-controller"),
        spec=client.V1DeploymentSpec(
            replicas=3,
            selector={"matchLabels": {"app": "real-time-data-processing-controller"}},
            template=client.V1PodTemplateSpec(
                metadata=client.V1ObjectMeta(labels={"app": "real-time-data-processing-controller"}),
                spec=client.V1PodSpec(
                    containers=[
                        client.V1Container(
                            name="real-time-data-processing-controller",
                            image="real-time-data-processing-controller:latest",
                            ports=[8080],
                        )
                    ]
                )
            )
        )
    )
    client.AppsV1Api().create_deployment(namespace, deployment)

    # 创建一个名为 "real-time-data-processing-service" 的 Service
    service = client.V1Service(
        api_version="v1",
        kind="Service",
        metadata=client.V1ObjectMeta(name="real-time-data-processing-service"),
        spec=client.V1ServiceSpec(
            selector={"app": "real-time-data-processing-controller"},
            ports=[client.V1ServicePort(port=8080)],
            type="ClusterIP"
        )
    )
    client.CoreV1Api().create_namespaced_service(namespace, service)

# 创建实时计算和数据处理系统
create_real_time_data_processing_system()
```

在上述代码中，我们首先加载了 kubernetes 配置，然后创建了一个名为 "real-time-data-processing" 的 Namespace。接着，我们创建了一个名为 "real-time-data-processing-controller" 的 Deployment，该 Deployment 包含了三个实例的实时计算和数据处理控制器容器。最后，我们创建了一个名为 "real-time-data-processing-service" 的 Service，该 Service 用于暴露实时计算和数据处理控制器的端口。
