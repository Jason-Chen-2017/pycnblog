```python
import numpy as np

def quantum_gate(state, gate):
    if gate == 'X':
        state = np.array([[0, 1], [1, 0]]) @ state
    elif gate == 'Y':
        state = np.array([[0, -1j], [1j, 0]]) @ state
    elif gate == 'Z':
        state = np.array([[1, 0], [0, -1]]) @ state
    return state

def quantum_entanglement(state, entanglement_gates):
    for gate in entanglement_gates:
        state = quantum_gate(state, gate)
    return state

state = np.array([1, 0])
entanglement_gates = ['X', 'Y', 'Z']
entangled_state = quantum_entanglement(state, entanglement_gates)
print(entangled_state)
```
