```python
import numpy as np
import matplotlib.pyplot as plt

# 输入函数
def input_function(t):
    return np.sin(t)

# 不定积分函数
def definite_integral(t):
    return np.trapz(input_function(t), t)

# 时间轴

# 输出

# 绘制

plt.xlabel('Time')
plt.ylabel('Output')
plt.title('System Output')
plt.show()
```
