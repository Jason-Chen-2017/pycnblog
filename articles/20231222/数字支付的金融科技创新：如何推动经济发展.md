## 1.Python实现AES对称加密和解密

```python
import hashlib
import base64

# 对称加密算法AES
def aes_encrypt(data, key):
    key = hashlib.sha256(key.encode()).digest()
    iv = hashlib.sha256(key).digest()
    cipher = AES.new(key, AES.MODE_CBC, iv)
    ciphertext = cipher.encrypt(data)
    return base64.b64encode(ciphertext).decode()

def aes_decrypt(ciphertext, key):
    key = hashlib.sha256(key.encode()).digest()
    iv = hashlib.sha256(key).digest()
    cipher = AES.new(key, AES.MODE_CBC, iv)
    decrypt_data = cipher.decrypt(base64.b64decode(ciphertext))
    return decrypt_data
```

上述代码实现了AES对称加密和解密的具体实现。首先，我们使用SHA256算法对密钥进行哈希处理，得到加密和解密的密钥。然后，我们使用AES算法进行加密和解密操作。最后，我们使用base64编码对加密后的数据进行编码，以便在传输时不会被破坏。

# 生成RSA密钥对
key = RSA.generate(2048)
private_key = key.export_key()
public_key = key.publickey().export_key()

# 非对称加密算法RSA
def rsa_encrypt(data, public_key):
    cipher = PKCS1_OAEP.new(public_key)
    ciphertext = cipher.encrypt(data)
    return ciphertext

def rsa_decrypt(ciphertext, private_key):
    private_key = RSA.import_key(private_key)
    decrypt_data = PKCS1_OAEP.new(private_key).decrypt(ciphertext)
    return decrypt_data
```

from Crypto.PublicKey import ECC
from Crypto.Signature import DSS
from Crypto.Hash import SHA256

# 生成ECDSA密钥对
key = ECC.generate(curve=ECC.SECP256k1())
private_key = key.export_key()
public_key = key.public_key().export_key()

# 数字签名算法ECDSA
def ecdsa_sign(data, private_key):
    hash_data = SHA256.new(data.encode())
    signer = DSS.new(key, 'fips-186-3')
    signature = signer.sign(hash_data)
    return signature

def ecdsa_verify(data, signature, public_key):
    hash_data = SHA256.new(data.encode())
    verifier = DSS.new(public_key, 'fips-186-3')
    try:
        verifier.verify(hash_data, signature)
        return True
    except:
        return False
```
