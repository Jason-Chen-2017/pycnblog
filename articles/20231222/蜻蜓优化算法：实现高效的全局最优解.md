```python
import random

def firefly_algorithm(f, x_min, x_max, n_fireflies, n_iterations):
    # 初始化蜻蜓群
    fireflies = [np.random.uniform(x_min, x_max, size=len(x_min)) for _ in range(n_fireflies)]

    # 计算适应度
    fitness = [f(x) for x in fireflies]

    # 主循环
    for _ in range(n_iterations):
        for i in range(n_fireflies):
            # 更新蜻蜓的位置
            for j in range(n_fireflies):
                if fitness[j] > fitness[i]:
                    r = np.linalg.norm(fireflies[i] - fireflies[j])
                    beta = 1 / (1 + r ** 2)
                    fireflies[i] += beta * (fireflies[j] - fireflies[i]) + random.uniform(-1, 1)

            # 更新适应度
            fitness[i] = f(fireflies[i])

            # 更新最佳解
            if fitness[i] < min(fitness):
                best_solution = fireflies[i]

    return best_solution
```
