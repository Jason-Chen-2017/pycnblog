import numpy as np

def two_body_problem(G, m1, m2, r1, r2, v1, v2):
    r = r1 - r2
    v = v1 - v2
    dr = r.copy()
    dv = v.copy()
    dt = 1
    t = 0
    while np.linalg.norm(dr) > 1e-6:
        dr = dr - dt * (G * m1 * m2 * dr / np.linalg.norm(dr)**3 + dv)
        dv = dv - dt * (-G * m1 * m2 * v / np.linalg.norm(dr)**3 - (G * m1 * m2 / np.linalg.norm(dr)**3 + np.dot(dr, dv)) * dr / np.linalg.norm(dr))
        t += dt
    return t, dr, dv
