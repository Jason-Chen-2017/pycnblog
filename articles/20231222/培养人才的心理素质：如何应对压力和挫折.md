        self.self_adjustment = SelfAdjustment()
        self.communication = Communication()

    def manage_pressure(self):
        start_time = time.time()
        end_time = start_time + self.time_management.set_time()
        while time.time() < end_time:
            self.self_adjustment.adjust()
            self.communication.communicate()

if __name__ == "__main__":

    pressure_management.manage_pressure()
