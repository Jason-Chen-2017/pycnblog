## 4.2 深度学习代码实例

在这个例子中，我们将使用Python的TensorFlow库来实现一个简单的神经网络模型：

```python
import tensorflow as tf

# 定义神经网络
class Net(tf.keras.Model):
    def __init__(self):
        super(Net, self).__init__()
        self.dense1 = tf.keras.layers.Dense(64, activation='relu')
        self.dense2 = tf.keras.layers.Dense(32, activation='relu')
        self.dense3 = tf.keras.layers.Dense(10, activation='softmax')

    def call(self, x):
        x = self.dense1(x)
        x = self.dense2(x)
        return self.dense3(x)

# 加载数据
(X_train, y_train), (X_test, y_test) = tf.keras.datasets.mnist.load_data()
X_train = X_train.reshape(-1, 784).astype('float32') / 255
X_test = X_test.reshape(-1, 784).astype('float32') / 255

# 训练模型
model = Net()
model.compile(optimizer=tf.keras.optimizers.Adam(), loss='sparse_categorical_crossentropy', metrics=['accuracy'])
model.fit(X_train, y_train, epochs=10)

# 评估模型
loss, accuracy = model.evaluate(X_test, y_test)
print("Accuracy:", accuracy)
```

## 4.4 自然语言处理代码实例

在这个例子中，我们将使用Python的NLTK库来实现一个简单的文本分类模型：

```python
import nltk
from nltk.classify import NaiveBayesClassifier
from nltk.corpus import stopwords

# 加载数据
data = [
    ("这是一篇关于人工智能的文章", "technology"),
    ("这是一篇关于健康的文章", "health"),
    ("这是一篇关于旅行的文章", "travel"),
]

# 预处理数据
def preprocess(text):
    words = nltk.word_tokenize(text)
    words = [word.lower() for word in words if word not in stopwords.words('english')]
    return words

# 训练模型
featuresets = [(preprocess(text), category) for (text, category) in data]
classifier = NaiveBayesClassifier.train(featuresets)

# 评估模型
test_text = "这是一篇关于人工智能的文章"
test_features = preprocess(test_text)
predicted_category = classifier.classify(test_features)
print("Predicted Category:", predicted_category)
```
