import networkx as nx

def topological_sort(graph):
    return list(nx.topological_sort(graph))

def main():
    tasks = ['A', 'B', 'C', 'D', 'E', 'F', 'G']
    dependencies = [(None, 'A'), ('None', 'B'), ('A', 'C'), ('B', 'D'), ('C', 'E'), ('D', 'F'), ('E', 'G')]
    graph = nx.DiGraph()
    for task, dep in zip(tasks, dependencies):
        graph.add_node(task)
        if dep:
            graph.add_edge(dep, task)
    result = topological_sort(graph)
    print(result)

if __name__ == '__main__':
    main()
