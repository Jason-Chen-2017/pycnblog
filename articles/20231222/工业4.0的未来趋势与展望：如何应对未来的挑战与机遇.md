import numpy as np
import matplotlib.pyplot as plt

# 模拟信号生成
def generate_signal(freq, amplitude, sample_rate):
    t = np.linspace(0, 1, sample_rate, False)
    signal = amplitude * np.sin(2 * np.pi * freq * t)
    return signal

# 采样
def sample(signal, sample_rate):
    samples = np.linspace(0, len(signal) / sample_rate, len(signal), False)
    sample_values = interp1d(signal, samples, kind='linear')(samples)
    return sample_values

# 量化
def quantize(sample_values, bits):
    quantized_values = np.round(sample_values * (2 ** bits), decimals=0)
    return quantized_values

# 信号生成、采样、量化
freq = 5
amplitude = 1
sample_rate = 1000
signal = generate_signal(freq, amplitude, sample_rate)
samples = sample(signal, sample_rate)
quantized_values = quantize(samples, 8)

plt.title('Original Signal')
plt.xlabel('Time')
plt.ylabel('Amplitude')

plt.figure()

### 4.2 机器学习：线性回归

在机器学习中，我们需要通过数据学习模式，实现模式的捕捉和预测。具体的代码实例如下：

```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error

# 数据生成
def generate_data(n_samples):
    X = np.random.rand(n_samples, 1)
    y = 2 * X + np.random.randn(n_samples, 1)
    return X, y

# 数据分割
def split_data(X, y, train_ratio):
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=1-train_ratio, random_state=0)
    return X_train, X_test, y_train, y_test

# 线性回归
def linear_regression(X, y):
    model = LinearRegression()
    model.fit(X, y)
    return model

# 数据生成、分割、线性回归
n_samples = 100
X, y = generate_data(n_samples)
X_train, X_test, y_train, y_test = split_data(X, y, 0.8)

model = linear_regression(X_train, y_train)

# 预测与评估
y_pred = model.predict(X_test)
mse = mean_squared_error(y_test, y_pred)

# 结果绘制
plt.figure()
plt.scatter(X_test, y_test, color='blue', label='True')
plt.plot(X_test, y_pred, color='red', label='Predict')
plt.title('Linear Regression')
plt.xlabel('X')
plt.ylabel('y')
plt.legend()

plt.figure()
plt.plot(mse, color='red')
plt.title('Mean Squared Error')
plt.xlabel('Iteration')
plt.ylabel('Error')

plt.show()
```

在这个例子中，我们首先生成了一组数据，然后通过线性回归算法学习了数据中的模式，最后通过预测来评估模型的效果。
