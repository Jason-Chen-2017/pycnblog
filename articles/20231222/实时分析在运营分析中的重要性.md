## 4.1 代码实例

我们以一个简单的实时流量监控案例为例，来详细解释实时分析在运营分析中的具体应用。

```python
import socket
import time
import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression

# 创建socket对象
s = socket.socket()
s.bind(('127.0.0.1', 9999))
s.listen(5)

# 数据收集与存储
data = []
while True:
    conn, addr = s.accept()
    print('Connect by', addr)
    while True:
        data = conn.recv(1024)
        if not data:
            break
        data = data.decode()
        data_list = data.split(',')
        data_list = [float(x) for x in data_list]
        data.append(data_list)
    conn.close()

# 数据处理与分析
df = pd.DataFrame(data[1:], columns=['时间', '流量'])
df['时间'] = pd.to_datetime(df['时间'])
df.set_index('时间', inplace=True)
df.resample('1H').mean()

# 选择算法
model = LinearRegression()

# 训练模型
X = df.index.values.reshape(-1, 1)
y = df['流量'].values
model.fit(X, y)

# 评估模型
y_pred = model.predict(X)
print('实际值：', y)
print('预测值：', y_pred)
```
