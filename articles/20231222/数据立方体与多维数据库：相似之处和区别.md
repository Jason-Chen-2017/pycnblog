# 4.具体代码实例和详细解释说明
## 4.1 数据立方体的代码实例
以下是一个简单的数据立方体的代码实例：

```python
from pandas import DataFrame

# 创建一个数据立方体
class Cube:
    def __init__(self, data):
        self.data = data

    def query(self, dimensions, measures, level):
        result = []
        for row in self.data[dimensions]:
            value = row[measures]
            result.append((level, value))
        return result

# 加载数据
data = DataFrame({
    'Product': ['A', 'B', 'C'],
    'Year': [2015, 2015, 2016],
    'Sales': [100, 200, 150]
})

# 创建一个数据立方体
cube = Cube(data)

# 定义查询
dimensions = ['Product', 'Year']
measures = 'Sales'
level = 'Year'

# 执行查询
result = cube.query(dimensions, measures, level)
print(result)
```
