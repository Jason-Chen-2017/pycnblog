```
import matplotlib.pyplot as plt

# 生成随机信号
t = np.linspace(0, 1, 1000)
signal = np.sin(2 * np.pi * 5 * t)

# 生成脉冲信号
pulse = np.zeros(len(t))
pulse[::100] = 1

# 数字信号处理
digital_signal = signal * pulse

# 绘制信号
plt.figure()
plt.plot(t, signal, label='原始信号')
plt.plot(t, digital_signal, label='数字信号')
plt.legend()
plt.show()
```
