# 4. 具体代码实例和详细解释说明
# 4.1 边缘计算代码实例
在这个代码实例中，我们将使用 Python 编程语言实现一个简单的边缘计算应用，即计算边缘设备的温度和湿度值，并将结果存储到本地文件中。

```python
import time
import json

# 模拟边缘设备的温度和湿度值
def get_temperature_humidity():
    temperature = 25 + 0.5 * time.time()
    humidity = 45 + 0.5 * time.time()
    return temperature, humidity

# 存储温度和湿度值到文件
def save_temperature_humidity(temperature, humidity):
    data = {
        'temperature': temperature,
        'humidity': humidity,
        'timestamp': time.time()
    }
    with open('temperature_humidity.json', 'a') as f:
        f.write(json.dumps(data) + '\n')

if __name__ == '__main__':
    while True:
        temperature, humidity = get_temperature_humidity()
        save_temperature_humidity(temperature, humidity)
        time.sleep(1)
```

# 4.2 物联网数据分析代码实例
在这个代码实例中，我们将使用 Python 编程语言实现一个简单的物联网数据分析应用，即从边缘设备获取温度和湿度值，并计算平均值。

```python
import json
import requests

# 获取边缘设备的温度和湿度值
def get_temperature_humidity():
    url = 'http://edge_device_ip/temperature_humidity.json'
    response = requests.get(url)
    data = json.loads(response.text)
    return data['temperature'], data['humidity']

# 计算温度和湿度值的平均值
def calculate_average(temperatures, humidities):
    avg_temperature = sum(temperatures) / len(temperatures)
    avg_humidity = sum(humidities) / len(humidities)
    return avg_temperature, avg_humidity

if __name__ == '__main__':
    temperatures = []
    humidities = []
    while len(temperatures) < 100:
        temperature, humidity = get_temperature_humidity()
        temperatures.append(temperature)
        humidities.append(humidity)
    avg_temperature, avg_humidity = calculate_average(temperatures, humidities)
    print(f'平均温度: {avg_temperature}, 平均湿度: {avg_humidity}')
```
