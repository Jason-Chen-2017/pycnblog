```python
import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

# 加载数据
data = pd.read_csv('city_data.csv')

# 分析关键因素
factors = data[['GDP', 'justice_rate', 'income_gini']]
factors.describe()

# 制定策略
strategies = data[['education', 'healthcare', 'public_security']]
strategies.describe()

# 评估策略效果
X_train, X_test, y_train, y_test = train_test_split(factors, strategies, test_size=0.2, random_state=42)
model = LinearRegression()
model.fit(X_train, y_train)
score = model.score(X_test, y_test)
print('R^2:', score)

# 监控和调整
monitor = pd.DataFrame(columns=['date', 'strategy', 'effect'])
for strategy in strategies.columns:
    for i in range(10):
        X_train, X_test, y_train, y_test = train_test_split(factors, strategies[strategy], test_size=0.2, random_state=42)
        model = LinearRegression()
        model.fit(X_train, y_train)
        score = model.score(X_test, y_test)
        monitor = monitor.append({'date': i, 'strategy': strategy, 'effect': score}, ignore_index=True)
```
