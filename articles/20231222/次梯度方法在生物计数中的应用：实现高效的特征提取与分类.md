    return np.sum((data - np.dot(theta, data.T)) ** 2)

# 初始化模型参数

# 设置学习率

# 设置迭代次数

# 使用次梯度方法进行训练
for i in range(iterations):
    # 计算梯度
    gradient = 2 * np.dot(data, data.T - np.dot(theta, data.T))

    # 更新模型参数
    theta = theta - learning_rate * gradient

    # 打印当前迭代的损失值
    print('Iteration {}: Loss: {}'.format(i, loss_function(theta)))

# 打印最终的模型参数
print('Final model parameters:', theta)
