```python
from kafka import KafkaProducer
from kafka import KafkaConsumer
import json

# 实时数据收集
def collect_realtime_data():
    producer = KafkaProducer(bootstrap_servers='localhost:9092')
    for i in range(10):
        data = {'value': i}
        producer.send('realtime_data_topic', value=json.dumps(data).encode('utf-8'))

# 实时数据处理
def process_realtime_data():
    consumer = KafkaConsumer('realtime_data_topic', bootstrap_servers='localhost:9092', value_deserializer=lambda m: json.loads(m.decode('utf-8')))
    for msg in consumer:
        value = msg.value['value']
        print('Received:', value)

# 实时数据分析
def analyze_realtime_data():
    consumer = KafkaConsumer('realtime_data_topic', bootstrap_servers='localhost:9092', value_deserializer=lambda m: json.loads(m.decode('utf-8')))
    for msg in consumer:
        value = msg.value['value']
        print('Analyzed:', value)

if __name__ == '__main__':
    collect_realtime_data()
    process_realtime_data()
    analyze_realtime_data()
```
