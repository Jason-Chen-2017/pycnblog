```python
# 创建优化问题
prob = LpProblem('CanReGenScheduling', LpMinimize)

# 创建变量
x1 = LpVariable('x1', lowBound=0)
x2 = LpVariable('x2', lowBound=0)

# 添加约束
prob += x1 + x2 == 100
prob += x1 <= 50
prob += x2 <= 50

# 求解优化问题

# 输出结果

```
