```python
# 计算被评审者的总评分
def calculate_total_score(reviewers, scorers):
    total_scores = {}
    for reviewer in reviewers:
        total_scores[reviewer] = sum(scorers[reviewer])
    return total_scores

# 打印被评审者的总评分
print(calculate_total_score(reviewers, scorers))
```
