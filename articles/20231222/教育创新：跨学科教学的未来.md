```python
# 定义不同学科领域的知识和技能
natural_science = {'物理学': ['力学', '热学', '电磁学'], '化学学': ['化学反应', '化学定律'], '生物学': ['生物系统', '遗传学']}
humanities = {'历史学': ['世界历史', '中国历史', '古典文化'], '文学': ['诗歌', '小说', '剧本'], '艺术': ['绘画', '雕塑', '音乐']}
social_science = {'政治学': ['政治体系', '国际关系'], '经济学': ['宏观经济', '微观经济'], '法学': ['法律制度', '法律理论']}

# 找出两个或多个学科领域之间的联系和关系
def find_relations(knowledge):
    relations = []
    for key in knowledge.keys():
        relations.append(key)
    return relations

# 将不同学科的知识和技能相互融合，形成一个新的知识体系
def fusion_knowledge(natural_science, humanities, social_science):
    fusion_knowledge = {}
    relations = find_relations(natural_science) + find_relations(humanities) + find_relations(social_science)
    for relation in relations:
        if relation in natural_science:
            fusion_knowledge[relation] = natural_science[relation]
        elif relation in humanities:
            fusion_knowledge[relation] = humanities[relation]
        elif relation in social_science:
            fusion_knowledge[relation] = social_science[relation]
    return fusion_knowledge

# 打印新的知识体系
print(fusion_knowledge(natural_science, humanities, social_science))
```
