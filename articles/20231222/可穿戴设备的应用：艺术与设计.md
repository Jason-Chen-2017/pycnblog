# 定义一个函数，用于计算两点间的距离
def distance(p1, p2):
    return math.sqrt((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2)

# 定义一个函数，用于计算两个线段之间的距离
def segment_distance(p1, p2, p3, p4):
    # 计算向量
    v1 = (p2[0] - p1[0], p2[1] - p1[1])
    v2 = (p4[0] - p3[0], p4[1] - p3[1])

    # 计算叉积
    cross = v1[0] * v2[1] - v1[1] * v2[0]

    # 计算距离
