# 4.具体代码实例和详细解释说明

## D3.js

以下是一个使用 D3.js 创建简单柱状图的代码示例：

```javascript
// 加载数据
d3.csv("data.csv").then(function(data) {
  // 数据处理
  var x = d3.scaleBand().domain(data.map(function(d) {
    return d.category;
  })).range([0, 300]).padding(0.1);

  var y = d3.scaleLinear().domain([0, d3.max(data, function(d) {
    return d.value;
  })]).range([0, 200]);

  // 创建可视化
  var svg = d3.select("body").append("svg")
    .attr("width", 400)
    .attr("height", 300);

  svg.selectAll("rect")
    .data(data)
    .enter()
    .append("rect")
    .attr("x", function(d) {
      return x(d.category);
    })
    .attr("y", function(d) {
      return y(d.value);
    })
    .attr("width", x.bandwidth())
    .attr("height", function(d) {
      return 200 - y(d.value);
    });
});
```

## Highcharts

以下是一个使用 Highcharts 创建简单柱状图的代码示例：

```javascript
// 加载数据
Highcharts.getJSON("data.json").then(function(data) {
  // 数据处理
  var categories = data.map(function(d) {
    return d.category;
  });

  var values = data.map(function(d) {
    return d.value;
  });

  // 创建图表
  Highcharts.chart("container", {
    chart: {
      type: "column"
    },
    title: {
      text: "Simple Column Chart"
    },
    xAxis: {
      categories: categories
    },
    yAxis: {
      min: 0,
      title: {
        text: "Value"
      }
    },
    series: [{
      name: "Value",
      data: values
    }]
  });
});
```

## Plotly

以下是一个使用 Plotly 创建简单柱状图的代码示例：

```python
import plotly.graph_objs as go

# 加载数据
data = [
  {"category": "A", "value": 10},
  {"category": "B", "value": 20},
  {"category": "C", "value": 30},
]

# 创建可视化
fig = go.Figure(data=[go.Bar(x=data["category"], y=data["value"])])

# 更新可视化
fig.update_layout(title_text="Simple Bar Chart")

# 显示可视化
fig.show()
```
