```python
def sliding_average(data_stream, window_size):
    window = deque(maxlen=window_size)
    total = 0
    count = 0
    for data in data_stream:
        window.append(data)
        total += data
        count += 1
        if count >= window_size:
            yield total / window_size
            total -= window.popleft()
            count -= 1

```
