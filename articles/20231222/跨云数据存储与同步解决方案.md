```python
import hashlib
import threading
import time

class CrossCloudStorage:
    def __init__(self, cloud_storage_list):
        self.cloud_storage_list = cloud_storage_list
        self.data_chunks = {}

    def hash_data(self, data):
        return hashlib.sha256(data.encode()).hexdigest()

    def split_data(self, data):
        hash_data = self.hash_data(data)
        if hash_data in self.data_chunks:
            return self.data_chunks[hash_data]
        else:
            chunk_list = []
            chunk_size = 1024
            start = 0
            while start < len(data):
                end = start + chunk_size
                chunk = data[start:end]
                chunk_list.append(chunk)
                start += chunk_size
            self.data_chunks[hash_data] = chunk_list
            return chunk_list

    def store_data(self, chunk, cloud_storage):
        cloud_storage.store(chunk)

    def sync_data(self, cloud_storage_list):
        for cloud_storage in cloud_storage_list:
            for chunk in cloud_storage.list_chunks():
                if chunk not in self.data_chunks:
                    self.data_chunks[chunk] = []
                self.data_chunks[chunk].extend(cloud_storage.get_chunk(chunk))

    def get_data(self, chunk):
        if chunk in self.data_chunks:
            return ''.join(self.data_chunks[chunk])
        else:
            return None

class CloudStorage:
    def __init__(self, storage_id):
        self.storage_id = storage_id

    def store(self, data):
        pass

    def list_chunks(self):
        return []

    def get_chunk(self, chunk):
        return []

# 示例使用
cloud_storage_list = [CloudStorage('google'), CloudStorage('aws'), CloudStorage('azure')]
cross_cloud_storage = CrossCloudStorage(cloud_storage_list)
data = 'this is a sample data'
data_chunks = cross_cloud_storage.split_data(data)
for chunk in data_chunks:
    for cloud_storage in cloud_storage_list:
        cross_cloud_storage.store_data(chunk, cloud_storage)
cross_cloud_storage.sync_data(cloud_storage_list)
result = cross_cloud_storage.get_data(cross_cloud_storage.hash_data(data))
print(result)
```
