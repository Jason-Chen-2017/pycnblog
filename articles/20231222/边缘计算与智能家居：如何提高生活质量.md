```python
import time
import requests

# 设备生成环境数据
def generate_environment_data():
    data = {'temperature': 25, 'humidity': 50, 'pressure': 1013}
    return data

# 设备将环境数据传输到边缘计算节点
def transfer_environment_data(data):
    url = 'http://edge-node/data'
    response = requests.post(url, json=data)
    return response

# 边缘计算节点对环境数据进行处理
def process_environment_data(data):
    temperature = data['temperature']
    humidity = data['humidity']
    pressure = data['pressure']
    print(f'temperature: {temperature}, humidity: {humidity}, pressure: {pressure}')

# 边缘计算节点将处理结果传输回设备
def transfer_processed_data(data):
    url = 'http://device/data'
    response = requests.post(url, json=data)
    return response

if __name__ == '__main__':
    data = generate_environment_data()
    response = transfer_environment_data(data)
    process_environment_data(response.json())
    processed_data = {'temperature': temperature, 'humidity': humidity, 'pressure': pressure}
    response = transfer_processed_data(processed_data)
```

4. `transfer_processed_data(data)` 函数：这个函数用于将边缘计算节点对环境数据的处理结果传输回设备。
