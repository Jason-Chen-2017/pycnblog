```python
def classify_and_index(resources_dir, categories):
    index = {}
    for category in categories:
        index[category] = []
    for root, dirs, files in os.walk(resources_dir):
        for file in files:
```

```python
import hashlib
import sqlite3

def register_student(student_id, password, name, email):
    conn = sqlite3.connect('students.db')
    cursor = conn.cursor()
    cursor.execute('CREATE TABLE IF NOT EXISTS students (id, password, name, email)')
    password_hash = hashlib.sha256(password.encode()).hexdigest()
    cursor.execute('INSERT INTO students (id, password, name, email) VALUES (?, ?, ?, ?)', (student_id, password_hash, name, email))
    conn.commit()
    conn.close()
    return student_id

def verify_student(student_id, password):
    conn = sqlite3.connect('students.db')
    cursor = conn.cursor()
    cursor.execute('SELECT id, password FROM students WHERE id = ?', (student_id,))
    row = cursor.fetchone()
    conn.close()
    if row is None:
        return False
    password_hash = hashlib.sha256(password.encode()).hexdigest()
    return row[1] == password_hash
```

```python
import json

def evaluate_test(student_id, test_id, answers):
    conn = sqlite3.connect('tests.db')
    cursor = conn.cursor()
    cursor.execute('CREATE TABLE IF NOT EXISTS test_results (student_id, test_id, score)')
    correct_answers = cursor.execute('SELECT answers FROM tests WHERE test_id = ?', (test_id,)).fetchone()[0]
    score = 0
    for question in json.loads(correct_answers):
        if answers[question['id']] == question['answer']:
            score += 1
    cursor.execute('INSERT INTO test_results (student_id, test_id, score) VALUES (?, ?, ?)', (student_id, test_id, score))
    conn.close()
    return score
```
