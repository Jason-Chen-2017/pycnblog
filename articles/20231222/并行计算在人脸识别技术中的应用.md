### 4.2 任务并行的具体代码实例

```python
import numpy as np
import multiprocessing

def face_detection(image):
    # 人脸检测
    face = detect_face(image)
    if face is None:
        return None

    # 人脸特征提取
    features = extract_features(face)

    # 人脸识别
    person = recognize_face(features)
    return person

def main():
    images = load_images()
    pool = multiprocessing.Pool(processes=len(images))
    persons = pool.map(face_detection, images)
    pool.close()
    pool.join()
    print("人脸识别结果：", persons)

if __name__ == "__main__":
    main()
```
