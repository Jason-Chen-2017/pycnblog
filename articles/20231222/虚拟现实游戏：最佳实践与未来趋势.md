class HMD:

    def update_head_pose(self, head_rotation, head_translation):
        self.head_pose = np.dot(head_rotation, self.head_pose) + head_translation

    def project_image(self, image, fov, aspect_ratio):
        # 计算图像投影矩阵
        projection_matrix = self.calculate_projection_matrix(fov, aspect_ratio)
        # 投影图像
        projected_image = np.dot(projection_matrix, image)
        return projected_image

    def calculate_projection_matrix(self, fov, aspect_ratio):
        # 计算投影矩阵
        projection_matrix = np.array([[np.tan(fov * np.pi / 180) / aspect_ratio, 0, 0],
