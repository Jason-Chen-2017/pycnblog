import qiskit

# 创建二级系统
qc = qiskit.QuantumCircuit(2)

# 初始化系统状态
qc.initialize([1, 0], 0)
qc.initialize([0, 1], 1)

# 测量系统状态
qc.measure([0, 1], [0, 1])

# 执行测量
aer_sim = qiskit.Aer.get_backend('aer_simulator')
qobj = qiskit.execute(qc, aer_sim)
result = qobj.result()

# 获取测量结果
counts = result.get_counts()
print(counts)
