    return v0 + (tf-t0)*np.mean(v0, v)

    return v0 + 2*(tf-t0)*np.mean(v0, v)

def dynamic_planning(t0, tf, v0):
    t = np.linspace(t0, tf, 100)
    v = np.zeros(len(t))
    v[0] = v0
    for i in range(1, len(t)):
        v[i] = min(v[i-1], (v[i-1] + a(t[i-1], v0, t0, t[i]))/(1+t[i]-t[i-1]))
