```python
})

# 查询指定客户信息
def get_customer_info(customer_id):
    return customer_info[customer_info['customer_id'] == customer_id]

# 更新客户信息
def update_customer_info(customer_id, **kwargs):
    update_data = {}
    for key, value in kwargs.items():
        update_data[key] = value
    customer_info.loc[customer_info['customer_id'] == customer_id, :] = update_data

# 删除客户信息
def delete_customer_info(customer_id):
    customer_info = customer_info[customer_info['customer_id'] != customer_id]
```
