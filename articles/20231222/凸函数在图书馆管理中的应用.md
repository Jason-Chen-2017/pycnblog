```python
def cost_function(x):
    return x**2 + 2*x + 1

def demand_function(x):
    return x + 1

    x_L = 0
    x_R = 10
    epsilon = 1e-6
    while x_R - x_L > epsilon:
        x_M = (x_L + x_R) / 2
        if demand_function(x_M) >= d:
            x_L = x_M
        else:
            x_R = x_M
    return x_M

d = 5
x_star = find_x_star(d)

print("最小化成本的值：", cost_function(x_star))
```
