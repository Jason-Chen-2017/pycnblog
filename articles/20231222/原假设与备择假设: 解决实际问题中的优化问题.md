```python
import numpy as np

def original_hypothesis(x):
    return x**2

def alternative_hypothesis(x):
    return (x-1)**2

def evaluate_performance(hypothesis, x):
    return hypothesis(x)

x = np.linspace(-10, 10, 100)
original_performance = np.array([evaluate_performance(original_hypothesis, x[i]) for i in range(len(x))])
alternative_performance = np.array([evaluate_performance(alternative_hypothesis, x[i]) for i in range(len(x))])

best_original_x = x[np.argmin(original_performance)]
best_alternative_x = x[np.argmin(alternative_performance)]

print("Best original x:", best_original_x)
print("Best alternative x:", best_alternative_x)
```
