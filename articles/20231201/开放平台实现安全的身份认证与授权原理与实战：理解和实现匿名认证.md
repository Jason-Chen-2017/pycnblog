import hashlib

# 用户注册
def register(username, password):
    # 生成密钥
    key = hashlib.sha256(password.encode()).hexdigest()
    # 存储用户信息
    user_info = {
        'username': username,
        'key': key
    }
    # 返回用户信息
    return user_info

# 用户登录
def login(username, password, user_info):
    # 使用密钥加密身份信息
    encrypted_info = hashlib.sha256((username + password).encode()).hexdigest()
    # 与用户信息进行比较
    if user_info['username'] == username and user_info['key'] == encrypted_info:
        return True
    else:
        return False

# 授权管理
def authorization(user_info):
    # 确定用户可以访问哪些资源和执行哪些操作
    # 这里是一个示例，实际应用中需要根据具体需求进行调整
    if user_info['username'] == 'admin':
        return {
            'resources': ['all'],

        }
    else:
        return {
            'resources': ['limited'],
            'operations': ['limited']
        }
