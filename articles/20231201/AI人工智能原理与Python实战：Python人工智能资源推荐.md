```python
import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

# 加载数据
data = pd.read_csv('data.csv')

# 计算用户-物品交互矩阵
user_item_matrix = data.pivot_table(index='user_id', columns='item_id', values='rating', fill_value=0)

# 计算协同过滤
similarity_matrix = cosine_similarity(user_item_matrix)

# 计算内容基于的推荐
content_matrix = data.pivot_table(index='user_id', columns='item_id', values='content', fill_value=0)

# 计算混合推荐
hybrid_matrix = (alpha * similarity_matrix + (1 - alpha) * content_matrix)

# 推荐物品
recommended_items = hybrid_matrix.sum(axis=1).sort_values(ascending=False)
```

在上述代码中，我们首先加载了数据，然后计算了用户-物品交互矩阵、协同过滤、内容基于的推荐和混合推荐。最后，我们根据混合推荐的结果推荐物品。
