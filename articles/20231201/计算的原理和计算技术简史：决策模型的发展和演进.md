决策模型的核心概念包括决策、状态、动作、奖励、策略等。以下是这些概念的详细解释：

```python
transition_probability = {
    ('state1', 'action1'): 0.7,
    ('state1', 'action2'): 0.3,
    ('state2', 'action1'): 0.6,
    ('state2', 'action2'): 0.4,
    ('state3', 'action1'): 0.5,
    ('state3', 'action2'): 0.5,
}

# 定义动态规划函数
def dynamic_programming(state, action):
    return reward_function.get((state, action), 0) + np.log(transition_probability.get((state, action), 1))

# 计算动态规划表
def dynamic_programming_table(states, actions):
    dp_table = [[0] * len(actions) for _ in states]
    for state in states:
        for action in actions:
            dp_table[state.index(state)][action.index(action)] = dynamic_programming(state, action)
    return dp_table

# 选择最佳动作
def select_best_action(state):
    dp_table = dynamic_programming_table(states, actions)
    best_action = max(actions, key=lambda action: dp_table[state.index(state)][action.index(action)])
    return best_action

# 测试
state = 'state1'
print(select_best_action(state))  # 输出: action1
```
