在图数据分析中，我们需要处理的数据是由节点（nodes）和边（edges）组成的图。节点表示数据实体，如人、商品、文章等，而边表示实体之间的关系。图数据分析的目标是从这些节点和边中抽取有意义的信息，以解决各种问题。

def dfs(G, root):
    visited = set()
    stack = [root]
    while stack:
        node = stack.pop()
        if node not in visited:
            visited.add(node)
            stack.extend(G.neighbors(node))
    return visited

def bfs(G, start):
    visited = set()
    queue = [start]
    while queue:
        node = queue.pop(0)
        if node not in visited:
            visited.add(node)
            queue.extend(G.neighbors(node))
    return visited

# 图的匹配
def max_matching(G):
    matching = set()
    visited = set()
    for node in G.nodes():
        if node not in visited:
            stack = [node]
            while stack:
                node = stack.pop()
                visited.add(node)
                for neighbor in G.neighbors(node):
                    if neighbor not in visited and G[node][neighbor]["weight"] > 0:
                        stack.append(neighbor)
                        if neighbor not in matching:
                            matching.add(neighbor)
    return matching

def scc(G):
    visited = set()
    stack = [node for node in G.nodes()]
    while stack:
        node = stack.pop()
        if node not in visited:
            scc = []
            stack.append(node)
            while stack:
                node = stack.pop()
                visited.add(node)
                scc.append(node)
                for neighbor in G.neighbors(node):
                    if neighbor not in visited:
                        stack.append(neighbor)
