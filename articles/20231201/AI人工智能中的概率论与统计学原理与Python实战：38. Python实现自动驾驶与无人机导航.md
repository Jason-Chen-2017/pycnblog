```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import least_squares

# 定义路径规划函数
def path_planning(x, y, z):
    a = np.array([[1, 0, 0], [0, 1, 0], [0, 0, 1]])
    b = np.array([[x[0]], [y[0]], [z[0]]])
    c = np.array([[x[1]], [y[1]], [z[1]]])
    return a.dot(b) - c

# 定义目标函数
def residuals(params, x, y, z):
    x0, y0, z0 = params
    return path_planning(x, y, z) - np.array([[x0], [y0], [z0]])

# 定义初始参数
params0 = np.array([[0], [0], [0]])

# 定义目标函数的梯度
def jacobian(params, x, y, z):
    a = np.array([[1, 0, 0], [0, 1, 0], [0, 0, 1]])
    return np.vstack((a, -a))

# 定义目标函数的初始梯度
jacobian0 = jacobian(params0, x, y, z)

# 使用最小二乘法求解参数
params, covariance = least_squares(residuals, params0, jacobian=jacobian, jacobian0=jacobian0)

# 绘制路径规划结果
x = np.linspace(x[0], x[1], 100)
y = np.linspace(y[0], y[1], 100)
z = np.linspace(z[0], z[1], 100)
X, Y, Z = np.meshgrid(x, y, z)
plt.contourf(X, Y, Z, path_planning(X, Y, Z), 10, cmap='RdBu')
plt.xlabel('x')
plt.ylabel('y')
plt.zlabel('z')
plt.show()
```
