```
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split

# 加载iris数据集
iris = load_iris()

# 将数据集拆分为训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(iris.data, iris.target, test_size=0.2, n_informative_features=0)

# 进行选择性繁殖
features = X_train.drop(columns=X_test.columns)

for n in range(100):
    # 交叉遗传
    index = np.random.rand(X_train.shape[0])
    parent1 = X_train.iloc[index, :-4]
    parent2 = X_train.iloc[index, 4:]
    child = np.concat(parent1, parent2)
    
    # 选择一定比例的优秀特征
    num_features = int(X_train.shape[0] * 0.8)
    features = features[:num_features]
    
    # 生成新特征
    new_features = np.concat(features, child)
    
    # 更新原始数据
    X_train = np.concat(X_train, new_features)
    y_train = np.concat(y_train, new_features)
```

最后，使用神经网络模型对新生成的特征进行训练，得到高效的决策和预测模型：
