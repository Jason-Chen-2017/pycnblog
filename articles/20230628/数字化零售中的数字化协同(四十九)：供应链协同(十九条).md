```
<button onclick="getInventory()">查询库存</button>

<script>
  function getInventory() {
    const productId = "123";
    const warehouseId = "101";
    const token = "{token}";

    fetch("/inventory/" + productId + "/" + warehouseId + ",?token=" + token)
     .then(response => response.json())
     .then(data => {
        const inventory = data.inventory;
        const message = data.message;

        if (inventory < 0) {
          message = "该商品不存在";
        } else {
          document.getElementById("inventory").innerHTML = inventory + "个";
        }
      });
  }
</script>
```
