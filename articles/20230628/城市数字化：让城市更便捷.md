```python
import requests
import json

class WeatherAPI:
    def __init__(self):
        self.url = "https://api.openweathermap.org/data/2.5/weather?q=%E5%9C%A8%E6%8F%8A&appid=你的APPID&lang=zh_CN"

    def get_weather(self, city):
        response = requests.get(self.url)
        data = json.loads(response.text)
        return data["weather"][0]

    def update_weather(self, city):
        response = requests.get(self.url)
        data = json.loads(response.text)
        weather = data["weather"][0]
        city_weather = weather["description"]
        return city_weather

if __name__ == "__main__":
    api = WeatherAPI()
    city = "北京"
    weather = api.update_weather(city)
    print(weather)
```
