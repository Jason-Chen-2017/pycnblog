    # 如果开启的图像数量超过阈值，门锁自动开启

            image = process_image(image)

            # 检测图像是否包含门锁
            if cv2.countNonZero(image, detection_threshold) > 0:

                break
        else:
            print("门锁未开启，请打开门锁！")

if __name__ == "__main__":
    main()
