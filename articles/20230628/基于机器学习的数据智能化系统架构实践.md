import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers

# 数据预处理
def preprocess_data(data):
    # 去除重复数据
    df = data.drop_duplicates()
    # 缺失数据处理
    df = df.dropna()
    # 特征工程
    features = pd.get_dummies(df[['userID', 'userType', 'user_last_login']])
    # 合并特征
    data = df.join(features)
    return data

# 特征工程
def feature_engineering(data):
    # 用户ID特征
    data['userID_feature'] = data['userID'].apply(lambda x: x.astype(int))
    data['userID_feature'] = data['userID_feature'].astype(str)
    # 用户类型特征
    data['userType_feature'] = data['userType'].astype(str)
    data['userType_feature'] = data['userType_feature'].astype(int)
    # 用户最近登录时间特征
    data['user_last_login_time'] = data['user_last_login_time'].apply(lambda x: x.astype(int))
    data['user_last_login_time'] = data['user_last_login_time'].astype(str)
    # 构建新的特征
    data['userID_feature'] = keras.Column(name='userID_feature', axis=1)
    data['userType_feature'] = keras.Column(name='userType_feature', axis=1)
    data['user_last_login_time_feature'] = keras.Column(name='user_last_login_time_feature', axis=1)
    # 将特征添加到数据中
    return data

# 模型选择与训练
def model_selection(data):
    # 特征工程
    data = feature_engineering(data)
    # 模型选择
    model = keras.models.Sequential()
    model.add(layers.Dense(64, activation='relu', input_shape=(data.shape[1],)))
    model.add(layers.Dense(32, activation='relu'))
    model.add(layers.Dense(1, activation='linear'))
    # 编译模型
    model.compile(optimizer='adam', loss='mean_squared_error', metrics=['mae'])
    # 训练模型
    model.fit(data, epochs=50, batch_size=32, validation_split=0.2, verbose=0)
    # 评估模型
    score = model.evaluate(data, verbose=0)
    return model, score

# 模型部署
def model_deployment(data, model, score):
    # 将模型部署到线上
    model.deploy(index=0)
    # 定期更新模型
    model.fit(data, epochs=5, batch_size=32, verbose=0)
    return model, score

# 推荐模型
def recommendation_model(data):
    # 数据预处理
    data = preprocess_data(data)
    # 特征工程
    data = feature_engineering(data)
    # 模型选择与训练
    model, score = model_selection(data)
    # 推荐模型
    recommendations = model(data)
    # 将推荐结果返回
    return recommendations, score
