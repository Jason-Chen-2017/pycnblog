4.3. 核心代码实现
```python
import os
import numpy as np
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers

# 定义训练集和测试集的路径
train_data_path = './data/train/'
test_data_path = './data/test/'

# 定义模型架构
model = keras.Sequential()
model.add(layers.Reshape((224, 224, 3)))
model.add(layers.Conv2D(32, (3, 3), activation='relu', input_shape=(224, 224, 3)))
model.add(layers.MaxPooling2D(pool_size=(2, 2)))
model.add(layers.Conv2D(64, (3, 3), activation='relu'))
model.add(layers.MaxPooling2D(pool_size=(2, 2)))
model.add(layers.Conv2D(128, (3, 3), activation='relu'))
model.add(layers.MaxPooling2D(pool_size=(2, 2)))
model.add(layers.Flatten())
model.add(layers.Dense(1024, activation='relu'))
model.add(layers.Dense(4, activation='softmax'))

# 编译模型
model.compile(optimizer='adam',
              loss='categorical_crossentropy',
              metrics=['accuracy'])

# 加载训练集和测试集
train_data = keras.preprocessing.image.ImageFolder(train_data_path, target_mode='categorical')
test_data = keras.preprocessing.image.ImageFolder(test_data_path, target_mode='categorical')

# 训练模型
history = model.fit(train_data, epochs=20, batch_size=16, validation_split=0.2)

# 在测试集上评估模型
test_loss, test_acc = model.evaluate(test_data)
print('Test accuracy:', test_acc)

# 使用模型进行预测
predictions = model.predict(test_data)
```
