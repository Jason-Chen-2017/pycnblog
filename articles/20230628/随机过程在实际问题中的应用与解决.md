#include <ctime>

// 生成随机数
int randomNumber() {

}

// 生成序列
int sequence[100];

void generateSequence(int length) {
    for (int i = 0; i < length; i++) {
        int x = randomNumber();
        sequence[i] = x;

    // 输出序列

    }
