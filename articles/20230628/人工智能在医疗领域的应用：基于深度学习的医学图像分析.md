-------------

import tensorflow as tf
import numpy as np

# 预处理
def preprocess(data):
    # 裁剪
    data = data[0:200, :]
    # 标准化
    data = (data - 128) / 255.0
    return data

# CNN模型
def cnn(input_data):
    # 卷积层
    conv1 = tf.keras.layers.Conv2D(32, (3, 3), padding='same')(input_data)
    conv2 = tf.keras.layers.Conv2D(64, (3, 3), padding='same')(conv1)
    conv3 = tf.keras.layers.Conv2D(128, (3, 3), padding='same')(conv2)
    conv4 = tf.keras.layers.Flatten()(conv3)
    conv5 = tf.keras.layers.Dense(256, activation='relu')(conv4)
    conv6 = tf.keras.layers.Dense(10, activation='softmax')(conv5)
    model = tf.keras.models.Model(inputs=input_data, outputs=conv6)
    return model

# 全连接层
def fcl(input_data):
    # 将卷积层输出扁平化
    input_data = input_data.reshape(-1, 128 * 128 * 3)
    # 计算Dense层输出
    output = cnn(input_data)
    # 计算全连接层输出
    output = tf.keras.layers.Dense(10, activation='softmax')(output)
    return output

# 模型训练
model = tf.keras.models.Sequential([
    tf.keras.layers.Dense(256, activation='relu', input_shape=(128, 128, 3)),
    tf.keras.layers.Dense(10, activation='softmax')
])
