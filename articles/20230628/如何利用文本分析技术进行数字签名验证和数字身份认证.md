import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader
import numpy as np
import pandas as pd

class TextAnalyzer(nn.Module):
    def __init__(self, vocab_size, tag_to_ix, embedding_dim, max_word_len):
        super(TextAnalyzer, self).__init__()
        self.embedding_dim = embedding_dim
        self.max_word_len = max_word_len
        self.vocab_size = vocab_size
        self.tag_to_ix = tag_to_ix
        self.tag = None
        self.optimizer = optim.Adam(self.parameters(), lr=0.001)

    def forward(self, text):
        # 预处理
         words = [word for word in text.split() if len(word) > 0]
         words = [word for word in words if not word in self.stopword_set]
         words = [self.tag_to_ix[word] if self.tag is not None else self.tag_to_ix[word]]
         words = [self.tag] if self.tag else words
         words = np.array(words)
         words = torch.array(words)
         words = words.reshape(1, -1)

         # 特征提取
         features = []
         for word in words:
            feature = []
            for i in range(self.max_word_len):
                if word[i] == 0:
                    feature.append(0)
                else:
                    feature.append(1)
             features.append(feature)

         # 模型训练
         for epoch in range(100):
            loss = 0
            for i in range(128):
                for j in range(128):
                    word = [self.tag_to_ix[self.tag] if self.tag is not None else self.tag_to_ix[word]]
                    feature = features[i][j]
                    output = self.optimizer.zero_grad()
                    output.set_data(feature)
                    output.backward()
                    loss += (feature - output.data) ** 2
                    output.data = 0
             loss /= 128
             print(f'Epoch: {epoch+1}, Loss: {loss}')

    def start_epoch(self):
        self.optimizer.zero_grad()

    def tag(self, word):
        self.tag = word

    def stopword_set(self):
        stopwords = set()
         stopwords.add('a')
         stopwords.add('an')
         stopwords.add('the')
         stopwords.add('in')
         stopwords.add('un')
         stopwords.add('ic')
         stopwords.add('ing')
         stopwords.add('ed')
         return stopwords
