def recommend(user_id, user_behavior_data, item_data, user_interest_data):
    # 计算用户之间的相似度
    user_similarity = cosine_similarity(user_behavior_data[user_id], user_behavior_data)

    # 计算商品之间的相似度
    item_similarity = cosine_similarity(item_data)

    # 计算推荐得分
    recommendation_score = user_similarity * item_similarity * interest_similarity

    # 返回推荐结果
    return recommendation_score
