def open_excel(file):
    # 打开Excel文件
    workbook = openpyxl.load_workbook(file)
    # 读取数据
    sheet = workbook.active
    data = []
    for row in sheet.iter_rows(min_row=2, values_only=True):
        data.append(row)
    return data

# 处理数据
def process_data(data):
    # 对数据进行处理
    processed_data = []
    for row in data:
        processed_data.append(row[0] + row[1])
    return processed_data

# 写入Excel文件
def write_excel(file, data):
    # 打开Excel文件
    workbook = openpyxl.Workbook()
    # 创建工作表
    sheet = workbook.active
    # 写入数据
    for row in data:
        sheet.append(row)
    # 保存Excel文件
    workbook.save(file)

# 结束任务
def end_task():
    # 结束任务
    print("任务执行完成")

# 启动任务
start_task("input.xlsx", "output.xlsx")
