```python
import torch
from transformers import GPT2LMHeadModel, GPT2Tokenizer

model = GPT2LMHeadModel.from_pretrained('gpt2')
tokenizer = GPT2Tokenizer.from_pretrained('gpt2')

def generate_product_description(user_profile, product_features):
    prompt = f"Based on the user profile: {user_profile} and the product features: {product_features}, generate a personalized product description."
    input_ids = tokenizer.encode(prompt, return_tensors='pt')
    output = model.generate(input_ids, max_length=200, num_return_sequences=1, do_sample=True, top_k=50, top_p=0.95, num_beams=1)
    return tokenizer.decode(output[0], skip_special_tokens=True)
```
