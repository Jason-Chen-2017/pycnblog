import openai

openai.api_key = "your_api_key"

def gpt3_query(prompt, max_tokens=100, temperature=0.7):
    response = openai.Completion.create(
        engine="text-davinci-002",
        prompt=prompt,
        max_tokens=max_tokens,
        n=1,
        stop=None,
        temperature=temperature,
    )
    return response.choices[0].text.strip()

import torch
import torch.nn as nn
import dgl
import dgl.function as fn

class GraphRecommender(nn.Module):
    def __init__(self, num_users, num_items, emb_dim):
        super().__init__()
        self.user_emb = nn.Embedding(num_users, emb_dim)
        self.item_emb = nn.Embedding(num_items, emb_dim)

    def forward(self, g):
        with g.local_scope():
            g.ndata['h'] = torch.cat([self.user_emb.weight, self.item_emb.weight], dim=0)
            g.update_all(fn.copy_u('h', 'm'), fn.mean('m', 'h'))
            user_emb = g.ndata['h'][:num_users]
            item_emb = g.ndata['h'][num_users:]
            return user_emb, item_emb
