import pandas as pd
from sklearn.cluster import KMeans

# 数据预处理
data = [ [1, 2, 3, 4, 5, 6], [1, 4, 6, 2, 5, 3], [1, 2, 3, 5, 6, 4], [1, 3, 5, 6, 4, 2], [2, 4, 6, 1, 3, 5], [2, 5, 6, 1, 4, 3]]

    features.append(sum(data[i]))

    y.append(1)

new_data = np.array([ [1, 2], [1, 4], [1, 2], [2, 5], [2, 6] ])

for i in range(len(new_data)):
