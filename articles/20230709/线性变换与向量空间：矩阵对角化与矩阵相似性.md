import matplotlib.pyplot as plt

class MatrixOperations:
    def __init__(self):
        self.A = np.array([[1, 2], [3, 4]])
        self.D = np.array([[1, 0], [0, 1]])

    def diagonal_transform(self):
        # 将矩阵 A 转化为对角矩阵 D
        D = self.A.diagonal()
        return D

    def matrix_similarity(self, other):
        # 计算矩阵 A 与 other 矩阵的相似性
        A_other = other.A
        D_other = other.D
        return A_other.shape[0] == D_other.shape[1]
