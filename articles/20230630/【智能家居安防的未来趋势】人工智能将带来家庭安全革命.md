```
#include <stdio.h>
#include <string.h>
#include <math.h>

// 门锁控制结构体
typedef struct {
    int status;
    int lock_state;
} DoorLock;

// 初始化门锁
void init_door_lock(DoorLock *lock) {
    lock->status = 0;
    lock->lock_state = LOCK_STATE_UNLOCKED;
}

// 设置门锁状态
void set_door_lock_status(DoorLock *lock, int status) {
    lock->status = status;
    lock->lock_state = LOCK_STATE_LOCKED;
}

// 设置门锁锁定状态
void set_door_lock_state(DoorLock *lock, int lock_state) {
    lock->lock_state = lock_state;
    lock->status = 0;
}

// 解锁门锁
void unlock_door_lock(DoorLock *lock) {
    if (lock->status == 1) {
        set_door_lock_status(lock, LOCK_STATE_UNLOCKED);
    }
}
```
