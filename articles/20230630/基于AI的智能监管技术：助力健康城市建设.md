4.3. 核心代码实现
--------------

```
# Python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.neighbors import KNeighborsClassifier

# 读取数据
data = pd.read_csv('data.csv')

# 将数据分为训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(data.drop(['target'], axis=1), data['target'], test_size=0.2, random_state=0)

# 准备训练数据
train_data = {'target': y_train.values, 'X': X_train.values}

# 准备测试数据
test_data = {'target': y_test.values, 'X': X_test.values}

# 训练模型
model1 = LinearRegression()
model1.fit(train_data['X'], train_data['target'])

# 预测
test_data['pred'] = model1.predict(test_data['X'])

# 可视化
plt.plot(test_data['X'], test_data['pred'])
plt.show()
```
