4.3. 核心代码实现

```
# 导入需要的库
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# 读取数据
df = pd.read_csv("data.csv")

# 清洗数据
df = df[["user_id", "user_name", "age", "gender", "history"]]
df = df.dropna()

# 处理缺失值
df = df[["user_id", "user_name", "age", "gender", "history"]]
df["age"] = df["age"].astype(int)
df["gender"] = df["gender"].astype(str)

# 将特征转换为数值
features = df[["age", "gender"]]

# 将特征进行归一化处理
features = (features - features.mean()) / features.std()

# 将数据集分为训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(features, df["history"], test_size=0.3)

# 创建线性回归模型并进行训练
model = LinearRegression()
model.fit(X_train.values, y_train.values)

# 进行预测
y_pred = model.predict(X_test.values)

# 计算模型的均方误差
mse = mean_squared_error(y_test.values, y_pred.values)
print("Mean Squared Error:", mse)

# 分析模型的评估指标
print("R-squared:", model.score(X_test.values, y_test.values))
```
