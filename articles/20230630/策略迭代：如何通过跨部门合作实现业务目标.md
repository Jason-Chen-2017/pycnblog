```python
import random

def strategy_evaluation(business_strategy):
    # 定义一个策略评估函数，用于计算策略的值
    def evaluate_strategy(strategy):
        # 这里可以使用某种排序算法对策略进行排序
        # 如果策略长度为 0，则默认排序方式为：升序
        return strategy[-1]  # 策略最后一层的元素就是该策略的值

    # 从左到右扫描策略，对每个元素计算评估值
    for i in range(len(strategy) - 1):
        # 取出当前层的一个元素
        current_strategy = strategy[i]
        # 计算该策略的值
        this_value = evaluate_strategy(current_strategy)
        # 输出当前策略的值
        print(f"当前策略的值为：{this_value}")

# 定义一个策略列表
business_strategy = [random.random() for _ in range(10)]

# 对策略列表进行排序
business_strategy.sort(reverse=True)

# 输出排序后的策略列表
print(f"排序后的策略列表为：{business_strategy}")

# 策略迭代
for i in range(len(business_strategy) - 1):
    # 随机取出一个策略
    strategy = business_strategy[i]
    # 计算策略的值
    value = evaluate_strategy(strategy)
    # 输出策略的值
    print(f"当前策略的值为：{value}")
```

* 从左到右扫描策略列表，对每个元素计算评估值，并输出当前策略的值。
* 对策略列表进行排序，使得值越小的策略排在前面。
* 使用随机数生成随机策略，并使用上面定义的函数计算策略值。
