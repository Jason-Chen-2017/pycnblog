import pandas as pd
import numpy as np
import time
import datetime

class SmartHome:
    def __init__(self):
        self.temperature = 25  # 初始温度
        self.time = 0  # 当前时间

    def set_temperature(self, temperature):
        self.temperature = temperature

    def get_temperature(self):
        return self.temperature

    def update_time(self):
        self.time += 1

    def is_day(self):
        return not (self.time % 12 == 0 or self.time % 7 == 0)

    def save_data(self, file):
        data = {
            'time': self.time,
            'temperature': self.temperature
        }
        df = pd.DataFrame(data)
        df.to_csv(file, index=False)

    def load_data(self, file):
        data = pd.read_csv(file)
        self.time = data['time']
        self.temperature = data['temperature']



def main():
    # 初始化智能家居系统
    home = SmartHome()

    # 读取之前的数据
    file = 'data.csv'
    home.load_data(file)

    # 开始记录数据
    start_time = time.time()

    # 循环记录数据
    while True:
        # 记录当前时间
        current_time = time.time()

        # 更新温度
        home.set_temperature(30)  # 设置当前温度为30度

        # 记录当前时间
        home.update_time()

        # 判断是否为白天
        if home.is_day():
            # 获取之前记录的数据
            data = home.save_data('data.csv')
            # 分析数据，以发现用户需求和行为
            print(data)

        # 循环记录数据
        time.sleep(60)

        # 记录当前时间
        end_time = time.time()

        # 计算时间差
        interval = end_time - start_time

        # 输出每分钟温度和时间
        print('Time: {:.2f}s Temperature: {:.2f}℃'.format(interval, home.temperature))

if __name__ == '__main__':
    main()
