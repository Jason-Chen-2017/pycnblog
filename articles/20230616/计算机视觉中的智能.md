```python

import numpy as np

# Load医学影像
img = cv2.imread('医学影像.png', cv2.IMREAD_GRAYSCALE)

# Apply thresholding to get binary image
threshold = 0.5
thresh = cv2.threshold(img, 0, threshold, cv2.THRESH_BINARY_INV + cv2.THRESH_OTSU)[1]

# Extract edges from the binary image
edges = cv2.Canny(thresh, 50, 150)

# Apply深度学习算法
result = cv2.dnn.blobFromImage(edges, 50, 5, 256, 1)

# Print结果
cv2.imshow('医学影像', result)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
