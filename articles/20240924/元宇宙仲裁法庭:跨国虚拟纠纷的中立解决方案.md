```python
# MetaVerseArbitration.sol

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract MetaVerseArbitration is ERC721, ERC721URIStorage, Pausable, Ownable {
    uint256 public nextTokenId;
    mapping(uint256 => address) private _tokenOwner;
    mapping(address => uint256) private _balances;
    mapping(uint256 => uint256) private _tokenApprovals;

    constructor() ERC721("MetaVerseArbitration", "MVA") {
        nextTokenId = 1;
    }

    function mint(string memory tokenURI) public onlyOwner {
        _mint(msg.sender, nextTokenId);
        _setTokenURI(nextTokenId, tokenURI);
        _balances[msg.sender]++;
        nextTokenId++;
    }

    function transferFrom(address from, address to, uint256 tokenId) public {
        require(_tokenOwner[tokenId] == from, "Not token owner");
        _transfer(from, to, tokenId);
    }

    function approve(address to, uint256 tokenId) public {
        _approve(msg.sender, to, tokenId);
    }

    function balanceOf(address owner) public view returns (uint256 balance) {
        return _balances[owner];
    }

    function tokenOfOwnerByIndex(address owner, uint256 index) public view returns (uint256 tokenId) {
        require(index < balanceOf(owner), "Index out of bounds");
        return _tokenOwner[msg.sender][index];
    }

    function pause() public onlyOwner {
        _pause();
    }

    function unpause() public onlyOwner {
        _unpause();
    }
}
```
