```python
import numpy as np

def queue_simulation(lambda_, mu, num_trials):
    wait_times = []
    for _ in range(num_trials):
        wait_time = 0
        t = 0
        while True:
            if np.random.rand() < lambda_ / mu:
                wait_time += t
                break
            t += 1
        wait_times.append(wait_time)
    return np.mean(wait_times)

lambda_ = 1.0
mu = 2.0
num_trials = 1000
average_wait_time = queue_simulation(lambda_, mu, num_trials)
print("Average wait time:", average_wait_time)
```
