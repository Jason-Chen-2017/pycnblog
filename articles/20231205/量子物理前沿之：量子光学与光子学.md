### 4.1.1 光电效应

```python
import numpy as np
import matplotlib.pyplot as plt

# 光电效应的公式
def photovoltaic_effect(E, h, c, q):
    # 光子的能量
    E_photon = h * c / wavelength
    # 电子的能量
    E_electron = q * V
    # 光电效应的强度
    I = E_photon * E / (E_electron + E_photon)
    return I

# 光电效应的参数
wavelength = 500e-9  # 光子的波长，单位为米
h = 6.626e-34  # 平面波数量子 h，单位为J·s
c = 299792458  # 光速 c，单位为m/s
q = 1.602e-19  # 电子的电荷 q，单位为C
V = 1.0  # 电压，单位为V

# 计算光电效应的强度
I = photovoltaic_effect(E, h, c, q)
print("光电效应的强度为：", I, "A")
```

```python
import numpy as np
import matplotlib.pyplot as plt

# 光纤的传播公式
def fiber_propagation(E, n1, n2, A):
    # 光子的能量
    E_photon = h * c / wavelength
    # 光纤的传播常数
    k = (n2 - n1) / (2 * n1)
    # 光纤的传播强度
    I = E_photon * E * k
    return I

# 光纤的传播参数
wavelength = 500e-9  # 光子的波长，单位为米
h = 6.626e-34  # 平面波数量子 h，单位为J·s
c = 299792458  # 光速 c，单位为m/s
n1 = 1.0  # 光纤内部的折射率
n2 = 1.5  # 光纤外部的折射率
A = 1.0  # 光纤的交叉面积，单位为m^2

# 计算光纤的传播强度
I = fiber_propagation(E, n1, n2, A)
print("光纤的传播强度为：", I, "A")
```
