import os
import zipfile
import hashlib
import requests
from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes

# 数据压缩
def compress_data(data):
    with zipfile.ZipFile('backup.zip', 'w') as zipf:
        zipf.write(data)

# 加密
def encrypt_data(data, key):
    cipher = AES.new(key, AES.MODE_EAX)
    ciphertext, tag = cipher.encrypt_and_digest(data)
    return ciphertext, tag

# 上传到云计算平台
def upload_to_cloud(file_path):
    url = 'https://cloud.example.com/upload'
    with open(file_path, 'rb') as f:
        response = requests.post(url, files={'file': f})
    return response.status_code

# 下载并解密
def download_and_decrypt(file_path):
    url = 'https://cloud.example.com/download'
    response = requests.get(url)
    with open(file_path, 'wb') as f:
        f.write(response.content)

    with open(file_path, 'rb') as f:
        data = f.read()
        key = get_random_bytes(16)
        ciphertext, tag = encrypt_data(data, key)
        cipher = AES.new(key, AES.MODE_EAX)
        data = cipher.decrypt_and_verify(ciphertext, tag)
        return data

# 解压缩
def decompress_data(data):
    with zipfile.ZipFile('backup.zip', 'r') as zipf:
        zipf.extractall()

# 主函数
def main():
    data = 'This is a sample data.'
    compress_data(data)
    encrypted_data, tag = encrypt_data(data.encode(), get_random_bytes(16))
    status_code = upload_to_cloud('backup.zip')
    if status_code == 200:
        downloaded_data = download_and_decrypt('backup.zip')
        decompress_data(downloaded_data)
        print('Backup and recovery successful.')
    else:
        print('Backup and recovery failed.')

if __name__ == '__main__':
    main()
