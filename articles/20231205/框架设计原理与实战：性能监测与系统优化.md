```python
    if len(arr) <= 1:

    pivot = arr[0]
    left = [x for x in arr[1:] if x < pivot]
```

```python
def binary_search(arr, target):
    left = 0
    right = len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1
```
