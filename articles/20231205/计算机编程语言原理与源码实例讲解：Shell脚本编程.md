# 输出变量值

# 文件操作
file="/etc/passwd"
if [ -f "$file" ]; then
    echo "File exists"
else
    echo "File not exists"
fi

# 进程管理
process="$(ps -ef | grep -v grep | grep -c $0)"
if [ $process -gt 0 ]; then
    echo "Process exists"
else
    echo "Process not exists"
fi

# 用户管理
user="$(who | grep $name)"
if [ -z "$user" ]; then
    echo "User not exists"
else
    echo "User exists"
fi
