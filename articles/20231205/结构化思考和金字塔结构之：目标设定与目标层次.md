    def __init__(self):
        self.targets = []

    def add_target(self, target):
        self.targets.append(target)

    def set_target(self, target_name, target_value):
        target = Target(target_name, target_value)
        self.add_target(target)

    def get_target_value(self, target_name):
        for target in self.targets:
            if target.name == target_name:
                return target.value
        return None

    def __init__(self, name, value):
        self.name = name
        self.value = value

target_setting.set_target("价格竞争力", 85)

# 获取目标值

price_competitiveness = target_setting.get_target_value("价格竞争力")

print("产品质量：", product_quality)
