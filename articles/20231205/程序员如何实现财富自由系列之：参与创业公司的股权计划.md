# 4.具体代码实例和详细解释说明

```python
import math

def calculate_pv(earnings, shares):
    return earnings / shares

def calculate_gv(pv, initial_value):
    return pv - initial_value

def calculate_loan(gv, rate):
    return gv / rate

def calculate_sale(gv, price):
    return gv / price

# 示例数据
earnings = 100000
shares = 1000
initial_value = 50000
rate = 0.05
price = 100

# 计算股权价值
pv = calculate_pv(earnings, shares)
print("股权价值:", pv)

# 计算股权增值
gv = calculate_gv(pv, initial_value)
print("股权增值:", gv)

# 计算股权抵押
loan = calculate_loan(gv, rate)
print("股权抵押:", loan)

# 计算股权转让
sale = calculate_sale(gv, price)
print("股权转让:", sale)
```
