    n = len(s2)

        for j in range(n):

def resource_allocation(tasks, resources):
    tasks.sort(key=lambda x: x[1], reverse=True)
    result = []
    for task in tasks:
        if resources >= task[0]:
            result.append(task)
            resources -= task[0]
    return result
