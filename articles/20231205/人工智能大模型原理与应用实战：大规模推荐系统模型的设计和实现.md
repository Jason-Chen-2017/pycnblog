```python
from sklearn.metrics.pairwise import cosine_similarity

# 基于内容的推荐算法
def content_based_recommendation(user_preferences, items):
    # 计算商品之间的相似度
    item_similarity = cosine_similarity(items)
    
    # 根据用户的兴趣推荐相似的商品
    recommended_items = []
    for item in items:
        if item in user_preferences:
            similar_items = item_similarity[items.index(item)]
            recommended_items.extend(list(set(items) - set(user_preferences) - set(similar_items)))
    return recommended_items

# 基于协同过滤的推荐算法
def collaborative_filtering_recommendation(user_ratings, items):
    # 计算用户之间的相似度
    user_similarity = cosine_similarity(user_ratings)
    
    # 根据用户的相似度推荐相关的商品
    recommended_items = []
    for user in user_ratings:
        similar_users = user_similarity[user]
        for similar_user in similar_users:
            if similar_user != user:
                recommended_items.extend(list(set(items) - set(user_ratings[user]) - set(user_ratings[similar_user])))
    return recommended_items
```
