```python
import numpy as np
from sklearn.semi_supervised import LabelSpreading

# 数据预处理
X = np.array([[1, 2], [3, 4], [5, 6], [7, 8]])
Y = np.array([[1], [2], [3], [4]])
Z = np.array([[9, 10], [11, 12], [13, 14], [15, 16]])

# 初始化
label_spreading = LabelSpreading(kernel='katz', alpha=0.1, n_iter=100)
label_spreading.fit(X, Y)

# 迭代更新
Y_pred = label_spreading.predict(Z)

# 输出结果
print(Y_pred)
```
