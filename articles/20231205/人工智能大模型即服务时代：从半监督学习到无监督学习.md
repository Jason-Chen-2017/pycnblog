```python
                           random_state=42, n_clusters_per_class=1)

# 生成一个无监督学习数据集
Z = np.random.rand(1000, 20)

# 创建一个半监督学习模型
model = LabelSpreading(kernel='knn', k=5, alpha=0.1)

# 对有监督数据集进行有监督学习
model.fit(X, y)

# 对无监督数据集进行无监督学习
model.fit(Z, None)

# 获取最终的模型
final_model = model.fit_transform(X)
```
