import java.util.ArrayList;
import java.util.List;
import org.json.JSONObject;
import org.json.JSONReader;
import org.json.JSONWriter;
import org.json.me.json. MeJson;
import org.json.me.json.MeJson.Builder;
import org.json.me.json.MeJson.Value;
import org.json.me.json.object.ObjectBuilder;
import org.json.me.json.object.ObjectModel;
import org.json.me.json.object.Style;
import org.json.me.json.text.Text;
import java.util.HashMap;
import java.util.Map;

public class Robot {
    private List<Style> styles;

    public Robot(List<Style> styles) {
        this.styles = styles;
    }

    public void setStyles(List<Style> styles) {
        this.styles = styles;
    }

    public Value getValue(String key) {
        JSONObject obj = new JSONObject();
        for (Style style : styles) {
            obj.setValue(key, style.getValue());
        }
        return obj.get(key);
    }

    public void setValue(String key, Value value) {
        obj.setValue(key, value);
    }

    public JSONObject getJSONObject() {
        JSONObject obj = new JSONObject();
        for (Style style : styles) {
            obj.setValue(style.getKey(), getValue(style.getKey()));
        }
        return obj;
    }

    public void setJSONObject(JSONObject obj) {
        this.obj = obj;
    }

    public void writeToJSON(String fileName) {
        try {
            JSONWriter.Builder writer = new JSONWriter.Builder(fileName);
            writer.setIndent(4);
            for (Style style : styles) {
                writer.writeValue(style.getValue(), style.getKey());
            }
            writer.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public void readFromJSON(String fileName) {
        try {
            JSONObject obj = JSONReader.parseText(fileName);
            for (Style style : styles) {
                setValue(style.getKey(), obj.getValue(style.getKey()));
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
