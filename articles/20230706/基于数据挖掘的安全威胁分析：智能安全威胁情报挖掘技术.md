import pandas as pd
import numpy as np
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.pipeline import Pipeline
from sklearn.model_selection import train_test_split

# 读取数据
def read_data(data_path):
    return pd.read_csv(data_path)

# 数据预处理
def preprocess_data(data):
    # 清洗
    data = data.dropna()
    data = data[["IP地址", "端口号", "协议类型"]]
    # 去重
    data = data.drop_duplicates()
    # 统一格式
    data["协议类型"] = "，" + data["协议类型"]
    return data

# 特征选择
def feature_extraction(text):
    vectorizer = CountVectorizer()
    vectorizer.fit(text)
    return vectorizer.transform(text)

# 数据挖掘
def data_mining(data):
    # 特征选择
    features = feature_extraction(data["协议类型"])
    # 标签
    labels = data.主效应
    # 建立分类器
    clf = MultinomialNB()
    clf.fit(features, labels)
    # 预测
    return clf.predict(features)

# 将各个模块进行集成
def integrate_module(data):
    data_processed = read_data("data.csv")
    data_processed = preprocess_data(data_processed)
    features = feature_extraction(data_processed["协议类型"])
    labels = data_processed["主效应"]
    clf = MultinomialNB()
    clf.fit(features, labels)
    return clf

# 运行
data = integrate_module("data")
情报挖掘 = data_mining(data)
