4.3. 核心代码实现

智能家居控制面板的核心代码主要包括传感器数据采集、数据传输、数据处理和智能决策等模块。以下是一个简单的代码示例：

```
import random
import numpy as np
import datetime

class Sensor:
    def __init__(self, device, sensor):
        self.device = device
        self.sensor = sensor

    def read(self):
        return self.sensor.read()

class SmartHomeController:
    def __init__(self):
        self.sensors = []

    def add_sensor(self, device, sensor):
        self.sensors.append(Sensor(device, sensor))

    def read_sensors(self):
        return [s.read() for s in self.sensors]

    def set_device(self, device, action):
        if action == "on":
            for sensor in self.sensors:
                sensor.device = device
        elif action == "off":
            for sensor in self.sensors:
                sensor.device = None

    def run(self):
        while True:
            readings = self.read_sensors()
            device = None
            action = None
            for sensor in readings:
                if sensor.read() > 0:
                    device = sensor.device
                    if action == "on":
                        action = "off"
                    elif action == "off":
                        action = "on"
                elif device is not None:
                    action = None
                    break
            if device is not None:
                action = device.action
                if action == "on":
                    device.action = "off"
                elif action == "off":
                    device.action = "on"
                    break
            if action == "on":
                device.action = "on"
            elif action == "off":
                device.action = "off"
            print(device.action)

# Example usage
sensor1 = Sensor("device1", "temperature")
sensor2 = Sensor("device2", "light")
sensor3 = Sensor("device3", " door")
smart_home_controller = SmartHomeController()
smart_home_controller.add_sensor(sensor1, sensor1)
smart_home_controller.add_sensor(sensor2, sensor2)
smart_home_controller.add_sensor(sensor3, sensor3)
smart_home_controller.run()
```
