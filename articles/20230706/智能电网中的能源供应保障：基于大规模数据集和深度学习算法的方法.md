2.2.5. 代码实例和解释说明

```
import numpy as np
import pandas as pd
import tensorflow as tf
from sklearn.model_selection import train_test_split

# 读取数据
df = pd.read_csv('energy_data.csv')

# 数据预处理
df = df[['temperature', 'humidity', 'gas_price', 'energy_demand']]
df.dropna(inplace=True)
df = df.dropna(inplace=True)

# 特征工程
X = df[['temperature', 'humidity', 'gas_price', 'energy_demand']]

# 计算特征间相关性
corr = df.corr()

# 选择特征
X = X[['temperature', 'humidity', 'gas_price']]

# 添加激活函数
def demand_predict(X):
    return np.maximum(X, 0)

# 构建模型
model = tf.keras.models.Sequential([
    tf.keras.layers.Dense(64, activation='relu', input_shape=(X.shape[1],)),
    tf.keras.layers.Dense(64, activation='relu'),
    tf.keras.layers.Dense(1)
])

# 编译模型
model.compile(loss='mean_squared_error', optimizer='adam')

# 训练模型
history = model.fit(X, demand_predict, epochs=100)

# 评估模型
mse = model.evaluate(X, demand_predict)

# 绘制训练集和测试集数据
train = history.train_data
test = history.test_data

# 绘制训练集和测试集需求预测
train_demand = train.demand_predict(train.temperature)
test_demand = test.demand_predict(test.temperature)
```
