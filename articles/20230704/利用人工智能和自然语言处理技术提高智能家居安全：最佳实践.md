using namespace std;

void main(int argc, char** argv)
{
    //读取文件
    VideoCapture(0);
    Mat video(640, 480, CV_8UC3, argv[1]);
    //创建用于存放检测框的矩形框
    Rect rect(0,0,640,480);
    //循环遍历每一帧
    while(true)
    {
        //读取
        if(video.read(rect) == true)
        {
            //转换为灰度图像
            Mat gray;
            cvtColor(video, gray, COLOR_BGR2GRAY);
            //寻找轮廓
            Mat contoursMat;
            findContours(gray, contoursMat, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);
            //绘制轮廓
            drawContours(video, contoursMat, -1, (int)coutil.arabic(0), 1);
            //检测物体
            vector<vector<vector<int>>> objects;
            //循环遍历每一帧
            for(int i = 0; i < 400; i++)
            {
                //循环遍历每一个轮廓
                for(int j = 0; j < (int)contoursMat.cols; j++)
                {
                    double area = cv::contourArea(contoursMat, contoursMat.at<Point2f>(j));
                    if(area > 100)
                    {
                        vector<vector<int>> obj;
                        //绘制边框
                        drawContours(video, contoursMat, -1, (int)coutil.arabic(0), 1);
                        //计算面积
                        obj.push_back(coutil.arabic(0));
                        coutil.clearContour(video, contoursMat);
                        coutil.drawContours(video, obj, -1, (int)coutil.arabic(0), 1);
                        objects.push_back(obj);
                    }
                }
            }
            //检测到的物体保存到文件
            for(int i = 0; i < (int)objects.size(); i++)
            {
                //保存到文件
                cv::imwrite(argv[2], objects[i]);
            }
            //显示每一帧
            video.show();
        }
        else
        {
            break;
        }
    }
    //释放
    video.release();
    return 0;
}
