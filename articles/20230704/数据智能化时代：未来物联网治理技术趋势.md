def adaptive_regulation(temperature, humidity, target_temperature, tolerance):
    # 定义温度和湿度变化范围

    humidity_change = (humidity - target_humidity) / tolerance

    new_temperature = temperature + temperature_change
    new_humidity = humidity + humidity_change
