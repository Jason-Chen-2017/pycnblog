      for j in range(len(item_list)):
        if i == 0 or j == 0:
          similarities.append(0)
        elif query_list[i - 1] == item_list[j - 1]:
          similarities.append(1)
    # 返回货物的相似度
    return sum(similarities) / len(query_list)

      elif item.类别 == 'B':

      else:
        categories.append('C')
    # 返回货物的类别
    return categories
  
  # 计算距离
  def calculate_distance(query_list, item_list):
    # 计算两个列表之间的欧几里得距离
    sum_x2 = 0
    sum_x1 = 0
    for i in range(len(query_list)):
      for j in range(len(item_list)):
        sum_x2 += (query_list[i] - item_list[j]) ** 2
        sum_x1 += (query_list[i] - item_list[j]) ** 2
    # 计算距离的平方和
    return sum_x2 + sum_x1
  
  # 计算相似度
  def calculate_similarity(query_list, item_list):
    # 计算货物的相似度
    similarities = []
    for i in range(len(query_list)):
      for j in range(len(item_list)):
        if i == 0 or j == 0:
          similarities.append(0)
        elif query_list[i - 1] == item_list[j - 1]:
          similarities.append(1)
    # 计算相似度的和
    return sum(similarities)
  
  # 按照距离和相似度对货物进行排序
  sorted_items = space_query(query_list, item_list)
