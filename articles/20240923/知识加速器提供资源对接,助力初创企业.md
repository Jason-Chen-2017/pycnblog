### 5.2 源代码详细实现

以下是知识加速器后端的源代码实现：

```python
# 导入 Flask 框架
from flask import Flask, request, jsonify
# 导入 MySQL Connector
import mysql.connector

# 创建 Flask 应用
app = Flask(__name__)

# 连接 MySQL 数据库
conn = mysql.connector.connect(
    host="localhost",
    user="root",
    password="password",
    database="knowledge_accelerator"
)

# 创建数据库表
cursor = conn.cursor()
cursor.execute("""
    CREATE TABLE IF NOT EXISTS demand (
        id INT AUTO_INCREMENT PRIMARY KEY,
        company_name VARCHAR(255),
        demand_type VARCHAR(255),
        demand_detail TEXT
    )
""")

cursor.execute("""
    CREATE TABLE IF NOT EXISTS resource (
        id INT AUTO_INCREMENT PRIMARY KEY,
        resource_type VARCHAR(255),
        resource_detail TEXT
    )
""")

# 注册 API 接口
@app.route('/register_demand', methods=['POST'])
def register_demand():
    data = request.json
    company_name = data['company_name']
    demand_type = data['demand_type']
    demand_detail = data['demand_detail']
    
    cursor.execute("""
        INSERT INTO demand (company_name, demand_type, demand_detail)
        VALUES (%s, %s, %s)
    """, (company_name, demand_type, demand_detail))
    
    conn.commit()
    return jsonify({'status': 'success', 'message': 'Demand registered successfully.'})

@app.route('/register_resource', methods=['POST'])
def register_resource():
    data = request.json
    resource_type = data['resource_type']
    resource_detail = data['resource_detail']
    
    cursor.execute("""
        INSERT INTO resource (resource_type, resource_detail)
        VALUES (%s, %s)
    """, (resource_type, resource_detail))
    
    conn.commit()
    return jsonify({'status': 'success', 'message': 'Resource registered successfully.'})

# 启动 Flask 应用
if __name__ == '__main__':
    app.run(debug=True)
```
