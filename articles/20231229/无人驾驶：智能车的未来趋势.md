def least_squares(X, y):

    XTY = np.dot(X.T, y)

def gradient_descent(X, y, alpha, iterations):
    m = len(y)
    theta = np.zeros(X.shape[1])
    for i in range(iterations):
        predictions = X.dot(theta)
        errors = predictions - y
        theta -= alpha / m * X.T.dot(errors)
    return theta

def logistic_regression(X, y, alpha, iterations):
    m = len(y)
    theta = np.zeros(X.shape[1])
    h = np.vectorize(lambda x: 1 / (1 + np.exp(-x)))

def rule_representation(knowledge):
    rules = {}
    for rule in knowledge:

def frame_representation(knowledge):
    frames = {}
    for frame in knowledge:
        slots = frame.split(':')
        slots = dict(zip(slots[:-1], slots[-1].split(',')))
        frames[frame.strip()] = slots
    return frames
