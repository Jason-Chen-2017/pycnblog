```python
import numpy as np
import cvxopt

# 原始医学图像

# 压缩映射
W = np.dot(cv2.DCT(np.float32(image)), np.float32(image))

# 压缩后的医学图像
compressed_image = W * image

# 压缩感知恢复算法
def compressive_sensing(compressed_image, alpha):
    C = cvxopt.matrix(compressed_image.reshape(-1))
    D = cvxopt.matrix(np.eye(compressed_image.shape[0]))
    G = cvxopt.matrix(np.zeros((compressed_image.shape[0], 1)))
    h = cvxopt.matrix(np.inf * np.ones(compressed_image.shape[0]))
    A = cvxopt.matrix(np.eye(compressed_image.shape[0]))
    b = cvxopt.matrix(np.zeros(compressed_image.shape[0]))
    G = cvxopt.matrix(np.eye(compressed_image.shape[0]))
    y = cvxopt.matrix(np.zeros(compressed_image.shape[0]))
    solver = cvxopt.solvers.qpSDP(C, D, G, h, A, b, y)
    solution = solver(['x'])
    x = solution['x'].value
    x = x.reshape(image.shape)
    return x

# 压缩感知恢复后的医学图像
recovered_image = compressive_sensing(compressed_image, alpha)

# 显示恢复后的医学图像
cv2.imshow('Recovered Medical Image', recovered_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
