    def __init__(self, user_history):
        self.user_history = user_history
        self.recommend_list = []

    def recommend(self):
        for history in self.user_history.get_history():
            # 根据用户购物历史，找出用户喜欢的商品
            if history.get_product() in self.recommend_list:

        return self.recommend_list
