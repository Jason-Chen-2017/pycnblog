import re
import nltk
from nltk.corpus import stopwords

def preprocess_text(text):
    # 清洗文本
    text = re.sub(r'[^\w\s]', '', text)
    # 分词
    words = nltk.word_tokenize(text)
    # 去除停用词
    words = [word for word in words if word not in stopwords.words('english')]
    # 返回预处理后的文本
    return words
```

from nltk.chunk import ne_chunk

def relation_extraction(text):
    # 进行命名实体识别
    named_entities = ne_chunk(text)
    # 提取实体和关系
    entities = []
    for entity in named_entities.subtrees():
        if entity.label() == 'NE':
            entities.append((entity.text, entity.label()))
    # 返回实体关系图
    return entities

import networkx as nx

def build_network(data):
    # 建立网络
    G = nx.Graph()
    # 添加节点
    for index, row in data.iterrows():
        G.add_node(row['name'], attributes=row.drop('name'))
    # 添加边
    for index, row in data.iterrows():
        for relation in row['relations']:
            G.add_edge(row['name'], relation['object'], attributes=relation.drop('object'))
    # 返回建立的网络
    return G
```
