def sequence_alignment(seq1, seq2):
    # 序列比对
    alignment_score = 0
    for i in range(len(seq1)):
        for j in range(len(seq2)):
            alignment_score += seq1[i] * seq2[j]
    return alignment_score

def genome_alignment(genome1, genome2):
    # 基因组比对
    alignment_score = 0
    for i in range(len(genome1)):
        for j in range(len(genome2)):
            alignment_score += genome1[i] * genome2[j]
    return alignment_score

    # 基因组分析
    analysis_score = 0
    for i in range(len(genome1)):
        for j in range(len(genome2)):
            analysis_score += genome1[i] * genome2[j]
