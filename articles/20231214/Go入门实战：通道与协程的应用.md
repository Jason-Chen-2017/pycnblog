Go语言是一种强类型、垃圾回收、并发简单的编程语言，由Google开发。Go语言的设计目标是让程序员更好地编写并发程序，并提供更好的性能。Go语言的并发模型主要由协程和通道组成。

协程是一种轻量级的用户级线程，它们可以在同一时间执行多个任务。协程与线程不同的是，协程的调度由程序员自己控制，而不是由操作系统自动调度。协程的调度是通过Go语言的调度器来实现的，调度器会根据协程的执行情况来调度它们。

通道是Go语言中的一种数据结构，它用于实现并发程序的同步和通信。通道是一种特殊的数据结构，它可以用来传递数据，而不是传递指针或引用。通道的设计目的是为了解决并发编程中的数据竞争问题，以及提高程序的性能和可读性。

			fmt.Println("协程1执行", i)
			time.Sleep(time.Second)
		}
	}()

	go func() {
		for i := 0; i < 5; i++ {
			fmt.Println("协程2执行", i)
			time.Sleep(time.Second)
		}
	}()

	// 等待协程完成
	time.Sleep(time.Second * 10)

	// 通道示例
	ch := make(chan int)

	// 将数据发送到通道
	go func() {
		for i := 0; i < 5; i++ {
			ch <- i
			fmt.Println("协程发送数据", i)
			time.Sleep(time.Second)
		}
		close(ch)
	}()

	// 从通道中接收数据
	for v := range ch {
		fmt.Println("协程接收数据", v)
	}

	// 关闭通道
	// close(ch)
}
