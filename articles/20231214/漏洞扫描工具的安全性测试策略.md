```python
from bs4 import BeautifulSoup

def scan(url):
    # 发送请求

    # 分析响应
    soup = BeautifulSoup(response.text, 'html.parser')
    # 找到漏洞
    for tag in soup.find_all('input'):
        if tag.get('type') == 'text':
            print('漏洞发现：可能存在SQL注入漏洞')

# 使用示例

```
