```python
# 数据处理流程的描述
pipeline = dvc.Pipeline()

@pipeline.task
def read_data(path):
    # 数据源的读取
    data = pd.read_csv(path)
    return data

@pipeline.task
def clean_data(data):
    # 数据处理任务的执行
    data = data.dropna()
    return data

@pipeline.task
def write_data(data, path):
    # 数据目标的写入
    data.to_csv(path)
    return path

# 数据处理流程的执行
pipeline()
```
