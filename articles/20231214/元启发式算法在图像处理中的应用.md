# 识别物体

for label, color in zip(labels, label_colors):
    mask = np.zeros_like(blur)
    mask[blur == label] = 1
    objects.append((label, color, mask))

# 合成图像
new_img = np.zeros_like(img)
for label, color, mask in objects:
    new_img[mask > 0] = color

# 显示结果
cv2.imshow('new_img', new_img)
cv2.waitKey(0)
cv2.destroyAllWindows()
