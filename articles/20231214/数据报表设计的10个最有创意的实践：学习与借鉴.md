### 4.1 交互式数据报表

```python
import pandas as pd
import matplotlib.pyplot as plt

# 创建数据库
db = pd.DataFrame({
    'name': ['Alice', 'Bob', 'Charlie', 'David', 'Eve'],
    'age': [25, 30, 35, 40, 45],
    'salary': [50000, 60000, 70000, 80000, 90000]
})

# 创建报表模板
template = '''
<html>
<head>
    <title>交互式数据报表</title>
</head>
<body>
    <h1>交互式数据报表</h1>
    <table>
        <tr>
            <th>名字</th>
            <th>年龄</th>
            <th>薪资</th>
        </tr>
        {% for row in rows %}
        <tr>
            <td>{{ row.name }}</td>
            <td>{{ row.age }}</td>
            <td>{{ row.salary }}</td>
        </tr>
        {% endfor %}
    </table>
</body>
</html>
'''

# 生成报表
def generate_report(query):
    results = db.query(query)
    rendered_template = template.render(rows=results.to_dict('records'))
    return rendered_template

# 显示报表
def display_report(report):
    print(report)

# 执行查询
query = 'age > 35'
report = generate_report(query)
display_report(report)
```
