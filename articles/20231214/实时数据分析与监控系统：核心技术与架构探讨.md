```python
import requests

def fetch_data(url):
    response = requests.get(url)
    if response.status_code == 200:
        return response.text
    else:
        return None
```

### 4.2数据处理

```python
import pandas as pd

def process_data(data):
    df = pd.read_csv(data)
    df = df[df['value'] > 100]
    df = df.groupby('timestamp').mean()
    return df
```

### 4.3数据存储

```python
import sqlite3

def store_data(df):
    conn = sqlite3.connect('data.db')
    df.to_sql('data', conn, if_exists='replace')
    conn.close()
```
