```python
def needman_wunsch(a, b):
    dp = [[0] * (len(b) + 1) for _ in range(len(a) + 1)]
    for i in range(1, len(a) + 1):
        for j in range(1, len(b) + 1):
            score = 0 if a[i - 1] == b[j - 1] else -1
            dp[i][j] = max(dp[i - 1][j - 1] + score,
                           dp[i - 1][j] + 1,
                           dp[i][j - 1] + 1)
    return dp[-1][-1]
```
