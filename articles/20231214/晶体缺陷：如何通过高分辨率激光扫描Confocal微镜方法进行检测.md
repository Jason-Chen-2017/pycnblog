在这种方法中，激光扫描Confocal微镜通过对晶体表面进行高分辨率扫描，来检测晶体缺陷。当激光光线通过晶体表面传播时，晶体缺陷会导致光线的散射和吸收。通过分析光线的散射和吸收情况，我们可以确定晶体缺陷的位置、形状和大小。

```python
import numpy as np
from scipy.optimize import least_squares

def light_propagation(x, y, I0, alpha, d, T):
    return I0 * np.exp(-alpha * d) * T

def cost_function(params, x, y, I, I0, alpha, d, T):
    I0, alpha, d, T = params
    return np.sum((light_propagation(x, y, I0, alpha, d, T) - I)**2)

# 设置参数
x = np.linspace(-1, 1, 100)
y = np.linspace(-1, 1, 100)

alpha = 0.1
d = np.sqrt(x**2 + y**2)
T = np.exp(-alpha * d)

# 优化
params = least_squares(cost_function, (I0, alpha, d, T), args=(x, y, I, I0, alpha, d, T)).x

# 输出结果
print("I0 =", params[0])
print("alpha =", params[1])

print("T =", params[3])
```
