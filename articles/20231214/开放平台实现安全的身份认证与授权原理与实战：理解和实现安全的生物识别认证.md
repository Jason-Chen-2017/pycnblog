import numpy as np

def extract_features(image):
    # 预处理生物样本
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    gray = cv2.resize(gray, (256, 256))
    gray = cv2.rotate(gray, cv2.ROTATE_90_CLOCKWISE)

    # 提取生物特征

    return features
