import random
import numpy as np

# 适应度函数
def fitness_function(x):
    return x ** 2

# 选择操作
def selection(population):
    selected_individuals = []
    for _ in range(population_size):
        individual = random.choice(population)
        selected_individuals.append(individual)
    return selected_individuals

# 变异操作
def mutation(individual):
    mutated_individual = individual.copy()
    mutation_probability = 0.1
    for i in range(len(individual)):
        if random.random() < mutation_probability:
            mutated_individual[i] = random.randint(-10, 10)
    return mutated_individual

# 交叉操作
def crossover(parent1, parent2):
    crossover_point = random.randint(1, len(parent1) - 1)
    child1 = parent1[:crossover_point] + parent2[crossover_point:]
    child2 = parent2[:crossover_point] + parent1[crossover_point:]
    return child1, child2

# 遗传算法的主要流程
def genetic_algorithm(population, population_size, mutation_probability, max_generations):
    for _ in range(max_generations):
        selected_individuals = selection(population)
        mutated_individuals = []
        for individual in selected_individuals:
            mutated_individual = mutation(individual, mutation_probability)
            mutated_individuals.append(mutated_individual)
        population = []
        for i in range(population_size):
            child1, child2 = crossover(mutated_individuals[i], mutated_individuals[i - 1])
            population.append(child1)
            population.append(child2)
    return population

# 初始化种群
population = [random.randint(-10, 10) for _ in range(population_size)]

# 遗传算法的主要流程
population = genetic_algorithm(population, population_size, mutation_probability, max_generations)

# 输出最优解
best_individual = max(population, key=fitness_function)
print("最优解：", best_individual)
