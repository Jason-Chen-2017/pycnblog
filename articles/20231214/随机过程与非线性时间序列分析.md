import numpy as np
import pandas as pd
from statsmodels.tsa.seasonal import seasonal_decompose
from statsmodels.tsa.stattools import adfuller

# 加载时间序列数据
data = pd.read_csv('data.csv', index_col='date', parse_dates=True)

# 对时间序列数据进行预处理
data = data.dropna()

# 对时间序列数据进行非线性变换
decomposition = seasonal_decompose(data, model='multiplicative')
trend = decomposition.trend
seasonal = decomposition.seasonal
residual = decomposition.resid

# 对非线性变换后的数据进行线性分析
dft = adfuller(trend)
print('trend ADF Statistic:', dft[0])
print('trend p-value:', dft[1])

dfs = adfuller(seasonal)
print('seasonal ADF Statistic:', dfs[0])
print('seasonal p-value:', dfs[1])

dfr = adfuller(residual)
print('residual ADF Statistic:', dfr[0])
print('residual p-value:', dfr[1])
