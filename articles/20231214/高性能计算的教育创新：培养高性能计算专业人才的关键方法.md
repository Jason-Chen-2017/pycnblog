```python
import os
from multiprocessing import Pool

def sort_nums(nums, start, end):
    result = nums[start:end]
    result.sort()
    return result

def parallel_sort(nums):
    num_processes = os.cpu_count()
    num_chunks = len(nums) // num_processes
    pool = Pool(num_processes)
    results = pool.map(sort_nums, [(nums, i * num_chunks, (i + 1) * num_chunks) for i in range(num_processes)])
    sorted_nums = []
    for result in results:
        sorted_nums.extend(result)
    return sorted_nums

nums = [5, 2, 8, 1, 9, 3, 6, 4, 7]
result = parallel_sort(nums)
print(result)
```
