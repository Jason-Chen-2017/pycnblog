具体代码实例和详细解释说明如下：

```python
import cv2
import numpy as np
import tensorflow as tf

# 图像预处理
def preprocess(img):
    img = cv2.resize(img, (224, 224))
    img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    img = img / 255.0
    return img

# 图像分割
def segmentation(img):
    edges = cv2.Canny(img, 100, 200)
    contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    for contour in contours:
        x, y, w, h = cv2.boundingRect(contour)
        cv2.rectangle(img, (x, y), (x + w, y + h), (0, 255, 0), 2)
    return img

# 图像特征提取
def feature_extraction(img):
    corners = cv2.goodFeaturesToTrack(img, maxCorners=100, qualityLevel=0.01, blockSize=3)
    corners = np.int0(corners)
    return corners

# 图像识别
def image_recognition(img):
    model = tf.keras.models.load_model('cnn_model.h5')
    img = preprocess(img)
    img = np.expand_dims(img, axis=0)
    prediction = model.predict(img)
    return prediction

# 主程序
def main():
    img = segmentation(img)
    corners = feature_extraction(img)
    prediction = image_recognition(img)
    print(prediction)

if __name__ == '__main__':
    main()
```

这个代码实例主要包括图像预处理、图像分割、图像特征提取和图像识别的步骤。通过这个代码实例，我们可以看到图像识别技术在农业行业的应用具有很大的可行性和实用性。
