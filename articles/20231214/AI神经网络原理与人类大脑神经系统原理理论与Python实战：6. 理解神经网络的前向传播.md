# 定义神经网络的输入、隐藏层和输出层的节点数量

output_nodes = 2

# 定义神经网络的权重和偏置

# 定义输入数据
inputs = np.array([[0, 0, 1]])

# 前向传播算法
hidden_layer_inputs = np.dot(inputs, weights) + biases
hidden_layer_outputs = 1 / (1 + np.exp(-hidden_layer_inputs))
outputs = np.dot(hidden_layer_outputs, weights.T) + biases

# 输出结果
print(outputs)
