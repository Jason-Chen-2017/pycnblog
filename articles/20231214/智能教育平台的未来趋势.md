def calculate_learning_habits_score(learning_habits):
    # 计算学生的学习习惯得分
    learning_habits_score = 0
    for habit in learning_habits:
        if habit == 'regular':
            learning_habits_score += 1
        elif habit == 'consistent':
            learning_habits_score += 2
        elif habit == 'irregular':
            learning_habits_score += 0.5
    return learning_habits_score

# 计算学生的能力得分
def calculate_abilities_score(abilities):
    # 计算学生的能力得分
    abilities_score = 0
    for ability in abilities:
        if ability == 'high':
            abilities_score += 2
        elif ability == 'medium':
            abilities_score += 1
        elif ability == 'low':
            abilities_score += 0.5
    return abilities_score

# 根据学习习惯得分和能力得分推荐学习资源
def recommend_resources_based_on_scores(learning_habits_score, abilities_score):
    # 根据学习习惯得分和能力得分推荐学习资源
    recommended_resources = []
    if learning_habits_score >= 3:
        recommended_resources.append('regular_resources')
    if abilities_score >= 2:
        recommended_resources.append('high_ability_resources')
    return recommended_resources
