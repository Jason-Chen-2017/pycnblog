# 创建量子电路

# 设置量子门
qc.h(0)
qc.cx(0, 1)
qc.h(1)

# 将量子电路转换为类似的量子电路

transpiled_qc = transpile(qc, simulator)

# 执行量子电路

# 绘制结果
plot_histogram(counts)
