```python
from scipy.optimize import linprog

def solve_conflict(A, B):
    # 求解线性方程组
    c = np.array([1, 1])
    A_ub = np.hstack([A, np.eye(len(A))])
    b_ub = np.hstack([B, np.zeros(len(B))])
    bounds = [(0, None)] * len(A)
    
    result = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds)
    return result.x
```
