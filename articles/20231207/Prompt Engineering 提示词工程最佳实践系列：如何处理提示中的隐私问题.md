```python
import random

def mask_data(data):
    masked_data = []
    for item in data:
        masked_item = []
        for value in item:
            masked_value = str(value) + str(random.randint(1, 100))
            masked_item.append(masked_value)
        masked_data.append(masked_item)
    return masked_data

data = [["name", "John"], ["age", 25], ["address", "123 Main St"]]
masked_data = mask_data(data)
print(masked_data)
```
