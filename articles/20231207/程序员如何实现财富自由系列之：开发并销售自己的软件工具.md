# 定义一个函数，用于计算两个向量之间的距离
def distance(vector1, vector2):
    # 计算两个向量之间的距离
    return np.linalg.norm(vector1 - vector2)

# 主函数
if __name__ == "__main__":
    # 定义两个向量
    vector1 = np.array([1, 2, 3])
    vector2 = np.array([4, 5, 6])

    # 计算两个向量之间的距离

    # 打印结果
    print(result)
