```python
import nltk
from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords

def handle_ambiguity(prompt):
    # 提取关键词
    keywords = word_tokenize(prompt)
    # 分析语法结构
    syntax_structure = nltk.pos_tag(keywords)
    # 设计合适的提示
    new_prompt = ""
    for word, tag in syntax_structure:
        if tag == "NN":
            new_prompt += word + " "
        else:
            new_prompt += word + " "
    return new_prompt

# 示例
prompt = "What is the capital of France?"
new_prompt = handle_ambiguity(prompt)
print(new_prompt)
```

在上述代码中，我们首先使用 NLTK 库来提取关键词和分析语法结构。然后，我们根据语法结构设计合适的提示。最后，我们返回新的提示。
