import tensorflow as tf

data = pd.read_csv('data.csv')

# 定义目标函数

objective += x

# 定义约束条件
constraint1 = pulp.LpConstraint(x, lowBound=10, upperBound=20)
constraint2 = pulp.LpConstraint(x, lowBound=20, upperBound=30)

# 添加约束条件

objective += constraint2
