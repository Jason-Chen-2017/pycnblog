```csharp
using UnityEngine;
using System.Collections;

public class GameController : MonoBehaviour {
    private GameObject player;
    private GameObject enemy;

    void Start () {
        player = GameObject.Find ("Player");
        enemy = GameObject.Find ("Enemy");
    }

    void Update () {
        // 更新玩家对象的状态
        player.transform.position = new Vector3 (player.transform.position.x + 1, player.transform.position.y, player.transform.position.z);

        // 更新敌人对象的状态
        enemy.transform.position = new Vector3 (enemy.transform.position.x - 1, enemy.transform.position.y, enemy.transform.position.z);

        // 检测玩家和敌人之间的碰撞
        if (Physics.CheckSphere (player.transform.position, 1f)) {
            Debug.Log ("玩家与敌人碰撞");
        }
    }
}
```

在这个代码实例中，我们创建了一个名为GameController的脚本，它包含了一个Start方法和一个Update方法。Start方法用于初始化游戏对象，Update方法用于更新游戏对象的状态和检测碰撞。
