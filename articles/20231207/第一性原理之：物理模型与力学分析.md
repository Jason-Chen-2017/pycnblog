```python
import numpy as np

def force(r, k):
    return -k * r

def energy(r, k):
    return 0.5 * k * r**2

def heat(r, k):
    return energy(r, k)

r = np.array([1.0, 2.0, 3.0])
k = 1.0

force_r = force(r, k)
energy_r = energy(r, k)
heat_r = heat(r, k)

print("Force:", force_r)
print("Energy:", energy_r)
print("Heat:", heat_r)
```
