import pandas as pd
import numpy as np

# 数据预处理
# 数据清洗
def clean_data(data):
    # 处理错误、异常、重复等问题
    data = data.dropna()  # 删除缺失值
    data = data.drop_duplicates()  # 删除重复值
    data = data.replace(np.nan, 0)  # 替换缺失值为0
    return data

# 数据转换
def transform_data(data):
    # 处理格式、类型、单位等问题
    data['age'] = data['age'].astype(int)  # 转换为整型
    data['gender'] = data['gender'].map({'M': 1, 'F': 0})  # 映射为0和1
    data['height'] = data['height'] / 100  # 单位转换
    return data

# 数据整合
def integrate_data(data):
    # 处理不同格式、类型、单位等问题
    data = pd.get_dummies(data, columns=['gender'])  # 一 hot编码
    data = data.fillna(0)  # 填充缺失值
    return data

# 数据清洗的具体实例
data = pd.read_csv('data.csv')
data = clean_data(data)
data = transform_data(data)
data = integrate_data(data)

# 输出结果
print(data)
