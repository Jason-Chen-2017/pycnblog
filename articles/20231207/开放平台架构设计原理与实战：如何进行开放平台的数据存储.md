# 4.具体代码实例和详细解释说明

在本节中，我们将通过一个具体的代码实例来详细解释如何进行开放平台的数据存储。

```python
import sqlite3
from sqlite3 import Error

def create_connection(database):
    conn = None
    try:
        conn = sqlite3.connect(database)
    except Error as e:
        print(e)

    return conn

def create_table(conn, table_name, columns):
    try:
        cursor = conn.cursor()
        cursor.execute(f"CREATE TABLE {table_name} ({columns})")
        conn.commit()
    except Error as e:
        print(e)

def insert_data(conn, table_name, data):
    try:
        cursor = conn.cursor()
        cursor.executemany(f"INSERT INTO {table_name} VALUES (?)", data)
        conn.commit()
    except Error as e:
        print(e)

def main():
    database = "example.db"
    table_name = "example_table"
    columns = ", ".join(["id INTEGER PRIMARY KEY", "name TEXT", "age INTEGER"])
    data = [("Alice", 25), ("Bob", 30), ("Charlie", 35)]

    conn = create_connection(database)
    create_table(conn, table_name, columns)
    insert_data(conn, table_name, data)

if __name__ == "__main__":
    main()
```

在上述代码中，我们首先创建了一个SQLite数据库连接，然后创建了一个名为"example_table"的表，其中包含了"id"、"name"和"age"三个列。最后，我们插入了一些数据到这个表中。
