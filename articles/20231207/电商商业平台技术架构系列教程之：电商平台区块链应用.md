### 4.1 创建一个简单的区块链

以下是创建一个简单的区块链的代码实例：

```python
import hashlib

# 创建一个简单的区块链
class BlockChain:
    def __init__(self):
        self.chain = []
        self.create_genesis_block()

    def create_genesis_block(self):
        genesis_block = {
            'index': 0,
            'timestamp': 1546300800,
            'data': 'Genesis Block',
            'nonce': 100,
            'hash': self.hash(genesis_block)
        }
        self.chain.append(genesis_block)

    def hash(self, block):
        block_string = str(block['index']) + str(block['timestamp']) + str(block['data']) + str(block['nonce'])
        return hashlib.sha256(block_string.encode()).hexdigest()

    def add_block(self, data):
        index = len(self.chain)
        timestamp = 1546300800 + index
        nonce = 0
        block = {
            'index': index,
            'timestamp': timestamp,
            'data': data,
            'nonce': nonce,
            'hash': self.hash(block)
        }
        self.chain.append(block)
        return block

# 创建一个区块链实例
block_chain = BlockChain()

# 添加一个交易
block_chain.add_block('First Transaction')

# 获取区块链数据
print(block_chain.chain)
```
