```python

    user_roles = {}

        user_roles[role_id] = True
    return user_roles
```

```python
def role_based_access_control(user_roles, resource_roles):
    access_rights = {}
    for role_id in user_roles:
        if role_id in resource_roles:
            access_rights[role_id] = resource_roles[role_id]
        else:
            access_rights[role_id] = []
    return access_rights
```
