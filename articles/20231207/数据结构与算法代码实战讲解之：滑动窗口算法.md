def sliding_window(arr, k):
    window_start = 0
    window_end = 0
    window_sum = 0
    max_sum = float('-inf')

    while window_end < len(arr):
        window_sum += arr[window_end]
        while window_sum > k:
            window_sum -= arr[window_start]
            window_start += 1
        max_sum = max(max_sum, window_sum)
        window_end += 1

    return max_sum
