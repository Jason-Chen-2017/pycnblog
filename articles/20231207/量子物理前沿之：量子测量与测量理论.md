# 测量量子态
qc.measure([0, 1], [0, 1])  # 对第0个量子比特和第1个量子比特进行测量

# 执行量子电路
simulator = Aer.get_backend('qasm_simulator')
job = simulator.run(qc)
result = job.result()

# 绘制测量结果的直方图
plot_histogram(result.get_counts())
