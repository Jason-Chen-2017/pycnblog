```python
# 深度优先搜索
def dfs(G, root):
    visited = set()
    stack = [root]
    while stack:
        node = stack.pop()
        if node not in visited:
            visited.add(node)
            stack.extend(G.neighbors(node))
    return visited

# 广度优先搜索
def bfs(G, root):
    visited = set()
    queue = [root]
    while queue:
        node = queue.pop(0)
        if node not in visited:
            visited.add(node)
            queue.extend(G.neighbors(node))
    return visited

# 发现社群
communities = []
for node in G.nodes():
    community = dfs(G, node)
    communities.append(community)
```
