```python
from sklearn.metrics.pairwise import pairwise_distances
from sklearn.metrics.pairwise import cosine_similarity

def recommend(user_id, user_ratings, movies):
    # 计算用户的兴趣
    user_similarities = cosine_similarity(user_ratings)

    # 计算用户与其他用户的相似性
    user_similarities = 1 - user_similarities

    # 计算新闻信息的相似性
    movie_similarities = cosine_similarity(movies)

    # 计算用户与新闻信息的相似性
    user_movie_similarities = pairwise_distances(user_ratings, movies)

    # 计算推荐的新闻信息
    recommended_movies = user_movie_similarities * user_similarities * movie_similarities

    # 返回推荐的新闻信息
    return recommended_movies
```

```python
from sklearn import svm
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

def classify(X_train, y_train, X_test):
    # 创建支持向量机模型
    model = svm.SVC()

    # 训练模型
    model.fit(X_train, y_train)

    # 预测测试集的结果
    y_pred = model.predict(X_test)

    # 计算分类准确率
    accuracy = accuracy_score(y_test, y_pred)

    # 返回分类准确率
    return accuracy
```
