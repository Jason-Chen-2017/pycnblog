import numpy as np
import pandas as pd
from scipy.optimize import minimize

# 读取考古遗迹数据

def objective_function(x):
    return np.sum((X * x - y) ** 2)

# 定义约束条件
def constraint_function(x):
    return np.sum(x) - 1

# 设置优化参数
bounds = ((0, None),) * X.shape[1]
constraints = ({'type': 'eq', 'fun': constraint_function},)
