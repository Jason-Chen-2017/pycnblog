   ```python
   import requests
   import json

   def get_user_data(username):
       url = f"https://graph.facebook.com/v12.0/{username}"
       headers = {
           'Authorization': 'Bearer {username}',
           'Content-Type': 'application/json',
       }
       response = requests.get(url, headers=headers)
       data = json.loads(response.text)
       return data['user']
   ```
