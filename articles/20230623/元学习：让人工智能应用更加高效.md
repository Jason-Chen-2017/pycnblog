```python
import torch.nn as nn
import torch.nn.functional as F

class TextClassification(nn.Module):
    def __init__(self):
        super(TextClassification, self).__init__()
        self.fc1 = nn.Linear(100, 5)
        self.fc2 = nn.Linear(5, 2)
        self.fc3 = nn.Linear(2, 1)

    def forward(self, x):
        x = F.relu(self.fc1(x))
        x = self.fc2(x)
        x = F.relu(x)
        x = self.fc3(x)
        return x
```
