import pandas as pd
from sklearn.model_selection import train_test_split

data = pd.read_csv("data.csv")
X = data["X_train"]
y = data["y_train"]

# 训练模型
model = LogisticRegression()
model.fit(X, y)

# 对模型进行预测
predictions = model.predict(X)

# 比较预测结果和实际结果
accuracy = accuracy_score(y, predictions)
print("Accuracy:", accuracy)

# 使用模型进行预测

# 计算准确率
