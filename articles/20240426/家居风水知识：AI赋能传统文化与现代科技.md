### 5.1 基于Python的家居风水评估系统

```python
# 导入必要的库
import numpy as np

# 定义房间类
class Room:
    def __init__(self, length, width):
        self.length = length
        self.width = width

    # 计算房间面积
    def area(self):
        return self.length * self.width

    # 计算房间周长
    def perimeter(self):
        return 2 * (self.length + self.width)

    # 计算房间形状因子
    def shape_factor(self):
        return 4 * np.pi * self.area() / self.perimeter()**2

# 创建房间对象
room = Room(5, 4)

# 打印房间信息
print("房间面积:", room.area())
print("房间周长:", room.perimeter())
print("房间形状因子:", room.shape_factor())
```
