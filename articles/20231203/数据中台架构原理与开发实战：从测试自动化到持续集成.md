```python
import csv
import json

def csv_to_json(csv_file, json_file):
    with open(csv_file, 'r') as f:
        reader = csv.reader(f)
        data = []
        for row in reader:
            data.append(dict(zip(reader[0], row)))

    with open(json_file, 'w') as f:
        json.dump(data, f)

csv_to_json('data.csv', 'data.json')
```
