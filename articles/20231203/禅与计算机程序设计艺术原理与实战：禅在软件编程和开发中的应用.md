```python

    return a + b

# 定义一个函数来实现减法操作
def subtract(a, b):
    return a - b

# 定义一个函数来实现乘法操作
def multiply(a, b):
    return a * b

# 定义一个函数来实现除法操作
def divide(a, b):
    return a / b

# 主函数
def main():
    # 输入两个数字
    num1 = int(input("请输入第一个数字："))
    num2 = int(input("请输入第二个数字："))

    # 选择运算符
    operator = input("请选择运算符（+、-、*、/）：")

    # 根据选择的运算符来调用相应的函数
    if operator == "+":
        result = add(num1, num2)
    elif operator == "-":
        result = subtract(num1, num2)
    elif operator == "*":
        result = multiply(num1, num2)
    elif operator == "/":
        result = divide(num1, num2)
    else:
        print("输入的运算符有误，请重新输入")
        return

    # 输出结果
    print("结果为：", result)

# 调用主函数
```
