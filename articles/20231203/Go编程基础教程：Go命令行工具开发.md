	flag.Parse()

	// 文件操作
	filename := flag.Arg(0)
	content, err := ioutil.ReadFile(filename)
	if err != nil {

		os.Exit(1)
	}

	// 错误处理
	fmt.Printf("Content: %s\n", string(content))
}
