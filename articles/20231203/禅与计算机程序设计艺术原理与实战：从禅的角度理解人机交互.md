```python
class Calculator:
    def __init__(self):
        self.input_num = 0
        self.operate_num = 0
        self.result = 0

    def input_number(self, num):
        self.input_num = num

    def operate(self, operate):
        self.operate_num = operate

    def equal(self):
        if self.operate_num == '+':
            self.result = self.input_num + self.operate_num
        elif self.operate_num == '-':
            self.result = self.input_num - self.operate_num
        elif self.operate_num == '*':
            self.result = self.input_num * self.operate_num
        elif self.operate_num == '/':
            self.result = self.input_num / self.operate_num

    def show_result(self):
        print(self.result)

calculator = Calculator()
calculator.input_number(5)
calculator.operate('+')
calculator.equal()
calculator.show_result()
```
