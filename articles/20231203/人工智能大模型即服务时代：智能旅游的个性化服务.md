```python
from sklearn.metrics.pairwise import cosine_similarity

# 降维
pca = PCA(n_components=2)
data_pca = pca.fit_transform(data)

# 计算用户之间的相似度
similarity = cosine_similarity(data_pca)

# 推荐
def recommend(user_id, top_n=10):
    user_data = data[user_id]
    user_pca = pca.transform([user_data])
    similarity_scores = list(enumerate(similarity[user_id]))
    similarity_scores = sorted(similarity_scores, key=lambda x: x[1], reverse=True)
    recommended_users = [i[0] for i in similarity_scores[1:top_n+1]]
    return recommended_users

# 测试
user_id = 0
print(recommend(user_id))
```
