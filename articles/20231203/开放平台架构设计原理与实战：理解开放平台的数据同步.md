```python
import socket

def send_data(data, host, port):
    # 创建一个TCP/IP套接字
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

    # 连接到服务器
    server_address = (host, port)
    sock.connect(server_address)

    try:
        # 发送数据
        sock.sendall(data)
    finally:
        # 关闭套接字
        sock.close()

def receive_data(host, port):
    # 创建一个TCP/IP套接字
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

    # 绑定套接字到服务器地址
    server_address = (host, port)
    sock.bind(server_address)

    # 开始监听
    sock.listen(1)

    while True:
        # 接收数据
        print('waiting for connection...')
        connection, client_address = sock.accept()
        try:
            # 接收数据
            data = connection.recv(1024)
            print('received {!r}'.format(data))
        finally:
            # 关闭套接字
            connection.close()

if __name__ == '__main__':
    send_data(b'Hello, world', 'localhost', 10000)
    receive_data('localhost', 10000)
```

在上述代码中，我们首先创建了一个TCP/IP套接字，然后连接到服务器。接下来，我们发送数据到服务器，并在服务器端接收数据。最后，我们关闭套接字。

### 4.3数据一致性

在本节中，我们将提供一个使用Paxos算法实现数据一致性的代码实例。

```python
import random

class Paxos:
    def __init__(self, nodes):
        self.nodes = nodes
        self.values = {}

    def propose(self, value):
        node = random.choice(self.nodes)
        self.values[node] = value
        self.round = 0
        while True:
            self.round += 1
            self.send_prepare(node)
            if self.accepted:
                return self.values[node]
            self.send_accept(node)

    def send_prepare(self, node):
        max_value = None
        for n in self.nodes:
            if n != node:
                value = self.values.get(n, None)
                if max_value is None or value > max_value:
                    max_value = value
        if max_value is None:
            self.values[node] = value
            self.accepted = True
        else:
            self.values[node] = max_value
            self.send_accept(node)

    def send_accept(self, node):
        max_value = None
        for n in self.nodes:
            if n != node:
                value = self.values.get(n, None)
                if max_value is None or value > max_value:
                    max_value = value
        if max_value is None:
            self.accepted = True
        else:
            self.values[node] = max_value

if __name__ == '__main__':
    nodes = ['node1', 'node2', 'node3']
    paxos = Paxos(nodes)
    value = paxos.propose('Hello, world')
    print(value)
```

在上述代码中，我们首先创建了一个Paxos实例，然后提出一个值。接下来，我们进行一轮Paxos算法，直到达到一致性。最后，我们输出一致的值。
