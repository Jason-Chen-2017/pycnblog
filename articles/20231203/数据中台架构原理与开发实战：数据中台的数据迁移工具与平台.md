假设我们需要将数据从一个MySQL数据库迁移到另一个MySQL数据库，我们可以使用以下代码实现数据迁移：

```python
import mysql.connector
from mysql.connector import Error

def migrate_data(source_host, source_user, source_password, source_database, target_host, target_user, target_password, target_database):
    try:
        # 创建源数据库连接
        source_connection = mysql.connector.connect(host=source_host, user=source_user, password=source_password, database=source_database)
        print("Source database connection established.")

        # 创建目标数据库连接
        target_connection = mysql.connector.connect(host=target_host, user=target_user, password=target_password, database=target_database)
        print("Target database connection established.")

        # 获取源数据库游标
        source_cursor = source_connection.cursor()

        # 获取目标数据库游标
        target_cursor = target_connection.cursor()

        # 获取源数据库表列表
        source_cursor.execute("SHOW TABLES")
        source_tables = source_cursor.fetchall()

        # 遍历源数据库表
        for table in source_tables:
            # 获取源数据库表结构
            source_cursor.execute(f"DESCRIBE {table[0]}")
            source_columns = source_cursor.fetchall()

            # 获取目标数据库表结构
            target_cursor.execute(f"DESCRIBE {table[0]}")
            target_columns = target_cursor.fetchall()

            # 判断目标数据库表是否存在
            if not target_columns:
                # 创建目标数据库表
                target_cursor.execute(f"CREATE TABLE {table[0]} ({', '.join([f'{column[0]} {column[1]}' for column in source_columns])})")
                target_connection.commit()

            # 获取源数据库表数据
            source_cursor.execute(f"SELECT * FROM {table[0]}")
            source_data = source_cursor.fetchall()

            # 插入目标数据库表数据
            target_cursor.executemany(f"INSERT INTO {table[0]} VALUES ({', '.join(['%s'] * len(source_columns))})")
            target_connection.commit()

        # 关闭数据库连接
        source_cursor.close()
        target_cursor.close()
        source_connection.close()
        target_connection.close()

        print("Data migration completed.")

    except Error as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    source_host = "localhost"
    source_user = "root"
    source_password = "123456"
    source_database = "test"
    target_host = "localhost"
    target_user = "root"
    target_password = "123456"
    target_database = "test"

    migrate_data(source_host, source_user, source_password, source_database, target_host, target_user, target_password, target_database)
```

在这个代码中，我们首先导入了`mysql.connector`模块，并定义了一个名为`migrate_data`的函数，该函数接受源数据库和目标数据库的连接信息作为参数。

然后，我们使用`mysql.connector.connect`方法创建了源数据库和目标数据库的连接，并获取了数据库游标。

接下来，我们使用`SHOW TABLES`语句获取了源数据库中的所有表，并遍历了每个表。对于每个表，我们首先获取了表的结构，然后判断目标数据库中是否存在该表。如果不存在，我们创建了目标数据库中的表。

然后，我们使用`SELECT * FROM`语句获取了源数据库中的表数据，并使用`INSERT INTO`语句将数据插入到目标数据库中。

最后，我们关闭了数据库连接和游标，并打印了数据迁移的完成信息。
