```python
import nltk
from nltk.tokenize import word_tokenize
from nltk.tag import pos_tag

def handle_missing_info(prompt):
    tokens = word_tokenize(prompt)
    tagged = pos_tag(tokens)

    missing_info = []
    for i in range(len(tagged)):
        if tagged[i][1] == 'NN' or tagged[i][1] == 'VB':
            missing_info.append(tagged[i][0])

    context_info = []
    for i in range(len(tagged)):
        if tagged[i][1] == 'DT' or tagged[i][1] == 'IN':
            context_info.append(tagged[i][0])

    return missing_info, context_info

prompt = "请帮我找到一本关于人工智能的书籍"
missing_info, context_info = handle_missing_info(prompt)

print("缺失信息：", missing_info)
print("上下文信息：", context_info)
```
