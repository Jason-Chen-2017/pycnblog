# 监控与日志
from prometheus import Prometheus

prometheus = Prometheus()
prometheus.add_metric("request_count", "requests_total", "method", "path")

from kibana import Kibana

logstash.add_input("product-service:8081")

logstash.add_output("elasticsearch")

kibana.add_index("product-service")
kibana.add_index("order-service")
