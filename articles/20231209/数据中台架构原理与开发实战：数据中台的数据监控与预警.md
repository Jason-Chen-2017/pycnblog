import logging
from data_monitor import DataMonitor

# 定义数据监控的规则
data_monitor = DataMonitor(data_source="db", data_type="temperature", time_range=(0, 24))

# 定义数据监控的触发条件
data_monitor.set_trigger_condition(temperature_threshold=30)

# 实现数据监控的实现
while True:
    # 获取数据
    data = data_monitor.get_data()

    # 处理数据
    if data is not None:
        # 判断数据是否超出阈值
        if data["temperature"] > data_monitor.temperature_threshold:
            # 触发预警
            data_monitor.trigger_alert(data)

    # 等待下一次监控
    time.sleep(1)
