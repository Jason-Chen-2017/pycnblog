```python
import neo4j

# 连接到Neo4j图数据库
driver = neo4j.GraphDatabase.driver("bolt://localhost:7687", auth=("neo4j", "your_password"))

# 构建图
def build_graph(data):
    # 创建节点
    for person in data:
        with driver.session() as session:
            session.run("CREATE (p:Person {name: $name})", name=person["name"])

    # 创建边
    for relation in data["relations"]:
        from_person_name = relation["from_person"]
        to_person_name = relation["to_person"]
        relation_type = relation["type"]
        relation_weight = relation["weight"]

        with driver.session() as session:
            session.run("MATCH (p1:Person {name: $from_person_name}), (p2:Person {name: $to_person_name}) MERGE (p1)-[:$type]->(p2) SET p1.$weight = $weight",
                        from_person_name=from_person_name, to_person_name=to_person_name, type=relation_type, weight=relation_weight)

# 测试数据
data = {
    "persons": [
        {"name": "Alice"},
        {"name": "Bob"},
        {"name": "Charlie"}
    ],
    "relations": [
        {"from_person": "Alice", "to_person": "Bob", "type": "FRIEND", "weight": 1},
        {"from_person": "Alice", "to_person": "Charlie", "type": "FRIEND", "weight": 1},
        {"from_person": "Bob", "to_person": "Charlie", "type": "FRIEND", "weight": 1}
    ]
}

# 构建图
build_graph(data)

# 关闭连接
driver.close()
```

import networkx as nx

def analyze_graph(data):
    # 构建网络
    G = nx.Graph()

    # 添加节点
    for person in data["persons"]:
        G.add_node(person["name"])

    # 添加边
    for relation in data["relations"]:
        from_person_name = relation["from_person"]
        to_person_name = relation["to_person"]
        relation_type = relation["type"]
        relation_weight = relation["weight"]

        G.add_edge(from_person_name, to_person_name, type=relation_type, weight=relation_weight)

    # 计算中心性
    centrality = nx.degree_centrality(G)

    # 返回结果
    return centrality

# 测试数据
