```python
import rdflib

from rdflib.namespace import RDF, RDFS

# 创建一个RDF图
g = Graph()

# 添加实体
g.add((rdflib.URIRef("http://example.com/entity/1"), RDF.type, Namespace("http://example.com/ns#Person")))
g.add((rdflib.URIRef("http://example.com/entity/2"), RDF.type, Namespace("http://example.com/ns#Organization")))

# 添加属性
g.add((rdflib.URIRef("http://example.com/entity/1"), Namespace("http://example.com/ns#name"), Literal("Alice")))
g.add((rdflib.URIRef("http://example.com/entity/2"), Namespace("http://example.com/ns#name"), Literal("Company")))

# 添加关系
g.add((rdflib.URIRef("http://example.com/entity/1"), Namespace("http://example.com/ns#worksFor"), rdflib.URIRef("http://example.com/entity/2")))

# 存储知识图谱到数据库
g.serialize(destination="knowledge_graph.ttl", format="ttl")
```

```python
import spacy

# 加载语言模型
nlp = spacy.load("en_core_web_sm")

# 文本预处理
text = "Alice works for Company."
doc = nlp(text)

# 词性标注
for token in doc:
    print(f"{token.text}: {token.pos_}")

# 命名实体识别
for entity in doc.ents:
    print(f"{entity.text}: {entity.label_}")

# 依存关系分析
for token in doc:
    for dep in token.dep_:
        print(f"{token.text}: {dep}")

# 语义分析结果输出
output = "Alice works for Company."
print(output)
```
