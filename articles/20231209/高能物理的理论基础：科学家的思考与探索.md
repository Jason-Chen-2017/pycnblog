import random

def init_random_generator(seed):
    random.seed(seed)

def generate_collision_event(probability_distribution):
    x = random.uniform(0, 1)
    for x_max, probability in probability_distribution.items():
        if x < probability:
            return x_max
    return None

def main():
    probability_distribution = {
        'x_max1': 0.1,
        'x_max2': 0.3,
        'x_max3': 0.6
    }
    init_random_generator(12345)
    for _ in range(1000):
        x_max = generate_collision_event(probability_distribution)
        if x_max is not None:
            print(x_max)

if __name__ == '__main__':
    main()
