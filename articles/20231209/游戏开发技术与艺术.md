## 4.1 计算3D模型的旋转

```python
import numpy as np

def rotate_3d(matrix, angle, axis):
    """
    计算3D模型的旋转
    """
    axis = axis / np.linalg.norm(axis)
    cos_angle = np.cos(angle)
    sin_angle = np.sin(angle)
    I = np.eye(4)
    axis_axis = axis * axis.T
    I_cos = I - cos_angle * axis_axis
    I_sin = sin_angle * axis_axis
    return np.dot(matrix, np.dot(I_cos, I + I_sin))
```

def texture_mapping(texture, normal, light_direction, camera_position):
    """
    计算纹理映射
    """
    n_dot_l = np.dot(normal, light_direction)
    if n_dot_l < 0:
        n_dot_l = 0
