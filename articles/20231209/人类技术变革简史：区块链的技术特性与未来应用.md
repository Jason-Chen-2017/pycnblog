## 1.4 区块链技术的具体代码实例和详细解释说明

以下是一个简单的区块链技术的具体代码实例和详细解释说明：

```python
import hashlib
import time

class Block:
    def __init__(self, index, previous_hash, timestamp, data, hash):
        self.index = index
        self.previous_hash = previous_hash
        self.timestamp = timestamp
        self.data = data
        self.hash = hash

    def calculate_hash(self):
        encoded_str = str(self.index) + str(self.previous_hash) + str(self.timestamp) + str(self.data)
        return hashlib.sha256(encoded_str.encode('utf-8')).hexdigest()

def create_genesis_block():
    return Block(0, "0", time.time(), "Genesis Block", "0")

def create_next_block(last_block):
    index = last_block.index + 1
    previous_hash = last_block.hash
    timestamp = time.time()
    data = "New Block"
    hash = last_block.calculate_hash()
    return Block(index, previous_hash, timestamp, data, hash)

def create_blockchain():
    genesis_block = create_genesis_block()
    blockchain = [genesis_block]
    return blockchain

def add_block_to_blockchain(blockchain, new_block):
    blockchain.append(new_block)
    return blockchain

def is_valid_block(candidate_block, last_block):
    if candidate_block.hash != candidate_block.calculate_hash():
        return False
    if candidate_block.previous_hash != last_block.hash:
        return False
    if candidate_block.index != last_block.index + 1:
        return False
    return True

def is_valid_blockchain(blockchain):
    for i in range(1, len(blockchain)):
        if not is_valid_block(blockchain[i], blockchain[i-1]):
            return False
    return True

def main():
    blockchain = create_blockchain()
    for i in range(10):
        new_block = create_next_block(blockchain[-1])
        blockchain = add_block_to_blockchain(blockchain, new_block)
    print(blockchain)
    print(is_valid_blockchain(blockchain))

if __name__ == '__main__':
    main()
```
