### 4.1数据预处理的代码实例

```python
import pandas as pd
import numpy as np

# 读取原始数据
data = pd.read_csv('data.csv')

# 数据清洗
data = data.dropna()

# 数据转换
data['age'] = data['age'].astype(int)

# 数据规范化
data['age'] = (data['age'] - np.mean(data['age'])) / np.std(data['age'])
```
