```python
import nltk
from nltk.tokenize import word_tokenize
from nltk.tag import pos_tag

def correct_syntax_error(text):
    tokens = word_tokenize(text)
    tagged = pos_tag(tokens)
    corrected_tokens = []

    for i in range(len(tagged)):
        if tagged[i][1] == 'NNP':
            corrected_tokens.append(tagged[i][0].capitalize())
        else:
            corrected_tokens.append(tagged[i][0].lower())

    corrected_text = ' '.join(corrected_tokens)
    return corrected_text

text = "I am going to the store to buy some groceries."
corrected_text = correct_syntax_error(text)
print(corrected_text)
```
