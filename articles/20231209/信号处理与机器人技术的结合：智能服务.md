```python
import numpy as np
import matplotlib.pyplot as plt

# 生成一段信号
t = np.linspace(0, 1, 100)
s = np.sin(2 * np.pi * 10 * t)

# 对信号进行滤波处理
filtered_s = s
filtered_s = np.convolve(filtered_s, np.ones(30) / 30)

# 绘制信号和滤波后的信号
plt.plot(t, s, label='Original Signal')
plt.plot(t, filtered_s, label='Filtered Signal')
plt.legend()
plt.show()
```
