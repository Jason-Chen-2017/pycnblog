        "action": "deny_loan"
    },

        "condition": "credit_score > 600",
        "action": "grant_loan"
    }
}

    "r21": {
        "condition": "age > 18",
        "action": "grant_loan"
    },
    "r22": {
        "condition": "age < 18",
        "action": "deny_loan"
    },
    "r23": {
        "condition": "credit_score > 600",
        "action": "grant_loan"
    }
}

# 抽取相同的规则

# 匹配规则
def match_rule(rule, data):
    condition = rule["condition"]
    action = rule["action"]
    if eval(condition):
        return action
    return None

# 组合规则
def combine_rules(rules):
    rule_set = set()
    for rule in rules:
        action = match_rule(rule, data)
        if action:
            rule_set.add(action)
    return rule_set

    "age": 20,
    "credit_score": 700
}
