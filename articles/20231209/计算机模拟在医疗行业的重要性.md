```python
import numpy as np
from scipy.integrate import odeint

# 定义微分方程
def dydt(y, t, a, b):
    dy = np.array([a * y[0] - b * y[1], a * y[1] + b * y[0]])
    return dy

# 初始条件
y0 = np.array([1, 0])

# 时间点
t = np.linspace(0, 10, 100)

# 参数
a = 0.5
b = 0.1

# 求解微分方程
y = odeint(dydt, y0, t, args=(a, b))

# 输出结果
print(y)
```

2. 然后，设置初始条件，以及时间点。
