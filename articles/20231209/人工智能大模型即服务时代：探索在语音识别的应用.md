import numpy as np
import tensorflow as tf
from flask import Flask, request

def preprocess(audio_file):
    y, sr = librosa.load(audio_file)
    y = librosa.effects.reduce_noise(y, sr)
    y = librosa.effects.trim(y)
    y = librosa.effects.normalize(y)
    return y, sr

# 模型训练
def train_model(audio_data, labels):
    model = tf.keras.models.Sequential([
        tf.keras.layers.Input(shape=(audio_data.shape[1],)),
        tf.keras.layers.Dense(256, activation='relu'),
        tf.keras.layers.Dense(128, activation='relu'),
        tf.keras.layers.Dense(64, activation='relu'),
        tf.keras.layers.Dense(32, activation='relu'),
        tf.keras.layers.Dense(16, activation='relu'),
        tf.keras.layers.Dense(1, activation='sigmoid')
    ])
    model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
    model.fit(audio_data, labels, epochs=10)
    return model
