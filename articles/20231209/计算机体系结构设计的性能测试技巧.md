```python
import time
import random

def test_function():
    # 模拟一个计算任务
    start_time = time.time()
    result = random.randint(1, 1000000)
    end_time = time.time()
    return result, end_time - start_time

def performance_test():
    # 模拟多个计算任务
    tasks = [test_function() for _ in range(1000)]
    # 计算吞吐量
    throughput = len(tasks) / sum(task[1] for task in tasks)
    # 计算平均延迟
    average_latency = sum(task[1] for task in tasks) / len(tasks)
    print("吞吐量:", throughput)
    print("平均延迟:", average_latency)

if __name__ == "__main__":
    performance_test()
```

1.Q: 性能测试与性能优化有什么区别？

2.Q: 性能测试和压力测试有什么区别？
A: 性能测试是用于评估系统在正常工作负载下的性能，而压力测试是用于评估系统在极高工作负载下的性能。
