import numpy as np
import random

def data_generate(data, num):
    new_data = []
    for _ in range(num):
        index = random.randint(0, len(data) - 1)
        new_data.append(data[index])
    return new_data

import cv2
import numpy as np

def data_extend(data, num):
    new_data = []
    for d in data:
        for _ in range(num):
            new_data.append(cv2.transpose(d))
            new_data.append(cv2.rotate(d, 90))
            new_data.append(cv2.resize(d, (int(d.shape[1] * 0.5), int(d.shape[0] * 0.5))))
    return new_data
```

import numpy as np
import random

def data_mix(data, num):
    new_data = []
    for d in data:
        for _ in range(num):
            new_data.append(d)
            new_data.append(d + np.random.normal(0, 1, d.shape))
            new_data.append(d + np.random.choice([-1, 1], size=d.shape, dtype=np.int))
    return new_data
```
