class RuleEngine:
    def __init__(self):
        self.rules = []

    def add_rule(self, name, condition, action):
        self.rules.append((name, condition, action))

    def process(self, data):
        for name, condition, action in self.rules:
            if condition(data):
                data = action(data)
                break
        return data
