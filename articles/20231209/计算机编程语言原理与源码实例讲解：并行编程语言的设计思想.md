```cpp
#include <mutex>

    std::mutex mtx;
    int sum = 0;
    int n = 10000;

    std::thread t1([&] {
        for (int i = 0; i < n; ++i) {
            std::lock_guard<std::mutex> lock(mtx);
            sum += i;
        }
    });

    std::thread t2([&] {
        for (int i = 0; i < n; ++i) {
            std::lock_guard<std::mutex> lock(mtx);
            sum += i * i;
        }
    });

    t1.join();
    t2.join();

    std::cout << "sum = " << sum << std::endl;

    return 0;
}
```
