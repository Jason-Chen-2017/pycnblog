data = data / data.max()

# 计算相似度
similarity = cosine_similarity(data)

# 推荐
def recommend(user_id, n_recommend):
    user_preferences = data.loc[user_id]
    similarities = similarity[user_id]
    recommendations = np.dot(similarities, data.T)
    top_n_recommendations = recommendations.nlargest(n_recommend)
    return top_n_recommendations.index
