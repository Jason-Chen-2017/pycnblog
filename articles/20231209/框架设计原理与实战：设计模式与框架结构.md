### 4.2日志框架的具体代码实例
以下是一个简单的日志框架的具体代码实例：

```python
# 日志模块
class Logger:
    def __init__(self):
        self.log_list = []

    def log(self, message):
        self.log_list.append(message)

# 日志输出模块
class Output:
    def __init__(self):
        self.output_list = []

    def output(self, message):
        self.output_list.append(message)

# 日志框架
class LogFramework:
    def __init__(self):
        self.logger = Logger()
        self.output = Output()

    def log(self, message):
        self.logger.log(message)
        self.output.output(message)

# 使用日志框架
framework = LogFramework()
framework.log("这是一个日志消息")
```
