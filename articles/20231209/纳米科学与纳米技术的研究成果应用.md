```python
import numpy as np
import scipy.constants as spc

def calculate_delta_g(T, R, ln_k):
    delta_g = - R * T * ln_k
    return delta_g

T = 300  # Kelvin
R = spc.R_gas  # Gas constant in J/(mol*K)
ln_k = np.log(K)  # Equilibrium constant

delta_g = calculate_delta_g(T, R, ln_k)
print("Delta G =", delta_g, "J/mol")
```
