## 4.1 代码实例

以下是一个具体的代码实例，用于监测老年人的心率、血压等生理指标：

```python
import time
import Adafruit_BME280
import Adafruit_ADS1x15

# 初始化传感器
bme280 = Adafruit_BME280.BME280()
ads1115 = Adafruit_ADS1x15.ADS1115()

# 设置传感器参数
bme280.sea_level_pressure_hPa = 1013.25
bme280.calibration_pressure_hPa = 1013.25
bme280.calibration_temp_C = 25

# 监测心率
def monitor_heart_rate():
    while True:
        # 获取心率数据
        heart_rate = ads1115.read_adc(0)
        # 处理心率数据
        heart_rate = heart_rate * 1000 / 65536
        # 发布心率数据
        print("Heart rate: ", heart_rate)
        time.sleep(1)

# 监测血压
def monitor_blood_pressure():
    while True:
        # 获取血压数据
        pressure = bme280.read_pressure()
        # 处理血压数据
        pressure = pressure * 100
        # 发布血压数据
        print("Blood pressure: ", pressure)
        time.sleep(1)

# 主函数
if __name__ == '__main__':
    # 启动心率监测
    monitor_heart_rate()
    # 启动血压监测
    monitor_blood_pressure()
```
