import cv2

# 读取图像

# 使用Canny边缘检测算法检测道路边缘
edges = cv2.Canny(gray, 50, 150)

lines = cv2.HoughLines(edges, 1, np.pi / 180, 200)

# 计算道路线的平均位置

# 绘制道路线
cv2.line(img, (avg_line[0][0], avg_line[0][1]), (avg_line[0][2], avg_line[0][3]), (0, 255, 0), 3)
