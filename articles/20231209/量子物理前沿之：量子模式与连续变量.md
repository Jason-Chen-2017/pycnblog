```python
# 定义函数
def f(x):
    return x**2 + 1

# 定义初始值

# 定义步长

# 定义迭代次数

# 初始化数组

dy = np.diff(y) / h

# 计算积分
z = np.cumsum(dy)

# 计算数值解

y_num[1:] = z[:-1]

print(x)
print(y)
print(y_num)
```
