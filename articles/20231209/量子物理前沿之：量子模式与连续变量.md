## 4.1量子模式的代码实例
以下是一个简单的量子模式代码实例，用于计算粒子在不同状态下的概率分布：

```python
import numpy as np

def wavefunction(x):
    A = 1
    B = 1
    return np.sqrt(A) * np.exp(B * x)

def probability_distribution(x):
    return np.abs(wavefunction(x)) ** 2

x = np.linspace(-10, 10, 1000)
probability = probability_distribution(x)

import matplotlib.pyplot as plt
plt.plot(x, probability)
plt.xlabel('x')
plt.ylabel('Probability')
plt.title('Quantum Model')
plt.show()
```

```python
import numpy as np

def pdf(x, mu, sigma):
    return 1 / (np.sqrt(2 * np.pi * sigma ** 2)) * np.exp(-(x - mu) ** 2 / (2 * sigma ** 2))

mu = 0
sigma = 1
x = np.linspace(-10, 10, 1000)
probability = pdf(x, mu, sigma)

import matplotlib.pyplot as plt
plt.plot(x, probability)
plt.xlabel('x')
plt.ylabel('Probability')
plt.title('Continuous Variable')
plt.show()
```
