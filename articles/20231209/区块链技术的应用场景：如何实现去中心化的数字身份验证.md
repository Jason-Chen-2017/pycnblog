```python
import hashlib

class Blockchain:
    def __init__(self):
        self.chain = []

    def add_block(self, data):
        prev_hash = self.chain[-1]['hash'] if self.chain else '0'
        hash_data = data + prev_hash
        hash_data_sha = hashlib.sha256(hash_data.encode()).hexdigest()
        self.chain.append({'data': data, 'hash': hash_data_sha})

    def get_chain(self):
        return self.chain

blockchain = Blockchain()
blockchain.add_block('第一块区块')
blockchain.add_block('第二块区块')
```
