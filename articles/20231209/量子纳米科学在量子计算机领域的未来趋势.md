import quantum_tk as qtk

class HadamardGate(qtk.QuantumGate):
    def __init__(self):
        super().__init__()

    def operation(self, qreg):
        for q in qreg:
            q.h()

# 创建一个量子寄存器
qreg = qtk.QuantumRegister(1)

# 实例化Hadamard门
hadamard_gate = HadamardGate()

# 应用Hadamard门
hadamard_gate.operation(qreg)

# 测量量子寄存器

print(result)
