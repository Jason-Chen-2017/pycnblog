def clean_data(data):
    # 处理缺失值
    data = data.fillna(data.mean())
    # 处理数据类型
    data = data.astype({'age': 'int', 'gender': 'category'})
    # 处理数据标准化
    data = data.apply(lambda x: (x - x.mean()) / x.std())
    return data

# 数据转换
def transform_data(data):
    # 一元一变
    data['age_group'] = pd.cut(data['age'], bins=[0, 18, 35, 50, 65, 100], labels=['0-18', '19-34', '35-49', '50-64', '65+'])
    # 二元一变
    data['gender_category'] = data['gender'].map({'male': 1, 'female': 0})
    # 多元一变
    data['region_category'] = data['region'].map({'east': 1, 'west': 2, 'north': 3, 'south': 4})
    return data

# 数据融合
def merge_data(customer_data, purchase_data, browse_data, social_data):
    # 合并客户信息和购买行为信息

    # 合并客户信息和浏览行为信息

    # 合并客户信息和社交媒体行为信息
    customer_social_data = pd.merge(customer_browse_data, social_data, on='customer_id')
    return customer_social_data
