import numpy as np
from scipy.integrate import quad

# 定义常数
k = 1.380649e-23  # 布林常数，J/K
h = 6.62607015e-34  # 普朗克常数，J·s
c = 2.99792458e8  # 光速，m/s
Tcmb = 2.725  # 宇宙背景辐射温度，K

# 计算辐射能量密度
def integrand(x):
    return 2 * np.pi**2 / 15 * (k * Tcmb / (h**3 * c**2)) * x**3 / (np.exp(x) - 1)

integral_energy_density, error = quad(integrand, 0, np.inf)
print("辐射能量密度：", integral_energy_density, "J/m^3")

# 计算辐射波长分布
def integrand_wavelength(u):
    return 8 * np.pi / c * u**3 / (np.exp(u) - 1)

u = h * c / (k * Tcmb)
integral_wavelength_distribution, error = quad(integrand_wavelength, 0, u)
print("辐射波长分布：", integral_wavelength_distribution, "1/m")

# 计算辐射波数分布
def integrand_frequency(u):
    return 8 * np.pi / c * u**3 / (np.exp(u) - 1)

u = h * c / (k * Tcmb)
u_frequency = u / (h / (k * Tcmb))
integral_frequency_distribution, error = quad(integrand_frequency, 0, u_frequency)
print("辐射波数分布：", integral_frequency_distribution, "1/Hz")

在上述代码中，我们首先定义了布林常数、普朗克常数、光速和宇宙背景辐射温度等常数。然后，我们定义了三个积分函数，分别用于计算辐射能量密度、辐射波长分布和辐射波数分布。最后，我们使用`scipy.integrate.quad`函数进行积分计算，并输出结果。
