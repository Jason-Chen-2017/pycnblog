import java.util.ArrayList;
import java.util.List;

public class Dijkstra {
    public static void main(String[] args) {
        // 初始化
        List<Node> nodes = new ArrayList<>();
        nodes.add(new Node("A", 0));
        nodes.add(new Node("B", 1));
        nodes.add(new Node("C", 2));
        nodes.add(new Node("D", 3));
        nodes.add(new Node("E", 4));

        // 选择起点
        Node start = nodes.get(0);

        // 选择最短路径
        Node current = start;
        while (current != null) {
            for (Node neighbor : current.neighbors) {
                if (neighbor.distance > current.distance + current.weight) {
                    neighbor.distance = current.distance + current.weight;
                    neighbor.previous = current;
                }
            }
            current = current.previous;
        }

        // 输出结果
        for (Node node : nodes) {
            System.out.println(node.name + ":" + node.distance);
        }
    }
}

class Node {
    String name;
    int distance;
    int weight;
    Node previous;
    List<Node> neighbors;

    public Node(String name, int weight) {
        this.name = name;
        this.weight = weight;
        this.neighbors = new ArrayList<>();
    }
}

        if j == 0:
            dp[i][j] = dp[i - 1][j + 1] + 1
        else:
