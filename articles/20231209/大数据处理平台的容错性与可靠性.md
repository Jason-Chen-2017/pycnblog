    # 模拟数据库操作
    from_balance = get_balance(from_account)
    to_balance = get_balance(to_account)

    if from_balance < amount:
        print("Insufficient balance in from account")
        return False

    # 更新from_account的余额
    update_balance(from_account, from_balance - amount)

    # 更新to_account的余额
    update_balance(to_account, to_balance + amount)

    return True

def get_balance(account):
    # 模拟数据库操作
    return get_account_balance(account)

def update_balance(account, balance):
    # 模拟数据库操作
    update_account_balance(account, balance)

# 创建两个线程，分别执行transfer_money函数
thread1 = threading.Thread(target=transfer_money, args=("from_account", "to_account", 100))
thread2 = threading.Thread(target=transfer_money, args=("from_account", "to_account", 100))

# 启动线程

thread2.join()
