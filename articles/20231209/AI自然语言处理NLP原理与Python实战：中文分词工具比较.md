import torch.nn as nn
import torch.optim as optim

        self.embedding = nn.Embedding(vocab_size, embedding_dim)

        self.fc = nn.Linear(hidden_dim * 2, output_dim)

    def forward(self, x):
        x = self.embedding(x)
        x, _ = self.lstm(x)
        x = self.fc(x)
        return x
