    noise = np.random.normal(0, 1, (batch_size, z_dim))
    # 生成图像
    generated_images = generator.predict(noise)
    # 训练判别器
    discriminator.trainable = True
    for epoch in range(num_epochs):
        # 训练判别器
        discriminator.trainable = True
        for _ in range(int(batch_size / 64)):
            # 训练判别器
            img_batch = generated_images[_]
            label = np.ones((batch_size, 1))
            discriminator.train_on_batch(img_batch, label)
        # 训练生成器
        discriminator.trainable = False
        for _ in range(int(batch_size / 64)):
            # 训练生成器
            noise_batch = np.random.normal(0, 1, (batch_size, z_dim))
            generated_images_batch = generator.predict(noise_batch)
            label = np.ones((batch_size, 1))
            discriminator.train_on_batch(generated_images_batch, label)

# 定义生成模型的主函数

    # 设置参数
    z_dim = 100
    batch_size = 64
    num_epochs = 100
    # 定义生成器和判别器网络
    generator = generator_network(z_dim)
    discriminator = discriminator_network((num_pixels, num_pixels, 1))
    # 训练生成器和判别器
