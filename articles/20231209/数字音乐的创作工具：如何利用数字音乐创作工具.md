```python
import numpy as np
import matplotlib.pyplot as plt

# 生成一个随机音频信号
def generate_audio_signal(length, amplitude, sample_rate):
    signal = np.random.rand(length) * amplitude
    return signal

# 对音频信号进行滤波处理
def filter_audio_signal(signal, cutoff_frequency, sample_rate):
    cutoff_frequency = np.pi * cutoff_frequency / sample_rate
    filtered_signal = signal
    for i in range(len(signal)):
        filtered_signal[i] = signal[i] * np.sin(cutoff_frequency * i)
    return filtered_signal

# 对多个音频信号进行混音处理
def mix_audio_signals(signals, sample_rate):
    mixed_signal = np.zeros(len(signals[0]))
    for i in range(len(signals)):
        mixed_signal += signals[i]
    return mixed_signal

# 生成一个简单的数字音乐作品
def generate_music_piece(length, amplitude, sample_rate, cutoff_frequency):
    signal = generate_audio_signal(length, amplitude, sample_rate)
    filtered_signal = filter_audio_signal(signal, cutoff_frequency, sample_rate)
    mixed_signal = mix_audio_signals([signal, filtered_signal], sample_rate)
    return mixed_signal

# 生成一个简单的数字音乐作品
length = 1000
amplitude = 0.5
sample_rate = 44100
cutoff_frequency = 2000

music_piece = generate_music_piece(length, amplitude, sample_rate, cutoff_frequency)

# 绘制音频信号的波形图
plt.plot(music_piece)
plt.xlabel('Time')
plt.ylabel('Amplitude')
plt.title('A Simple Digital Music Piece')
plt.show()
```
