```python
import numpy as np

class MDP:
    def __init__(self, states, actions, transition_prob, reward_prob):
        self.states = states
        self.actions = actions
        self.transition_prob = transition_prob
        self.reward_prob = reward_prob

    def transition(self, state, action):
        next_state = np.random.choice(self.states, p=self.transition_prob[state, action])
        reward = np.random.choice(np.arange(-1, 1), p=self.reward_prob[state, action])
        return next_state, reward
```

import numpy as np

class HMM:
    def __init__(self, states, transition_prob, emission_prob):
        self.states = states
        self.transition_prob = transition_prob
        self.emission_prob = emission_prob

    def predict(self, state, observation):
        next_state = np.random.choice(self.states, p=self.transition_prob[state])
