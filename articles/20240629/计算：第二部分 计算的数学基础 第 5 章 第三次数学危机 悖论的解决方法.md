    return y in x

def is_not_member(x, y):
    return not is_member(x, y)

def russell_paradox():
    R = {x for x in range(10) if is_not_member(x, range(10))}
    return R

- 由于我们在`is_not_member`函数中使用了`range(10)`，因此R集合中的元素只能取0-9之间的整数。
