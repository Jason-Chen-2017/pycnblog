import numpy as np
import matplotlib.pyplot as plt

# 创建神经元数组
neurons = np.array([[1, 0], [0, 1]])

# 创建连接矩阵
connections = np.array([[0, 1], [1, 0]])

# 迭代计算神经元激活

    activations = np.dot(neurons, connections)
    neurons = activations / np.sum(activations)

plt.show()
