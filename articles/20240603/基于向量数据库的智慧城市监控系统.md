from vectordb import VectorDB
from vectordb.index import FaissIndex

db = VectorDB()
index = FaissIndex()
db.create_index(index)

# 向数据库插入数据
db.insert({'id': 1, 'timestamp': 1234567890, 'location': [1.0, 2.0], 'data': [1, 2, 3]})

# 查询数据库

print(results)
