```python
import networkx as nx

G = nx.DiGraph()
G.add_edge('A', 'B', weight=1)
G.add_edge('B', 'C', weight=2)
G.add_edge('C', 'A', weight=3)
G.add_edge('A', 'D', weight=4)
G.add_edge('D', 'C', weight=5)

source = 'A'
target = 'C'

path = nx.dijkstra_path(G, source, target, weight='weight')
print(path)
```
