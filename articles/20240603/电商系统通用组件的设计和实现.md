```python
import hashlib
import hmac

def generate_hash(user_id):
    m = hashlib.sha256()
    m.update(user_id.encode('utf-8'))
    return m.hexdigest()

def encrypt_data(data, key):
    mac = hmac.new(key.encode('utf-8'), data.encode('utf-8'), hashlib.sha256)
    return mac.hexdigest()
```
