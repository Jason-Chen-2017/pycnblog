```python
class EcommerceTeam:
    def __init__(self, sales, customer_satisfaction, new_product_launch_speed, market_share):
        self.sales = sales
        self.customer_satisfaction = customer_satisfaction
        self.new_product_launch_speed = new_product_launch_speed
        self.market_share = market_share

    def evaluate_performance(self):
        sales_score = sum(self.sales)
        customer_satisfaction_score = sum(self.customer_satisfaction)
        new_product_launch_speed_score = sum(self.new_product_launch_speed)
        market_share_score = sum(self.market_share)

        total_score = (sales_score + customer_satisfaction_score + new_product_launch_speed_score + market_share_score) / 4

        return total_score
```
