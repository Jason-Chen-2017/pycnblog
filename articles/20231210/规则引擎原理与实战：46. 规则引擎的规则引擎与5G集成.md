```python
import requests
from requests.exceptions import RequestException

class RuleEngine:
    def __init__(self, api_url):
        self.api_url = api_url

    def get_rule(self, rule_id):
        try:
            response = requests.get(f'{self.api_url}/rule/{rule_id}')
            if response.status_code == 200:
                return response.json()
            else:
                raise RequestException(response.text)
        except RequestException as e:
            print(e)

    def create_rule(self, rule):
        try:
            response = requests.post(f'{self.api_url}/rule', json=rule)
            if response.status_code == 201:
                return response.json()
            else:
                raise RequestException(response.text)
        except RequestException as e:
            print(e)

    def update_rule(self, rule_id, rule):
        try:
            response = requests.put(f'{self.api_url}/rule/{rule_id}', json=rule)
            if response.status_code == 200:
                return response.json()
            else:
                raise RequestException(response.text)
        except RequestException as e:
            print(e)

    def delete_rule(self, rule_id):
        try:
            response = requests.delete(f'{self.api_url}/rule/{rule_id}')
            if response.status_code == 204:
                return True
            else:
                raise RequestException(response.text)
        except RequestException as e:
            print(e)
            return False

# 使用规则引擎与5G网络集成
rule_engine = RuleEngine('http://5g-network.com/api')

# 获取规则
rule = rule_engine.get_rule(1)
print(rule)

# 创建规则
rule = {
    'name': 'test_rule',
    'condition': 'event == "5G_network"',
    'action': 'send_alert'
}
new_rule = rule_engine.create_rule(rule)
print(new_rule)

# 更新规则
rule['name'] = 'updated_test_rule'
updated_rule = rule_engine.update_rule(1, rule)
print(updated_rule)

# 删除规则
deleted = rule_engine.delete_rule(1)
print(deleted)
```

在上述代码中，我们创建了一个名为`RuleEngine`的类，该类用于与5G网络集成。`RuleEngine`类提供了四个方法：`get_rule`、`create_rule`、`update_rule`和`delete_rule`。这些方法用于实现规则引擎与5G网络之间的数据交换、事件处理和错误处理。
