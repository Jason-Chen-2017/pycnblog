```python
# 创建规则引擎实例

# 创建规则
rule1 = engine.create_rule("rule1", "if age > 18 and gender = 'male' then grant access")

# 维护规则
engine.update_rule(rule1, "if age > 18 and gender = 'male' then grant access")
engine.delete_rule(rule2)

# 执行规则
engine.execute_rule("rule1")

# 监控规则
engine.monitor_rule("rule1")
```
