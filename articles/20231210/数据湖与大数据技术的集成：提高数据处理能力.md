```python
from hadoop.conf import Configuration
from hadoop.mapreduce import Job

# 创建一个Configuration实例
conf = Configuration()

# 创建一个Job实例
job = Job(conf)

# 设置输入和输出路径
job.set_input_path("/data_lake/data.txt")
```
