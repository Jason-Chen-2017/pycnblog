```python
import pandas as pd
import psycopg2

# 数据抽取
def extract_data(database_connection, table_name):
    query = f"SELECT * FROM {table_name};"
    data = pd.read_sql(query, database_connection)
    return data

# 数据转换
def transform_data(data, target_schema):
    data = data.rename(columns=target_schema)
    return data

# 数据加载
def load_data(data, target_table):
    data.to_sql(target_table, target_database_connection, if_exists='replace')

# 数据抽取
database_connection = psycopg2.connect(database="mydatabase", user="myuser", password="mypassword")
table_name = "mytable"
data = extract_data(database_connection, table_name)

# 数据转换
target_schema = {"column1": "column_1", "column2": "column_2"}
data = transform_data(data, target_schema)

# 数据加载
target_database_connection = psycopg2.connect(database="mytargetdatabase", user="mytargetuser", password="mypassword")
target_table = "mytargettable"
load_data(data, target_table)
```

在上述代码中，我们首先使用`extract_data`函数对数据库进行数据抽取。然后，我们使用`transform_data`函数对抽取到的数据进行转换。最后，我们使用`load_data`函数将转换后的数据加载到目标数据库中。
