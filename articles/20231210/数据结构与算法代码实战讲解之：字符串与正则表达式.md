# 字符串匹配算法
def kmp_match(pattern, text):
    pattern_len = len(pattern)
    text_len = len(text)
    next = [-1] * pattern_len
    j = -1
    for i in range(pattern_len):
        while j >= 0 and pattern[i] != pattern[j]:
            j = next[j]
        j += 1
        next[i] = j
    j = -1
    for i in range(text_len):
        while j >= 0 and pattern[j] != text[i]:
            j = next[j]
        j += 1
        if j == pattern_len:
            return i - j
    return -1

# 正则表达式匹配算法
def dfa_match(pattern, text):
    pattern_len = len(pattern)
    text_len = len(text)
    dfa = build_dfa(pattern)
    state = 0
    for i in range(text_len):
        c = text[i]
        if c in dfa[state]:
            state = dfa[state][c]
        else:
            state = -1
            break
    return state != -1

# 构建DFA
def build_dfa(pattern):
    states = {0: {}, 1: {}}
    for i, c in enumerate(pattern):
        if c not in states[0]:
            states[0][c] = i + 1
        if c not in states[1]:
            states[1][c] = i + 1
    states[0]['$'] = 1
    states[1]['$'] = 1
    return states
