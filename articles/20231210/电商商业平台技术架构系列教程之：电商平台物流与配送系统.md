```python
from requests.exceptions import RequestException

class DeliverySystem:
    def __init__(self, api_key):
        self.api_key = api_key

    def get_shipping_rates(self, origin, destination, weight, dimensions):
        url = f'https://api.example.com/shipping-rates?api_key={self.api_key}&origin={origin}&destination={destination}&weight={weight}&dimensions={dimensions}'
        try:
            response = requests.get(url)
            response.raise_for_status()
            return response.json()
        except RequestException as e:
            print(e)
            return None

    def get_shipping_labels(self, shipment_id, label_format):
        url = f'https://api.example.com/shipping-labels?api_key={self.api_key}&shipment_id={shipment_id}&label_format={label_format}'
        try:
            response = requests.get(url)
            response.raise_for_status()
            return response.json()
        except RequestException as e:
            print(e)
            return None

# 使用示例
delivery_system = DeliverySystem('your_api_key')
shipping_rates = delivery_system.get_shipping_rates('origin', 'destination', 10, '2x2x2')
shipping_labels = delivery_system.get_shipping_labels('shipment_id', 'pdf')
```
