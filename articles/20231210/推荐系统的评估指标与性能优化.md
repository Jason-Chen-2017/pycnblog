```python

# 电影的特征信息

    'movie_1': {
        'type': '科幻',
        'year': 2020,
        'rating': 8.5
    },
    'movie_2': {
        'type': '悬疑',
        'year': 2019,
        'rating': 7.8
    },
    'movie_3': {

        'year': 2021,
        'rating': 9.0

movie_feature_vectors = []
for movie_id, movie_info in movie_features.items():
    feature_vector = [movie_info['type'], movie_info['year'], movie_info['rating']]
    movie_feature_vectors.append(feature_vector)

movie_feature_matrix = np.array(movie_feature_vectors)
```

```python
    'user_1': {
        'genre': '科幻',
        'year': 2020,

    },
    'user_2': {
        'genre': '悬疑',
        'year': 2019,
        'rating': 7.8
    },
    'user_3': {
        'genre': '喜剧',
        'year': 2021,
        'rating': 9.0
    }
}

# 构建用户的兴趣向量
user_interest_vectors = []
for user_id, user_info in user_interests.items():
    interest_vector = [user_info['genre'], user_info['year'], user_info['rating']]
```

最后，我们可以使用余弦相似度来计算电影之间的相似度，并根据相似度排序，选择与用户兴趣最相似的电影，构建推荐列表。我们可以使用以下代码来实现这个功能：
