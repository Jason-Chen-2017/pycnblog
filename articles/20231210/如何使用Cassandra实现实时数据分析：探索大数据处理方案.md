    id UUID PRIMARY KEY,
    data TEXT
);

# 存储数据

# 分析数据
SELECT data FROM my_keyspace.my_table WHERE id = uuid();

# 数据分析算法

import numpy as np

data = pd.read_csv('data.csv')
data['data'] = data['data'].astype(float)

mean = np.mean(data['data'])
std = np.std(data['data'])

print('Mean:', mean)
print('Standard Deviation:', std)
