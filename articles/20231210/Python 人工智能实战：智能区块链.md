```python
from eth_account import Account
from eth_abi import encode_single

# 创建一个智能合约的实例
contract = web3.eth.contract(address=CONTRACT_ADDRESS, abi=CONTRACT_ABI)

# 创建一个账户实例
account = Account.create("PRIVATE_KEY")

# 调用智能合约的一个方法
method = contract.functions.some_method()
transaction = method.buildTransaction({
    'from': account.address,
    'gas': 200000,
    'gasPrice': web3.toWei('20', 'gwei'),
})
signed_transaction = account.signTransaction(transaction)
transaction_hash = web3.eth.sendRawTransaction(signed_transaction.rawTransaction)
transaction_receipt = web3.eth.waitForTransactionReceipt(transaction_hash)

# 解码智能合约的返回值
result = contract.events.SomeEvent.processReceipt(transaction_receipt)
```
