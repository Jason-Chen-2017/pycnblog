```python
import docker

class RuleEngine:
    def __init__(self):
        self.client = docker.from_env()

    def add_rule(self, rule):
        container = self.client.containers.create(image='rule-engine', rule=rule)
        container.start()

    def execute_rule(self, event):
        container = self.client.containers.get('rule-engine')
        response = container.exec_run('execute', event)
        return response.decode('utf-8')
```
