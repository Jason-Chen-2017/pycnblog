```go
package main

import (
    "fmt"
    "os"
    "path/filepath"
    "github.com/fsouza/go-dockerclient"
)

func main() {
    client, err := docker.NewClient(docker.Options{
        Host: "unix:///var/run/docker.sock",
    })
    if err != nil {
        fmt.Println(err)
        os.Exit(1)
    }

    image, err := client.ImageBuild(context.Background(), filepath.Join(os.Getcwd(), "Dockerfile"), nil)
    if err != nil {
        fmt.Println(err)
        os.Exit(1)
    }

    fmt.Println("Image built:", image.ID)
}
```

- 运行Docker容器：使用Go语言的并发模型，轻松地创建和管理Docker容器。具体代码实例如下：

```go
package main

import (
    "fmt"
    "os"
    "path/filepath"
    "github.com/fsouza/go-dockerclient"
)

func main() {
    client, err := docker.NewClient(docker.Options{
        Host: "unix:///var/run/docker.sock",
    })
    if err != nil {
        fmt.Println(err)
        os.Exit(1)
    }

    container, err := client.ContainerCreate(context.Background(), &docker.CreateContainerOptions{
        Image: image.ID,
    })
    if err != nil {
        fmt.Println(err)
        os.Exit(1)
    }

    fmt.Println("Container created:", container.ID)
}
```

- 管理Docker容器：使用Go语言的内置库和工具，轻松地进行Docker容器的管理和监控。具体代码实例如下：

```go
package main

import (
    "fmt"
    "os"
    "path/filepath"
    "github.com/fsouza/go-dockerclient"
)

func main() {
    client, err := docker.NewClient(docker.Options{
        Host: "unix:///var/run/docker.sock",
    })
    if err != nil {
        fmt.Println(err)
        os.Exit(1)
    }

    if err != nil {
        fmt.Println(err)

}
```
