```python
def recursive_function(n):
    if n == 1:
        return 1
    else:
        return n + recursive_function(n - 1)

def iterative_function(n):
    result = 0
    for i in range(1, n + 1):
        result += i
    return result

def dynamic_programming_function(n):
    dp = [0] * (n + 1)
    dp[1] = 1
    for i in range(2, n + 1):
        dp[i] = dp[i - 1] + i
    return dp[n]

print(recursive_function(5))  # 15
print(iterative_function(5))  # 15
print(dynamic_programming_function(5))  # 15
```
