def quick_sort(arr, left, right, less_than):
    if left < right:
        pivot_index = partition(arr, left, right, less_than)
        quick_sort(arr, left, pivot_index - 1, less_than)
        quick_sort(arr, pivot_index + 1, right, less_than)

def partition(arr, left, right, less_than):
    pivot = arr[right]
    i = left - 1
    for j in range(left, right):
        if less_than(arr[j], pivot):
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    arr[i + 1], arr[right] = arr[right], arr[i + 1]
    return i + 1

def less_than_int(a, b):
    return a < b

def less_than_float(a, b):
    return a < b

arr = [3, 1, 4, 1, 5, 9, 2]
quick_sort(arr, 0, len(arr) - 1, less_than_int)
print(arr)  # [1, 1, 2, 3, 4, 5, 9]

arr = [3.14, 2.71, 1.61, 0.5]
quick_sort(arr, 0, len(arr) - 1, less_than_float)
print(arr)  # [0.5, 1.61, 2.71, 3.14]
