```python

P = pydotplus.Dot(graph_type='digraph')
D = pydotplus.Dot(graph_type='digraph')

# 定义条件依赖关系
S.add_edge(P, D)

# 绘制贝叶斯网络
S.set_name('S')
P.set_name('P')
D.set_name('D')
```
