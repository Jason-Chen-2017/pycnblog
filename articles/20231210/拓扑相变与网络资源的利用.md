import heapq

def dijkstra(graph, start):
    distances = [float('inf')] * len(graph)
    distances[start] = 0
    pq = [(0, start)]
    while pq:
        current_distance, current_vertex = heapq.heappop(pq)
        if current_distance > distances[current_vertex]:
            continue
        for neighbor, weight in graph[current_vertex].items():
            distance = current_distance + weight
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(pq, (distance, neighbor))
    return distances

def kruskal(graph):
    roots = {i: i for i in range(len(graph))}
    edges = sorted(graph.items(), key=lambda x: x[1])
    result = []
    for edge in edges:
        u, v = edge[0], edge[1][0]
        root_u, root_v = find_root(roots, u), find_root(roots, v)
        if root_u != root_v:
            result.append(edge)
            union(roots, u, v)
    return result

def prim(graph):
    visited = [False] * len(graph)
    distances = [float('inf')] * len(graph)
    distances[0] = 0
    result = []
    while True:
        min_distance = float('inf')
        for i in range(len(graph)):
            if not visited[i] and distances[i] < min_distance:

        if min_distance == float('inf'):

        result.append(min_vertex)

            if not visited[neighbor] and distances[neighbor] > distances[min_vertex] + weight:
                distances[neighbor] = distances[min_vertex] + weight
    return result

def find_root(roots, vertex):
    while vertex != roots[vertex]:
        roots[vertex] = roots[roots[vertex]]
        vertex = roots[vertex]
    return vertex

    roots[find_root(roots, u)] = find_root(roots, v)
