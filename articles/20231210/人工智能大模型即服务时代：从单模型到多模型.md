import numpy as np

def model_fusion(predictions, weights):
    fusion_result = np.sum(predictions * weights) / np.sum(weights)
    return fusion_result

predictions = np.array([0.8, 0.7, 0.9])
weights = np.array([0.3, 0.4, 0.3])
fusion_result = model_fusion(predictions, weights)
print(fusion_result)
```

from sklearn.model_selection import cross_val_score
from sklearn.ensemble import RandomForestClassifier

def model_selection(X, y, model):
    scores = cross_val_score(model, X, y, cv=5)
    return np.mean(scores)

X = np.array([[1, 2], [3, 4], [5, 6]])
y = np.array([0, 1, 0])
model = RandomForestClassifier()
score = model_selection(X, y, model)
print(score)
```
