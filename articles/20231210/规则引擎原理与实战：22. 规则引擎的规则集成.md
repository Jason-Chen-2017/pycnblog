# 读取多个规则文件或规则集合

for rule_file in rules_files:
    # 读取规则文件
    with open(rule_file, 'r') as f:
        rules = f.readlines()

    # 解析规则文件
    for rule in rules:
        # 提取规则
        rule = Rule(rule)

        # 添加规则到规则库
        rule_library.add_rule(rule)

# 根据规则库中的规则来执行决策和操作
rule_library.execute()
