```python

    {"condition": "credit_score > 700 and income > 50000", "action": "approve"},
    {"condition": "credit_score <= 700 and income > 60000", "action": "approve"},
    {"condition": "credit_score <= 700 and income <= 60000", "action": "reject"},
]

# 读取问题输入
credit_score = 650
income = 55000

# 评估规则
for rule in rules:
    if eval(rule["condition"]):
        action = rule["action"]
        break
else:
    action = "unknown"

# 执行决策
if action == "approve":
    print("贷款批准")
else:
    print("贷款拒绝")
```
