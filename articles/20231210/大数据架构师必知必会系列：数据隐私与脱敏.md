import random

def replace_sensitive_info(info):
    # 对需要脱敏的个人信息进行分类
    sensitive_info = info['name'], info['age'], info['address']

    # 根据需要选择适合的脱敏算法
    replace_rules = {
        'name': lambda name: name.replace('*', '*'),
        'age': lambda age: str(age) + '*',
        'address': lambda address: address.replace('*', '*')
    }

    # 对需要脱敏的个人信息进行处理
    for key, value in sensitive_info.items():
        if key in replace_rules:
            sensitive_info[key] = replace_rules[key](value)

    # 对处理后的个人信息进行验证
    if all(value.isalpha() for value in sensitive_info.values()):
        return sensitive_info
    else:
        return None

# 测试代码
