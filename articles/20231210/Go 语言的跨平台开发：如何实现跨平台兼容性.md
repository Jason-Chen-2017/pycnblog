	}

		return

	fmt.Printf("文件大小: %d 字节\n", info.Size())

```go
package main

import (
	"fmt"
	"net"
)

func main() {
	conn, err := net.Dial("tcp", "localhost:8080")
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	defer conn.Close()

	conn.Write([]byte("Hello, World!\n"))
	buf := make([]byte, 1024)
	n, err := conn.Read(buf)
	if err != nil {
		fmt.Println("Error:", err)
		return
	}

	fmt.Println(string(buf[:n]))
}
```
