import numpy as np

def euler_method(f, x0, y0, h, t_end):
    t = np.arange(x0, t_end, h)
    y = np.zeros_like(t)
    y[0] = y0

    for i in range(len(t) - 1):
        y[i + 1] = y[i] + h * f(t[i], y[i])

    return t, y

def f(t, y):
    return y * np.exp(-t)

x0 = 0
y0 = 1
h = 0.1
t_end = 1

t, y = euler_method(f, x0, y0, h, t_end)

print("t:", t)
print("y:", y)

接下来，我们设置了初始条件`x0`、`y0`、步长`h`和结束时间`t_end`，并调用`euler_method`函数来解决微分方程问题。最后，我们打印出求解后的时间`t`和解`y`。
