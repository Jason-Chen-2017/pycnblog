```python
data = pd.DataFrame(query_result)
data = data.dropna()
data = data.fillna(0)

# 数据转换
data['node'] = data['node'].astype(str)
data['edge'] = data['edge'].astype(str)

# 数据聚合
data_agg = data.groupby('node').agg({'edge': 'count'})
```

```python
def analyze_information_propagation(G, start_node):
    # 使用BFS算法进行信息传播分析
    queue = [start_node]
    visited = set()
    visited.add(start_node)
    propagation_order = []
    while queue:
        node = queue.pop(0)
        propagation_order.append(node)
        for neighbor in G[node].neighbors():
            if neighbor not in visited:
                queue.append(neighbor)
                visited.add(neighbor)
    return propagation_order
```
