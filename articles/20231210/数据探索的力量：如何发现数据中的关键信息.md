```python
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# 数据清洗
data = pd.read_csv('data.csv')
data = data.dropna()  # 删除缺失值
data['age'] = data['age'].astype(int)  # 数据类型转换
data['gender'] = data['gender'].map({'M': 0, 'F': 1})  # 数据格式转换
data = data[data['age'] > 18]  # 数据过滤

# 数据可视化
sns.barplot(x='age', y='height', data=data)
plt.show()

# 数据聚类
from sklearn.cluster import KMeans
kmeans = KMeans(n_clusters=3)
kmeans.fit(data[['age', 'height']])
labels = kmeans.labels_

# 数据分析
from scipy.stats import pearsonr
r, p = pearsonr(data['age'], data['height'])
print('Pearson 相关系数:', r)
```
