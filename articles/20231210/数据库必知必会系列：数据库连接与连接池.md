```python
import mysql.connector

# 创建数据库连接池
def create_connection_pool(host, user, password, database):
    pool = mysql.connector.pooling.MySQLConnectionPool(
        pool_name='my_pool',
        pool_size=5,
        host=host,
        user=user,
        password=password,
        database=database
    )
    return pool

# 获取数据库连接
def get_connection(pool):
    connection = pool.get_connection()
    return connection

# 释放数据库连接
def release_connection(connection):
    pool = connection.get_connection_pool()
    pool.release_connection(connection)

# 主程序
if __name__ == '__main__':
    host = 'localhost'
    user = 'root'
    password = '123456'
    database = 'test'

    # 创建数据库连接池
    pool = create_connection_pool(host, user, password, database)

    # 获取数据库连接
    connection = get_connection(pool)

    # 执行数据库操作
    cursor = connection.cursor()
    cursor.execute('SELECT * FROM users')
    results = cursor.fetchall()

    # 释放数据库连接
    release_connection(connection)
```

在上述代码实例中，我们首先创建了一个数据库连接池，并设置了数据库连接池的大小为5。然后，我们通过`get_connection`函数来获取数据库连接，并执行数据库操作。最后，我们通过`release_connection`函数来释放数据库连接。
