    user_similarity = np.zeros((user_matrix.shape[0], user_matrix.shape[0]))

    for i in range(user_matrix.shape[0]):
        for j in range(i + 1, user_matrix.shape[0]):
            if user_neighbors[i][j] > 0:
                user_similarity[i, j] = user_similarity[j, i] = cosine(user_matrix[i, :], user_matrix[j, :])

    user_prediction = np.zeros((user_matrix.shape[0], item_matrix.shape[1]))

    return user_prediction
