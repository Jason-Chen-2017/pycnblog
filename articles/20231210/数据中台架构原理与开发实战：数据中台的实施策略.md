```python
import pandas as pd

# 连接数据源
df1 = pd.read_csv('data1.csv')
df2 = pd.read_csv('data2.csv')

# 映射数据源
df1.columns = ['id', 'name', 'age']
df2.columns = ['id', 'name', 'gender']

# 转换数据源
df1['gender'] = df1['id'].map(df2.set_index('id')['gender'])

# 合并数据源
df = pd.concat([df1, df2], ignore_index=True)

# 数据清洗
df = df.dropna()
df = df[df['age'] != '']
df = df[df['age'] != 'null']

# 数据存储
df.to_csv('data_integration.csv', index=False)
```
