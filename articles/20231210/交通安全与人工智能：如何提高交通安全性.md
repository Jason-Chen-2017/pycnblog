```python

# 定义控制策略的参数
control_parameters = np.array([0.1, 0.2, 0.3])

# 实现控制策略
def control_strategy(current_position, target_position, control_parameters):
    distance = np.linalg.norm(current_position - target_position)
    control_output = control_parameters * distance
    return control_output

# 显示控制策略的输出
print(control_strategy([10, 10], [20, 20], control_parameters))
```
