```python
import numpy as np
from scipy.optimize import linprog

def traffic_light_control(data):
    # 定义交通信号灯的状态
    states = ['red', 'yellow', 'green']
    # 定义交通信号灯的控制策略
    control_policy = np.zeros(len(states))
    # 定义交通信号灯的权重
    weights = np.ones(len(states))
    # 使用贪心算法求解最优控制策略
    result = linprog(control_policy, A_eq=weights, b_eq=data, bounds=[(0, 1) for _ in states])
    # 返回最优控制策略
    return result.x
```
