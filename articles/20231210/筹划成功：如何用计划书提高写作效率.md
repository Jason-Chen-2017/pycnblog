class PlanBook:
    def __init__(self, name, date, status):
        self.name = name
        self.date = date
        self.status = status
        self.projects = []

    def add_project(self, project):
        self.projects.append(project)

    def remove_project(self, project):
        self.projects.remove(project)

    def update_project(self, project, name, date, status):
        project.name = name
        project.date = date
        project.status = status

    def complete_project(self, project):
        project.status = "完成"

        self.projects.remove(project)

    def __init__(self, name, date, status):
        self.name = name
        self.date = date
        self.status = status
        self.tasks = []

    def add_task(self, task):
        self.tasks.append(task)

    def remove_task(self, task):
        self.tasks.remove(task)

    def update_task(self, task, name, date, status):
        task.name = name
        task.date = date
        task.status = status

    def complete_task(self, task):
        task.status = "完成"

    def delete_task(self, task):
        self.tasks.remove(task)

class Task:
    def __init__(self, name, date, priority, status):
        self.name = name
        self.date = date
        self.priority = priority
        self.status = status

    def update_task(self, name, date, priority, status):
        self.name = name
        self.date = date
        self.priority = priority
        self.status = status

    def complete_task(self):
        self.status = "完成"

    def delete_task(self):
        self.status = "删除"
