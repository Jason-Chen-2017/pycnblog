### 5.1数据预处理代码实例

```rust
fn preprocess_data(data: &mut Vec<f64>) {
    // 去除噪声
    data.retain(|&x| x.abs() < 10.0);

    // 填充缺失值
    for i in 0..data.len() {
        if data[i] == 0.0 {
            data[i] = data[i - 1];
        }
    }

    // 转换变量
    for i in 0..data.len() {
        data[i] = data[i].powi(2);
    }
}
```

### 5.2数据清洗代码实例

```rust
fn clean_data(data: &mut Vec<f64>) {
    // 数据校验
    for i in 0..data.len() {
        if data[i] < 0.0 {
            panic!("数据校验失败，数据值不能为负数");
        }
    }

    // 数据纠正
    for i in 0..data.len() {
        if data[i] > 100.0 {
            data[i] = 100.0;
        }
    }

    // 数据过滤
    data.retain(|&x| x > 10.0);
}
```
