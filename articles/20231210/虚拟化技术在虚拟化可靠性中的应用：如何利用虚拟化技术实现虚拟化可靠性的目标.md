```python
class VirtualMachine:
    def __init__(self, name, resources):
        self.name = name
        self.resources = resources

    def request_resource(self, resource):
        if resource in self.resources:
            self.resources.remove(resource)
        else:
            print(f"{self.name} 没有 {resource} 资源")

    def release_resource(self, resource):
        self.resources.append(resource)

class VirtualizationPlatform:
    def __init__(self, virtual_machines):
        self.virtual_machines = virtual_machines

    def allocate_resource(self, resource, virtual_machine):
        for vm in self.virtual_machines:
            if resource in vm.resources:
                vm.release_resource(resource)
                virtual_machine.request_resource(resource)
                print(f"{virtual_machine.name} 获得了 {resource} 资源")
                break
        else:
            print(f"没有找到 {resource} 资源的虚拟机")

    def deallocate_resource(self, resource, virtual_machine):
        for vm in self.virtual_machines:
            if resource in vm.resources:
                vm.release_resource(resource)
                virtual_machine.request_resource(resource)
                print(f"{virtual_machine.name} 释放了 {resource} 资源")
                break
        else:
            print(f"没有找到 {resource} 资源的虚拟机")

# 创建虚拟机
vm1 = VirtualMachine("vm1", ["CPU", "Memory", "Disk"])
vm2 = VirtualMachine("vm2", ["CPU", "Memory", "Disk"])

# 创建虚拟化平台
platform = VirtualizationPlatform([vm1, vm2])

# 分配资源
platform.allocate_resource("CPU", vm1)
platform.allocate_resource("Memory", vm2)

# 释放资源
platform.deallocate_resource("CPU", vm1)
platform.deallocate_resource("Memory", vm2)
```
