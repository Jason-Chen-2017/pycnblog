(1)数据预处理:对原始数据进行清洗、去除停用词、去除标点符号、去除数字等操作，确保数据质量。

```python
import re
import nltk
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer

nltk.download('punkt')
nltk.download('wordnet')

def preprocess(text):
    # 去除标点符号
    text = text.lower()
    # 去除数字
    text = re.sub(r'\d', '', text)
    # 去除停用词
    stop_words = set(stopwords.words('english'))
    text = [word for word in text.split() if word not in stop_words]
    # 词性标注
    lemmatizer = WordNetLemmatizer()
    text = [lemmatizer.lemmatize(word) for word in text]
    # 去停用词并词性标注后的文本
    return''.join(text)
```

(2)特征提取:从预处理后的文本数据中提取出对威胁情报有用的特征信息，如词汇、词性、词义等。

```python
import nltk
from nltk.corpus import wordnet

nltk.download('punkt')
nltk.download('wordnet')

def extract_features(text):
    # 词性标注
    lemmatizer = WordNetLemmatizer()
    text = [lemmatizer.lemmatize(word) for word in text]
    # 词频统计
    word_freq = nltk.FreqDist(text)
    # 取出词频大于1的词
    words = word_freq.keys()[word_freq.values()[0] > 1]
    # 词频统计结果
    freq_dict = word_freq.from_keys(words)
    # 词频
    word_freq = nltk.FreqDist(freq_dict)
    # 取出词频大于1的词
    words = word_freq.keys()[word_freq.values()[0] > 1]
    # 词性标注
    pos_dict = nltk.adapted.pos_tag(words)
    text = [[lemmatizer.lemmatize(word) for word, pos in pos_dict.items()] for word in text]
    # 词性标注结果
    word_性和lemmatizer_tag = nltk.adapted.pos_tag(text)
    return word_性和lemmatizer_tag
```

(3)建模训练:利用机器学习算法，对提取出的特征信息进行建模训练，得到模型参数。

```python
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.naive_bayes import MultinomialNB
from sklearn.metrics import accuracy_score

# 读取数据
data = pd.read_csv('data.csv')

# 数据划分
X = data.drop('label', axis=1)
y = data['label']

# 数据划分训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)

# 特征提取
features = extract_features(X_train)

# 模型选择
clf = MultinomialNB()

# 模型训练
clf.fit(X_train, y_train)

# 模型评估
y_pred = clf.predict(X_test)

# 计算准确率
accuracy = accuracy_score(y_test, y_pred)
print('Accuracy:', accuracy)
```
