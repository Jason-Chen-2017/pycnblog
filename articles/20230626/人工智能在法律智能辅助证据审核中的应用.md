import numpy as np
import tensorflow as tf
from tensorflow import keras
import os

# 读取数据
def read_data(data_dir):
    data = []
    for filename in os.listdir(data_dir):
        if filename.endswith('.txt'):
            with open(os.path.join(data_dir, filename), 'r', encoding='utf-8') as f:
                content = f.read()
            data.append(content)
    return data

# 特征提取

    features = []
    for word in nltk.word_tokenize(text):
        feature = np.array([int(word.lower() in [j, w]) for j in range(30)])
        features.append(feature)
    return features

# 模型训练
def train_model(data):
    model = keras.Sequential()
    model.add(keras.layers.Dense(32, activation='relu', input_shape=(None, 128)))
    model.add(keras.layers.Dense(64, activation='relu'))
    model.add(keras.layers.Dense(2, activation='softmax'))
    model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
    model.fit(data, epochs=5, batch_size=32)
    return model

# 对数据进行训练
data = read_data('text_data')
model = train_model(data)

# 应用模型进行审核
def apply_model(text):
    features = extract_features(text)
    predictions = model.predict(features)
    return np.argmax(predictions)

# 测试
test_text = '这是一份合同，您需要审核的内容如下：'
result = apply_model(test_text)
