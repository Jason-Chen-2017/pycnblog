```
public static String hashPassword(String password) {
    StringBuffer result = new StringBuffer();
    int len = password.length();
    for (int i = 0; i < len; i++) {
        result.append(password.charAt(i) ^ (int) (Math.random() * 128));
    }
    return result.toString();
}
```
