```
import numpy as np
import pandas as pd

def calculate_future_sales(data):
    future_sales = data.shift(-1).apply(lambda x: x[0] * np.exp(-0.01 * x[1]))
    return future_sales.shift(-1)

def calculate_future_cost(data):
    future_cost = data.shift(-1).apply(lambda x: x[0] * np.exp(-0.02 * x[1]))
    return future_cost.shift(-1)

def calculate_net_income(data):
    future_sales = calculate_future_sales(data)
    future_cost = calculate_future_cost(data)
    return (future_sales - future_cost)

def main():
    data = pd.read_csv('finance_data.csv')
    result = calculate_net_income(data)
    print(result)

if __name__ == '__main__':
    main()
```
