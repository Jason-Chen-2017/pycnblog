```python
import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler
from scipy.sparse import csr_matrix
from scipy.sparse.matrix import linalg
from scipy.sparse import linalg
import numpy as np

class DataProcess:
    def __init__(self, data):
        self.data = data

    def preprocess(self):
        self.data['age'] = np.array(self.data['age'].map(int))
        self.data['gender'] = np.array(self.data['gender'].map(str))
        self.data.dropna(inplace=True)
        self.data['消费金额'] = np.array(self.data['消费金额'].map(float))
        self.data['消费次数'] = np.array(self.data['消费次数'].map(int))

        # 身高处理
        self.data['height'] = (self.data['身高'] - 180) / 100 * 2

    def cluster(self):
        # 将数据进行归一化处理
        scaler = StandardScaler()
        self.data = pd.DataFrame(scaler.fit_transform(self.data), columns=['身高', '消费金额', '消费次数', '性别', '年龄'])

        # 构建相关性学习数据矩阵
        self.matrix = csr_matrix([self.data[['身高', '消费金额', '消费次数']], columns=['身高', '消费金额', '消费次数']).T)
        self.clusters = linalg.solve(self.matrix, k=0)

    def train(self):
        # 训练相关性学习模型
        self.model = linalg.solve(self.matrix, k=0)

    def test(self):
        # 测试相关性学习模型
        res = linalg.solve(self.model, k=0)

# 读取原始数据
data = pd.read_csv('data.csv')

# 进行数据预处理
process = DataProcess(data)
process.preprocess()

# 对数据进行聚类
process.cluster()

# 对数据进行训练
process.train()

# 对数据进行测试
print(res)
```
