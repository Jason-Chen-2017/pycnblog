4.3. 核心代码实现
```python
import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
import requests

class WearableData:
    def __init__(self, device, api_key):
        self.device = device
        self.api_key = api_key
        self.api_url = "https://api.example.com/data"
        self.data = []

    def collect_data(self):
        response = requests.get(self.api_url)
        if response.status_code == 200:
            data = response.json()
            self.data.append(data["data"])
        else:
            print(f"Error: {response.status_code}")

    def upload_data(self):
        response = requests.post(self.api_url, data=self.data, json=True)
        if response.status_code == 200:
            print("Data uploaded successfully.")
        else:
            print(f"Error: {response.status_code}")

# Example usage:
api_key = "your_api_key_here"
wearable = WearableData("SmartSensor", api_key)
wearable.collect_data()
wearable.upload_data()
```
