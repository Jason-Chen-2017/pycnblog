void init()  // 初始化游戏
{
    // 设置游戏变量初始值

    score = 0;
    obstacle[x][y] = 0;
}

void update()  // 更新游戏状态
{
    // 移动游戏角色
    y -= MOVING_SPEED;

    // 检查游戏角色是否撞墙
    if (y < 0 || y > HEIGHT)
    {
        return;
    }

    // 检查游戏角色是否撞上障碍物
    for (int z = 0; z < HEIGHT; z++)
    {
        if (y == obstacle[x][z])
        {
            return;
        }
    }

    // 如果游戏角色没有撞墙,就可以更新游戏分数
    if (y > HEIGHT - 1)
    {
        score++;
    }
