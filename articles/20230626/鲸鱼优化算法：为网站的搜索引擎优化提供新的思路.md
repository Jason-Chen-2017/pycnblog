### 4.3 核心代码实现

```
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np

class DeepSEO(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(DeepSEO, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_size)
        self.fc2 = nn.Linear(hidden_size, output_size)

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        x = self.fc2(x)
        return x

# 读取数据集
train_data = []
test_data = []
with open('train_data.txt', 'r') as f:
    for line in f:
        train_data.append([float(line.strip()), int(line.strip())])
with open('test_data.txt', 'r') as f:
    for line in f:
        test_data.append([float(line.strip()), int(line.strip())])

# 训练数据
train_inputs = []
train_outputs = []
for line in train_data:
    input_data, output_data = line
    train_inputs.append(input_data)
    train_outputs.append(output_data)

# 测试数据
test_inputs = []
test_outputs = []
for line in test_data:
    input_data, output_data = line
    test_inputs.append(input_data)
    test_outputs.append(output_data)

# 设置神经网络参数
input_size = len(train_inputs[0])
hidden_size = 64
output_size = len(test_outputs[0])
learning_rate = 0.01

# 创建神经网络实例
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
model = DeepSEO(input_size, hidden_size, output_size)

# 定义损失函数和优化器
criterion = nn.MSELoss()
optimizer = optim.SGD(model.parameters(), lr=learning_rate, momentum=0.9)

# 训练模型
for epoch in range(100):
    running_loss = 0.0
    for i, input_data in enumerate(train_inputs):
        output_data = train_outputs[i]
        model.zero_grad()
        output = model(input_data)
        loss = criterion(output, output_data)
        loss.backward()
        optimizer.step()
        running_loss += loss.item()

    print('Epoch {}: running loss = {}'.format(epoch+1, running_loss))

# 测试模型
correct = 0
total = 0
with torch.no_grad():
    for input_data, output_data in test_inputs:
        output = model(input_data)
        _, predicted = output.max(1)
        total += output.size(0)
        correct += (predicted == output_data).sum().item()

print('Accuracy on test data = {}%'.format(100*correct/total))
```
