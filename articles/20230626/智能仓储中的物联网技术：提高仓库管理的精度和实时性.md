4.3. 核心代码实现

```python
import numpy as np
import pandas as pd
import json
import requests
import time

class IoTWarehouse:
    def __init__(self, host='192.168.0.100', port=5000):
        self.host = host
        self.port = port
        self.data = {}

    def send_data(self, data):
        response = requests.post('http://' + self.host + ':' + str(self.port) + '/api/data', data=data)
        return response.content

    def receive_data(self):
        while True:
            data = self.receive_data()
            if data:
                print('接收到数据：', data)
                self.data = data
                
    def receive_data_realtime(self):
        while True:
            data = self.receive_data_realtime()
            if data:
                print('接收到数据：', data)
                
    def initialize(self):
        while True:
            self.receive_data_realtime()
            time.sleep(10)

    def query_data(self, data_key):
        return self.data.get(data_key)

    def create_data(self, data_key, data):
        self.data[data_key] = data

    def read_data(self, data_key):
        return self.query_data(data_key)

    def write_data(self, data_key, data):
        self.create_data(data_key, data)

# 示例：发送数据
iot_warehouse = IoTWarehouse()
data = {'product_id': 1, 'name': 'Apple', 'quantity': 10, 'price': 20, 'inventory_quantity': 0, 'inventory_price': 100}
iot_warehouse.send_data(data)

# 示例：接收数据并打印
print('接收到数据：', iot_warehouse.receive_data())

# 示例：查询数据
data_key = 'product_id'
print('查询数据：', iot_warehouse.query_data(data_key))

# 示例：创建数据
iot_warehouse.create_data('product_id', 5)

# 示例：写入数据
iot_warehouse.write_data('product_id', 6)
```
