import random
import numpy as np
import math

# 生成足够长的股票数据
stock_data = generate_stock_data(1000, 10)

# 将生成的股票数据序列归一化到[-1, 1]区间
stock_data_norm = (stock_data - 0.5) / 0.5

# 根据股票数据序列生成随机过程
def generate_random_process(stock_data_norm):
    # 定义随机过程
    def stock_process(t):
        # 定义一维随机变量
        noise = np.random.normal(scale=0.1, size=1)
        # 定义一阶矩
        C = 0.025
        # 计算一阶矩
        M1 = np.cumsum(noise**2)
        M2 = 2 * np.cumsum((noise - C)**2)
        # 计算二阶矩
        S2 = 2 * M1 * (1 - M2)
        # 计算随机变量
        return C + S2 * np.sin(0.01 * (t - T) / 100)
    
    # 生成随机过程
    process = stock_process(len(stock_data_norm))
    
    # 将随机过程以T为周期，生成足够长的数据
    process_T = 2000
    process_steps = int(process_T / len(stock_data_norm))
    process_data = []
    for i in range(process_steps):
        process_data.append(process.next())
    
    # 计算预测误差
    predictions = [math.random.normal(scale=1, size=1) for _ in range(len(process_data))]
    error = [math.sqrt(sum((prediction - stock_data)**2)) for prediction in predictions]
    
    return process_data, predictions, error

# 生成足够长的股票数据
process_data, predictions, error = generate_random_process(stock_data)

# 计算预测误差
predictions_sqrt_error = [math.sqrt(sum((prediction - stock_data)**2)) for prediction in predictions]

# 绘制预测误差
import matplotlib.pyplot as plt
plt.plot(len(process_data), predictions_sqrt_error)
plt.xlabel('步数')
plt.ylabel('预测误差')
plt.show()
