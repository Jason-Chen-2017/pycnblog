import datetime

        self.name = None
        self. fingerprint = None
        self.arrival_time = None
        self.leave_time = None

    def set_fingerprint(self, fingerprint):
        self.fingerprint = fingerprint

    def check_fingerprint(self):
        return self.fingerprint == self.name

# Test code for checking fingerprint
def test_fingerprint(fingerprint):
    print("Fingerprint:", fingerprint)
    time.sleep(10)
    print("Checking fingerprint...")
    if fingerprint == self.name:
        print("Fingerprint is correct!")
    else:
        print("Fingerprint is incorrect!")

# Main code for attendance system
def main():
    while True:
        arrival_time = input("Please enter arrival time (YYYY-MM-DD HH:MM:SS): ")
        leave_time = input("Please enter leave time (YYYY-MM-DD HH:MM:SS): ")
        try:
            fingerprint = test_fingerprint(arrival_time + " " + leave_time)
            attendance = Attendance()
            attendance.set_fingerprint(fingerprint)
            attendance.check_fingerprint()
