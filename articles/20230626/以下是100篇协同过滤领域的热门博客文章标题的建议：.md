  user_data = pd.read_csv('user_data.csv')
  item_data = pd.read_csv('item_data.csv')
  
  # 数据预处理
  user_data['user_id'] = user_data.apply(lambda x: x.split('_')[0], axis=1)
  user_data['item_id'] = user_data.apply(lambda x: x.split('_')[1], axis=1)
  
  # 用户行为模型
  user_behavior = user_data.apply(lambda x: x.items, axis=1)
  user_behavior = user_behavior.apply(lambda x: x[1])
  user_behavior = user_behavior.apply(lambda x: x[0])
  
  # 推荐模型
  classifier = MultinomialNB()
  recommendations = classifier.fit_predict(user_behavior)
  
  # 测试与评估
  tt = np.array([accuracy_score(user_data['item_id'], recommendations)])
  print('Test TP:', tt[0][0])
  print('Test TN:', tt[0][1])
  print('Test FP:', tt[0][2])
