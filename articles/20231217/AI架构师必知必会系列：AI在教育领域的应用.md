```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 文本数据预处理
vectorizer = TfidfVectorizer()
X = vectorizer.fit_transform(data['resource_description'])

# 计算资源之间的相似度
similarity_matrix = cosine_similarity(X)

# 根据学生的学习历史，推荐个性化的学习资源
def recommend_resource(student_history):
    student_history_vector = vectorizer.transform(student_history)
    similarity_to_resources = similarity_matrix[student_history_vector]
    recommended_resources = similarity_to_resources.argsort()[::-1]
    return recommended_resources
```
