from datetime import datetime, timedelta

def task_scheduler(task, interval, start_time):

        current_time = datetime.now()
        if (current_time - start_time) % interval == 0:
            task()
            start_time = current_time

2. 定义一个任务函数，该函数将在定时任务中执行。
