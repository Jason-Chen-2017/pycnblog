## 4.2 漏洞挖掘代码实例

### 4.2.1 数据流分析实例

```python
import re

def data_flow_analysis(code):
    patterns = [
        r'.*leak\(.*\).*',
        r'.*overflow\(.*\).*',
    ]
    results = []
    for pattern in patterns:
        matches = re.findall(pattern, code)
        if matches:
            results.append((pattern, matches))
    return results

# 使用
code = '''
def leak(data):
    print(data)
    return data

def overflow(buffer, data):
    buffer += data
    return buffer
'''
results = data_flow_analysis(code)
print(results)
```

### 4.2.2 控制流分析实例

```python
import re

def control_flow_analysis(code):
    patterns = [
        r'.*if\s+\(.*\).*else.*',
        r'.*while\s+\(.*\).*',
    ]
    results = []
    for pattern in patterns:
        matches = re.findall(pattern, code)
        if matches:
            results.append((pattern, matches))
    return results

# 使用
code = '''
if True:
    print('if true')
else:
    print('if false')

while True:
    print('while true')
'''
results = control_flow_analysis(code)
print(results)
```
