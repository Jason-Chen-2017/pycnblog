```python
# 规则引擎的具体代码实例

# 规则编写
class Rule:
    def __init__(self, condition, action):
        self.condition = condition
        self.action = action

# 规则执行
def execute_rule(rules, facts):
    for rule in rules:
        if rule.condition(facts):
            rule.action(facts)

# 规则管理
class RuleBase:
    def __init__(self):
        self.rules = []

    def add_rule(self, rule):
        self.rules.append(rule)

    def remove_rule(self, rule):
        self.rules.remove(rule)

# 规则查询
class RuleQuery:
    def __init__(self, rule_base):
        self.rule_base = rule_base

    def query(self, condition):
        return [rule for rule in self.rule_base.rules if condition(rule.condition)]

# 使用规则引擎
rule_base = RuleBase()
rule_base.add_rule(Rule(lambda x: x > 10, lambda x: print(x, "大于10")))
rule_base.add_rule(Rule(lambda x: x < 5, lambda x: print(x, "小于5")))
rule_base.add_rule(Rule(lambda x: x == 10, lambda x: print(x, "等于10")))

facts = [5, 10, 15]
execute_rule(rule_base.rules, facts)
```
