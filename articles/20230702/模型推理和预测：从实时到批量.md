4.3. 核心代码实现
---------------------

以下是实现实时推理和批量预测的核心代码实现：

```python
import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

# 读取数据
data = pd.read_csv('data.csv')

# 将数据分为训练集和测试集
train, test = train_test_split(data, test_size=0.2, random_state=0)

# 准备输入数据
input_data = keras.utils.to_categorical(train.drop(['label'], axis=1), num_classes=len(data))

# 准备目标数据
target_data = keras.utils.to_categorical(test.drop(['label'], axis=1), num_classes=len(data))

# 构建模型
model = Sequential()
model.add(Dense(32, activation='relu', input_shape=(input_data.shape[1],)))
model.add(Dropout(0.2))
model.add(Dense(32, activation='relu'))
model.add(Dropout(0.2))
model.add(Dense(1, activation='sigmoid'))

# 编译模型
model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])

# 训练模型
model.fit(train, target_data, epochs=50, batch_size=32, validation_split=0.1)

# 在测试集上进行预测
predictions = model.predict(test)

# 计算准确率
accuracy = accuracy_score(test.label, predictions)
print('Accuracy: {:.2f}%'.format(accuracy * 100))
```
