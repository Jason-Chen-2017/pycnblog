```
import nltk
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.pipeline import make_pipeline

nltk.download('punkt')
nltk.download('wordnet')

def preprocess_text(text):
    # 去除停用词
    nltk.download('stopwords')
    stop_words = set(stopwords.words('english'))
    return [word for word in nltk.word_tokenize(text) if word not in stop_words]

def main():
    # 加载数据
    data = open('data.txt')
    # 数据清洗
    data = [line.strip() for line in data]
    # 数据集划分
    X = []
    y = []
    for line in data:
        if len(line.split('    ')) > 1:
            feature, label = line.split('    ')
            X.append(feature.lower())
            y.append(label)
    # 特征工程
    vectorizer = CountVectorizer()
    X = vectorizer.fit_transform(X)
    # 标签划分
    clf = MultinomialNB()
    y_result = clf.fit_predict(X)
    # 输出结果
    for label in y_result:
        print(' '.join([x for x in vectorizer.transform([label])]))

if __name__ == '__main__':
    main()
```
