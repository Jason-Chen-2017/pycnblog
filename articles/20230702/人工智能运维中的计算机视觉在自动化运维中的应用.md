```python
import numpy as np

# 加载图像
img = cv2.imread('image.jpg')

# 特征点检测
ret, thresh = cv2.threshold(img, 20, 255, cv2.THRESH_BINARY)

# 寻找轮廓
contours, hierarchy = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

# 分类
classifier = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml', cv2.CASCADE_SCALE_IMAGE)
objects = classifier.detectMultiScale(thresh, 1.3, 5)

# 绘制轮廓
img = cv2.drawContours(img, contours, -1)

# 显示结果
cv2.imshow('image', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
