```
// Payment request
public class PaymentRequest {
    private String paymentMethod;
    private double amount;
    private String paymentToken;

    public PaymentRequest(String paymentMethod, double amount, String paymentToken) {
        this.paymentMethod = paymentMethod;
        this.amount = amount;
        this.paymentToken = paymentToken;
    }

    public String getPaymentMethod() {
        return paymentMethod;
    }

    public double getAmount() {
        return amount;
    }

    public String getPaymentToken() {
        return paymentToken;
    }
}

// Payment server response
public class PaymentServerResponse {
    private String success;
    private PaymentRequest request;

    public PaymentServerResponse(String success, PaymentRequest request) {
        this.success = success;
        this.request = request;
    }

    public String getSuccess() {
        return success;
    }

    public PaymentRequest getRequest() {
        return request;
    }
}

// Payment transaction
public class PaymentTransaction {
    private PaymentServerResponse response;
    private PaymentRequest request;

    public PaymentTransaction(PaymentServerResponse response, PaymentRequest request) {
        this.response = response;
        this.request = request;
    }

    public PaymentServerResponse getResponse() {
        return response;
    }

    public PaymentRequest getRequest() {
        return request;
    }
}
```
