// 线性搜索算法
public int linearSearch(int[] data, int size) {
    int index = -1;
    for (int i = 0; i < size - 1; i++) {
        if (data[i] == data[i + 1]) {
            index = i;
            break;
        }
    }
    return index;
}
```
3.2.2. 快速排序算法
```
// 快速排序算法
public void quickSort(int[] data, int size) {
    int n = size - 1;
    int i = 0;
    int j = n - 1;
    while (i < j) {
        while (i < j && data[i] < data[j]) {
            j--;
        }
        data[i], data[j] = data[j], data[i];
        while (i < j && data[i] > data[j]) {
            i++;
        }
        data[j], data[i] = data[i], data[j];
        i++;
        j--;
    }
}
```

```
// 动态规划算法
public int dynamicPow(int base, int n) {
    if (n == 0) {
        return 1;
    }
    int result = 1;
    for (int i = 1; i <= n; i++) {
        result *= base;
        result /= i;
    }
    return result;
}
