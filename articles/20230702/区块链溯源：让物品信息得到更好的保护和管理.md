typedef enum {
    INITIAL_BLOCK,
    ADD_BLOCK,
    UPDATE_BLOCK,
    DELETE_BLOCK

typedef struct {
    uint256 id;
    string name;
    string price;
    string source;
    string destination;
} item;

block_type current_block_type = INITIAL_BLOCK;

// 存储物品信息的结构体
struct item_info {
    item id;
    string name;
    string price;
    string source;
    string destination;
};

// 存储物品信息的数组
item_info items[10000];

// 存储物品信息的数据结构体
struct item_struct {
    item_info data;
};

items[1] = new_item;

// 将新添加的物品信息存储到链表中
void add_item(block_type type, item_struct item) {
    if (current_block_type == ADD_BLOCK) {
        items[sizeof(items)/sizeof(items[0]) - 1] = item;
    }
}

// 修改物品信息
void update_item(block_type type, item_struct item) {
    if (current_block_type == UPDATE_BLOCK) {
        for (int i = 0; i < sizeof(items); i++) {
            if (items[i].id == item.id) {
                items[i].data = item;
                break;
            }
        }
    }
}

// 删除物品信息
void delete_item(block_type type, uint256 id) {
    if (current_block_type == DELETE_BLOCK) {
        for (int i = 0; i < sizeof(items); i++) {
            if (items[i].id == id) {
                int j = -1;
                while (j < sizeof(items) && items[j].id!= id) {
                    j++;
                }
                if (j < sizeof(items)) {
                    items[j] = items[i];
                    break;
                }
                break;
            }
        }
    }
}

// 将物品信息存储到区块链中
void store_item(block_type type, item_struct item) {
    if (current_block_type == INITIAL_BLOCK || current_block_type == ADD_BLOCK) {
        item_struct new_item = item;
        switch (current_block_type) {
            case INITIAL_BLOCK:
                break;
            case ADD_BLOCK:
                add_item(type, new_item);
                break;
            case UPDATE_BLOCK:
                update_item(type, new_item);
                break;
            case DELETE_BLOCK:
                delete_item(type, new_item.id);
                break;
        }
    }
}
