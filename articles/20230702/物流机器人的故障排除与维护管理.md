```python
# 导入所需库
import numpy as np
import cv2

# 定义物体定位算法
def detect_position(image, objects):
    # 遍历检测到的物体
    for obj in objects:
        # 计算物体与图像的坐标差
        x, y, w, h = obj
        # 绘制矩形框
        cv2.rectangle(image, (x, y), (x + w, y + h), (0, 255, 0), 2)
    
    # 返回定位结果
    return (x + 1, y + 1)
```
