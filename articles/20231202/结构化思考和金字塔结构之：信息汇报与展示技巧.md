import networkx as nx
import matplotlib.pyplot as plt

def draw_hierarchy(hierarchy):
    G = nx.DiGraph()
    for i in range(len(hierarchy)):
        G.add_node(hierarchy[i][0], label=hierarchy[i][0])
        for j in range(i+1, len(hierarchy)):
            G.add_edge(hierarchy[i][0], hierarchy[j][0], label=hierarchy[j][1])

    pos = nx.spring_layout(G)
    nx.draw(G, pos, with_labels=True, node_size=3000, node_color='skyblue', font_size=10)
    plt.show()

hierarchy = [('Main Topic', 'Subtopic 1'), ('Subtopic 1', 'Subtopic 1.1'), ('Subtopic 1', 'Subtopic 1.2'), ('Subtopic 1', 'Subtopic 1.3'), ('Main Topic', 'Subtopic 2'), ('Subtopic 2', 'Subtopic 2.1'), ('Subtopic 2', 'Subtopic 2.2'), ('Subtopic 2', 'Subtopic 2.3')]
