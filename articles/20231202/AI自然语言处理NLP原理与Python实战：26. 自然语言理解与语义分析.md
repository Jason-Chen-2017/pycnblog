# 4.具体代码实例和详细解释说明

在本节中，我们将通过具体的Python代码实例来说明自然语言理解与语义分析的实现细节。

```python
import nltk
from nltk.tokenize import word_tokenize
from nltk.tag import pos_tag
from nltk.chunk import ne_chunk
from nltk.parse import dependency_graph

# 文本预处理
text = "昨天，我去了北京。"
tokens = word_tokenize(text)
tagged = pos_tag(tokens)
named_entities = ne_chunk(tagged)

# 依存关系解析
dependency_graph = dependency_graph(tagged)

# 语义分析
semantic_roles = []
for node in dependency_graph.nodes():
    if node.label() == 'ROOT':
        semantic_roles.append(node.children())

# 语义网络构建
semantic_network = {}
for role in semantic_roles:
    for entity in role:
        if entity not in semantic_network:
            semantic_network[entity] = []
        semantic_network[entity].append(role)

# 语义网络分析
for entity in semantic_network:
    print(entity, semantic_network[entity])
```
