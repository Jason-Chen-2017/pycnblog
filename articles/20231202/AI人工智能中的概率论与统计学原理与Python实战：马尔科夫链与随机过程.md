```python
import numpy as np

# 定义马尔科夫链的状态和状态转移概率
states = ['A', 'B', 'C']
transition_probabilities = np.array([
    [0.5, 0.3, 0.2],
    [0.4, 0.5, 0.1],
    [0.3, 0.4, 0.3]
])

# 定义初始状态
initial_state = 'A'

# 定义迭代次数
iterations = 1000

# 定义状态转移矩阵
state_transition_matrix = np.eye(len(states))

# 计算状态转移矩阵的幂
state_transition_matrix_power = np.linalg.matrix_power(state_transition_matrix, iterations)

# 计算平稳分布
steady_state_distribution = np.linalg.solve(transition_probabilities, np.ones(len(states)))

# 计算期望值
expectation = np.dot(steady_state_distribution, transition_probabilities)

# 计算方差
variance = np.dot(steady_state_distribution, np.dot(transition_probabilities, np.dot(steady_state_distribution.T, transition_probabilities.T))) - np.dot(expectation, np.dot(expectation, transition_probabilities))

# 输出结果
print("平稳分布:", steady_state_distribution)
print("期望值:", expectation)
print("方差:", variance)
```
