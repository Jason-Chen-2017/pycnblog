import matplotlib.pyplot as plt

import time

start_time = time.time()
for i in range(1000000):
    x = np.random.randn(1000000)
    y = np.exp(-x**2)
end_time = time.time()
print('放下思维的时间：', end_time - start_time)

# 心灵的和谐
import threading

def heart_harmony():
    x = np.random.randn(1000000)
    y = np.exp(-x**2)

start_time = time.time()
threading.Thread(target=heart_harmony).start()
end_time = time.time()
print('心灵的和谐的时间：', end_time - start_time)
