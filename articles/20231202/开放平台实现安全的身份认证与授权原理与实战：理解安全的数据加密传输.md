def encrypt(data, key):
    cipher = AES.new(key, AES.MODE_EAX)
    ciphertext, tag = cipher.encrypt_and_digest(data)
    return cipher.nonce, ciphertext, tag

# 解密数据
def decrypt(nonce, ciphertext, tag, key):
    cipher = AES.new(key, AES.MODE_EAX, nonce=nonce)
    return cipher.decrypt_and_verify(ciphertext, tag)

def generate_rsa_key():
    key = RSA.generate(2048)
    public_key = key.publickey()
    private_key = key
    return public_key, private_key

    return ciphertext

    return data

    digest = SHA256.new(data)
    signature = pkcs1_15.new(private_key).sign(digest)
    return signature

    return pkcs1_15.new(public_key).verify(digest, signature)

    # 加密数据
    encrypted_data = encrypt_rsa(data, public_key)

    # 解密数据
    decrypted_data = decrypt_rsa(encrypted_data, private_key)

    # 签名

    # 验签
    is_valid = verify(data, signature, public_key)

    # 打印结果

    print('Is valid signature:', is_valid)
