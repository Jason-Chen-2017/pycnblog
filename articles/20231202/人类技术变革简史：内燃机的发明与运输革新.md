class InternalCombustionEngine:
    def __init__(self, fuel, air, combustion, power):
        self.fuel = fuel
        self.air = air
        self.combustion = combustion
        self.power = power

    def run(self):
        # 燃料与空气的混合
        mixed = self.fuel + self.air

        # 燃烧反应
        reaction = self.combustion(mixed)

        # 能量输出
        output = self.power(reaction)

    def stop(self):
        # 停止燃料的流动
        self.fuel = 0

        # 停止空气的进入
        self.air = 0

        # 停止能量的输出
        self.power = 0
