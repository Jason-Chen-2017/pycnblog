## 2.核心概念与联系

结构化思考是一种将复杂问题分解为简单问题的思考方法。它的核心思想是将问题分解为一系列相互关联的子问题，然后逐步解决这些子问题，最后将解决的子问题组合成原问题的解决方案。结构化思考可以帮助我们更好地理解问题，提高问题解决的效率和质量。

金字塔结构是一种将信息或知识分层组织的框架。它的核心思想是将信息或知识分为多个层次，每个层次包含一定范围的信息或知识，并且每个层次与其他层次之间存在一定的联系。金字塔结构可以帮助我们更好地组织信息或知识，提高信息或知识的管理和应用效率。

结构化思考和金字塔结构之间的联系是，结构化思考可以帮助我们更好地组织信息或知识，而金字塔结构则是一种将结构化思考应用于信息或知识管理的具体实现。结构化思考和金字塔结构相互补充，可以帮助我们更好地解决信息共享和知识管理的问题。

```python
# 创建用户表
class User(db.Model):
    __tablename__ = 'users'
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(64), unique=True, index=True)
    password_hash = db.Column(db.String(128))
    email = db.Column(db.String(128), unique=True, index=True)

# 创建信息表
class Information(db.Model):
    __tablename__ = 'information'
    id = db.Column(db.Integer, primary_key=True)
    category = db.Column(db.String(64))
    title = db.Column(db.String(128))
    content = db.Column(db.Text)

# 创建评论表
class Comment(db.Model):
    __tablename__ = 'comment'
    id = db.Column(db.Integer, primary_key=True)
    content = db.Column(db.Text)
    user_name = db.Column(db.String(64))
    information_title = db.Column(db.String(128))

# 实现用户注册功能
@app.route('/register', methods=['GET', 'POST'])
def register():
    if request.method == 'POST':
        username = request.form['username']
        password = request.form['password']
        email = request.form['email']
        user = User(username=username, password_hash=generate_password_hash(password), email=email)
        db.session.add(user)
        db.session.commit()
        flash('注册成功')
        return redirect(url_for('login'))
    return render_template('register.html')

# 实现信息发布功能
@app.route('/publish', methods=['GET', 'POST'])
def publish():
    if request.method == 'POST':
        category = request.form['category']
        title = request.form['title']
        content = request.form['content']
        information = Information(category=category, title=title, content=content)
        db.session.add(information)
        db.session.commit()
        flash('发布成功')
        return redirect(url_for('index'))
    return render_template('publish.html')

# 实现信息查询功能
@app.route('/query', methods=['GET', 'POST'])
def query():
    if request.method == 'POST':
        category = request.form['category']
        keyword = request.form['keyword']
        information = Information.query.filter_by(category=category).filter(information.title.like('%' + keyword + '%')).all()
        return render_template('query.html', information=information)
    return render_template('query.html')

# 实现信息评论功能
@app.route('/comment', methods=['GET', 'POST'])
def comment():
    if request.method == 'POST':
        content = request.form['content']
        user_name = request.form['user_name']
        information_title = request.form['information_title']
        comment = Comment(content=content, user_name=user_name, information_title=information_title)
        db.session.add(comment)
        db.session.commit()
        flash('评论成功')
        return redirect(url_for('index'))
    return render_template('comment.html')
```
