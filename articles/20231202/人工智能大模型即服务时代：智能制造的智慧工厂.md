def objective_function(x):
    return x[0]**2 + x[1]**2

# 定义约束条件
def constraint_function(x):
    return x[0] + x[1] - 1

# 训练优化模型
result = minimize(objective_function, x0=[0.5, 0.5], constraints=[{'type': 'eq', 'fun': constraint_function}])
