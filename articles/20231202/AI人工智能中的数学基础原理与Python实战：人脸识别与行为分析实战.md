```python
import numpy as np

# 加载图像

# 将图像转换为灰度图像

faces = face_cascade.detectMultiScale(gray, scaleFactor=1.1, minNeighbors=5, minSize=(30, 30), flags=cv2.CASCADE_SCALE_IMAGE)

# 绘制人脸框
for (x, y, w, h) in faces:
    cv2.rectangle(img, (x, y), (x+w, y+h), (255, 0, 0), 2)

# 显示图像

cv2.waitKey(0)
cv2.destroyAllWindows()
```
