```go
    data := []byte("Hello, World!")

    // 将数据转换为二进制
    binaryData := make([]byte, len(data))
    for i, v := range data {
        binaryData[i] = byte(v >> 8)
        binaryData[i+1] = byte(v & 0xFF)
    }

    // 对数据进行位运算
    for i := 0; i < len(binaryData); i++ {
        binaryData[i] ^= key[i % len(key)]
    }

    // 生成加密后的数据
    encryptedData := make([]byte, len(binaryData))
    for i := 0; i < len(binaryData); i++ {
        encryptedData[i] = binaryData[i]
    }

    // 打印加密后的数据
    fmt.Println("加密后的数据:", string(encryptedData))

    // 解密数据
    decryptedData := make([]byte, len(encryptedData))
    for i := 0; i < len(encryptedData); i++ {
        decryptedData[i] = encryptedData[i] ^ key[i % len(key)]
    }

    // 将解密后的数据转换回字符串形式
    fmt.Println("解密后的数据:", string(decryptedData))
}
```

在这个代码实例中，我们首先初始化了密钥（`key`）和要加密的数据（`data`）。然后，我们将数据转换为二进制形式，并对其进行位运算。最后，我们生成加密后的数据（`encryptedData`）并将其打印出来。同时，我们还实现了数据的解密操作，将解密后的数据（`decryptedData`）转换回字符串形式并打印出来。
