```python
class Calculator:
    def __init__(self):
        self.num1 = 0
        self.num2 = 0

    def set_num1(self, num1):
        self.num1 = num1

    def set_num2(self, num2):
        self.num2 = num2

    def add(self):
        return self.num1 + self.num2

    def sub(self):
        return self.num1 - self.num2

    def mul(self):
        return self.num1 * self.num2

    def div(self):
        return self.num1 / self.num2

calc = Calculator()
calc.set_num1(5)
calc.set_num2(3)
print(calc.add())  # 输出: 8
print(calc.sub())  # 输出: 2
print(calc.mul())  # 输出: 15
print(calc.div())  # 输出: 1.6666666666666667
```
