```python
import threading
import time

class DataSync:
    def __init__(self):
        self.data = {}
        self.lock = threading.Lock()

    def update(self, key, value):
        with self.lock:
            self.data[key] = value

    def get(self, key):
        with self.lock:
            return self.data.get(key)

if __name__ == '__main__':
    sync = DataSync()
    threading.Thread(target=sync.update, args=('key1', 'value1')).start()
    threading.Thread(target=sync.update, args=('key2', 'value2')).start()
    time.sleep(1)
    print(sync.get('key1'))
    print(sync.get('key2'))
```
