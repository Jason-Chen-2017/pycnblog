   ```python
   import numpy as np

   def fill_missing_values(x, mode='mean'):
       if mode == 'mean':
           x_fill = np.mean(x)
       elif mode == 'median':
           x_fill = np.median(x)
       elif mode == 'min':
           x_fill = np.min(x)
       elif mode == 'max':
           x_fill = np.max(x)
       return x_fill

   def typecast(x, type):
       return x.astype(type)

   def format(x, format):
       return x.astype(format)
   ```
