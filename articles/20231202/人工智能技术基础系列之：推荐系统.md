```python
import numpy as np
from scipy.sparse import csr_matrix
from scipy.sparse.linalg import svds

# 用户-目标评分矩阵
R = csr_matrix([
    [5, 0, 0, 0, 0],
    [0, 4, 0, 0, 0],
    [0, 0, 3, 0, 0],
    [0, 0, 0, 2, 0],
    [0, 0, 0, 0, 1
])

# 用户-用户相似度矩阵
Sim = np.array([
    [0, 0.5, 0.6, 0.7, 0.8],
    [0.5, 0, 0.6, 0.7, 0.8],
    [0.6, 0.6, 0, 0.5, 0.4],
    [0.7, 0.7, 0.5, 0, 0.4],
    [0.8, 0.8, 0.4, 0.4, 0
])

# 矩阵分解
U, sigma, Vt = svds(R.T.dot(R), k=5)

# 计算预测评分
pred = np.dot(U, np.dot(sigma, Vt.T))

# 计算评估指标
mse = np.mean((R - pred) ** 2)
```
