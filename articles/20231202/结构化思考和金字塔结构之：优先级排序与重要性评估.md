# 4.具体代码实例和详细解释说明

在本节中，我们将通过具体的代码实例来解释优先级排序和重要性评估的概念和算法。

```python
# 定义问题和部分
problem = {
    "name": "问题名称",
    "parts": [
        {"name": "部分名称1", "importance": 0.5},
        {"name": "部分名称2", "importance": 0.3},
        {"name": "部分名称3", "importance": 0.2}
    ]
}

# 计算问题的优先级
def calculate_priority(problem):
    priority = 0
    for part in problem["parts"]:
        priority += part["importance"]
    return priority

# 计算问题的重要性
def calculate_importance(problem):
    importance = 0
    for part in problem["parts"]:
        importance += part["importance"]
    return importance

# 优先级排序
def sort_by_priority(problems):
    return sorted(problems, key=lambda x: x["priority"], reverse=True)

# 重要性评估
def evaluate_importance(problems):
    return sorted(problems, key=lambda x: x["importance"], reverse=True)

# 测试代码
problems = [problem]
sorted_by_priority = sort_by_priority(problems)
sorted_by_importance = evaluate_importance(problems)

print("按优先级排序的问题：", sorted_by_priority)
print("按重要性评估的问题：", sorted_by_importance)
```
