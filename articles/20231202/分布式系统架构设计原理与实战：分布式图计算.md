def connected_components(graph):
    components = []
    visited = set()

    def dfs(node):
        if node in visited:
            return
        visited.add(node)
        for neighbor in graph[node].neighbors:
            dfs(neighbor)
        components.append(visited)
        visited.clear()

    for node in graph.nodes:
        dfs(node)
    return components
