from github import GHRepo
import os

token = 'your_github_token'

def extract_files(repo, path):
    try:
        gh_repository = GHRepo(repo)
        files = gh_repository.get_files(repo.body, path)
        
        output_folder = os.path.join(os.getcwd(), path)
        os.mkdir(output_folder)
        
        for content in files.contents:
            print(content.path)
            content.download_to(output_folder)
    except GHError as err:
        print(err)

extract_files(path, 'some_dir/some_file.txt')
