        block = {
            'index': len(self.chain) + 1,
            'timestamp': time.time(),
            'proof': proof,
            'previous_hash': previous_hash
        }
        self.chain.append(block)
        return block

            if hash_operation[:4] == '0000':
                check_proof = True
            else:
                new_proof += 1

### 1.4.2 加密货币代码实例

```python
import hashlib
import hmac
import os
import time

class CryptoCurrency:
    def __init__(self, public_key, private_key):
        self.public_key = public_key
        self.private_key = private_key

    def sign(self, message):
        signature = hmac.new(self.private_key, message.encode(), digestmod=hashlib.sha256).digest()
        return signature

    def verify(self, message, signature):
        return hmac.compare_digest(signature, hmac.new(self.public_key, message.encode(), digestmod=hashlib.sha256).digest())

    def generate_address(self):
        address = os.urandom(20).hex()
        return address

    def send_transaction(self, recipient_address, amount):
        transaction = {
            'from_address': self.generate_address(),
            'to_address': recipient_address,
            'amount': amount
        }
        signature = self.sign(json.dumps(transaction))
        return transaction, signature
```
