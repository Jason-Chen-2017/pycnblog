import React, { useState, useEffect } from 'react';
import axios from 'axios';

const TranslationService = {
  async translate(text, targetLanguage) {
    const apiKey = 'YOUR_GOOGLE_CLOUD_TRANSLATION_API_KEY';
    const url = `https://translation.googleapis.com/language/translate/v2?key=${apiKey}`;

    const response = await axios.post(url, {
      q: text,
      target: targetLanguage,
    });

    return response.data.data.translations[0].translatedText;
  },
};

const LanguageSelector = () => {
  const [text, setText] = useState('');
  const [translatedText, setTranslatedText] = useState('');

  const languages = ['en', 'zh', 'fr', 'es', 'ja'];

  const handleLanguageChange = (event) => {
    TranslationService.translate(text, event.target.value).then((result) => {
      setTranslatedText(result);
    });
  };

  return (
    <div>
      <textarea
        value={text}
        onChange={(event) => setText(event.target.value)}
        placeholder="Enter text to translate..."
      />
      <select value={translatedText.language} onChange={handleLanguageChange}>
        {languages.map((lang) => (
          <option key={lang} value={lang}>
            {lang}
          </option>
        ))}
      </select>
      <p>{translatedText.text}</p>
    </div>
  );
};

export default LanguageSelector;
