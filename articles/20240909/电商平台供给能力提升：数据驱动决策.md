        self.cart = {}

    def add_item(self, user_id, item_id):
        if user_id in self.cart:
            self.cart[user_id].append(item_id)

            self.cart[user_id] = [item_id]

### 2. 如何实现一个商品搜索功能？

**答案：** 可以使用倒排索引实现商品搜索功能。

```python
class SearchEngine:
    def __init__(self):
        self.index = {}

    def add_product(self, product_id, tags):
        for tag in tags:
            if tag not in self.index:
                self.index[tag] = []
            self.index[tag].append(product_id)

    def search_products(self, query):
        query_tags = query.split()
        result = set()
        for tag in query_tags:
            if tag in self.index:
                result.update(self.index[tag])
        return list(result)
```

import pika

class OrderProcessor:

    def process_order(self, order_id):
        self.channel.basic_publish(exchange='', routing_key='orders', body=str(order_id))
        self.connection.close()

    def consume_orders(self):
        def callback(ch, method, properties, body):
            order_id = int(body)
            print(f"Processing order {order_id}")
            # 处理订单逻辑
            ch.basic_ack(delivery_tag=method.delivery_tag)

        self.channel.basic_consume(queue='orders', on_message_callback=callback, auto_ack=True)
        self.channel.start_consuming()
