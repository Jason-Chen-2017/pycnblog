b1 = np.random.rand(1, 1)

# 前向传播

a1 = np.tanh(z1)

# 计算损失
loss = np.mean((a1 - y)**2)

# 反向传播
dz1 = 2 * (a1 - y)
da1 = dz1 * (1 - np.power(a1, 2))

# 更新权重和偏置
dw1 = np.dot(X.T, da1)
db1 = np.sum(da1, axis=0, keepdims=True)

# 学习率
alpha = 0.1

# 更新参数
w1 -= alpha * dw1
b1 -= alpha * db1
