def generate_logic_expression(doc):
    logic_expression = []
    for token in doc:
        if token.pos_ == 'NOUN':
            logic_expression.append(f"{token.text} = '{token.lemma_}'")
        elif token.pos_ == 'VERB':
            logic_expression.append(f"{token.text}()")
    return ' '.join(logic_expression)

logic_expression = generate_logic_expression(doc)
