### 1. 请编写一个程序，实现一个简单的任务管理工具，支持以下功能：
- 添加任务
- 删除任务
- 完成任务
- 查看任务列表

**答案：** 

```python
# 任务管理工具

class TaskManager:
    def __init__(self):
        self.tasks = []

    def add_task(self, task):
        self.tasks.append(task)
        print(f"任务'{task}'已添加。")

    def delete_task(self, task):
        if task in self.tasks:
            self.tasks.remove(task)
            print(f"任务'{task}'已删除。")
        else:
            print(f"任务'{task}'不存在。")

    def complete_task(self, task):
        if task in self.tasks:
            self.tasks.remove(task)
            print(f"任务'{task}'已完成。")
        else:
            print(f"任务'{task}'不存在。")

    def show_tasks(self):
        print("任务列表：")
        for task in self.tasks:
            print(task)

# 测试
manager = TaskManager()
manager.add_task("购买水果")
manager.add_task("看电影")
manager.show_tasks()
manager.delete_task("看电影")
manager.complete_task("购买水果")
manager.show_tasks()
```

### 2. 请编写一个程序，实现一个简单的习惯跟踪工具，支持以下功能：
- 添加习惯
- 删除习惯
- 完成习惯
- 查看习惯列表
- 统计完成次数

**答案：**

```python
# 习惯跟踪工具

class HabitTracker:
    def __init__(self):
        self.habits = {}

    def add_habit(self, habit):
        self.habits[habit] = 0
        print(f"习惯'{habit}'已添加。")

    def delete_habit(self, habit):
        if habit in self.habits:
            del self.habits[habit]
            print(f"习惯'{habit}'已删除。")
        else:
            print(f"习惯'{habit}'不存在。")

    def complete_habit(self, habit):
        if habit in self.habits:
            self.habits[habit] += 1
            print(f"习惯'{habit}'已完成。")
        else:
            print(f"习惯'{habit}'不存在。")

    def show_habits(self):
        print("习惯列表：")
        for habit, count in self.habits.items():
            print(f"{habit}：{count}次")

    def habit_statistics(self, habit):
        if habit in self.habits:
            print(f"{habit}的完成次数：{self.habits[habit]}次")
        else:
            print(f"习惯'{habit}'不存在。")

# 测试
tracker = HabitTracker()
tracker.add_habit("早起")
tracker.add_habit("锻炼")
tracker.show_habits()
tracker.complete_habit("早起")
tracker.habit_statistics("早起")
tracker.show_habits()
tracker.delete_habit("锻炼")
tracker.show_habits()
```
