**1. 编写一个函数，实现两个有序链表的合并。**

**答案：**

```python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def merge_sorted_list(l1, l2):
    dummy = ListNode()
    current = dummy
    while l1 and l2:
        if l1.val < l2.val:
            current.next = l1
            l1 = l1.next
        else:
            current.next = l2
            l2 = l2.next
        current = current.next
    current.next = l1 or l2
    return dummy.next
```

```python
def str_match(s, p):
    def KMP_next(p):
        next = [0] * len(p)
        j = 0
        for i in range(1, len(p)):
            while j > 0 and p[j] != p[i]:
                j = next[j - 1]
            if p[j] == p[i]:
                j += 1
            next[i] = j
        return next

    next = KMP_next(p)
    j = 0
    for i in range(len(s)):
        while j > 0 and s[i] != p[j]:
            j = next[j - 1]
        if s[i] == p[j]:
            j += 1
        if j == len(p):
            return i - j + 1
    return -1
```

**3. 实现一个优先队列，支持插入、删除和获取最大值操作。**

```python
import heapq

class PriorityQueue:
    def __init__(self):
        self.heap = []

    def push(self, item, priority):
        heapq.heappush(self.heap, (-priority, item))

    def pop(self):
        return heapq.heappop(self.heap)[1]

    def get_max(self):
        return self.heap[0][1]
```
