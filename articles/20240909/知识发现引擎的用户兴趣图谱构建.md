### 1. 预处理数据

**题目：** 给定一个文本文件，编写一个程序，将其中的文本数据进行预处理，包括分词、去除停用词、词干提取等。

**答案：**

（以下为 Python 代码示例）

```python
import nltk
from nltk.corpus import stopwords
from nltk.stem import PorterStemmer

# 下载停用词库
nltk.download('stopwords')

def preprocess_text(text):
    # 分词
    tokens = nltk.word_tokenize(text)
    
    # 去除停用词
    stop_words = set(stopwords.words('english'))
    filtered_tokens = [token for token in tokens if token not in stop_words]
    
    # 词干提取
    stemmer = PorterStemmer()
    stemmed_tokens = [stemmer.stem(token) for token in filtered_tokens]
    
    return stemmed_tokens

text = "This is a sample text for preprocessing."
preprocessed_text = preprocess_text(text)
print(preprocessed_text)
```

### 2. 基于协同过滤的推荐算法

**题目：** 实现一个基于用户的协同过滤推荐算法，根据用户的历史行为数据，为用户推荐相似用户喜欢的商品。

**答案：**

（以下为 Python 代码示例）

```python
import numpy as np
from collections import defaultdict

# 假设用户行为数据为用户-物品评分矩阵
user_item_matrix = np.array([[1, 0, 1, 1],
                              [0, 1, 1, 0],
                              [1, 1, 0, 1],
                              [1, 0, 1, 0]])

# 计算用户之间的相似度矩阵
def compute_similarity_matrix(user_item_matrix):
    num_users, num_items = user_item_matrix.shape
    similarity_matrix = np.zeros((num_users, num_users))
    
    for i in range(num_users):
        for j in range(num_users):
            if i != j:
                common_items = np.where(np.logical_and(user_item_matrix[i, :] == 1, user_item_matrix[j, :] == 1))[0]
                if len(common_items) > 0:
                    similarity_matrix[i][j] = np.dot(user_item_matrix[i, common_items], user_item_matrix[j, common_items])
                    
    return similarity_matrix

similarity_matrix = compute_similarity_matrix(user_item_matrix)

# 根据相似度矩阵为用户推荐物品
def recommend_items(user_id, similarity_matrix, user_item_matrix, k=3):
    similar_users = np.argsort(similarity_matrix[user_id])[::-1][:k]
    recommended_items = set()
    
    for user in similar_users:
        items = np.where(user_item_matrix[user, :] == 1)[0]
        recommended_items.update(items)
    
    return list(recommended_items)

user_id = 0
recommended_items = recommend_items(user_id, similarity_matrix, user_item_matrix)
print("Recommended items:", recommended_items)
```

### 3. 基于主题模型的文本分类

**题目：** 使用 LDA（Latent Dirichlet Allocation）主题模型对一组文本进行分类，并输出每个文本的主题分布。

**答案：**

（以下为 Python 代码示例）

```python
import gensim
from gensim.models import LdaModel

# 假设文本数据为列表
texts = [['apple', 'banana', 'orange'],
         ['apple', 'orange', 'mango'],
         ['banana', 'mango', 'apple'],
         ['orange', 'mango', 'banana']]

# 分词并转换为词袋模型
dictionary = gensim.corpora.Dictionary(texts)
corpus = [dictionary.doc2bow(text) for text in texts]

# 使用 LDA 模型进行主题建模
lda_model = LdaModel(corpus, num_topics=3, id2word=dictionary, passes=10)

# 输出每个文本的主题分布
for i, text in enumerate(texts):
    print(f"Text {i+1} topic distribution:")
    print(lda_model.get_document_topics(corpus[i]))
    print()
```
