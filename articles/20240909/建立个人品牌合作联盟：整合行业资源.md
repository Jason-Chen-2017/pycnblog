### 6. 求最大公约数和最小公倍数

**题目：** 编写一个函数，计算两个数的最大公约数和最小公倍数。

```go
func gcd(a, b int) int {
    for b != 0 {
        t := b
        b = a % b
        a = t
    }
    return a
}

func lcm(a, b int) int {
    return a / gcd(a, b) * b
}
```

### 7. 字符串压缩和解压

**题目：** 编写一个函数，实现字符串的压缩和解压功能。

```go
func compress(s string) string {
    var b strings.Builder
    prev := s[0]
    count := 1
    for _, c := range s[1:] {
        if c == prev {
            count++
        } else {
            b.WriteByte(prev)
            if count > 1 {
                b.WriteString(strconv.Itoa(count))
            }
            prev = c
            count = 1
        }
    }
    b.WriteByte(prev)
    if count > 1 {
        b.WriteString(strconv.Itoa(count))
    }
    return b.String()
}

func decompress(s string) string {
    var result strings.Builder
    for i := 0; i < len(s); i++ {
        if i > 0 && s[i] >= '0' && s[i] <= '9' {
            j := i
            for j < len(s) && s[j] >= '0' && s[j] <= '9' {
                j++
            }
            repeat := j - i - 1
            result.WriteString(strings.Repeat(string(s[i-1]), repeat))
            i = j
        } else {
            result.WriteByte(s[i])
        }
    }
    return result.String()
}
```

### 8. 快排算法实现

**题目：** 编写一个快速排序算法。

```go
func quickSort(arr []int) {
    if len(arr) <= 1 {
        return
    }
    pivot := arr[len(arr)/2]
    left, right := 0, len(arr)-1
    for i := 0; i <= right; i++ {
        if arr[i] < pivot {
            arr[left], arr[i] = arr[i], arr[left]
            left++
        } else if arr[i] > pivot {
            arr[right], arr[i] = arr[i], arr[right]
            right--
            i--
        }
    }
    quickSort(arr[:left])
    quickSort(arr[left:])
}
```
