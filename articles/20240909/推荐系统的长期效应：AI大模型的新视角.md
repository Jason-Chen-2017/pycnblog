### 4.1 算法编程题1：实现一个简单的协同过滤推荐算法

**题目描述：**
实现一个基于用户协同过滤的推荐算法，给定一个用户-物品评分矩阵，预测用户未评分的物品评分。

**答案解析：**
协同过滤推荐算法可以分为基于用户和基于物品的协同过滤。以下是一个基于用户协同过滤的简单实现：

```python
import numpy as np

def similarity_matrix(rating_matrix):
    """ 计算用户间的相似度矩阵 """
    # 使用余弦相似度计算相似度
    similarity = np.dot(rating_matrix.T, rating_matrix) / (
        np.linalg.norm(rating_matrix, axis=1) * np.linalg.norm(rating_matrix, axis=0)
    )
    return similarity

def collaborative_filtering(rating_matrix, user_index, k=5):
    """ 基于用户协同过滤推荐算法 """
    # 计算用户相似度矩阵
    similarity = similarity_matrix(rating_matrix)
    
    # 计算用户间的相似度得分
    user_similarity = similarity[user_index]
    scores = np.dot(user_similarity, rating_matrix) - user_similarity[user_index]
    
    # 选择最相似的 k 个用户，并计算预测评分
    top_k_indices = np.argsort(scores)[-k:]
    predicted_rating = np.mean(rating_matrix[top_k_indices] / np.linalg.norm(rating_matrix[top_k_indices]))
    
    return predicted_rating

# 测试数据
rating_matrix = np.array([[5, 3, 0, 1], [4, 0, 0, 1], [1, 1, 0, 5], [1, 0, 0, 4], [0, 1, 5, 4]])

# 预测未评分的物品评分
predicted_rating = collaborative_filtering(rating_matrix, 0)
print("Predicted rating:", predicted_rating)
```

def content_similarity(user_features, item_features, similarity_threshold=0.5):
    """ 计算用户和物品的相似度 """
    similarity = np.dot(user_features, item_features)
    return similarity >= similarity_threshold

def content_based_recommending(user_features_matrix, item_features_matrix, user_index, k=5):
    """ 基于内容推荐算法 """
    # 计算用户和物品的相似度
    similarities = np.dot(user_features_matrix[user_index], item_features_matrix.T)
    
    # 选择最相似的 k 个物品
    top_k_indices = np.argsort(similarities)[-k:]
    top_k_items = item_features_matrix[top_k_indices]
    
    # 计算预测评分
    predicted_ratings = np.dot(user_features_matrix[user_index], top_k_items) / np.linalg.norm(top_k_items)
    
    return predicted_ratings
