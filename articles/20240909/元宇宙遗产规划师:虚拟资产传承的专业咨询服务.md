### 1. 虚拟资产评估系统

```go
package main

import (
    "encoding/json"
    "fmt"
)

type Asset struct {
    Name   string `json:"name"`
    Value  int    `json:"value"`
}

type User struct {
    Username string     `json:"username"`
    Assets   []Asset    `json:"assets"`
}

func EvaluateAssets(user User) (User, int) {
    totalValue := 0
    for _, asset := range user.Assets {
        totalValue += asset.Value
    }
    return user, totalValue
}

func main() {
    user := User{
        Username: "Alice",
        Assets: []Asset{
            {"Virtual Land", 5000},
            {"Virtual Item", 200},
            {"Virtual Coin", 100},
        },
    }

    evaluatedUser, totalValue := EvaluateAssets(user)
    fmt.Printf("User: %v\n", evaluatedUser)
    fmt.Printf("Total Value: %d\n", totalValue)
}
```

### 2. 虚拟资产继承权分配算法

```go
package main

import (
    "encoding/json"
    "fmt"
)

type Asset struct {
    Name     string `json:"name"`
    Value    int    `json:"value"`
    Allocation []分配 `json:"分配"`
}

type 分配 struct {
    Name     string  `json:"name"`
    Percentage float64 `json:"percentage"`
}

type User struct {
    Username string      `json:"username"`
    Assets   []Asset     `json:"assets"`
    Heirs    []Heir      `json:"heirs"`
}

type Heir struct {
    Name       string  `json:"name"`
    Percentage float64 `json:"percentage"`
}

func AllocateAssets(user User) User {
    for _, asset := range user.Assets {
        totalPercentage := 0.0
        for _, heir := range user.Heirs {
            totalPercentage += heir.Percentage
        }

        for _, heir := range user.Heirs {
            allocation := (asset.Value * heir.Percentage) / totalPercentage
            asset.Allocation = append(asset.Allocation, 分配{heir.Name, allocation})
        }
    }
    return user
}

func main() {
    user := User{
        Username: "Alice",
        Assets: []Asset{
            {"Virtual Land", 5000},
            {"Virtual Item", 200},
            {"Virtual Coin", 100},
        },
        Heirs: []Heir{
            {"Bob", 0.5},
            {"Charlie", 0.3},
            {"Dave", 0.2},
        },
    }

    allocatedUser := AllocateAssets(user)
    fmt.Printf("User: %v\n", allocatedUser)
}
```
