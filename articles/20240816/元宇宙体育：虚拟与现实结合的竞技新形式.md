```python
from OculusPythonSDK import OculusPythonSDK
from OculusPythonSDK.sensors import Sensor
import time

# 初始化Oculus Rift
sdk = OculusPythonSDK()
headset = sdk.get_headset()
headset.connect()

# 实时采集头显位置和姿态
while True:
    data = headset.get_data()
    x, y, z = data['x'], data['y'], data['z']
    print(f"Headset position: ({x}, {y}, {z})")
    time.sleep(0.1)

# 断开连接
headset.disconnect()
```
