```python
import numpy as np
import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

# 数据收集与预处理
data = pd.read_csv('user_behavior_data.csv')
data['user_id'] = data['user_id'].astype('int')
data['item_id'] = data['item_id'].astype('int')
data['behavior'] = data['behavior'].astype('int')

# 算法模型选择与训练
user_item_matrix = data.pivot_table(index='user_id', columns='item_id', values='behavior', fill_value=0)
user_item_matrix = user_item_matrix.fillna(0)

# 预测与评估
similarity_matrix = cosine_similarity(user_item_matrix)

# 推荐
def recommend(user_id, n_recommend=10):
    user_item_matrix_user = user_item_matrix.loc[user_id].dropna()
    similarity_user_item = similarity_matrix.loc[user_id]
    similarity_user_item = similarity_user_item[similarity_user_item.notnull()]
    similarity_user_item = similarity_user_item.sort_values(ascending=False)
    recommend_items = user_item_matrix.loc[similarity_user_item.index].loc[user_id].dropna().sort_values(ascending=False)[:n_recommend]
    return recommend_items

user_id = 12345
recommend_items = recommend(user_id)
print(recommend_items)
```

在这个代码实例中，我们首先对用户行为数据进行预处理，然后使用协同过滤算法对用户行为数据进行训练。最后，我们根据用户的历史行为，生成个性化的商品推荐列表。
