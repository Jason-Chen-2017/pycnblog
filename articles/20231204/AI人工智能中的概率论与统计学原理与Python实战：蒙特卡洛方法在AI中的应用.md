    y = random.uniform(-1, 1)
    return (x, y)

    x, y = point
    return x**2 + y**2 <= 1

def estimate_pi(n):
    points_in_circle = 0
    for _ in range(n):
        point = generate_random_point()
        if is_point_in_circle(point):
            points_in_circle += 1
    pi = 4 * points_in_circle / n
    return pi

n = 100000
pi = estimate_pi(n)
print("Estimated pi:", pi)
