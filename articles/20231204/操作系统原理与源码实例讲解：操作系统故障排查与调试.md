class Process:
    def __init__(self, pid, arrival_time, burst_time):
        self.pid = pid
        self.arrival_time = arrival_time
        self.burst_time = burst_time

def FCFS_schedule(processes):
    processes.sort(key=lambda x: x.arrival_time)
    current_time = 0
    for process in processes:
        if process.arrival_time >= current_time:
            current_time = process.arrival_time
        process.waiting_time = current_time - process.arrival_time
        current_time += process.burst_time
        process.turnaround_time = current_time

processes = [Process(1, 0, 5), Process(2, 2, 3), Process(3, 4, 8)]
FCFS_schedule(processes)
for process in processes:
    print(process.pid, process.waiting_time, process.turnaround_time)
```

def best_fit(processes, memory):
    for process in processes:
        for free_space in memory:
            if free_space >= process.size:
                process.allocation = free_space
                memory.remove(free_space)
                break

processes = [Process(1, 0, 5), Process(2, 2, 3), Process(3, 4, 8)]
memory = [10, 15, 20, 25, 30]
best_fit(processes, memory)
for process in processes:
    print(process.pid, process.allocation)
```
