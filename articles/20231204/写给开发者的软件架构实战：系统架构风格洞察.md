```python
class Component:
    def __init__(self, name, performance):
        self.name = name
        self.performance = performance

    def organize(self):
        return self.performance

    def combine(self):
        return self.performance

    def interact(self):
        return self.performance

class System:
    def __init__(self, components):
        self.components = components

    def organize(self):
        return sum(component.organize() for component in self.components)

    def combine(self):
        return sum(component.combine() for component in self.components)

    def interact(self):
        return sum(component.interact() for component in self.components)

# 创建组件
component1 = Component("Component1", 10)
component2 = Component("Component2", 20)

# 创建系统
system = System([component1, component2])

# 组件组织
organize_result = system.organize()
print("组件组织的性能指标：", organize_result)

# 组件组合
combine_result = system.combine()
print("组件组合的性能指标：", combine_result)

# 组件交互
interact_result = system.interact()
print("组件交互的性能指标：", interact_result)
```
