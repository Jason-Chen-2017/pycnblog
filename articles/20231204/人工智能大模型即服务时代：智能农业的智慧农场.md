        # 设置监控数据的发送时间间隔

            # 设置监控数据的发送标志
            FLAG = True

            # 设置监控数据的发送循环
            while FLAG:
                # 设置监控数据的发送时间
                START_TIME = time.time()

                # 设置监控数据的发送循环
                while COUNT < SLEEP:
                    # 设置监控数据的发送时间
                    END_TIME = time.time()

                    # 设置监控数据的发送时间间隔
                    TIME_DIFF = END_TIME - START_TIME

                    # 设置监控数据的发送标志
                    if TIME_DIFF >= INTERVAL:
                        # 设置监控数据的发送标志
                        FLAG = False

                        # 设置监控数据的发送循环
                        break

                    # 设置监控数据的发送次数
                    COUNT += 1

                # 设置监控数据的发送循环

    break
