    inconsistencies = []

            inconsistencies.append(i)
    if inconsistencies:
        # 删除不一致信息
        prompt = ' '.join([token for i, token in enumerate(tokens) if i not in inconsistencies])
        # 修改不一致信息
        prompt = ' '.join([token for i, token in enumerate(tokens) if i not in inconsistencies])
        # 添加不一致信息
        prompt = ' '.join([token for i, token in enumerate(tokens) if i not in inconsistencies])

    tagged = nltk.pos_tag(tokens)
    inconsistencies = []
    for i in range(len(tagged)):
        if tagged[i][1] == 'NN' and tagged[i+1][1] == 'NN':
