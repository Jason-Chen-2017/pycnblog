以下是一个简单的设备驱动程序的代码实例：

```c
#include <stdio.h>
#include <stdlib.h>
#include <sys/ioctl.h>
#include <unistd.h>

#define DEVICE_FILE "/dev/mydevice"

int main() {
    int fd;
    unsigned char data;

    // 打开设备文件
    fd = open(DEVICE_FILE, O_RDWR);
    if (fd < 0) {
        perror("open");
        return -1;
    }

    // 读取设备数据
    if (ioctl(fd, IOCTL_READ, &data) < 0) {
        perror("ioctl");
        close(fd);
        return -1;
    }

    // 处理读取到的数据
    printf("read data: %d\n", data);

    // 关闭设备文件
    close(fd);

    return 0;
}
```

在这个代码实例中，我们创建了一个简单的设备驱动程序，它通过设备文件与设备进行数据交互。我们首先打开了设备文件，然后使用ioctl函数读取设备数据。最后，我们处理读取到的数据并关闭设备文件。
