import numpy as np
from scipy.integrate import odeint

# 轨道运动的数学模型
def dydt(state, t, G, r):
    r = np.array(r)
    r_dot = state[:3]
    r_ddot = state[3:]
    r_ddot = -G * r / np.linalg.norm(r) ** 3 * r + r_dot

    return np.concatenate((r_dot, r_ddot))

# 轨道初始条件
G = 6.67430e-11  # 引力常数
r0 = np.array([150000000, 0, 0])  # 轨道半径
v0 = np.array([0, 0, 0])  # 轨道速度

# 轨道运动的解析解
sol = odeint(dydt, v0, [0, 10000], args=(G, r0), t_eval=[0, 10000], mxstep=100000)

# 轨道的稳定性和稳定区域
def stability(r, G):
    r = np.array(r)
    r_norm = np.linalg.norm(r)
    return G * r_norm ** 2 / (4 * np.pi ** 2)

# 轨道的稳定性和稳定区域
r_stable = np.array([150000000, 150000000, 150000000])
G_stable = stability(r_stable, G)
