from sklearn.metrics.pairwise import pairwise_distances
from sklearn.metrics.pairwise import cosine_similarity

# 首先，我们需要计算用户之间的相似度
def calculate_similarity(user_matrix):
    similarity_matrix = pairwise_distances(user_matrix, metric='cosine')
    return similarity_matrix

# 然后，我们可以使用相似度矩阵来推荐内容
def recommend_content(user_matrix, item_matrix, similarity_matrix, user_id):
    # 首先，我们需要计算用户的兴趣爱好
    user_interest = user_matrix[user_id]

    # 然后，我们可以使用相似度矩阵来推荐内容
    recommended_items = cosine_similarity(user_interest.reshape(1, -1), similarity_matrix)

    # 最后，我们可以根据推荐得分来推荐内容
    recommended_items = recommended_items.flatten()
    recommended_items = recommended_items[recommended_items > 0]
    recommended_items = item_matrix[recommended_items]

    return recommended_items
