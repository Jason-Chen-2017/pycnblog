}

# 发送API请求
response = requests.post(api_url, json=package_info)

# 解析API响应
response_data = json.loads(response.text)

# 处理快递物流数据

    print("快递包裹信息：", response_data["data"])
else:
    print("错误信息：", response_data["message"])
