```python
import numpy as np

# 定义有向图
graph = {
    'A': [('A', 'B', 1), ('A', 'C', 1)],
    'B': [('B', 'C', 1)],
    'C': [('C', 'D', 1)],
    'D': []
}

# 计算有向图的行列式
def adjacency_matrix(graph):
    n = len(graph)
    A = np.zeros((n, n))
    for node, edges in graph.items():
        for edge in edges:
            A[node.index(edge[0]), edge[0].index(node)] = edge[2]
    return A

# 计算有向图的秩
def rank(A):
    U, _ = np.linalg.qr(A)
    return np.sum(np.all(U[:, i] != 0, axis=0) for i in range(U.shape[1]))

# 主程序
A = adjacency_matrix(graph)
r = rank(A)
print("行列式:", A)
print("秩:", r)
```

在这个代码实例中，我们首先定义了一个有向图，然后使用 `adjacency_matrix` 函数计算有向图的行列式，最后使用 `rank` 函数计算有向图的秩。
