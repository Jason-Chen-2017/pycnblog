# 进行数据归一化
data['feature1'] = data['feature1'].values / 255.0
data['feature2'] = data['feature2'].values / 10.0
data['feature3'] = data['feature3'].values / 2.0
data['feature4'] = data['feature4'].values / 5.0

# 进行数据标准化

# 进行数据一致性验证
