```python
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score, confusion_matrix, f1_score

# 数据集
train_data = ['example_1', 'example_2', 'example_3']
test_data = ['example_4', 'example_5', 'example_6']

# 数据预处理
X_train = train_data[0]
X_test = test_data[0]
y_train = train_data[1]
y_test = test_data[1]

# 特征提取
X_train_features = ['text', 'image']
X_test_features = ['text', 'image']

# 特征选择
X_train_features = [feature for feature in X_train_features if feature not in ['image']]
X_test_features = [feature for feature in X_test_features if feature not in ['image']]

# 模型训练
clf = RandomForestClassifier(n_estimators=100, random_state=42)
clf.fit(X_train, y_train)

# 模型评估
accuracy = clf.score(X_test, y_test)
conf_matrix = confusion_matrix(y_test, clf.predict(X_test))
f1_score(y_test, clf.predict(X_test), average='weighted')

# 模型部署
X_train_features = ['text', 'image']
clf.predict(X_train_features)
```
