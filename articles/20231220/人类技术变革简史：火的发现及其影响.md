```python
def burn(fuel, oxidant):
    reaction = fuel + oxidant
    heat = reaction * 1000  # 燃烧产生的热量
    light = heat / 100  # 燃烧产生的光
    return heat, light

fuel = "纸"
oxidant = "氧气"

print(f"燃烧{fuel}产生的热量为：{heat}J")
print(f"燃烧{fuel}产生的光为：{light}lumen")
```
