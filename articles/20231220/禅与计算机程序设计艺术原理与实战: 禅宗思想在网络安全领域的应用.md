    K = (g, d, e)
    return K

def encrypt(M, e, p):
    C = M ** e % p
    return C

def decrypt(C, d, p):
    M = C ** d % p
    return M

p = random.getrandbits(2048)
while p % 2 == 0 or p % 3 == 0:
    p = random.getrandbits(2048)
K = generate_keys(p)

C = encrypt(M, K[2], p)
print(f"密钥对：{K}")
