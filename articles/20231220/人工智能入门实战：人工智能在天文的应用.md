import numpy as np

def fill_missing_values(data, method='interpolation'):
    if method == 'interpolation':
        # 插值法填充缺失值
        data = np.interp(np.arange(len(data)), np.arange(len(data))[~np.isnan(data)], data[~np.isnan(data)])
    elif method == 'nearest_neighbors':
        # 最邻近法填充缺失值
        data = np.nan_to_num(data, nan=data[np.argmin(np.abs(data - data[~np.isnan(data)]))])
    return data
```

import cv2

def detect_edges(image):
    # Sobel算子检测边缘
    sobelx = cv2.Sobel(image, cv2.CV_64F, 1, 0, ksize=3)
    sobely = cv2.Sobel(image, cv2.CV_64F, 0, 1, ksize=3)
    gradient_magnitude = np.sqrt(sobelx**2 + sobely**2)
    gradient_direction = np.arctan2(sobely, sobelx)
    return gradient_magnitude, gradient_direction
```
