import numpy as np
import matplotlib.pyplot as plt

# 线性规划示例
def simplex_x(c, A, b):
    n, m = len(c), len(b)
    basis = np.zeros(m)
    free = np.zeros(n-m)
    x = np.zeros(n)
    for i in range(m):
        if A[i, -1] > 0:
            pivot = i
            break
    basis[pivot] = 1
    x[pivot] = (b[pivot] - np.dot(A[pivot, :pivot], basis[:pivot])) / A[pivot, -1]
    for i in range(n-m):
        free[i] = (c[i] - np.dot(c[:pivot], basis[:pivot])) / A[pivot, -1]
    for i in range(m):
        if i != pivot:
            for j in range(n):
                if j != pivot:
                    x[j] -= A[i, j] * x[pivot]
            b[i] -= A[i, -1] * x[pivot]
    return x

# 遗传算法示例
def genetic_algorithm(fitness_func, population_size, generations, mutation_rate, crossover_rate):
    population = np.random.rand(population_size, len(fitness_func))
    for _ in range(generations):
        fitness = fitness_func(population)
        new_population = np.random.rand(population_size, len(fitness_func))
        for i in range(population_size):
            parent1 = population[np.random.choice(population_size, 2, replace=False)]
            parent2 = population[np.random.choice(population_size, 2, replace=False)]
            if np.random.rand() < crossover_rate:
                crossover_point = np.random.randint(1, len(parent1))
                child1 = np.concatenate((parent1[:crossover_point], parent2[crossover_point:]))
                child2 = np.concatenate((parent2[:crossover_point], parent1[crossover_point:]))
            else:
                child1, child2 = parent1, parent2
            if np.random.rand() < mutation_rate:
                child1 = child1 + np.random.rand(len(child1)) * 0.1
                child2 = child2 + np.random.rand(len(child2)) * 0.1
            new_population[i] = child1 if np.random.rand() < 0.5 else child2
        population = new_population
    best_individual = population[np.argmax(fitness)]
    return best_individual

# 粒子群优化示例
def particle_swarm_optimization(fitness_func, population_size, generations, w, c1, c2):
    population = np.random.rand(population_size, len(fitness_func))
    velocity = np.random.rand(population_size, len(fitness_func))
    personal_best = population.copy()
    global_best = personal_best.copy()
    for _ in range(generations):
        fitness = fitness_func(population)
        for i in range(population_size):
            if fitness[i] > fitness[global_best[i]]:
                global_best[i] = population[i]
        for i in range(population_size):
            r1, r2 = np.random.rand(), np.random.rand()
            velocity[i] = w * velocity[i] + c1 * r1 * (personal_best[i] - population[i]) + c2 * r2 * (global_best[i] - population[i])
            population[i] += velocity[i]
        personal_best = population.copy()
    return global_best
