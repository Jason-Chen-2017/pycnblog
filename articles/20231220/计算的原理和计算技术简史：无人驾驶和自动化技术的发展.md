```python
import numpy as np

def control(speed, steering, throttle, brake):
    # 计算加速度
    acceleration = throttle - brake
    # 计算转向力
    torque = steering
    # 更新车辆速度和方向
    speed += acceleration
    steering += torque
    return speed, steering

# 初始化车辆状态
speed = 0
steering = 0

# 模拟驾驶过程
for t in range(100):
    # 更新车辆状态
    speed, steering = control(speed, steering, 0.5, 0.1)
    # 输出车辆状态
    print(f"Time: {t}, Speed: {speed}, Steering: {steering}")
```
