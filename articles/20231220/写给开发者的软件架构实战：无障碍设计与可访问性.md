```python
import tkinter as tk
from tkinter import messagebox

class MyApp(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("无障碍设计和可访问性示例")
        self.geometry("400x300")

        self.entry = tk.Entry(self)
        self.entry.pack()

        self.button = tk.Button(self, text="提交")
        self.button.pack()

        self.button.bind("<Return>", self.submit)
        self.entry.bind("<Return>", self.submit)

    def submit(self, event):
        value = self.entry.get()
        if value:
            messagebox.showinfo("提示", "您输入的内容是：{}".format(value))
        else:
            messagebox.showwarning("警告", "请输入内容")

if __name__ == "__main__":
    app = MyApp()
    app.mainloop()
```
