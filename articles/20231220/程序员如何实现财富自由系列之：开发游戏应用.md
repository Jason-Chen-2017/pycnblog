### 4.1.2 Box2D碰撞检测代码实例

```c++
#include <Box2D/Box2D.h>

b2Vec2 position(0.0f, 0.0f);
b2BodyDef bodyDef;
b2Body* body = nullptr;

void createBody() {
    bodyDef.position = position;
    body = new b2World(b2Vec2(0.0f, 0.0f)).CreateBody(&bodyDef);
}

void checkCollision() {
    b2PolygonShape shape;
    b2FixtureDef fixtureDef;
    fixtureDef.shape = &shape;
    fixtureDef.density = 1.0f;
    body->CreateFixture(&fixtureDef);
}
```

### 4.1.3 FMOD播放音频代码实例

```c++
#include <fmod.hpp>
#include <fmod_studio.hpp>

FMOD::Studio::System* system;
FMOD::Studio::EventInstance* instance;

void playSound() {
    FMOD_STUDIO_EVENT_DESCRIPTOR eventDesc;
    eventDesc.name = "event:/Music/Music";
    eventDesc.userData = nullptr;

    FMOD_RESULT result = system->createEvent(&eventDesc, FMOD_STUDIO_PLAYBACK_DEFAULT, &instance);
    if (result == FMOD_OK) {
        instance->start();
    }
}
```
