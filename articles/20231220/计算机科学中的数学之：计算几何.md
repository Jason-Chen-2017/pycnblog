2. 线：在计算几何中，线是两个点的连线。线可以用向量表示，可以用来表示几何对象的边、直线等。

3. 面：在计算几何中，面是三个点的平面。面可以用向量表示，可以用来表示几何对象的面、平面等。

4. 多边形：在计算几何中，多边形是一组连续的点，它们构成一个闭合的多边形。多边形可以用向量表示，可以用来表示几何对象的多边形、三角形等。

5. 凸包：在计算几何中，凸包是一组点的凸包，它们构成一个凸包。凸包可以用向量表示，可以用来表示几何对象的凸包、凸多边形等。

```python
import math

def closest_pair(points):
    points.sort(key=lambda x: x[1])
    min_dist = float('inf')
    for i in range(len(points) - 1):
        x1, y1 = points[i]
        x2, y2 = points[i + 1]
        dist = math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)
        if dist < min_dist:
            min_dist = dist
    return min_dist

points = [(1, 2), (3, 4), (5, 6), (7, 8)]
print(closest_pair(points))
```

```python
def convex_hull(points):
    points.sort(key=lambda x: x[1])
    hull = []
    for p in points:
        while len(hull) >= 2 and cross(hull[-2], hull[-1], p) < 0:
            hull.pop()
        hull.append(p)
    return hull

def cross(a, b, c):
    return (b[0] - a[0]) * (c[1] - a[1]) - (b[1] - a[1]) * (c[0] - a[0])

points = [(1, 2), (3, 4), (5, 6), (7, 8)]
print(convex_hull(points))
```

```python
def point_in_polygon(p, polygon):
    n = len(polygon)
    cross = 0
    for i in range(n):
        v1 = polygon[i] - p
        v2 = polygon[(i + 1) % n] - p
        if v1[1] > v2[1]:
            v1, v2 = v2, v1
        if v1[1] <= 0 and v2[1] > 0 and cross_product(v1, v2) < 0:
            cross += 1
    return cross % 2 == 0

def cross_product(v1, v2):
    return v1[0] * v2[1] - v1[1] * v2[0]

polygon = [(0, 0), (2, 0), (2, 2), (0, 2)]
p = (1, 1)
print(point_in_polygon(p, polygon))
```

```python
def polygon_area(polygon):
    area = 0
    for i in range(len(polygon)):
        p1 = polygon[i]
        p2 = polygon[(i + 1) % len(polygon)]
        area += p1[0] * p2[1] - p1[1] * p2[0]
```
