```python
from sklearn.linear_model import LinearRegression
from sklearn.optimize import minimize

# 构建多变量回归模型
model = LinearRegression()
model.fit(data[['temp', 'humidity', 'light']], data['yield'])

# 优化模型参数
def objective_function(params):
    model = LinearRegression(fit_intercept=False)
    model.coef_ = params
    return model.score(data[['temp', 'humidity', 'light']], data['yield'])

params = minimize(objective_function, np.zeros(3), bounds=[(-1, 1) for _ in range(3)])
```
