# 4.具体代码实例和详细解释说明

在本节中，我们将通过一个简单的代码实例来说明领域定义与表示的具体实现。我们将实现一个简单的数学表达式求值器，该求值器可以处理加法、减法、乘法和除法四种基本运算。

```python
import re

class ExpressionEvaluator:
    def __init__(self):
        self.operators = {'+': 1, '-': 1, '*': 2, '/': 2}
        self.precedence = {'+': 1, '-': 1, '*': 2, '/': 2}

    def parse(self, expression):
        tokens = re.split(r'\s+', expression)
        operators = []
        operands = []
        for token in tokens:
            if token in self.operators:
                if len(operands) == 0:
                    operands.append(token)
                else:
                    while len(operands) > 0 and self.precedence[operators[-1]] <= self.precedence[token]:
                        operands.append(operators.pop())
                    operands.append(token)
            else:
                operands.append(token)
        while len(operators) > 0:
            operands.append(operators.pop())
        return operands

    def evaluate(self, expression):
        operands = self.parse(expression)
        stack = []
        for operand in operands:
            if operand.isdigit():
                stack.append(int(operand))
            else:
                right = stack.pop()
                left = stack.pop()
                if operand == '+':
                    stack.append(left + right)
                elif operand == '-':
                    stack.append(left - right)
                elif operand == '*':
                    stack.append(left * right)
                elif operand == '/':
                    stack.append(left / right)
        return stack[0]

if __name__ == '__main__':
    evaluator = ExpressionEvaluator()
    expression = '3 + 5 * 2 - 4 / 2'
    result = evaluator.evaluate(expression)
    print(result)
```
