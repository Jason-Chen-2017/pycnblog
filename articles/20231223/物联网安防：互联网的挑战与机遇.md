import hmac

import socket

    "device_id": "1234567890",
    "device_type": "sensor",
    "device_ip": "192.168.1.100"
}

# 用户信息
user_info = {
    "user_id": "1234567890",
    "user_type": "admin"
}

# 设备认证
def device_authentication(device_info, user_info):
    # 生成设备认证请求
    device_request = json.dumps(device_info)
    # 生成用户认证请求
    user_request = json.dumps(user_info)
    # 生成HMAC认证签名
    key = b"secret_key"
    signature = hmac.new(key, msg=user_request.encode(), digestmod=hashlib.sha256).digest()
    # 返回认证签名
    return signature

# 主程序
if __name__ == "__main__":
    # 获取设备认证签名
    signature = device_authentication(device_info, user_info)
    # 发送设备认证请求
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.connect(("192.168.1.100", 8080))
    sock.sendall(bytes(device_request, "utf-8"))
    response = sock.recv(1024)
    sock.close()
    # 解析响应
    response_dict = json.loads(response.decode())
    # 判断设备认证是否成功
    if response_dict["status"] == "success":
        print("设备认证成功")
    else:
        print("设备认证失败")
