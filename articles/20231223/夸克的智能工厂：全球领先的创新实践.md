    # 生产任务的时间和资源需求
    time_demand = tasks[:, 0]
    resource_demand = tasks[:, 1]

    # 生产线资源的时间和资源需求

    line_resource_demand = line_resources[:, 1]

    # 线性规划问题
    c = np.zeros(len(tasks))
    A = np.hstack((np.eye(len(tasks)), -tasks))
    b = np.hstack((np.full(len(tasks), resources[0]), resources[1]))

    # 调度生产任务
    result = linprog(c, A_ub=A, b_ub=b, bounds=[(0, 1) for _ in range(len(tasks))])
