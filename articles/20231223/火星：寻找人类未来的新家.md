import matplotlib.pyplot as plt

    data = np.loadtxt(file_path)
    return data

def preprocess_data(data):
    data = data - np.mean(data)
    data = data / np.std(data)
    return data

def analyze_data(data):
    slope = np.polyfit(np.arange(len(data)), data, 1)
    return slope

def plot_data(slope):
    plt.plot(np.arange(len(slope)), slope)
    plt.xlabel('Index')
    plt.ylabel('Slope')
    plt.title('Firestar Topography Analysis')
    plt.show()

if __name__ == '__main__':
    file_path = 'firestar_topography.txt'
    data = load_data(file_path)
    data = preprocess_data(data)
    slope = analyze_data(data)
    plot_data(slope)
