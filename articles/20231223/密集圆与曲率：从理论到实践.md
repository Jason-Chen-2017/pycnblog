### 2.3 密集圆

密集圆是密集圆算法的核心概念，它是一个圆形区域，用于描述目标物体的边界框和曲率。密集圆通常由一个圆心和一个半径组成，其中圆心表示目标物体的中心，半径表示目标物体的大小。

import cv2
import numpy as np

def detect_bounding_box(image):
    gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    edges = cv2.Canny(gray_image, 100, 200)
    contours, hierarchy = cv2.findContours(edges, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)
    for contour in contours:
        x, y, w, h = cv2.boundingRect(contour)
        cv2.rectangle(image, (x, y), (x + w, y + h), (0, 255, 0), 2)
    return image

def compute_curvature(bounding_box):
    x, y, w, h = bounding_box
    k = (x - w / 2 - x - w / 2) / (y - h / 2 - y + h / 2)
    return k

def build_dense_circle(bounding_box, curvature):
    x, y, w, h = bounding_box
    r = np.sqrt((w / 2) ** 2 + (h / 2) ** 2) / 2
    cx = x + r * np.cos(np.arctan2(h / 2, w / 2))
    cy = y + r * np.sin(np.arctan2(h / 2, w / 2))
    return cx, cy, r

bounding_box = detect_bounding_box(image)
curvature = compute_curvature(bounding_box)
dense_circle = build_dense_circle(bounding_box, curvature)

cv2.circle(bounding_box, (int(dense_circle[0]), int(dense_circle[1])), int(dense_circle[2]), (0, 0, 255), 2)
cv2.imshow('Dense Circle', bounding_box)
cv2.waitKey(0)
cv2.destroyAllWindows()
