## 1. D3.js

D3.js是一个非常强大的数据可视化库，它可以创建各种类型的数据可视化组件，例如线图、柱状图、饼图等。D3.js使用HTML、CSS和SVG来渲染图表，并且可以根据数据和配置选项动态地更新图表。

以下是一个简单的D3.js示例代码：

```javascript
// 导入D3.js库
var d3 = require('d3');

// 创建一个简单的数据集
var data = [
  { x: 0, y: 8 },
  { x: 1, y: 2 },
  { x: 2, y: 4 },
  { x: 3, y: 6 },
  { x: 4, y: 8 }
];

// 创建一个简单的线图
var svg = d3.select('body')
  .append('svg')
  .attr('width', 500)
  .attr('height', 500);

var line = d3.line()
  .x(function(d) { return d.x; })
  .y(function(d) { return d.y; });

svg.append('path')
  .datum(data)
  .attr('d', line);
```

在这个示例中，我们首先导入了D3.js库，然后创建了一个简单的数据集。接着，我们创建了一个简单的线图，并使用`d3.line()`函数绘制了线路。

## 2. Chart.js

Chart.js是一个轻量级的数据可视化库，它可以创建各种类型的数据可视化组件，例如线图、柱状图、饼图等。Chart.js使用HTML5的canvas元素来渲染图表，并且可以根据数据和配置选项动态地更新图表。

以下是一个简单的Chart.js示例代码：

```javascript
// 导入Chart.js库
var Chart = require('chart.js');

// 创建一个简单的数据集
var data = {
  labels: ['Red', 'Blue', 'Yellow', 'Green', 'Purple', 'Orange'],
  datasets: [{
    label: '# of Votes',
    data: [12, 19, 3, 5, 2, 3],
    backgroundColor: [
      'rgba(255, 99, 132, 0.2)',
      'rgba(54, 162, 235, 0.2)',
      'rgba(255, 206, 86, 0.2)',
      'rgba(75, 192, 192, 0.2)',
      'rgba(153, 102, 255, 0.2)',
      'rgba(255, 159, 64, 0.2)'
    ],
    borderColor: [
      'rgba(255, 99, 132, 1)',
      'rgba(54, 162, 235, 1)',
      'rgba(255, 206, 86, 1)',
      'rgba(75, 192, 192, 1)',
      'rgba(153, 102, 255, 1)',
      'rgba(255, 159, 64, 1)'
    ],
    borderWidth: 1
  }]
};

// 创建一个简单的饼图
var ctx = document.getElementById('chart').getContext('2d');
var chart = new Chart(ctx, {
  type: 'pie',
  data: data,
  options: {
    responsive: true,
    maintainAspectRatio: false
  }
});
```

在这个示例中，我们首先导入了Chart.js库，然后创建了一个简单的数据集。接着，我们创建了一个简单的饼图，并使用`new Chart()`函数绘制了饼图。

  ['Year', 'Sales', 'Expenses'],
  [2014, 1000, 400],
