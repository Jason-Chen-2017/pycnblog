```python
import time
import board
import busio
import adafruit_bme280

# 初始化BME280传感器
i2c = busio.I2C(board.SCL, board.SDA)
bme280 = adafruit_bme280.Adafruit_BME280_I2C(i2c)

# 主程序循环
while True:
    # 获取心率数据
    heart_rate = bme280.heart_rate
    print("心率：", heart_rate, "bpm")

    # 等待1秒
    time.sleep(1)
```
