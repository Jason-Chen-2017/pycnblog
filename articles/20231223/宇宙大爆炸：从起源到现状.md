```python
import numpy as np
import matplotlib.pyplot as plt

def expand(R0, H, t):
    R = R0 * np.exp(H * t)
    return R

R0 = 1
H = 0.21
t = np.linspace(0, 13.8, 100)

R = expand(R0, H, t)

plt.plot(t, R)
plt.xlabel('Time (Gyr)')
plt.ylabel('Radius (Gpc)')
plt.title('Universe Expansion')
plt.show()
```
