## 4.1 代码实例

```

CREATE DATABASE medical;

-- 使用新创建的数据库
USE medical;

-- 创建一个新的数据表
CREATE TABLE patient (
    id INT PRIMARY KEY,
    name VARCHAR(255),
    age INT,
    gender ENUM('male', 'female'),
    disease VARCHAR(255)
);

-- 插入数据
INSERT INTO patient (id, name, age, gender, disease) VALUES (1, 'John Doe', 30, 'male', 'flu');
INSERT INTO patient (id, name, age, gender, disease) VALUES (2, 'Jane Smith', 25, 'female', 'cold');

-- 查询数据
SELECT * FROM patient;

-- 更新数据
UPDATE patient SET age = 31 WHERE id = 1;

-- 删除数据
DELETE FROM patient WHERE id = 2;
```
