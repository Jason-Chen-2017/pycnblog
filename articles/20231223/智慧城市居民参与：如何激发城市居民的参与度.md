```python
import numpy as np

def evaluate_participation(data):
    participation = np.sum(data[:, 0] * data[:, 1]) / np.sum(data[:, 1])
    return participation

def boost_participation(data, alpha=0.1):
    participation = evaluate_participation(data)
    new_participation = participation + alpha * (data[:, 2] - participation)
    return new_participation

data = np.array([[1, 3, 5], [2, 4, 6], [3, 5, 7]])
participation = evaluate_participation(data)
new_participation = boost_participation(data)

print("参与度:", participation)
print("新参与度:", new_participation)
```
