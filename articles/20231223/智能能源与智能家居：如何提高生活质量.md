```python
class Battery:
    def __init__(self, capacity):
        self.capacity = capacity
        self.energy = 0

    def charge(self, power):
        self.energy += power
        if self.energy > self.capacity:
            self.energy = self.capacity

    def discharge(self, power):
        self.energy -= power
        if self.energy < 0:
            self.energy = 0

battery = Battery(100)
battery.charge(50)
battery.discharge(20)
```
