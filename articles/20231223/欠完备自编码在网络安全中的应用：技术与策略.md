encoding_dim = 32  # 编码器输出的维度

hidden_layer = Dense(encoding_dim, activation='relu')(input_layer)
output_layer = Dense(input_dim, activation='sigmoid')(hidden_layer)

autoencoder = Model(input_layer, output_layer)
