```python
import numpy as np
from scipy.spatial import distance

# 评价指标和权重

# 事物评分
scores = {
    '北方城市': [80, 70, 85],
    '南方城市': [70, 80, 75],
    '西部城市': [60, 65, 80],
    '东部城市': [75, 70, 78],
}

# 构建模糊关系矩阵
def fuzzy_relation_matrix(scores):
    relation_matrix = []
    for city in scores:
        relation_row = []
        for i in range(len(scores[city])):
            relation_row.append(1 / scores[city][i])
        relation_matrix.append(relation_row)
    return relation_matrix

# 求模糊综合评价结果
def fuzzy_comprehensive_evaluation(weights, relation_matrix):
    evaluation = []
    for city in relation_matrix:
        evaluation.append(np.dot(weights, city))
    return evaluation

# 主程序
relation_matrix = fuzzy_relation_matrix(scores)
evaluation = fuzzy_comprehensive_evaluation(weights, relation_matrix)
print(evaluation)
```
