```python
import matplotlib.pyplot as plt

def direct_antigen_capture(antigen, antibody, capture_ratio):
    captured_antigen = antigen * capture_ratio

    return captured_antigen, free_antigen

antigen = 100
```

```python
import numpy as np
import matplotlib.pyplot as plt

def cell_dynamics(cell_num, growth_rate, death_rate, time):
    dt = 1
    cell_list = [cell_num]
    for _ in range(time // dt):
        new_cell_list = []
        for cell in cell_list:
            new_cell_list.append(cell + growth_rate * dt)
            if np.random.rand() < death_rate * dt:
                new_cell_list.append(0)
        cell_list = new_cell_list
    return cell_list

cell_num = 100
growth_rate = 0.01
death_rate = 0.005
time = 100

cell_list = cell_dynamics(cell_num, growth_rate, death_rate, time)
plt.plot(cell_list)
plt.xlabel("Time")
plt.ylabel("Cell number")
plt.show()
```
