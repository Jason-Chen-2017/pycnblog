# 4.2代码实例
以下是一个简单的Python代码实例，用于从Twitter获取数据，并进行基本的数据分析。

```python
import tweepy
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.decomposition import LatentDirichletAllocation

# 设置Twitter API密钥和密码
consumer_key = 'your_consumer_key'
consumer_secret = 'your_consumer_secret'
access_token = 'your_access_token'
access_token_secret = 'your_access_token_secret'

# 设置Twitter API参数
auth = tweepy.OAuthHandler(consumer_key, consumer_secret)
auth.set_access_token(access_token, access_token_secret)
api = tweepy.API(auth)

# 获取Twitter数据
tweets = api.search(q='health', count=100, lang='en')

# 提取数据
tweet_texts = [tweet.text for tweet in tweets]

# 数据预处理
tweet_texts_clean = [tweet.lower() for tweet in tweet_texts if tweet.lower().find('http') == -1 and tweet.lower().find('www') == -1 and tweet.lower().find('https') == -1]

# 数据分析
vectorizer = TfidfVectorizer()
tfidf_matrix = vectorizer.fit_transform(tweet_texts_clean)
lda = LatentDirichletAllocation(n_components=2)
lda.fit(tfidf_matrix)

# 数据可视化
plt.scatter(lda.transform(tfidf_matrix)[0:100, 0], lda.transform(tfidf_matrix)[0:100, 1])
plt.show()
```
