$$
\frac{dT}{dt} = \frac{1}{c_p} \frac{dQ}{dt}

其中，$T$ 表示大气温度，$t$ 表示时间，$c_p$ 表示大气热容，$Q$ 表示热量。

## 4.2 仿真技术实现

```python
import numpy as np
import matplotlib.pyplot as plt

# 定义大气模型参数
cp = 1000.0  # 大气热容
T0 = 288.0  # 初始大气温度
Q0 = 0.0  # 初始热量
dt = 3600.0  # 时间步长

# 定义仿真技术函数
def simulate(T0, Q0, cp, dt):
    T = np.zeros(int(1.0 / dt))
    T[0] = T0
    Q = np.zeros(int(1.0 / dt))
    Q[0] = Q0

    for i in range(int(1.0 / dt)):
        T[i + 1] = T[i] + dt / cp * Q[i]
        Q[i + 1] = Q[i]

    return T, Q

# 运行仿真
T, Q = simulate(T0, Q0, cp, dt)

# 绘制大气温度变化曲线
plt.plot(T)
plt.xlabel('Time (s)')
plt.ylabel('Temperature (K)')
plt.title('Atmospheric Temperature Variation')
plt.show()
```
