3. 随机森林：随机森林是一种集成学习方法，可以用于对客户风险进行多分类。随机森林的主要优点是高准确率、高泛化能力、高效率。

```python
import numpy as np
import random

class RandomForest:
    def __init__(self, n_estimators=100, max_depth=10):
        self.n_estimators = n_estimators
        self.max_depth = max_depth
        self.trees = [self._build_tree(X_train, y_train) for i in range(self.n_estimators)]

    def _build_tree(self, X, y):
        if len(np.unique(y)) == 1:
            return self._majority_vote(X)
        if len(X.columns) == 1:
            return self._majority_vote(X)
        best_feature = self._select_best_feature(X, y)
        threshold = self._find_best_threshold(X[:, best_feature], y)
        left_idx = np.array(range(len(y)))[X[:, best_feature] <= threshold]
        right_idx = np.array(range(len(y)))[X[:, best_feature] > threshold]
        left_tree = self._build_tree(X.iloc[left_idx].drop(best_feature, axis=1), y[left_idx])
        right_tree = self._build_tree(X.iloc[right_idx].drop(best_feature, axis=1), y[right_idx])
        return self._construct_tree(best_feature, threshold, left_tree, right_tree)

    def _majority_vote(self, X):
        return np.argmax(np.bincount(X))

    def _select_best_feature(self, X, y):
        best_feature, best_gain = None, -1
        for feature in X.columns:
            gain = self._gain(X[:, feature], y)
            if gain > best_gain:
                best_gain = gain
                best_feature = feature
        return best_feature

    def _find_best_threshold(self, X_column, y):
        threshold_gain = {}
        for threshold in np.unique(X_column):
            left_idx = np.array(range(len(y)))[X_column <= threshold]
            right_idx = np.array(range(len(y)))[X_column > threshold]
            left_gain = self._gain(X_column[left_idx], y[left_idx])
            right_gain = self._gain(X_column[right_idx], y[right_idx])
            gain = left_gain + right_gain - len(y) / 2
            threshold_gain[threshold] = gain
        return np.argmax(threshold_gain.values())

    def _gain(self, X_column, y):
        entropy = self._entropy(y)
        left_idx = np.array(range(len(y)))[X_column <= self._median(X_column)]
        right_idx = np.array(range(len(y)))[X_column > self._median(X_column)]
        left_entropy = self._entropy(y[left_idx])
        right_entropy = self._entropy(y[right_idx])
        return entropy - (left_entropy * len(y[left_idx]) + right_entropy * len(y[right_idx])) / len(y)

    def _entropy(self, y):
        hist = np.bincount(y)
        p = hist / len(y)
        return -np.sum(p * np.log2(p))

    def _median(self, X_column):
        sorted_values = np.sort(X_column)
        n = len(sorted_values)
        return (sorted_values[n // 2] + sorted_values[(n - 1) // 2]) / 2

    def _construct_tree(self, feature, threshold, left_tree, right_tree):
        return {'feature': feature, 'threshold': threshold, 'left': left_tree, 'right': right_tree}
```

```python
}

```

我们可以使用随机森林算法来进行车辆风险评估。首先，我们需要将数据分为训练集和测试集：

```python
from sklearn.model_selection import train_test_split

X = df.drop('风险等级', axis=1)
```

