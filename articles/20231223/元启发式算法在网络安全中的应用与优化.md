import random

def initialize_ants(n, n_ants):
    ants = []
    for i in range(n_ants):
        ant = [random.randint(0, n - 1) for _ in range(n)]
        ants.append(ant)
    return ants

def pheromone_update(ants, pheromone, alpha):
    for ant in ants:
        for i in range(len(ant) - 1):
            pheromone[ant[i], ant[i + 1]] += pheromone[ant[i], ant[i + 1]] * alpha

def ant_system(n, n_ants, pheromone, alpha, beta, evaporation_rate, max_iter):
    for _ in range(max_iter):
        ants = initialize_ants(n, n_ants)
        pheromone_total = np.sum(pheromone)
        for ant in ants:
            for i in range(len(ant) - 1):
                pheromone_current = pheromone[ant[i], ant[i + 1]] / pheromone_total
                pheromone_next = pheromone[ant[i], ant[i + 1]] * (pheromone_current ** alpha) * (1 / (1 + len(ant))) ** beta
                pheromone[ant[i], ant[i + 1]] = pheromone_next
        pheromone_total = np.sum(pheromone)
        pheromone -= pheromone * evaporation_rate
    return ants
