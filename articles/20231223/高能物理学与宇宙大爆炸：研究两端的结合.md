import numpy as np
import matplotlib.pyplot as plt

def collision(particle1, particle2):
    # 计算碰撞时间
    t = (particle1.mass + particle2.mass) / (2 * particle1.energy)
    # 计算碰撞位置
    x = particle1.position + particle1.velocity * t
    # 计算碰撞后粒子的速度和方向
    v = particle1.velocity + particle2.velocity
    particle1.energy = np.sqrt(v**2 + particle1.mass**2)
    particle2.energy = np.sqrt(v**2 + particle2.mass**2)
    return x, particle1, particle2

# 初始化粒子
particle1 = Particle(mass=1, energy=100, position=[0, 0], velocity=[1, 0])
particle2 = Particle(mass=1, energy=100, position=[1, 0], velocity=[-1, 0])

# 模拟碰撞

# 绘制粒子轨迹
plt.plot([0, x], [0, 0], 'o-')
plt.xlim(-10, 10)
plt.ylim(-10, 10)
plt.show()

## 6.1 高能物理学常见问题

Q: 高能粒子与普通粒子有什么区别？

A: 高能粒子通常指具有较高能量的粒子，如电子、中子、反物质粒子等。普通粒子指具有较低能量的粒子，如氢原子核、氢子等。

Q: 高能物理学与核物理学有什么区别？

A: 高能物理学主要研究高能粒子的性质和行为，而核物理学主要研究核子和核粒子的性质和行为。高能物理学和核物理学在一定程度上相互关联，但是它们具有相对独立的研究内容和方法。
