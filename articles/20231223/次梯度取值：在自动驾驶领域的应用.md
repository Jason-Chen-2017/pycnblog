```python
def gradient_descent(X, y, theta, learning_rate, iterations):
    m = len(y)
    for i in range(iterations):
        hypothesis = np.dot(X, theta)
        gradient = (1 / m) * np.dot(X.T, (hypothesis - y))
        theta -= learning_rate * gradient
    return theta
```
