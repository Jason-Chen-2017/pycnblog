## 4.1 交通信号灯控制
### 4.1.1 获取交通信号灯的实时视频
```python
import cv2

# 获取交通信号灯的实时视频
cap = cv2.VideoCapture('traffic_light.mp4')
```
### 4.1.2 分析视频中的交通情况
```python
import numpy as np

# 定义交通信号灯的颜色
colors = ['red', 'yellow', 'green']

# 定义交通信号灯的状态
status = ['stop', 'prepare', 'go']

# 定义车辆的颜色
vehicle_colors = ['red', 'green', 'yellow']

# 定义车辆的状态
vehicle_status = ['stop', 'go']

# 分析视频中的交通情况
while True:
    ret, frame = cap.read()
    if not ret:
        break

    # 检测交通信号灯
    for color in colors:
        for status in status:
            rect = cv2.selectROI('Select ROI', frame, fromCenter=False, showCrosshair=True)
            mask = cv2.inRange(frame, np.array([color]), np.array([color]))
            count = cv2.countNonZero(mask)
            if count > 0:
                print(f'{color} {status} detected')

    # 检测车辆
    for color in vehicle_colors:
        for status in vehicle_status:
            rect = cv2.selectROI('Select ROI', frame, fromCenter=False, showCrosshair=True)
            mask = cv2.inRange(frame, np.array([color]), np.array([color]))
            count = cv2.countNonZero(mask)
            if count > 0:
                print(f'{color} {status} detected')

    cv2.imshow('Traffic Light', frame)

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```
