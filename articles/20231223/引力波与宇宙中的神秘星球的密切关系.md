```python
import matplotlib.pyplot as plt

# 设置引力波的振动强度、频率和相位
h_plus = 1e-21
frequency = 1e2  # Hz
phase = np.pi / 4

# 生成引力波的时间序列
time = np.linspace(0, 1, 1000)

# 绘制引力波的时间序列
plt.plot(time, wave)
plt.xlabel('Time (s)')
plt.ylabel('Amplitude')
plt.title('Gravitational Wave Time Series')
plt.show()
```
