import numpy as np

# 生态保护措施和生态保护需求的对应关系表
match_table = {
    'forest_protection': ['tropical_rainforest', 'temperate_forest', 'boreal_forest'],
    'wetland_restoration': ['coastal_wetland', 'inland_wetland'],
    'coral_reef_conservation': ['coral_reef', 'mangrove'],
}

# 生态保护措施和生态保护需求的成本和优先级
cost_table = {
    'forest_protection': {'tropical_rainforest': 1000, 'temperate_forest': 800, 'boreal_forest': 600},
    'wetland_restoration': {'coastal_wetland': 1200, 'inland_wetland': 900},
    'coral_reef_conservation': {'coral_reef': 1400, 'mangrove': 1100},
}

# 匹配生态保护措施和生态保护需求
def match_protection_measures(match_table, cost_table):
    matched_measures = []
    for measure, areas in match_table.items():
        total_cost = sum(cost_table[measure].values())
        matched_areas = []
        for area in areas:
            area_cost = cost_table[measure][area]
            if total_cost - area_cost >= 0:
                matched_areas.append(area)
                total_cost -= area_cost
        matched_measures.append((measure, matched_areas))
    return matched_measures

matched_measures = match_protection_measures(match_table, cost_table)
print(matched_measures)
