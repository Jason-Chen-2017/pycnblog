### 4.2.5 Python 推荐系统代码实例

```python
from sklearn.metrics.pairwise import cosine_similarity

# 加载数据
data = pd.read_csv('data.csv')
user_ids = data['user_id']
item_ids = data['item_id']
ratings = data['rating']

# 数据预处理
user_id_map = {}
item_id_map = {}
user_id_counter = 0
item_id_counter = 0
for user_id, item_id, rating in zip(user_ids, item_ids, ratings):
    if user_id not in user_id_map:
        user_id_map[user_id] = user_id_counter
        user_id_counter += 1
    if item_id not in item_id_map:
        item_id_map[item_id] = item_id_counter
        item_id_counter += 1

# 构建用户-项目矩阵
user_item_matrix = np.zeros((len(user_id_map), len(item_id_map)))
for user_id, item_id, rating in zip(user_ids, item_ids, ratings):
    user_item_matrix[user_id_map[user_id]][item_id_map[item_id]] = rating

# 计算用户-项目矩阵的余弦相似度
similarity = cosine_similarity(user_item_matrix.T)

# 推荐系统
def recommend(user_id, similarity, num_recommendations=5):
    user_id = user_id_map[user_id]
    similarity = similarity[user_id]
    recommendations = np.argsort(-similarity)[1:num_recommendations + 1]
    return recommendations

# 测试推荐系统
user_id = 0
recommendations = recommend(user_id, similarity)
print('推荐项目:', recommendations)
```
