import numpy as np

def cultural_similarity(cultural_features):
    n = cultural_features.shape[1]
    numerator = np.sum(np.min(cultural_features, axis=0))
    denominator_i = np.sqrt(np.sum(cultural_features**2, axis=0))
    denominator_j = denominator_i.copy()
    return numerator / (denominator_i * denominator_j)

cultural_features = np.array([[1, 2], [3, 4], [5, 6]])

def pca(data, n_components=2):
