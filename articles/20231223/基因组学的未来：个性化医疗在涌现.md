### 3.1.1最长公共子序列（LCS）算法

最长公共子序列（LCS）算法是一种用于找到两个序列中最长公共子序列的算法。这个算法可以用于解决基因组组装中的一些问题，如找到两个相邻的DNA片段之间的最长公共子序列。

LCS算法的基本思想是动态规划。我们可以使用一个二维表格来存储每个位置的最长公共子序列长度。然后，我们可以使用这个表格来找到最长公共子序列。

LCS算法的时间复杂度是O(m*n)，其中m和n分别是两个序列的长度。这意味着LCS算法的时间复杂度是线性的，所以它可以用于解决大规模的基因组组装问题。

def lcs(X, Y):
    m = len(X)
    n = len(Y)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1):
        for j in range(n + 1):
            if i == 0 or j == 0:
                dp[i][j] = 0
            elif X[i - 1] == Y[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
    result = ""
    i = m
    j = n
    while i > 0 and j > 0:
        if X[i - 1] == Y[j - 1]:
            result = X[i - 1] + result
            i -= 1
            j -= 1
        elif dp[i - 1][j] > dp[i][j - 1]:
            i -= 1
        else:
            j -= 1
    return result

import numpy as np

def emit(observation, hidden_state):
    # 观测值
    observation = np.array([observation])
    # 隐藏状态
    hidden_state = np.array([hidden_state])
    # 模型参数
    transition_probability = np.array([[0.5, 0.5]])
    emission_probability = np.array([[0.5, 0.5]])
    # 计算概率
    emission_probability = np.dot(emission_probability, observation)
    transition_probability = np.dot(transition_probability, hidden_state)
    return emission_probability, transition_probability

def viterbi(observations, hidden_states):
    # 初始化变量
    V = np.zeros((len(observations), len(hidden_states)))
    P = np.zeros((len(observations), len(hidden_states)))
    # 遍历观测值
    for t in range(len(observations)):
        # 遍历隐藏状态
        for i in range(len(hidden_states)):
            # 计算概率
            emission_probability, transition_probability = emit(observations[t], hidden_states[i])
            # 更新变量
            if t == 0:
                V[t][i] = emission_probability
                P[t][i] = 1.0
            else:
                V_temp = V[t - 1][:]
                P_temp = P[t - 1][:]
                for j in range(len(hidden_states)):
                    if transition_probability[j] > P_temp[j]:
                        V_temp[j] = 0.0
                        P_temp[j] = 0.0
                V_temp[i] = emission_probability + np.max(V[t - 1])
                P_temp[i] = transition_probability
                V[t] = V_temp
                P[t] = P_temp
    # 返回最大概率和路径
    path = np.argmax(V[-1])
    path = [path] + [np.argmax(P[:, path]) for _ in range(len(observations) - 1)]
    return path
