def needman_wunsch(seq1, seq2):
    len1 = len(seq1)
    len2 = len(seq2)
    score_matrix = [[0 for _ in range(len2 + 1)] for _ in range(len1 + 1)]
    gap_penalty = -1
    for i in range(len1 + 1):
        score_matrix[i][0] = gap_penalty * i
    for j in range(len2 + 1):
        score_matrix[0][j] = gap_penalty * j
    for i in range(1, len1 + 1):
        for j in range(1, len2 + 1):
            match = seq1[i - 1] == seq2[j - 1]
            score_matrix[i][j] = max(score_matrix[i - 1][j - 1] + match,
                                     score_matrix[i - 1][j] + gap_penalty,
                                     score_matrix[i][j - 1] + gap_penalty)
    traceback = [['' for _ in range(len2 + 1)] for _ in range(len1 + 1)]
    i, j = len1, len2
    while i > 0 or j > 0:
        match = seq1[i - 1] == seq2[j - 1]
        if i > 0 and j > 0 and score_matrix[i][j] == score_matrix[i - 1][j - 1] + match:
            traceback[i][j] = 'diag'
            i -= 1
            j -= 1
        elif i > 0 and score_matrix[i][j] == score_matrix[i - 1][j] + gap_penalty:
            traceback[i][j] = 'up'
            i -= 1
        else:
            traceback[i][j] = 'left'
            j -= 1
    align1, align2 = '', ''
    while i > 0 or j > 0:
        if traceback[i][j] == 'diag':
            align1 += seq1[i - 1]
            align2 += seq2[j - 1]
            i -= 1
            j -= 1
        elif traceback[i][j] == 'up':
            align1 += seq1[i - 1]
            align2 += '-'
            i -= 1
        else:
            align1 += '-'
            align2 += seq2[j - 1]
            j -= 1
    return align1, align2
