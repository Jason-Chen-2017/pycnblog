def force(x, y):
    return -x * y

# 构建数学模型
def dydt(y, x):
    return -y + force(x, y)

# 求解数学模型
def solve(x0, y0, h, t_end):
    y = y0
    t = 0
    while t < t_end:
        y_new = y + h * dydt(y, x0)
        y = y_new
        t += h
    return y

# 主程序

t_end = 10.0

# 绘图

plt.xlabel('x')
plt.ylabel('y')
plt.title('Strongly Related Physics Example')
