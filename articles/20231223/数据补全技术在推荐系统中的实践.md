## 3.1 协同过滤

协同过滤是一种基于用户行为的推荐算法，它通过对用户的行为数据进行分析，找出相似的用户或者物品，从而生成个性化的推荐列表。协同过滤可以分为基于用户的协同过滤和基于物品的协同过滤。

### 3.1.1 基于用户的协同过滤

基于用户的协同过滤是一种基于用户之间的相似性的协同过滤方法。它通过计算用户之间的相似性，找出与目标用户最相似的其他用户，从而生成个性化的推荐列表。基于用户的协同过滤可以分为以下几种方法：

1. 欧氏距离：欧氏距离是一种用于计算两个用户之间相似性的方法，它通过计算两个用户的行为数据之间的欧氏距离，从而得到用户之间的相似性。
2. 皮尔逊相关系数：皮尔逊相关系数是一种用于计算两个用户之间相似性的方法，它通过计算两个用户的行为数据之间的皮尔逊相关系数，从而得到用户之间的相似性。
3. 用户-用户协同过滤：用户-用户协同过滤是一种基于用户的协同过滤方法，它通过计算用户之间的相似性，找出与目标用户最相似的其他用户，从而生成个性化的推荐列表。

### 3.1.2 基于物品的协同过滤

基于物品的协同过滤是一种基于物品之间的相似性的协同过滤方法。它通过计算物品之间的相似性，找出与目标物品最相似的其他物品，从而生成个性化的推荐列表。基于物品的协同过滤可以分为以下几种方法：

## 4.1 协同过滤

### 4.1.1 基于用户的协同过滤

```python
import numpy as np
from scipy.spatial.distance import cosine

# 用户行为数据
user_behavior_data = {
    'user1': [1, 2, 3],
    'user2': [2, 3, 4],
    'user3': [1, 3, 5],
}

# 计算用户之间的相似性
def user_similarity(user_behavior_data):
    similarity = {}
    for user1 in user_behavior_data:
        for user2 in user_behavior_data:
            if user1 != user2:
                similarity[(user1, user2)] = 1 - cosine(user_behavior_data[user1], user_behavior_data[user2])
    return similarity

# 找出与目标用户最相似的其他用户
def find_similar_users(user_similarity, target_user):
    similar_users = []
    max_similarity = 0
    for user, similarity in user_similarity.items():
        if user == target_user:
            continue
        if similarity > max_similarity:
            max_similarity = similarity
            similar_users = [user]
        elif similarity == max_similarity:
            similar_users.append(user)
    return similar_users

# 生成个性化的推荐列表
def recommend(user_behavior_data, target_user):
    similar_users = find_similar_users(user_similarity, target_user)
    recommended_items = set()
    for user in similar_users:
        recommended_items.update(set(user_behavior_data[user]))
    return list(recommended_items)

# 测试
print(recommend(user_behavior_data, 'user1'))
```

## 4.2 矩阵分解

### 4.2.1 SVD

```python
import numpy as np
from scipy.sparse.linalg import svds

# 用户行为数据
user_behavior_data = {
    'user1': [1, 2, 3],
    'user2': [2, 3, 4],
    'user3': [1, 3, 5],
}

# 将用户行为数据转换为矩阵
user_behavior_matrix = np.array([user_behavior_data[user] for user in user_behavior_data])

# 进行奇异值分解
U, sigma, Vt = svds(user_behavior_matrix, k=2)

# 重构矩阵
sigma_new = np.diag(np.sqrt(np.maximum(0, sigma)))
reconstructed_matrix = U @ sigma_new @ Vt

# 测试
print(reconstructed_matrix)
```

### 4.2.2 NMF

```python
import numpy as np
from scipy.optimize import minimize
from scipy.sparse.linalg import norm

# 用户行为数据
user_behavior_data = {
    'user1': [1, 2, 3],
    'user2': [2, 3, 4],
    'user3': [1, 3, 5],
}

# 将用户行为数据转换为矩阵
user_behavior_matrix = np.array([user_behavior_data[user] for user in user_behavior_data])

# 非负矩阵分解
def nmf_loss(W, H, H0):
    return norm(user_behavior_matrix - W @ H0, ord=2) ** 2

# 优化非负矩阵分解
def nmf_optimize(W, H, H0, max_iter=100, tol=1e-6):
    for _ in range(max_iter):
        grad_W = -2 * H @ H0.T
        grad_H = -2 * W @ H0.T
        W_new, H_new = minimize(nmf_loss, (W, H), args=(H0,), method='BFGS', jac=True)
        if np.linalg.norm(grad_W) < tol and np.linalg.norm(grad_H) < tol:
            break
        W, H = W_new, H_new
    return W, H

# 初始化W和H
W = np.random.rand(len(user_behavior_data), 2)
H = np.random.rand(len(user_behavior_data), 2)

# 优化非负矩阵分解
W_optimized, H_optimized = nmf_optimize(W, H, H0=np.sqrt(np.maximum(0, np.diag(np.dot(user_behavior_matrix, user_behavior_matrix.T))))

# 重构矩阵
reconstructed_matrix = W_optimized @ H_optimized

# 测试
print(reconstructed_matrix)
```
