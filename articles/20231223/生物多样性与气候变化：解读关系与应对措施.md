## 4.1 代码实例

```python
import numpy as np
import matplotlib.pyplot as plt

# 生物多样性数据
x = np.array([1, 2, 3, 4, 5])
# 气候变化数据
y = np.array([2, 4, 6, 8, 10])

# 使用数学模型公式来描述生物多样性与气候变化之间的关系

beta = np.polyval(alpha, np.mean(x))

# 绘制生物多样性与气候变化之间的关系
plt.scatter(x, y)
plt.plot(x, alpha * x + beta, 'r')
plt.xlabel('生物多样性')
plt.ylabel('气候变化')
plt.show()
```
