```python
class ShoppingCart:
    def __init__(self):
        self.goods_list = []

    def add_goods(self, goods):
        self.goods_list.append(goods)

    def total_price(self):
        total_price = 0
        for goods in self.goods_list:
            total_price += goods.price + goods.shipping_fee + goods.insurance_fee
        return total_price
```

```python
class CheckoutStrategy:
    def __init__(self, shopping_cart, coupons):
        self.shopping_cart = shopping_cart
        self.coupons = coupons

    def total_price_after_coupon(self):
        total_price = self.shopping_cart.total_price()
        for coupon in self.coupons:
            if coupon.validity_period > datetime.now():
                total_price -= min(coupon.discount, total_price)
```
