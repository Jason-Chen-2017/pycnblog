```python

# 加载数据集

# 数据预处理
# 数据清洗

data['age'] = data['age'].astype(int)  # 数据类型转换
data['gender'] = data['gender'].map({'M': 1, 'F': 0})  # 数据格式转换

# 数据校验
# 数据检查
data = data[data['age'] >= 0]  # 数据范围检查
data = data[data['age'] <= 150]  # 数据范围检查

# 数据验证
data = data[data['gender'] == 0 | data['gender'] == 1]  # 数据规则验证

# 数据过滤
# 数据筛选
data = data[data['age'] > 30]  # 特定条件筛选

# 数据去除
data = data.dropna(subset=['phone'])  # 缺失值去除

# 数据排除
data = data[data['gender'] == 1]  # 异常值排除
```
