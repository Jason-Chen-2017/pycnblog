```python
# 聚类算法
kmeans = KMeans(n_clusters=5)
kmeans.fit(X_train)
y_cluster = kmeans.predict(X_test)

# 模型选择
lr = LogisticRegression()

# 参数优化
lr.fit(X_train, y_train)

# 模型评估
accuracy = lr.score(X_test, y_test)
print('Accuracy:', accuracy)
```
