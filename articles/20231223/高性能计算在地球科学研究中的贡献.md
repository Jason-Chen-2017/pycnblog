# 定义地球气候模型的核心算法
def earth_climate_model(T, dt):
    # 计算地球的热量变化
    dT = (-k * T + Q * dt) / m
    # 更新地球的温度
    T = T + dT * dt
    return T

    # 可视化和展示地球气候模型的预测结果

plt.xlabel('Time')
plt.ylabel('Temperature')
plt.legend()
plt.show()
