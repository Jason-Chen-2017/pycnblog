### 4.1数据纠错技术的具体代码实例

```python
import numpy as np

def hamming_encode(data, k):
    n = len(data)
    code = np.pad(data, (k-1, k-1), 'constant', constant_values=(0, 0))
    return code

def hamming_decode(code, k):
    n = len(code)
    data = code
    for i in range(k):
        error_bit = data[n-1-i] ^ code[n-1-i]
        data = np.where(data == 1, data ^ error_bit, data)
    return data

data = np.array([1, 0, 1, 1, 0])
k = 3
encoded_data = hamming_encode(data, k)
decoded_data = hamming_decode(encoded_data, k)
print(decoded_data)
```
