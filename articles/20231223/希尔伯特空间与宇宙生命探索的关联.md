# 快速傅里叶变换（FFT）
def fft_basis(n):
    x = np.linspace(0, 1, n, endpoint=False)
    x_pi = x * 2 * np.pi
    return np.exp(-1j * x_pi) / np.sqrt(n)

# 奇异值分解（SVD）
def svd_basis(A):
    U, s, V = np.linalg.svd(A, full_matrices=False)
    return U, s, V

# 格拉马什正交化（Gram-Schmidt）
def gram_schmidt(basis):
    result = []
    for v in basis:
        proj = np.dot(np.outer(result, v), np.array(result))
        v_orth = v - proj
        result.append(v_orth / np.linalg.norm(v_orth))
    return result

# 误差衰减法
def error_decay(A, basis, max_iter=1000):
    error = np.inf
    for i in range(max_iter):
        residual = A - np.dot(np.outer(basis, basis.T), A)
        basis = residual / np.linalg.norm(residual)
        error = np.linalg.norm(residual)
        print(f"Iteration {i+1}: Error = {error}")
    return basis
