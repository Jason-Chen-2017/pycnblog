import numpy as np
import matplotlib.pyplot as plt

def init_ice_sheet(x, y, A0, T0):
    A = A0 * np.ones((x, y))
    T = T0 * np.ones((x, y))
    return A, T

def compute_temperature(A, T, k, Ta, Tw):
    return Ta - T / k

def compute_melt_rate(Ta, Tw):
    return k * (Ta - Tw)

def update_ice_sheet(A, T, k, melt_rate, dt):
    A -= melt_rate * dt
    return A

def main():
    x = 100
    y = 100
    A0 = 1.0
    T0 = 0.0
    k = 1.0
    Ta = 273.15
    Tw = 273.15
    dt = 1.0
    steps = 1000

    A, T = init_ice_sheet(x, y, A0, T0)

    for _ in range(steps):
        Ta = compute_temperature(A, T, k, Ta, Tw)
        melt_rate = compute_melt_rate(Ta, Tw)
        A = update_ice_sheet(A, T, k, melt_rate, dt)

    plt.imshow(A, cmap='Greys')
    plt.colorbar()
    plt.show()

if __name__ == '__main__':
    main()
