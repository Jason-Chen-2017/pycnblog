```python
import pandas as pd

# 设备生成数据
data = np.random.rand(100, 4)

# 边缘设备对数据进行处理
def edge_computing(data):
    # 数据滤波
    filtered_data = np.mean(data, axis=1)
    # 数据聚合
    aggregated_data = np.sum(filtered_data)
    return aggregated_data

# 边缘设备将处理结果发送回中央服务器
result = edge_computing(data)
print(result)
```
