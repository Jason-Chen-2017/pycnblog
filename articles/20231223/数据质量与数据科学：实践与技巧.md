### 4.1.1 数据验证

```python
import pandas as pd

# 加载数据
data = pd.read_csv('data.csv')

# 验证数据准确性
def validate_data(data):
    # 检查数据是否缺失
    if data.isnull().sum().sum() > 0:
        print('数据缺失')
    # 检查数据是否符合预期格式
    if not data.dtypes.apply(lambda x: isinstance(x, (int, float))) == True:
        print('数据格式不符合预期')
    # 检查数据是否在预期范围内
    if (data['age'] < 0).sum() > 0 or (data['age'] > 150).sum() > 0:
        print('数据范围不符合预期')

validate_data(data)
```

### 4.1.2 数据清洗

```python
# 清洗数据
def clean_data(data):
    # 删除缺失的数据
    data = data.dropna()
    # 修改或补充缺失的数据
    data['age'] = data['age'].fillna(data['age'].mean())
    return data
```
