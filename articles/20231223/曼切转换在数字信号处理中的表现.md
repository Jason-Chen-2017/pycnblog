import numpy as np
import matplotlib.pyplot as plt

def manchester_convolutional_coding(data, taps, T):
    N = len(data)
    encoded_data = np.zeros(N * (T + 1))

    for i in range(N):
        start = i * T
        end = start + T
        encoded_data[start:end] = data[i]
        encoded_data[end] = 1 - data[i]

    return encoded_data

data = np.array([1, 0, 1, 0])
taps = [1, -1]
T = 2

encoded_data = manchester_convolutional_coding(data, taps, T)

plt.plot(encoded_data)
plt.show()
