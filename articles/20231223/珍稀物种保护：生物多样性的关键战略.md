# 4.具体代码实例和详细解释说明
## 4.1 数据预处理
```python
import pandas as pd
import numpy as np

# 读取生物资源数据
data = pd.read_csv('genetic_data.csv')

# 数据清洗
data = data.dropna()

# 数据整理
data = data[['species', 'population', 'habitat', 'threat']]

# 数据归一化
data['population'] = (data['population'] - data['population'].mean()) / data['population'].std()
data['habitat'] = (data['habitat'] - data['habitat'].mean()) / data['habitat'].std()
data['threat'] = (data['threat'] - data['threat'].mean()) / data['threat'].std()
```
## 4.2 特征提取
```python
from sklearn.feature_extraction.text import TfidfVectorizer

# 构建TF-IDF向量化器
vectorizer = TfidfVectorizer()

# 对生物资源数据进行特征提取
X = vectorizer.fit_transform(data['species'])
y = data['threat']
```
## 4.3 模型构建
```python
from sklearn.ensemble import RandomForestClassifier

# 构建随机森林分类器
clf = RandomForestClassifier()

# 训练模型
clf.fit(X, y)
```
## 4.4 模型评估
```python
from sklearn.metrics import accuracy_score, precision_score, recall_score

# 对模型进行评估
y_pred = clf.predict(X)

# 计算精度
accuracy = accuracy_score(y, y_pred)

# 计算精确度
precision = precision_score(y, y_pred)

# 计算召回率
recall = recall_score(y, y_pred)

print(f'精度：{accuracy}, 精确度：{precision}, 召回率：{recall}')
```
