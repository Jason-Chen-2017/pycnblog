import numpy as np

def entropy(prob):
    return -np.sum(prob * np.log2(prob))

def entropy_weight_allocation(resources, entropy_values):
    probabilities = np.array([entropy_values[i] / (entropy_values[0] + entropy_values[1]) for i in range(2)])
    allocation = np.zeros(len(resources))
    for i, resource in enumerate(resources):
        for j, probability in enumerate(probabilities):
            allocation[i] += probability * resource[j]
    return allocation

resources = [{'X': 10, 'Y': 20}, {'X': 30, 'Y': 10}]
entropy_values = [entropy(resource['X'] + resource['Y']) for resource in resources]
print(entropy_weight_allocation(resources, entropy_values))
