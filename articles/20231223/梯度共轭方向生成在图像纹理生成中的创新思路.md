```python
import matplotlib.pyplot as plt

# 定义样式空间
style_space = np.random.rand(100, 100)

# 在样式空间中进行梯度共轭方向搜索
def gradient_ascent(style_space, learning_rate=0.01, num_iterations=1000):
    x = np.random.rand(100, 100)
    for _ in range(num_iterations):
        gradient = np.random.rand(100, 100)
        x = x + learning_rate * gradient
    return x

# 在样式空间中进行梯度共轭方向生成
def generate_texture(style_space):
    x = gradient_ascent(style_space)
    return x

# 生成图像纹理
texture = generate_texture(style_space)
plt.imshow(texture, cmap='gray')
plt.show()
```
