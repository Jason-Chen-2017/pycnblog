$$
M{x(t)} = \int_{0}^{\infty} x(t) t^{-s} dt

```python
import numpy as np

def mellin_transform(x, s):
    return np.fft.fft(x)

def inverse_mellin_transform(M, s):
    return np.fft.ifft(M)

x_recovered = inverse_mellin_transform(M, s)
```
