```python
def calculate_coverage(test_cases, code_paths):
    covered_paths = set()
    for test_case in test_cases:
        for path in code_paths:
            if test_case in path:
                covered_paths.add(path)
    coverage = len(covered_paths) / len(code_paths)
    return coverage
```
