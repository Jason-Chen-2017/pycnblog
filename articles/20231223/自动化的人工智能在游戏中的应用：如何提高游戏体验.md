```python
class BayesianNetworkNode:
    def __init__(self, probability_distribution):
        self.probability_distribution = probability_distribution

    def get_probability(self, evidence):
        if evidence not in self.probability_distribution:
            return 0
        else:
            return self.probability_distribution[evidence]

def create_bayesian_network(nodes, edges, evidence):
    network = {}
    for node in nodes:
        network[node] = BayesianNetworkNode({})
    for edge in edges:
        network[edge[0]].probability_distribution[edge[1]] = 1
    for node in network:
        if node not in evidence:
            network[node].probability_distribution[None] = 1
    return network

def execute_bayesian_network(network, evidence):
    if evidence is None:
        return list(network.values())
    else:
        return [network[node].get_probability(evidence) for node in network]
```
