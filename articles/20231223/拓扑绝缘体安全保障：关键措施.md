```python
def shortest_path(graph, start, end):
    return nx.shortest_path(graph, start, end, method='dijkstra')

def fault_location(graph, fault_node):
    shortest_paths = []
    for node in graph.nodes():
        if node != fault_node:
            shortest_paths.append(shortest_path(graph, fault_node, node))
    return shortest_paths

# 构建电力系统的有向图
graph = nx.DiGraph()
graph.add_edge('A', 'B')
graph.add_edge('B', 'C')
graph.add_edge('C', 'D')
graph.add_edge('D', 'E')
graph.add_edge('E', 'F')

# 故障定位
fault_node = 'C'
shortest_paths = fault_location(graph, fault_node)
print(shortest_paths)
```
