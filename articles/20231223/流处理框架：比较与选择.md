# 定义数据源

               .from_elements([1, 2, 3, 4, 5])
               .assign_timestamps_and_watermarks()
               .key_by(1))

# 定义数据处理器

                  .from_collection([1, 2, 3, 4, 5])

                  .map(lambda x: x * 2)
                  .to_append_stream(data_sink, "value")
                  .key_by(1))
