# 4.具体代码实例和详细解释说明

```python
import numpy as np

def stock_price_simulation(S, K, T, r, sigma, N):
    dt = T / N
    W = np.random.standard_normal(N)
    Z = np.cumsum(np.sqrt(dt) * W)
    S_final = S * np.exp((r - 0.5 * sigma ** 2) * T + sigma * Z)
    return S_final

def call_option_price(S, K, T, r, sigma, N):
    call_price = 0
    for i in range(N):
        S_final = stock_price_simulation(S, K, T, r, sigma, N)
        call_price += max(S_final - K, 0)
    call_price /= N
    return call_price

S = 100
K = 100
T = 1
r = 0.05
sigma = 0.2
N = 100000

call_price = call_option_price(S, K, T, r, sigma, N)
print("Monte Carlo method call option price: ", call_price)
```
