```python
import numpy as np
import matplotlib.pyplot as plt

# 模拟交通流量的生成
def generate_traffic_flow(num_cars, flow_rate, time_step):
    return np.random.poisson(flow_rate * time_step, num_cars)

# 模拟交通信号灯的控制
def traffic_light_control(traffic_flow, time_step, green_time, red_time):
    cars_in_green = 0
    cars_in_red = 0
    state = 'green'

    while cars_in_green + cars_in_red < traffic_flow:
        if state == 'green':
            cars_in_green += generate_traffic_flow(green_time, flow_rate, time_step)
            if cars_in_green >= traffic_flow:
                state = 'red'
        else:
            cars_in_red += generate_traffic_flow(red_time, flow_rate, time_step)
            if cars_in_red >= traffic_flow:
                state = 'green'

    return state

# 模拟交通信号灯的运行
def simulate_traffic_light(flow_rate, green_time, red_time, time_step, num_steps):
    traffic_flow = generate_traffic_flow(num_steps, flow_rate, time_step)
    state = traffic_light_control(traffic_flow, time_step, green_time, red_time)
    return state

# 测试代码
flow_rate = 100
green_time = 5
red_time = 5
time_step = 1
num_steps = 100

state = simulate_traffic_light(flow_rate, green_time, red_time, time_step, num_steps)
print(f'交通信号灯最终状态为：{state}')
```
