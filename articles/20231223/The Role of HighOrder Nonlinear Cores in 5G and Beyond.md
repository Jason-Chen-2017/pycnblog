```python
import numpy as np
import matplotlib.pyplot as plt

def high_order_nonlinear_core(x, h):
    y = np.zeros_like(x)
    for i in range(len(x)):
        y[i] = np.sum(h * x[i - :])
    return y

x = np.random.rand(1000)
h = np.random.rand(1000)
y = high_order_nonlinear_core(x, h)

plt.plot(x, label='Input Signal')
plt.plot(y, label='Output Signal')
plt.legend()
plt.show()
```
