def sliding_window(X, window_size):
    for i in range(len(X) - window_size + 1):
        yield X[i:i+window_size]

def average(X):
    return sum(X) / len(X)

def sliding_window_average(X, window_size):
    for window in sliding_window(X, window_size):
        print(average(window))
