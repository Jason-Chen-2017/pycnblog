## 1.背景介绍

### 1.1 开放数据的概念与特点

- 推动社会创新：开放数据可以被各种应用开发者所使用，为创新提供数据支持，推动社会创新。

然后，我们创建一个`app.py`文件，实现数据共享平台的后端功能：

```python
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///data.db'
db = SQLAlchemy(app)

class Data(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(100), nullable=False)
    data = db.Column(db.Text, nullable=False)

@app.route('/data', methods=['POST'])
def upload_data():
    data = request.json
    new_data = Data(name=data['name'], data=data['data'])
    db.session.add(new_data)
    db.session.commit()
    return jsonify({'message': '数据上传成功'}), 201

@app.route('/data/<int:id>', methods=['GET'])
def get_data(id):
    data = Data.query.get_or_404(id)
    return jsonify({'name': data.name, 'data': data.data}), 200

@app.route('/data/<int:id>', methods=['PUT'])
def update_data(id):
    data = Data.query.get_or_404(id)
    data.name = request.json['name']
    data.data = request.json['data']
    db.session.commit()
    return jsonify({'message': '数据更新成功'}), 200

@app.route('/data/<int:id>', methods=['DELETE'])
def delete_data(id):
    data = Data.query.get_or_404(id)
    db.session.delete(data)
    db.session.commit()
    return jsonify({'message': '数据删除成功'}), 200

if __name__ == '__main__':
    app.run(debug=True)
```

document.getElementById('upload-form').addEventListener('submit', async (event) => {
    event.preventDefault();
    const name = document.getElementById('name').value;
    const data = document.getElementById('data').value;
    const response = await fetch('/data', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({ name, data })
    });
    if (response.ok) {
        alert('数据上传成功');
        location.reload();
    } else {
        alert('数据上传失败');
    }
});

fetch('/data').then(response => response.json()).then(data => {
    const dataList = document.getElementById('data-list');
    data.forEach(item => {
        const li = document.createElement('li');
        li.innerHTML = `<strong>${item.name}</strong>: ${item.data}`;
        dataList.appendChild(li);
    });
});
