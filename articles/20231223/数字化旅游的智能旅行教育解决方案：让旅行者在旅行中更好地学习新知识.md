```python
import numpy as np

def cosine_similarity(a, b):
    dot_product = np.dot(a, b)
    norm_a = np.linalg.norm(a)
    norm_b = np.linalg.norm(b)
    return dot_product / (norm_a * norm_b)

def recommend(user_id, user_ratings, item_ratings, num_recommendations):
    user_ratings_matrix = np.array(user_ratings)
    item_ratings_matrix = np.array(item_ratings)
    user_ratings_matrix = user_ratings_matrix - np.mean(user_ratings_matrix, axis=0)
    item_ratings_matrix = item_ratings_matrix - np.mean(item_ratings_matrix, axis=0)
    similarity_matrix = np.dot(user_ratings_matrix, item_ratings_matrix.T) / (np.sqrt(np.dot(user_ratings_matrix.T, user_ratings_matrix)) * np.sqrt(np.dot(item_ratings_matrix.T, item_ratings_matrix)))
    recommendations = np.argsort(-similarity_matrix[user_id])[:num_recommendations]
    return recommendations
```

在这个代码实例中，我们首先定义了一个计算余弦相似度的函数 `cosine_similarity`，然后定义了一个根据用户ID、用户评分矩阵、项目评分矩阵和推荐数量推荐项目的函数 `recommend`。在 `recommend` 函数中，我们首先将用户评分矩阵和项目评分矩阵进行标准化，然后计算相似度矩阵，最后返回推荐项目。
