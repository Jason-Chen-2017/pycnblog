def triangle_mesh(vertices, faces):
    for face in faces:
        for i in range(len(face) - 1):
            vertex1, vertex2 = vertices[face[i]], vertices[face[i+1]]
            edge = vertex2 - vertex1
            edge /= np.linalg.norm(edge)
            edge *= 0.5
            vertices.append(vertex1 + edge)

# 光线追踪算法
def ray_casting(camera, scene):
    rays = camera.generate_rays()
    colors = np.zeros(rays.shape[:2], dtype=np.float32)
    for ray in rays:
        hit, obj = scene.intersect(ray)
        if hit:
            color = obj.material.color
            attenuation = obj.material.attenuation
            colors += attenuation * color
        else:
            color = camera.background_color
            attenuation = 1.0
            colors += attenuation * color
    return colors
