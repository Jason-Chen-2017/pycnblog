```python
import gps3

def get_gps_data():
    session = gps3.Session()
    session.start()
    while True:
        fix = session.read()
        if fix:
            print(f"latitude: {fix.latitude}, longitude: {fix.longitude}")
        time.sleep(1)
    session.stop()

get_gps_data()
```
