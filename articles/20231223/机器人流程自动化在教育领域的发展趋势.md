```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 读取教学资源数据
teaching_resources = pd.read_csv('teaching_resources.csv')

# 文本特征提取
tfidf_vectorizer = TfidfVectorizer(stop_words='english')
teaching_resources['tfidf_features'] = tfidf_vectorizer.fit_transform(teaching_resources['description'])

# 智能推荐
def recommend_resources(resource, top_n=5):
    similarity = cosine_similarity(teaching_resources['tfidf_features'], resource['tfidf_features'].reshape(1, -1))
    recommended_resources = teaching_resources.iloc[similarity.argsort()[:top_n]]
    return recommended_resources

# 测试智能推荐
resource = teaching_resources.iloc[0]
recommended_resources = recommend_resources(resource)
print(recommended_resources)
```
