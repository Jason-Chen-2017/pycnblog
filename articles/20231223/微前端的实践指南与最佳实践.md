## 4.具体代码实例和详细解释说明

### 4.1代码实例

```javascript
// 定义一个微前端应用程序
class MicroFrontend {
  constructor(name, performance, extendability, stability) {
    this.name = name;
    this.performance = performance;
    this.extendability = extendability;
    this.stability = stability;
  }

  // 计算性能
  calculatePerformance() {
    return this.performance;
  }

  // 计算可扩展性
  calculateExtendability() {
    return this.extendability;
  }

  // 计算稳定性
  calculateStability() {
    return this.stability;
  }
}

// 创建一个微前端应用程序实例
const microFrontend1 = new MicroFrontend('microFrontend1', 0.9, 0.8, 0.7);
const microFrontend2 = new MicroFrontend('microFrontend2', 0.8, 0.9, 0.6);
const microFrontend3 = new MicroFrontend('microFrontend3', 0.7, 0.7, 0.8);

// 计算性能
const performance1 = microFrontend1.calculatePerformance();
const performance2 = microFrontend2.calculatePerformance();
const performance3 = microFrontend3.calculatePerformance();

// 计算可扩展性
const extendability1 = microFrontend1.calculateExtendability();
const extendability2 = microFrontend2.calculateExtendability();
const extendability3 = microFrontend3.calculateExtendability();

// 计算稳定性
const stability1 = microFrontend1.calculateStability();
const stability2 = microFrontend2.calculateStability();
const stability3 = microFrontend3.calculateStability();

// 输出结果
console.log(`性能: ${performance1}, ${performance2}, ${performance3}`);
console.log(`可扩展性: ${extendability1}, ${extendability2}, ${extendability3}`);
console.log(`稳定性: ${stability1}, ${stability2}, ${stability3}`);
```

答案：微前端架构是将一个大型应用程序拆分成多个独立的前端应用程序，这些应用程序可以独立开发、部署和运行。而单页面应用程序（SPA）是将一个大型应用程序的所有功能和页面放在一个页面中，这些功能和页面通过JavaScript动态加载和卸载。
