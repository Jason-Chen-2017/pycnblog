## 4.1 数据收集与预处理
```python
import pandas as pd

# 加载客户行为数据
customer_data = pd.read_csv('customer_data.csv')

# 数据清洗
customer_data = customer_data.dropna()

# 数据转换
customer_data['age'] = customer_data['age'].astype(int)

# 数据整合
customer_data = pd.concat([customer_data, pd.read_csv('market_data.csv')])

# 数据归一化
customer_data = (customer_data - customer_data.min()) / (customer_data.max() - customer_data.min())
```

## 4.3 营销活动优化
```python
# 目标设定
target = 1000

# 策略制定
strategy = 'increase_advertising'

# 渠道选择
if strategy == 'increase_advertising':
    channels = ['facebook', 'instagram', 'google']

# 活动执行
for channel in channels:
    customer_data = customer_data.merge(pd.DataFrame({channel: 1}), how='left')

# 效果评估
customer_data['sales'] = customer_data['sales'] + customer_data[channel] * 100
customer_data['sales'] = customer_data['sales'].apply(lambda x: max(0, x))
result = customer_data['sales'].sum()
print(f'目标达成率: {result / target * 100}%')
```
