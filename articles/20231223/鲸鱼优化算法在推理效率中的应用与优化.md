```python
def init_whale_swarm(d, N):
    X = np.random.rand(N, d)
    return X

def update_whale_position(X, A, B, C, U, r, X_best):
    V = A * np.exp(-B * np.power(np.abs(r), 2)) * U * (C * X_best - X) + B * np.abs(r) * (X_best - X)
    X_new = X + V
    return X_new
```

```python
def rosenbrock(X):
    return np.sum(100 * (X[:, 1] - np.power(X[:, 0], 2))**2 + (1 - X[:, 0])**2)
```

### 3.4 定义鲸鱼优化算法的主函数

接下来，我们需要定义鲸鱼优化算法的主函数，例如初始化鲸鱼群、更新鲸鱼群中每个鲸鱼的位置和速度等。

```python
def whale_optimization(d, N, A, B, C, max_iter):
    X = init_whale_swarm(d, N)
    X_best = X[np.argmin(rosenbrock(X))]
    for t in range(max_iter):
        r = np.random.rand(N, 1) - 0.5
        for i in range(N):
            X[i] = update_whale_position(X[i], A, B, C, r, X[i], X_best)
            if rosenbrock(X[i]) < rosenbrock(X_best):
                X_best = X[i]
    return X_best
```
