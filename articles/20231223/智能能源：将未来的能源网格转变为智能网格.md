import numpy as np

# 能源资源状态估计
def estimate_energy_resource_state(y, w):
    N = len(y)
    x_hat = np.zeros(N)
    for i in range(N):
        x_hat[i] = np.sum(w[i] * y[i])
    return x_hat

# 能源资源调度
def schedule_energy_resource(y, x, w):
    N = len(y)
    u = np.zeros(N)
    for i in range(N):
        u[i] = np.argmin(np.sum(np.power(np.abs(y[i] - x[i]), 2) * w[i]))
    return u

# 能源资源自动控制
def control_energy_resource(y, x, Kp, Ki, Kd):
    N = len(y)
    u = np.zeros(N)
    for i in range(N):
        e = y[i] - x[i]
        u[i] = Kp * e + Ki * np.sum(e) + Kd * np.diff(e)
    return u
