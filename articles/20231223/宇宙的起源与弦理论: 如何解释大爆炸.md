import numpy as np
import matplotlib.pyplot as plt

def string_vibration_pattern(frequency, amplitude):
    time = np.linspace(0, 2 * np.pi, 1000)
    vibration = amplitude * np.sin(frequency * time)
    plt.plot(time, vibration)
    plt.xlabel('Time')
    plt.ylabel('Amplitude')
    plt.title('String Vibration Pattern')
    plt.show()

frequency = 1

# 4.2 量子场论的计算

```python
import numpy as np
import matplotlib.pyplot as plt

def quantum_field(x, mass, coupling_constant):
    energy = np.sqrt(mass**2 + (coupling_constant * x)**2)
    field = energy / np.sqrt(2 * np.pi) * np.exp(-energy**2 / 2)
    plt.plot(x, field)
    plt.xlabel('Position')
    plt.ylabel('Field')
    plt.title('Quantum Field')
    plt.show()

mass = 1
coupling_constant = 1
quantum_field(x, mass, coupling_constant)
```
