    return a * b

    if b == 0:
        raise ValueError("Cannot divide by zero")
    return a / b

    a = float(input("Enter the first number: "))
    b = float(input("Enter the second number: "))
    try:
        print("Sum:", add(a, b))
        print("Difference:", subtract(a, b))

    except ValueError as e:
        print(e)
