# 4.具体代码实例和详细解释说明

## 4.1 线性回归模型

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成数据
np.random.seed(0)
x = np.random.rand(100, 1)
y = 3 * x + 2 + np.random.rand(100, 1)

# 训练模型
def linear_regression(x, y, alpha=0.01, iterations=1000):
    m, n = x.shape
    theta = np.zeros((n, 1))
    for _ in range(iterations):
        predictions = x.dot(theta)
        errors = predictions - y
        gradient = (x.T).dot(errors) / m
        theta -= alpha * gradient
    return theta

theta = linear_regression(x, y)

# 预测
x_test = np.linspace(0, 1, 100)
y_test = 3 * x_test + 2
plt.scatter(x, y, marker='o')
plt.plot(x_test, y_test, 'r', linewidth=2)
plt.show()
```

## 4.2 逻辑回归模型

```python
import numpy as np
from sklearn.linear_model import LogisticRegression

# 生成数据
np.random.seed(0)
x = np.random.rand(100, 1)
y = (np.random.rand(100, 1) > 0.5).astype(int)

# 训练模型
model = LogisticRegression()
model.fit(x.reshape(-1, 1), y)

# 预测
x_test = np.linspace(0, 1, 100)
y_test = (np.random.rand(100, 1) > 0.5).astype(int)
plt.scatter(x, y, marker='o')
plt.plot(x_test, model.predict(x_test.reshape(-1, 1)).ravel(), 'r', linewidth=2)
plt.show()
```
