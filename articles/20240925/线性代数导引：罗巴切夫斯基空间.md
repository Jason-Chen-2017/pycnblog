import numpy as np
from scipy.spatial import distance

def hyperbolic_angle(d, r):
    """
    计算罗巴切夫斯基空间中两条直线的夹角。
    
    参数：
    d -- 直线之间的距离
    r -- 双曲半径
    
    返回：

    """
    return np.arccos(1 - 2 * (distance.euclidean((0, 0), (d, d)) / r)**2)

def hyperbolic_distance(p1, p2, r):
    """
    计算罗巴切夫斯基空间中两点之间的距离。
    
    参数：
    p1, p2 -- 两点的坐标
    r -- 双曲半径
    
    返回：

    """
