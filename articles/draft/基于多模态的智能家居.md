import requests
from bs4 import BeautifulSoup

class门锁：

        self. lock_value = ""
        self. lock_time = ""

        url = f"https://example.com/lock_key?name={self.name}&password={self.password}"

        data = response.json()
        self.lock_key = data['key']
        self.lock_value = data['value']
        self.lock_time = data['time']

    def unlock_click(self):
        url = f"https://example.com/UnlockKey?name={self.name}&password={self.password}"
        response = requests.post(url)
        data = response.json()
        self.name = data['name']
        self.password = data['password']
        self.lock_time = data['time']
