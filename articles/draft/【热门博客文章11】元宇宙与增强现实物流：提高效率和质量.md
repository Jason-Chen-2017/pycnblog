using System.Collections;

public class InventorySystem : MonoBehaviour
{
    public List<Product> products = new List<Product>();

    private List<Transform> entities = new List<Transform>();

    private void Start()
    {
        // 加载产品资源
        ProductProductResource productProductResource = new ProductProductResource(products);
        StartCoroutine(LoadProducts());
    }

    IEnumerator LoadProducts()
    {
        // 加载产品资源
        yield return productProductResource.Load();

        // 创建实体
        Transform[] entities = Create entities;

        // 更新实体
        foreach (Transform entity in entities)
        {
            // 更新实体状态
            entity.transform.position = Vector3.Lerp(entity.transform.position, Vector3.forward, 0.01f);
            entity.transform.name = "Product 1";
            entity.transform.scale.x = 1.0f;
            entity.transform.scale.y = 1.0f;
            entity.transform.scale.z = 1.0f;
        }

        // 加载其他产品资源
        StartCoroutine(LoadProducts());
    }

    IEnumerator Create entities()
    {
        // 创建产品实体
        foreach (var product in products)
        {
            // 创建产品实体
            Transform entity = Instantiate(product);

            // 添加实体到实体列表
            entities.Add(entity);

            // 添加实体到场景
            foreach (var transform in entities)
            {
                transform.localScale = new Vector3(1, 1, 1);
            }
        }
    }
}
