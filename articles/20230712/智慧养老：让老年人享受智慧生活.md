### 4.4. 代码讲解说明

这里以一个简单的物联网模块为例，介绍如何实现老年人智能床的智能控制：

```python
import socket
import time

class SmartBed:
    def __init__(self, socket_address):
        self.socket_address = socket_address
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.socket.bind((socket_address, 80))
        self.socket.listen(1)

    def send_data(self, data):
        self.socket.sendall(data)

    def receive_data(self):
        data, client_address = self.socket.accept()
        print(f"Received data from {client_address}")
        return data

    def main(self):
        print("Smart Bed is ready to receive data.")

        while True:
            data = self.receive_data()
            print(f"Received data: {data}")
            print("Smart Bed:", data)
            control_command = input("Control command (send, receive, query, exit): ")
            if control_command.lower() == "send":
                self.send_data(data)
            elif control_command.lower() == "receive":
                self.receive_data()
            elif control_command.lower() == "query":
                print("Control query:", data)
            elif control_command.lower() == "exit":
                print("Exiting...")
                break
            else:
                print("Invalid control command")

if __name__ == "__main__":
    smart_bed = SmartBed("127.0.0.1:5545")
    print("Connecting to Smart Bed...")
    print("Press Enter to send a control command")
    control_command = input("Control command (send, receive, query, exit): ")
    smart_bed.main()
```
