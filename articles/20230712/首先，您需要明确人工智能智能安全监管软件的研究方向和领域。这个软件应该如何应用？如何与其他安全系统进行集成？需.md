        for line in f:
            data.append(line.strip())
    return data

    data = []
    for line in data:
        if re.search(r'^\d{2,}$', line):
            data.append(line.strip())
        else:
            data.append(line)
    return data

# 数据收集

    收集的数据包括网络和设备信息，包括 IP 地址、MAC 地址、操作系统、文件名等。
    return data

# 模型训练
def train_model(data, model):
    model.fit(data, epochs=10)
    return model

# 模型测试
def test_model(data, model):
    predictions = model.predict(data)
    return predictions

# 核心模块实现

    # 读取数据
    data = read_data('security_data.csv')
    data = clean_data(data)
