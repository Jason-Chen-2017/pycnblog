```python
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.ensemble import decision_tree
from sklearn.metrics import accuracy_score

# Step 1: Data Collection
data = pd.read_csv('data.csv')

# Step 2: Data Cleaning
# Remove duplicate data
data.drop_duplicates(inplace=True)

# Step 3: Data Analysis
# Extract useful information
data['score'] = data['score'].apply(lambda x: 0 if x < 50 else x)
data['teacher_score'] = data['teacher_score'].apply(lambda x: 0 if x < 50 else x)

# Step 4: Data推荐
# Split data into training and test sets
X_train, X_test, y_train, y_test = train_test_split(data.drop('recommended', axis=1), data['recommended'], test_size=0.2)

# Tree model
clf = decision_tree(criterion='accuracy',
                  random_state=0)
clf.fit(X_train, y_train)

# Predict score
recommended = clf.predict(X_test)
```
