```python
import time
from collections import defaultdict

# 定义数据源
def data_source():
    while True:
        yield {'user_id': 'user1', 'timestamp': time.time()}
        time.sleep(1)

# 定义数据流处理操作
def process_data(data):
    user_id = data['user_id']
    timestamp = data['timestamp']
    user_data = defaultdict(int)
    user_data[user_id] += 1
    return user_data

# 定义数据接收器
def data_receiver(data):
    for user_id, count in data.items():
        print(f'用户 {user_id} 的访问次数为 {count}')

# 定义滑动窗口操作
def sliding_window(window_size, data):
    window = []
    while True:
        for item in data:
            window.append(item)
            if len(window) > window_size:
                window.pop(0)
        yield window

# 实现流式数据处理算法
def stream_processing(data_source, process_data, data_receiver, sliding_window, window_size):
    window = sliding_window(window_size, data_source)
    for data in window:
        processed_data = process_data(data)
        data_receiver(processed_data)

# 启动流式数据处理
stream_processing(data_source, process_data, data_receiver, sliding_window, window_size)
```
