```python
import numpy as np

def multiple_integral(f, a, b, c, d):
    dx = np.linspace(a, b, 100)
    dy = np.linspace(c, d, 100)
    x, y = np.meshgrid(dx, dy)
    z = f(x, y)
    integral = np.trapz(np.trapz(z, x), y)
    return integral

def f(x, y):
    return x * y

a = 0
b = 1
c = 0
d = 1

integral = multiple_integral(f, a, b, c, d)
print(integral)
```
