## 4.1 简单的元启发式算法实现

```python
import random

def init_solution():
    return random.randint(1, 100)

def evaluate_solution(solution):
    return -solution**2

def search_neighbor(solution):
    return random.randint(solution - 10, solution + 10)

def update_solution(solution, neighbor):
    if evaluate_solution(neighbor) > evaluate_solution(solution):
        return neighbor
    else:
        return solution

def terminate_condition():
    return False

solution = init_solution()
while not terminate_condition():
    neighbor = search_neighbor(solution)
    solution = update_solution(solution, neighbor)

print(solution)
```
