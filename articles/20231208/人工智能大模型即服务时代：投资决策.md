```python
import tensorflow as tf
from tensorflow.keras.layers import Input, Dense, BatchNormalization
from tensorflow.keras.models import Model

# 创建生成对抗网络
def generate_adversarial_network():
    # 生成器
    generator_input = Input(shape=(100,))
    x = Dense(128, activation='relu')(generator_input)
    x = BatchNormalization()(x)
    x = Dense(128, activation='relu')(x)
    x = BatchNormalization()(x)
    x = Dense(100, activation='sigmoid')(x)
    generator = Model(generator_input, x)

    # 判别器
    discriminator_input = Input(shape=(100,))
    x = Dense(128, activation='relu')(discriminator_input)
    x = BatchNormalization()(x)
    x = Dense(128, activation='relu')(x)
    x = BatchNormalization()(x)
    x = Dense(1, activation='sigmoid')(x)
    discriminator = Model(discriminator_input, x)

    # 生成对抗网络
    z = Input(shape=(100,))
    image = generator(z)
    validity = discriminator(image)

    # 生成对抗网络模型
    model = Model(z, validity)
    return model
```
