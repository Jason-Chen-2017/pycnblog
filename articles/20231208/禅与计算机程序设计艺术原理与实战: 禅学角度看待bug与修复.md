    for i in range(len(program)):
        for j in range(i, len(program)):
            if program[i:j+1] == "bug":
                dp[i][j] = 1
            else:
                dp[i][j] = 0
    return dp

def fix_bug(program, dp):
    # 使用回溯算法来解决bug问题
    def backtrack(i, j):
        if i > j:
            return
        if dp[i][j]:
            print("找到bug：", program[i:j+1])
            return
        for k in range(i, j):
            backtrack(i, k)
            backtrack(k+1, j)
    backtrack(0, len(program)-1)
