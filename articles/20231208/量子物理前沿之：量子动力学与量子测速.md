### 4.1 量子动力学的具体代码实例

```python
import numpy as np
from scipy.linalg import eigh

# 定义波函数
def wave_function(x):
    return np.exp(-x**2)

# 求解波函数方程
def solve_schrodinger_equation(x, t):
    H = np.diag(np.arange(x.size) * 2)
    psi = np.array([wave_function(x[i]) for i in range(x.size)])
    e, v = eigh(H)
    return np.dot(v, np.exp(-1j * e * t) * np.dot(np.linalg.inv(v), psi))

# 分析波函数的解
def analyze_wave_function(x, t):
    psi = solve_schrodinger_equation(x, t)
    return np.abs(psi)**2
```
