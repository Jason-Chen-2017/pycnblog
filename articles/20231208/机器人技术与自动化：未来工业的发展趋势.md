### 4.1 线性回归

线性回归是一种简单的监督学习方法，通过训练数据集来训练模型，使模型能够对新的输入数据进行预测。线性回归的数学模型公式为：

$$
y = \beta_0 + \beta_1x_1 + \beta_2x_2 + ... + \beta_nx_n
$$

具体代码实例：

```python
import numpy as np

# 定义训练数据集
X = np.array([[1, 1], [1, 2], [2, 2], [2, 3]])
y = np.array([1, 2, 3, 4])

# 定义模型参数
beta_0 = 0
beta_1 = 0
beta_2 = 0

# 定义损失函数
def loss(y_pred, y):
    return np.sum((y_pred - y)**2)

# 定义梯度
def grad(y_pred, y):
    return 2 * (y_pred - y)

# 定义优化算法
def optimize(beta_0, beta_1, beta_2, X, y, learning_rate):
    for _ in range(1000):
        y_pred = beta_0 + beta_1 * X[:, 0] + beta_2 * X[:, 1]
        grad_beta_0 = grad(y_pred, y) * X[:, 0]
        grad_beta_1 = grad(y_pred, y) * X[:, 1]
        grad_beta_2 = grad(y_pred, y) * X[:, 2]
        beta_0 -= learning_rate * grad_beta_0
        beta_1 -= learning_rate * grad_beta_1
        beta_2 -= learning_rate * grad_beta_2
    return beta_0, beta_1, beta_2

# 训练模型
beta_0, beta_1, beta_2 = optimize(beta_0, beta_1, beta_2, X, y, learning_rate=0.01)

# 预测
y_pred = beta_0 + beta_1 * X[:, 0] + beta_2 * X[:, 1]
print(y_pred)
```

    return np.sum(y * np.log(y_pred) + (1 - y) * np.log(1 - y_pred))

def grad(y_pred, y):
    return y_pred - y

# 定义优化算法
def optimize(beta_0, beta_1, beta_2, X, y, learning_rate):

        beta_1 -= learning_rate * grad_beta_1

beta_0, beta_1, beta_2 = optimize(beta_0, beta_1, beta_2, X, y, learning_rate=0.01)
