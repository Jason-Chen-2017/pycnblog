```python
import pandas as pd

# 读取数据
data = pd.read_csv('data.csv')

# 检查数据质量
def check_data_quality(data):
    # 检查数据准确性
    accuracy = data.apply(lambda x: x.isnull().sum() / len(x))
    # 检查数据完整性
    completeness = data.isnull().sum().sum() / len(data)
    # 检查数据一致性
    consistency = data.duplicated().sum() / len(data)
    # 检查数据时效性
    timeliness = data['timestamp'].max() - data['timestamp'].min()
    # 检查数据可用性
    availability = data.shape[0] / (data.shape[0] + data.isnull().sum().sum())
    return accuracy, completeness, consistency, timeliness, availability

# 获取数据质量指标
accuracy, completeness, consistency, timeliness, availability = check_data_quality(data)
```
