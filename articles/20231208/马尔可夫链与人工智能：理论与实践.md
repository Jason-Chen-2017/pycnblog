## 4.1 实例背景

假设我们有一个简单的天气预报任务，需要根据当前天气来预测下一个天气。我们可以使用马尔可夫链来建模这个任务，其状态可以表示为“晴天”和“雨天”，转移概率可以表示为从当前天气转移到下一个天气的概率。

## 4.2 实例代码

```python
import numpy as np

# 初始化马尔可夫链的状态与转移概率
states = ['sunny', 'rainy']
transition_probabilities = np.array([[0.8, 0.2], [0.5, 0.5]])

# 训练马尔可夫链的转移概率
def train_transition_probabilities(data):
    # 提取状态集合和转移概率
    states = set(data)
    transition_probabilities = np.zeros((len(states), len(states)))
    for state1 in states:
        for state2 in states:
            count = 0
            for data in data:
                if data == state1:
                    count += 1
            transition_probabilities[states.index(state1), states.index(state2)] = count / len(data)
    return transition_probabilities

# 推理马尔可夫链的下一个状态
def predict_next_state(current_state, transition_probabilities):
    # 计算下一个状态的概率
    probabilities = transition_probabilities[states.index(current_state)]
    # 选择下一个状态
    next_state = np.random.choice(states, p=probabilities)
    return next_state

# 训练马尔可夫链的转移概率
data = ['sunny', 'rainy', 'sunny', 'rainy', 'sunny', 'rainy', 'sunny', 'rainy', 'sunny', 'rainy']
transition_probabilities = train_transition_probabilities(data)

# 推理马尔可夫链的下一个状态
current_state = 'sunny'
next_state = predict_next_state(current_state, transition_probabilities)
print(next_state)  # Output: 'rainy'
```
