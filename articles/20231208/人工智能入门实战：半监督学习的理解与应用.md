```python
import numpy as np
from sklearn.semi_supervised import LabelSpreading
from sklearn.datasets import make_classification

# 生成有标签的数据
X, y = make_classification(n_samples=1000, n_features=20, n_informative=10,
                           n_redundant=10, n_classes=10, n_clusters_per_class=1,
                           weights=[0.1, 0.9], flip_y=0, random_state=42)

# 生成无标签的数据
X_unlabeled = np.random.randn(1000, 20)

# 创建半监督学习模型
model = LabelSpreading(kernel='knn', alpha=0.1, n_jobs=-1)

# 训练模型
model.fit(X, y, X_unlabeled)

# 预测标签
y_pred = model.predict(X_unlabeled)
```
