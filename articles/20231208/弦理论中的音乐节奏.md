```python
import numpy as np
import matplotlib.pyplot as plt

# 创建音频波
def create_audio_wave(frequency, time_length, amplitude):
    time = np.linspace(0, time_length, time_length)
    wave = amplitude * np.sin(2 * np.pi * frequency * time)
    return wave

# 创建音乐节奏
def create_rhythm(frequency, time_length, amplitude, rhythm):
    wave = create_audio_wave(frequency, time_length, amplitude)
    rhythm_wave = wave * rhythm
    return rhythm_wave

# 创建音乐节奏
frequency = 440
time_length = 1
amplitude = 1
rhythm = 0.5
wave = create_rhythm(frequency, time_length, amplitude, rhythm)

# 绘制音频波
plt.plot(wave)
plt.xlabel('Time')
plt.ylabel('Amplitude')
plt.title('Music Rhythm')
plt.show()
```
