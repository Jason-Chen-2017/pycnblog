```python
import numpy as np
from qiskit import QuantumCircuit, Aer, transpile, assemble
from qiskit.visualization import plot_histogram

# 创建一个4个量子比特的量子电路
qc = QuantumCircuit(4)

# 创建一个4个元素的输入序列
input_sequence = [1, 0, 1, 1]

# 将输入序列转换为量子状态
for i in range(4):
    qc.x(i)

# 添加H门
qc.h(0)
qc.h(1)
qc.h(2)
qc.h(3)

# 添加CNOT门
qc.cx(0, 1)
qc.cx(1, 2)
qc.cx(2, 3)

# 执行量子电路
simulator = Aer.get_backend('qasm_simulator')
job = simulator.run(qc, input_sequence)

# 获取结果
result = job.result()
counts = result.get_counts()

# 绘制结果
plot_histogram(counts)
```
