## 4.1 缓存服务器的代码实例

以下是一个简单的缓存服务器的代码实例：

```python
import time

class CacheServer:
    def __init__(self):
        self.data = {}
        self.timeout = 3600

    def add(self, key, value):
        self.data[key] = value
        self.data[key]['timestamp'] = time.time()
        self.data[key]['timeout'] = self.timeout

    def get(self, key):
        if key in self.data:
            self.data[key]['timestamp'] = time.time()
            return self.data[key]['value']
        else:
            return None

    def update(self, key, value):
        if key in self.data:
            self.data[key]['value'] = value
            self.data[key]['timestamp'] = time.time()
        else:
            return None

    def delete(self, key):
        if key in self.data:
            del self.data[key]
        else:
            return None

    def sync(self, other):
        for key in self.data:
            if key in other.data:
                if self.data[key]['timestamp'] < other.data[key]['timestamp']:
                    self.data[key] = other.data[key]

if __name__ == '__main__':
    cache1 = CacheServer()
    cache2 = CacheServer()

    cache1.add('key1', 'value1')
    cache2.add('key1', 'value2')

    cache1.sync(cache2)

    print(cache1.get('key1'))  # 输出: value2
```
