```python
# 信号采集

# 信号分析
F = np.fft.fft(f)

# 信号处理
filtered_f = np.fft.ifft(F * np.array([0, 1, 0]))

# 信号处理结果的可视化

plt.plot(t, filtered_f, label='处理后的信号')
plt.legend()
plt.show()
```
