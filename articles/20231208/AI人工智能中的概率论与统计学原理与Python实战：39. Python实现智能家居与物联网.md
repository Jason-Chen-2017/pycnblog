```python
client = mqtt.Client()
client.connect('mqtt_server', 1883, 60)

# 订阅主题
client.subscribe('smart_home/topic')

# 发布消息
client.publish('smart_home/topic', 'Hello, MQTT!')

# 循环接收消息
while True:
    message = client.recv()
    print(message.topic, message.payload.decode('utf-8'))
```
