```python
import numpy as np
import matplotlib.pyplot as plt

# 模拟信号
t = np.arange(0.0, 1.0, 0.01)
s = 4 * np.sin(2 * np.pi * t)

# 数字信号处理
fs = 1000.0
T = 1.0 / fs
x = np.zeros(int(len(t) * fs))
x[int(T * len(t))::int(T * fs)] = s

# 处理后的数字信号
y = x * np.hamming(int(len(t) * fs))

# 将处理后的数字信号转换回模拟信号
f, t, S = np.interp(t, [0, T, 2 * T, 3 * T], y)

# 绘制模拟信号和数字信号
plt.figure(figsize=(10, 5))
plt.plot(t, s, label='模拟信号')
plt.plot(t, y, label='数字信号')
plt.legend()
plt.show()
```
