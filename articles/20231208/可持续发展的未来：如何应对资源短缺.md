```python
def dynamic_programming(resources, demands):
    # 初始化动态规划表
    dp = [[0] * (len(demands) + 1) for _ in range(len(resources) + 1)]

    # 填充动态规划表
    for i in range(len(resources) + 1):
        for j in range(len(demands) + 1):
            if i == 0 or j == 0:
                dp[i][j] = 0
            elif resources[i - 1] >= demands[j - 1]:
                dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - 1] + resources[i - 1])
            else:
                dp[i][j] = dp[i - 1][j]

    # 返回最优解
    return dp[len(resources)][len(demands)]
```
