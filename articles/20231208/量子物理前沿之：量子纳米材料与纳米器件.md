import numpy as np
from qiskit import QuantumCircuit, Aer, transpile, assemble

# 创建量子态
qc = QuantumCircuit(2)
qc.h(0)

# 操作量子态
qc.cx(0, 1)

# 执行量子计算

statevector = result.get_statevector(qc)
print(statevector)

from scipy.constants import e, m, h, c
from scipy.special import jn
from scipy.integrate import quad

# 计算多极模式的电磁场
def calculate_multipole_field(r, l, m, n, k):
    rho = r * np.sqrt(l * (l + 1))
    integral = quad(lambda x: x * jn(m, x) * jn(m, rho / x), 0, rho)[0]
    return integral / (n * rho * jn(m, k * r))

# 计算电磁场的总强度
def calculate_total_field(r, l, m, k):
    total_field = 0
    for n in range(1, 100):
        total_field += calculate_multipole_field(r, l, m, n, k)
    return total_field

# 计算电磁场的强度
