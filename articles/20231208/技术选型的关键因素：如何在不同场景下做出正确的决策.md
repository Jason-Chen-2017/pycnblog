import pandas as pd
from sklearn.tree import DecisionTreeClassifier
from sklearn.svm import SVC
from sklearn.neural_network import MLPClassifier

# 收集数据
data = pd.read_csv('data.csv')

# 预处理数据
data = data.dropna()
data = data.apply(lambda x: (x - x.mean()) / x.std())

# 训练模型
X = data.drop('technology', axis=1)
y = data['technology']

clf1 = DecisionTreeClassifier()
clf1.fit(X, y)

clf2 = SVC()
clf2.fit(X, y)

clf3 = MLPClassifier()
clf3.fit(X, y)

# 评估模型
accuracy1 = clf1.score(X, y)
accuracy2 = clf2.score(X, y)
accuracy3 = clf3.score(X, y)

print('Decision Tree Accuracy:', accuracy1)
print('Support Vector Machine Accuracy:', accuracy2)
print('Neural Network Accuracy:', accuracy3)
