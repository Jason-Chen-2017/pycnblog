```python

    def __init__(self):
        self.data = {}

    def add(self, key, value):
        self.data[key] = value

    def get(self, key):
        return self.data.get(key)

# 实现数据库

    def add(self, key, value):
        super().add(key, value)

    def get(self, key):
        return super().get(key)
```
