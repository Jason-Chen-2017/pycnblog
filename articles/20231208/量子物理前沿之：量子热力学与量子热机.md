## 4.1 量子热力学的基本概念

我们将通过以下代码实例来详细解释量子热力学的基本概念：

```python
import numpy as np
import matplotlib.pyplot as plt

# 定义波函数
def wave_function(x):
    return np.exp(-x**2)

# 计算粒子的态熵
def entropy(wave_function):
    return -np.sum(wave_function**2 * np.log(wave_function**2))

# 计算粒子的热容
def heat_capacity(wave_function, temperature):
    return -np.sum(wave_function**2 * np.log(wave_function**2)) / temperature

# 计算粒子的热运动
def heat_motion(wave_function, temperature):
    return np.sum(wave_function**2 * np.exp(-x**2 / temperature**2))

# 计算粒子的热力学性质
def thermodynamic_properties(wave_function, temperature):
    return entropy(wave_function), heat_capacity(wave_function, temperature), heat_motion(wave_function, temperature)

# 绘制粒子的热力学性质
def plot_thermodynamic_properties(entropy, heat_capacity, heat_motion):
    plt.plot(temperature, entropy, label='粒子的态熵')
    plt.plot(temperature, heat_capacity, label='粒子的热容')
    plt.plot(temperature, heat_motion, label='粒子的热运动')
    plt.legend()
    plt.show()

# 主程序
if __name__ == '__main__':
    temperature = np.linspace(0, 10, 100)
    wave_function = np.exp(-x**2)
    entropy, heat_capacity, heat_motion = thermodynamic_properties(wave_function, temperature)
    plot_thermodynamic_properties(entropy, heat_capacity, heat_motion)
```

在上述代码中，我们首先定义了波函数，然后计算了粒子的态熵、热容、热运动等热力学性质。最后，我们绘制了粒子的热力学性质的图像。
