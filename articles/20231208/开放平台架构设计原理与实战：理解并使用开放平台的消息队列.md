```python
import queue
import threading
import time

# 消息队列管理器
class MessageQueueManager:
    def __init__(self):
        self.queue = queue.Queue()

    def send_message(self, message):
        self.queue.put(message)

    def get_message(self):
        return self.queue.get()

# 生产者
class Producer(threading.Thread):
    def __init__(self, message_queue_manager):
        threading.Thread.__init__(self)
        self.message_queue_manager = message_queue_manager

    def run(self):
        for i in range(10):
            message = f"Message {i}"
            self.message_queue_manager.send_message(message)
            print(f"Producer sent message: {message}")
            time.sleep(1)

# 消费者
class Consumer(threading.Thread):
    def __init__(self, message_queue_manager):
        threading.Thread.__init__(self)
        self.message_queue_manager = message_queue_manager

    def run(self):
        while True:
            message = self.message_queue_manager.get_message()
            if message is None:
                break
            print(f"Consumer received message: {message}")
            time.sleep(1)

# 主程序
def main():
    message_queue_manager = MessageQueueManager()
    producer = Producer(message_queue_manager)
    consumer = Consumer(message_queue_manager)

    producer.start()
    consumer.start()

    producer.join()
    consumer.join()

if __name__ == "__main__":
    main()
```
