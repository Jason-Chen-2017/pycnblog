### 4.1.1 条件概率模型

条件概率模型的一个简单实现如下：

```python
import numpy as np

def condition_probability_model(corpus, word):
    word_count = 0
    total_count = 0
    for sentence in corpus:
        for w in sentence:
            if w == word:
                word_count += 1
            total_count += 1
    return word_count / total_count
```

### 4.1.2 隐马尔可夫模型

隐马尔可夫模型的一个简单实现如下：

```python
import numpy as np

def hidden_markov_model(corpus, word):
    transition_matrix = np.zeros((len(corpus), len(corpus)))
    emission_probability = np.zeros((len(corpus), len(corpus)))
    for sentence in corpus:
        for i in range(len(sentence) - 1):
            transition_matrix[sentence[i], sentence[i + 1]] += 1
        for i in range(len(sentence)):
            emission_probability[sentence[i], word] += 1
    return transition_matrix, emission_probability
```

import numpy as np
import keras

def rnn(corpus, word, embedding_dim, hidden_dim, output_dim):
    model = keras.Sequential()
    model.add(keras.layers.Embedding(output_dim, embedding_dim, input_length=len(corpus)))
    model.add(keras.layers.LSTM(hidden_dim))
    model.add(keras.layers.Dense(output_dim, activation='softmax'))
    model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
    model.fit(corpus, np.array([word]), epochs=10, batch_size=32)
    return model
```

import spacy

nlp = spacy.load('en_core_web_sm')

def text_summarization(text, summary_length):
    doc = nlp(text)
    summary = ""
    for i in range(summary_length):
        max_score = 0
        max_token = None
        for token in doc:
            score = 0
            if token.dep_ == 'punct':
                score += 1
            if token.head.dep_ == 'punct':
                score += 1
            if score > max_score:
                max_score = score
                max_token = token
        summary += max_token.text
        if max_token.dep_ == 'punct':
            summary += "."
    return summary
```
