```python
import random

# 定义随机变量
x = random.uniform(-1, 1)

# 定义概率分布
f = lambda x: 1 / 2 * (1 - x**2)

# 定义目标函数
g = lambda x: x**2

# 生成随机样本
n = 10000
samples = [g(x) for _ in range(n)]

# 计算目标函数的期望值

# 输出结果
print("The estimated mean of the target function is:", mean)
```
