```
    when
        $customer : Customer( $amount : amount )
        $amount < 100
    then
        modify($customer) { it.amount = $amount * 0.95; }
end

// 目标规则引擎：JBoss Drools
rule "Customer Discount"
    when
        $customer : Customer( $amount : amount )
        $amount < 100
    then
        modify($customer) { it.amount = $amount * 0.95; }
end
```
