```python
from sklearn.ensemble import IsolationForest
from sklearn.preprocessing import StandardScaler

# Isolation Forest异常检测
def isolation_forest_anomaly(data):
    scaler = StandardScaler()
    data = scaler.fit_transform(data)
    clf = IsolationForest(contamination=0.1)
    clf.fit(data)
    return clf

# Z-score异常检测
def z_score_anomaly(data):
    mu = np.mean(data)
    std = np.std(data)
    threshold = 3
    anomalies = [x for x in data if abs(x - mu) > threshold * std]
    return anomalies
```
