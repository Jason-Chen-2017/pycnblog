模糊集是一种用于表示不确定性信息的集合。模糊集的元素可以是具体的数值或者是范围。例如，一个血压值可以被表示为一个模糊集，其中元素可以是具体的血压值（如120mmHg）或者是一个范围（如110-130mmHg）。

        relation = fuzzy_set.similarity_scores_

        relation = fuzzy_set.cluster_centers_

    plt.ylabel('Heart Rate')

    rules = []
    for i in range(len(element) - 1):
        condition = (element[i] <= 'Blood Pressure', '<=', element[i + 1])
        result = ('High Blood Pressure', 'if', condition)
        rules.append(result)
    return rules

rules = define_rule(element, relation)

# 根据模糊规则，建立模糊控制规则
def build_control_rule(rules):
    control_rules = []
    for rule in rules:
        if rule[0] == 'High Blood Pressure':
            control_rules.append(rule)
    return control_rules

control_rules = build_control_rule(rules)

# 根据模糊控制，实现医学诊断的自动化

    diagnosis = []

        for rule in control_rules:

                condition = rule[2]
                if condition(row):
                    diagnosis.append(rule[0])
                    break
    return diagnosis

diagnosis = diagnose(data, control_rules)
print(diagnosis)
