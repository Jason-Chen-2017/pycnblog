```python
def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i+1, n):
            if arr[min_idx] > arr[j]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

arr = [5, 2, 8, 3, 1]
result = selection_sort(arr)
print(result)
```

2.搜索算法实例：
```python
def binary_search(arr, target):
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

arr = [1, 2, 3, 4, 5]
target = 3
result = binary_search(arr, target)
print(result)
```

3.图论算法实例：

```python
from collections import defaultdict

def dijkstra(graph, start):
    n = len(graph)
    dist = [float("inf")] * n
    dist[start] = 0
    visited = [False] * n

    for _ in range(n):
        min_dist = float("inf")
        min_node = -1
        for i in range(n):
            if not visited[i] and dist[i] < min_dist:
                min_dist = dist[i]
                min_node = i
        if min_node == -1:
            break
        visited[min_node] = True
        for i, d in graph[min_node].items():
            if not visited[i] and dist[min_node] + d < dist[i]:
                dist[i] = dist[min_node] + d
    return dist

graph = defaultdict(dict)
graph[0][1] = 4
graph[0][2] = 2
graph[1][3] = 1
graph[2][1] = 1
graph[2][3] = 5
graph[3][4] = 1

start = 0
result = dijkstra(graph, start)
print(result)
```
