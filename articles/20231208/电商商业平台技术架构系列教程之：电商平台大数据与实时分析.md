from pandas import DataFrame

# 数据实时处理
def process_data(data):
    df = DataFrame(data, columns=['age', 'gender', 'date'])
    df['date'] = pd.to_datetime(df['date'])
    df['day_of_week'] = df['date'].dt.day_of_week
    return df

# 数据实时分析
def analyze_data(df):
    mean_age = df['age'].mean()
    median_age = df['age'].median()
    std_age = df['age'].std()
    return mean_age, median_age, std_age

# 数据实时预测
def predict_data(df):
    model = RandomForestClassifier()
    model.fit(df.drop('label', axis=1), df['label'])
    y_pred = model.predict(df.drop('label', axis=1))
    return y_pred

# 数据可视化和报表
def visualize_data(df):
    plt.boxplot(df['age'])
    plt.show()

# 数据处理流程
def data_processing(data):
    df = DataFrame(data, columns=['age', 'gender', 'date'])
    df['date'] = pd.to_datetime(df['date'])
    df['day_of_week'] = df['date'].dt.day_of_week
    return df

# 数据处理流程
def main():
    for message in consumer:
        data = message.value
        df = data_processing(data)
        mean_age, median_age, std_age = analyze_data(df)
