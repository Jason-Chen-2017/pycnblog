rules = [
    {
        "condition": "age >= 18",

    },
    {
        "condition": "age < 18",
        "action": "deny_access"
    }
]

# 定义工作流程

    "condition1",

    "condition2",
    "action2"
]

# 定义事件监听器

    "event1": "condition1",
    "event2": "condition2"
}

# 定义执行引擎
executor = {
    "condition1": "grant_access",
    "condition2": "deny_access"
}

# 监听事件

    if eval(condition):
        # 触发规则

        # 执行规则
