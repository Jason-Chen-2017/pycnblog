```python
import numpy as np
from collections import Counter

# 文本数据
text = "我爱你，你爱我。"

# 对文本进行分词
words = text.split()

# 计算每个词的出现概率
word_count = Counter(words)
word_probability = {word: count / len(words) for word, count in word_count.items()}

# 构建马尔可夫链模型
transition_matrix = np.zeros((len(words), len(words)))
for i in range(len(words)):
    for j in range(len(words)):
        if words[i] == words[j]:
            transition_matrix[i, j] = 1

# 根据马尔可夫链模型，选择文本中的关键信息
key_words = [word for word, probability in word_probability.items() if probability > 0.5]

# 将选择的关键信息组合成一个新的文本摘要
summary = " ".join(key_words)

print(summary)
```
