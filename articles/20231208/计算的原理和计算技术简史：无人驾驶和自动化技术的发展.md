```python
import cv2
import numpy as np
from sklearn.svm import SVC
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

# 加载数据集
data = np.load('data.npy')
labels = np.load('labels.npy')

# 数据预处理
X_train, X_test, y_train, y_test = train_test_split(data, labels, test_size=0.2, random_state=42)

# 训练模型
model = SVC(kernel='linear')
model.fit(X_train, y_train)

# 测试模型
y_pred = model.predict(X_test)
accuracy = accuracy_score(y_test, y_pred)
print('Accuracy:', accuracy)
```

# 4.2自动化技术的具体代码实例
自动化技术的具体代码实例可以通过使用计算机控制、机器人技术和人工智能技术来实现生产过程的自动化。以下是一个自动化技术的具体代码实例：

```python
import rospy
from geometry_msgs.msg import Twist

# 初始化节点
rospy.init_node('autonomous_vehicle_controller', anonymous=True)

# 设置速度和方向
def set_speed_and_direction(linear_speed, angular_speed):
    pub = rospy.Publisher('/cmd_vel', Twist, queue_size=10)
    rate = rospy.Rate(10)

    while not rospy.is_shutdown():
        twist = Twist()
        twist.linear.x = linear_speed
        twist.angular.z = angular_speed
        pub.publish(twist)
        rate.sleep()

# 主函数
if __name__ == '__main__':
    set_speed_and_direction(0.5, 0.5)
```
