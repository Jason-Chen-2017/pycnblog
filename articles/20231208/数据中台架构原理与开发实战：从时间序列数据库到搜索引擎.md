```python
import datetime

def partition_by_time(data, time_granularity):
    partitions = []
    current_partition = []
    for value in data:
        timestamp = value['timestamp']
        current_timestamp = timestamp.replace(microsecond=0)
        if current_partition and current_partition[-1]['timestamp'] + time_granularity < current_timestamp:
            partitions.append(current_partition)
            current_partition = []
        current_partition.append(value)
    if current_partition:
        partitions.append(current_partition)
    return partitions
```

## 3.搜索引擎的搜索和排名代码实例

### 3.1搜索引擎的搜索实现

```python
import re

def search(query, index):
    query_words = query.split()
    results = []
    for word in query_words:
        results = search_word(word, results, index)
    return results

def search_word(word, results, index):
    word = word.lower()
    for position, value in index.items():
        if re.search(word, value):
            results.append((position, value))
    return results
```

### 3.2搜索引擎的排名实现

```python
def rank(query, index, rank_factors):
    query_words = query.split()
    rank_scores = {}
    for word in query_words:
        rank_scores = rank_word(word, rank_scores, index, rank_factors)
    ranked_results = sorted(rank_scores.items(), key=lambda x: x[1], reverse=True)
    return ranked_results

def rank_word(word, rank_scores, index, rank_factors):
    word = word.lower()
    for position, value in index.items():
        if re.search(word, value):
            for factor in rank_factors:
                rank_scores[(position, value)] += factor[word]
    return rank_scores
```
