    return a + b

def subtract(a, b):
    return a - b

def multiply(a, b):
    return a * b

def divide(a, b):
    if b == 0:
        raise ValueError("Cannot divide by zero")
    return a / b

    a = int(input("Enter first number: "))
    b = int(input("Enter second number: "))
    print("Sum:", add(a, b))
    print("Difference:", subtract(a, b))
    print("Product:", multiply(a, b))
    print("Quotient:", divide(a, b))

if __name__ == "__main__":
    main()
