## 4.1酒店预订推荐

我们可以使用Python的Scikit-learn库来实现酒店预订推荐系统。首先，我们需要收集一些数据，包括用户的历史行为数据和酒店的特征数据。然后，我们可以使用协同过滤算法来推荐合适的酒店。

```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

# 用户的历史行为数据
user_behavior = {
    'user1': ['hotel1', 'hotel2', 'hotel3'],
    'user2': ['hotel4', 'hotel5', 'hotel6'],
    'user3': ['hotel7', 'hotel8', 'hotel9'],
}

# 酒店的特征数据
hotel_features = {
    'hotel1': {'location': '北京', 'price': 200, 'rating': 4},
    'hotel2': {'location': '上海', 'price': 300, 'rating': 5},
    'hotel3': {'location': '广州', 'price': 250, 'rating': 3},
    'hotel4': {'location': '深圳', 'price': 350, 'rating': 4},
    'hotel5': {'location': '成都', 'price': 400, 'rating': 5},
    'hotel6': {'location': '杭州', 'price': 320, 'rating': 4},
    'hotel7': {'location': '重庆', 'price': 280, 'rating': 3},
    'hotel8': {'location': '西安', 'price': 220, 'rating': 4},
    'hotel9': {'location': '南昌', 'price': 210, 'rating': 3},
}

# 计算用户之间的相似度
user_similarity = cosine_similarity(user_behavior)

# 计算用户对酒店的预测评分
user_predicted_scores = {}
for user, items in user_behavior.items():
    user_predicted_scores[user] = {}
    for item in items:
        user_predicted_scores[user][item] = 0
        for other_user, other_items in user_behavior.items():
            if user == other_user:
                continue
            for other_item in other_items:
                user_predicted_scores[user][item] += user_similarity[user][other_user] * hotel_features[other_item][user]
                user_predicted_scores[user][item] /= np.sqrt(sum([user_similarity[user][other_user]**2 for other_user in user_behavior[user]]))

# 推荐最高评分的酒店
recommended_hotels = {}
for user, items in user_behavior.items():
    recommended_hotels[user] = []
    for item in items:
        recommended_hotels[user].append((item, user_predicted_scores[user][item]))
    recommended_hotels[user] = sorted(recommended_hotels[user], key=lambda x: x[1], reverse=True)

print(recommended_hotels)
```

在这个例子中，我们首先收集了用户的历史行为数据和酒店的特征数据。然后，我们使用了协同过滤算法来计算用户之间的相似度，并根据这个相似度来计算用户对酒店的预测评分。最后，我们根据预测评分来推荐最高评分的酒店。
