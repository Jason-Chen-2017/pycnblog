随机算法的具体代码实例和详细解释说明如下：

1. 随机排序：

```python
import random

def random_sort(arr):
    for i in range(len(arr) - 1, 0, -1):
        j = random.randint(0, i)
        arr[i], arr[j] = arr[j], arr[i]
    return arr

arr = [1, 2, 3, 4, 5]
print(random_sort(arr))
```

2. 随机选择：

```python
import random

def random_select(arr, k):
    n = len(arr)
    index = random.randint(0, n - 1)
    return arr[index]

arr = [1, 2, 3, 4, 5]
k = 3
print(random_select(arr, k))
```

3. 随机生成：

```python
import random

def random_generate(min_value, max_value, num):
    return [random.randint(min_value, max_value) for _ in range(num)]

min_value = 1
max_value = 10
num = 5
print(random_generate(min_value, max_value, num))
```

4. 随机分配：

```python
import random

def random_allocate(arr, num):
    result = [[] for _ in range(num)]
    for i in range(len(arr)):
        index = random.randint(0, num - 1)
        result[index].append(arr[i])
    return result

arr = [1, 2, 3, 4, 5]
num = 2
print(random_allocate(arr, num))
```

```python
def event_probability(event, event_space):
    n_event = event_space.count(event)
    n_space = len(event_space)
    return n_event / n_space

event = "A"
probability = event_probability(event, event_space)
print(probability)
```
