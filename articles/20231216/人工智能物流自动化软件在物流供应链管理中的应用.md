具体的代码实现如下：

```python
import networkx as nx
from networkx.algorithms import shortest_paths

def auto_transport_planning(transport_info):
    # 根据物流运输的信息，自动规划运输的路线
    # 例如，根据运输的地址，自动计算运输的路线
    G = nx.Graph()
    G.add_nodes_from(transport_info.keys())
    G.add_edges_from([(source, target, {'weight': weight}) for source, target, weight in transport_info.values()])
    shortest_path = shortest_paths.dijkstra_path(G, source='source', target='target')
    return shortest_path
```
