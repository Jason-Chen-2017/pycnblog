import numpy as np
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense

# 生成器
def generator(z):
    model = Sequential()
    model.add(Dense(128, input_dim=z, activation='relu'))
    model.add(Dense(64, activation='relu'))
    model.add(Dense(32, activation='relu'))
    model.add(Dense(16, activation='relu'))
    model.add(Dense(8, activation='relu'))
    model.add(Dense(4, activation='tanh'))
    return model

# 判别器
def discriminator(x):
    model = Sequential()
    model.add(Dense(16, input_dim=x.shape[1], activation='relu'))
    model.add(Dense(32, activation='relu'))
    model.add(Dense(64, activation='relu'))
    model.add(Dense(128, activation='relu'))
    model.add(Dense(1, activation='sigmoid'))
    return model

# 训练模型
def train(epochs, batch_size, z_dim):
    # 生成器和判别器
    generator = generator(z_dim)
    discriminator = discriminator(z_dim)

    # 优化器
    generator_optimizer = tf.keras.optimizers.Adam(0.0002, 0.5)
    discriminator_optimizer = tf.keras.optimizers.Adam(0.0002, 0.5)

    # 噪声
    noise = np.random.normal(0, 1, (batch_size, z_dim))

    for epoch in range(epochs):
        # 训练生成器
        noise = np.random.normal(0, 1, (batch_size, z_dim))
        z = np.concatenate((noise, noise), axis=1)
        z = np.concatenate((z, noise), axis=1)
        z = np.concatenate((z, noise), axis=1)
        z = np.concatenate((z, noise), axis=1)

        z = np.reshape(z, (batch_size, 100))

        generated_images = generator.predict(z)

        # 训练判别器
        real_images = np.random.normal(0, 1, (batch_size, 16))
        real_images = np.reshape(real_images, (batch_size, 4))

        d_loss_real = discriminator.train_on_batch(real_images, np.ones((batch_size, 1)))
        d_loss_fake = discriminator.train_on_batch(generated_images, np.zeros((batch_size, 1)))

        # 训练生成器
        noise = np.random.normal(0, 1, (batch_size, z_dim))
        z = np.concatenate((noise, noise), axis=1)
        z = np.concatenate((z, noise), axis=1)
        z = np.concatenate((z, noise), axis=1)
        z = np.concatenate((z, noise), axis=1)

        z = np.reshape(z, (batch_size, 100))

        g_loss = discriminator.train_on_batch(z, np.ones((batch_size, 1)))

        # 更新优化器
        generator_optimizer.update(generator.trainable_variables, g_loss)
        discriminator_optimizer.update(discriminator.trainable_variables, d_loss_real + d_loss_fake)

    return generator
