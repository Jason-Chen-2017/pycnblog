```python
import time

class Counter:
    def __init__(self):
        self.count = 0

    def increment(self):
        self.count += 1

    def display(self):
        print(f"Count: {self.count}")

def main():
    counter = Counter()

    while True:
        counter.increment()
        counter.display()
        time.sleep(1)

if __name__ == "__main__":
    main()
```
