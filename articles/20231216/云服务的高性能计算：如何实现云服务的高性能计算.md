# 读取数据
data = pd.read_csv('data.csv')

# 数据清洗

# 数据过滤
data = data[data['feature1'] > 0]

# 数据转换
data['feature2'] = data['feature2'].astype('float32')

# 数据分析

import matplotlib.pyplot as plt

# 数据探索
print(data.describe())

# 数据挖掘

# 聚类分析
kmeans = KMeans(n_clusters=3)
data['cluster'] = kmeans.fit_predict(data[['feature1', 'feature2']])

# 模型构建
from sklearn.ensemble import RandomForestClassifier

# 模型选择
model = RandomForestClassifier(n_estimators=100)

# 模型训练
model.fit(data[['feature1', 'feature2']], data['label'])

# 模型评估
from sklearn.metrics import accuracy_score

# 结果解释
y_pred = model.predict(data[['feature1', 'feature2']])
print('Accuracy:', accuracy_score(data['label'], y_pred))
