```python
def count_even_numbers(start, end):
    count = 0
    for i in range(start, end + 1):
        if i % 2 == 0:
            count += 1
    return count

if __name__ == '__main__':
    start = 0
    end = 100
    result = count_even_numbers(start, end)
    print(result)
```
