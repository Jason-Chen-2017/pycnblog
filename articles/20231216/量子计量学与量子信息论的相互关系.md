# 准备两个量子态，分别为 |0⟩ 和 |1⟩
|0⟩ = np.array([1, 0])

# 对两个量子态进行 Hadamard 变换，分别为 H

# 对两个量子态进行 CNOT 变换，分别为 CNOT

CNOT_01 = np.dot(CNOT, np.kron(H_0, |1⟩))

# 对两个量子态进行 Hadamard 变换，分别为 H

H_CNOT_01 = np.dot(H, CNOT_01)
