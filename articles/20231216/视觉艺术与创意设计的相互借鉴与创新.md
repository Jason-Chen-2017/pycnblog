## 4.1 生成艺术作品的代码实例

```python
import random
import numpy as np
import matplotlib.pyplot as plt

# 生成随机颜色
def random_color():
    return np.random.rand(3,)

# 生成随机形状
def random_shape():
    return np.random.randint(2, size=10)

# 生成随机大小
def random_size():
    return np.random.randint(10, size=10)

# 生成艺术作品
def generate_artwork():
    colors = [random_color() for _ in range(10)]
    shapes = [random_shape() for _ in range(10)]
    sizes = [random_size() for _ in range(10)]

    fig, ax = plt.subplots()
    for i, color in enumerate(colors):
        ax.plot(shapes[i], sizes[i], color=color, marker='o', linestyle='-')

    ax.set_xlim(-5, 5)
    ax.set_ylim(-5, 5)
    ax.set_aspect('equal', adjustable='box')
    plt.show()

if __name__ == '__main__':
    generate_artwork()
```

## 4.3 制作动画的代码实例

```python
import random
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation

# 生成随机颜色
def random_color():
    return np.random.rand(3,)

# 生成随机形状
def random_shape():
    return np.random.randint(2, size=10)

# 生成随机大小
def random_size():
    return np.random.randint(10, size=10)

# 制作动画
def animate(i):
    colors = [random_color() for _ in range(10)]
    shapes = [random_shape() for _ in range(10)]
    sizes = [random_size() for _ in range(10)]

    fig, ax = plt.subplots()
    for i, color in enumerate(colors):
        ax.plot(shapes[i], sizes[i], color=color, marker='o', linestyle='-')

    ax.set_xlim(-5, 5)
    ax.set_ylim(-5, 5)
    ax.set_aspect('equal', adjustable='box')

    return [ax]

if __name__ == '__main__':
    fig, ax = plt.subplots()
    ani = FuncAnimation(fig, animate, interval=1000)
    plt.show()
```
