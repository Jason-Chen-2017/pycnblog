import numpy as np
import matplotlib.pyplot as plt

# 超声波波动的数值模拟
def wave_simulation(x, t, c):
    u = np.zeros(x.shape)
    for i in range(x.shape[0]):
        u[i] = c * x[i] * t[i]
    return u

# 超声波散射计算
def scattering_calculation(k, r, l):
    jl = np.zeros(r.shape)
    hl = np.zeros(r.shape)
    for i in range(r.shape[0]):
        jl[i] = np.sin(r[i])
        hl[i] = np.sin(r[i]) - r[i] * np.cos(r[i])
    return (jl / hl) ** 2

# 超声波波动的吸收计算
def absorption_calculation(rho, c, delta):
    alpha = 0.5 * rho * c * np.gradient(delta)
    return alpha

# 主程序
x = np.linspace(0, 1, 100)

c = 1500

l = 1
r = np.arange(0, 10, 0.1)

u = wave_simulation(x, t, c)
scattering = scattering_calculation(k, r, l)
absorption = absorption_calculation(1000, c, u)
