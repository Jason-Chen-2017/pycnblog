import hashlib
import json
from time import time

class Block:
    def __init__(self, index, previous_hash, timestamp, data, hash):
        self.index = index
        self.previous_hash = previous_hash
        self.timestamp = timestamp
        self.data = data
        self.hash = hash

    @staticmethod
    def hash(block):
        block_string = json.dumps(block.__dict__, sort_keys=True).encode()
        return hashlib.sha256(block_string).hexdigest()

    @property
    def hash(self):
        return self._hash

    @hash.setter
    def hash(self, hash):
        self._hash = hash

    @staticmethod
    def create_genesis_block(data):
        return Block(0, "0", time(), data, "0")

class Blockchain:
    def __init__(self):
        self.chain = [Block.create_genesis_block("Genesis Block")]

    def add_block(self, data):
        index = len(self.chain)
        previous_hash = self.chain[-1].hash
        timestamp = time()
        block = Block(index, previous_hash, timestamp, data, self.hash(self.chain[-1]))
        self.chain.append(block)

    @staticmethod
    def hash(block):
        block_string = json.dumps(block.__dict__, sort_keys=True).encode()
        return hashlib.sha256(block_string).hexdigest()

    @property
    def hash(self):
        return self._hash

    @hash.setter
    def hash(self, hash):
        self._hash = hash

    @staticmethod
    def proof_of_work(previous_block, difficulty):
        nonce = 0
        while Block.hash(Block(previous_block, nonce, time(), difficulty)).startswith("0" * difficulty):
            nonce += 1
        return nonce

    def add_block_with_proof(self, data, proof):
        index = len(self.chain)
        previous_hash = self.chain[-1].hash
        timestamp = time()
        block = Block(index, previous_hash, timestamp, data, proof)
        self.chain.append(block)

    @staticmethod
    def is_valid_proof(previous_block, proof, difficulty):
        return Block.hash(Block(previous_block, proof, time(), difficulty)).startswith("0" * difficulty)

    def valid_chain(self):
        for i in range(1, len(self.chain)):
            if not Block.is_valid_proof(self.chain[i - 1], self.chain[i].proof, self.chain[i].difficulty):
                return False
        return True

# 创建区块链网络

    def __init__(self):

    def add_transaction(self, transaction):
        self.transactions.append(transaction)

    def valid_transactions(self):
        valid_transactions = []
        for transaction in self.transactions:
