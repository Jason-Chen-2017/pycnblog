### 1.4.1 量子仿真代码实例

以下是一个简单的量子仿真代码实例，它通过模拟一个简单的量子系统来预测系统的行为：

```python
import numpy as np
import scipy.linalg

# 定义量子系统的波函数
def wave_function(t):
    return np.exp(-t**2 / 2)

# 定义量子系统的运动符
def hamiltonian():
    return np.array([[0, -1j], [1j, 0]])

# 进行时间演化
def time_evolution(t):
    return scipy.linalg.expm(-1j * hamiltonian() * t)

# 计算物理量
def expectation_value(state, operator):
    return np.dot(state.conj().T, np.dot(operator, state))

# 初始化量子状态向量
state = np.array([1, 0])

# 进行时间演化
time = 0.1
evolved_state = time_evolution(time) @ state

# 计算能量
energy = expectation_value(evolved_state, hamiltonian())

print("能量:", energy)
```

```python
import numpy as np

    return np.exp(-x**2 / 2)

# 定义量子系统的运动符
def hamiltonian(x):
    return -1j * np.diff(wave_function(x), x)

# 初始化波函数
```
