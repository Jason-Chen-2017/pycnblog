        task_start = np.random.randint(0, task.size - task_size + 1)
        task_end = task_start + task_size
        tasks.append(task[task_start:task_end])
    return tasks

def allocate_resources(tasks, nodes):
    allocated_resources = []
    for task in tasks:
        resource = np.random.randint(1, 100)
        allocated_resources.append(resource)

    nodes = np.random.rand(10)
    tasks = divide_task(task, nodes)

    main()
