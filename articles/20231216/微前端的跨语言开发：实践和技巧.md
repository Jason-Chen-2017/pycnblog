```html
  </template>

      constructor() {
        super();
        this.attachShadow({mode: 'open'});
      }
      connectedCallback() {
        this.shadowRoot.innerHTML = `<div>Hello, world!</div>`;
      }
    }
    customElements.define('my-app', MyApp);
  </script>
</dom-module>
```
