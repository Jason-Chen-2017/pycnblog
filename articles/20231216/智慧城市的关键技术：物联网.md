import random

    def __init__(self, pin):
        self.pin = pin
        self.value = 0

    def read(self):
        self.value = random.randint(0, 100)
        return self.value

# 创建一个传感器对象
sensor = Sensor(2)

# 主程序循环
while True:
    # 读取传感器值
    value = sensor.read()

    # 输出传感器值
    print("Sensor value:", value)

    # 延时一段时间
    time.sleep(1)
