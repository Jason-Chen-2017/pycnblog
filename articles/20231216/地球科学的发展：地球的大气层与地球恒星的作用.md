```python
import numpy as np
import matplotlib.pyplot as plt

# 定义大气层和恒星的参数
atmosphere_parameters = {'CO2': 400, 'CH4': 1800, 'N2O': 330, 'O3': 350}
sun_parameters = {'luminosity': 3.828e26, 'radius': 6.96e8, 'temperature': 5778}

# 定义大气层和恒星的数学模型
def calculate_temperature(atmosphere, sun):
    # 计算大气层的辐射平衡温度
    balance_temperature = (sun['luminosity'] * atmosphere['CO2'] * 4.34e-11) / (4.0 * np.pi * sun['radius'] ** 2)

    # 计算大气层的温度反馈
    feedback_temperature = (sun['luminosity'] * atmosphere['CH4'] * 4.34e-11) / (4.0 * np.pi * sun['radius'] ** 2)

    # 计算地球的平均温度
    average_temperature = balance_temperature + feedback_temperature

    return average_temperature

# 计算地球的气候
average_temperature = calculate_temperature(atmosphere_parameters, sun_parameters)
print('地球的平均温度为：', average_temperature, 'K')

# 绘制大气层和恒星的影响
plt.plot(atmosphere_parameters['CO2'], average_temperature, label='大气层')
plt.plot(sun_parameters['luminosity'], average_temperature, label='恒星')
plt.xlabel('参数')
plt.ylabel('温度')
plt.legend()
plt.show()
```
