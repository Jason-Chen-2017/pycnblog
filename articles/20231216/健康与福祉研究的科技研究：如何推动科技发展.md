# 定义线性回归算法
def linear_regression(X, y):
    # 计算X的逆矩阵
    X_inv = np.linalg.inv(X.T @ X)
    # 计算权重
    w = X_inv @ (X.T @ y)
    return w

# 使用线性回归算法预测结果
