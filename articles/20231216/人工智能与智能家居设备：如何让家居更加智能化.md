## 4.4 语音识别代码实例

以下是一个简单的语音识别代码实例，用于识别语音命令：

```python
import numpy as np
import librosa
import torch
from torch import nn, optim

# 数据加载
audio_data = np.load('voice_data.wav')

# 数据预处理
audio_data = librosa.to_mono(audio_data)
audio_data = librosa.effects.normalize(audio_data)

# 模型构建
model = nn.Sequential(
    nn.Conv1d(1, 32, kernel_size=3, stride=2, padding=1),
    nn.ReLU(),
    nn.MaxPool1d(kernel_size=2, stride=2),
    nn.Conv1d(32, 64, kernel_size=3, stride=2, padding=1),
    nn.ReLU(),
    nn.MaxPool1d(kernel_size=2, stride=2),
    nn.Flatten(),
    nn.Linear(64 * 8 * 2, 128),
    nn.ReLU(),
    nn.Linear(128, 64),
    nn.ReLU(),
    nn.Linear(64, 1)
)

# 模型训练
criterion = nn.BCEWithLogitsLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

for epoch in range(10):
    optimizer.zero_grad()
    output = model(audio_data.unsqueeze(0))
    loss = criterion(output, torch.tensor([1]).unsqueeze(0))
    loss.backward()
    optimizer.step()

# 模型测试
pred = torch.sigmoid(model(audio_data.unsqueeze(0)))
```

## 4.5 语音合成代码实例

以下是一个简单的语音合成代码实例，用于生成语音命令：

```python
import numpy as np
import librosa
import torchaudio
from torchaudio.datasets import Librispeech
from torchaudio.transforms import MelSpectrogram, LogMelSpectrogram

# 文本加载
text = "请打开灯光"

# 文本转换
mel_spectrogram = LogMelSpectrogram()(text)

# 语音生成
waveform = torchaudio.composite.generate_waveform(
    mel_spectrogram,
    sampling_rate=16000,
    duration=len(text) / 16000,
    n_iter=100,
    n_jobs=1
)

# 语音处理
waveform = librosa.effects.reverb(waveform, room=0.5, duration=1.5)
waveform = librosa.effects.normalize(waveform)

# 语音保存
librosa.output.write_wav('voice_command.wav', waveform, 16000)
```
