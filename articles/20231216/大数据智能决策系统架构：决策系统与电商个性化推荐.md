```python
import scrapy

class UserBehaviorSpider(scrapy.Spider):
    name = 'user_behavior'
    allowed_domains = ['example.com']
    start_urls = ['http://example.com/user_behavior']

    def parse(self, response):
        for user_behavior in response.xpath('//div[@class="user_behavior"]'):
            yield {
                'user_id': user_behavior.xpath('@data-user-id').extract_first(),
                'item_id': user_behavior.xpath('@data-item-id').extract_first(),
                'behavior': user_behavior.xpath('@data-behavior').extract_first(),
                'timestamp': user_behavior.xpath('@data-timestamp').extract_first(),
            }
```

```python
from pyspark import SparkContext

sc = SparkContext()
hadoop_conf = sc._conf
hadoop_conf.set("fs.defaultFS", "hdfs://localhost:9000")
hadoop_conf.set("fs.hdfs.impl", "org.apache.hadoop.hdfs.DistributedFileSystem")

user_behavior_rdd = sc.textFile("hdfs://localhost:9000/user_behavior.csv")
user_behavior_rdd = user_behavior_rdd.map(lambda line: line.split(','))
```

```python
from pyspark.sql import SparkSession

spark = SparkSession.builder.appName("data_processing").getOrCreate()
user_behavior_df = spark.createDataFrame(user_behavior_rdd)

user_behavior_df = user_behavior_df.withColumn("behavior", user_behavior_df["behavior"].cast("int"))
user_behavior_df = user_behavior_df.withColumn("timestamp", user_behavior_df["timestamp"].cast("long"))
user_behavior_df = user_behavior_df.filter(user_behavior_df["behavior"] == 1)
```
