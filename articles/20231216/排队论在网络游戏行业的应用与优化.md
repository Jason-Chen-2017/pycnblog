## 4.1 实现先来先服务排队策略的代码实例

```python
class QueueSystem:
    def __init__(self):
        self.queue = []
        self.serving = None

    def enqueue(self, player):
        self.queue.append(player)

    def dequeue(self):
        if self.serving is None:
            self.serving = self.queue.pop(0)
        else:
            self.serving = self.serving.next

    def serve(self, time):
        if self.serving is None:
            return
        self.serving.time += time
        if self.serving.time <= 0:
            self.serving.next = self.queue.pop(0)
            self.serving = None

```

```python
    def dequeue(self):
        if self.serving is None:
            self.serving = self.queue[0]
            self.queue.pop(0)
        else:
            min_time = float('inf')
            min_player = None
            for player in self.queue:
                if player.time < min_time:
                    min_time = player.time
                    min_player = player
            self.serving = min_player
            self.queue.remove(min_player)

        if self.serving is None:
            return
        self.serving.time += time
        if self.serving.time <= 0:
```
