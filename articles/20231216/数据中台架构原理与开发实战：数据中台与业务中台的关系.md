    correct_data = 0
    total_data = len(data)
    for item in data:
        if is_correct(item):
            correct_data += 1
    data_quality = (correct_data / total_data) * 100

    cleaned_data = []
    for item in data:
        if is_clean(item):
            cleaned_data.append(item)
    clean_data_rate = (len(cleaned_data) / len(data)) * 100
    return cleaned_data, clean_data_rate
```

### 4.1.3数据集成

def integrate_data(data1, data2):
    data_1_len = len(data1)
    data_2_len = len(data2)
    integrated_data = []
    for i in range(data_1_len):
        for j in range(data_2_len):
            if data1[i] == data2[j]:
                integrated_data.append(data1[i])
    data_integration_rate = (len(integrated_data) / (data_1_len + data_2_len)) * 100
    return integrated_data, data_integration_rate
```
