### 5.2 源代码详细实现

以下是一个简单的元宇宙养老保险系统的代码实例：

```python
# app.py

from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from ethereum.contract import Contract

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///metaverse_pension.db'
db = SQLAlchemy(app)

class PensionAccount(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.String(100), unique=True)
    balance = db.Column(db.Float)

@app.route('/deposit', methods=['POST'])
def deposit():
    user_id = request.form['user_id']
    amount = float(request.form['amount'])

    account = PensionAccount.query.get(user_id)
    if account:
        account.balance += amount
    else:
        account = PensionAccount(user_id=user_id, balance=amount)

    db.session.add(account)
    db.session.commit()

    return jsonify({'status': 'success', 'balance': account.balance})

@app.route('/withdraw', methods=['POST'])
def withdraw():
    user_id = request.form['user_id']
    amount = float(request.form['amount'])

    account = PensionAccount.query.get(user_id)
    if account and account.balance >= amount:
        account.balance -= amount
        db.session.commit()
        return jsonify({'status': 'success', 'balance': account.balance})
    else:
        return jsonify({'status': 'failure', 'message': 'insufficient balance'})

if __name__ == '__main__':
    db.create_all()
    app.run(debug=True)
```

- **取款功能**：定义了 /withdraw 接口，接收用户 ID 和取款金额，检查余额并更新养老金账户余额。
