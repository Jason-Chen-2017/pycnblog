# app.py

from flask import Flask, request, jsonify
from models import User, Course

app = Flask(__name__)

@app.route('/users', methods=['POST'])
def create_user():
    data = request.json
    user = User(name=data['name'], email=data['email'])
    # 保存用户到数据库
    return jsonify({'status': 'success', 'user': user.to_dict()})

@app.route('/courses', methods=['POST'])
def create_course():
    data = request.json
    course = Course(name=data['name'], price=data['price'])
    # 保存课程到数据库
    return jsonify({'status': 'success', 'course': course.to_dict()})

@app.route('/users/<int:user_id>/courses', methods=['POST'])
def enroll_course(user_id):
    user = User.query.get(user_id)
    course = Course.query.get(request.json['course_id'])
    user.courses.append(course)
    # 更新用户信息到数据库
    return jsonify({'status': 'success', 'user': user.to_dict()})

if __name__ == '__main__':
    app.run(debug=True)
```

# app.py

from flask import Flask, request, jsonify
from models import User, Course

app = Flask(__name__)

@app.route('/users', methods=['POST'])
def create_user():
    data = request.json
    user = User(name=data['name'], email=data['email'])
    # 保存用户到数据库
    return jsonify({'status': 'success', 'user': user.to_dict()})

@app.route('/courses', methods=['POST'])
def create_course():
    data = request.json
    course = Course(name=data['name'], price=data['price'])
    # 保存课程到数据库
    return jsonify({'status': 'success', 'course': course.to_dict()})

@app.route('/users/<int:user_id>/courses', methods=['POST'])
def enroll_course(user_id):
    user = User.query.get(user_id)
    course = Course.query.get(request.json['course_id'])
    user.courses.append(course)
    # 更新用户信息到数据库
    return jsonify({'status': 'success', 'user': user.to_dict()})

if __name__ == '__main__':
    app.run(debug=True)
```
