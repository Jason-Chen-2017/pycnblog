    if cluster not in knowledge:
        knowledge[cluster] = []
    knowledge[cluster].append(data['cleaned_text'].iloc[i])

    if cluster not in knowledge:
        knowledge[cluster] = []
    knowledge[cluster].append(data['cleaned_text'].iloc[i])

# 问答系统
chatbot = ChatBot('KnowledgeDiscoveryBot')
