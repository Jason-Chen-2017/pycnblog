import time
import threading

def work(time_block):
    print(f"开始工作，时间块：{time_block}")
    time.sleep(time_block)
    print("工作完成")

def rest(rest_time):

    time_block = 25 * 60  # 时间块长度（秒）

if __name__ == "__main__":
    main()
