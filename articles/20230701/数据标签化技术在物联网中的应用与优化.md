4.4. 代码讲解说明

下面是一个简单的 Python 代码示例，用于实现数据预处理、特征提取和标签分配等功能：

```python
import numpy as np
import torch
from torch.utils.data import Dataset

class DataLabelingDataset(Dataset):
    def __init__(self, data):
        self.data = data

    def __getitem__(self):
        return self.data

    def __len__(self):
        return len(self.data)

preprocess_func = lambda x: x.astype('float') / 255.

# 数据预处理
data = [preprocess_func([x] for x in self.data) for _ in range(self.len())]

# 特征提取
features = []
for i in range(self.len()):
    for j in range(self.len()):
        if i!= j:
            features.append([self.data[i], self.data[j]])

# 标签分配
labels = []
for i in range(self.len()):
    labels.append(i)

# 保存数据
torch.save('data.pkl', {'data': data, 'features': features, 'labels': labels})
```
