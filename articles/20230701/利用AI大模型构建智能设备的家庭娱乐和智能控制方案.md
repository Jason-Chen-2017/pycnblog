9.3.1 智能音响

```
import numpy as np
import pymysql

class AudioSystem:
    def __init__(self):
        self.音响 = pymysql.connect(
            host='家庭服务器',
            user='USER名',
            password='PASSWORD',
            database='智能音响数据库'
        )

    def play_music(self, music_title, volume):
        query = f"SELECT * FROM 音乐库 WHERE 音乐标题 = {music_title}"
        result = self.音响.query(query)
        music_id = result[0]['音乐ID']
        self.音响.play(music_id, volume)

    def query_music(self, music_title):
        query = f"SELECT * FROM 音乐库 WHERE 音乐标题 = {music_title}"
        return self.音响.query(query)
```

9.3.3 智能电视

```
import numpy as np
import cv2

class TelevisionSystem:
    def __init__(self):
        self.cap = cv2.VideoCapture(0)
        self.cap.set(cv2.CAP_PROP_POS_FRAMES, 0)

    def show_video(self, video_path):
        self.cap.read()
        while True:
            ret, frame = self.cap.read()
            if ret:
                cv2.imshow('视频', frame)
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    break
                elif cv2.waitKey(10) & 0xFF == ord('s'):
                    self.show_video(video_path)
                elif cv2.waitKey(10) & 0xFF == ord('p'):
                    break
                elif cv2.waitKey(10) & 0xFF == ord('b'):
                    break
                else:
                    break
                ret, frame = self.cap.read()
            else:
                break

    def play_video(self, video_path):
        self.show_video(video_path)
```
