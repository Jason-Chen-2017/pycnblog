4.3. 核心代码实现

以下是一个简单的智能合约实现：

```
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract MyToken is ERC20, Ownable {
    using SafeMath for uint256;

    // 构造函数
    constructor() ERC20("My Token", "MT") {}

    // 部署函数
    deploy(address[] memory _buyers) external payable {
        _transfer(address(this), _buyers[0], _amount);
    }

    // 调用函数
    function() external payable {
        uint256 _amount = msg.value.mul(100);
        _transfer(address(this), payable(msg.sender), _amount);
    }

    // 构造函数调用
    constructorCaller(address payable _recipient, uint256 _amount) external payable {
        _transfer(address(this), _recipient, _amount);
    }

    // 事件
    event Transfer(address from, address to, uint256 value);

    // 安全性保护
    function _transfer(address to, uint256 value) internal override {
        require(msg.sender == address(this), "ERC20: transfer to the wrong contract");
        super._transfer(to, value);
    }

    function _call() internal override(ERC20) {
        uint256 amount = msg.value.mul(100);
        super._call(amount);
    }
}
```

这个智能合约实现了ERC20标准，包括ERC20代币的发行、代币的交易和代币提现等功能。部署这个合约需要使用Truffle网络的智能合约部署器。

```

```

最后，使用Truffle模拟器进行合约测试。在模拟器中，可以测试合约的发行、交易和提现等功能。
