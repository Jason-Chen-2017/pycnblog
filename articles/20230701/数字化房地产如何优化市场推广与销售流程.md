def calculate_similarity(data, query):
    # 构造相似度矩阵
    matrix = np.array([
        [0, 1, 1],
        [1, 0, 0],
        [1, 0, 0]
    ])
    
    # 遍历矩阵，计算相似度
    for i in range(len(data)):
        for j in range(i+1, len(data)):
            # 元素相加
            sum = np.sum([a*b for a, b in matrix])
            # 相似度
            similarity = (data[i] - data[j])**2 / sum**2
            # 添加元素

            matrix[j][i] = similarity
