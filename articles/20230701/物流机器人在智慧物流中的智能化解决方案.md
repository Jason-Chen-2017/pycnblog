------------------

物流机器人主要包括感知、决策、执行三个主要部分。感知部分主要负责收集环境信息，例如激光雷达、摄像头等传感器；决策部分负责对收集到的信息进行处理和分析，例如图像识别、语音识别等；执行部分主要负责实现物流机器人的行动，例如移动、搬运等。

import numpy as np
import rospy
from geometry_msgs.msg import PoseStamped

class robot:
    def __init__(self):
        self.bridge = CvBridge()
        self.image_sub = rospy.Subscriber("/camera/image", Image, self.image_callback)
        self.cmd_vel_pub = rospy.Publisher("/cmd_vel", PoseStamped, queue_size=1)

    def image_callback(self, image_msg):
        try:
            image = self.bridge.imgmsg_to_cv2(image_msg, "bgr8")
            cv_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
            contours, _ = cv2.findContours(cv_image, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
            for contour in contours:
                x, y, w, h = cv2.boundingRect(contour)
                rospy.loginfo("Contour with [%f, %f, %f, %f]" % (x, y, w, h))
        except CvBridgeError as e:
            rospy.logwarn_throttle(1, "CvBridge Error %s" % e)

    def cmd_vel_callback(self, msg):
        rospy.loginfo("Received command velocity message")

    def run(self):
        rate = rospy.Rate(10)
        while not rospy.is_shutdown():
            try:
                self.image_sub.ack()
                self.image_sub.info( camera_frame="/camera/image")
                image_data = self.bridge.imgmsg_to_cv2(self.image_sub, "bgr8")
                cv_image = cv2.cvtColor(image_data, cv2.COLOR_BGR2GRAY)
                contours, _ = cv2.findContours(cv_image, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
                for contour in contours:
                    x, y, w, h = cv2.boundingRect(contour)
                    self.bridge.send_message("/cmd_vel", PoseStamped, position=msg.pose.position, orientation=msg.pose.orientation)
                rate.sleep()
            except rospy.ServiceException as e:
                rospy.logwarn_throttle(1, "ServiceException %s" % e)
