1. 排序算法的代码实例：
```python
def sort(arr):
    for i in range(len(arr)):
        for j in range(i+1, len(arr)):
            if arr[i] > arr[j]:
                arr[i], arr[j] = arr[j], arr[i]
    return arr

arr = [3, 2, 1, 4]
sorted_arr = sort(arr)
print(sorted_arr)
```
2. 搜索算法的代码实例：

```python
def search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

arr = [3, 2, 1, 4]
target = 2
index = search(arr, target)
if index != -1:
    print("找到了，位置为：", index)
else:
    print("没有找到")
```
3. 分治算法的代码实例：
```python
def divide_conquer(arr, low, high):
    if low == high:
        return arr[low]
    mid = (low + high) // 2
    left_result = divide_conquer(arr, low, mid)
    right_result = divide_conquer(arr, mid + 1, high)
    return left_result + right_result

arr = [3, 2, 1, 4]
result = divide_conquer(arr, 0, len(arr) - 1)
print(result)
```
4. 动态规划算法的代码实例：
```python
def dp(arr):
    n = len(arr)
    dp = [0] * n
    for i in range(n):
        for j in range(i, n):
            if j == i:
                dp[j] = arr[i]
            else:
                dp[j] = max(dp[j], dp[j-1] + arr[j])
    return dp[n-1]

arr = [3, 2, 1, 4]
result = dp(arr)
print(result)
```
