## 4.1 社交网络分析代码实例

```python
import networkx as nx
import matplotlib.pyplot as plt

# 创建社交网络
G = nx.Graph()

# 添加节点
G.add_node('Alice')
G.add_node('Bob')
G.add_node('Charlie')

# 添加边
G.add_edge('Alice', 'Bob')
G.add_edge('Alice', 'Charlie')

# 计算度
degree_centrality = nx.degree_centrality(G)
print(degree_centrality)

# 计算中心性
betweenness_centrality = nx.betweenness_centrality(G)
print(betweenness_centrality)

# 绘制社交网络
pos = nx.spring_layout(G)
nx.draw(G, pos, with_labels=True)
plt.show()
```

## 4.2 评论分析代码实例

```python
import re
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.decomposition import LatentDirichletAllocation

# 读取评论数据
comments = ['I love this course!', 'This course is terrible.']

# 清洗评论数据
def clean_comments(comments):
    cleaned_comments = []
    for comment in comments:
        comment = re.sub(r'\W+', ' ', comment)
        cleaned_comments.append(comment)
    return cleaned_comments

# 清洗评论数据
cleaned_comments = clean_comments(comments)

# 词频统计
tfidf = TfidfVectorizer()
tfidf_matrix = tfidf.fit_transform(cleaned_comments)

# 主题模型
lda = LatentDirichletAllocation(n_components=2)
lda_matrix = lda.fit_transform(tfidf_matrix)

# 主题分析
topics = lda.components_
for i in range(len(topics)):
    print('Topic', i+1)
    for j in range(len(topics[i])):
        print(cleaned_comments[j], ':', topics[i][j])
```
