import numpy as np

def collaborative_filtering(user_item_ratings, user_item_ratings_matrix, k):
    user_item_ratings_matrix = np.array(user_item_ratings_matrix)
    user_item_ratings_matrix = user_item_ratings_matrix.T

    user_item_ratings_matrix_mean = np.mean(user_item_ratings_matrix, axis=0)
    user_item_ratings_matrix_std = np.std(user_item_ratings_matrix, axis=0)

    user_item_ratings_matrix_normalized = (user_item_ratings_matrix - user_item_ratings_matrix_mean) / user_item_ratings_matrix_std

    user_item_ratings_matrix_transposed_normalized = user_item_ratings_matrix_normalized.T

    similarity_matrix = np.dot(user_item_ratings_matrix_normalized, user_item_ratings_matrix_transposed_normalized) / (k * np.sqrt(np.sum(np.square(user_item_ratings_matrix_normalized), axis=1) * np.sum(np.square(user_item_ratings_matrix_transposed_normalized), axis=1)))

    return similarity_matrix
