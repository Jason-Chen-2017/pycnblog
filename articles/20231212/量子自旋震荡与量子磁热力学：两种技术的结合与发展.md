# 定义自旋系统的跃迁矩阵

qc.h(0)
qc.dx(0)

# 解析自旋系统的哈密顿方程

Eigenvectors = np.linalg.eig(E)

# 绘制自旋矢量在不同磁场下的动态行为
plot_histogram(Eigenvalues)
