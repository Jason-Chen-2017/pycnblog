```python
import numpy as np
from astropy.coordinates import SkyCoord
from astropy.time import Time

# 计算行星的轨道
def calculate_orbit(planet, time):
    # 获取行星的轨道信息
    orbit = planet.orbit
    # 获取行星的位置信息
    position = planet.position(time)
    # 计算行星的速度
    velocity = planet.velocity(time)
    # 计算行星的轨道信息
    return orbit, position, velocity

# 获取行星的轨道信息
planet = SkyCoord(ra=12, dec=34, distance=100, frame='icrs')
time = Time('2022-01-01')
orbit, position, velocity = calculate_orbit(planet, time)

# 输出结果
print('行星的轨道信息：', orbit)
print('行星的位置信息：', position)
print('行星的速度信息：', velocity)
```
