```python
import pandas as pd

# 读取数据
data = pd.read_csv('data.csv')

# 数据清洗
data = data.dropna()

# 数据转换
data['date'] = pd.to_datetime(data['date'])
data['day'] = data['date'].dt.day

# 数据集成
data = data.groupby('day').mean()
```
