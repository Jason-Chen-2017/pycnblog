```python
import cv2
import numpy as np

# 加载电子图像

# 增强图像对比度
img_enhanced = cv2.equalizeHist(img)

# 滤波图像
img_filtered = cv2.medianBlur(img_enhanced, 5)

# 二值化图像
ret, img_binary = cv2.threshold(img_filtered, 127, 255, cv2.THRESH_BINARY)

# 检测边缘
edges = cv2.Canny(img_binary, 50, 150)

# 识别缺陷
contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

# 绘制缺陷轮廓
for contour in contours:
    cv2.drawContours(img_binary, [contour], -1, (0, 255, 0), 2)

# 显示结果
cv2.imshow('Enhanced Image', img_enhanced)
cv2.imshow('Filtered Image', img_filtered)
cv2.imshow('Binary Image', img_binary)
cv2.imshow('Edges', edges)
cv2.imshow('Contours', img_binary)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
