def partition_data(data, num_nodes):
    data_size = len(data)
    partition_size = data_size // num_nodes
    partitions = []
    for i in range(num_nodes):
        start = i * partition_size
        end = (i + 1) * partition_size
        partitions.append(data[start:end])
    return partitions
```
