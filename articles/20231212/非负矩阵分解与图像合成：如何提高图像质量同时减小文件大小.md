def nmf(X, n_components, n_iter=100, tol=1e-6):
    W = np.random.rand(X.shape[0], n_components)
    H = np.random.rand(n_components, X.shape[1])

    for _ in range(n_iter):
        V = np.dot(W, H)
        delta = np.dot(X - V, X - V)
