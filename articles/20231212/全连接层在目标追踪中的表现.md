import tensorflow as tf

# 定义全连接层
def fully_connected_layer(input_layer, num_output, activation_function):
    # 定义权重和偏置
    weights = tf.Variable(tf.random_normal([input_layer.shape[1], num_output]))
    biases = tf.Variable(tf.zeros([num_output]))

    # 计算输出
    output = tf.add(tf.matmul(input_layer, weights), biases)

    # 应用激活函数
    if activation_function == 'relu':
        output = tf.nn.relu(output)
    elif activation_function == 'sigmoid':
        output = tf.nn.sigmoid(output)

    return output

# 使用全连接层
