### 4.1 机器学习代码实例

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.svm import LinearSVC

# 文本数据
texts = ["这是一个法律案例", "这是另一个法律案例"]

# 文本预处理
vectorizer = TfidfVectorizer()
vectorizer.fit_transform(texts)

# 模型训练
classifier = LinearSVC()
classifier.fit(vectorizer.transform(texts), [0, 1])

# 模型预测
predictions = classifier.predict(vectorizer.transform(["这是一个法律案例"]))
print(predictions)  # [1]
```

### 4.2 深度学习代码实例

```python
import tensorflow as tf

# 文本数据
texts = ["这是一个法律案例", "这是另一个法律案例"]

# 文本预处理
tokenizer = tf.keras.preprocessing.text.Tokenizer()
tokenizer.fit_on_texts(texts)
sequences = tokenizer.texts_to_sequences(texts)

# 模型构建
model = tf.keras.Sequential([
    tf.keras.layers.Embedding(input_dim=len(tokenizer.word_index) + 1, output_dim=128, input_length=len(sequences[0])),
    tf.keras.layers.LSTM(128),
    tf.keras.layers.Dense(1, activation='sigmoid')
])

# 模型训练
model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
model.fit(sequences, [0, 1], epochs=10, batch_size=1)

# 模型预测
predictions = model.predict(tokenizer.texts_to_sequences(["这是一个法律案例"]))
print(predictions)  # [1]
```
