### 4.1 使用Golang的数据库/sql包

Golang的数据库/sql包是一个用于访问数据库的标准库。我们可以使用这个包来连接数据库、执行查询、插入、更新和删除操作等。以下是一个简单的代码实例：

```go
package main

import (
	"database/sql"
	"fmt"
	"log"
)

func main() {
	// 连接数据库
	db, err := sql.Open("mysql", "root:password@tcp(localhost:3306)/test")
	if err != nil {
		log.Fatal(err)
	}
	defer db.Close()

	// 执行查询操作
	rows, err := db.Query("SELECT * FROM users")
	if err != nil {
		log.Fatal(err)
	}
	defer rows.Close()

	// 遍历查询结果
	for rows.Next() {
		var id int
		var name string
		err := rows.Scan(&id, &name)
		if err != nil {
			log.Fatal(err)
		}
		fmt.Println(id, name)
	}
}
```

### 4.2 使用Golang的gorm库

gorm是一个基于Golang的ORM库，它提供了简洁的API和高性能的数据库操作。我们可以使用gorm库来构建高性能的数据库应用。以下是一个简单的代码实例：

```go
package main

import (
	"fmt"
	"github.com/jinzhu/gorm"
	_ "github.com/jinzhu/gorm/dialects/mysql"
)

type User struct {
	ID   int
	Name string
}

func main() {
	// 连接数据库
	db, err := gorm.Open("mysql", "root:password@tcp(localhost:3306)/test")
	if err != nil {
		fmt.Println("连接数据库失败", err)
		return
	}
	defer db.Close()

	// 创建用户
	user := User{Name: "John"}
	db.Create(&user)

	// 查询用户
	var user2 User
	db.First(&user2, user.ID)
	fmt.Println(user2.Name)

	// 更新用户
	user2.Name = "Jack"
	db.Save(&user2)

	// 删除用户
	db.Delete(&user2)
}
```
