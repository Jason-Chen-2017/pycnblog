```java
import org.opentsdb.core.TSDB;
import org.opentsdb.core.DataPoint;
import org.opentsdb.core.TSD;

public class OpenTSDBExample {
    public static void main(String[] args) {
        // 创建时间序列
        String metric = "my_metric";
        String tags = "host:my_host";
        String timestamp = "1515350400000";
        double value = 100.0;

        // 存储时间序列
        TSDB tsdb = new TSDB("localhost:4242");
        TSD tsd = tsdb.getTSD(metric, tags);
        tsd.put(Long.parseLong(timestamp), value);

        // 查询时间序列
        double result = tsd.first().getValue();
        System.out.println(result);
    }
}
```

3.查询时间序列：通过获取时间序列的实例并获取第一个数据点来查询时间序列数据。
