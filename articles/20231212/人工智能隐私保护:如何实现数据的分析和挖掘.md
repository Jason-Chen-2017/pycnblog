### 4.2 隐私保护的具体代码实例

```python
import numpy as np

# 数据掩码
def mask_data(data, mask):
    masked_data = np.zeros_like(data)
    for i in range(data.shape[0]):
        masked_data[i] = data[i] * mask
    return masked_data

# 数据脱敏
def anonymize_data(data, k):
    anonymized_data = data.copy()
    for column in data.columns:
        unique_values = data[column].unique()
        if len(unique_values) > k:
            anonymized_data[column] = np.random.randint(0, k, size=data.shape[0])
    return anonymized_data
```
