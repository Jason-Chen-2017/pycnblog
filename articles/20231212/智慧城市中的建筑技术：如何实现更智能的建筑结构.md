## 4.3 通信系统代码实例

```python
import socket

# 建立通信网络
def create_socket():
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.bind(('localhost', 12345))
    s.listen(5)
    return s

# 编码传感器数据
def encode_data(data):
    encoded_data = data.astype(np.float32).tobytes()
    return encoded_data

# 编码控制命令
def encode_command(command):
    encoded_command = command.astype(np.int32).tobytes()
    return encoded_command

# 传输传感器数据
def send_data(s, data):
    encoded_data = encode_data(data)
    s.sendall(encoded_data)

# 传输控制命令
def send_command(s, command):
    encoded_command = encode_command(command)
    s.sendall(encoded_command)

# 解码传感器数据
def decode_data(data):
    decoded_data = np.frombuffer(data, dtype=np.float32)
    return decoded_data

# 解码控制命令
def decode_command(data):
    decoded_command = np.frombuffer(data, dtype=np.int32)
    return decoded_command

# 接收传感器数据
def receive_data(s):
    data = s.recv(1024)
    return decode_data(data)

# 接收控制命令
def receive_command(s):
    command = s.recv(1024)
    return decode_command(command)
```
