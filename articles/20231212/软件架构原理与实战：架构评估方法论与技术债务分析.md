```python
import ast

def cyclomatic_complexity(source_code):
    tree = ast.parse(source_code)
    complexity = 0
    for node in ast.walk(tree):
        if isinstance(node, ast.If):
            complexity += 1
        elif isinstance(node, ast.While):
            complexity += 1
        elif isinstance(node, ast.For):
            complexity += 1
    return complexity
```
