    def update_velocity(self, target, frequency, noise):

def bat_algorithm(problem, max_iterations, num_bats, loudness_range, visibility_range):
    bats = [Bat(np.random.rand(problem.shape), np.random.rand(problem.shape), np.random.uniform(loudness_range[0], loudness_range[1]), np.random.uniform(visibility_range[0], visibility_range[1])) for _ in range(num_bats)]

    for t in range(max_iterations):

        for bat in bats:

            frequency = np.random.uniform(0, 1)
            noise = np.random.uniform(-1, 1)
