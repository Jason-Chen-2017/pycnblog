```python
import cv2
import numpy as np

def preprocess_image(image):
    # 读取图像
    img = cv2.imread(image)
    # 转换为灰度图像
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    # 进行二值化处理
    ret, binary = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    # 返回处理后的图像
    return binary
```
