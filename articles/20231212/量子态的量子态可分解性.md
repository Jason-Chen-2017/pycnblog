import numpy as np
from qiskit import QuantumCircuit, Aer, transpile, assemble
from qiskit.visualization import plot_histogram

# 创建量子电路

# 设置量子比特的初始状态
qc.h(0)
qc.cx(0, 1)

# 将量子电路编译为优化后的量子电路
optimized_qc = transpile(qc, backend=Aer.get_backend('qasm_simulator'))

# 将优化后的量子电路编译为量子门集
quantum_program = assemble(optimized_qc)

# 执行量子门集
result = quantum_program.run()

# 绘制结果
plot_histogram(result.get_counts())
