    data = data.transform()

    # 数据统计

    data = data.map()
    return data

# 主函数

    # 导入数据集
    data = knime.data.DataTable.import_csv("data.csv")
    # 数据预处理
    data = clean_data(data)

    data = analyze_data(data)
    # 数据可视化
    data = visualize_data(data)
