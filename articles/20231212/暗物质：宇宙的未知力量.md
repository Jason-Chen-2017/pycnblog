import numpy as np
from scipy.optimize import minimize

def dark_matter_potential(positions, masses, G):
    r = np.linalg.norm(positions, axis=1)
    potential = -G * np.sum(masses * r)
    return potential

def dark_matter_force(positions, masses, G):
    r = np.linalg.norm(positions, axis=1)
    force = G * masses / r**3 * positions
    return force

def dark_matter_constraint(positions, masses, G):
    potential = dark_matter_potential(positions, masses, G)
    return potential

def dark_matter_optimize(positions, masses, G, initial_guess):
    constraints = {'type': 'eq', 'fun': dark_matter_constraint}
    bounds = ((0, np.inf),) * len(positions)
    result = minimize(dark_matter_potential, initial_guess, args=(masses, G), bounds=bounds, constraints=constraints)
    return result.x

# 初始化星系的位置和质量
positions = np.array([[0, 0], [1, 0], [0, 1]])
masses = np.array([1, 1, 1])

# 设置引力常数

# 计算暗物质的位置
initial_guess = np.array([[0, 0], [0, 0], [0, 0]])
result = dark_matter_optimize(positions, masses, G, initial_guess)

# 输出结果
print(result)
