# 4.具体代码实例和详细解释说明

## 4.1 代码实例

以下是一个简单的图数据库和图计算的代码实例：

```python
# 定义图数据结构
class Graph:
    def __init__(self):
        self.nodes = {}

    def add_node(self, node_id, node_data):
        self.nodes[node_id] = node_data

    def add_edge(self, node_id1, node_id2, edge_data):
        self.nodes[node_id1].edges[node_id2] = edge_data
        self.nodes[node_id2].edges[node_id1] = edge_data

# 定义图算法
def shortest_path(graph, start_node_id, end_node_id):
    visited = set()
    queue = [(start_node_id, [start_node_id])]

    while queue:
        current_node_id, path = queue.pop(0)

        if current_node_id == end_node_id:
            return path

        if current_node_id not in visited:
            visited.add(current_node_id)
            for neighbor_node_id, edge_data in graph.nodes[current_node_id].edges.items():
                if neighbor_node_id not in visited:
                    queue.append((neighbor_node_id, path + [neighbor_node_id]))

# 使用图数据结构和图算法
graph = Graph()
graph.add_node(1, {'name': 'A'})
graph.add_node(2, {'name': 'B'})
graph.add_node(3, {'name': 'C'})
graph.add_edge(1, 2, {'weight': 1})
graph.add_edge(1, 3, {'weight': 2})
graph.add_edge(2, 3, {'weight': 3})

shortest_path_result = shortest_path(graph, 1, 3)
print(shortest_path_result)  # [1, 2, 3]
```
