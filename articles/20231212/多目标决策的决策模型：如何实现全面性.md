import numpy as np
from scipy.optimize import linprog

# 定义目标权重
weights = np.array([1.0, 1.0])

# 定义决策变量
decision_variables = np.array([x1, x2, x3, x4, x5, x6])

# 定义决策约束
constraints = (
    {'type': 'ineq', 'fun': lambda x: x1 + x2 + x3 + x4 + x5 + x6 - 100},
    {'type': 'ineq', 'fun': lambda x: x1 + x2 + x3 + x4 - 50},
    {'type': 'ineq', 'fun': lambda x: x2 + x3 + x4 - 30},
    {'type': 'ineq', 'fun': lambda x: x3 + x4 - 20},
    {'type': 'ineq', 'fun': lambda x: x4 - 10},
    {'type': 'ineq', 'fun': lambda x: x1 + x2 + x3 + x4 + x5 + x6 - 100},
    {'type': 'ineq', 'fun': lambda x: x1 + x2 + x3 + x4 - 50},
    {'type': 'ineq', 'fun': lambda x: x2 + x3 + x4 - 30},
    {'type': 'ineq', 'fun': lambda x: x3 + x4 - 20},
    {'type': 'ineq', 'fun': lambda x: x4 - 10},
)

# 定义目标函数
objective_function = lambda x: weights[0] * (x1 + x2 + x3 + x4 + x5 + x6) + weights[1] * (x1 + x2 + x3 + x4)

# 求解决策问题
result = linprog(objective_function, decision_variables, constraints, method='simplex')

# 输出决策结果
print(result)
