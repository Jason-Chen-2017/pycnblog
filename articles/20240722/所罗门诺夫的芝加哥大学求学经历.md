import numpy as np

def greedy_coloring(graph):
    n = len(graph)
    colors = np.zeros(n, dtype=int)
    for i in range(n):
        neighbors = graph[i]
        if colors[i] == 0:
            color = 1
            while any(color in colors[neighbors]):
                color += 1
            colors[i] = color
    return colors
