import sympy as sp
import sympy.sets as sets

# 定义Lusin公式
def Lusin(A):
    return A.union(A.complement()) == sp.S.Reals

# 定义Baire集合
def Baire(A):
    return not sp.Eq(A, sp.Union(*[A] for _ in range(sp.oo)))

# 验证Lusin公式

print(Lusin(A))

# 验证Baire集合不可测
