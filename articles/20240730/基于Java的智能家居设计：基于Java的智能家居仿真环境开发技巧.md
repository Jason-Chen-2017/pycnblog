```java
import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class SmartLightControl extends JFrame implements ActionListener {
    private JLabel label;
    private JButton button;
    
    public SmartLightControl() {
        setTitle("Smart Light Control");
        setSize(400, 300);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        
        label = new JLabel("Smart Light Control");
        label.setBounds(50, 100, 200, 50);
        add(label);
        
        button = new JButton("Turn On");
        button.setBounds(100, 200, 100, 50);
        button.addActionListener(this);
        add(button);
    }
    
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == button) {
            label.setText("Light Turned On");
        }
    }
    
    public static void main(String[] args) {
        new SmartLightControl().setVisible(true);
    }
}
```
