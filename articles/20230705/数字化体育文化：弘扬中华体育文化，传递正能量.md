```python
import numpy as np
import tensorflow as tf

# 数据预处理
data = np.array([
    [123, 123, 123, 123],
    [123, 222, 222, 222],
    [222, 333, 333, 333],
    [333, 444, 444, 444],
])

    features.append(row)

    [123, 222, 222, 222],
    [222, 333, 333, 333],
    [333, 444, 444, 444],
])

test_loss = model.evaluate(test_data)
print('Test loss:', test_loss)

# 模型优化
model.fit(test_data, epochs=10, batch_size=32)
```
