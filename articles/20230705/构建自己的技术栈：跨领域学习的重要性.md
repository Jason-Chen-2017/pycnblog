# -*- coding: utf-8 -*-

import random

class Solution:
    def __init__(self):
        self.max_num = 1000

    def transform_remainder(self, num, div):
        remainder = num % div
        return (remainder + div - div) % div

    def inverse_transform_remainder(self, num, div):
        return (num // div - div) % div

    def apply_remainder(self, num, operation):
        return operation(num, operation) % self.max_num

    def __repr__(self):
        return f"InverseTransformRemainder({self.max_num}, {div})"


def main():
    s = Solution()
    num = random.randint(1, 100)
    div = random.randint(2, 10)
    print(s.transform_remainder(num, div))  # 输出结果
    print(s.inverse_transform_remainder(num, div))  # 输出结果
    print(s.apply_remainder(num, s.transform_remainder))  # 输出结果
    print(s.apply_remainder(num, s.inverse_transform_remainder))  # 输出结果

if __name__ == "__main__":
    main()
