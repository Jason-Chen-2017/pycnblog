下面是一个简单的Python代码实现：
```python
import random
import numpy as np

def generate_random_number(seed):
    return np.random.randint(0, 1)

def simulate_system(num_steps, temperature, initial_state):
    state = initial_state
    for _ in range(num_steps):
        random_number = generate_random_number(seed)
        new_state = (1 - temperature) * state + random_number
        state = new_state
    return state

def main():
    # 设置随机数种子
    seed = 100

    # 设置模拟温度
    temperature = 100

    # 设置模拟系统的初始状态
    initial_state = [1, 0, 0]

    # 模拟系统的状态，直到达到停止条件
    num_steps = 1000
    states = [simulate_system(num_steps, temperature, initial_state) for _ in range(num_steps)]

    return states

# 计算模拟结果
states = main()
```
