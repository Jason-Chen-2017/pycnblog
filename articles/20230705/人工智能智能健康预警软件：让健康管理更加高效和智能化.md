import random
import time

class DataCollector:
    def __init__(self, device, port):
        self.device = device
        self.port = port

    def collect_data(self):
        data = []
        while True:
            if not self.device.is_connected():
                time.sleep(1)
                continue

            data.append(self.device.read_data())

            if len(data) >= 100:
                self.device.write_data(data[:-1])
                data = []

        return data
