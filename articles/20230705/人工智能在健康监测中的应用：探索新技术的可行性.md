import numpy as np
import pandas as pd

def preprocess(data):
    # 处理缺失值
    data = data.fillna(0)
    # 处理异常值
    data = data[(data!= 0) % 1]
    # 处理重复值
    data = data.drop_duplicates()
    # 处理噪声值
    data = data[(data - np.mean(data)) % 10000]
    return data

# 处理数据
