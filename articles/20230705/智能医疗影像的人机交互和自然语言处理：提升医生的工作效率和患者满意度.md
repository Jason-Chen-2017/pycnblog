import numpy as np
import tensorflow as tf

class Word2Vec(object):
    def __init__(self, size, min_count, sg=0.05):
        self.size = size
        self.min_count = min_count
        self.sg = sg
        self.word_to_vec = {}
        self.vec_to_word = {}

    def word_to_vec(self, word, epochs=100):
        # convert word to vectors
        for epoch in range(epochs):
            self.word_to_vec[word] = np.array([self.sg * np.random.randn(1, self.size) for _ in range(self.size)])

    def vec_to_word(self, vec, epochs=100):
        # convert vectors to words
        for epoch in range(epochs):
            self.vec_to_word[vec] = np.argmax(vec, axis=1)

    def train(self, data, epochs=100):
        # train word vectors
        for epoch in range(epochs):
            for word, word_vec in data.items():
                self.word_to_vec[word] = word_vec

            # train word index
            for word, word_vec in data.items():
                self.vec_to_word[word] = word_vec

    def test(self, data):
        # test word vectors
        total = 0
        correct = 0
        for word, word_vec in data.items():
            try:
                index = self.vec_to_word[word]
                total += 1
                if index == word:
                    correct += 1
            except:
                pass

        # calculate accuracy
        accuracy = correct / total

        return accuracy
