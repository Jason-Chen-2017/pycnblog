import numpy as np
import re
import nltk
nltk.download('punkt')

def preprocess_text(text):
    # 去除标点符号
    text = text.lower()
    # 去除停用词
    text = nltk.corpus.stopwords.words('english') + text
    # 去除数字
    text = re.sub('\d', '', text)
    # 分词
    text = nltk.word_tokenize(text)
    # 去除词形变化
    text = text.变化_words()
    return text

def analyze_text(text):
    # 对文本进行分词
    text = nltk.word_tokenize(text)
    # 统计词频
    word_count = nltk.FreqDist(text)
    # 分析文本
    for word, freq in word_count.items():
        if freq > 50:
            print('{}: {}'.format(word, freq))

# 法律文本分析
text = '被告人张三因涉嫌故意杀人罪，被判处有期徒刑十年。'
result = analyze_text(text)

# 法律文本推理
text = '根据我国刑法的规定，犯故意杀人罪，应当判处死刑。'
result = analyze_text(text)
