import sqlite3
import os
import re

# 连接数据库
conn = sqlite3.connect('政府印刷数据.db')

# 数据库操作函数
def create_table():
    cursor = conn.cursor()
    cursor.execute('''CREATE TABLE IF NOT EXISTS government_print
                   (id TEXT PRIMARY KEY AUTOINCREMENT,
                   name TEXT NOT NULL,
                   description TEXT,
                   publish_time TIMESTAMP NOT NULL,
                   is_deleted INTEGER NOT NULL,
                   FOREIGN KEY (user_id) REFERENCES user_info(id))''')
    conn.commit()

# 数据插入函数
def insert_data(name, description):
    cursor = conn.cursor()
    cursor.execute('''INSERT INTO government_print (name, description)
                      VALUES (?,?)''', (name, description))
    conn.commit()

# 数据查询函数
def query_data(name):
    cursor = conn.cursor()
    cursor.execute('''SELECT * FROM government_print WHERE name =?''', (name,))
    row = cursor.fetchone()
    if row:
        return row
    else:
        return None

# 数据修改函数
def update_data(id, name, description):
    cursor = conn.cursor()
    cursor.execute('''UPDATE government_print
                      SET name =?, description =? WHERE id =?''', (name, description, id))
    conn.commit()

# 数据导出函数
def export_data(name):
    cursor = conn.cursor()
    cursor.execute('''SELECT * FROM government_print WHERE name =?''', (name,))
    rows = cursor.fetchall()
    for row in rows:
        print(row)

# 关闭数据库连接
def close_connection():
    cursor.close()
    conn.close()

# 运行函数
if __name__ == '__main__':
    create_table()
    print('政府印刷数据表创建成功！')

    while True:
        # 显示菜单
        print('1. 查询数据')
        print('2. 插入数据')
        print('3. 修改数据')
        print('4. 导出数据')
        print('5. 关闭连接')
        choice = int(input('请输入操作编号：'))

        if choice == 1:
            name = input('请输入要查询的数据名称：')
            result = query_data(name)
            if result:
                print('正在查询数据，请稍候...')
                print(result)
            else:
                print('查无相关数据，请重新查询！')
        elif choice == 2:
            name = input('请输入要插入的数据名称：')
            description = input('请输入数据描述：')
            result = insert_data(name, description)
            if result:
                print('插入数据成功！')
                print('数据插入成功！')
            else:
                print('插入数据失败，请重新尝试！')
        elif choice == 3:
            id = input('请输入要修改的数据编号：')
            name = input('请输入要修改的数据名称：')
            description = input('请输入数据修改内容：')
            result = update_data(id, name, description)
            if result:
                print('修改数据成功！')
                print('数据修改成功！')
            else:
                print('修改数据失败，请重新尝试！')
        elif choice == 4:
            name = input('请输入要导出的数据名称：')
            result = export_data(name)
            if result:
                print('导出数据成功！')
                print('数据导出成功！')
            else:
                print('导出数据失败，请重新尝试！')
        elif choice == 5:
            close_connection()
            break
        else:
            print('输入有误，请重新输入！')
