import jieba
from pprint import pprint


def zh_to_拼音(zh):
    pinyin = []
    for i in range(len(zh)):
        if zh[i] in 'abcdefghijklmnopqrstuvwxyz':
            pinyin.append('a')
        else:
            pinyin.append('b')
    return pinyin


def preprocess(text):
    words = jieba.cut(text)
    pinyin = zh_to_拼音(text)
    return pinyin


def analyze(text):
    words = preprocess(text)
    pinyin = zh_to_拼音(text)
    # 添加标点符号
    words = [f'{pinyin[0][i]}{pinyin[1][i]}{pinyin[2][i]...}{pinyin[-1][i]}']
    return''.join(words)


def main():
    text = '这是一道作业题'
    result = analyze(text)
    pprint(result)


if __name__ == '__main__':
    main()
