### 3.2. 核心模块实现

首先，需要实现自然语言处理的基本功能，包括文本预处理、文本分类、信息提取等。以下是一个简单的示例实现：

```python
import spacy
from spacy.tokenize import word_tokenize
from spacy.parsing.preprocessing import STOPWORDS
import re

nlp = spacy.load('en_core_web_sm')

def preprocess(text):
    # 去除停用词
    stop_words = set(STOPWORDS)
    words = [word for word in nlp.vocab if word not in stop_words]
    # 去除标点符号
    words = [word.lower() for word in words]
    # 去除数字
    words = [word for word in words if not re.match(r'\d', word)]
    # 分词
    words = word_tokenize(text)
    # 保留关键词
    keywords = [word for word in words if nlp.word.is_stop(word) or nlp.word.is_punct(word)]
    return''.join(keywords)

def classify(text):
    # 分类
    doc = nlp(text)
    for entity in doc.ents:
        if entity.label_ in ('R', 'F'):
            return entity.text.lower()
    return ''

def extract_info(text):
    # 提取信息
    doc = nlp(text)
    for entity in doc.ents:
        if entity.label_ in ('R', 'F'):
            return entity.text.lower()
    return ''

preprocessed_text = preprocess(text)
classified_text = classify(preprocessed_text)
info = extract_info(classified_text)

# 输出原始文本
print(text)
print('经过预处理后的文本：', preprocessed_text)
print('经过分类后的文本：', classified_text)
print('提取到的信息：', info)
```
