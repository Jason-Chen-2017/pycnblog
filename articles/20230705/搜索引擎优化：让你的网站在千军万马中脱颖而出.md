// Google Algorithm
public class GoogleAlgorithm {
    public static int[] getPageRank(int url, int alpha=0.85, int beta=0.4, int iterations=100, int n=20, int startPage=1) {
        int sum = 0;
        int cnt = 0;
        while (cnt < url == 0) {
            sum = sum * beta + (1-beta) * cnt;
            cnt = cnt + 1;
        }
        int maxVal = (int) Math.sqrt(sum);
        int[] result = new int[url];
        for (int i = 0; i < url; i++)
            result[i] = (int) Math.exp(-alpha * result[i]);
        int minVal = (int) Math.min(1, maxVal);
        for (int i = 0; i < url; i++) {
            result[i] = minVal * Math.pow(result[i], -1) + result[i];
            minVal = minVal * minVal;
        }
        return result;
    }
