import numpy as np
from sklearn.cluster import KMeans

def user_interest_clustering(data, k):
    # 初始化 K-Means 模型
    kmeans = KMeans(n_clusters=k, random_state=0)
    # 训练模型
    kmeans.fit(data)
    # 获取聚类结果
    clusters = kmeans.predict(data)
    # 输出聚类结果
    print("User Interest Clusters:", clusters)

# 示例数据

import tensorflow as tf

def attention Mechanism(inputs, hidden_size):
    # 输入层和隐藏层的尺寸必须相同
    query = inputs
    key = inputs
    value = inputs

    # 计算注意力得分
    attention_scores = tf.matmul(query, key, transpose_b=True) / (hidden_size ** 0.5)

    # 应用 softmax 函数得到注意力权重
    attention_weights = tf.nn.softmax(attention_scores)

    # 计算注意力输出
    attention_output = tf.matmul(attention_weights, value)

    return attention_output

# 示例数据
