```python

# 删除重复数据
df.drop_duplicates(inplace=True)

# 填补缺失值
df.fillna(df.mean(), inplace=True)

# 删除异常值
q1 = df['age'].quantile(0.25)
q3 = df['age'].quantile(0.75)
iqr = q3 - q1
df = df[~((df['age'] < (q1 - 1.5 * iqr)) | (df['age'] > (q3 + 1.5 * iqr)))]
```
