```python
import json

# 数据库连接

c = conn.cursor()

# 创建设备表
c.execute('''CREATE TABLE IF NOT EXISTS devices (device_id TEXT PRIMARY KEY, device_data TEXT)''')

# 插入设备数据
def insert_device_data(device_id, device_data):
    c.execute("INSERT INTO devices (device_id, device_data) VALUES (?, ?)", (device_id, json.dumps(device_data)))
    conn.commit()

# 获取设备数据
def get_device_data(device_id):
    c.execute("SELECT device_data FROM devices WHERE device_id=?", (device_id,))
    result = c.fetchone()
    if result:
        return json.loads(result[0])
    else:
        return None

# 关闭数据库连接
conn.close()
```
