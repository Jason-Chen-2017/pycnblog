        uint id;

    def __init__(self, state_size, action_size):
        super(DRLModel, self).__init__()
        self.fc1 = tf.keras.layers.Dense(64, activation='relu', input_shape=(state_size,))
        self.fc2 = tf.keras.layers.Dense(64, activation='relu')
