```python
def collaborative_filtering(train_data, user_id, k=10):
    # 计算用户相似度矩阵
    similarity_matrix = compute_similarity_matrix(train_data)

    # 获取目标用户与其他用户的相似度排名
    similar_users = sorted(similarity_matrix[user_id].items(), key=lambda x: x[1], reverse=True)[:k]

    # 计算推荐列表
    recommended_items = []
    for user, similarity in similar_users:
        for item in train_data[user]:
            if item not in train_data[user_id]:
                recommended_items.append(item)

    return recommended_items
```
