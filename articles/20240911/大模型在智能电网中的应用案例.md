```python
def energy_optimization(distributed_energy, storage_system, demand):
    # 初始化动态规划表
    dp = [[0] * (len(demand) + 1) for _ in range(len(distributed_energy) + 1)]

    # 动态规划过程
    for i in range(1, len(distributed_energy) + 1):
        for j in range(1, len(demand) + 1):
            if distributed_energy[i - 1] >= demand[j - 1]:
                dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - 1] + distributed_energy[i - 1])
            else:
                dp[i][j] = dp[i - 1][j]

    # 返回最优解
    return dp[len(distributed_energy)][len(demand)]
```
