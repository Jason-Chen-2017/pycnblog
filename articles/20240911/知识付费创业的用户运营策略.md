#### 2. 算法编程题

**问题1：** 请使用 Python 编写一个基于 K-均值算法的用户聚类程序，实现对用户数据的聚类。

**答案：** 
```python
import numpy as np

def kmeans(data, k, max_iter):
    centroids = data[np.random.choice(data.shape[0], k, replace=False)]
    for i in range(max_iter):
        # 计算距离
        distances = np.linalg.norm(data - centroids, axis=1)
        # 分配簇
        clusters = np.argmin(distances, axis=1)
        # 更新中心点
        centroids = np.array([data[clusters == j].mean(axis=0) for j in range(k)])
    return centroids

# 测试
data = np.random.rand(100, 2)
centroids = kmeans(data, 3, 100)
print(centroids)
```

**问题2：** 请使用 Python 编写一个基于矩阵分解的协同过滤算法，实现用户和物品的推荐。

**答案：**
```python
import numpy as np

def matrix_factorization(R, K, iter_num=5, lambda_=0.1):
    N, M = R.shape
    # 初始化用户和物品的潜在特征矩阵
    U = np.random.rand(N, K)
    V = np.random.rand(M, K)
    
    for _ in range(iter_num):
        # 更新用户特征
        for i in range(N):
            for k in range(K):
                e = R[i] - np.dot(U[i], V[:, k])
                U[i, k] = U[i, k] + lambda_ * (U[i, k] - e / np.sqrt(np.sum(U[i]**2) * np.sum(V[:, k]**2)))
        
        # 更新物品特征
        for j in range(M):
            for k in range(K):
                e = R[:, j] - np.dot(U[:, k], V[j, k])
                V[j, k] = V[j, k] + lambda_ * (V[j, k] - e / np.sqrt(np.sum(U[:, k]**2) * np.sum(V[j]**2)))
    
    # 预测评分
    pred = np.dot(U, V.T)
    return pred

# 测试
R = np.array([[5, 3, 0, 1],
              [4, 0, 0, 1],
              [1, 1, 0, 5],
              [1, 0, 0, 4],
              [0, 1, 5, 4]])
pred = matrix_factorization(R, 2)
print(pred)
```
