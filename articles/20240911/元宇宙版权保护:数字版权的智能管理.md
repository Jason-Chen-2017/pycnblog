### 3. 数字版权保护的智能合约设计

**题目：** 请设计一个简单的数字版权保护的智能合约。

**答案：** 下面是一个简单的数字版权保护的智能合约设计：

```solidity
pragma solidity ^0.8.0;

contract DigitalRightsManagement {
    // 数字内容ID与版权所有者映射
    mapping(uint256 => address) public contentOwners;

    // 许可证结构
    struct License {
        address licensee;
        uint256 validityStart;
        uint256 validityEnd;
        bool active;
    }

    // 数字内容ID与许可证映射
    mapping(uint256 => License[]) public licenses;

    // 事件
    event LicenseGranted(uint256 contentId, address licensee, uint256 validityStart, uint256 validityEnd);
    event LicenseRevoked(uint256 contentId, address licensee);

    // 授予许可证
    function grantLicense(uint256 contentId, address licensee, uint256 validityStart, uint256 validityEnd) external {
        require(contentOwners[contentId] == msg.sender, "Only the content owner can grant licenses");
        require(validityStart < validityEnd, "Invalid validity period");

        licenses[contentId].push(License({
            licensee: licensee,
            validityStart: validityStart,
            validityEnd: validityEnd,
            active: true
        }));

        emit LicenseGranted(contentId, licensee, validityStart, validityEnd);
    }

    // 撤销许可证
    function revokeLicense(uint256 contentId, address licensee) external {
        require(contentOwners[contentId] == msg.sender, "Only the content owner can revoke licenses");

        for (uint256 i = 0; i < licenses[contentId].length; i++) {
            if (licenses[contentId][i].licensee == licensee) {
                licenses[contentId][i].active = false;
                emit LicenseRevoked(contentId, licensee);
                return;
            }
        }
    }

    // 检查内容许可状态
    function checkLicense(uint256 contentId, address licensee) external view returns (bool) {
        for (uint256 i = 0; i < licenses[contentId].length; i++) {
            if (licenses[contentId][i].licensee == licensee && licenses[contentId][i].active && block.timestamp >= licenses[contentId][i].validityStart && block.timestamp <= licenses[contentId][i].validityEnd) {
                return true;
            }
        }
        return false;
    }
}
```

**解析：** 此智能合约通过映射和结构体实现了数字内容的版权管理，包括许可证的授予、撤销和检查功能。智能合约的设计确保了版权的自动化和透明化。
