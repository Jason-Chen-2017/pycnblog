#### 1. 马尔可夫链模拟

**题目：** 编写一个程序，模拟一个简单的马尔可夫链，并计算状态转移概率矩阵。

**答案：**

```python
import numpy as np

def markov_chain(transitions, states, steps):
    # 初始化状态转移矩阵
    transition_matrix = np.zeros((states, states))
    for i in range(states):
        for j in range(states):
            transition_matrix[i][j] = transitions[i][j]
    
    # 初始化初始状态
    initial_state = np.random.choice(states)
    current_state = initial_state
    
    # 模拟马尔可夫链
    states_visited = [current_state]
    for _ in range(steps - 1):
        next_state = np.random.choice(states, p=transition_matrix[current_state])
        current_state = next_state
        states_visited.append(current_state)
    
    # 输出状态转移概率矩阵
    print("State Transition Matrix:")
    print(transition_matrix)
    
    # 输出状态访问次数
    state_counts = np.zeros(states)
    for state in states_visited:
        state_counts[state] += 1
    print("State Counts:", state_counts)

# 测试
transitions = [
    [0.5, 0.5],
    [0.2, 0.8]
]
states = 2
steps = 10
markov_chain(transitions, states, steps)
```

import numpy as np

def mcmc_sampler(f, initial_state, num_steps, proposal_stddev):
    # 初始化 MCMC 模拟
    states = [initial_state]
    for _ in range(num_steps):
        # 生成建议状态
        proposal = np.random.normal(states[-1], proposal_stddev)
        # 计算接受概率
        acceptance_probability = min(1, f(proposal) / f(states[-1]))
        # 更新状态
        if np.random.uniform(0, 1) < acceptance_probability:
            states.append(proposal)
        else:
            states.append(states[-1])
    return states
