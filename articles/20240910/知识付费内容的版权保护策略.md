```python
import numpy as np
import cv2

def embed_watermark(image, watermark):
    # 将水印图像缩放到与原始图像相同大小
    watermark = cv2.resize(watermark, image.shape[:2][::-1])

    # 将水印图像转换为灰度图像
    watermark_gray = cv2.cvtColor(watermark, cv2.COLOR_BGR2GRAY)

    # 对水印图像进行二值化处理
    _, watermark_binary = cv2.threshold(watermark_gray, 128, 255, cv2.THRESH_BINARY)

    # 将水印图像嵌入到原始图像的像素中
    for i in range(image.shape[0]):
        for j in range(image.shape[1]):
            if watermark_binary[i, j] == 255:
                image[i, j] = 255

    return image
```
