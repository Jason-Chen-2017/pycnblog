        self.production_scale = 1000
        self.cost_structure = [500, 300, 200]

    def adjust_production(self, economic_state):
        if economic_state == 'up':

            self.cost_structure[0] -= 50
        elif economic_state == 'down':
            self.production_scale -= 100

company = Company('ABC Company')
company.adjust_production('up')
