**实例代码：**

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract SupplyChain {
    // 存储商品信息
    struct Product {
        string id;
        string producer;
        string productionDate;
        string transportRoute;
    }

    // 商品列表
    mapping(string => Product) public products;

    // 上链商品
    function addProduct(string memory id, string memory producer, string memory productionDate, string memory transportRoute) public {
        products[id] = Product(id, producer, productionDate, transportRoute);
    }

    // 查询商品信息
    function getProduct(string memory id) public view returns (string memory, string memory, string memory, string memory) {
        Product memory product = products[id];
        return (product.id, product.producer, product.productionDate, product.transportRoute);
    }
}
```

```solidity
pragma solidity ^0.8.0;
```

```solidity
contract SupplyChainCollaboration {
    // 存储订单信息
    struct Order {
        string id;
        string supplier;
        string buyer;
        string status;
    }

    // 订单列表
    mapping(string => Order) public orders;
```

```solidity
    // 上链订单
    function addOrder(string memory id, string memory supplier, string memory buyer, string memory status) public {
        orders[id] = Order(id, supplier, buyer, status);
    }

    // 查询订单信息
    function getOrder(string memory id) public view returns (string memory, string memory, string memory, string memory) {

        return (order.id, order.supplier, order.buyer, order.status);
    }
}
```
