   func bubbleSort(arr []int) {
       n := len(arr)
       for i := 0; i < n - 1; i++ {
           for j := 0; j < n - 1 - i; j++ {
               if arr[j] > arr[j+1] {
                   arr[j], arr[j+1] = arr[j+1], arr[j]
               }
           }
       }
   }

   ```go
   func binarySearch(arr []int, target int) int {
       low, high := 0, len(arr) - 1
       for low <= high {
           mid := (low + high) / 2
           if arr[mid] == target {
               return mid
           } else if arr[mid] < target {
               low = mid + 1
           } else {
               high = mid - 1
           }
       }
       return -1
   }
   ```

   ```go
   type Cache struct {
       data map[int]int
       capacity int
   }

   func NewCache(capacity int) *Cache {
       return &Cache{
           data: make(map[int]int),
           capacity: capacity,
       }
   }

   func (c *Cache) Set(key int, value int) {
       if _, ok := c.data[key]; ok {
           delete(c.data, key)
       }
       c.data[key] = value
       if len(c.data) > c.capacity {
           oldestKey := c.data[c.capacity]
           delete(c.data, oldestKey)
       }
   }

   func (c *Cache) Get(key int) int {
       if value, ok := c.data[key]; ok {
           return value
       }
       return -1
   }
   ```

### 4. 编程实战题

**问题：** 请解决以下编程实战题，并简要说明解题思路。

**题目：** 实现一个函数，判断一个字符串是否是回文。

```go
func isPalindrome(s string) bool {
    runes := []rune(s)
    left, right := 0, len(runes) - 1
    for left < right {
        if runes[left] != runes[right] {
            return false
        }
        left++
        right--
    }
    return true
}
```

**解题思路：** 使用双指针法，分别从字符串的两端开始遍历，比较对应的字符是否相同。如果遇到不同的字符，直接返回 false。遍历完成后，返回 true。
