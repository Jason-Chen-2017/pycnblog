#### 6. 设计一个简单的投票系统，实现投票和统计功能。

**题目：** 设计一个投票系统，要求能够接收用户投票并统计最终结果。

**答案：**
```python
class VotingSystem:
    def __init__(self):
        self.votes = {}

    def vote(self, candidate):
        if candidate in self.votes:
            self.votes[candidate] += 1
        else:
            self.votes[candidate] = 1

    def get_results(self):
        max_votes = 0
        most_voted_candidate = None
        for candidate, votes in self.votes.items():
            if votes > max_votes:
                max_votes = votes
                most_voted_candidate = candidate
        return most_voted_candidate, max_votes

# 测试
system = VotingSystem()
system.vote("Alice")
system.vote("Bob")
system.vote("Alice")
print(system.get_results())  # 输出 ('Alice', 2)
```

#### 7. 实现一个文本分类器，使用人类计算辅助标注数据。

**题目：** 实现一个基于朴素贝叶斯算法的文本分类器，其中部分数据标注由人类计算完成。

**答案：**
```python
from collections import defaultdict
from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords

def tokenize(text):
    tokens = word_tokenize(text.lower())
    return [token for token in tokens if token not in stopwords.words('english')]

class TextClassifier:
    def __init__(self):
        self词汇表 = defaultdict(int)
        self文档总数 = 0
        self类别概率 = defaultdict(int)
        self条件概率 = defaultdict(int)

    def train(self, data):
        for 文档，类别 in data:
            self文档总数 += 1
            self类别概率[类别] += 1
            tokens = tokenize(文档)
            for token in tokens:
                self词汇表[token] += 1
                self条件概率[(类别, token)] += 1

    def predict(self, 文档):
        tokens = tokenize(文档)
        max_probability = 0
        predicted_category = None
        for 文档类别，类别概率 in self类别概率.items():
            probability = 1
            for token in tokens:
                if (文档类别，token) in self条件概率:
                    probability *= (self条件概率[(文档类别，token)] / self词汇表[token])
                else:
                    probability *= (1 / self词汇表[token])
            probability *=类别概率
            if probability > max_probability:
                max_probability = probability
                predicted_category = 文档类别
        return predicted_category

# 测试
data = [
    ("I love this book", "positive"),
    ("This is a great movie", "positive"),
    ("I hate this product", "negative"),
    ("I am not satisfied with the service", "negative"),
]
classifier = TextClassifier()
classifier.train(data)
print(classifier.predict("I love this movie"))  # 输出 'positive'
```
