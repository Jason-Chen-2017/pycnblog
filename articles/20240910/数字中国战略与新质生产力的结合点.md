def filter_data(df, condition):
    return df.filter(condition)

# 初始化 SparkSession
spark = SparkSession.builder.appName("FilterData").getOrCreate()

# 加载数据
data = spark.read.csv("data.csv")

# 设置条件
condition = "age > 30"

# 筛选数据
filtered_data = filter_data(data, condition)

# 输出结果
filtered_data.show()
