```python
import torch
from torchvision import models, transforms
from torch.nn import functional as F

class MultiModalModel(torch.nn.Module):
    def __init__(self):
        super(MultiModalModel, self).__init__()
        self.conv = models.resnet50(pretrained=True)
        self.rnn = torch.nn.GRU(input_size=2048, hidden_size=512, num_layers=2, batch_first=True)
        self.fc = torch.nn.Linear(512, 10)

    def forward(self, x, y):
        x = self.conv(x)
        x = x.view(x.size(0), -1)
        y = self.rnn(y)
        x = torch.cat((x, y), 1)
        x = self.fc(x)
        return x

model = MultiModalModel()
input_image = torch.randn(4, 3, 224, 224)
input_text = torch.randn(4, 20, 300)
output = model(input_image, input_text)
```
