def calculate_cooperation_score(team_members):
    total_communication_quality = 0
    for member in team_members:
        total_communication_quality += member.communication_quality
    return total_communication_quality / len(team_members)

team_members = [
    {"name": "Alice", "communication_quality": 8},
    {"name": "Bob", "communication_quality": 7},
    {"name": "Charlie", "communication_quality": 9},
]

cooperation_score = calculate_cooperation_score(team_members)
print("Team cooperation score:", cooperation_score)
