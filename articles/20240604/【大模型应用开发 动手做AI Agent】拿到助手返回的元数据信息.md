import os

    return creation_time

    creator = os.getlogin()
    return creator

    data_type = type(data)
    return data_type

# 获取数据来源

    source = urllib.parse.urlunparse(data["url"])
    return source

    permission = os.access(data["path"], os.R_OK)
    return permission
