```python
import numpy as np
from sklearn.cluster import KMeans

# 数据
data = np.random.rand(1000, 3)
# 分割数据
kmeans = KMeans(n_clusters=10)
kmeans.fit(data)
data_split = [data[kmeans.labels_ == i] for i in range(10)]
```
