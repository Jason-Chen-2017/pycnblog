```python
import torch
import torch.nn as nn
import torch.optim as optim

class MetaLearner(nn.Module):
    def __init__(self):
        super(MetaLearner, self).__init__()
        self.model = nn.Sequential(
            nn.Linear(100, 50),
            nn.ReLU(),
            nn.Linear(50, 10)
        )

    def forward(self, x):
        return self.model(x)

def train(model, data, target, optimizer, loss_func):
    model.zero_grad()
    output = model(data)
    loss = loss_func(output, target)
    loss.backward()
    optimizer.step()

optimizer = optim.Adam(model.parameters())
for epoch in range(100):
    train(model, data, target, optimizer, loss_func)
```
