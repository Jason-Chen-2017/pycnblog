```python
import numpy as np
from scipy.optimize import minimize

# 航班调度数据
n = 5
costs = np.array([10, 20, 30, 40, 50])
constraints = np.array([1, 1, 1, 0, 0])

# 目标函数
def objective(x):
    return np.sum(costs * x)

# 约束条件
def constraint(x):
    return np.sum(x) - 1

# 调度方案
x0 = np.array([0.2, 0.3, 0.4, 0.1, 0])

# 求解
result = minimize(objective, x0, method='SLSQP', constraints={'type': 'eq', 'fun': constraint})

print("最优调度方案:", result.x)
print("最优成本:", result.fun)
```
