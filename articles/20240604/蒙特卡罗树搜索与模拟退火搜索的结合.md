import numpy as np

def monte_carlo_tree_search(S, A, T, R, s0, max_depth=100):
    root = Node(s0, None, None, 1)
    for depth in range(max_depth):
        node = root
        s = s0
        for _ in range(depth):
            a = np.random.choice(A, p=T[s, :])
            s = np.random.choice(S, p=T[s, a])
            node = node.children[a]
        node.expand(A, T, R)
    return root.best_action

class Node:
    def __init__(self, state, parent, action, N):
        self.state = state
        self.parent = parent
        self.action = action
        self.N = N
        self.children = {}
        self.Q = 0

    def expand(self, A, T, R):
        for a in A:
            if a not in self.children:
                self.children[a] = Node(np.random.choice(T[self.state, a]), self, a, 0)
        self.Q = np.mean([self.children[a].Q for a in self.children])

    def best_action(self):
        return max(self.children.items(), key=lambda x: x[1].Q / x[1].N)[0]
