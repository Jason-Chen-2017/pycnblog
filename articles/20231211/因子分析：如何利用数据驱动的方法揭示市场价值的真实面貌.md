import pandas as pd
from sklearn.linear_model import LinearRegression

factor_data = pd.read_csv('factor_data.csv')

factor_data['factor_change'] = factor_data['factor'].pct_change()

# 对因子分析模型进行训练
model = LinearRegression()
model.fit(stock_data[['price_change', 'factor_change']], stock_data['price_change'])

# 对因子分析模型进行评估
score = model.score(stock_data[['price_change', 'factor_change']], stock_data['price_change'])
print('模型准确性：', score)

# 对因子分析模型进行优化
opt_model = LinearRegression(fit_intercept=False)
opt_model.fit(stock_data[['price_change', 'factor_change']], stock_data['price_change'])

# 对因子分析模型进行应用
pred_price_change = opt_model.predict(stock_data[['price_change', 'factor_change']])
print('预测结果：', pred_price_change)
