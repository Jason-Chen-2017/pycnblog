import matplotlib.pyplot as plt

# 生成光信号
t = np.linspace(0, 1, 100)
I0 = 1000
alpha = 0.1
I = I0 * np.exp(-alpha * t)

# 光电子转换器的转换
R = 1000
V = R * I

# 光电子成像系统的成像
h = np.ones((5, 5)) / 25
f = np.fft.fft2(I)
f_shifted = np.fft.fftshift(f)
f_conj = np.conj(f_shifted)
h_shifted = np.fft.fftshift(h)
h_conj = np.conj(h_shifted)
g = np.multiply(f_conj, h_conj)
f_final = np.fft.ifftshift(g)
f_final = np.fft.ifft2(f_final)

# 绘制成像结果
plt.imshow(np.abs(f_final), cmap='gray')
