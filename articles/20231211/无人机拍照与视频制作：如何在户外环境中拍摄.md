```python
import socket

class DataTransfer:
    def __init__(self, host, port):
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.socket.connect((host, port))

    def send(self, data):
        self.socket.send(data)

    def receive(self):
        data = self.socket.recv(1024)
        return data

    def close(self):
        self.socket.close()

```
