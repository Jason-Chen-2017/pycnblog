```python
import hashlib
import time

# 创建区块链
def create_blockchain():
    blockchain = []
    previous_hash = '0' * 64
    for i in range(100):
        nonce = 0
        while True:
            hash = hashlib.sha256(str(nonce).encode('utf-8')).hexdigest()
            if hash[:4] == previous_hash[:4]:
                break
            nonce += 1
        blockchain.append({
            'index': i,
            'timestamp': time.time(),
            'transactions': [],
            'previous_hash': previous_hash,
            'nonce': nonce,
            'hash': hash
        })
        previous_hash = hash
    return blockchain

# 验证交易
def verify_transaction(transaction):
    sender_wallet = create_wallet()
    receiver_wallet = create_wallet()
    sender_wallet.balance -= transaction['amount']
    receiver_wallet.balance += transaction['amount']
    if sender_wallet.balance < 0 or receiver_wallet.balance < 0:
        return False
    return True
```
