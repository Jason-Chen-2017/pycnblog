def connected_components(graph):
    visited = set()
    components = []
    for node in graph.nodes():
        if node not in visited:
            component = []
            dfs(graph, node, component)
            components.append(component)
    return components

def dfs(graph, node, component):
    visited.add(node)
    component.append(node)
    for neighbor in graph.neighbors(node):
        if neighbor not in visited:
            dfs(graph, neighbor, component)
```

def traffic_analysis(graph, traffic):
    result = []
    for node in graph.nodes():
        traffic_node = traffic[node]
        neighbors = graph.neighbors(node)
        for neighbor in neighbors:
            traffic_neighbor = traffic[neighbor]
            result.append((node, neighbor, traffic_node, traffic_neighbor))
    return result

def attack_detection(graph, attack_patterns):
    result = []
    for pattern in attack_patterns:
        nodes = pattern.nodes()
        for node in nodes:
            if node not in graph.nodes():
                break
        else:
            for i in range(len(nodes) - 1):
                edge = (nodes[i], nodes[i + 1])
                if edge not in graph.edges():
                    break
            else:
                result.append(pattern)
    return result
```
