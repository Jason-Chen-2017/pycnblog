#include <ctime>

    int samples[100000];

        samples[i] = rand() % 100;
    }

    // 计算属性值
    double mean = 0;
    for (int i = 0; i < 100000; i++) {
        mean += samples[i];
    }
    mean /= 100000;

    double std = 0;
    for (int i = 0; i < 100000; i++) {
        std += pow(samples[i] - mean, 2);
    }
    std = sqrt(std / 100000);

    std::cout << "Standard Deviation: " << std << std::endl;

    return 0;
}
