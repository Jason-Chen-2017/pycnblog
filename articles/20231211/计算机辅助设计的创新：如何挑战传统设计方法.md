```python
from scipy.optimize import minimize

# 定义目标函数
def objective(x):
    return x[0]**2 + x[1]**2 - 10*x[0] + 10*x[1]

# 定义约束条件
def constraint(x):
    return x[0]**2 + x[1]**2 - 400

# 初始化参数
x0 = [1.0, 1.0]

# 设置优化算法参数
options = {'disp': True}

# 执行优化
res = minimize(objective, x0, method='SLSQP', options=options, constraints={'type': 'eq', 'fun': constraint})

# 输出结果
print(res.x)
```
