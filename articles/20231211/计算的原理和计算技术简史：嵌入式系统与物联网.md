def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    else:
        pivot = arr[0]
        less = [i for i in arr[1:] if i <= pivot]
        greater = [i for i in arr[1:] if i > pivot]

def binary_search(arr, target):
    left = 0
    right = len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

arr = [1, 2, 3, 4, 5]
target = 3
print(binary_search(arr, target))
```

def multiply(a, b):
    if len(a) == 1 or len(b) == 1:
        return [a[0] * b[0]]
    else:
        a1 = a[:len(a) // 2]
        a2 = a[len(a) // 2:]
        b1 = b[:len(b) // 2]
        b2 = b[len(b) // 2:]
        return multiply(a1, b1) + multiply(a2, b2)

def fibonacci(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        dp = [0] * (n + 1)
        dp[0] = 0
        dp[1] = 1
        for i in range(2, n + 1):
            dp[i] = dp[i - 1] + dp[i - 2]
        return dp[n]

n = 10
print(fibonacci(n))
```

5. 贪心算法的代码实例：

```python
def coin_change(coins, amount):
    dp = [float("inf")] * (amount + 1)
    dp[0] = 0
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] = min(dp[i], dp[i - coin] + 1)
    return dp[amount]

coins = [1, 2, 5]
amount = 11
print(coin_change(coins, amount))
```

def subset_sum(arr, target):
    def backtrack(arr, target, start, end, path):
        if target == 0:
            return True
        if target < 0:
            return False
        for i in range(start, end + 1):
            if arr[i] > target:
                continue
            if backtrack(arr, target - arr[i], i + 1, end, path + [arr[i]]):
                return True
        return False

    arr = sorted(arr)
    for i in range(len(arr)):
        if backtrack(arr, target, i, len(arr) - 1, []):
            return True
    return False

arr = [2, 3, 5, 7]
target = 8
print(subset_sum(arr, target))
