import numpy as np

class WhaleOptimization:
    def __init__(self, num_whales, search_space, max_iter):
        self.num_whales = num_whales
        self.search_space = search_space
        self.max_iter = max_iter
        self.positions = np.random.uniform(search_space[0], search_space[1], num_whales)
        self.velocities = np.zeros(num_whales)
        self.best_position = self.positions[np.argmin(self.fitness(self.positions))]

    def fitness(self, positions):
        # 计算鲸鱼群的适应度
        fitness = np.zeros(self.num_whales)
        for i in range(self.num_whales):
            fitness[i] = self.objective_function(positions[i])
        return fitness

    def objective_function(self, position):
        # 定义目标函数
        return position**2

    def update_positions(self):
        for i in range(self.num_whales):
            r1, r2 = np.random.rand(2)
            c1, c2 = np.random.rand(2, self.num_whales)
            self.velocities[i] = (2 * r1 * self.velocities[i] + c1 * self.positions[i] - c2 * self.best_position)
            self.positions[i] += self.velocities[i]

    def update_best_position(self):
        if np.min(self.positions) < self.best_position:
            self.best_position = np.min(self.positions)

    def run(self):
        for _ in range(self.max_iter):
            self.update_positions()
            self.update_best_position()
        return self.best_position
