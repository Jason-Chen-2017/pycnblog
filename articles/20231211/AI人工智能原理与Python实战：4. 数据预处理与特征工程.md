import numpy as np

# 数据清洗
def data_cleaning(data, noise, fill):
    data_cleaned = data - noise
    data_cleaned = np.where(data_cleaned == 0, fill, data_cleaned)
    data_cleaned = (data_cleaned - np.min(data_cleaned)) / (np.max(data_cleaned) - np.min(data_cleaned))
    return data_cleaned
