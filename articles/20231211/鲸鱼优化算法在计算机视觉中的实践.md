import numpy as np

class WhaleOptimization:
    def __init__(self, num_whales, search_space, max_iterations):
        self.num_whales = num_whales
        self.search_space = search_space
        self.max_iterations = max_iterations
        self.whales = np.random.uniform(self.search_space[0], self.search_space[1], self.num_whales)
        self.best_whale = np.min(self.whales)

    def update_whales(self):
        for i in range(self.num_whales):
            A = 2 * np.random.rand() - 1
            C = 2 * np.random.rand()
            D = 2 * np.random.rand() - 1
            rand = np.random.rand()
            self.whales[i] = self.whales[i] + A * D * self.whales[i] + C * rand
            if np.random.rand() < 0.5:
                self.whales[i] = self.search_space[0] + 0.5 * (self.search_space[1] - self.search_space[0]) * np.random.rand(2)

    def run(self):
        for t in range(self.max_iterations):
            self.update_whales()
            self.best_whale = np.min(self.whales)
            if np.random.rand() < 0.01:
                self.whales = np.random.uniform(self.search_space[0], self.search_space[1], self.num_whales)
        return self.best_whale
