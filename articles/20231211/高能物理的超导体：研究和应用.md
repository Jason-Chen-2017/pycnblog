    if type_of_superconductor == 'typeI':

        return 2

def superconductor_performance(type_of_superconductor, temperature):
    performance = superconductor_spectrum(temperature) * superconductor_temperature(type_of_superconductor)
    return performance

# 主程序

    # 选择超导体类型
    type_of_superconductor = 'typeII'
    # 选择测试温度
