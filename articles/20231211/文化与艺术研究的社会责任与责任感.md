X_train = X_train.toarray()

tfidf_transformer = TfidfTransformer()

# 文化与艺术数据分析后的矩阵
X_train_tfidf = X_train_tfidf.toarray()
X_test_tfidf = X_test_tfidf.toarray()

# 文化与艺术数据分析算法
classifier = MultinomialNB()
