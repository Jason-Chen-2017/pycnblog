1. Python：Python是一种非常流行的编程语言，可以用于文本分析、机器学习和数据挖掘等任务。以下是一个Python代码实例，用于文本分析：

```python
import re
import nltk
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize

# 读取文本
with open('text.txt', 'r', encoding='utf-8') as f:
    text = f.read()

# 去除标点符号
text = re.sub(r'[^\w\s]', '', text)

# 分词
tokens = word_tokenize(text)

# 去除停用词
stop_words = set(stopwords.words('english'))
tokens = [word for word in tokens if word.lower() not in stop_words]

# 词频统计
word_freq = nltk.FreqDist(tokens)

# 打印词频统计结果
print(word_freq)
```

3. Java：Java是一种非常流行的编程语言，可以用于文本分析、机器学习和数据挖掘等任务。以下是一个Java代码实例，用于数据挖掘：

```java
import weka.core.Instances;
import weka.core.converters.ConverterUtils.DataSource;
import weka.classifiers.trees.J48;

public class Main {
    public static void main(String[] args) throws Exception {
        // 加载数据
        DataSource source = new DataSource("data.arff");
        Instances data = source.getDataSet();

        // 设置训练集和测试集
        data.randomize(new java.util.Random(1));
        int trainSize = (int) (0.8 * data.numInstances());
        Instances train = new Instances(data, 0, trainSize);
        Instances test = new Instances(data, trainSize, data.numInstances() - trainSize);

        // 训练模型
        J48 j48 = new J48();
        j48.buildClassifier(train);

        // 预测结果
        int correct = 0;
        for (int i = 0; i < test.numInstances(); i++) {
            double prediction = j48.classifyInstance(test.instance(i));
            if (prediction == test.classAttribute().value((int) test.instance(i).classValue())) {
                correct++;
            }
        }
        System.out.println("Accuracy: " + (double) correct / test.numInstances());
    }
}
```
