import hashlib
import json
from time import time

# 创建区块
def create_block(index, timestamp, data):
    block = {
        'index': index,
        'timestamp': timestamp,
        'data': data,
        'previous_hash': get_previous_hash(block),
        'hash': get_hash(block)
    }
    return block

# 获取区块链
def get_blockchain():
    blockchain = []
    with open('blockchain.json', 'r') as f:
        blockchain = json.load(f)
    return blockchain

# 获取区块链的哈希值
def get_hash(block):
    block_string = json.dumps(block, sort_keys=True).encode()
    return hashlib.sha256(block_string).hexdigest()

# 获取前一个区块的哈希值

    previous_block = get_last_block()
    return previous_block['hash']

    blockchain = get_blockchain()
