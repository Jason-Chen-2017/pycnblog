import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# 数据分析算法
def data_analysis(data):
    mean = np.mean(data)
    std = np.std(data)
    return mean, std

# 决策支持系统算法
def decision_support_system(x, a, b, c):
    return (a*x + b)/c

# 信息交流算法
def information_exchange(x, theta):
    return 1/(1+np.exp(-x))

# 数据分析
data = np.random.randn(1000)
mean, std = data_analysis(data)
print("Mean:", mean)
print("Standard Deviation:", std)

# 决策支持系统
x = np.linspace(-10, 10, 100)
a, b, c = 1, 0, 1
y = decision_support_system(x, a, b, c)
plt.plot(x, y)
plt.xlabel('x')
plt.ylabel('y')
plt.title('Decision Support System')
plt.show()

# 信息交流
x = np.linspace(-10, 10, 100)
theta = 0
y = information_exchange(x, theta)
plt.plot(x, y)
plt.xlabel('x')
plt.ylabel('y')
plt.title('Information Exchange')
plt.show()
