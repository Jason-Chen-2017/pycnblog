```python
import numpy as np
from scipy.optimize import leastsq

# 粒子加速算法
def accelerate(energy, mass, c):
    return energy / c

# 粒子检测算法
def detect(distance, radius, angle):
    return (radius * angle) / 2

# 数据处理算法
def process(data, m, b):
    return m * data + b

# 实验数据
data = np.array([1, 2, 3, 4, 5])

# 粒子加速
energy = accelerate(10, 1, 3e8)

# 粒子检测
distance = detect(10, 1, np.deg2rad(45))

# 数据处理
m, b = leastsq(lambda x: np.sum((data - process(data, x[0], x[1])) ** 2), [0, 0])

# 结果解释
print("粒子能量:", energy)
print("粒子距离:", distance)
print("数据趋势:", m, b)
```
