```python
class Counter:
    def __init__(self, initial_value=0):
        self.value = initial_value

    def increment(self):
        self.value += 1

    def decrement(self):
        self.value -= 1

    def get_value(self):
        return self.value

counter = Counter(initial_value=0)
counter.increment()
counter.decrement()
print(counter.get_value())  # 输出: -1
```
