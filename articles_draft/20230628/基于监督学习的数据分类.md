4.3. 核心代码实现

```python
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.svm import SVR
from sklearn.metrics import accuracy_score

# 读取数据
data = pd.read_csv('data.csv')

# 数据预处理
# 将文本数据转换为数字
data['text'] = data['text'].astype('int')
# 去除停用词
stopwords = set(pd.read_csv('stopwords.txt')['text'])
data['text'] = data['text'].apply(lambda x:''.join([word for word in x.split() if word.lower() not in stopwords]))

# 标准化
scaler = StandardScaler()
data[['text', 'label']] = scaler.fit_transform(data[['text', 'label']])

# 将数据分为训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(data.drop(['label'], axis=1), data['label'], test_size=0.2, n_informative_features=1)

# 创建并训练SVR模型
model = SVR(kernel='rbf', C=1.0, epsilon=0.1)
model.fit(X_train, y_train)

# 对测试集进行预测
y_pred = model.predict(X_test)

# 评估模型
accuracy = accuracy_score(y_test, y_pred)
print("Accuracy:", accuracy)

# 应用模型进行预测
test_data = pd.DataFrame({'text': ['Apple', 'Banana', 'Orange', 'Grape'], 'label': [0, 1, 2, 0]})
pred_labels = model.predict(test_data)

print(test_data)
```
