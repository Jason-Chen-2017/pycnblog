import numpy as np
import pandas as pd
import spacy
import knowledge_graph

import random

nlp = spacy.load('en_core_web_sm')

def preprocess(text):
    doc = nlp(text)
    for token in doc.token_list:
        if token.is_stop!= True and token.is_punct!= True:
            token.remove(None)
    doc = doc.unsqueeze(0)
    return doc

def create_knowledge_graph(text):
     entities, relations = [], []
     for token in nlp.vocab:
        if token.is_stop!= True and token.is_punct!= True:
            if token.is_noun() or token.is_verb():
                entities.append(token.text)
                relations.append('HAS_BEEN_SEEN')
            else:
                pass
     knowledges = [{"from": entity, "to": relation} for entity, relation in relations]
     return knowledge_graph.create_graph(entities, relations, '杏仁')

def classify_news_articles(texts):
     knowledge_graph = knowledge_graph.KnowledgeGraph()
     for text in texts:
        doc = preprocess(text)
        doc = doc.expand('POS', ['NOUN', 'VERB'])
        for token in doc:
            if token.is_stop!= True and token.is_punct!= True:
                if token.is_noun():
                    entity = token.text
                    relations = ['HAS_BEEN_SEEN']
                    knowledge_graph.add_entity(entity, relations)
                else:
                    pass
     return knowledge_graph

def main():
     texts = [
        '新闻1', '新闻2', '新闻3', '新闻4', '新闻5'
     ]
     knowledge_graph = classify_news_articles(texts)
     for text, entity, relations in knowledge_graph.items():
        print(f'{text}: {entity}, {relations}')

if __name__ == '__main__':
    main()
