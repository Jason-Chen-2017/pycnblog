```python
import nltk
from nltk.corpus import stopwords

from sklearn.pipeline import Pipeline
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.pipeline import VaderSentimentAnalyzer
from sklearn.metrics import f1_score
nltk.download('vader_lexicon')

def preprocess_text(text):
    # 去除停用词
    text = nltk.corpus.stopwords.words('english') + [text]
    # 分词
    text = word_tokenize(text)
    # 提取特征
    vectorizer = CountVectorizer()
    feature = vectorizer.fit_transform(text)
    # 情感分析
    pipeline = Pipeline([
        ('text-vectorizer', vectorizer),
        ('vader-sentiment-analyzer', VaderSentimentAnalyzer()),
        ('text-classifier', '朴素贝叶斯')
    ])
    data = pipeline.fit_transform(text)
    # 返回特征
    return feature
```
