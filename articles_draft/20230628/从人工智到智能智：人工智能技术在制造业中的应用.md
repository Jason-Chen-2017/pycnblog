import tensorflow as tf
import cv2

# 数据预处理
def preprocess_image(image):
    # 读取图像
    img_array = image.read()
    # 转为灰度
    gray_array = cv2.cvtColor(img_array, cv2.COLOR_BGR2GRAY)
    # 数据归一化
    stretch_factor = 1.2
    new_array = cv2.resize(gray_array, (int(gray_array.shape[1] * stretch_factor), int(gray_array.shape[0] * stretch_factor)))(gray_array)
    # 模糊处理
    blur = cv2.GaussianBlur(new_array, (5, 5), 0)
    # 边缘检测
    edges = cv2.Canny(blur, 50, 200)
    # 数据保存
    return edges

# 模型实现
def create_model():
    model = tf.keras.models.Sequential()
    model.add(tf.keras.layers.Conv2D(32, (3, 3), activation='relu', input_shape=(224, 224, 3)))
    model.add(tf.keras.layers.MaxPooling2D((2, 2)))
    model.add(tf.keras.layers.Conv2D(64, (3, 3), activation='relu'))
    model.add(tf.keras.layers.MaxPooling2D((2, 2)))
    model.add(tf.keras.layers.Flatten())
    model.add(tf.keras.layers.Dense(64, activation='relu'))
    model.add(tf.keras.layers.Dense(10, activation='softmax'))
    model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])
    return model
