    'physics': 0.4,
    'chemistry': 0.2
}

# 推荐算法
def recommend(student_info):
    recommendations = []
    for subject in student_info:
        if student_info[subject] > 0.5:
            recommendations.append(subject)
    return recommendations

# 学习算法
def learn(student_info, recommendations):
    learning_resources = []
    for recommendation in recommendations:
        learning_resources.append(get_learning_resources(recommendation))
    return learning_resources

# 交互算法
def interact(student_info, learning_resources):
    interaction_strategies = []
    for learning_resource in learning_resources:
        interaction_strategies.append(get_interaction_strategies(student_info, learning_resource))
    return interaction_strategies

# 获取学习资源
def get_learning_resources(subject):
    return [f'{subject}_resource_{i}' for i in range(1, 11)]

# 获取交互策略
def get_interaction_strategies(student_info, learning_resource):
    return [f'{learning_resource}_interaction_{i}' for i in range(1, 6)]

# 主函数
def main():
    recommendations = recommend(student_info)
    learning_resources = learn(student_info, recommendations)
    interaction_strategies = interact(student_info, learning_resources)
    print('推荐课程：', recommendations)
    print('学习资源：', learning_resources)
