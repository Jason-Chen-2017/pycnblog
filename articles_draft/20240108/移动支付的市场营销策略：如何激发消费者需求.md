        self.balance += amount

    def make_payment(self, amount):
        if self.balance >= amount:
            self.balance -= amount
            self.transactions.append((amount, "商家"))
        else:
            raise ValueError("Insufficient funds")

    def get_balance(self):
        return self.balance

    def get_transactions(self):
        return self.transactions

# 创建一个移动支付对象

# 向移动支付对象添加钱

# 查看移动支付对象的交易记录
