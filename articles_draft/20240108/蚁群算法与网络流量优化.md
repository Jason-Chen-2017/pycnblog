### 4.1 问题描述

假设我们有一个简单的网络，包括 5 个节点和 7 条边。我们的目标是找到一个最小的流量分配方案，使得网络流量分配得当。

import random

def init_ants():
    ants = []
    for _ in range(100):
        ant = [random.randint(0, 4) for _ in range(5)]
        ants.append(ant)
    return ants

def pheromone_attraction(ant, graph):
    pheromone = 0
    for edge in graph.edges():
        pheromone += graph[edge][0]
    return pheromone / len(graph.edges())

def ant_decision(ant, graph, pheromone_coef, evaporation_coef):
    decision = {}
    for edge in graph.edges():
        decision[edge] = pheromone_attraction(ant, graph) * pheromone_coef
    return decision

    for edge in graph.edges():
        graph[edge][0] *= (1 - evaporation_coef)

    return ant
