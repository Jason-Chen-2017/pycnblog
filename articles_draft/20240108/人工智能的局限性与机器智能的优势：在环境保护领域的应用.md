def sigmoid(x):

def cost_function(X, y, theta):
    m = len(y)
    h = X @ theta
    err = (1 / m) * np.sum((h - y) ** 2)
    return err

def gradient_descent(X, y, theta, alpha, iterations):
    m = len(y)
    for i in range(iterations):
        h = X @ theta
        error = h - y
        theta -= (alpha / m) * X.T @ error
    return theta
