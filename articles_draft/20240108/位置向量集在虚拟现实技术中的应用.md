## 4.具体代码实例和详细解释说明

### 4.1代码实例

```python
import numpy as np

class PositionVectorSet:
    def __init__(self):
        self.positions = []
        self.directions = []

    def add_object(self, position, direction):
        self.positions.append(position)
        self.directions.append(direction)

    def update_position(self, index, new_position):
        self.positions[index] = new_position

    def calculate_distance(self, index1, index2):
        position1 = self.positions[index1]
        position2 = self.positions[index2]
        return np.linalg.norm(position1 - position2)

    def calculate_direction(self, index1, index2):
        position1 = self.positions[index1]
        position2 = self.positions[index2]
        return position1 - position2

# 创建位置向量集
pvs = PositionVectorSet()

# 添加对象
pvs.add_object(np.array([0, 0, 0]), np.array([1, 0, 0]))
pvs.add_object(np.array([1, 0, 0]), np.array([0, 1, 0]))

# 更新对象的位置
pvs.update_position(0, np.array([1, 0, 0]))

# 计算对象之间的距离
distance = pvs.calculate_distance(0, 1)
print("Distance:", distance)

# 计算对象之间的方向
direction = pvs.calculate_direction(0, 1)
print("Direction:", direction)
```
