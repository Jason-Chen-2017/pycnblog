from sklearn.tree import DecisionTreeClassifier

class BedSheetConfig:
    def __init__(self, bed_sheets):
        self.bed_sheets = bed_sheets

    def config_bed_sheets(self):
        X = []
        y = []
        for bed_sheet in self.bed_sheets:
            X.append([bed_sheet.color, bed_sheet.size])
            y.append(bed_sheet.color)

        clf = DecisionTreeClassifier()
        clf.fit(X, y)
        return clf
```

import sqlite3

class BedSheetUsage:
    def __init__(self, db_name):
        self.conn = sqlite3.connect(db_name)
        self.cursor = self.conn.cursor()
        self.cursor.execute('''CREATE TABLE IF NOT EXISTS usage
                              (id INTEGER PRIMARY KEY, bed_sheet_id INTEGER,
                               use_times INTEGER, use_time STRING)''')

    def add_usage(self, bed_sheet_id, use_times, use_time):
        self.cursor.execute('''INSERT INTO usage (bed_sheet_id, use_times, use_time)
                               VALUES (?, ?, ?)''', (bed_sheet_id, use_times, use_time))
        self.conn.commit()

    def get_usages(self):
        self.cursor.execute('''SELECT * FROM usage''')
        return self.cursor.fetchall()
```
