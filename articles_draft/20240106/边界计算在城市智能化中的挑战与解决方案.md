# 读取原始数据
data = pd.read_csv('sensor_data.csv')

# 数据清洗
data_clean = data.dropna()

# 数据压缩
data_compressed = data_clean.groupby('sensor_id').sum()

    anonymized_data = data.replace(data[0], '*')
