# 4.具体代码实例和详细解释说明
在实际应用中，我们可以使用以下代码实例来实现泛化能力：

## 4.1 线性回归模型
```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

# 生成数据
np.random.seed(0)
x = np.random.rand(100, 1)
y = 2 * x + 1 + np.random.randn(100, 1) * 0.1

# 训练模型
model = LinearRegression()
model.fit(x, y)

# 预测
x_test = np.linspace(0, 1, 100)
y_test = model.predict(x_test.reshape(-1, 1))

# 绘图
plt.scatter(x, y, label='原始数据')
plt.plot(x_test, y_test, color='red', label='预测结果')
plt.legend()
plt.show()
```
## 4.2 逻辑回归模型
```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LogisticRegression

# 生成数据
np.random.seed(0)
x = np.random.rand(100, 1)
y = 1 / (1 + np.exp(-2 * x - 1)) + np.random.randn(100, 1) * 0.1
y = np.where(y > 0.5, 1, 0)

# 训练模型
model = LogisticRegression()
model.fit(x, y)

# 预测
x_test = np.linspace(0, 1, 100)
y_test = model.predict(x_test.reshape(-1, 1))

# 绘图
plt.scatter(x, y, label='原始数据')
plt.plot(x_test, y_test, color='red', label='预测结果')
plt.legend()
plt.show()
```
## 4.3 支持向量机模型
```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.svm import SVC

# 生成数据
np.random.seed(0)
x = np.random.rand(100, 2)
y = np.where(x[:, 0] > 0.5, 1, -1) + np.random.randn(100, 1) * 0.1

# 训练模型
model = SVC(kernel='linear')
model.fit(x, y)

# 预测
x_test = np.linspace(-1, 1, 100)
y_test = model.predict(x_test.reshape(-1, 2))

# 绘图
plt.scatter(x[:, 0], x[:, 1], c=y, cmap='viridis')
plt.plot(x_test, -model.dual_coef_[0] * x_test - model.intercept_[0] / model.coef_[0], color='red')
plt.show()
```
