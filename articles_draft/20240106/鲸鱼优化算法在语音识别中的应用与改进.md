1. 初始化鲸鱼群的位置，即将所有鲸鱼的位置随机分配在搜索空间中。
2. 计算每个鲸鱼的适应度，即评估每个鲸鱼在搜索空间中的表现。
3. 找到最佳鲸鱼，即适应度最高的鲸鱼。
4. 更新其他鲸鱼的位置，以逐渐靠近最佳鲸鱼。

import numpy as np
import matplotlib.pyplot as plt

# 初始化鲸鱼群的位置
def init_whale_positions(num_whales, search_space):
    return np.random.uniform(search_space[0], search_space[1], (num_whales, 2))

# 计算鲸鱼之间的距离
def calculate_distance(whale_positions):
    return np.sqrt(np.sum((whale_positions - whale_positions[:, np.argmax(whale_positions, axis=0)][:, np.newaxis])**2, axis=1))

# 更新鲸鱼的位置
def update_whale_positions(whale_positions, best_whale_pos, A, distance):
    return whale_positions - A * distance

# 鲸鱼优化算法主体
def whale_optimization_algorithm(search_space, num_whales, max_iterations):
    whale_positions = init_whale_positions(num_whales, search_space)
    best_whale_pos = whale_positions[:, np.argmax(whale_positions, axis=0)]
    A = 2

    for t in range(max_iterations):
        distance = calculate_distance(whale_positions)
        whale_positions = update_whale_positions(whale_positions, best_whale_pos, A, distance)
        best_whale_pos = whale_positions[:, np.argmax(whale_positions, axis=0)]
        A *= 0.99

    return best_whale_pos

# 测试鲸鱼优化算法
search_space = [0, 100]
num_whales = 10
max_iterations = 100
best_whale_pos = whale_optimization_algorithm(search_space, num_whales, max_iterations)

print("最佳鲸鱼位置：", best_whale_pos)
