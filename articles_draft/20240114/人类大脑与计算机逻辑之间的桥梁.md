def sigmoid(x):

def backpropagation(x, y, theta):
    m = len(y)
    predictions = sigmoid(np.dot(x, theta))
    predictions = np.where(predictions >= 0.5, 1, 0)
    loss = -(y * np.log(predictions) + (1 - y) * np.log(1 - predictions))
    loss = np.sum(loss) / m
    d_theta = np.dot(x.T, (predictions - y)) / m
    return d_theta, loss
