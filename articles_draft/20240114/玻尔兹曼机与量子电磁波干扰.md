```python
import numpy as np

class BoltzmannMachine:
    def __init__(self, N, W, a):
        self.N = N
        self.W = W
        self.a = a
        self.x = np.random.randint(0, 2, size=(N, 1))

    def energy(self):
        E = -np.dot(self.a, self.x) - np.dot(self.W, np.dot(self.x, self.x.T))
        return E

    def probability(self):
        Z = np.prod(1 + np.exp(-self.a - np.dot(self.W, self.x)))
        P = np.exp(-self.energy()) / Z
        return P

    def sample(self):
        P = self.probability()
        u = np.random.rand()
        for i in range(self.N):
            u -= P[i]
            if u <= 0:
                self.x[i, 0] = 1 - self.x[i, 0]
                break

    def train(self, epochs, learning_rate):
        for epoch in range(epochs):
            for i in range(self.N):
                self.sample()
            for i in range(self.N):
                for j in range(i + 1, self.N):
                    W_ij = self.W[i, j]
                    W_ij_new = W_ij + learning_rate * (self.x[i] * self.x[j] - np.mean(self.x[i]) * np.mean(self.x[j]))
                    self.W[i, j] = W_ij_new

# 示例使用
N = 5
W = np.random.rand(N, N)
a = np.random.rand(N)
bm = BoltzmannMachine(N, W, a)
for epoch in range(1000):
    bm.train(1, 0.1)
    print(bm.energy())
```
