```python
import numpy as np

def kernel(x, y):
    return np.dot(x, y.T)

def kernel_matrix(X):
    K = np.zeros((X.shape[0], X.shape[0]))
    for i in range(X.shape[0]):
        for j in range(X.shape[0]):
            K[i, j] = kernel(X[i], X[j])
    return K

def gram_matrix(X):
    G = np.zeros((X.shape[0], X.shape[0]))
    for i in range(X.shape[0]):
        G[i, i] = 1
    return G

def project(K, G, x):
    return np.dot(np.dot(K, G), x)

X = np.array([[1, 2], [3, 4], [5, 6]])
G = gram_matrix(X)
K = kernel_matrix(X)
x = np.array([[7, 8], [9, 10]])
y = project(K, G, x)
print(y)
```
