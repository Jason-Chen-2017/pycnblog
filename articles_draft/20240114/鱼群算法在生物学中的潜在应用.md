class Fish:

        self.y = y
        self.vx = vx
        self.vy = vy

    def update_position(self, dt):
        self.x += self.vx * dt
        self.y += self.vy * dt

    def update_velocity(self, dt, fish_list):
        r = np.zeros(2)
        w = np.zeros(2)
        for f in fish_list:
            if f != self:
                r += f.position - self.position
                w += 1 / np.linalg.norm(f.position - self.position)
        r /= np.linalg.norm(r)
        u = self.alpha * r + self.beta * self.velocity
        self.velocity += u * dt

    def run(self, dt, fish_list, iterations):
        for _ in range(iterations):
            self.update_position(dt)
            self.update_velocity(dt, fish_list)

def main():
    num_fish = 50
    dt = 0.01
    iterations = 1000
