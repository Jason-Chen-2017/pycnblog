```python
import cv2
import numpy as np

def detect_object(image, object_class):
    # Load the pre-trained object detection model
    net = cv2.dnn.readNetFromCaffe('deploy.prototxt', 'model.caffemodl')

    # Pre-process the image
    blob = cv2.dnn.blobFromImage(image, 1.0, (300, 300), (104, 117, 123))
    net.setInput(blob)

    # Perform object detection
    detections = net.forward()

    # Filter the detections by class
    class_id = int(object_class)
    conf_threshold = 0.5
    detections = detections[0, 0, :, :]
    indices = np.where(detections[:, -1] >= conf_threshold)[0].astype('int32')

    for i in indices:
        box = detections[i, 0:4] * np.array([image.shape[1], image.shape[0], image.shape[1], image.shape[0]])
        (startX, startY, endX, endY) = box.astype('int')

        # Draw the bounding box
        cv2.rectangle(image, (startX, startY), (endX, endY), (0, 255, 0), 2)

    return image
```
