# 4.具体代码实例和详细解释说明

## 4.1 弹性资源分配的代码实例

```python
class ResourceManager:
    def __init__(self):
        self.resources = {}

    def add_resource(self, resource_name, resource_value):
        if resource_name not in self.resources:
            self.resources[resource_name] = resource_value
        else:
            self.resources[resource_name] += resource_value

    def remove_resource(self, resource_name, resource_value):
        if resource_name in self.resources:
            self.resources[resource_name] -= resource_value
            if self.resources[resource_name] <= 0:
                del self.resources[resource_name]

    def get_resource(self, resource_name):
        return self.resources.get(resource_name, 0)

resource_manager = ResourceManager()
resource_manager.add_resource("CPU", 10)
resource_manager.add_resource("Memory", 20)
resource_manager.add_resource("Storage", 30)

print(resource_manager.get_resource("CPU"))  # Output: 10
print(resource_manager.get_resource("Memory"))  # Output: 20
print(resource_manager.get_resource("Storage"))  # Output: 30

resource_manager.remove_resource("CPU", 5)
print(resource_manager.get_resource("CPU"))  # Output: 5
```

class Scheduler:
    def __init__(self):
        self.tasks = []

    def add_task(self, task):
        self.tasks.append(task)

    def remove_task(self, task):
        self.tasks.remove(task)

    def get_tasks(self):
        return self.tasks

scheduler = Scheduler()
scheduler.add_task("Task1")
scheduler.add_task("Task2")
scheduler.add_task("Task3")

print(scheduler.get_tasks())  # Output: ['Task1', 'Task2', 'Task3']
scheduler.remove_task("Task2")
