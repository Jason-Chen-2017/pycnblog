```python
import numpy as np

# 人才特征数据
data = np.array([
    [90, 80, 85, 95],
    [85, 85, 90, 80],
    [90, 95, 85, 80],
    [80, 85, 80, 90]
])

# 人才特征权重
weights = np.array([0.3, 0.3, 0.3, 0.1])

# 人才评估和优化
def evaluate_and_optimize(data, weights):
    # 计算人才的总评分
    scores = np.sum(data * weights, axis=1)
    # 计算人才的优化得分
    optimized_scores = (scores - np.min(scores)) / (np.max(scores) - np.min(scores))
    # 计算人才的排名得分
    ranks = np.sum(optimized_scores * weights, axis=1)
    return scores, optimized_scores, ranks

# 调用评估和优化函数
scores, optimized_scores, ranks = evaluate_and_optimize(data, weights)

# 输出结果
print("Scores:", scores)
print("Optimized Scores:", optimized_scores)
print("Ranks:", ranks)
```
