## 4.1 模糊神经元的实现

以下是一个简单的模糊神经元的实现代码：

```python
import numpy as np

class FuzzyNeuron:
    def __init__(self, inputs, outputs):
        self.inputs = inputs
        self.outputs = outputs
        self.weights = np.random.rand(len(inputs))
        self.bias = np.random.rand()

    def forward(self, inputs):
        weighted_sum = np.dot(inputs, self.weights) + self.bias
        output = self.activation(weighted_sum)
        return output

    def activation(self, x):
        return 1 / (1 + np.exp(-x))

    def backward(self, error):
        d_weights = error * self.activation(self.weights.dot(inputs) + self.bias) * self.activation(self.weights.dot(inputs) + self.bias)
        d_bias = error * self.activation(self.weights.dot(inputs) + self.bias)
        return d_weights, d_bias
```

```python
import numpy as np

class FuzzyNeuralNetwork:
    def __init__(self, inputs, hidden, outputs):
        self.inputs = inputs
        self.hidden = hidden
        self.outputs = outputs
        self.weights1 = np.random.rand(len(hidden), len(inputs))
        self.bias1 = np.random.rand(len(hidden))
        self.weights2 = np.random.rand(len(outputs), len(hidden))
        self.bias2 = np.random.rand(len(outputs))

    def forward(self, inputs):
        hidden_inputs = np.dot(inputs, self.weights1) + self.bias1
        hidden_outputs = [neuron.forward(hidden_inputs) for neuron in self.hidden]
        output_inputs = np.dot(hidden_outputs, self.weights2) + self.bias2
        output = [neuron.forward(output_inputs) for neuron in self.outputs]
        return output

    def backward(self, error):
        d_weights2, d_bias2 = self.outputs[0].backward(error)
        for neuron in self.outputs[1:]:
            d_weights2, d_bias2 = neuron.backward(d_weights2)
        d_weights1 = [neuron.backward(hidden_outputs) for neuron, hidden_outputs in zip(self.hidden, hidden_outputs)]
        for neuron in self.hidden:
            d_weights1, d_bias1 = neuron.backward(d_weights1)
        return d_weights1, d_bias1, d_weights2, d_bias2
```
