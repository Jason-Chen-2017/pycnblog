import matplotlib.pyplot as plt

# 生产设备（制造机器人）
class ManufacturingRobot:
    def __init__(self):
        self.temperature = 0

    def start_work(self):
        self.temperature = np.random.uniform(20, 30)
        print("生产设备开始工作，温度为：", self.temperature)

    def stop_work(self):
        print("生产设备停止工作")

# 传感器（温度传感器）
class TemperatureSensor:
    def __init__(self):
        self.temperature = 0

    def measure_temperature(self):
        self.temperature = np.random.uniform(20, 30)
        print("传感器监测到的温度为：", self.temperature)
        return self.temperature

# 控制器（工业控制器）
class IndustrialController:
    def __init__(self):
        self.temperature = 0

    def collect_temperature_data(self, temperature):
        self.temperature = temperature
        print("控制器收集到的温度数据为：", self.temperature)

    def send_temperature_data(self):
        print("控制器将温度数据发送到计算机上")

# 计算机（工业计算机）
class IndustrialComputer:
    def __init__(self):
        self.temperature = 0

    def receive_temperature_data(self, temperature):
        self.temperature = temperature
        print("计算机收到的温度数据为：", self.temperature)

    def analyze_temperature_data(self):
        print("计算机对收集到的温度数据进行分析")

    def find_problems_and_optimizations(self):
        print("计算机发现生产过程中的问题和优化机会")

# 主程序
if __name__ == "__main__":
    manufacturing_robot = ManufacturingRobot()
    manufacturing_robot.start_work()

    temperature_sensor = TemperatureSensor()
    temperature_sensor.measure_temperature()

    industrial_controller = IndustrialController()
    industrial_controller.collect_temperature_data(temperature_sensor.temperature)
    industrial_controller.send_temperature_data()

    industrial_computer = IndustrialComputer()
    industrial_computer.receive_temperature_data(industrial_controller.temperature)
    industrial_computer.analyze_temperature_data()
    industrial_computer.find_problems_and_optimizations()

    manufacturing_robot.stop_work()
