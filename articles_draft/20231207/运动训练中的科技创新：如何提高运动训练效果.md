    return x[0]**2 + x[1]**2

# 定义约束条件
def constraint_function(x):
    return x[0] + x[1] - 10

# 创建优化问题
problem = {'type': 'min', 'fun': objective_function, 'constraints': [{'type': 'eq', 'fun': constraint_function}]}

# 解决优化问题
