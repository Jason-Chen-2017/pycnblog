```python
import numpy as np
import pandas as pd
from statsmodels.tsa.arima.model import ARIMA

# 加载数据
data = pd.read_csv('energy_data.csv')

# 数据预处理
data = data['energy_data'].values.reshape(-1, 1)

# 训练ARIMA模型
model = ARIMA(data, order=(1, 1, 1))
model_fit = model.fit(disp=0)

# 预测能源需求

print(predictions)
```
