```python
class LogisticsManager:
    def __init__(self, resources, processes, standards):
        self.resources = resources
        self.processes = processes
        self.standards = standards

    def allocate_resources(self, process_id, required_resources):
        for resource in self.resources:
            if resource.id in required_resources:
                required_resources.remove(resource.id)
                process = next((p for p in self.processes if p.id == process_id), None)
                process.resources.append(resource)
        if required_resources:
            raise Exception("Not enough resources to allocate")

    def check_compliance(self, process_id, standard_id):
        process = next((p for p in self.processes if p.id == process_id), None)
        standard = next((s for s in self.standards if s.id == standard_id), None)
        if not standard.rules(process):
            raise Exception(f"Process {process_id} is not compliant with standard {standard_id}")
```
