```python
import time
import random

class Sensor:
    def __init__(self, id, name):
        self.id = id
        self.name = name

    def get_data(self):
        pass

class TemperatureSensor(Sensor):
    def __init__(self, id, name):
        super().__init__(id, name)

    def get_data(self):
        return random.uniform(20, 30)

class HumiditySensor(Sensor):
    def __init__(self, id, name):
        super().__init__(id, name)

    def get_data(self):
        return random.uniform(30, 60)

class Controller:
    def __init__(self):
        self.temperature = 25
        self.humidity = 50

    def control(self, data):
        pass

def main():
    sensors = [TemperatureSensor(1, 'temperature'), HumiditySensor(2, 'humidity')]
    controller = Controller()

    while True:
        for sensor in sensors:
            data = sensor.get_data()
            controller.control(data)
            print(f'{sensor.name}: {data}')

        time.sleep(1)

if __name__ == '__main__':
    main()
```

在这个代码实例中，我们首先定义了一个基类`Sensor`，并且创建了两个子类`TemperatureSensor`和`HumiditySensor`。然后我们创建了一个`Controller`类，用于控制家居设备。在`main`函数中，我们创建了两个传感器和一个控制器，并且在一个无限循环中获取传感器的数据，并将其传递给控制器进行处理。
