```cpp
#include <iostream>
#include <cmath>

// 控制系统
class Controller {
public:
    Controller(double kp, double ki, double kd)
        : kp_(kp), ki_(ki), kd_(kd) {}

    double control(double error, double prev_error, double delta_t) {
        double control_output = kp_ * error + ki_ * prev_error + kd_ * (error - prev_error) / delta_t;
        return control_output;
    }

private:
    double kp_;
    double ki_;
    double kd_;
};

int main() {
    double kp = 100;
    double ki = 100;
    double kd = 100;

    double prev_error = 0;
    double error = 1;
    double delta_t = 0.1;

    Controller controller(kp, ki, kd);

    for (int i = 0; i < 10; ++i) {
        double control_output = controller.control(error, prev_error, delta_t);
        std::cout << "Control output: " << control_output << std::endl;
        prev_error = error;
        error = 0;
    }

    return 0;
}
```
