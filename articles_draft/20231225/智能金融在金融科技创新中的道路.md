```python
from sklearn.linear_model import LinearRegression

y_train = np.array([1, 2, 3, 4, 5])

# 测试集
X_test = np.array([[6], [7], [8], [9], [10]])

# 训练计算机程序

y_pred = model.predict(X_test)

# 输出预测结果
print(y_pred)
```
