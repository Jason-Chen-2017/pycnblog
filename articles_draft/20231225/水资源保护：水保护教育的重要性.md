import random

# 制定水资源保护教育计划
def plan_education(population):
    education_plan = []
    for i in range(population):
        education_plan.append({
            'target': random.choice(['increase_awareness', 'promote_behavior']),
            'content': random.choice(['water_shortage', 'water_saving']),
            'method': random.choice(['lecture', 'competition', 'material'])
        })
    return education_plan

# 组织水资源保护教育活动
def organize_activity(education_plan):
    for plan in education_plan:
        if plan['target'] == 'increase_awareness':
            lecture(plan['content'])
        elif plan['target'] == 'promote_behavior':
            if plan['method'] == 'competition':
                competition(plan['content'])
            elif plan['method'] == 'material':
                material(plan['content'])

# 评估水资源保护教育效果
def evaluate_effect(education_plan):
    effect = 0
    for plan in education_plan:
        if plan['target'] == 'increase_awareness':
            effect += increase_awareness(plan['content'])
        elif plan['target'] == 'promote_behavior':
            effect += promote_behavior(plan['content'])
    return effect

# 举办水资源保护教育讲座
def lecture(content):
    print(f'举办{content}的讲座')

# 组织水资源保护竞赛
def competition(content):
    print(f'组织{content}的竞赛')

# 发布水资源保护宣传材料
def material(content):
    print(f'发布{content}的宣传材料')

# 增加水资源保护意识
def increase_awareness(content):
    awareness = random.randint(1, 100)
    print(f'增加{content}的水资源保护意识，水资源保护意识提高{awareness}%')
    return awareness

# 改善水资源保护行为
def promote_behavior(content):
    behavior = random.randint(1, 100)
    print(f'改善{content}的水资源保护行为，水资源保护行为改善{behavior}%')
    return behavior

# 主程序
if __name__ == '__main__':
    population = 1000
    education_plan = plan_education(population)
    organize_activity(education_plan)
    effect = evaluate_effect(education_plan)
    print(f'水资源保护教育效果：{effect}%')
