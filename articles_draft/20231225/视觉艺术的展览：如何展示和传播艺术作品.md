import numpy as np

def convert_to_digital(image_path):
    # 读取图片
    image = cv2.imread(image_path)
    # 转换为数字形式
    digital_image = np.array(image, dtype=np.uint8)

### 4.2 人工智能展览

例如，实现智能推荐功能，可以使用Python语言编写的代码实现：

```python
from sklearn.neighbors import NearestNeighbors

def recommend(user_id, user_data, item_data):
    # 创建邻近模型
    nn = NearestNeighbors(n_neighbors=5, metric='cosine')
    nn.fit(np.vstack((user_data, item_data)))
    # 获取相似用户
    similar_users = nn.kneighbors(user_data[user_id].reshape(1, -1), return_distance=False)
    # 获取相似用户的喜好项目
    liked_items = [user_data[uid] for uid in similar_users[0]]
    # 计算相似度
    similarity = np.mean([np.dot(user_data[user_id], item) for item in liked_items], axis=1)
    # 推荐项目
    recommended_items = item_data[np.argsort(-similarity)]
    return recommended_items
```

### 4.3 大数据展览

例如，实现异常检测功能，可以使用Python语言编写的代码实现：

```python
from sklearn.ensemble import IsolationForest

def detect_anomalies(data, n_estimators=100, contamination=0.01):
    # 创建异常森林模型
    clf = IsolationForest(n_estimators=n_estimators, contamination=contamination)
    # 训练模型
    clf.fit(data)
    # 预测异常
    preds = clf.predict(data)
    # 返回异常数据
    anomalies = data[preds == -1]
    return anomalies
```
