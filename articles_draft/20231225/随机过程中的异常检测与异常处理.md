```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机过程数据
np.random.seed(0)
T = 100
X = np.random.normal(loc=0, scale=1, size=T)

# 异常检测
def detect_anomaly(X, threshold):
    anomaly = []
    for i in range(T):
        if abs(X[i]) > threshold:
            anomaly.append(X[i])
    return anomaly

# 异常处理
def handle_anomaly(X, anomaly):
    X_handled = np.zeros(T)
    for i in range(T):
        if i in anomaly:
            X_handled[i] = np.mean(X)
    return X_handled

# 异常检测与异常处理
anomaly = detect_anomaly(X, 2)
X_handled = handle_anomaly(X, anomaly)

# 结果可视化
plt.figure(figsize=(10, 6))
plt.plot(X, label='Random Process')
plt.plot(X_handled, label='Handled Anomaly')
plt.legend()
plt.show()
```

```R
  X_handled <- rep(0, T)
  X_handled[anomaly] <- mean(X)
  return(X_handled)
```
