# 创建数据库
def create_database():
    conn = sqlite3.connect('example.db')
    cursor = conn.cursor()
    cursor.execute('CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT, age INTEGER)')
    conn.commit()
    conn.close()

# 插入数据

    conn = sqlite3.connect('example.db')
    cursor = conn.cursor()
    cursor.execute('INSERT INTO users (name, age) VALUES (?, ?)', (name, age))

    conn.close()

    conn = sqlite3.connect('example.db')
    cursor = conn.cursor()
    cursor.execute('SELECT * FROM users WHERE name = ?', (name,))
    rows = cursor.fetchall()
    conn.close()
    return rows

    conn = sqlite3.connect('example.db')
    cursor = conn.cursor()
    cursor.execute('DELETE FROM users WHERE name = ?', (name,))
    conn.commit()

## 4.2Java代码实例

我们将使用Java编写一个简单的数据库系统，该系统可以在不同平台上运行和访问数据。

```java
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

public class DatabaseExample {
    public static void main(String[] args) {
        try {
            // 加载驱动
            Class.forName("org.sqlite.JDBC");
            // 创建连接
            Connection connection = DriverManager.getConnection("jdbc:sqlite:example.db");
            // 创建表
            connection.createStatement().execute("CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT, age INTEGER)");
            // 插入数据
            PreparedStatement insertStatement = connection.prepareStatement("INSERT INTO users (name, age) VALUES (?, ?)");
            insertStatement.setString(1, "Alice");
            insertStatement.setInt(2, 25);
            insertStatement.executeUpdate();
            // 查询数据
            PreparedStatement queryStatement = connection.prepareStatement("SELECT * FROM users WHERE name = ?");
            queryStatement.setString(1, "Alice");
            ResultSet resultSet = queryStatement.executeQuery();
            while (resultSet.next()) {
                System.out.println("ID: " + resultSet.getInt("id") + ", Name: " + resultSet.getString("name") + ", Age: " + resultSet.getInt("age"));
            }
            // 删除数据
            PreparedStatement deleteStatement = connection.prepareStatement("DELETE FROM users WHERE name = ?");
            deleteStatement.setString(1, "Alice");
            deleteStatement.executeUpdate();
        } catch (ClassNotFoundException | SQLException e) {
            e.printStackTrace();
        }
    }
}
```

在上面的代码中，我们使用了Java Database Connectivity（JDBC）API来实现数据库系统。JDBC是一个跨平台的数据库API，可以在Windows、Linux、MacOS等平台上运行和访问数据。
