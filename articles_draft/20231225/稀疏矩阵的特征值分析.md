import numpy as np

def jacobi(A, x0, tol=1e-6, max_iter=1000):
    n = A.shape[0]
    D = np.diag(np.diag(A))
    D_inv = np.linalg.inv(D)
    x = x0
    for i in range(max_iter):
        x_new = D_inv @ (A @ x)
        if np.linalg.norm(x_new - x) < tol:
            break
        x = x_new
    return x

A = np.array([[1, 0, 0], [0, 2, 0], [0, 0, 3]])
x0 = np.array([1, 1, 1])
x = jacobi(A, x0)
print(x)
