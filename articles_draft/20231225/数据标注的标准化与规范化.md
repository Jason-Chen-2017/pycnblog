```python
import numpy as np

def preprocess_data(image):
    # 将图像转换为灰度图像
    gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    # 对灰度图像进行二值化处理
    binary_image = cv2.threshold(gray_image, 0, 255, cv2.THRESH_BINARY | cv2.THRESH_OTSU)[1]
    return binary_image
```

```python
    # 遍历图像中的每个像素点
```
