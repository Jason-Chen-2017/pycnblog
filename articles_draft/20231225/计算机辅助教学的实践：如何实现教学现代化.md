import nltk
from nltk.corpus import wordnet

# 初始化词汇表
vocab = set()

# 加载词汇表
nltk.download('wordnet')
for synset in wordnet.all_synsets():
    for lemma in synset.lemmas():
        vocab.add(lemma.name())

# 计算词向量
word_vectors = {}
for word in vocab:
    word_vectors[word] = sum(frequencies(word) for synset in wordnet.all_synsets(word) for lemma in synset.lemmas())

# 计算单词相似度
def similarity(word1, word2):
    return word_vectors[word1].dot(word_vectors[word2]) / (norm(word_vectors[word1]) * norm(word_vectors[word2]))

# 测试单词相似度
print(similarity('apple', 'orange'))
