# 时间步长

# 空间步长

# 时间步长
t_end = 100

# 空间步长

# 初始化浓度

# 时间迭代
for t in range(int(t_end / dt)):
    C_new = np.zeros((100, 1))
    for i in range(1, 99):
        C_new[i] = C[i] + dt * (D * (C[i + 1] - 2 * C[i] + C[i - 1]) / dx**2 - V * (C[i + 1] - C[i - 1]) / dx + R(i * dx))
    C = C_new

# 绘制浓度分布
plt.plot(np.linspace(0, x_end, 100), C)
plt.xlabel('Space')
plt.ylabel('Concentration')
plt.show()
