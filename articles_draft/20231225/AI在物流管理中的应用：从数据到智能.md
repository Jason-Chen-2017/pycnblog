```python
import networkx as nx

# 创建图
G = nx.Graph()

# 添加节点
G.add_node('A')
G.add_node('B')
G.add_node('C')
G.add_node('D')
G.add_node('E')

# 添加边

G.add_edge('A', 'C', weight=15)
G.add_edge('B', 'C', weight=20)
G.add_edge('B', 'D', weight=10)
G.add_edge('C', 'D', weight=15)
G.add_edge('C', 'E', weight=20)
G.add_edge('D', 'E', weight=5)

# 最短路径算法
def dijkstra(G, start, end):
    # 初始化距离
    dist = {node: float('inf') for node in G.nodes()}
    dist[start] = 0

    # 初始化优先级队列
    q = [(0, start)]

    # 初始化访问标记
    visited = set()

    while q:
        # 获取当前最短距离的节点
        current_distance, current_node = heappop(q)

        # 如果当前节点已经访问过，则跳过
        if current_node in visited:
            continue

        # 标记当前节点为访问
        visited.add(current_node)

        # 如果当前节点是目标节点，则结束循环
        if current_node == end:
            break

        # 遍历当前节点的邻居
        for neighbor, weight in G.edges(current_node, data=True):
            # 如果邻居节点未访问过，并且当前路径距离小于邻居节点的最短距离
            if neighbor not in visited and current_distance + weight < dist[neighbor]:
                # 更新邻居节点的最短距离
                dist[neighbor] = current_distance + weight
                # 将邻居节点加入优先级队列
                heappush(q, (current_distance + weight, neighbor))

    # 返回最短距离
    return dist[end]

# 计算最短路径
print(dijkstra(G, 'A', 'E'))
```

在这个代码实例中，我们首先创建了一个无权图，然后将节点和边添加到图中。接着，我们实现了Dijkstra算法，用于计算从起点‘A’到终点‘E’的最短路径。最后，我们调用Dijkstra算法并打印了最短路径。
