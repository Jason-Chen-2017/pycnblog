```python
# 创建一个量子电路
qc = quil.Program()

# 添加一个量子比特
qc += quil.H(0)

# 添加一个量子门
qc += quil.CNOT(0, 1)

# 将量子电路编译成二进制格式

qvm.warm_start()

# 运行量子电路
result = qvm.run(qc_binary)

# 打印结果
print(result)
```
