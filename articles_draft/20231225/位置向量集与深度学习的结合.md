### 4.1代码实例

import numpy as np
import tensorflow as tf

# 定义位置向量集
def generate_position_set(n, size):
    P = []
    for i in range(n):
        p = np.random.randint(0, size, size=size)
        P.append(p)
    return P

# 融合位置信息
def fuse_position(P, F):
    X = P + F
    return X

# 编码位置信息
def encode_position(P):
    E = np.tanh(P)
    return E

# 传播位置信息
def propagate_position(P, F):
    X = tf.concat([P, F], axis=1)
    return X

# 训练和预测
def train_and_predict(X, y, model):
    model.fit(X, y, epochs=10)
    preds = model.predict(X)
    return preds

# 生成数据
n = 1000
size = 10
X = np.random.randn(n, size)
y = np.random.randn(n)

# 定义模型
model = tf.keras.Sequential([
    tf.keras.layers.Dense(64, activation='relu'),
    tf.keras.layers.Dense(32, activation='relu'),
    tf.keras.layers.Dense(1)
])

# 融合位置信息
P = generate_position_set(n, size)

# 编码位置信息

# 传播位置信息
X = propagate_position(P, X)

# 训练和预测
preds = train_and_predict(X, y, model)
