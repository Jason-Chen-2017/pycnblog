import random

    # 初始化鲸鱼群的位置和速度
    positions = np.random.uniform(-10, 10, size=(n_whales, 1))
    velocities = np.zeros((n_whales, 1))

    # 设置终止条件
    best_position = positions[np.argmin(np.apply_along_axis(objective_function, 1, positions))]
    best_value = objective_function(best_position)
    tolerance = 1e-6
    max_iterations = 100

    for iteration in range(max_iterations):
        # 更新鲸鱼群的位置和速度
        for i in range(n_whales):
            A = 2 * random.random() - 1
            C = 2 * A * random.random() - A
            r = random.random()

            if r < 0.7:
                # 捕食
                D = np.subtract(positions[np.argmin(np.apply_along_axis(objective_function, 1, positions)), 0], positions[i, 0])

                # 漂流
                D = np.subtract(positions[i, 0], positions[np.argmin(np.apply_along_axis(objective_function, 1, positions)), 0])
            else:
                # 潜行
                D = np.subtract(positions[i, 0], positions[np.argmin(np.apply_along_axis(objective_function, 1, positions)), 0])

            velocities[i, 0] = velocities[i, 0] + C * r * D
            positions[i, 0] = positions[i, 0] + velocities[i, 0]

        # 更新全局最佳解
        current_best_position = positions[np.argmin(np.apply_along_axis(objective_function, 1, positions))]
        current_best_value = objective_function(current_best_position)

        if np.abs(current_best_value - best_value) < tolerance:
            break

        best_value = current_best_value
        best_position = current_best_position

    return best_position, best_value
