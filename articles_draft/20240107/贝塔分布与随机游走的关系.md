def transition_probability(i, j):
    return (gamma(a[i] + b[j]) / (gamma(a[i]) * gamma(b[j]))) * (p[i] ** (a[i] - 1) * q[i] ** (b[j] - 1))

        current_state = next_state
        path.append(current_state)
