### 1.4.1 求导法则的Python实现

```python
import numpy as np

def derivative(f, h=1e-5):
    x = np.linspace(-10, 10, 1000)
    y = f(x)
    y_derivative = np.zeros_like(x)
    for i in range(1, len(x)):
        y_derivative[i] = (y[i] - y[i-1]) / h
    return y_derivative
```

### 1.4.2 积分法则的Python实现

import numpy as np

def integral(f, a, b, h=1e-5):
    x = np.linspace(a, b, 1000)
    y = f(x)
    integral_result = np.zeros(len(x))
    integral_result[0] = y[0]
    for i in range(1, len(x)):
