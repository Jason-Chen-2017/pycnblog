### 4.1 医疗图像处理代码实例

```python
import numpy as np
import cv2
import os

def load_data():
    # 加载医疗图像数据
    data = []
    labels = []
    for filename in os.listdir('data/medical_images'):
        img = cv2.imread('data/medical_images/' + filename)
        data.append(img)
        label = np.array([1 if 'positive' in filename else 0])
        labels.append(label)
    return data, labels

def train_model():
    # 训练医疗图像分类模型
    X_train, y_train = load_data()
    model = cv2.createSVM(C=1, kernel_type=cv2.SVM_RBF, gamma=0.5, degree=3)
    model.train(X_train, cv2.Ptr(y_train))
    return model

def test_model():
    # 测试医疗图像分类模型
    X_test, y_test = load_data()
    accuracy = model.test(X_test, cv2.Ptr(y_test))
    print('Accuracy: {:.2f}%'.format(accuracy * 100))
```

### 4.3 医疗文本处理代码实例

```python
import numpy as np
import nltk
import os

def load_data():
    # 加载医疗文本数据
    data = []
    labels = []
    for filename in os.listdir('data/medical_text'):
        text = open('data/medical_text/' + filename, 'r').read()
        data.append(text)
        label = np.array([1 if 'positive' in filename else 0])
        labels.append(label)
    return data, labels

def train_model():
    # 训练医疗文本分类模型
    X_train, y_train = load_data()
    model = nltk.NaiveBayesClassifier.train(X_train, y_train)
    return model

def test_model():
    # 测试医疗文本分类模型
    X_test, y_test = load_data()
    accuracy = model.classify(X_test)
    print('Accuracy: {:.2f}%'.format(accuracy * 100))
```
