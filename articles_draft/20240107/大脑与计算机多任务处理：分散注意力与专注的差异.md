```python
def task1():
    start_time = time.time()

        time.sleep(1)
    end_time = time.time()
    return end_time - start_time

def task2():
    start_time = time.time()
    for _ in range(5):
        time.sleep(1)
    end_time = time.time()

    t1 = task1()
    t2 = task2()
    return t1 + t2

result = brain_multitasking()

```

### 4.2 计算机多任务处理代码实例

计算机多任务处理的代码实例可以使用Python编程语言实现。以下是一个简单的计算机多任务处理示例：

```python
import time
import multiprocessing

def task1():
    start_time = time.time()
    for _ in range(5):
        time.sleep(1)
    end_time = time.time()
    return end_time - start_time

def task2():
    start_time = time.time()
    for _ in range(5):
        time.sleep(1)
    end_time = time.time()
    return end_time - start_time

def computer_multitasking():
    pool = multiprocessing.Pool()
    t1 = pool.apply_async(task1)
    t2 = pool.apply_async(task2)
    result1 = t1.get()
    result2 = t2.get()
    pool.close()
    pool.join()
    return result1 + result2

result = computer_multitasking()
print("计算机多任务处理结果:", result)
```
