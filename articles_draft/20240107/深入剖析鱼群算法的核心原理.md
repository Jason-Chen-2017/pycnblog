```python
import numpy as np

def fish_swarm_optimization(x, pbest, gbest, w, c1, c2, n, iter):
    for t in range(iter):
        r1 = np.random.rand()
        r2 = np.random.rand()
        for i in range(n):
            if pbest[i] < x[i]:
                pbest[i] = x[i]
            r1 = np.random.rand()
            r2 = np.random.rand()
            x[i] = w * x[i] + c1 * r1 * (pbest[i] - x[i]) + c2 * r2 * (gbest - x[i])
        if np.max(x) > np.max(gbest):
            gbest = x
    return gbest
```
