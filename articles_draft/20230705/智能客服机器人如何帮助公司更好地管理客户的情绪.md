```  
import re

class NaturalLanguageProcessing:
    def __init__(self):
        self.pattern = re.compile('^(\S+)(\s+(\S+))?$')

    def preprocess(self, text):
        # 去除HTML标签
        text = text.lower()
        # 去除表情符号
        text = text.translate(str.maketrans('', '', string.punctuation))
        # 去除关键词
        return [word for word in self.pattern.finditer(text) if not word.group(1)]

    def text_to_vector(self, text):
        # 将文本转换成向量
        return [int(word) for word in self.pattern.finditer(text) if not word.group(1)]

    def vector_to_text(self, vectors):
        # 将向量转换成文本
        return''.join([str(word) for word in vectors])
```
