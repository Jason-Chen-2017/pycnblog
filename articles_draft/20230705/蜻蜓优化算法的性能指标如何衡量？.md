    data = data.dropna()
    return data

def partition(data, max_part):
    # 对数据进行分治
    data_list = []
    for i in range(0, len(data), max_part):
        data_list.append(data[i:i + max_part])
    return data_list

def search(data):
    # 对子问题进行蜻蜓搜索，找到最优解
    best = 0
    best_score = 0
    for i in range(len(data)):

        for j in range(len(data)):

                score += (data[i] - data[j]) ** 2
        best = score
        best_score = score
    return best, best_score

    # 更新全局最优解
    new_best = score
    while new_best > best_score:
