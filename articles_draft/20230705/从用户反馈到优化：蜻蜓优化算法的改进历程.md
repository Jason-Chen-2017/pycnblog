```python
def fetch_keywords(url, keyword_length, min_score=0.8, max_score=1.0):
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    keywords = []
    for keyword in soup.stripped_strings:
        if len(keyword) >= keyword_length:
            score = max_score
            if score > min_score:
                keywords.append(keyword)
    return keywords

def sort_keywords(keywords):
    keywords.sort(key=len, reverse=True)
    return keywords

def main():
    keywords = fetch_keywords('https://www.baidu.com/')
    keywords = sort_keywords(keywords)
    for keyword in keywords:
        print(keyword)

if __name__ == '__main__':
    main()
```
