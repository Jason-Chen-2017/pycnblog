        (-1, 1, 1),

        (-1, -1, -1),

    # 创建一个立方体的面
    faces = [

        (4, 5, 6, 7),
        (0, 4, 5, 1),

        (2, 6, 7, 3),

    ]
    # 创建一个立方体的顶点数据
    vertex_data = np.array(vertices)
    # 创建一个立方体的面数据
    face_data = np.array(faces)
