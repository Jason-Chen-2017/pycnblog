# 数据清洗

    # 去重

    # 填充

    # 纠正
    data = data.correct_data()
    return data

# 数据校验
def check_data(data):
    # 验证
    data = data.validate()
    # 过滤
    data = data.filter()
    # 转换
    data = data.transform()
    return data

# 数据标准化
def standardize_data(data):
    # 归一化
    data = data.normalize()
    # 转换
    data = data.transform()
    # 映射
    data = data.map()
    return data

# 数据合并
def merge_data(data1, data2):
    # 匹配
    data = data1.merge(data2, on='key')
    # 融合
    data = data.merge(data2, on='key', how='inner')
    # 整合
    data = data.merge(data2, on='key', how='outer')
    return data
