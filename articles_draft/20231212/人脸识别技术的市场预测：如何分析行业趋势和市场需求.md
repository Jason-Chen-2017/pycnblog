## 4.3 特征匹配

```python
import cv2
import numpy as np

def match_features(features1, features2):
    matcher = cv2.FlannBasedMatcher(dict(algorithm=0, trees=5), {})
    matches = matcher.knnMatch(features1, features2, k=2)
    good_matches = []
    for m, n in matches:
        if m.distance < 0.7 * n.distance:
            good_matches.append(m)
    return good_matches
```

```python
import cv2

def face_recognition(image1, image2):
    if len(good_matches) > 10:
        h, w = image1.shape[0:2]
        img1_small = cv2.resize(image1, (w//8, h//8))
        img2_small = cv2.resize(image2, (w//8, h//8))
        bf = cv2.BFMatcher(cv2.NORM_L2, crossCheck=True)
        matches = bf.match(img1_small, img2_small)
        matches = sorted(matches, key=lambda x: x.distance)
        if len(matches) > 10:
            return True
        else:
            return False
    else:
        return False
```
