### 4.1 先来先服务（FCFS）
```python
import queue

class Process:
    def __init__(self, name, arrival_time, service_time):
        self.name = name
        self.arrival_time = arrival_time
        self.service_time = service_time

def fcfs_schedule(processes):
    queue = queue.Queue()
    for process in processes:
        queue.put(process)

    waiting_time = 0
    total_waiting_time = 0

    while not queue.empty():
        process = queue.get()
        waiting_time = max(waiting_time, process.arrival_time)
        total_waiting_time += waiting_time - process.arrival_time
        waiting_time += process.service_time

    return total_waiting_time / len(processes)

processes = [Process("P1", 0, 5), Process("P2", 2, 3), Process("P3", 4, 8)]
print(fcfs_schedule(processes))
```
