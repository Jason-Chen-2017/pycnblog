2. 光照计算：

```python
import numpy as np

# 光照模型
class Light:
    def __init__(self, position, color):
        self.position = position
        self.color = color

    def get_direction(self, point):
        return self.position - point

    def get_intensity(self, point):
        return np.dot(self.get_direction(point), -1)

# 光照效果
class Material:
    def __init__(self, color, diffuse_coefficient, specular_coefficient):
        self.color = color
        self.diffuse_coefficient = diffuse_coefficient
        self.specular_coefficient = specular_coefficient

    def get_diffuse_color(self, light, point):
        return self.color * light.get_intensity(point) * self.diffuse_coefficient

    def get_specular_color(self, light, point, normal, view_direction):
        return self.color * light.get_intensity(point) * self.specular_coefficient * np.maximum(0, np.dot(normal, view_direction))

# 光照计算
def calculate_light(point, normal, material, light):
    view_direction = -1 * normal
    diffuse_color = material.get_diffuse_color(light, point)
    specular_color = material.get_specular_color(light, point, normal, view_direction)
    return diffuse_color + specular_color
```
