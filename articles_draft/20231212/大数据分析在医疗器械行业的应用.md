```java
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

public class DataAnalysis {
    public static void main(String[] args) {
        // 读取数据
        List<String[]> dataList = new ArrayList<>();
        try (BufferedReader br = new BufferedReader(new FileReader("data.csv"))) {
            String line;
            while ((line = br.readLine()) != null) {
                String[] values = line.split(",");
                dataList.add(values);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }

        // 数据预处理
        List<String[]> dataListClean = new ArrayList<>();
        for (String[] values : dataList) {
            boolean allValuesValid = true;
            for (String value : values) {
                if (value == null || value.isEmpty()) {
                    allValuesValid = false;
                    break;
                }
            }
            if (allValuesValid) {
                dataListClean.add(values);
            }
        }

        // 数据探索
        double[] means = new double[dataListClean.size()];
        double[] stdDevs = new double[dataListClean.size()];
        for (int i = 0; i < dataListClean.size(); i++) {
            String[] values = dataListClean.get(i);
            double sum = 0;
            double sumSquares = 0;
            int count = 0;
            for (String value : values) {
                double valueDouble = Double.parseDouble(value);
                sum += valueDouble;
                sumSquares += valueDouble * valueDouble;
                count++;
            }
            means[i] = sum / count;
            stdDevs[i] = Math.sqrt(sumSquares / count - means[i] * means[i]);
        }
        System.out.println("Means: " + Arrays.toString(means));
        System.out.println("Standard Deviations: " + Arrays.toString(stdDevs));
    }
}
```
