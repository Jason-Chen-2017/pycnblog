# 目标函数
def objective_function(x):
    return x[0]**2 + x[1]**2

# 约束条件
def constraint_function(x):
    return x[0] + x[1] - 10

# 初始化变量
x0 = [2, 3]

# 设置约束
constraints = ({'type': 'eq', 'fun': constraint_function})

# 优化
result = minimize(objective_function, x0, constraints=constraints)

# 输出结果
print('Optimal solution:', result.x)
