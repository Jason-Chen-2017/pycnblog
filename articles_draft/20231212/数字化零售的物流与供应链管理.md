# 创建供应链风险评估模型
def supply_chain_risk_assessment(data):
    risk_score = 0
    for row in data:
        risk_score += row['risk_score']
    return risk_score / len(data)

# 供应链管理策略优化
def supply_chain_management_strategy_optimization(risk_score):
    if risk_score < 0.5:
        return '优'
    elif risk_score >= 0.5 and risk_score < 0.7:
        return '良'
    elif risk_score >= 0.7 and risk_score < 0.9:
        return '中'
    else:
        return '差'
