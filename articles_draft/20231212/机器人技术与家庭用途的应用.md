## 4.3 机器人的导航与路径规划

```python
import numpy as np

class Controller:
    def navigate(self, target):
        # 计算机器人的导航路径
        path = []
        current_position = self.get_current_position()
        target_position = target

        while not np.array_equal(current_position, target_position):
            direction = np.subtract(target_position, current_position)
            distance = np.linalg.norm(direction)
            step = 0.1 * distance
            new_position = current_position + step * direction / distance
            path.append(new_position)
            current_position = new_position

        return path

    def move(self, point):
        # 控制机器人移动到指定点
        current_position = self.get_current_position()
        direction = np.subtract(point, current_position)
        distance = np.linalg.norm(direction)
        step = 0.1 * distance
        new_position = current_position + step * direction / distance
        self.go_to(new_position)
```
