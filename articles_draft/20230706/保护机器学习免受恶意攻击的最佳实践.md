def malicious_sql_injection(url, data):

    cursor.execute('SELECT * FROM users WHERE name =?', (data[0],))
    result = cursor.fetchone()
    cursor.execute('SELECT * FROM users WHERE age >?', (data[1],))
    result = cursor.fetchone()
    cursor.execute('SELECT * FROM users WHERE age <?', (data[2],))
    result = cursor.fetchone()
    cursor.execute('SELECT * FROM users')
    rows = cursor.fetchall()
    for row in rows:
        print(row)
    conn.close()
