### 4.3. 核心代码实现

以下是一个简单的核心代码实现：

```
#include <WiFi.h>
#include <WiFiClient.h>
#include <WiFiServer.h>
#include <Arduino.h>

const int port = 80;
const int ledPin = 25;
const int buzzer = 9;

WiFiServer server(port);

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(buzzer, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  WiFiClient client = server.available();
  if (client) {
    Serial.println("New client");
    String request = client.readStringUntil('\r');
    client.flush();
    client.println('ACK');
    client.readStringUntil('\r');

    // process request
    String requestData = "";
    int command = client.readInt();
    switch (command) {
      case 0:
        // led
        digitalWrite(ledPin,!digitalRead(ledPin));
        break;
      case 1:
        // buzzer
        digitalWrite(buzzer,!digitalRead(buzzer));
        break;
      case 2:
        // request data
        requestData = request;
        break;
    }

    // send data
    client.println('<' + requestData);
    client.println('>');
  }

  // buzzer
  digitalWrite(buzzer,!digitalRead(buzzer));
  delay(100);
  digitalWrite(buzzer, digitalRead(buzzer));
  delay(100);
}
```
