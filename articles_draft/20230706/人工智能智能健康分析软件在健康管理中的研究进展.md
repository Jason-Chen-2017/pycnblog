3.3. 核心代码实现

下面是一个简单的Python代码实现，用于读取用户提供的健康数据，并计算家庭中各种疾病的患病风险。
```python
import numpy as np
import pandas as pd

def calculate_family_health_risk(data):
    # 家庭成员数量
    num_family_members = len(data)
    
    # 各种疾病的患病风险
    disease_risks = {
        "高血压": 0.2,
        "糖尿病": 0.5,
        "高胆固醇": 0.3,
        "肥胖": 0.4,
        "吸烟": 0.1,
        "饮酒": 0.2
    }
    
    # 计算患病风险
    risk_sum = 0
    for disease in disease_risks:
        risk_sum += data[disease] * disease_risks[disease]
    
    # 平均患病风险
    mean_risk = risk_sum / num_family_members
    
    # 返回平均患病风险
    return mean_risk

# 读取用户提供的健康数据
data = [
    {'体重 (kg)': 65, '血压 (mmHg)': 120, '血红蛋白 (g/L)': 130},
    {'体重 (kg)': 70, '血压 (mmHg)': 135, '血红蛋白 (g/L)': 125},
    {'体重 (kg)': 75, '血压 (mmHg)': 140, '血红蛋白 (g/L)': 110},
    {'体重 (kg)': 60, '血压 (mmHg)': 110, '血红蛋白 (g/L)': 120},
    {'体重 (kg)': 75, '血压 (mmHg)': 120, '血红蛋白 (g/L)': 130}
]

# 计算家庭中各种疾病的患病风险
risk = calculate_family_health_risk(data)

print("家庭中各种疾病的患病风险：", risk)
```
