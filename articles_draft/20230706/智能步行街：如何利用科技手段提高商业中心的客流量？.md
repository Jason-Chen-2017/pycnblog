import numpy as np
import pandas as pd
import random
import datetime

class TrafficData:
    def __init__(self):
        self.data = []

    def add_data(self, data):
        self.data.append(data)

    def get_average_flow(self):
        return np.mean(self.data)

class TrafficLight:
    def __init__(self, data, max_flow):
        self.data = data
        self.max_flow = max_flow

    def predict_flow(self):
        predicted_flow = np.interpolate(self.data, [0] * len(self.data) + [self.max_flow], method='linear')
        return predicted_flow

def calculate_rent(data, name):
    rent = 0
    for item in data:
        if item > 0:
            rent += item
    return rent

def main():
    # 定义商业中心智能步行街数据
    traffic_data = TrafficData()
    data = traffic_data.get_average_flow()
    traffic_light = TrafficLight(data, 10000)
    predicted_flow = traffic_light.predict_flow()
    rent_data = calculate_rent(data, 'A')

    print('平均客流量：', predicted_flow)
    print('客流量最大值：', np.max(predicted_flow))
    print('平均租金：', round(rent_data, 2))

if __name__ == '__main__':
    main()
