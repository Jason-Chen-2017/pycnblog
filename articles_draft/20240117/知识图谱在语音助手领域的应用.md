class Entity:

def search_entity(kg, query):
    for entity in kg.entities:
        if query in entity.name:
            return entity
    return None

```

import random

def generate_answer(kg, query):
    entity = search_entity(kg, query)
    if entity:
        answers = []
        for relation in kg.relations:
            if entity.name in relation.name:
                answers.append(relation.name)
        return random.choice(answers)
    else:
        return "I don't know the answer."

answer = generate_answer(kg, 'Apple')
print(answer)
```
