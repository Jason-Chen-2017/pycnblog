g = Graph()

ns = Namespace("http://example.org/")

# 创建实体
person = URIRef(ns.person)

# 创建关系
has_job = URIRef(ns.has_job)

# 创建属性
name = URIRef(ns.name)
age = URIRef(ns.age)

# 创建约束
age_constraint = URIRef(ns.age_constraint)

# 添加实体、关系、属性和约束到知识图谱
g.add((person, name, Literal("Alice")))
g.add((person, age, Literal(30)))
g.add((person, has_job, job))
g.add((job, age_constraint, Literal(35, datatype=XSD.integer)))

# 查询实体、关系和属性
query = """
SELECT ?person ?name ?age ?job
WHERE {
    ?person rdf:type ns:person .
    ?person ns:name ?name .
    ?person ns:age ?age .
    ?person ns:has_job ?job .
    FILTER (?age <= 35)
}
"""

# 执行查询
results = g.query(query)

# 打印查询结果
for row in results:
    print(row)
