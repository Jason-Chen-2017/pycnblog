具体代码实例如下：

```python
import time
import smtplib
from email.mime.text import MIMEText

# 性能指标计算公式的实现
def calculate_performance_metrics(metrics):
    avg_response_time = sum(metrics) / len(metrics)
    max_response_time = max(metrics)
    ninety_percent_response_time = sorted(metrics)[int(len(metrics) * 0.9)]
    return avg_response_time, max_response_time, ninety_percent_response_time

# 报警触发条件的实现
def trigger_alarm(avg_response_time, max_response_time, ninety_percent_response_time):
    if avg_response_time > 200 or max_response_time > 500 or ninety_percent_response_time > 300:
        return True
    else:
        return False

# 报警通知方式的实现
def send_email_notification(subject, body):
    sender = "your_email@example.com"
    receiver = "your_email@example.com"
    password = "your_password"

    msg = MIMEText(body)
    msg['Subject'] = subject
    msg['From'] = sender
    msg['To'] = receiver

    server = smtplib.SMTP('smtp.example.com', 587)
    server.starttls()
    server.login(sender, password)
    server.sendmail(sender, receiver, msg.as_string())
    server.quit()

# 主程序
metrics = [100, 200, 300, 400, 500]  # 示例性能指标
avg_response_time, max_response_time, ninety_percent_response_time = calculate_performance_metrics(metrics)

if trigger_alarm(avg_response_time, max_response_time, ninety_percent_response_time):
    send_email_notification("MySQL 查询速度报警", "查询速度超过阈值")
```
