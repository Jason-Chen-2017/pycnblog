import numpy as np

def cosine_similarity(u, v):
    sim = np.dot(u, v) / (np.linalg.norm(u) * np.linalg.norm(v))
    return sim

def collaborative_filtering(user_ratings, num_neighbors=5):
    user_similarity = {}
    for u in user_ratings.keys():
        user_similarity[u] = {}
        for v in user_ratings.keys():
            if u != v:
                user_similarity[u][v] = cosine_similarity(user_ratings[u], user_ratings[v])

    user_recommendations = {}
    for u in user_ratings.keys():
        neighbors = sorted(user_similarity[u].items(), key=lambda x: x[1], reverse=True)[:num_neighbors]
        for v, sim in neighbors:
            if v not in user_recommendations[u]:
                user_recommendations[u].append(v)

    return user_recommendations

from Crypto.Util.Padding import pad, unpad

def encrypt(plaintext, key):
    cipher = AES.new(key, AES.MODE_ECB)
    ciphertext = cipher.encrypt(pad(plaintext.encode(), AES.block_size))
    return ciphertext

def decrypt(ciphertext, key):
    cipher = AES.new(key, AES.MODE_ECB)
    plaintext = unpad(cipher.decrypt(ciphertext), AES.block_size)
