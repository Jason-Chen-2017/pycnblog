## 4.3. 核心代码实现

```python
import numpy as np
import matplotlib.pyplot as plt
from OpenGL.GL import *
from OpenGL.GLU import *

# 初始化 OpenGL
glClearColor(0.0, 0.0, 0.0, 1.0, 1.0)  # 设置背景色为黑色
glEnable(GL_DEPTH_TEST)  # 开启深度测试

# 加载模型
model = models.loadModel("robot.obj")  # 加载机器人模型的OBJ文件

# 定义机器人的坐标和颜色
base_position = [0, 0, -50], [0, 25, 0], [100, 0, -50]  # 机器人的初始位置和颜色

# 循环遍历手术过程
for operation in ["FINGER_GRIP", "HAND_GRIP", "LEG_GRIP", "FOAM", "NONE"]:
    # 切换到手术模式
    glPushMatrix()
    glColorMode(GL_MODE_BUFFER)
    
    # 设置操作颜色
    if operation == "FINGER_GRIP":
        glColor(0, 0, 0, 1)  # 蓝色，表示手指颜色
    elif operation == "HAND_GRIP":
        glColor(0, 0, 0, 0.5)  # 红色，表示手套颜色
    elif operation == "LEG_GRIP":
        glColor(0, 0, 0, 1)  # 蓝色，表示腿部颜色
    elif operation == "FOAM":
        glColor(0, 0, 0, 1)  # 浅蓝色，表示泡沫颜色
    else:
        glColor(0, 0, 255, 1)  # 白色，表示无特定颜色
    
    # 绘制手术野
    glBegin(GL_POLYGON)
    for vertex in base_position:
        glVertexPoint(1, GL_FALSE)
        glNormalize(0, 0, 1)
        glColor(0, 0, 0, 1)
    glEnd(GL_POLYGON)
    
    # 渲染图形
    glRenderPointe(GL_POLYGON, 0)
    glNormalize(0, 0, 1)
    glColor(0, 0, 0, 1)
    glEnd()
    
    # 交换 GL 状态
    glPopMatrix()
    
    # 恢复 OpenGL 设置
    glEnable(GL_DEPTH_TEST)
    glColorMode(GL_MODE_CUTOUT)

# 关闭 OpenGL
glutTerminate()
```
