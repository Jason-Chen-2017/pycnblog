
然后，编写以下 Python 脚本：

```python
import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.tree import DecisionTreeRegressor
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_squared_error

# 读取数据
def read_data(data_path):
    data = pd.read_csv(data_path)
    return data

# 数据预处理
def preprocess(data):
    data['转向时间'] = pd.to_datetime(data['转向时间'])
    data.set_index('转向时间', inplace=True)
    data.sort_index(inplace=True)
    return data

# 机器学习模型实现
def machine_learning_model(data):
    # 线性回归模型
    lr = LinearRegression()
    lr.fit(data.iloc[:, :-1].values, data.iloc[:, -1].values)
    return lr

# 深度学习模型实现
def deep_learning_model(data):
    # 决策树回归模型
    dt = DecisionTreeRegressor(random_state=0)
    dt.fit(data.iloc[:, :-1].values, data.iloc[:, -1].values)
    return dt

# 数据处理
data = read_data('marketing_data.csv')
preprocessed_data = preprocess(data)

# 机器学习模型
lr = machine_learning_model(preprocessed_data)

# 预测
predictions = lr.predict(preprocessed_data)

# 深度学习模型
dt = deep_learning_model(preprocessed_data)

# 比较两个模型的性能
mse = mean_squared_error(preprocessed_data.iloc[:, :-1].values, predictions)
rmse = np.sqrt(mse)
print("Mean Squared Error (MSE):", mse)
print("Root Mean Squared Error (RMSE):", rmse)
```
