# 4.1 云计算代码实例

在这里，我们将提供一个简单的云计算代码实例，它可以让用户在网络上获取计算资源，从而实现资源的共享和分配。

```python
import time

class CloudComputing:
    def __init__(self):
        self.resources = []

    def add_resource(self, resource):
        self.resources.append(resource)

    def get_resource(self, user):
        for resource in self.resources:
            if resource.available:
                resource.available = False
                user.assign_resource(resource)
                print(f"User {user.id} assigned resource {resource.id}")
                break
        else:
            print("No available resources")

class Resource:
    def __init__(self, id, available):
        self.id = id
        self.available = available

class User:
    def __init__(self, id):
        self.id = id

    def assign_resource(self, resource):
        self.resource = resource

if __name__ == "__main__":
    cloud = CloudComputing()
    resource1 = Resource(1, True)
    resource2 = Resource(2, True)
    user1 = User(1)

    cloud.add_resource(resource1)
    cloud.add_resource(resource2)
    cloud.get_resource(user1)
```
