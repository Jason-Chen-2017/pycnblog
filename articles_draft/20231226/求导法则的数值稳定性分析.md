前向差分求导法则的数学模型公式为：

$$
y_{i+1} = y_i + h \cdot f(x_i, y_i)
$$

```python
import numpy as np

def forward_difference(f, x0, y0, h, N):
    x = np.linspace(x0, x0 + N * h, N + 1)
    y = np.zeros(N + 1)
    y[0] = y0

    for i in range(N):
        y[i + 1] = y[i] + h * f(x[i], y[i])

    return x, y

# 测试函数
def test_function(x, y):
    return x * y

# 参数设置
x0 = 0
y0 = 1
h = 0.1
N = 10

x, y = forward_difference(test_function, x0, y0, h, N)
print("x:", x)
print("y:", y)
```
