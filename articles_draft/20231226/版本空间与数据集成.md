```python
import numpy as np

# 将不同的知识表示为向量空间
def phi(R):
    K = np.zeros(len(R))
    for i, r in enumerate(R):
        K[i] = r
    return K

# 比较不同的知识表示
def sim(K_i, K_j):
    return np.dot(K_i, K_j) / (np.linalg.norm(K_i) * np.linalg.norm(K_j))

# 组合不同的知识表示
def combine(K_i, K_j, w_i, w_j):
    return (w_i * K_i + w_j * K_j) / (w_i + w_j)

# 实现数据集成
def integrate(K_i, K_j, w_i, w_j):
    K_comb = combine(K_i, K_j, w_i, w_j)
    return K_comb

# 示例知识表示
R_i = [1, 2, 3]
R_j = [4, 5, 6]

# 知识表示向量空间
K_i = phi(R_i)
K_j = phi(R_j)

# 知识表示权重
w_i = 0.5
w_j = 0.5

# 数据集成
D_integrated = integrate(K_i, K_j, w_i, w_j)

print(D_integrated)
```
