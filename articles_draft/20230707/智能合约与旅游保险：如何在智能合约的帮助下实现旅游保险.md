```
}

contract Insurance {
    address payable owner;
    mapping(address => uint256) public claims;

    constructor() public {
        owner = msg.sender;
    }

    function claim(address payable claimant, uint256 amount) public payable {
        require(msg.sender == owner, "You are not the owner of the contract.");
        require(amount <= 100000000, "Amount must be greater than 0.");

        // 移除之前的索赔
        clients() wheremsgid == address(this) {
            claims[msg.sender] = 0;
        }

        claims[claimant] = amount;
        clients() wheremsgid == address(this) {
            cout << "Claim submitted." << endl;
        }
    }
}
```
