def preprocess(text):
    "Preprocess the text."
    return re.sub('<[^>]*>', '', text).strip()

def ngrams(text, n):
    "Generate n-grams from the text."
    n = n-1
    while len(text) > 0 and n > 0:
        yield text[-n:]
        text = text[-1:]
        n -= 1

def lemmatize(words):
    "Lemmatize the words."
    return [word.lower() for word in words]

def pos_tag(words):
    "Add part-of-speech tags to the words."
