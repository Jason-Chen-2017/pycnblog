import MySQL

class UserProfile:
    def __init__(self, user_id):
        self.user_id = user_id
        self.username = "johndoe"
        self.email = "johndoe@example.com"
        self.username_md5 = "9842159296789234"

        self.is_active = True
        self.favorites = ["product1", "product2"]
    
    def get_user_profile(self):
        sql_query = "SELECT * FROM users WHERE id = %s AND username_md5 = %s ORDER BY created_at DESC LIMIT 1"
        user_profile = MySQL.connect(
            host="localhost",
            user="root",
            passwd="password",
            db="database",
            charset="utf8"
        )
        user_profile.execute(sql_query, (self.user_id, self.username_md5))
        result = user_profile.fetchone()
        user_profile.close()
        return result

# 推荐模块
class ProductRecommendation:
    def __init__(self, user_profile):
        self.user_profile = user_profile
    
    def get_recommendations(self, user_id, num=10):
        sql_query = "SELECT * FROM products WHERE id NOT IN (SELECT id FROM user_profiles WHERE username_md5 = %s ORDER BY points DESC LIMIT %s) AND user_id = %s ORDER BY points DESC LIMIT %s"
        recommendations = []
        user_profile = self.user_profile
        user_profile.execute(sql_query, (user_profile.username_md5, num, user_id))
        result = user_profile.fetchall()
        for item in result:
            recommendations.append(item[1])
        recommendations = sorted(recommendations, key=lambda x: x.points, reverse=True)
        return recommendations
