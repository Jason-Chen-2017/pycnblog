4.4. 代码讲解说明

```python
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score

# 准备数据集
user_data = pd.read_csv('user_data.csv')
event_data = pd.read_csv('event_data.csv')

# 数据预处理
def preprocess_data(data):
    # 去重
    data = data.drop_duplicates()
    # 排序
    data = data.sort_values(by='event_time')
    # 划分训练集和测试集
    X = data.drop(['user_id', 'event_type'], axis=1)
    y = data['event_action']
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)
    # 回归模型训练
    model = LogisticRegression()
    model.fit(X_train, y_train)
    # 预测
    y_pred = model.predict(X_test)
    # 计算准确率
    accuracy = accuracy_score(y_test, y_pred)
    print('Accuracy: ', accuracy)
    return model

# 预测用户需求
def predict_user_demand(model):
    user_id = 'user_1'
    event_type = 'click'
    event_time = '2022-03-01 14:30:00'
    # 数据预处理
    preprocessed_data = preprocess_data(user_data)
    preprocessed_data = preprocess_data.dropna()
    # 特征工程
    X = preprocessed_data[['user_id', 'event_time', 'event_type']]
    y = preprocessed_data['event_action']
    # 模型训练
    model = model
    model.fit(X, y)
    # 预测
    demand = model.predict([[user_id, event_time, event_type]])[0]
    print('User demand: ', demand)

# 执行营销活动
def execute_marketing_activity(model, user_id, event_type):
    if event_type == 'click':
        # 执行点击活动
        activity = 'click'
        print('Activity: ', activity)
    else:
        # 执行其他活动
        activity = 'none'
        print('Activity: ', activity)

# 应用示例
```
