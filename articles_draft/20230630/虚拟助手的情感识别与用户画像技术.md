import numpy as np

def preprocess(text):  # 对输入文本进行预处理
    # 分词
    words = np.split(text,'')
    # 去除停用词
    words = [word for word in words if word not in [' ，。', '？', '！']]
    # 词性标注
    pos = np.partition(words, axis=0)[0]
    words = pos
    # 命名实体识别
    words = [word for word in words if word.startswith('N')]
    # 删除词性标注中的N
    words = [word for word in words if word.startswith('N') and word not in pos]
    # 去除标点符号
    words = [word for word in words if word.isspace() and word not in ['。', '？', '！']]
    # 将所有处理后的单词进行排序
    words = sorted(words)
    return''.join(words)

def emotional_label(text):  # 对输入文本进行情感极性判定
    words = preprocess(text)
    labels = []
    for word in words:
        # 情感极性判定
        if word.endswith('.'):
            labels.append('正面')
        else:
            labels.append('负面')
    return''.join(labels)

# 训练数据
