    ['user2', 'item2'],
    ['user2', 'item3'],
    ['user3', 'item1'],

    ['user3', 'item3'],
]

# 建模
def build_model(data):
    # 创建一个有向图
    graph = {}
    # 计算每个节点的入度和出度
    in_degree = {}
    out_degree = {}
    # 遍历数据
    for item in data:
        # 更新入度和出度
        in_degree[item[1]] = in_degree.get(item[1], 0) + 1
        out_degree[item[0]] = out_degree.get(item[0], 0) + 1
        # 更新有向图
        graph[item[0]] = graph.get(item[0], set()).union({item[1]})
    return graph, in_degree, out_degree

    # 计算每个节点的概率
    p = {}
    # 遍历数据
    for item in data:
        # 更新概率
        p[item[0]] = p.get(item[0], 0) + 1
        p[item[1]] = p.get(item[1], 0) + 1
    # 计算概率的归一化因子
    norm = sum(p.values())
    # 更新概率
    for key in p.keys():
        p[key] = p[key] / norm
    # 更新有向图的概率

    return graph, p

    next_item = np.argmax([p[item] * graph[user][i] for i in range(len(graph[user]))])

    graph, p = optimize(graph, in_degree, out_degree)
