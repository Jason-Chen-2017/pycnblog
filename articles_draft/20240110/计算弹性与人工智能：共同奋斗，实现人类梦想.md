def get_resource(max_resource, demand):
    return max_resource * (demand / max_demand)

max_resource = 100
max_demand = 1000
current_demand = 500

while True:
    resource = get_resource(max_resource, current_demand)
    print("Current resource: ", resource)
    time.sleep(1)
