```python
def allocate_resource(customer_demands, logistics_resources):
    for demand in customer_demands:
        for resource in logistics_resources:
            if demand.demand_quantity <= resource.resource_quantity and \
               demand.demand_time <= resource.available_time:
                resource.resource_quantity -= demand.demand_quantity
                demand.demand_quantity = 0
                return True
    return False
```

```python
def schedule_resource(customer_demands, logistics_resources):
    customer_demands.sort(key=lambda demand: demand.demand_time)
    logistics_resources.sort(key=lambda resource: resource.available_time)
    
    while customer_demands:
        for resource in logistics_resources:
            if resource.resource_quantity > 0:
                demand = customer_demands.pop(0)
                resource.resource_quantity -= demand.demand_quantity
                demand.demand_quantity = 0
                break
    return logistics_resources
```
