class DecentralizedExchange:
    def __init__(self):
        self.orders = []

    def create_order(self, price, amount, side):
        order = {
            'price': price,
            'amount': amount,
            'side': side,
            'timestamp': time.time(),
            'order_id': len(self.orders)
        }
        self.orders.append(order)
        return order

    def match_orders(self):
        buy_orders = [order for order in self.orders if order['side'] == 'buy']
        sell_orders = [order for order in self.orders if order['side'] == 'sell']

        min_price_difference = float('inf')
        matched_buy_order = None
        matched_sell_order = None

        for buy_order in buy_orders:
            for sell_order in sell_orders:
                price_difference = abs(buy_order['price'] - sell_order['price'])
                if price_difference < min_price_difference:
                    min_price_difference = price_difference
                    matched_buy_order = buy_order
                    matched_sell_order = sell_order

        if matched_buy_order and matched_sell_order:
            buy_order['amount'] -= matched_sell_order['amount']
            sell_order['amount'] -= matched_buy_order['amount']

            if buy_order['amount'] == 0:
                buy_order['status'] = 'filled'
            if sell_order['amount'] == 0:
                sell_order['status'] = 'filled'

            return matched_buy_order, matched_sell_order

        return None, None
