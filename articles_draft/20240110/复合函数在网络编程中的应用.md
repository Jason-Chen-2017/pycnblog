import numpy as np

# 定义基本函数

    # 数据清洗
    return data.dropna()

    # 数据转换
    return data.astype(int)

def aggregate_data(data):
    # 数据聚合
    return data.sum()

# 定义复合函数
def process_data(data):
    # 使用复合函数处理数据
    data = clean_data(data)
    data = transform_data(data)
    data = aggregate_data(data)
    return data

# 测试数据
