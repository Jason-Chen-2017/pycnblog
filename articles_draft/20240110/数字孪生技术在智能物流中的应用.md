首先，我们需要收集物流资源的状态数据。假设我们有一个物流资源的状态数据列表：

```python
resource_status = [
    {'id': 1, 'location': 'A', 'speed': 10, 'temperature': 20, 'humidity': 50},
    {'id': 2, 'location': 'B', 'speed': 20, 'temperature': 25, 'humidity': 60},
    {'id': 3, 'location': 'C', 'speed': 30, 'temperature': 30, 'humidity': 70},
]
```

接下来，我们需要将物流资源的状态数据转换为数字模拟设备的状态数据。我们可以使用 Python 的 `json` 模块来实现这个功能：

```python
import json

def convert_resource_status(resource_status):
    digital_status = []
    for status in resource_status:
        digital_status.append(json.dumps(status))
    return digital_status

digital_status = convert_resource_status(resource_status)
print(digital_status)
```

接下来，我们需要建立物理世界与数字世界之间的一一对应关系，实现物理设备与数字模拟设备之间的实时同步。我们可以使用 Python 的 `socket` 模块来实现这个功能：

```python
import socket

def start_server():
    host = 'localhost'
    port = 12345
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server_socket.bind((host, port))
    server_socket.listen(5)
    print('Server is listening...')
    while True:
        client_socket, addr = server_socket.accept()
        print('Client connected:', addr)
        data = client_socket.recv(1024)
        print('Received data:', data.decode())
        client_socket.send('OK'.encode())
        client_socket.close()

def start_client():
    host = 'localhost'
    port = 12345
    client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    client_socket.connect((host, port))
    resource_status = json.dumps(resource_status)
    client_socket.send(resource_status.encode())
    data = client_socket.recv(1024)
    print('Received data:', data.decode())
    client_socket.close()

if __name__ == '__main__':
    start_server()
```
