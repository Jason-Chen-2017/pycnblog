$$

\end{cases}
$$

我们可以使用Python编写如下代码来实现这个方程组的解：

```python
import numpy as np

def objective_function(x):
    return np.sum((x - np.array([1, 1]))**2)

def jacobian(x):
    return np.array([[2, 1], [1, 1]])

def high_phase_newton_method(x0, tol=1e-6, max_iter=100):
    x = np.copy(x0)
    for i in range(max_iter):
        g = jacobian(x)
        H = np.linalg.inv(g.T @ g) @ g.T @ g
        dx = -H @ g @ x
        x += dx
        if np.linalg.norm(dx) < tol:
            break
    return x

x0 = np.array([0, 0])
x_optimal = high_phase_newton_method(x0)
print(x_optimal)
```
