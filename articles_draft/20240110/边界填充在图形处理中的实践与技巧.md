$$
f(x) = ax + b
$$

其中，$a$ 是斜率，$b$ 是截距。

具体操作步骤如下：

1. 找到边界上的两个邻近点。
2. 计算它们之间的斜率和截距。
3. 使用斜率和截距来计算新的点的值。

def barycentric_coordinates(x, y):
    x0, y0 = 0, 0
    x1, y1 = 1, 0
    x2, y2 = 0, 1
    return (1 - x - y) * x0 * y0 + x * x1 * y0 + y * x2 * y0, (1 - x - y) * x0 * y1 + x * x1 * y1 + y * x2 * y1

import numpy as np

def gaussian_kernel(sigma):
    x = np.linspace(-sigma, sigma, 2 * sigma + 1)
    y = np.linspace(-sigma, sigma, 2 * sigma + 1)
    x, y = np.meshgrid(x, y)
    return (1 / (2 * np.pi * sigma**2)) * np.exp(-(x**2 + y**2) / (2 * sigma**2))

def gaussian_filter(image, sigma):
