# 实现PID控制

Ki = 1
Kd = 1

def pid_control(error, prev_error, prev_control, dt):
    integral = prev_error + error * dt
    derivative = (error - prev_error) / dt
    control = Kp * error + Ki * integral + Kd * derivative
    return control
