# 4. 具体代码实例和详细解释说明
# 4.1 传感器数据收集的代码实例

以下是一个使用 Python 编程语言实现传感器数据收集的代码实例：

```python
import time
import random

class Sensor:
    def __init__(self, id, name):
        self.id = id
        self.name = name
        self.data = []

    def collect_data(self):
        self.data.append(random.randint(0, 100))

    def get_data(self):
        return self.data

# 创建传感器对象
temperature_sensor = Sensor(1, "Temperature")
humidity_sensor = Sensor(2, "Humidity")
light_sensor = Sensor(3, "Light")

# 收集传感器数据
for i in range(10):
    temperature_sensor.collect_data()
    humidity_sensor.collect_data()
    light_sensor.collect_data()
    time.sleep(1)

# 获取传感器数据
temperature_data = temperature_sensor.get_data()
humidity_data = humidity_sensor.get_data()
light_data = light_sensor.get_data()

# 计算平均温度、平均湿度、平均光线
temperature_avg = sum(temperature_data) / len(temperature_data)
print("Average Temperature: ", temperature_avg)

humidity_avg = sum(humidity_data) / len(humidity_data)
print("Average Humidity: ", humidity_avg)

light_avg = sum(light_data) / len(light_data)
print("Average Light: ", light_avg)
```

```python
class SmartController:
    def __init__(self):
        self.temperature = 25
        self.humidity = 50
        self.light = 100

    def control_temperature(self, target_temperature):
        if self.temperature < target_temperature:
            print("Turn on heater")
        elif self.temperature > target_temperature:
            print("Turn off heater")

    def control_humidity(self, target_humidity):
        if self.humidity < target_humidity:
            print("Turn on humidifier")
        elif self.humidity > target_humidity:
            print("Turn off humidifier")

    def control_light(self, target_light):
        if self.light < target_light:
            print("Turn on light")
        elif self.light > target_light:
            print("Turn off light")

# 创建智能控制对象
smart_controller = SmartController()

# 设置目标温度、目标湿度、目标光线
target_temperature = 24
target_humidity = 45
target_light = 80

# 实现智能控制
smart_controller.control_temperature(target_temperature)
smart_controller.control_humidity(target_humidity)
smart_controller.control_light(target_light)
```
