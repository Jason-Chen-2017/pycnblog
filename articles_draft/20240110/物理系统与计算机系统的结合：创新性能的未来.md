## 1. 傅里叶变换的Python实现

```python
import numpy as np
import matplotlib.pyplot as plt

# 定义信号
def signal(t):
    if -2 <= t <= 2:
        return np.sin(np.pi * t)
    else:
        return 0

# 计算傅里叶变换
def fft(signal, N):
    fft_result = np.fft.fft(signal, N)
    return fft_result

# 绘制信号和傅里叶变换
def plot(signal, fft_result):
    t = np.linspace(-4, 4, 1000)
    plt.plot(t, signal(t), label='信号')
    plt.plot(t, np.abs(fft_result), label='傅里叶变换')
    plt.legend()
    plt.show()

# 主程序
if __name__ == '__main__':
    N = 1024
    signal = np.array([signal(t) for t in range(-N//2, N//2)])
    fft_result = fft(signal, N)
    plot(signal, fft_result)
```

import matplotlib.pyplot as plt

def signal(t):
    if -2 <= t <= 2:
        return np.sin(np.pi * t)
    else:
        return 0

# 定义信号传输系统
def h(t):
    if -2 <= t <= 2:
        return np.exp(-np.pi * t**2)
    else:
        return 0

# 计算信号传输
def transmission(signal, h):
    transmission_result = np.convolve(signal, h)
    return transmission_result

# 绘制信号和信号传输
