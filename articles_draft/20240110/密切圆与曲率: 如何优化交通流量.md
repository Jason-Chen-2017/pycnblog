$$

## 4.1 密切圆的计算

以下是一个计算密切圆的Python代码实例：

```python
import numpy as np

def calculate_circle_center(cars):
    x_sum = 0
    y_sum = 0
    n = len(cars)
    for car in cars:
        x_sum += car['x']
        y_sum += car['y']
    x_center = x_sum / n
    y_center = y_sum / n
    return (x_center, y_center)

def calculate_circle_radius(cars, x_center, y_center):
    d_sum = 0
    n = len(cars)
    for i in range(n):
        car = cars[i]
        dx = car['x'] - x_center
        dy = car['y'] - y_center
        d_sum += np.sqrt(dx**2 + dy**2)
    radius = d_sum / n
    return radius

cars = [{'x': 0, 'y': 0, 'speed': 1}, {'x': 1, 'y': 0, 'speed': 1}, {'x': 2, 'y': 0, 'speed': 1}]
x_center, y_center = calculate_circle_center(cars)
radius = calculate_circle_radius(cars, x_center, y_center)
print('密切圆的中心:', (x_center, y_center))
print('密切圆的半径:', radius)
```

def calculate_curvature(points):
    n = len(points)
    if n < 2:
        return None
    dx_ddx = (points[1][0] - points[0][0], points[2][0] - points[1][0])
    dy_ddx = (points[1][1] - points[0][1], points[2][1] - points[1][1])
    d2x_ddx2 = (points[2][0] - 2 * points[1][0] + points[0][0], points[3][0] - 2 * points[2][0] + points[1][0])
