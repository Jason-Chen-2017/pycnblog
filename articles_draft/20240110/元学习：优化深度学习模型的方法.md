## 4.1 代码实例

我们以一个简单的超参数优化任务为例，来演示元学习的使用方法和原理。

```python
import numpy as np
import tensorflow as tf

# 生成训练数据
def generate_data():
    X = np.random.rand(100, 10)
    y = np.random.rand(100)
    return X, y

# 定义学习模型
def define_model():
    model = tf.keras.Sequential([
        tf.keras.layers.Dense(64, activation='relu'),
        tf.keras.layers.Dense(1)
    ])
    return model

# 定义元学习模型
def define_meta_model():
    meta_model = tf.keras.Sequential([
        tf.keras.layers.Dense(64, activation='relu'),
        tf.keras.layers.Dense(32, activation='relu'),
        tf.keras.layers.Dense(1)
    ])
    return meta_model

# 训练元学习模型
def train_meta_model(meta_model, X, y, epochs=100):
    optimizer = tf.keras.optimizers.Adam(learning_rate=0.001)
    meta_model.compile(optimizer=optimizer, loss='mse')
    meta_model.fit(X, y, epochs=epochs)
    return meta_model

# 使用元学习模型进行超参数优化
def optimize_hyperparameters(meta_model, X, y):
    hyperparameters = {
        'learning_rate': [0.01, 0.001, 0.0001],
        'batch_size': [32, 64, 128]
    }
    best_loss = np.inf
    best_hyperparameters = None
    for learning_rate in hyperparameters['learning_rate']:
        for batch_size in hyperparameters['batch_size']:
            model = define_model()
            model.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=learning_rate), loss='mse')
            loss = model.evaluate(X, y)
            if loss < best_loss:
                best_loss = loss
                best_hyperparameters = {
                    'learning_rate': learning_rate,
                    'batch_size': batch_size
                }
    return best_hyperparameters

# 主程序
if __name__ == '__main__':
    X, y = generate_data()
    meta_model = define_meta_model()
    meta_model = train_meta_model(meta_model, X, y)
    best_hyperparameters = optimize_hyperparameters(meta_model, X, y)
    print('Best hyperparameters:', best_hyperparameters)
```

在上面的代码实例中，我们首先生成了训练数据，并定义了学习模型和元学习模型。接着，我们训练了元学习模型，并使用元学习模型进行超参数优化。
