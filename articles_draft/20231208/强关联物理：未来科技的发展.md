### 4.2 粒子物理算法实例
```python
import numpy as np
from scipy.integrate import solve_ivp

# 初始化一个粒子系统
def f(t, y):
    return np.array([y[1], -y[0]])

y0 = np.array([1, 0])
t_span = (0, 10)

# 应用一个粒子模型
sol = solve_ivp(f, t_span, y0, method='RK45', rtol=1e-6, atol=1e-6)

# 计算粒子系统的能量和稳定性
E = np.dot(sol.y[:, -1], sol.y[:, -1])
stability = np.linalg.norm(sol.y[:, -1])

# 输出结果
print(E, stability)
```

### 4.3 光学算法实例
```python
import numpy as np
from scipy.optimize import minimize

# 初始化一个光学系统
def f(x):
    return np.sum((x - np.array([1, 0, 0]))**2)

x0 = np.array([0, 0, 0])

# 应用一个光分析方法
res = minimize(f, x0, method='BFGS')

# 计算光学系统的传输效率和分辨率
efficiency = res.fun
resolution = np.linalg.norm(res.x)

# 输出结果
print(efficiency, resolution)
```
