## 4. 具体代码实例和详细解释说明

### 4.1 数据收集

```python
import requests
import pandas as pd

def collect_data(platform):
    url = f"https://{platform}.com/data"
    response = requests.get(url)
    data = response.json()
    return pd.DataFrame(data)

platforms = ["google", "baidu", "tencent"]
data = []
for platform in platforms:
    data.append(collect_data(platform))

df = pd.concat(data, axis=0)
```

### 4.2 数据存储

```python
import sqlite3

def store_data(df, filename):
    conn = sqlite3.connect(filename)
    df.to_sql("ad_data", conn, if_exists="replace")
    conn.close()

store_data(df, "ad_data.db")
```

### 4.3 数据预处理

```python
import pandas as pd

def preprocess_data(df):
    df = df.dropna()
    df = df.drop_duplicates()
    df = df.reset_index(drop=True)
    return df

df = preprocess_data(df)
```

### 4.4 数据清洗

```python
import pandas as pd

def clean_data(df):
    df = df.fillna(df.mean())
    df = df.replace(to_replace=0, method='ffill')
    return df

df = clean_data(df)
```

### 4.5 数据转换

```python
import pandas as pd

def transform_data(df):
    df = df.groupby("platform").mean()
    df = df.reset_index()
    return df

df = transform_data(df)
```
