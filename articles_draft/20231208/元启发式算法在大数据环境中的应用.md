# 初始化
x = np.random.rand(1)

# 评估
f = lambda x: x ** 2

# 局部搜索
def local_search(x, f):
    dx = np.random.randn(1)
    x_new = x + dx
    if f(x_new) > f(x):
        x = x_new
    return x

# 全局搜索
def global_search(x, f, max_iter):
    for _ in range(max_iter):
        x = local_search(x, f)
    return x

# 随机性

    for _ in range(max_iter):
        x = np.random.rand(1)
        if f(x) > f(x):
            x = x
    return x

# 终止条件
max_iter = 1000

# 执行算法
