## 4.2 创建一个简单的游戏循环

```python
import pygame

def game_loop():
    pygame.init()

    # 设置游戏窗口大小
    screen_width = 800
    screen_height = 600
    screen = pygame.display.set_mode((screen_width, screen_height))

    # 设置游戏对象列表
    game_objects = [game_object]

    # 游戏循环
    while True:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                pygame.quit()
                return

        # 更新游戏对象
        for game_object in game_objects:
            game_object.update()

        # 绘制游戏对象
        screen.fill((0, 0, 0))
        for game_object in game_objects:
            pygame.draw.rect(screen, game_object.color, (game_object.x, game_object.y, game_object.width, game_object.height))

        # 更新游戏窗口
        pygame.display.flip()

# 调用游戏循环
game_loop()
```
