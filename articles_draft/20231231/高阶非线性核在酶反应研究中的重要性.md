import matplotlib.pyplot as plt

def reaction_rate(S, k1):
    return -k1 * S

def product_rate(S, k1, k2):
    return k1 * S - k2 * P

t = np.linspace(0, 10, 100)
S0 = 1
P0 = 0
k1 = 0.1
k2 = 0.01

dS = reaction_rate(S0, k1) * t

plt.plot(t, dS, label='Substrate')
plt.plot(t, dP, label='Product')
plt.xlabel('Time')
plt.ylabel('Concentration')
plt.legend()
plt.show()
