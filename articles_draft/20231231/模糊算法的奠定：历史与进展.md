import numpy as np

def fuzzy_union(A, B):
    C = np.zeros((len(A), len(B)))
    for i in range(len(A)):
        for j in range(len(B)):
            C[i][j] = max(A[i], B[j])
    return C

def fuzzy_intersection(A, B):
    C = np.zeros((len(A), len(B)))
    for i in range(len(A)):
        for j in range(len(B)):
            C[i][j] = min(A[i], B[j])
    return C

def fuzzy_difference(A, B):
    C = np.zeros((len(A), len(B)))
    for i in range(len(A)):
        for j in range(len(B)):
            C[i][j] = max(A[i] - B[j], 0)
    return C

A = np.array([[0.8, 0.6, 0.4], [0.7, 0.5, 0.3]])
B = np.array([[0.6, 0.4, 0.2], [0.5, 0.3, 0.1]])

print("和：", fuzzy_union(A, B))
print("积：", fuzzy_intersection(A, B))
print("差：", fuzzy_difference(A, B))
