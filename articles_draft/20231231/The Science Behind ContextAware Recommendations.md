def calculate_score(user, item, context):
    # 计算用户和商品之间的相似度
    similarity = cosine_similarity(user.features, item.features)
    # 计算上下文特征的影响
    context_weight = context.weight
    # 计算最终的推荐分数
    score = similarity * context_weight
    return score

class RecommendationSystem:
    def __init__(self, users, items, context):
        self.users = users
        self.items = items
        self.context = context
        self.scores = {}

    def recommend(self, user, context):
        # 计算用户和商品之间的相似度
        similarities = []
        for item in self.items:
            score = calculate_score(user, item, self.context)
            similarities.append(score)
        # 排序推荐内容
        sorted_items = sorted(zip(self.items, similarities), key=lambda x: x[1], reverse=True)
        # 返回推荐列表
        recommendations = [item[0] for item in sorted_items]
        return recommendations
