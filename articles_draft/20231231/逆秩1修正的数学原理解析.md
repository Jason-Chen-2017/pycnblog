    noise = noise_level * np.random.randn(len(signal))
    return signal + noise

# 逆秩估计
def rake_ordering(signal):
    N = len(signal)
    rake = np.zeros(N, dtype=complex)
    for n in range(N):
        rake[n] = signal[n] * np.exp(-2j * np.pi * n * n / N)
    return rake

# 相位修正
def phase_correction(rake):
    N = len(rake)
    phase = np.angle(rake)
    corrected_rake = rake * np.exp(-1j * phase)
    return corrected_rake

# 信号重构
def signal_reconstruction(corrected_rake):
    N = len(corrected_rake)
    reconstructed_signal = np.zeros(N, dtype=complex)
    for n in range(N):
        reconstructed_signal[n] = corrected_rake[n] * np.exp(2j * np.pi * n * n / N)
    return reconstructed_signal

# 主程序
if __name__ == '__main__':
    # 生成信号
