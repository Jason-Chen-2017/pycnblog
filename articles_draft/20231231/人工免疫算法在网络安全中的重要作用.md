def f(x):
    return 1 / (1 + np.linalg.norm(x))

def A(x, library):
    return (1 / (1 + np.linalg.norm(x))) * np.sum(np.abs(x - library) <= 0.1)

def g(x, library):
    return 1 / (1 + f(x)) * (1 - A(x, library))
