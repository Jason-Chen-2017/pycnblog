```python
import numpy as np

def portfolio_optimization(weights, returns, risk_free_rate, risk_aversion):
    n = len(returns)
    weight_vector = np.ones(n) * weights
    risk_premium = np.zeros(n)

    for i in range(n):
        risk_premium[i] = (returns[i] - risk_free_rate) * weight_vector[i]

    portfolio_return = np.sum(risk_premium)
    portfolio_variance = np.dot(weight_vector.T, np.dot(returns.cov(), weight_vector))
    sharpe_ratio = (portfolio_return - risk_free_rate) / np.sqrt(portfolio_variance)

    while risk_aversion * portfolio_variance > sharpe_ratio:
        weight_vector = weight_vector - risk_aversion * np.dot(returns.cov(), weight_vector)
        weight_vector = weight_vector / np.sum(weight_vector)

        risk_premium = np.zeros(n)
        for i in range(n):
            risk_premium[i] = (returns[i] - risk_free_rate) * weight_vector[i]

        portfolio_return = np.sum(risk_premium)
        portfolio_variance = np.dot(weight_vector.T, np.dot(returns.cov(), weight_vector))
        sharpe_ratio = (portfolio_return - risk_free_rate) / np.sqrt(portfolio_variance)

    return weight_vector
```

  - `returns`：资产的历史收益率。
