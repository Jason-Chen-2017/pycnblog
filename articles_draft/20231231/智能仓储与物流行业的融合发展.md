## 4.3 路径规划算法实例
```python
from heapq import heappush, heappop

# 图的邻接表表示
graph = {
    'A': [('B', 1), ('C', 4)],
    'B': [('A', 1), ('C', 2), ('D', 5)],
    'C': [('A', 4), ('B', 2), ('D', 3)],
    'D': [('B', 5), ('C', 3)]
}

# 路径规划算法
def dijkstra(graph, start, end):
    queue = [(0, start, [])]
    visited = set()

    while queue:
        dist, vertex, path = heappop(queue)
        if vertex not in visited:
            visited.add(vertex)
            path = path + [vertex]
            if vertex == end:
                return path, dist
            for neighbor, weight in graph[vertex]:
                if neighbor not in visited:
                    heappush(queue, (dist + weight, neighbor, path))

    return None

# 路径规划
path, cost = dijkstra(graph, 'A', 'D')
print(path, cost)
```
