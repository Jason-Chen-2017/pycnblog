import numpy as np

# 构建位置向量集
def build_lvs(locations):
    lvs = []
    for location in locations:
        vector = np.array(location) / np.linalg.norm(location)
        lvs.append(vector)
    return lvs

# 定义遗传算法的基本操作
def selection(population):
    # 选择最佳个体
    best_individual = min(population, key=lambda x: x[0])
    return best_individual

def crossover(parent1, parent2):
    # 交叉操作
    child = (parent1[0] * parent2[1] + parent1[1] * parent2[0]) / 2
    return child

def mutation(individual, mutation_rate):
    # 变异操作
    if np.random.rand() < mutation_rate:
        individual[0] += np.random.rand() - 0.5
        individual[1] += np.random.rand() - 0.5
    return individual

# 定义遗传算法的目标函数
def fitness_function(individual, locations, distances):
    # 计算货物运输距离
    distance = np.linalg.norm(individual - distances)
    return distance

# 使用遗传算法进行优化
def genetic_algorithm(lvs, locations, distances, mutation_rate, generations):
    population = [np.array([np.random.rand(), np.random.rand()]) for _ in range(100)]
    for _ in range(generations):
        population = sorted(population, key=lambda x: fitness_function(x, locations, distances))
        population = [selection(population) for _ in range(len(population))]
        population = [crossover(population[i], population[i+1]) for i in range(0, len(population), 2)]
        population = [mutation(individual, mutation_rate) for individual in population]
    return population[0]
