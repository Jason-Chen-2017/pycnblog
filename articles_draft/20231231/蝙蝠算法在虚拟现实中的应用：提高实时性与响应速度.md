import numpy as np

class Bat:
    def __init__(self, x, v, f):
        self.x = x
        self.v = v
        self.f = f

    def update_velocity(self, x_best, v_max, c, phi):
        self.v = self.v + c * np.random.rand() * (x_best - self.x) + phi * (v_max - self.v)

    def update_position(self, x_best):
        self.x = self.x + self.v

class BatAlgorithm:
    def __init__(self, num_bats, x_min, x_max, v_min, v_max, f_min, f_max, max_iter):
        self.num_bats = num_bats
        self.x_min = x_min
        self.x_max = x_max
        self.v_min = v_min
        self.v_max = v_max
        self.f_min = f_min
        self.f_max = f_max
        self.max_iter = max_iter
        self.bats = [Bat(np.random.uniform(x_min, x_max), np.random.uniform(v_min, v_max), np.random.uniform(f_min, f_max)) for _ in range(num_bats)]
        self.x_best = self.bats[0].x
        self.f_best = self.bats[0].f

    def run(self):
        for t in range(self.max_iter):
            x_best = self.bats[0].x
            f_best = self.bats[0].f
            for i in range(1, self.num_bats):
                if self.bats[i].f < f_best:
                    x_best = self.bats[i].x
                    f_best = self.bats[i].f

                if self.bats[i].f < f_best:

                    f_best = self.bats[i].f
