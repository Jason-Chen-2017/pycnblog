from knowledge_graph import KnowledgeGraph

def knowledge_graph(entities, relations):
    # 构建知识图谱
    kg = KnowledgeGraph()
    # 添加实体
    for entity in entities:
        kg.add_entity(entity)
    # 添加关系
    for relation in relations:
        kg.add_relation(relation)
    # 构建知识图谱图
    g = kg.build_graph()
    return g
