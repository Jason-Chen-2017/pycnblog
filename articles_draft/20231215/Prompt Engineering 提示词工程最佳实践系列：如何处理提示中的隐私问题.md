    # 使用AES加密算法加密敏感信息
    key = 'abcdefghijklmnop'
    cipher = AES.new(key, AES.MODE_EAX)
    ciphertext, tag = cipher.encrypt_and_digest(text.encode())

    # 处理提示中的敏感信息
