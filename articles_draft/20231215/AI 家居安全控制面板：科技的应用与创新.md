```python
    # 去除噪声
    data = np.nan_to_num(data)
    
    # 填充缺失值
    data = np.interp(np.arange(data.shape[0]),
                     np.arange(data.shape[0])[~np.isnan(data)],
                     data[~np.isnan(data)])
    
    return data
```
