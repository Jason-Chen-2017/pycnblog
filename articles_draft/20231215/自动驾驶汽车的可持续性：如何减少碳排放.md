import heapq
from collections import defaultdict

def a_star(graph, start, goal):
    # 初始化启发式函数
    heuristic = lambda: graph[start][goal]['weight']

    # 初始化开始节点
    start_node = start
    node_queue = [(0, start_node)]

    # 初始化已访问节点集合
    visited = set()

    while node_queue:
        # 获取当前节点和最短路径长度
        current_node, current_cost = heapq.heappop(node_queue)

        # 如果当前节点是目标节点，则返回最短路径长度
        if current_node == goal:
            return current_cost

        # 如果当前节点已经被访问过，则跳过
        if current_node in visited:
            continue

        # 添加当前节点到已访问节点集合
        visited.add(current_node)

        # 获取当前节点的邻居节点
        neighbors = graph[current_node]

        # 遍历当前节点的邻居节点
        for neighbor in neighbors:
            # 计算邻居节点的最短路径长度
            neighbor_cost = current_cost + graph[current_node][neighbor]['weight']

            # 如果邻居节点不在已访问节点集合中，并且邻居节点的最短路径长度小于邻居节点的最短路径长度，则更新邻居节点的最短路径长度
            if neighbor not in visited and neighbor_cost < graph[neighbor][goal]['weight']:
                graph[neighbor][goal]['weight'] = neighbor_cost
                heapq.heappush(node_queue, (neighbor_cost, neighbor))

    # 如果没有找到目标节点，则返回None
    return None
