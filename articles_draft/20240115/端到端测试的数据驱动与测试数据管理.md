import hashlib

def generate_test_data(param):
    data = {}
    for key, value in param.items():
        if isinstance(value, list):
            data[key] = random.choice(value)
        else:
            data[key] = value
    return data

# 测试数据存储
def store_test_data(data):
    # 使用哈希算法对数据进行存储
    hash_data = hashlib.sha256(str(data).encode()).hexdigest()
    return hash_data

def update_test_data(data):
    # 使用哈希算法对数据进行更新
    hash_data = hashlib.sha256(str(data).encode()).hexdigest()
    return hash_data

def delete_test_data(data):
    # 使用哈希算法对数据进行删除
    hash_data = hashlib.sha256(str(data).encode()).hexdigest()
    return hash_data

# 端到端测试的数据驱动与测试数据管理
def end_to_end_test(param):
