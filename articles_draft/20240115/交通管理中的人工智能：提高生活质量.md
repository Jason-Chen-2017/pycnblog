1. 交通规划中的最短路径算法实现：

```python
import math

def distance(x1, y1, x2, y2):
    return math.sqrt((x2 - x1)**2 + (y2 - y1)**2)

def dijkstra(graph, start, end):
    visited = set()
    distance = {node: float('inf') for node in graph}
    distance[start] = 0
    path = {node: None for node in graph}

    while len(visited) < len(graph):
        current_node = None
        for node in graph:
            if node not in visited and (current_node is None or distance[node] < distance[current_node]):
                current_node = node

        visited.add(current_node)
        for neighbor, weight in graph[current_node].items():
            new_distance = distance[current_node] + weight
            if new_distance < distance[neighbor]:
                distance[neighbor] = new_distance
                path[neighbor] = current_node

    return distance[end], path

graph = {
    'A': {'B': 1, 'C': 4},
    'B': {'A': 1, 'C': 2, 'D': 5},
    'C': {'A': 4, 'B': 2, 'D': 1},
    'D': {'B': 5, 'C': 1}
}

start = 'A'
end = 'D'
print(dijkstra(graph, start, end))
```

2. 交通信息化中的交通流量预测实现：

```python
def predict_traffic_flow(history_flow, alpha=0.5):
    future_flow = []
    for flow in history_flow:
        future_flow.append(alpha * flow + (1 - alpha) * 1000)
    return future_flow

history_flow = [1000, 1200, 1400, 1600, 1800, 2000, 2200, 2400, 2600, 2800, 3000]
predicted_flow = predict_traffic_flow(history_flow)
print(predicted_flow)
```

3. 交通安全中的事故预测实现：

```python
import numpy as np

def predict_accident(history_accident, threshold=0.5):
    predicted_accident = []
    for i in range(len(history_accident)):
        z = np.dot(history_accident[i], np.array([1, 2, 3]))
        p = 1 / (1 + np.exp(-z))
        predicted_accident.append(p > threshold)
    return predicted_accident

history_accident = [
    [1, 2, 3],
    [2, 3, 4],
    [3, 4, 5],
    [4, 5, 6],
    [5, 6, 7],
    [6, 7, 8],
    [7, 8, 9],
    [8, 9, 10],
    [9, 10, 11],
    [10, 11, 12]
]
predicted_accident = predict_accident(history_accident)
print(predicted_accident)
```
