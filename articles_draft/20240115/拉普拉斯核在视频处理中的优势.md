```python
import cv2
import numpy as np

def laplacian(image, ksize=1):
    return cv2.Laplacian(image, cv2.CV_64F, ksize)

def video_stabilization(video_path, output_path):
    cap = cv2.VideoCapture(video_path)
    fourcc = cv2.VideoWriter_fourcc(*'XVID')
    out = cv2.VideoWriter(output_path, fourcc, 20.0, (int(cap.get(cv2.CAP_PROP_FRAME_WIDTH)), int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))))

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        laplacian_image = laplacian(frame)
        stabilized_frame = cv2.addWeighted(frame, 0.7, laplacian_image, 0.3, 0)

        out.write(stabilized_frame)
        cv2.imshow('Stabilized Frame', stabilized_frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    out.release()
    cv2.destroyAllWindows()

# 实现视频稳定化
video_stabilization('input_video.mp4', 'output_video.mp4')
```
