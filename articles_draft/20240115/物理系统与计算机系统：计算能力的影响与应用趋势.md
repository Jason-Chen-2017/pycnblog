## 4.1 物理系统中的代码实例

物理系统中的代码实例包括：

1. 热力学方程的数值求解：

$$
\frac{\partial u}{\partial t} = \alpha \nabla^2 u
$$

代码实例：

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.sparse.linalg import spsolve
from scipy.sparse import tril

def solve_heat_equation(u0, alpha, T, dx, dy, dt):
    Nx, Ny = T.shape
    A = tril(alpha * (np.ones((Nx * Ny, Nx * Ny)) + dx**2 / dx**2 + dy**2 / dy**2))
    b = np.zeros(Nx * Ny)
    b[0] = u0[0, 0]
    b[-1] = u0[-1, -1]
    for i in range(1, Nx):
        b[i] = u0[i, 0] + u0[i - 1, 0]
        b[-i - 1] = u0[-i, -1] + u0[-i - 1, -1]
    x = np.linspace(0, 1, Nx)
    y = np.linspace(0, 1, Ny)
    X, Y = np.meshgrid(x, y)
    u = spsolve(A, b)
    return u, X, Y

u0 = np.array([[1, 0], [0, 1]])
alpha = 0.1
T = np.array([[0, 1], [1, 0]])
dx = 0.1
dy = 0.1
dt = 0.01

u, X, Y = solve_heat_equation(u0, alpha, T, dx, dy, dt)
plt.imshow(u, extent=[0, 1, 0, 1])
plt.colorbar()
plt.show()
```

def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[0]
    left = [x for x in arr[1:] if x < pivot]
    right = [x for x in arr[1:] if x >= pivot]
    return quick_sort(left) + [pivot] + quick_sort(right)

arr = [3, 6, 8, 10, 1, 2, 1]
print(quick_sort(arr))
