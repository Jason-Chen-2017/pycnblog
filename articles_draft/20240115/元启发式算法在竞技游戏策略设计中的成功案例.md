# 4.具体代码实例和详细解释说明

import numpy as np

def initialize_population(population_size, search_space):
    # 初始化搜索空间
    population = np.random.uniform(search_space[0], search_space[1], population_size)
    return population

def evaluate_fitness(population, evaluation_function):
    # 评估函数
    fitness = np.apply_along_axis(evaluation_function, 1, population)
    return fitness

def select_parents(population, fitness, selection_method):
    # 选择父代
    parents = selection_method(population, fitness)
    return parents

def crossover(parents, crossover_method):
    # 交叉操作
    offspring = crossover_method(parents)
    return offspring

def mutation(offspring, mutation_method):
    # 变异操作
    mutated_offspring = mutation_method(offspring)
    return mutated_offspring

def local_search(mutated_offspring, local_search_method):
    # 局部搜索
    local_search_result = local_search_method(mutated_offspring)
    return local_search_result

def global_search(local_search_result, global_search_method):
    # 全局搜索
    global_search_result = global_search_method(local_search_result)
    return global_search_result

def main():
    # 初始化搜索空间
    search_space = (0, 1)
    population_size = 100
    population = initialize_population(population_size, search_space)

    # 定义评估函数
    evaluation_function = lambda x: np.sin(2 * np.pi * x)

    # 评估函数
    fitness = evaluate_fitness(population, evaluation_function)

    # 选择父代
    selection_method = lambda population, fitness: np.argsort(fitness)[-10:]
    parents = select_parents(population, fitness, selection_method)

    # 交叉操作
    crossover_method = lambda parents: np.random.choice(parents, size=population_size, replace=True)
    offspring = crossover(parents, crossover_method)

    # 变异操作
    mutation_method = lambda offspring: np.random.uniform(search_space[0], search_space[1], offspring.shape)
    mutated_offspring = mutation(offspring, mutation_method)

    # 局部搜索
    local_search_method = lambda mutated_offspring: np.random.choice(mutated_offspring, size=population_size, replace=True)
    local_search_result = local_search(mutated_offspring, local_search_method)

    # 全局搜索
    global_search_method = lambda local_search_result: np.random.choice(local_search_result, size=population_size, replace=True)
    global_search_result = global_search(local_search_result, global_search_method)

    # 输出结果
    print("Global search result:", global_search_result)

if __name__ == "__main__":
    main()
