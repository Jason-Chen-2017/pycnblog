```python
import jieba
import re

def entity_recognition(text):
    # 对文本进行分词
    words = jieba.lcut(text)
    
    # 对分词结果进行过滤，只保留实体
    entities = []
    for word in words:
        if re.match(r'\b\w+\b', word):
            entities.append(word)
    
    return entities

# 测试
text = '北京气象局2020年1月1日报告，北京今天天气晴，最高温度2度，最低温度-8度。'
    
entities = entity_recognition(text)
print(entities)
```
