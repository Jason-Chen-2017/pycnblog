```python
import random
import time

def fast_thinking(problem):
    start_time = time.time()
    solutions = []
    for _ in range(10):
        solution = random.choice(problem.solutions)
        if problem.is_valid(solution):
            solutions.append(solution)
    end_time = time.time()
    return solutions, end_time - start_time

def slow_thinking(problem):
    start_time = time.time()
    solutions = []
    for solution in problem.solutions:
        if problem.is_valid(solution):
            solutions.append(solution)
    end_time = time.time()
    return solutions, end_time - start_time

problem = YourProblem()
fast_solutions, fast_time = fast_thinking(problem)
slow_solutions, slow_time = slow_thinking(problem)

print(f"快思维耗时：{fast_time}")
print(f"慢思维解决方案数量：{len(slow_solutions)}")
