import time
import random

def collect_data():
    while True:
        data = random.randint(0, 100)
        print("收集到的数据是：", data)
        time.sleep(1)

def transmit_data(data):
    print("正在传输数据...", data)
    time.sleep(1)

if __name__ == "__main__":
    collect_data()
    transmit_data(100)
```

import sqlite3

def store_data(data):
    conn = sqlite3.connect("data.db")
    cursor = conn.cursor()
    cursor.execute("CREATE TABLE IF NOT EXISTS data (id INTEGER PRIMARY KEY, value INTEGER)")
    for i in range(len(data)):
        cursor.execute("INSERT INTO data (value) VALUES (?)", (data[i],))
    conn.commit()
    conn.close()

if __name__ == "__main__":
    data = [10, 20, 30, 40, 50]
    store_data(data)
```
