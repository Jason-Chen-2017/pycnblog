```python
# 计算平移向量
translation_vector = np.array([1, 0, 0])

# 计算变换矩阵
transform_matrix = rotation_matrix @ translation_vector.reshape(-1, 1)

# 将偶对基与变换矩阵相乘
transformed_basis = basis @ transform_matrix

# 打印结果
print(transformed_basis)
```
