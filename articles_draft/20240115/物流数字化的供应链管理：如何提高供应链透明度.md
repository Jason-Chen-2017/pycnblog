import numpy as np

def transportation_optimization(dist_matrix, capacity):
    """
    dist_matrix: 距离矩阵，形状为 (n, n)
    capacity: 货物容量
    """
    # 初始化最短路径和最短距离
    shortest_path = np.zeros((n, n))
    shortest_distance = np.inf

    # 使用贪心算法找到最短路径
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if shortest_distance > dist_matrix[i][k] + dist_matrix[k][j]:
                    shortest_distance = dist_matrix[i][k] + dist_matrix[k][j]
                    shortest_path[i][j] = k

    # 使用动态规划算法优化最短路径
    for t in range(capacity):
        for i in range(n):
            for j in range(n):
                if shortest_path[i][j] == -1:
                    continue
                for k in range(n):
                    if shortest_path[i][k] != -1 and shortest_path[k][j] != -1:
                        if shortest_distance > dist_matrix[i][k] + dist_matrix[k][j]:
                            shortest_distance = dist_matrix[i][k] + dist_matrix[k][j]
                            shortest_path[i][j] = k

    return shortest_path, shortest_distance

## 4.2 供应链管理算法实例

在供应链管理算法中，我们可以使用 Python 编程语言来实现线性规划算法和动态规划算法。以下是一个简单的供应链管理算法实例：

```python
from scipy.optimize import linprog

def supply_chain_management(cost_matrix, demand):
    """
    cost_matrix: 成本矩阵，形状为 (m, n)
    demand: 需求向量，形状为 (n,)
    """
    # 初始化线性规划问题
    A = np.hstack([np.ones((m, 1)), cost_matrix])
    b = np.hstack([np.zeros(m), demand])
    c = np.hstack([np.zeros(m), -cost_matrix])

    # 使用线性规划算法解决问题
    result = linprog(c, A_ub=A, b_ub=b, method='highs')

    # 输出结果
    print("Status:", result.status)
    print("Optimal value:", result.fun)
    print("Optimal variables:", result.x)

    return result
```
