### 4.1 代码实例

```python
import pymysql

class SMS:
    def __init__(self, host, user, password, db):
        self.conn = pymysql.connect(host, user, password, db)
        self.cursor = self.conn.cursor()

    def insert_student(self, student_id, name, age):
        sql = "INSERT INTO student (student_id, name, age) VALUES (%s, %s, %s)"
        self.cursor.execute(sql, (student_id, name, age))
        self.conn.commit()

    def update_student(self, student_id, name):
        sql = "UPDATE student SET name = %s WHERE student_id = %s"
        self.cursor.execute(sql, (name, student_id))
        self.conn.commit()

    def delete_student(self, student_id):
        sql = "DELETE FROM student WHERE student_id = %s"
        self.cursor.execute(sql, student_id)
        self.conn.commit()

    def query_student(self, student_id):
        sql = "SELECT * FROM student WHERE student_id = %s"
        self.cursor.execute(sql, student_id)
        return self.cursor.fetchone()

# 使用示例
sms = SMS('localhost', 'root', 'password', 'school')
sms.insert_student(1, '张三', 20)
student = sms.query_student(1)
print(student)
sms.update_student(1, '张三更新')
sms.delete_student(1)
```
