```python
import numpy as np
import matplotlib.pyplot as plt

# 定义单位拉普拉斯核
def unit_laplacian(t, dt=1):
    return np.zeros_like(t) - dt

# 定义高斯拉普拉斯核
def gaussian_laplacian(t, sigma=1):
    return -np.heaviside(t, 0) * (1 / (sigma**2)) * np.exp(-t**2 / (2 * sigma**2))

# 定义多项式拉普拉斯核
def polynomial_laplacian(t, order=2):
    return np.polyval([-order, 1], t)

# 信道模型建立
def channel_model(t, laplacian_kernel):
    return laplacian_kernel(t) * np.random.randn(len(t))

# 调制解调
def modulation_demodulation(s, laplacian_kernel):
    return laplacian_kernel(s)

# 测试
t = np.arange(0, 10, 0.1)
s = np.sin(2 * np.pi * t)
laplacian_kernel = gaussian_laplacian(t)

# 信道模型建立
h = channel_model(t, laplacian_kernel)
y = s + h

# 调制解调
y_demodulated = modulation_demodulation(y, laplacian_kernel)

# 绘制结果
plt.figure()
plt.plot(t, s, label='信号')
plt.plot(t, h, label='信道噪声')
plt.plot(t, y, label='信号+信道噪声')
plt.plot(t, y_demodulated, label='解调后信号')
plt.legend()
plt.show()
```
