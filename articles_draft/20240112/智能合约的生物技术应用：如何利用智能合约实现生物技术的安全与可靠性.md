```solidity
pragma solidity ^0.5.0;

contract BiologicalDataProtection {
    address public owner;
    mapping(address => bool) public isAuthorized;

    event DataProtected(address indexed user, uint256 indexed dataId);

    constructor() public {
        owner = msg.sender;
    }

    function protectData(uint256 dataId) public {
        require(msg.sender == owner, "Only owner can protect data.");
        require(isAuthorized[msg.sender], "User is not authorized.");

        isAuthorized[msg.sender] = true;
        emit DataProtected(msg.sender, dataId);
    }

    function authorizeUser(address user) public {
        require(msg.sender == owner, "Only owner can authorize user.");

        isAuthorized[user] = true;
    }

    function revokeAuthorization(address user) public {
        require(msg.sender == owner, "Only owner can revoke authorization.");

        isAuthorized[user] = false;
    }
}
```
