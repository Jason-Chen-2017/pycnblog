def bat_algorithm(f, x_low, x_upp, n_bats, n_iter, alpha, gamma, beta, rho):

    x = x_low + rho * (x_upp - x_low)
    v = np.zeros_like(x)
    f_min = np.inf
    for i in range(n_bats):
        f_min = min(f_min, f(x[i]))

    # 蝙蝠的速度更新公式
    for t in range(n_iter):
        for i in range(n_bats):
            # 目标函数的梯度公式
            grad = np.zeros_like(x[i])
            for j in range(n_dim):
                x_temp = x[i].copy()
                x_temp[j] += 1
                grad[j] = (f(x_temp) - f(x[i])) / (x_temp[j] - x[i][j])

            # 蝙蝠的定位公式
            x_temp = x[i].copy()
            x_temp += grad * alpha
            if np.random.rand() > 0.5:
                x_temp += (x_upp - x_temp) * beta
            else:
                x_temp += (x_temp - x_low) * beta
            x_temp = np.clip(x_temp, x_low, x_upp)

            # 蝙蝠的速度更新公式
            v[i] = v[i] + (x[i] - x_temp) * gamma
            x[i] = x[i] + v[i]

            # 目标函数的梯度公式
            grad = np.zeros_like(x[i])
            for j in range(n_dim):
                x_temp = x[i].copy()
                x_temp[j] -= 1
                grad[j] = (f(x_temp) - f(x[i])) / (x_temp[j] - x[i][j])

            # 蝙蝠的定位公式
            x_temp = x[i].copy()
            x_temp -= grad * alpha
            if np.random.rand() > 0.5:
                x_temp += (x_upp - x_temp) * beta
            else:
                x_temp += (x_temp - x_low) * beta
            x_temp = np.clip(x_temp, x_low, x_upp)

            # 蝙蝠的速度更新公式
            v[i] = v[i] + (x[i] - x_temp) * gamma
            x[i] = x[i] + v[i]

    return x[np.argmin(f(x))], f_min
