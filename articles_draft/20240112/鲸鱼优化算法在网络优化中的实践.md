import numpy as np

def f(x):
    return -x**2 + 4*x

def update_position(x, v, pbest, gbest, w, c1, c2):
    r1 = np.random.rand()
    r2 = np.random.rand()
    v = w * v + c1 * r1 * (pbest - x) + c2 * r2 * (gbest - x)
    x = x + v
    return x

def whale_optimization_algorithm(x_init, n, max_iter, w, c1, c2):
    x = x_init
    v = np.zeros(n)
    pbest = x.copy()
    gbest = x.copy()
    f_best = np.min(f(x))

    for i in range(max_iter):
        for j in range(n):
            x[j] = update_position(x[j], v[j], pbest[j], gbest[j], w, c1, c2)
            f_value = f(x[j])
            if f_value < f_best[j]:
                f_best[j] = f_value
                pbest[j] = x[j]
                if f_value < f_best:
                    gbest = x.copy()
                    f_best = f_best[j]

    return gbest, f_best

x_init = np.random.rand(10, 1) * 100
n = 10
max_iter = 1000
w = 0.9
c1 = 2
c2 = 2

gbest, f_best = whale_optimization_algorithm(x_init, n, max_iter, w, c1, c2)
print("最佳解:", gbest)
print("最佳适应度值:", f_best)
