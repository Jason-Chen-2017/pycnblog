```python
def random_sequence(length, alphabet):
    return ''.join(random.choice(alphabet) for _ in range(length))

def similarity(seq1, seq2):
    count = sum(1 for a, b in zip(seq1, seq2) if a == b)
    return count / len(seq1)

def monte_carlo_alignment(seq1, seq2, iterations):
    total_similarity = 0
    for _ in range(iterations):
        seq1_rand = random_sequence(len(seq1), 'ACGT')
        seq2_rand = random_sequence(len(seq2), 'ACGT')
        similarity_rand = similarity(seq1_rand, seq2_rand)
        total_similarity += similarity_rand
    return total_similarity / iterations

seq1 = 'ATGCGATACG'
seq2 = 'ATGCGATACG'
iterations = 10000
print(monte_carlo_alignment(seq1, seq2, iterations))
```
