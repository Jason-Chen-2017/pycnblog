# 创建一个有向无环图

# 添加节点
G.add_node('A')
G.add_node('B')
G.add_node('C')

# 添加边
G.add_edge('A', 'B')
G.add_edge('B', 'C')

# 设置条件概率
G.nodes['A']['prob'] = 0.5
G.nodes['B']['prob'] = 0.5
G.nodes['C']['prob'] = 0.5

# 计算条件概率

print(P_A_given_B)
