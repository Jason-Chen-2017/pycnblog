import numpy as np

# 建立数字模型
def create_digital_model(shape, size, quality, performance):
    model = {
        'shape': shape,
        'size': size,
        'quality': quality,
        'performance': performance
    }
    return model

# 实现数据同步
def sync_data(physical, digital):
    digital['shape'] = physical['shape']
    digital['size'] = physical['size']
    digital['quality'] = physical['quality']
    digital['performance'] = physical['performance']

# 实现交互
def interact(physical, digital):
    if physical['shape'] != digital['shape']:
        digital['shape'] = physical['shape']
    if physical['size'] != digital['size']:
        digital['size'] = physical['size']
    if physical['quality'] != digital['quality']:
        digital['quality'] = physical['quality']
    if physical['performance'] != digital['performance']:
        digital['performance'] = physical['performance']

# 实现优化
def optimize(physical, digital):
    cost = abs(physical['shape'] - digital['shape']) + abs(physical['size'] - digital['size']) + abs(physical['quality'] - digital['quality']) + abs(physical['performance'] - digital['performance'])
    return cost

# 测试

    'shape': 10,
    'size': 20,
    'quality': 30,
    'performance': 40
}
