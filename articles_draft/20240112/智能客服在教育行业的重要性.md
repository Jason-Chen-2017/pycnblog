```python
import numpy as np

def recommend_resource(user_profile, resource_profile):
    # 计算用户与资源之间的相似度
    similarity = np.dot(user_profile, resource_profile) / (np.linalg.norm(user_profile) * np.linalg.norm(resource_profile))
    # 返回相似度最高的资源编号
    return np.argmax(similarity)
```
