       dot_product = np.dot(A, B)

       norm_B = np.linalg.norm(B)
       return dot_product / (norm_A * norm_B)

   def content_based_recommendation(user_profile, content_features, threshold):
       similarities = []
       for content in content_features:
           similarity = cosine_similarity(user_profile, content)

       return similarities
