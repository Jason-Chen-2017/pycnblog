3. 代码实例：

以下是一个使用Python和NLTK库进行文本分类的代码实例：

```python
import nltk
from nltk.classify import NaiveBayesClassifier
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize

# 数据预处理
def preprocess(text):
    tokens = word_tokenize(text)
    tokens = [t.lower() for t in tokens if t.isalpha()]
    tokens = [t for t in tokens if t not in stopwords.words('english')]
    return tokens

# 特征提取
def extract_features(tokens):
    return {t: True for t in tokens}

# 数据集
documents = [
    ('This is the first document.', 'positive'),
    ('This document is the second document.', 'negative'),
    ('And this is the third one.', 'positive'),
    ('Is this the first document?', 'negative')
]

# 数据预处理和特征提取
featuresets = [(extract_features(preprocess(d)), c) for (d, c) in documents]

# 模型训练
classifier = NaiveBayesClassifier.train(featuresets)

# 模型评估
print(classifier.classify(extract_features(preprocess('This is the fourth document.'))))
```
