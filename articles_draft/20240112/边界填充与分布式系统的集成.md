```python
class BoundaryFill:
    def __init__(self, data, transaction):
        self.data = data
        self.transaction = transaction

    def fill_data(self):
        filled_data = 0
        for i in range(len(self.data)):
            filled_data += self.data[i]
        return filled_data

    def fill_transaction(self):
        filled_transaction = 0
        for i in range(len(self.transaction)):
            filled_transaction += self.transaction[i]
        return filled_transaction

    def integration(self):
        integration_data = 0
        for i in range(len(self.data)):
            integration_data += self.data[i]
        integration_transaction = 0
        for i in range(len(self.transaction)):
            integration_transaction += self.transaction[i]
        return integration_data, integration_transaction

if __name__ == "__main__":
    data = [1, 2, 3]
    transaction = [4, 5, 6]
    boundary_fill = BoundaryFill(data, transaction)
    filled_data = boundary_fill.fill_data()
    filled_transaction = boundary_fill.fill_transaction()
    integration_data, integration_transaction = boundary_fill.integration()
    print("Filled Data:", filled_data)
    print("Filled Transaction:", filled_transaction)
    print("Integration Data:", integration_data)
    print("Integration Transaction:", integration_transaction)
```
