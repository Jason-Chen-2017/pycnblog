def greedy_resource_allocation(projects, budget):
    selected_projects = []
    remaining_budget = budget
    while remaining_budget > 0 and projects:
        project = max(projects, key=lambda x: x['profit'] / x['cost'])
        if project['cost'] <= remaining_budget:
            selected_projects.append(project)
            remaining_budget -= project['cost']
        projects.remove(project)
    return selected_projects

    {'name': 'Project A', 'cost': 100, 'profit': 150},
    {'name': 'Project B', 'cost': 200, 'profit': 250},
    {'name': 'Project C', 'cost': 150, 'profit': 200},
    {'name': 'Project D', 'cost': 300, 'profit': 350}
]

budget = 500
selected_projects = greedy_resource_allocation(projects, budget)
