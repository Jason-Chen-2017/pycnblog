$$

$$

$$

```python
from sklearn.metrics.pairwise import cosine_similarity

# 计算商品之间的相似度
def calculate_similarity(matrix):
    similarity = cosine_similarity(matrix)
    return similarity

# 获取最相似的商品
def get_similar_items(item_id, similarity):
    similar_items = sorted(enumerate(similarity[item_id]), key=lambda x: x[1], reverse=True)
    return similar_items[1:5]
```

```python
    vectorizer = TfidfVectorizer()
    item_matrix = vectorizer.fit_transform(user_items)
    user_interest = cosine_similarity(item_matrix, item_matrix)
    return user_interest
```
