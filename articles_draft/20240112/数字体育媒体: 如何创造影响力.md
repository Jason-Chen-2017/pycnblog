## 4.2 内容推荐

可以使用协同过滤算法来推荐内容：

```python
from scipy.spatial.distance import cosine
from scipy.cluster.vq import vq, kmeans
from sklearn.metrics.pairwise import pairwise_distances

# 计算用户之间的相似度
def user_similarity(ratings, user_id):
    user_ratings = ratings[user_id]
    distances = pairwise_distances(ratings, user_ratings)
    similarities = 1 - distances
    return similarities

# 推荐内容
def recommend_content(ratings, user_id, num_recommendations):
    user_similarities = user_similarity(ratings, user_id)
    user_similarities = user_similarities[user_id]
    similarities_sorted = sorted(user_similarities, reverse=True)
    recommendations = []
    for similarity in similarities_sorted[:num_recommendations]:
        recommended_user_id = user_similarities.tolist().index(similarity)
        recommendations.append(recommended_user_id)
    return recommendations
```
