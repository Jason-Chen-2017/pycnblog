# 4. 具体代码实例和详细解释说明

```python
import numpy as np

def vector_outer_product(A, B):
    # 计算向量A和向量B的模
    norm_A = np.linalg.norm(A)
    norm_B = np.linalg.norm(B)
    
    # 计算向量A和向量B之间的夹角
    angle = np.arccos(np.dot(A, B) / (norm_A * norm_B))
    
    # 计算外积
    outer_product = norm_A * norm_B * np.sin(angle)
    
    return outer_product

# 示例向量
A = np.array([1, 2, 3])
B = np.array([4, 5, 6])

# 计算外积
result = vector_outer_product(A, B)
print(result)
```
