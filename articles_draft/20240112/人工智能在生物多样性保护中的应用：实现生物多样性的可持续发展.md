from sklearn.base import BaseEstimator, RegressorMixin
import numpy as np

class GeneticAlgorithm(BaseEstimator, RegressorMixin):
    def __init__(self, population_size, mutation_rate, crossover_rate):
        self.population_size = population_size
        self.mutation_rate = mutation_rate
        self.crossover_rate = crossover_rate

    def fit(self, X, y):
        # 初始化种群
        population = np.random.rand(self.population_size, X.shape[1])

        for _ in range(1000):
            # 选择最佳个体
            best_individuals = np.argsort(y[population])[:self.population_size // 2]

            # 产生新一代
            new_population = np.random.rand(self.population_size, X.shape[1])
            for i in range(self.population_size // 2):
                new_population[i] = population[best_individuals[i]]
            for i in range(self.population_size // 2, self.population_size):
                parent1 = population[best_individuals[i - self.population_size // 2]]
                parent2 = population[best_individuals[i - self.population_size // 2 + 1]]
                child = np.random.randint(0, 2)
                if child == 0:
                    new_population[i] = parent1
                else:
                    new_population[i] = parent2

            # 变异
            for i in range(self.population_size):
                if np.random.rand() < self.mutation_rate:
                    new_population[i] += np.random.randn(X.shape[1]) * 0.1

            # 交叉
            for i in range(self.population_size):
                if np.random.rand() < self.crossover_rate:
                    child1 = np.random.randint(0, 2)
                    child2 = np.random.randint(0, 2)
                    new_population[i] = child1 * parent1 + child2 * parent2

            # 更新种群
            population = new_population

        # 返回最佳个体
        self.best_individual = population[np.argmin(y[population])]

    def predict(self, X):
        return self.best_individual
