3. 卷积的Python实现：

```python
import numpy as np

def convolution(x, h):
    N = len(x)
    y = np.zeros(N)
    for n in range(N):
        for m in range(N):
            y[n] += x[m] * h[n - m]
    return y

x = np.sin(2 * np.pi * 50 * t) + np.cos(2 * np.pi * 100 * t)
h = np.sin(2 * np.pi * 50 * t)
y = convolution(x, h)
plt.plot(y)
plt.show()
```

```python
import numpy as np

def filtering(x, h):
    N = len(x)
    y = np.zeros(N)
    for n in range(N):
        y[n] = x[n] * h[n]
    return y

h = np.sin(2 * np.pi * 50 * t)
y = filtering(x, h)

```

```python
import numpy as np

def audio_compression(x, ratio):
    N = len(x)
    y = np.zeros(N)
    for n in range(N):
        y[n] = x[n] * ratio
    return y

x = np.sin(2 * np.pi * 50 * t) + np.cos(2 * np.pi * 100 * t)
ratio = 0.5
y = audio_compression(x, ratio)
plt.plot(y)
plt.show()
```
