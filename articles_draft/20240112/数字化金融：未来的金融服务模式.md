```python
import hashlib
import os
import binascii

def create_block(index, timestamp, data, previous_hash):
    block = f"{index}{timestamp}{data}{previous_hash}".encode('utf-8')
    block_hash = hashlib.sha256(block).hexdigest()
    return block, block_hash

def create_genesis_block():
    return create_block(0, "2021-01-01", "Genesis Block", "0")

def create_new_block(previous_block, transaction):
    index = previous_block[0] + 1
    timestamp = str(int(time.time()))
    data = transaction
    previous_hash = previous_block[1]
    block, block_hash = create_block(index, timestamp, data, previous_hash)
    return block, block_hash

def add_transaction(transaction, blockchain):
    last_block = blockchain[-1]
    previous_hash = last_block[1]
    new_block, new_block_hash = create_new_block(last_block, transaction)
    blockchain.append(new_block)
    blockchain.append(new_block_hash)
    return blockchain

def is_valid_blockchain(blockchain):
    for i in range(1, len(blockchain)):
        current_block = blockchain[i]
        previous_block = blockchain[i - 1]
        if current_block[:32] != previous_block[32:64]:
            return False
    return True

def create_transaction(sender, recipient, amount):
    return f"{sender}{recipient}{amount}".encode('utf-8')

def main():
    blockchain = create_genesis_block()
    transaction = create_transaction("Alice", "Bob", 100)
    blockchain = add_transaction(transaction, blockchain)
    print(blockchain)
    print(is_valid_blockchain(blockchain))

if __name__ == "__main__":
    main()
```
