from sklearn.linear_model import LinearRegression

# 人类直觉数据
X_directive = np.array([[1], [2], [3], [4], [5]])
y_directive = np.array([2, 4, 6, 8, 10])

# 机器学习数据
X_ml = np.array([[1], [2], [3], [4], [5]])
y_ml = np.array([1, 3, 5, 7, 9])

# 融合数据
X_fusion = np.vstack((X_directive, X_ml))
y_fusion = np.hstack((y_directive, y_ml))

# 训练融合模型
model = LinearRegression()
model.fit(X_fusion, y_fusion)

# 预测
X_test = np.array([[6], [7]])
y_pred = model.predict(X_test)

print(y_pred)
