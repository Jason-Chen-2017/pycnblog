# 5.具体代码实例和解释说明

在实际应用中，可以使用以下代码实例来实现物流过程的实时监控和控制：

```python
import time

class Material:
    def __init__(self, name, weight):
        self.name = name
        self.weight = weight

class Vehicle:
    def __init__(self, name, capacity):
        self.name = name
        self.capacity = capacity

class Warehouse:
    def __init__(self, name, stock):
        self.name = name
        self.stock = stock

class Logistics:
    def __init__(self):
        self.materials = []
        self.vehicles = []
        self.warehouses = []

    def add_material(self, material):
        self.materials.append(material)

    def add_vehicle(self, vehicle):
        self.vehicles.append(vehicle)

    def add_warehouse(self, warehouse):
        self.warehouses.append(warehouse)

    def load_material(self, vehicle, material):
        if vehicle.capacity >= material.weight:
            vehicle.capacity -= material.weight
            material.weight = 0
            return True
        else:
            return False

    def unload_material(self, vehicle, material):
        if vehicle.capacity >= material.weight:
            vehicle.capacity += material.weight
            material.weight = 0
            return True
        else:
            return False

    def transport_material(self, vehicle, material):
        if self.load_material(vehicle, material):
            time.sleep(1)
            self.unload_material(vehicle, material)

if __name__ == "__main__":
    logistics = Logistics()

    material1 = Material("iron", 100)
    material2 = Material("steel", 200)
    material3 = Material("copper", 150)

    vehicle1 = Vehicle("truck", 300)
    vehicle2 = Vehicle("train", 500)

    warehouse1 = Warehouse("factory", 1000)
    warehouse2 = Warehouse("market", 500)

    logistics.add_material(material1)
    logistics.add_material(material2)
    logistics.add_material(material3)

    logistics.add_vehicle(vehicle1)
    logistics.add_vehicle(vehicle2)

    logistics.add_warehouse(warehouse1)
    logistics.add_warehouse(warehouse2)

    logistics.transport_material(vehicle1, material1)
    logistics.transport_material(vehicle2, material2)
    logistics.transport_material(vehicle1, material3)
```
