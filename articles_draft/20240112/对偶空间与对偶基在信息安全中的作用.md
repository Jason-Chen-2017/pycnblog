def rsa_key_gen(bits):
    p = random.getrandbits(bits)
    q = random.getrandbits(bits)
    while p == q or gcd(p-1, q-1) != 1:
        p = random.getrandbits(bits)
        q = random.getrandbits(bits)
    N = p * q
    phi_N = (p-1) * (q-1)
    d = pow(e, phi_N-1, N)
    return (e, N, d, N)

def rsa_encrypt(e, N, m):
    c = pow(m, e, N)
    return c

def rsa_decrypt(d, N, c):
    m = pow(c, d, N)
    return m

def gcd(a, b):
    while b != 0:
        a, b = b, a % b
    return a

    F = np.fft.fft(f)
    return F

    f = np.fft.ifft(F)
    return f
