# 使用并行计算
def parallel_add(v1, v2):
    # 将数据分块
    block_size = 2
    blocks = [v1[i:i+block_size] for i in range(0, len(v1), block_size)]
    blocks += [v2[i:i+block_size] for i in range(0, len(v2), block_size)]

    # 在每个块上执行向量加法
    results = [np.sum(block) for block in blocks]

    # 将结果汇总到一个单一的向量中
    result = np.concatenate(results)
    return result
