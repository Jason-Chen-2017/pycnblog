```python
class RuleEngine:
    def __init__(self):
        self.rules = []

    def load_rules(self, rules_file):
        with open(rules_file, 'r') as f:
            for line in f:
                self.rules.append(Rule(line))

    def match(self, data):
        matched_rules = []
        for rule in self.rules:
            if rule.match(data):
                matched_rules.append(rule)
        return matched_rules

    def execute(self, data):
        matched_rules = self.match(data)
        for rule in matched_rules:
            rule.execute(data)

    def run(self, data):
        self.load_rules('rules.txt')
        self.execute(data)
```
