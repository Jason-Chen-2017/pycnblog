```python
# 设置API地址

# 设置设备密码
device_password = "123456"

    print("设备配置成功")
else:
    print("设备配置失败")
```

### 4.3网络连接代码实例
```python
import socket

# 设置服务器地址
server_address = ("192.168.1.1", 12345)

# 创建套接字
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

# 连接服务器
sock.connect(server_address)

# 发送数据
data = "Hello, World!"
sock.sendall(data.encode())

# 接收数据
data = sock.recv(1024)
print(data.decode())

# 关闭套接字
sock.close()
```
