def gps_location(latitude, longitude, altitude):
    # 计算地球半径
    earth_radius = 6371.0

    # 计算经度差
    delta_longitude = np.radians(longitude - latitude)

    # 计算纬度差
    delta_latitude = np.radians(latitude)

    # 计算高度差
    delta_altitude = altitude

    # 计算距离
    a = np.sin(delta_latitude / 2)**2 + np.cos(np.radians(latitude)) * np.cos(np.radians(latitude)) * np.sin(delta_longitude / 2)**2
    c = 2 * np.arctan(np.sqrt(a), np.sqrt(1 - a))
    distance = earth_radius * c

    return distance
