## 4.1 卫星的轨道计算

```python
import numpy as np

def calculate_orbit(initial_position, initial_velocity, time):
    G = 6.67430e-11  # m^3 kg^-1 s^-2
    mass = 5.972e24  # kg
    position = np.array(initial_position)
    velocity = np.array(initial_velocity)

    acceleration = G * mass * position / np.linalg.norm(position)**3
    position += velocity * time
    velocity += acceleration * time

    return position, velocity
```

def calculate_earth_observation(position, time):
    G = 6.67430e-11  # m^3 kg^-1 s^-2
    mass = 5.972e24  # kg
    velocity = position[1:]
    acceleration = G * mass * position / np.linalg.norm(position)**3
