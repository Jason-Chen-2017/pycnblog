import numpy as np
import random

# 初始化鲸鱼群
def init_whale_swarm(num_whales, search_space):
    whales = np.random.uniform(search_space[0], search_space[1], num_whales)
    return whales

# 计算鲸鱼群的适应度
def compute_fitness(whales, fitness_func):
    fitness_values = [fitness_func(x) for x in whales]
    return fitness_values

# 更新鲸鱼群的位置
def update_whale_position(whales, A, D):
    new_whales = whales + A * D
    return new_whales

# 更新鲸鱼群的速度
def update_whale_speed(whales, X_best, R):
    new_speeds = np.array([
        alpha * X + beta * X_best + gamma * R
        for X, alpha, beta, gamma in zip(whales, A, B, C)
    ])
    return new_speeds

# 鲸鱼优化算法主函数
def whale_optimization(search_space, fitness_func, num_iterations):
    num_whales = 50
    A = np.array([random.uniform(0.5, 1.5) for _ in range(num_whales)])
    B = np.array([random.uniform(0.5, 1.5) for _ in range(num_whales)])
    C = np.array([random.uniform(0.5, 1.5) for _ in range(num_whales)])
    X_best = np.random.uniform(search_space[0], search_space[1])

    whales = init_whale_swarm(num_whales, search_space)
    fitness_values = compute_fitness(whales, fitness_func)

    for _ in range(num_iterations):
        for i in range(num_whales):
            R = np.random.randn(search_space.shape[0])
            D = update_whale_speed(whales, X_best, R)
            new_whales = update_whale_position(whales, A[i], D[i])
            fitness_values[i] = compute_fitness(new_whales, fitness_func)

        X_best = np.argmax(fitness_values)

    return X_best, fitness_values[X_best]

# 测试鲸鱼优化算法
if __name__ == '__main__':
    search_space = (-5, 5)
    fitness_func = lambda x: x**2
    num_iterations = 100

    X_best, best_fitness = whale_optimization(search_space, fitness_func, num_iterations)
    print('最佳解:', X_best)
    print('最佳适应度:', best_fitness)
