import pandas as pd

def process_data(data):
    # 数据清洗
    data = data.dropna()

    # 数据转换
    data['timestamp'] = pd.to_datetime(data['timestamp'])
    data['value'] = data['value'].astype(float)

    # 数据处理流程
    processed_data = data.groupby('timestamp').mean().reset_index()
