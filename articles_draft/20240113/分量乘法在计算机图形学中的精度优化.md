# 2.核心概念与联系

$$

a_1 & b_1 \\
a_2 & b_2 \\
a_3 & b_3
\end{bmatrix}
\begin{bmatrix}
c_1 & d_1 \\
c_2 & d_2 \\
c_3 & d_3
\end{bmatrix}
=
\begin{bmatrix}
a_1c_1 + b_1d_1 & a_1c_2 + b_1d_2 \\
a_2c_1 + b_2d_1 & a_2c_2 + b_2d_2 \\
a_3c_1 + b_3d_1 & a_3c_2 + b_3d_2
\end{bmatrix}
$$

    float e22 = a2 * c2 + b2 * d2;
