```python
import numpy as np

def monitor_psychological_indicators(data):
    # 计算心理指标的均值和标准差
    mean = np.mean(data)
    std = np.std(data)
    return mean, std

def analyze_psychological_indicators(old_mean, old_std, data_mean, data_std, alpha):
    new_mean = alpha * old_mean + (1 - alpha) * data_mean
    new_std = alpha * old_std + (1 - alpha) * data_std
    return new_mean, new_std

def adjust_psychological_indicators(old_mean, old_std, data_mean, data_std):
    new_mean = (old_mean + data_mean) / 2
    new_std = (old_std + data_std) / 2
    return new_mean, new_std
```
