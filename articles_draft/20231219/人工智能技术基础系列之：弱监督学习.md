```python
import numpy as np
import tensorflow as tf

# 数据准备
y = np.linspace(0, np.pi, 100)
x = np.sin(y) + np.random.normal(0, 0.1, 100)

# 模型训练

    tf.keras.layers.Dense(64, activation='relu', input_shape=(1,)),
```
