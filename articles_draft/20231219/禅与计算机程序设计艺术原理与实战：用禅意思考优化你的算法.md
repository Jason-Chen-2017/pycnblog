    m = len(s1)
    n = len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i - 1] == s2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1

    return dp[m][n]

def dijkstra(graph, start):

    dist[start] = 0

    while pq:
        _, u = heapq.heappop(pq)
        for v, w in graph[u].items():
