# 3.1 贝叶斯决策的核心算法原理
贝叶斯决策的核心算法原理是基于贝叶斯定理的。贝叶斯定理是一种概率推理方法，它可以帮助我们根据已有的信息来推断未知事件的概率。在推荐系统中，我们可以使用贝叶斯定理来计算用户对某个项目的喜好概率，从而为用户提供更准确的推荐。

贝叶斯定理的公式为：

$$
P(A|B) = \frac{P(B|A)P(A)}{P(B)}
$$

其中，$P(A|B)$ 表示当事件B发生时，事件A的概率；$P(B|A)$ 表示当事件A发生时，事件B的概率；$P(A)$ 表示事件A的概率；$P(B)$ 表示事件B的概率。

2. 收集项目的特征数据，例如项目的标签、描述、类别等。
3. 使用贝叶斯定理计算用户对某个项目的喜好概率。具体步骤如下：
   - 计算项目的条件概率$P(B|A)$，即当用户喜欢某个项目时，项目的概率。
   - 计算用户的条件概率$P(A|B)$，即当用户喜欢某个项目时，用户的概率。
   - 计算用户的概率$P(A)$，即用户的概率。

import numpy as np

# 用户的历史行为数据
user_history = [
    {'user_id': 1, 'item_id': 1, 'action': 'click'},
    {'user_id': 1, 'item_id': 2, 'action': 'click'},
    {'user_id': 1, 'item_id': 3, 'action': 'click'},
    {'user_id': 2, 'item_id': 1, 'action': 'click'},
    {'user_id': 2, 'item_id': 2, 'action': 'click'},
    {'user_id': 2, 'item_id': 3, 'action': 'click'},
    {'user_id': 2, 'item_id': 4, 'action': 'click'},
]

# 项目的特征数据
item_features = {
    'item_id_1': {'category': '电子产品', 'tags': ['手机', '智能']},
    'item_id_2': {'category': '电子产品', 'tags': ['平板', '智能']},
    'item_id_3': {'category': '电子产品', 'tags': ['手机', '苹果']},
    'item_id_4': {'category': '服装', 'tags': ['潮流', '裤子']},
}

# 计算项目的条件概率
def calc_conditional_probability(user_history, item_features):
    item_count = {}
    user_count = {}
    item_user_count = {}
    for history in user_history:
        user_id = history['user_id']
        item_id = history['item_id']
        action = history['action']
        if user_id not in user_count:
            user_count[user_id] = 0
        if item_id not in item_count:
            item_count[item_id] = 0
        if (user_id, item_id) not in item_user_count:
            item_user_count[(user_id, item_id)] = 0
        if action == 'click':
            user_count[user_id] += 1
            item_count[item_id] += 1
            item_user_count[(user_id, item_id)] += 1
    for user_id, count in user_count.items():
        for item_id, count in item_count.items():
            if user_id not in item_user_count or item_id not in item_user_count[user_id]:
                item_user_count[(user_id, item_id)] = 0
            item_user_count[(user_id, item_id)] += min(count, item_count[item_id])
    for user_id, item_id in item_user_count.keys():
        item_features[item_id]['probability'] = item_user_count[(user_id, item_id)] / user_count[user_id]
    return item_features

# 根据用户对某个项目的喜好概率，为用户推荐个性化的项目
def recommend(item_features, user_id):
    recommended_items = []
    for item_id, features in item_features.items():
        if 'probability' not in features:
            calc_conditional_probability(user_history, item_features)
        probability = features['probability']
        if probability > 0.5:
            recommended_items.append((item_id, probability))
    recommended_items.sort(key=lambda x: x[1], reverse=True)
    return recommended_items

# 测试
item_features = calc_conditional_probability(user_history, item_features)
recommended_items = recommend(item_features, 1)
print(recommended_items)
