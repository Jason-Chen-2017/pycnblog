    return pop

    # Calculate the drag forces
    drag_forces = fish_drag * (fish_speed**2) * fish_pos

    # Calculate the goal attraction forces
    goal_forces = goal_drag * (goal_vel**2) * (goal_pos - fish_pos)

    # Calculate the net forces
    net_forces = drag_forces + goal_forces

    return net_forces

    # Update the fish speeds
    pop = fish_speed + forces / fish_mass

    return pop

    # Update the fish positions
    pop = fish_pos + fish_speed

    return pop

def select_best(pop, fitness):
    # Select the fittest fish
    best_fish = pop[np.argmin(fitness)]

    return best_fish
