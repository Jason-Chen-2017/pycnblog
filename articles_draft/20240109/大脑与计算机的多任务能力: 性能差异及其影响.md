```python
import numpy as np

class BrainMultiTask:
    def __init__(self):
        self.input_size = 10
        self.hidden_size = 50
        self.output_size = 2
        self.weights = np.random.rand(self.hidden_size, self.input_size)
        self.bias = np.zeros(self.hidden_size)

    def forward(self, x):
        self.hidden = np.dot(x, self.weights) + self.bias
        self.output = 1 / (1 + np.exp(-self.hidden))
        return self.output

    def train(self, x, y, learning_rate):
        self.forward(x)
        self.predicted = self.forward(x)
        self.error = y - self.predicted
        self.delta = self.error * (self.predicted * (1 - self.predicted))
        self.weights -= learning_rate * np.dot(self.hidden.T, self.delta)
        self.bias -= learning_rate * np.sum(self.delta)

    def predict(self, x):
        return self.forward(x)
```
