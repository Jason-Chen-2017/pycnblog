# 定义物流规划问题
def tsp(distances):
    row_indices = np.arange(len(distances))
    col_indices = np.arange(len(distances[0]))
    cost = distances
    row_optimal = linear_sum_assignment(cost, ind2sub=lambda ind: (ind[0], ind[1] + len(ind[0])))
