```python
import numpy as np

def greedy_algorithm(capacity, demand):
    index = np.argsort(-capacity / demand)
    allocation = np.zeros(len(capacity))
    total_demand = 0
    for i in index:
        allocation[i] = min(capacity[i], demand - total_demand)
        total_demand += allocation[i]
    return allocation
```

2.价格竞争算法实现：

```python
import numpy as np

def auction_algorithm(supply, demand):
    price = np.full(len(supply), np.nan)
    for i in range(len(supply)):
        supply_i = supply[i]
        demand_i = demand[i]
        price[i] = (supply_i + demand_i) / 2
    return price
```
