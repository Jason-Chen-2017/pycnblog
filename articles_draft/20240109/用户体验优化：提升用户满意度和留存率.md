```python
import pandas as pd

data = pd.read_csv('user_behavior.csv')
data['timestamp'] = pd.to_datetime(data['timestamp'])
data['timestamp'] = (data['timestamp'] - pd.to_datetime('2021-01-01')).dt.days
data['product'] = data['product'].astype('category').cat.codes
```

### 4.3计算用户相似度

接下来，我们需要计算用户之间的相似度。例如，我们可以使用皮尔森相关系数来计算用户相似度。

```python
from scipy.stats import pearsonr

def user_similarity(user_a, user_b):
    similarity = pearsonr(user_a, user_b)
    return similarity[0]

user_a = data[data['user_id'] == 1]['product'].values
user_b = data[data['user_id'] == 2]['product'].values
similarity = user_similarity(user_a, user_b)
print(f'用户A和用户B的相似度：{similarity}')
```

### 4.4推断用户可能喜欢的项目

最后，我们需要根据用户相似度来推断用户可能喜欢的项目。例如，我们可以将用户的历史行为与其他用户的历史行为进行比较，并计算出相似度最高的项目。

```python
def recommend_items(user_id, data):
    user_history = data[data['user_id'] == user_id]['product'].values
    similarities = []
    for _, row in data.iterrows():
        if row['user_id'] != user_id:
            similarity = user_similarity(user_history, row['product'].values)
            similarities.append((row['user_id'], similarity))
    similarities.sort(key=lambda x: x[1], reverse=True)
    recommended_items = data[data['user_id'].isin([user_id] + [i[0] for i in similarities[:5]])]['product'].unique()
    return recommended_items

recommended_items = recommend_items(1, data)
print(f'用户1可能喜欢的项目：{recommended_items}')
```
