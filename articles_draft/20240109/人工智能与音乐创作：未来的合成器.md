```python
# 收集音乐数据

# 构建马尔可夫链模型
transition_matrix = np.zeros((len(data), len(data)))
for i in range(len(data)):
    for j in range(len(data)):
        transition_matrix[i, j] = count(data[i-1], data[j]) / sum(count(data[i-1], d) for d in data)

# 生成新音乐序列
new_data = []
current_data = data[0]
for _ in range(10):
    new_data.append(current_data)
    current_data = np.random.choice(data, p=transition_matrix[data.index(current_data)])

# 输出新音乐序列
print(new_data)
```
