```python
import numpy as np

class FishSwarmOptimization:
    def __init__(self, num_fish, fish_params, target_func):
        self.num_fish = num_fish
        self.fish_params = fish_params
        self.target_func = target_func
        self.positions = np.random.rand(num_fish, self.fish_params['dim'])
        self.velocities = np.zeros((num_fish, self.fish_params['dim']))
        self.best_position = self.positions[np.argmin(self.target_func(self.positions))]
        self.best_value = self.target_func(self.positions)

    def update_velocities(self):
        for i in range(self.num_fish):
            r1 = np.random.rand()
            r2 = np.random.rand()
            p1 = self.positions[i]
            p2 = self.positions[np.random.randint(self.num_fish)]
            p3 = self.positions[np.random.randint(self.num_fish)]
            c1 = 2 * self.fish_params['c1'] * r1
            c2 = 2 * self.fish_params['c2'] * r2
            self.velocities[i] = (c1 * p1 + c2 * p2 + self.fish_params['w'] * self.velocities[i])

    def update_positions(self):
        for i in range(self.num_fish):
            self.positions[i] += self.velocities[i]

    def run(self, max_iterations):
        for _ in range(max_iterations):
            self.update_velocities()
            self.update_positions()
            current_value = self.target_func(self.positions)
            if current_value < self.best_value:
                self.best_value = current_value
                self.best_position = self.positions[np.argmin(current_value)]
        return self.best_position, self.best_value
```

2. `fish_params`：鱼群参数，包括 `dim`（维度）、`c1`（关注权重）、`c2`（探索权重）和 `w`（鱼群中的权重）。
3. `target_func`：目标函数，用于评估鱼群中的鱼的优化效果。

接着，我们实现了三个方法：

1. `update_velocities`：更新鱼群中每个鱼的速度。
2. `update_positions`：更新鱼群中每个鱼的位置。
3. `run`：运行鱼群算法，直到达到最大迭代次数。
