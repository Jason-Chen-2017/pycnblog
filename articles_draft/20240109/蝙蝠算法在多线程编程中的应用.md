# 4. 具体代码实例和详细解释说明

在本节中，我们将通过一个具体的代码实例来详细解释蝙蝠算法的实现过程。

```python
import threading
import time

class Bat:
    def __init__(self, id, speed, blood_speed, sleep_time):
        self.id = id
        self.speed = speed
        self.blood_speed = blood_speed
        self.sleep_time = sleep_time
        self.is_sleeping = False

    def fly(self):
        print(f"蝙蝠{self.id}开始飞行")
        time.sleep(self.speed)
        print(f"蝙蝠{self.id}结束飞行")

    def feed(self):
        print(f"蝙蝠{self.id}开始吸血")
        time.sleep(self.blood_speed)
        print(f"蝙蝠{self.id}结束吸血")

    def sleep(self):
        self.is_sleeping = True
        print(f"蝙蝠{self.id}开始睡眠")
        time.sleep(self.sleep_time)
        self.is_sleeping = False
        print(f"蝙蝠{self.id}结束睡眠")

def main():
    num_bats = 5
    bats = []
    for i in range(num_bats):
        speed = 2 * i
        blood_speed = 3 * i
        sleep_time = 5 * i
        bat = Bat(i, speed, blood_speed, sleep_time)
        bats.append(bat)

    for bat in bats:
        bat.fly()
        bat.feed()
        while bat.is_sleeping:
            time.sleep(1)

    print("所有蝙蝠都完成飞行和吸血")

if __name__ == "__main__":
    main()
```
