```python
import numpy as np

class LogicPusher:
    def __init__(self):
        self.rules = []
        self.facts = []

    def add_rule(self, rule):
        self.rules.append(rule)

    def add_fact(self, fact):
        self.facts.append(fact)

    def push(self):
        for rule in self.rules:
            for fact in self.facts:
                if rule.matches(fact):
                    fact.apply(rule)

class Rule:
    def __init__(self, head, body):
        self.head = head
        self.body = body

    def matches(self, fact):
        return all(body.matches(fact) for body in self.body)

class Fact:
    def __init__(self, predicate, value):
        self.predicate = predicate
        self.value = value

    def apply(self, rule):
        print(f"Applying rule: {rule.head} to fact: {self.predicate}({self.value})")

# 示例规则

rule2 = Rule(Fact('is_even', x), Fact('is_even', y))

# 示例事实
fact1 = Fact('is_even', 4)
fact2 = Fact('is_even', 6)

# 创建逻辑推理引擎
logic_pusher = LogicPusher()

# 添加规则和事实
logic_pusher.add_rule(rule1)
logic_pusher.add_rule(rule2)
logic_pusher.add_fact(fact1)

# 推理
logic_pusher.push()
```
