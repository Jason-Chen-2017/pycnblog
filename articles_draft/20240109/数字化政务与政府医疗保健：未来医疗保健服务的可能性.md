```python
class EBS:
    def __init__(self):
        self.patient_info = {}
        self.doctor_info = {}
        self.treatment_plan = {}
        self.medication = {}

    def create_case_sheet(self):
        self.patient_info = {}
        self.doctor_info = {}
        self.treatment_plan = {}
        self.medication = {}
        return self

    def fill_case_sheet(self, patient_info, doctor_info, treatment_plan, medication):
        self.patient_info = patient_info
        self.doctor_info = doctor_info
        self.treatment_plan = treatment_plan
        self.medication = medication
        return self

    def save_case_sheet(self):
        with open('case_sheet.txt', 'w') as f:
            f.write(f'Patient Info: {self.patient_info}\n')
            f.write(f'Doctor Info: {self.doctor_info}\n')
            f.write(f'Treatment Plan: {self.treatment_plan}\n')
            f.write(f'Medication: {self.medication}\n')
        return self

    def access_case_sheet(self):
        with open('case_sheet.txt', 'r') as f:
            print(f'Patient Info: {f.readline().strip()}\n')
            print(f'Doctor Info: {f.readline().strip()}\n')
            print(f'Treatment Plan: {f.readline().strip()}\n')
            print(f'Medication: {f.readline().strip()}\n')
        return self
```

这个代码实现了一个简单的电子病例单系统，可以创建、填写、保存和访问病例单。在实际应用中，我们可以根据需要进行优化和拓展。
