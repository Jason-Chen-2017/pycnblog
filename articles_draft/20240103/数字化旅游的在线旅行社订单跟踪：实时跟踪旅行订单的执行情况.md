```python
from collections import deque

class DataStream:
    def __init__(self):
        self.stream = deque()

    def insert(self, x):
        self.stream.append(x)

    def remove(self, x):
        if x in self.stream:
            self.stream.remove(x)

    def find(self, x):
        return x in self.stream
```
