### 1.4.1 任务管理代码实例

```python
import numpy as np

class Task:
    def __init__(self, name, importance, urgency):
        self.name = name
        self.importance = importance
        self.urgency = urgency

tasks = [
    Task("任务A", 5, 3),
    Task("任务B", 7, 2),
    Task("任务C", 3, 6),
]

for task in tasks:
    priority = np.sum(np.array([task.importance, task.urgency]))
    print(f"任务 {task.name} 优先级：{priority}")
```

### 1.4.2 团队协作代码实例

```python
class TeamMember:
    def __init__(self, name, ability, task_demand):
        self.name = name
        self.ability = ability
        self.task_demand = task_demand

team_members = [
    TeamMember("成员A", 8, 2),
    TeamMember("成员B", 6, 3),
    TeamMember("成员C", 7, 1),
]

team_cooperation_efficiency = sum([team_member.ability / team_member.task_demand for team_member in team_members])
print(f"团队协作效率：{team_cooperation_efficiency}")
```

### 1.4.3 沟通方式选择代码实例

```python
class CommunicationMethod:
    def __init__(self, name, effectiveness, cost):
        self.name = name
        self.effectiveness = effectiveness
        self.cost = cost

communication_methods = [
    CommunicationMethod("方式A", 9, 2),
    CommunicationMethod("方式B", 7, 1),
    CommunicationMethod("方式C", 8, 3),
]

best_communication_method = max([communication_method.effectiveness / communication_method.cost for communication_method in communication_methods])
print(f"最佳沟通方式：{best_communication_method}")
```
