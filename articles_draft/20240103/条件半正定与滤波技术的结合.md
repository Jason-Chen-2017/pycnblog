import numpy as np

def conditional_semi_definite(x):
    A = np.zeros((len(x), len(x)))
    for i in range(len(x)):
        A[i, i] = x[i]**2
    return A

### 4.2滤波技术的Python实现

接下来，我们需要定义滤波器的impulse响应$h(t)$：

```python
import numpy as np

def filter_impulse_response(t, cutoff_freq, sample_rate):
    N = 4
    halfband = cutoff_freq / (sample_rate / 2)
    h = np.zeros(2 * N + 1)
    for i in range(N, N + halfband):
        h[i - N] = 0.5**(i - N)
    for i in range(N + halfband, 2 * N + 1):
        h[i - N] = 0.5**(i - N)
    h = h[N:-N]
    return h
```
