## 4.1 计算多边形的面积和外接圆的半径

```python
import numpy as np

def calculate_area(points):
    area = 0
    for i in range(len(points)):
        area += points[i][0] * points[(i + 1) % len(points)][1]
        area -= points[(i + 1) % len(points)][0] * points[i][1]
    return abs(area) / 2

def calculate_circumradius(points):
    area = calculate_area(points)
    circumradius = area / sum(p[0] * p[1] for p in points)
    return circumradius

points = np.array([[0, 0], [2, 0], [2, 2], [0, 2]])
area = calculate_area(points)
circumradius = calculate_circumradius(points)
print("面积:", area)
print("外接圆半径:", circumradius)
```

def is_intersect(points1, points2):
    for p1 in points1:
        for p2 in points2:
            if p1[0] <= p2[0] <= p1[2] and p2[0] <= p1[0] <= p2[2]:
                return True
    return False

points1 = np.array([[0, 0], [2, 0], [2, 2], [0, 2]])
points2 = np.array([[1, 1], [3, 1], [3, 3], [1, 3]])

def calculate_min_distance(points):
    min_distance = float('inf')
    for p1 in points:
        for p2 in points:
            distance = np.sqrt((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2)
            min_distance = min(min_distance, distance)
    return min_distance

points = np.array([[0, 0], [2, 0], [2, 2], [0, 2]])
min_distance = calculate_min_distance(points)
