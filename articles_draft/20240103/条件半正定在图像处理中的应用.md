## 3.1 图像边缘检测
### 3.1.1 二阶差分矩阵
在图像边缘检测中，我们通常使用二阶差分矩阵来描述图像的变化。对于一个二维图像f(x, y)，其二阶差分矩阵可以表示为：

$$
\begin{bmatrix}
f(x+1, y+1) - 2f(x+1, y) + f(x, y+1) & f(x+1, y+1) - 2f(x, y+1) + f(x-1, y+1) \\
f(x+1, y+1) - 2f(x+1, y-1) + f(x, y-1) & f(x+1, y) - 2f(x, y) + f(x-1, y)
\end{bmatrix}
$$

### 3.1.2 条件半正定检测
对于一个给定的点(x, y)，我们可以计算其二阶差分矩阵的特征值。如果这些特征值都大于零，则该点为边缘点。根据条件半正定的定义，如果矩阵A满足以下条件：

1. A是对称的，即A = A^T
2. 对于任何非零向量v，A*v >= 0

那么，矩阵A的特征值都是非负的。因此，我们可以使用条件半正定来判断一个点是否为边缘点。

## 4.1 图像边缘检测
```python
import numpy as np
import cv2

def is_positive_eigenvalue(matrix):
    eigenvalues = np.linalg.eigvals(matrix)
    return all(eigenvalue > 0 for eigenvalue in eigenvalues)

def detect_edges(image):
    image = cv2.GaussianBlur(image, (3, 3), 0)
    second_derivative = cv2.laplacian(image, cv2.CV_64F)
    edges = np.zeros_like(image, dtype=np.uint8)
    for y, x in np.ndindex(second_derivative.shape):
        if is_positive_eigenvalue(second_derivative[y-1:y+2, x-1:x+2]):
            edges[y, x] = 255
    return edges

edges = detect_edges(image)
cv2.imshow('Edges', edges)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
## 4.2 图像滤波

```python
def average_filter(image, filter_size):
    filter_matrix = np.ones((filter_size, filter_size), dtype=np.float32) / filter_size**2
    filtered_image = np.zeros_like(image, dtype=np.float32)
    for y in range(image.shape[0] - filter_size + 1):
        for x in range(image.shape[1] - filter_size + 1):
            filtered_image[y, x] = np.sum(image[y:y+filter_size, x:x+filter_size] * filter_matrix)
    return filtered_image

filtered_image = average_filter(image, 3)
cv2.imshow('Filtered Image', filtered_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
## 4.3 图像压缩
```python
import numpy as np
import cv2

def dct2(image):
    dct_image = cv2.dct(np.float32(image))
    return dct_image

def idct2(dct_image):
    idct_image = cv2.idct(np.float32(dct_image))
    return idct_image

def compress_image(image, quality_factor):
    dct_image = dct2(image)
    compressed_dct_image = np.round(dct_image * quality_factor).astype(np.float32)
    compressed_image = idct2(compressed_dct_image)
    return compressed_image

compressed_image = compress_image(image, 0.5)
cv2.imshow('Compressed Image', compressed_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
