    # 获取用户历史行为数据
    user_data = pd.read_csv('user_data.csv')
    
    # 获取用户购买过的商品ID
    bought_products = user_data[user_data['user_id'] == user_id]['product_id'].values
    
    # 获取用户未购买过的商品ID
    not_bought_products = user_data[user_data['user_id'] == user_id]['product_id'].nunique() - len(bought_products)

    recommendations = []
    for i in range(not_bought_products):
        product_id = model.predict([[i]])[0]
        recommendations.append(product_id)
    
    return recommendations
