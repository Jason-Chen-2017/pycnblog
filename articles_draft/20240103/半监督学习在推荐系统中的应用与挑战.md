import numpy as np
import pandas as pd
from scipy.spatial.distance import cosine

# 构建用户-物品矩阵
user_item_matrix = pd.DataFrame(np.random.randint(0, 5, size=(1000, 1000)), columns=['rating'])

# 填充矩阵
user_item_matrix.fillna(0, inplace=True)

# 计算相似度
user_similarity = pd.DataFrame(index=user_item_matrix.index, columns=user_item_matrix.index)

for i in range(user_item_matrix.shape[0]):
    for j in range(i + 1, user_item_matrix.shape[0]):
        user_similarity.loc[i, j] = cosine(user_item_matrix.iloc[i], user_item_matrix.iloc[j])
        user_similarity.loc[j, i] = user_similarity.loc[i, j]

# 推荐物品
def recommend(user_id, n_recommendations):
    user_row = user_item_matrix.loc[user_id]
    similar_users = user_similarity.loc[user_id].sort_values(ascending=False)
    similar_users = similar_users[similar_users != 0]
    similar_users = similar_users.index[1:n_recommendations + 1]

    recommended_items = user_item_matrix.loc[similar_users].mean()
    return recommended_items.sort_values(ascending=False)

# 测试推荐
user_id = 0
n_recommendations = 10
recommended_items = recommend(user_id, n_recommendations)
print(recommended_items)
