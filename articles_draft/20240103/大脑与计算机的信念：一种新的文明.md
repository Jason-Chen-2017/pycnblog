def loss_function(y_true, y_pred):
    return np.mean((y_true - y_pred) ** 2)

# 定义神经网络的前向传播
def forward_pass(X, W, b):
    Z = np.dot(X, W) + b
    A = sigmoid(Z)
    return A

def backward_pass(X, y, A, W, b):
    dZ = A - y
    dW = np.dot(X.T, dZ)

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# 测试
