import numpy as np

# 最大池化
def max_pooling(input_data, pool_size, stride):
    output_data = np.zeros((input_data.shape[0] - pool_size + 1, input_data.shape[1] - pool_size + 1))
    for i in range(output_data.shape[0]):
        for j in range(output_data.shape[1]):
            max_value = np.max(input_data[i:i+pool_size, j:j+pool_size])
            output_data[i, j] = max_value
    return output_data

# 平均池化
def average_pooling(input_data, pool_size, stride):
    output_data = np.zeros((input_data.shape[0] - pool_size + 1, input_data.shape[1] - pool_size + 1))
    for i in range(output_data.shape[0]):
        for j in range(output_data.shape[1]):
            avg_value = np.mean(input_data[i:i+pool_size, j:j+pool_size])
            output_data[i, j] = avg_value
    return output_data

# 随机池化
def random_pooling(input_data, pool_size, stride):
    output_data = np.zeros((input_data.shape[0] - pool_size + 1, input_data.shape[1] - pool_size + 1))
    for i in range(output_data.shape[0]):
        for j in range(output_data.shape[1]):
            random_value = np.random.randint(0, 256)
            output_data[i, j] = random_value
    return output_data
```
