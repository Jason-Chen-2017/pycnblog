```python
import numpy as np
import pandas as pd
from sklearn.preprocessing import MinMaxScaler
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error
from skfuzzy import fuzzify, control

# 数据预处理
data = pd.read_csv('house_price.csv')

data_scaled = scaler.fit_transform(data)

# 神经网络的构建
from skfuzzy.neural_network import ANFIS

x = data_scaled[:, :-1]
y = data_scaled[:, -1]

model = ANFIS(x, y, num_iterations=1000)

# 模糊逻辑的构建
from skfuzzy.control import Antecedent, Consequent, Rule

antecedent_area = Antecedent(np.arange(0, 1, 0.1), 'area')
antecedent_age = Antecedent(np.arange(0, 1, 0.1), 'age')
antecedent_location = Antecedent(np.arange(0, 1, 0.1), 'location')

consequent_price = Consequent(np.arange(0, 1, 0.1), 'price')

rule1 = Rule(antecedent_area, antecedent_age, antecedent_location, consequent_price)
rule2 = Rule(antecedent_area, antecedent_age, antecedent_location, consequent_price)

# 神经模糊系统的训练
model.train(x, y)

# 神经模糊系统的应用
predicted_price = model.predict(x)

# 评估模型性能
mse = mean_squared_error(y, predicted_price)
print('Mean Squared Error:', mse)
```
