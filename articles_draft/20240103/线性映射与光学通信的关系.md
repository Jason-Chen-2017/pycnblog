1. 对于任意向量$v_1$和$v_2$，有$T(v_1+v_2)=T(v_1)+T(v_2)$。

# 4.具体代码实例和详细解释说明

## 4.1Python实现光信号调制

```python
import numpy as np
import matplotlib.pyplot as plt

def pulse_shaping(data, symbol_rate, pulse_width):
    N = len(data)
    T = 1 / symbol_rate
    t = np.linspace(-T/2, T/2, int(2*T*N))
    pulse = np.blackman(int(pulse_width*N))
    pulse = pulse / np.sum(pulse)
    return np.convolve(data, pulse, mode='valid')

def modulate(data, pulse_shaping, symbol_rate):
    t = np.linspace(-1/symbol_rate, 1/symbol_rate, int(1/symbol_rate*len(data)))
    pulse_shaping = pulse_shaping * np.exp(1j*2*np.pi*t)
    return np.fft.fft(pulse_shaping)

data = np.array([1, 0, 1, 0])
symbol_rate = 1
pulse_width = 2
modulated_signal = modulate(data, pulse_shaping(data, symbol_rate, pulse_width), symbol_rate)

plt.figure()
plt.stem(modulated_signal.real)
plt.xlabel('Frequency (Hz)')
plt.ylabel('Amplitude')
plt.title('Modulated Signal')
plt.show()
```

```python
def demodulate(modulated_signal, symbol_rate):
    t = np.linspace(-1/symbol_rate, 1/symbol_rate, int(1/symbol_rate*len(modulated_signal)))
    pulse_shaping = np.blackman(int(2*symbol_rate*len(modulated_signal)))
    pulse_shaping = pulse_shaping / np.sum(pulse_shaping)
    return np.convolve(modulated_signal, np.fft.ifft(pulse_shaping), mode='valid').real

data = np.array([1, 0, 1, 0])
symbol_rate = 1
pulse_width = 2
modulated_signal = modulate(data, pulse_shaping(data, symbol_rate, pulse_width), symbol_rate)
demodulated_signal = demodulate(modulated_signal, symbol_rate)

plt.figure()
plt.stem(demodulated_signal)
plt.xlabel('Time (s)')
plt.ylabel('Amplitude')
plt.title('Demodulated Signal')
plt.show()
```
