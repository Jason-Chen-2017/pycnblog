### 1.4.1 高维积分示例

假设我们需要计算以下高维积分的值：

$$
I = \int_{0}^{1} \int_{0}^{1} \int_{0}^{1} e^{-(x^2 + y^2 + z^2)} dx dy dz
$$

我们可以使用蒙特卡罗方法进行估计，具体的代码实例如下：

```python
import numpy as np

def f(x, y, z):
    return np.exp(-(x**2 + y**2 + z**2))

def mc_integral(f, a, b, n_samples):
    x_samples = np.random.uniform(a, b, n_samples)
    y_samples = np.random.uniform(a, b, n_samples)
    z_samples = np.random.uniform(a, b, n_samples)
    integral = np.sum(f(x_samples, y_samples, z_samples)) / (n_samples * (b - a)**3)
    return integral

n_samples = 100000
integral = mc_integral(f, 0, 1, n_samples)
print("蒙特卡罗方法估计的积分值：", integral)
```

```python
import numpy as np

def language_model(sentence):
    words = sentence.split()
    word_count = np.zeros(len(vocab))
    for word in words:
        word_count[vocab.index(word)] += 1
    return word_count

vocab = ['the', 'cat', 'sat', 'on', 'the', 'mat']
sentences = ['the cat sat on the mat', 'the cat sat on the rug', 'the cat jumped on the mat']

word_counts = [language_model(sentence) for sentence in sentences]
average_word_count = np.mean(word_counts, axis=0)
print("平均词频：", average_word_count)
```
