plt.ylabel('Probability')
plt.title('Wave function probability')
plt.show()

    return np.abs(electric_field)**2 / (2 * np.epsilon_0)

import matplotlib.pyplot as plt

plt.contourf(X, Y, Z, energy_density_values, 50)
