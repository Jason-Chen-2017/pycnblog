```python
def gradient_descent(X, y, learning_rate, iterations):
    theta = np.zeros(1)
    for i in range(iterations):
        y_pred = np.dot(X, theta)
        loss_value = loss(y, y_pred)
        gradient = np.dot(X.T, (y_pred - y)) / len(y)
        theta -= learning_rate * gradient
```

