import threading
import time

class TaskScheduler:
    def __init__(self):
        self.tasks = []

    def add_task(self, task):
        self.tasks.append(task)

    def run_task(self):
        for task in self.tasks:
            task.run()

class Task:
    def __init__(self, name):
        self.name = name

    def run(self):
        print(f"Running task {self.name}")
        time.sleep(1)

scheduler = TaskScheduler()
task1 = Task("Task 1")
task2 = Task("Task 2")
scheduler.add_task(task1)
scheduler.add_task(task2)
scheduler.run_task()
```
