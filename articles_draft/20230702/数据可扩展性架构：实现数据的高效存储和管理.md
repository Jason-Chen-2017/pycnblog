### 4.3 核心代码实现

```python
import time
import random
import string
import numpy as np
import os
import json
from PYTHON.lib.config import Conf

class FileSystem:
    def __init__(self, redis_url, hadoop_url, hadoop_port, hadoop_user, hadoop_password):
        self.redis_url = redis_url
        self.hadoop_url = hadoop_url
        self.hadoop_port = hadoop_port
        self.hadoop_user = hadoop_user
        self.hadoop_password = hadoop_password

        self.redis = Conf.get_redis_client(redis_url)
        self.hadoop = Conf.get_hadoop_client(hadoop_url)

    def write_file(self, file_name, content):
        with self.redis.command("write", [file_name, content]) as command:
            result = command.result()

        return result

    def read_file(self, file_name):
        with self.redis.command("read", [file_name]) as command:
            result = command.result()

        return result.decode("utf-8")

    def delete_file(self, file_name):
        with self.redis.command("delete", [file_name]) as command:
            result = command.result()

        return result

    def list_files(self):
        with self.redis.command("ls", []) as command:
            result = command.result()

        return result.decode("utf-8").strip()

    def usage(self):
        print("Usage: python file_system.py <redis_url> <hadoop_url> <hadoop_port> <hadoop_user> <hadoop_password>")

if __name__ == '__main__':
    redis_url = "127.0.0.1:6379"
    hadoop_url = "hdfs://hadoop:9000"
    hadoop_port = 9000
    hadoop_user = "hadoop"
    hadoop_password = "hadoop"

    file_system = FileSystem(redis_url, hadoop_url, hadoop_port, hadoop_user, hadoop_password)

    # File writing
    file_name = "test.txt"
    content = "test content"
    file_status = file_system.write_file(file_name, content)
    print("Status: ", file_status)

    # File reading
    file_name = "test.txt"
    content = file_system.read_file(file_name)
    print("Content: ", content)

    # File deleting
    file_name = "test.txt"
    file_status = file_system.delete_file(file_name)
    print("Status: ", file_status)

    # File listing
    file_status = file_system.list_files()
    print("Files: ", file_status)
```
