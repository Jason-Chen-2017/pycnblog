def create_model(vocab_size, n_classes):

    embedding = word2vec.Word2Vec(vocab_size, n_classes)
    # 将输入层与嵌入层连接，产生新的特征
    outputs = embedding(inputs)

    model = keras.Model(inputs, outputs)
    return model
