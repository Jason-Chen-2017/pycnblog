# 数据预处理
def preprocess_data(data):
    # 数据清洗
    data = []
    for line in data:
        # 去除标点符号和空格
        line = line.translate(str.maketrans("", "", string.punctuation))
        # 去除换行符
        line = line.rstrip()
        # 将所有元素转换为小写
        line = line.lower()
        # 将所有元素组成一个列表
        data.append(line)
    # 数据去重
    data = sorted(data)
    # 数据格式化
    data = [": " + line for line in data]
    return data

# 数据标注
def label_data(data, label):
    # 循环遍历数据
    for i in range(len(data)):
        # 循环遍历标签
        for label_id in range(len(label)):
            # 判断当前元素是否为标签
            if label[i] == label_id:
                # 标记当前元素为标签
                data[i] = label_id
                # 返回当前元素
                return data
    # 如果循环遍历完所有的数据仍然没有找到标签
    return np.zeros(len(data), dtype=int)

# 数据校正
def normalize_data(data):
    # 数据归一化
    data = (data - np.min(data)) / (np.max(data) - np.min(data))
    # 返回数据
    return data

# 应用实例
# 数据集
data = ["计算机视觉 : 图像分类", "计算机视觉 : 目标检测", "计算机视觉 : 自然语言处理"]
# 标签
labels = [1, 0, 2]
# 数据预处理
preprocessed_data = preprocess_data(data)
# 数据标注
labeled_data = label_data(preprocessed_data, labels)
# 数据校正
normalized_data = normalize_data(labeled_data)
# 模型训练
model = keras.Sequential()
model.add(keras.layers.Dense(64, input_shape=(preprocessed_data.shape[1],), activation='relu'))
model.add(keras.layers.Dropout(0.2))
model.add(keras.layers.Dense(32, activation='relu'))
model.add(keras.layers.Dropout(0.2))
model.add(keras.layers.Dense(2, activation='softmax'))
model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
model.fit(normalized_data, labels, epochs=10)
