2.2. 技术原理介绍:算法原理,操作步骤,数学公式等

    def compile(self, loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy']):
        self.model.compile(loss=loss, optimizer=optimizer, metrics=metrics, epochs=1)

    def fit(self, x_train, y_train, epochs=1):
        self.model.fit(x_train, y_train, epochs=epochs, batch_size=32, validation_data=(x_train, y_train))
