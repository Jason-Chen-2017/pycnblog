import hadoop
import pyspark

# 初始化
def init():
    # 创建Hadoop环境
    hadoop.init()
    # 创建Spark环境
    spark = pyspark.Spark()
    # 加载配置文件
    config = hadoop.config.load()
    # 设置Spark与Hadoop的并行度
    spark.conf.set("spark.es.scale", 1)
    spark.conf.set("spark.processing.框架", "hadoop")
    # 创建配送数据表
    df = spark.read.format("csv").option("header", "true").option("inferSchema", "true")
    # 计算配送距离
    df = df.withColumn("配送距离", calculate_distance(df))
    # 计算配送时间
    df = df.withColumn("配送时间", calculate_time(df))
    # 保存结果
    df.write.mode("overwrite").csv("配送结果.csv", "text")

# 计算配送距离
def calculate_distance(df):
    distances = df.distance_from_source.apply(lambda row: row[0])
    return distances.sum()

# 计算配送时间
def calculate_time(df):
    df["配送时间"] = df["配送距离"] / df["配送速度"]
