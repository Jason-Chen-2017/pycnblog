from sklearn.metrics.pairwise import cosine_similarity

def recommend(user_id, items, ratings):
    user_ratings = ratings[user_id]
    user_ratings_norm = np.linalg.norm(user_ratings)
    similarity = cosine_similarity(user_ratings.reshape(1, -1), ratings)
    similarity_norm = np.linalg.norm(similarity, axis1=1)
    similarity_norm_transpose = np.transpose(similarity_norm)
    similarity_result = similarity_norm * similarity_norm_transpose
    similarity_result = np.where(similarity_result > 0, similarity_result, 0)
    similarity_result = similarity_result.flatten()

    recommended_items = [items[i] for i in similarity_result[:10]]
    return recommended_items
