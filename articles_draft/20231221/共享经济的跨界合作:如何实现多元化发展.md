```python
import threading
import time

class ResourceAllocator:
    def __init__(self):
        self.resources = []
        self.requests = []

    def add_resource(self, resource):
        self.resources.append(resource)

    def request_resource(self, resource, user):
        self.requests.append((resource, user))
        print(f"{user} is waiting for {resource}")
        while self.resources:
            if self.resources[0] == resource:
                self.resources.pop(0)
                print(f"{user} gets {resource}")
                return
            time.sleep(1)

if __name__ == "__main__":
    allocator = ResourceAllocator()
    allocator.add_resource("A")
    allocator.add_resource("B")
    allocator.request_resource("A", "User1")
    allocator.request_resource("B", "User2")
    allocator.request_resource("A", "User3")

class PriceAdjustor:
    def __init__(self):
        self.prices = {}

    def update_price(self, resource, price):
        if resource not in self.prices:
            self.prices[resource] = []
        self.prices[resource].append(price)

    def get_average_price(self, resource):
        if resource not in self.prices:
            return None
        return sum(self.prices[resource]) / len(self.prices[resource])

if __name__ == "__main__":
    adjustor = PriceAdjustor()
    adjustor.update_price("A", 100)
    adjustor.update_price("A", 100)
    adjustor.update_price("A", 150)
    print(f"Average price of A: {adjustor.get_average_price('A')}")
```
