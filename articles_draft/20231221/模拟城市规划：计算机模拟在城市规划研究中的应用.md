def update_model(model, r, K, d):
    new_model = np.copy(model)
    for i in range(model.shape[0]):
        for j in range(model.shape[1]):
            new_model[i, j] = model[i, j] + r * model[i, j] * (1 - model[i, j] / K) - d * model[i, j]
            if new_model[i, j] > 1:
