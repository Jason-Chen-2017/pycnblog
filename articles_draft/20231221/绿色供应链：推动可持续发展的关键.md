## 4.1 代码实例

```python
class GreenSupplyChain:
    def __init__(self):
        self.products = []
        self.suppliers = []
        self.transports = []
        self.wastes = []

    def add_product(self, product):
        self.products.append(product)

    def add_supplier(self, supplier):
        self.suppliers.append(supplier)

    def add_transport(self, transport):
        self.transports.append(transport)

    def add_waste(self, waste):
        self.wastes.append(waste)

    def calculate_environmental_cost(self):
        environmental_cost = 0
        for product in self.products:
            environmental_cost += product.environmental_cost
        for supplier in self.suppliers:
            environmental_cost += supplier.environmental_cost
        for transport in self.transports:
            environmental_cost += transport.environmental_cost
        for waste in self.wastes:
            environmental_cost += waste.environmental_cost
        return environmental_cost

    def calculate_efficiency(self):
        efficiency = 0
        for product in self.products:
            efficiency += product.efficiency
        for supplier in self.suppliers:
            efficiency += supplier.efficiency
        for transport in self.transports:
            efficiency += transport.efficiency
        return efficiency

    def calculate_sustainability(self):
        sustainability = 0
        for product in self.products:
            sustainability += product.sustainability
        for supplier in self.suppliers:
            sustainability += supplier.sustainability
        for transport in self.transports:
            sustainability += transport.sustainability
        return sustainability
```
