## 4.1 数据授权的具体代码实例
以下是一个简单的数据授权示例，使用Python编程语言实现：
```python
import hashlib
import hmac
import base64
import json

# 数据所有者加密数据并签名
def encrypt_and_sign(data, key):
    encrypted_data = hashlib.sha256(data.encode()).hexdigest()
    signature = hmac.new(key.encode(), msg=encrypted_data.encode(), digestmod=hashlib.sha256).hexdigest()
    return encrypted_data, signature

# 数据接收方验证数据的安全性和完整性
def verify(encrypted_data, signature, key):
    decrypted_data = hmac.new(key.encode(), msg=encrypted_data.encode(), digestmod=hashlib.sha256).hexdigest()
    return hmac.compare_digest(signature, decrypted_data)

# 数据所有者
data_owner_key = "my_secret_key"
data = "Hello, World!"
encrypted_data, signature = encrypt_and_sign(data, data_owner_key)
print("Encrypted data:", encrypted_data)
print("Signature:", signature)

# 数据接收方
data_recipient_key = "my_secret_key"
is_verified = verify(encrypted_data, signature, data_recipient_key)
print("Is verified:", is_verified)
```
在这个示例中，数据所有者使用HMAC和SHA-256算法来加密和签名数据，数据接收方使用相同的算法来验证数据的安全性和完整性。

```python
import hashlib
import json

# 创建一个区块
class Block:
    def __init__(self, index, previous_hash, timestamp, data, hash):
        self.index = index
        self.previous_hash = previous_hash
        self.timestamp = timestamp
        self.data = data
        self.hash = hash

# 创建一个区块链
class Blockchain:
    def __init__(self):
        self.chain = [self.create_genesis_block()]

    def create_genesis_block(self):
        return Block(0, "0", 0, "Genesis Block", self.calculate_hash(0, "0", 0, "Genesis Block"))

    def create_new_block(self, data):
        index = len(self.chain)
        previous_hash = self.calculate_hash(self.chain[index - 1])
        timestamp = str(time.time())
        block = Block(index, previous_hash, timestamp, data, self.calculate_hash(index, previous_hash, timestamp, data))
        self.chain.append(block)
        return block

    def calculate_hash(self, index, previous_hash, timestamp, data):
        return hashlib.sha256(f"{index}{previous_hash}{timestamp}{data}".encode()).hexdigest()

# 创建一个区块链实例
my_blockchain = Blockchain()
my_blockchain.create_new_block("First block")
my_blockchain.create_new_block("Second block")
my_blockchain.create_new_block("Third block")

# 打印区块链
for block in my_blockchain.chain:
    print(json.dumps(block.__dict__, default=lambda o: o.__str__(), sort_keys=True))
```
