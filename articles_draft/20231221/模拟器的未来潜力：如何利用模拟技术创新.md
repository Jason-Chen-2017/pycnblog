      $$
      F_{ij} = \frac{m_i m_j}{r_{ij}^2} \left( \frac{p_{ij}}{r_{ij}} + \frac{p_{ij} \cdot r_{ij}}{r_{ij}^2} r_{ij} \right)

# 定义粒子自由运动的力律

    return -9.8 * x

# 定义粒子自由运动的积分步骤
def integrate(x, y, dt):
    return x + dt * y, y - dt * force(x, y)

# 初始化粒子的位置和速度
x = 0
y = 0
vx = 1
vy = 0

# 设置时间步长和模拟时间

# 创建一个空列表来存储粒子的位置

# 使用朗日积分法进行模拟
