```python
import numpy as np

def path_planning(current_position, target_position, obstacles):
    # 计算目标位置与障碍物的距离
    distances = np.linalg.norm(target_position - obstacle, axis=1)
    # 找到最近的障碍物
    nearest_obstacle = np.argmin(distances)
    # 计算目标位置与障碍物之间的角度
    angle = np.arctan2(target_position[1] - obstacles[nearest_obstacle][1], target_position[0] - obstacles[nearest_obstacle][0])
    # 计算最佳的行驶方向
    direction = angle - np.arctan2(current_position[1] - target_position[1], current_position[0] - target_position[0])
    return direction
```
