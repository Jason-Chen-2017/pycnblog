### 3.1 欧几里得距离

欧几里得距离是一种用于计算两个向量之间距离的方法，它可以用于评估图像之间的相似性。欧几里得距离的公式如下：

$$
d(x, y) = \sqrt{\sum_{i=1}^{n} (x_i - y_i)^2}
$$

其中，$x$ 和 $y$ 是两个向量，$n$ 是向量的维度，$x_i$ 和 $y_i$ 是向量的各个元素。

```python
import numpy as np

def euclidean_distance(x, y):
    return np.sqrt(np.sum((x - y) ** 2))

x = np.array([1, 2, 3])
y = np.array([4, 5, 6])

print(euclidean_distance(x, y))
```

### 4.2 结构相似性计算

```python
import cv2
import numpy as np

def structure_similarity(img1, img2):
    gray1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
    gray2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)
    ret1, binary1 = cv2.threshold(gray1, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    ret2, binary2 = cv2.threshold(gray2, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
    return cv2.matchTemplate(binary1, binary2, cv2.TM_CCOEFF_NORMED)


print(structure_similarity(img1, img2))
```

### 4.3 特征相似性计算

```python
import cv2
import numpy as np

def feature_similarity(img1, img2):
    gray1 = cv2.cvtColor(img1, cv2.COLOR_BGR2GRAY)
    gray2 = cv2.cvtColor(img2, cv2.COLOR_BGR2GRAY)
    sift = cv2.SIFT_create()
    keypoints1, descriptors1 = sift.detectAndCompute(gray1, None)
    keypoints2, descriptors2 = sift.detectAndCompute(gray2, None)
    matcher = cv2.BFMatcher(cv2.NORM_L2)
    matches = matcher.match(descriptors1, descriptors2)
    return cv2.drawMatches(img1, keypoints1, img2, keypoints2, matches, None)


print(feature_similarity(img1, img2))
```
