**答案：**
```python
def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

def Feynman_prime_question(n):
    if is_prime(n):
        print(f"{n} 是素数。")
    else:
        print(f"{n} 不是素数。")

Feynman_prime_question(17)
```

    return all(lst[i] <= lst[i+1] for i in range(len(lst)-1))
