### 4.1 代码实例

import numpy as np

def initialize_population(population_size):
    return np.random.rand(population_size)

def calculate_survive_probability(cell, survive_max):
    return cell / survive_max

def calculate_divide_probability(cell, divide_max):
    return cell / divide_max

def update_population(population, survive_probabilities, divide_probabilities):
    new_population = np.zeros_like(population)
    for i in range(len(population)):
        if np.random.rand() < survive_probabilities[i]:
            new_population[i] = population[i]
        if np.random.rand() < divide_probabilities[i]:
            new_population[i] += 1
    return new_population

population_size = 100
survive_max = 10
divide_max = 10
population = initialize_population(population_size)

for t in range(100):
    survive_probabilities = np.array([calculate_survive_probability(cell, survive_max) for cell in population])
    divide_probabilities = np.array([calculate_divide_probability(cell, divide_max) for cell in population])
    population = update_population(population, survive_probabilities, divide_probabilities)

print(population)
