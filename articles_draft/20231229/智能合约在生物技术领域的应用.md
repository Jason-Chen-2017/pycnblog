```solidity
pragma solidity ^0.5.0;

contract BiotechProductTrade {
    address public seller;
    address public buyer;
    uint public price;
    uint public quantity;
    uint public timestamp;

    event TradeCreated(address indexed seller, uint price, uint quantity);
    event TradeCompleted(address indexed buyer, uint price, uint quantity);

    constructor(address _seller, uint _price, uint _quantity) public {
        seller = _seller;
        price = _price;
        quantity = _quantity;
        timestamp = block.timestamp;
        require(msg.sender == seller, "Seller must be the sender");
    }

    function completeTrade(address _buyer) public {
        buyer = _buyer;
        require(msg.sender == buyer, "Buyer must be the sender");
        emit TradeCompleted(buyer, price, quantity);
    }
}
```
