    return np.random.rand(num_photons)

# 光子传输
def transmit_photons(photons, transmission_matrix):
    return np.dot(photons, transmission_matrix)

# 光子检测

    return np.dot(transmitted_photons, detection_matrix.T)

# 主程序
if __name__ == '__main__':
    num_photons = 100
    transmission_matrix = np.random.rand(num_photons, num_photons)
    detection_matrix = np.random.rand(num_photons, num_photons)

    photons = generate_photons(num_photons)
    transmitted_photons = transmit_photons(photons, transmission_matrix)
    detected_photons = detect_photons(transmitted_photons, detection_matrix)
