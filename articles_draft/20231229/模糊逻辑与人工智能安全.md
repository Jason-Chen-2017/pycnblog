import numpy as np
from scipy.interpolate import interp1d

# 定义模糊集
def define_fuzzy_set(name, x, value):
    return lambda x: np.where(x == name, value, 0)

# 定义模糊语言
temperature_language = ['冷', '适中', '热']
temperature_values = [1, 2, 3]
temperature_set = [define_fuzzy_set(lang, np.arange(1, 5), val) for lang, val in zip(temperature_language, temperature_values)]

# 定义模糊关系
def define_fuzzy_relation(name, x, relation):
    return lambda x, y: np.where(x == name, relation, 0)

# 定义模糊逻辑规则
def define_fuzzy_logic_rule(name, x, rule):
    return lambda x, y: np.where(x == name, rule, 0)

# 定义模糊逻辑规则
cold_relation = define_fuzzy_relation('较低', np.arange(1, 5), 0.5)
hot_relation = define_fuzzy_relation('较高', np.arange(1, 5), 0.5)

# 推理和决策
def infer_and_decide(temperature, cooling_set, heating_set):
    cool_degree = interp1d(cooling_set, cooling_set, kind='linear')(temperature)
    heat_degree = interp1d(heating_set, heating_set, kind='linear')(temperature)
    if cool_degree < heat_degree:
        return '开启空调'
    else:
        return '关闭空调'

# 测试
temperature = np.array([2, 2.5, 3])

result = infer_and_decide(temperature, cooling_set, heating_set)
