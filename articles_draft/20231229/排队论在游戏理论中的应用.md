import numpy as np

def mms_simulation(lambda_, mu, s, t):
    ticks = np.arange(t)
    inter_arrival = np.random.exponential(scale=1/lambda_, size=t)
    service_time = np.random.exponential(scale=1/mu, size=t)
    queue_length = np.zeros(t)
    in_queue = np.zeros(t)

    for i in range(t):
        inter_arrival[i] = max(0, inter_arrival[i])
        if in_queue[i] < s and inter_arrival[i] > 0:
            in_queue[i] += 1
            queue_length[i] = in_queue[i]
        else:
            queue_length[i] = in_queue[i]

        if in_queue[i] > 0:
            inter_arrival[i+1] = inter_arrival[i] - service_time[i]
            service_time[i+1] = np.random.exponential(scale=1/mu, size=1)

    return queue_length
