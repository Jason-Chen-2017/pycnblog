$$

$$
\text{Throughput} = \frac{\text{NumberOfTasks}}{\text{TotalTime}}

$$

```python
import time

def preprocess_data(data):
    # 数据清洗
    data = np.nan_to_num(data)
    # 数据压缩
    data = np.compress(data < 100, data)
    return data

# 数据处理
def process_data(data):
    # 数据分析
    result = np.mean(data)
    return result

# 结果传输
def transfer_result(result):
    # 结果通过网络传输至中心服务器
    print("Transferring result:", result)
    time.sleep(1)
    return result

# 主函数
def main():
    # 收集数据
    data = np.random.rand(100)
    # 数据预处理
    data = preprocess_data(data)
    # 数据处理
    result = process_data(data)
    # 结果传输
    result = transfer_result(result)

if __name__ == "__main__":
    main()
```

答案：边缘计算在边缘设备上进行计算，而中心化计算在中心服务器上进行计算。边缘计算的特点是实时性、可扩展性和高效性，而中心化计算的特点是数据集中化、计算集中化和资源共享。
