import numpy as np

# 读取数据
data = pd.read_csv('water_data.csv')

# 计算水资源需求
def water_demand(agriculture, industry, city, ecology):
    return agriculture * 0.5 + industry * 0.3 + city * 0.2 + ecology * 0.1

# 计算总水资源需求
