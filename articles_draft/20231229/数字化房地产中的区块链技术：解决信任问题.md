```python
import hashlib
import json
import time

class Blockchain(object):
    def __init__(self):
        self.chain = []
        self.create_block(proof=1, previous_hash='0')
        self.nodes = set()

    def create_block(self, proof, previous_hash):
        block = {'index': len(self.chain) + 1,
                'timestamp': time.time(),
                'proof': proof,
                'previous_hash': previous_hash}
        self.chain.append(block)
        return block

    def get_last_block(self):
        return self.chain[-1]

    def new_transaction(self, sender, recipient, amount):
        transaction = {'sender': sender,
                      'recipient': recipient,
                      'amount': amount}
        self.chain.append(transaction)
        return transaction

    def new_block(self, proof, previous_hash):
        previous_blocks = self.chain[len(self.chain) - len(self.chain) // 50:]
        self.proof = previous_blocks[-1]['proof'] + 1

        block = self.create_block(proof, previous_hash)

        return block

    def proof_of_work(self, last_proof):
        proof = 0
        while self.valid_proof(last_proof, proof) is False:
            proof += 1
        return proof

    @staticmethod
    def valid_proof(last_proof, proof):
        if proof < last_proof:
            return False
        elif proof > last_proof + 1:
            return False
        else:
            return True

    def register_node(self, address):
        self.nodes.add(address)

    def valid_transaction(self, transaction):
        if transaction['amount'] <= 0:
            return False
        elif transaction['recipient'] == "":
            return False
        elif transaction['sender'] == "":
            return False
        return True

    def valid_chain(self, chain):
        last_block = chain[-1]
        current_index = self.chain[-1]['index']

        if last_block['index'] != current_index:
            return False
        for i in range(current_index - 1, -1, -1):
            block = self.chain[i]
            if block['hash'] != self.hash(block):
                return False
        return True

    def hash(self, block):
        block_string = json.dumps(block, sort_keys=True).encode()
        return hashlib.sha256(block_string).hexdigest()
```
