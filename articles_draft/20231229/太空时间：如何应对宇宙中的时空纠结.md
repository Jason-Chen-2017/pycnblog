## 4.1 时间同步代码实例

以下是一个使用 Python 编写的时间同步协议实现：

```python
import time
import socket

def get_time_server():
    # 获取时间服务器的 IP 地址和端口号
    time_server_ip = '127.0.0.1'
    time_server_port = 12345
    return (time_server_ip, time_server_port)

def send_time_request(time_server_ip, time_server_port):
    # 向时间服务器发送时间同步请求
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
        sock.connect((time_server_ip, time_server_port))
        sock.sendall(b'GET /time HTTP/1.1\r\nHost: {}\r\n\r\n'.format(time_server_ip))
        response = sock.recv(1024)
        return int(response.split()[1])

def update_time(current_time, time_difference):
    # 更新时间
    return current_time + time_difference

if __name__ == '__main__':
    current_time = int(time.time())
    time_server = get_time_server()
    time_difference = send_time_request(*time_server)
    adjusted_time = update_time(current_time, time_difference)
    print('Adjusted time:', adjusted_time)
```
