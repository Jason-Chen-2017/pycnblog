    I = np.array([k * f(x) for x in fireflies])

    # 主循环
    for _ in range(max_iter):
        for i in range(num_fireflies):
            # 更新光强
            I[i] = k * f(fireflies[i])

            # 生成新蜻蜓
            fireflies_new = fireflies[i] + beta * I[i] * np.random.rand(1, 2)

            # 更新蜻蜓群

                fireflies[i] = fireflies_new

    best_adaptability = np.max(I)
