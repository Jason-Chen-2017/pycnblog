## 4.1 噪声去除代码实例

以下是一个使用中值滤波算法实现噪声去除的Python代码实例：

```python
import cv2
import numpy as np

def median_filter(image, size):
    rows, cols = image.shape[:2]
    result = np.zeros((rows, cols), np.uint8)
    for i in range(rows):
        for j in range(cols):
            result[i][j] = np.median(image[max(0, i-size):min(rows, i+size+1), max(0, j-size):min(cols, j+size+1)])
    return result

size = 3
filtered_image = median_filter(image, size)
cv2.imshow('Filtered Image', filtered_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 4.2 增强代码实例

以下是一个使用对比度扩展算法实现图像增强的Python代码实例：

```python
import cv2
import numpy as np

def contrast_stretching(image, a, b):
    rows, cols, channels = image.shape
    result = np.zeros((rows, cols, channels), np.uint8)
    for i in range(rows):
        for j in range(cols):
            pixel = image[i][j]
            if pixel < a:
                result[i][j] = 0
            elif pixel > b:
                result[i][j] = 255
            else:
                result[i][j] = int((pixel - a) / (b - a) * 255)
    return result

a = 50
b = 150
stretched_image = contrast_stretching(image, a, b)
cv2.imshow('Stretched Image', stretched_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 4.3 边缘检测代码实例

以下是一个使用Sobel算法实现边缘检测的Python代码实例：

```python
import cv2
import numpy as np

def sobel_edge_detection(image):
    gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    sobelx = cv2.Sobel(gray_image, cv2.CV_64F, 1, 0, ksize=3)
    sobely = cv2.Sobel(gray_image, cv2.CV_64F, 0, 1, ksize=3)
    magnitude = np.sqrt(sobelx**2 + sobely**2)
    angle = np.arctan2(sobely, sobelx) * 180 / np.pi
    return magnitude, angle

magnitude, angle = sobel_edge_detection(image)
cv2.imshow('Magnitude', magnitude)
cv2.imshow('Angle', angle)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
