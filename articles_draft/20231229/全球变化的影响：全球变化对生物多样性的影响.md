## 4.1 气候模型代码实例
```python
import numpy as np
import matplotlib.pyplot as plt

def solar_radiation(lat, albedo):
    if lat < 66.5:
        return (1 - albedo) * 1361 * np.sin(np.pi * lat / 180)
    else:
        return 0

def longwave_radiation(temperature):
    return 3.7 * 10**-10 * temperature**4

def shortwave_radiation(albedo):
    return (1 - albedo) * 1361

def latent_heat(precipitation, evaporation):
    return precipitation - evaporation

def sensible_heat(temperature, humidity):
    return 0.622 * 5 * temperature * humidity

def climate_model(lat, albedo, temperature, precipitation, evaporation, humidity):
    solar = solar_radiation(lat, albedo)
    longwave = longwave_radiation(temperature)
    shortwave = shortwave_radiation(albedo)
    latent = latent_heat(precipitation, evaporation)
    sensible = sensible_heat(temperature, humidity)
    heat_balance = solar + longwave - shortwave - latent - sensible
    return heat_balance

lat = np.linspace(-90, 90, 360)
albedo = 0.3
temperature = 288
precipitation = 1000
evaporation = 500
humidity = 70

heat_balance = np.vectorize(climate_model)(lat, albedo, temperature, precipitation, evaporation, humidity)
plt.plot(lat, heat_balance)
plt.xlabel('Latitude')
plt.ylabel('Heat Balance')
plt.title('Climate Model')
plt.show()
```

import matplotlib.pyplot as plt

def logistic_growth(N, r, K):
    dN = r * N * (1 - N / K)
    return dN
