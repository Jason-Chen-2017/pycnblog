```python
import numpy as np

    {'type': 'trade', 'from': '文明A', 'to': '文明C', 'result': '成功'},
    {'type': 'culture', 'from': '文明B', 'to': '文明C', 'result': '传播'},
]

# 计算关系强度
def calculate_relation_strength(events):
    relation_strength = {}
    for event in events:
        from_civilization = event['from']
        to_civilization = event['to']
        if from_civilization not in relation_strength:
            relation_strength[from_civilization] = {}
        if to_civilization not in relation_strength:
            relation_strength[to_civilization] = {}
        relation_strength[from_civilization][to_civilization] = 1
        relation_strength[to_civilization][from_civilization] = 1
    return relation_strength

# 计算关系强度矩阵
def calculate_relation_strength_matrix(relation_strength):
    relation_strength_matrix = np.zeros((len(relation_strength), len(relation_strength)))
    for from_civilization, to_civilization in relation_strength.items():
        relation_strength_matrix[from_civilization][to_civilization] = relation_strength[from_civilization][to_civilization]
    return relation_strength_matrix

# 分析关系强度矩阵
def analyze_relation_strength_matrix(relation_strength_matrix):
    print("关系强度矩阵：")
    print(relation_strength_matrix)

# 主程序
if __name__ == "__main__":
    relation_strength = calculate_relation_strength(events)
    relation_strength_matrix = calculate_relation_strength_matrix(relation_strength)
    analyze_relation_strength_matrix(relation_strength_matrix)
```

2. `calculate_relation_strength_matrix`：将关系强度存储为矩阵形式。这可以帮助我们更好地理解不同文明之间的关系、冲突和合作。
