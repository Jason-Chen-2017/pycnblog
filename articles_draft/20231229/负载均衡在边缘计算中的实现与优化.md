```python
import numpy as np

class EdgeComputing:
    def __init__(self, nodes):
        self.nodes = nodes
        self.tasks = []
        self.weights = []

    def add_task(self, task, weight):
        self.tasks.append(task)
        self.weights.append(weight)

    def calculate_load(self):
        loads = np.zeros(len(self.nodes))
        for i, task in enumerate(self.tasks):
            load = np.sum(self.nodes[i] * self.weights[i])
            loads[i] = load
        return loads

    def distribute_tasks(self):
        loads = self.calculate_load()
        distributed_tasks = np.zeros(len(self.nodes))
        for i, load in enumerate(loads):
            distributed_tasks[i] = np.sum(loads) / len(loads)
        return distributed_tasks
```
