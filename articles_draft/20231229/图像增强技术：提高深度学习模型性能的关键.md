## 4.1 旋转

```python
import cv2
import numpy as np

def rotate(image, angle):
    height, width = image.shape[:2]
    center = (width // 2, height // 2)
    rotation_matrix = cv2.getRotationMatrix2D(center, angle, 1.0)
    rotated = cv2.warpAffine(image, rotation_matrix, (width, height))
    return rotated

angle = 45
rotated = rotate(image, angle)
cv2.imshow('Rotated', rotated)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## 4.2 滤波

```python
import cv2
import numpy as np

def median_filter(image, size):
    height, width = image.shape[:2]
    filtered = np.zeros((height, width), dtype=np.uint8)
    for i in range(height):
        for j in range(width):
            values = []
            for x in range(max(0, i - size // 2), min(height, i + size // 2) + 1):
                for y in range(max(0, j - size // 2), min(width, j + size // 2) + 1):
                    values.append(image[x][y])
            filtered[i][j] = np.median(values)
    return filtered

size = 3
filtered = median_filter(image, size)
cv2.imshow('Filtered', filtered)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
