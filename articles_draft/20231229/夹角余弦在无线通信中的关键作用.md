### 4.1 代码实例

```python
import numpy as np

def angle_cosine(a, b):
    dot_product = np.dot(a, b)
    magnitude_a = np.linalg.norm(a)
    magnitude_b = np.linalg.norm(b)
    cos_theta = dot_product / (magnitude_a * magnitude_b)
    return cos_theta

# 信号向量
signal = np.array([1, 0, 0])

# 干扰/噪声向量
noise = np.array([0, 1, 0])

# 计算夹角余弦值
cos_theta = angle_cosine(signal, noise)
print("夹角余弦值:", cos_theta)
```
