import numpy as np
from sklearn.linear_model import LinearRegression

# 定义一个滑动窗口算法
def sliding_window_algorithm(data, window_size, step_size):
    window = []
    for i in range(len(data)):
        if i < window_size:
            window.append(data[i])
        else:
            window.append(data[i])
            window.pop(0)
        if (i + 1) % step_size == 0:
            # 处理函数
            result = np.mean(window)
            print(result)

# 定义一个流式算法
def streaming_algorithm(data, processing_function):
    for i in range(len(data)):
        result = processing_function(data[i])
        print(result)

# 定义一个机器学习算法
def machine_learning_algorithm(data, model, training_strategy, prediction_strategy):
    # 训练策略
    model.fit(data[:int(len(data)*0.8)], data[int(len(data)*0.8):])
    # 预测策略
    prediction = model.predict(data)
    print(prediction)
