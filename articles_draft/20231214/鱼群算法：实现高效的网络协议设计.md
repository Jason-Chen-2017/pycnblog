class Fish:
    def __init__(self, position, velocity):
        self.position = position
        self.velocity = velocity

    def update_position(self):
        self.position += self.velocity

    def update_velocity(self, self_acceleration, crowd_acceleration):
        self.velocity += self_acceleration + crowd_acceleration

def update_fish_position(fish_list):
    for fish in fish_list:
        fish.update_position()

def update_fish_velocity(fish_list):
    for fish in fish_list:
        fish.update_velocity()

def main():
    # 初始化鱼群
    fish_list = [Fish(np.random.rand(), np.random.rand()) for _ in range(100)]

    # 更新鱼群的位置和速度
    for _ in range(1000):
        update_fish_position(fish_list)
        update_fish_velocity(fish_list)

if __name__ == "__main__":
    main()
