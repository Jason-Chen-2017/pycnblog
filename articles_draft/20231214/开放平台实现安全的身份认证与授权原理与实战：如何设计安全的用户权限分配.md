class AccessControl:
    def __init__(self):
        self.users = {}
        self.roles = {}
        self.permissions = {}

    def add_user(self, user, role):
        self.users[user] = role

    def add_role(self, role, permission):
        self.roles[role] = permission

    def add_permission(self, permission, resource, operation):
        self.permissions[permission] = (resource, operation)

    def check_permission(self, user, resource, operation):
        role = self.users.get(user, None)
        if role is None:
            return False

        permission = self.roles.get(role, None)
        if permission is None:
            return False

        resource, operation = self.permissions[permission]
        return resource == resource and operation == operation

access_control = AccessControl()
access_control.add_user('user1', 'admin')
access_control.add_role('admin', 'read_resource')
access_control.add_permission('read_resource', 'resource1', 'read')

user = 'user1'
resource = 'resource1'
operation = 'read'

if access_control.check_permission(user, resource, operation):
    print('Permission granted')
else:
    print('Permission denied')
