import socket
import time

def start_server():
    # 创建套接字
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

    # 绑定地址和端口
    sock.bind(('127.0.0.1', 8080))

    # 监听连接
    sock.listen(5)

    while True:
        # 接收连接
        conn, addr = sock.accept()

        # 处理请求
        request = conn.recv(1024)
        print(request)

        # 发送响应
        response = b"Hello, World!"
        conn.send(response)

        # 关闭连接
        conn.close()

if __name__ == '__main__':
    start_server()
