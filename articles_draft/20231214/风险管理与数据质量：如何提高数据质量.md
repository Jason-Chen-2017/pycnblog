### 4.2 数据校验

```python
import pandas as pd

# 读取数据
data = pd.read_csv('data.csv')

# 数据校验规则设计
rules = {
    'age': {'type': 'int', 'range': (18, 100)},
    'date': {'type': 'datetime', 'format': '%Y-%m-%d'}
}

# 数据校验算法实现
def check_data(data, rules):
    errors = []
    for column, rule in rules.items():
        if not isinstance(data[column].iloc[0], rule['type']):
            errors.append(f'{column} 数据类型错误')
        if rule['type'] == 'int' and not (rule['range'][0] <= data[column].iloc[0] <= rule['range'][1]):
            errors.append(f'{column} 数据值范围错误')
        if rule['type'] == 'datetime' and not isinstance(data[column].iloc[0], rule['format']):
            errors.append(f'{column} 日期格式错误')
    return errors

# 数据校验结果分析
errors = check_data(data, rules)
if errors:
    print('数据校验失败')
    print(errors)
else:
    print('数据校验成功')
```
