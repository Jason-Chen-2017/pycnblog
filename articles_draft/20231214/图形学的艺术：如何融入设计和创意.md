### 4.1 点的加法和减法

```python
# 点的加法
def add_point(p1, p2):
    return (p1[0] + p2[0], p1[1] + p2[1])

# 点的减法
def sub_point(p1, p2):
    return (p1[0] - p2[0], p1[1] - p2[1])
```

### 4.2 线段的加法和减法

```python
# 线段的加法
def add_segment(s1, s2):
    p1 = add_point(s1[0], s2[0])
    p2 = add_point(s1[1], s2[1])
    return (p1, p2)

# 线段的减法
def sub_segment(s1, s2):
    p1 = sub_point(s1[0], s2[0])
    p2 = sub_point(s1[1], s2[1])
    return (p1, p2)
```

### 4.3 多边形的加法和减法

```python
# 多边形的加法
def add_polygon(p1, p2):
    polygon1 = [p for p in p1]
    polygon2 = [p for p in p2]
    for p in polygon2:
        polygon1.append(p)
    return polygon1

# 多边形的减法
def sub_polygon(p1, p2):
    polygon1 = [p for p in p1]
    polygon2 = [p for p in p2]
    for i in range(len(polygon2)):
        polygon1.remove(polygon2[i])
    return polygon1
```

### 4.4 曲线的加法和减法

```python
# 曲线的加法
def add_curve(c1, c2):
    curve1 = c1
    curve2 = c2
    for i in range(len(curve1)):

    return curve1

# 曲线的减法
def sub_curve(c1, c2):
    curve1 = c1
```
