class DecisionTree:
    def __init__(self):
        self.root = None

    def fit(self, data, target):
        self.root = self.create_tree(data, target)

    def predict(self):
        return self.root.predict(data)

    def create_tree(self, data, target):
        if data.shape[0] == 0:
            return None

        # 这里选择的是决策树，所以要随机选择一个节点
        node = random.choice(data)

        # 将数据按照节点的特征进行划分
        left = [x for x in data if x < node]
        right = [x for x in data if x >= node]

        # 递归地对左右两个子节点进行训练
        self.fit(left, target)
        self.fit(right, target)

        # 将两个子节点的预测结果取平均值
        return (left.mean(axis=0) + right.mean(axis=0)) / 2
