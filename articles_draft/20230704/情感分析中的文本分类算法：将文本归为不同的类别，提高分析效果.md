def preprocess(texts):
    # 去除标点符号
    texts = [text.translate(str.maketrans("", "", "")) for text in texts]
    # 去除停用词
    texts = [text.lower() for text in texts]
    # 分词
    texts = " ".join(texts)
    return texts
```

2. 特征提取

```
# 特征提取
def extract_features(text):
    # 提取词
    words = nltk.word_tokenize(text)
    # 提取情感倾向
    sentiment = nltk.compat.nltk.sentiment.polarity_scores(words)
    # 提取主题
    for topic in nltk.themes.Topic.run_themes(words):
        print(topic)
    # 综合特征
    features = []
    for feature in sentiment:
        if feature.polarity >= 0.5:
            features.append("positive")
        else:
            features.append("negative")
    return features
```

3. 模型训练

```
# 模型训练
def train_model(texts, features):
    # 加载预训练的Word2Vec模型
    word_embeddings = word2vec.Word2Vec.load_word2vec_format(filename="word2vec.txt", size=64, ngrams=128, gens=1)
    # 加载特征
    vectorizer = nltk.HybridFeatures.TextVectorizer(feature_extractor=extract_features)
    # 构建数据
    train_data = vectorizer.fit_transform(texts)
    # 构建特征
    train_features = vectorizer.transform(features)
    # 训练模型
    model = nltk.models.Word2Vec.SVC(algorithm="lgb")
