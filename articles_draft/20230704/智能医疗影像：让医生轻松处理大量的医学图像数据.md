4.3. 核心代码实现

实现该方法的具体代码如下所示：
```python
import numpy as np
import tensorflow as tf
import cv2

# 数据预处理
def preprocess_image(image_path):
    # 将图像从文件中读取
    image = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE)
    # 对图像进行归一化处理
    image = image / 255.
    # 将图像转换为三维数组
    image = np.expand_dims(image, axis=0)
    # 将维度归一化
    image = image[:, :, ::-1]
    return image

# 模型训练
def train_model(model_path, train_data):
    # 加载数据集
    train_images = []
    train_labels = []
    for data in train_data:
        train_images.append(preprocess_image(data[0]))
        train_labels.append(data[1])
    # 将数据集转换为三维数组
    train_images = np.array(train_images, dtype=np.float32)
    train_labels = np.array(train_labels, dtype=np.float32)
    # 将模型转换为神经网络
    model = tf.keras.models.Sequential()
    model.add(tf.keras.layers.Conv2D(32, (3, 3), activation='relu', input_shape=(224, 224, 3)))
    model.add(tf.keras.layers.MaxPooling2D(pool_size=(2, 2)))
    model.add(tf.keras.layers.Conv2D(64, (3, 3), activation='relu'))
    model.add(tf.keras.layers.MaxPooling2D(pool_size=(2, 2)))
    model.add(tf.keras.layers.Conv2D(128, (3, 3), activation='relu'))
    model.add(tf.keras.layers.MaxPooling2D(pool_size=(2, 2)))
    model.add(tf.keras.layers.Flatten())
    model.add(tf.keras.layers.Dense(128, activation='relu'))
    model.add(tf.keras.layers.Dropout(0.5))
    model.add(tf.keras.layers.Dense(10, activation='softmax'))
    # 编译模型
    model.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])
    # 训练模型
    model.fit(train_images, train_labels, epochs=10)
    return model

# 模型测试
def test_model(model_path):
    # 加载模型
    model = tf.keras.models.load_model(model_path)
    # 对测试数据集进行预测
    predictions = model.predict(test_images)
    # 计算准确率
    accuracy = np.mean(predictions == test_labels)
    print('Accuracy:', accuracy)

# 主函数
if __name__ == '__main__':
    train_data = [('train_001.jpg', 0.1), ('train_002.jpg', 0.2), ('train_003.jpg', 0.3),...]
    test_data = [('test_001.jpg', 0.1), ('test_002.jpg', 0.2), ('test_003.jpg', 0.3),...]
    model_path ='model_trained.h5'
    # 训练模型
    train_model(model_path, train_data)
    # 测试模型
    test_model(model_path)
```
