        # 选择最优交易机会
        selected_opportunities = trading_opportunities.sort_values(by=0)[-1]
        
        # 执行交易
        if selected_opportunities.iloc[0] > 0:
            buy_price = selected_opportunities.iloc[0]
            sell_price = 0
            
            # 确认交易
            if sell_price > buy_price:
                # 卖出
                df.loc[df['Id'] == selected_opportunities.index[0], 'BuyPrice'] = sell_price
