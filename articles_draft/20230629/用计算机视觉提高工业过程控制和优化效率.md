4.4. 代码讲解说明

```python
import cv2
import torch
import numpy as np

class VideoCapture:
    def __init__(self, camera_id):
        self.camera = cv2.VideoCapture(camera_id)
        self.cap = self.camera.read()
        self.cap.set(cv2.CAP_PROP_FPS, 30)

    def is_ready(self):
        return self.cap.isOpened()

    def show_frame(self):
        ret, frame = self.cap.read()
        if ret:
            cv2.imshow('frame', frame)
            if cv2.waitKey(1) == ord('q'):
                cv2.destroyAllWindows()
                return
        else:
            return

    def run(self):
        while True:
            ret, frame = self.cap.read()
            if ret:
                cv2.imshow('frame', frame)
                if cv2.waitKey(1) == ord('q'):
                    cv2.destroyAllWindows()
                    return
            else:
                break

def vision_based_ surveillance(camera_id):
    cap = VideoCapture(camera_id)
    cap.run()
    return cap
```
