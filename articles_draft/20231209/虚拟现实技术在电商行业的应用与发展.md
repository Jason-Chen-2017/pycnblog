2. 图像处理算法的实现：

```python
import cv2
import numpy as np

# 图像输入
def image_input(image_path):
    image = cv2.imread(image_path)
    return image

# 图像处理
def image_process(image):
    gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    blur_image = cv2.GaussianBlur(gray_image, (5, 5), 0)
    edge_image = cv2.Canny(blur_image, 50, 150)
    return edge_image

# 图像输出
def image_output(edge_image, output_path):
    cv2.imwrite(output_path, edge_image)

# 主函数
if __name__ == '__main__':
    image = image_input(image_path)
    edge_image = image_process(image)
    image_output(edge_image, output_path)
```
