## 4.1 光子的生成、传播、相互作用
```python
import numpy as np
import matplotlib.pyplot as plt

# 生成光子
def generate_photon(wavelength, amplitude):
    return amplitude * np.exp(-1j * 2 * np.pi * wavelength * np.random.rand())

# 传播光子
def propagate_photon(photon, distance):
    return photon * np.exp(-1j * 2 * np.pi * distance / wavelength)

# 相互作用光子
def interact_photons(photon1, photon2, distance):
    return photon1 + photon2 * np.exp(-1j * 2 * np.pi * distance / wavelength)

# 生成光子的波动行为
def photon_wave(wavelength, amplitude, distance):
    photon = generate_photon(wavelength, amplitude)
    photon = propagate_photon(photon, distance)
    return photon

# 绘制光子的波动行为
def plot_photon_wave(wavelength, amplitude, distance):
    photon = photon_wave(wavelength, amplitude, distance)
    plt.plot(photon.real, photon.imag)
    plt.show()

# 设置参数
wavelength = 500e-9  # 光子波长
amplitude = 1.0     # 光子振幅
distance = 1.0e-2   # 光子传播距离

# 生成光子
photon = generate_photon(wavelength, amplitude)

# 传播光子
photon = propagate_photon(photon, distance)

# 绘制光子的波动行为
plot_photon_wave(wavelength, amplitude, distance)
```

## 4.2 光学图像处理
```python
import numpy as np
import matplotlib.pyplot as plt

# 读取图像
def read_image(file_path):
    return plt.imread(file_path)

# 光学滤波
def photon_filter(image, kernel):
    return np.convolve(image, kernel, mode='same')

# 光学变换
def photon_transform(image, transform_matrix):
    return np.dot(image, transform_matrix)

# 光学融合
def photon_fusion(image1, image2):
    return (image1 + image2) / 2.0

# 读取图像

# 光学滤波
kernel = np.array([[0, -1, 0], [-1, 4, -1], [0, -1, 0]])
filtered_image = photon_filter(image, kernel)

# 光学变换
transform_matrix = np.array([[1, 0], [0, -1]])
transformed_image = photon_transform(image, transform_matrix)

# 光学融合
fused_image = photon_fusion(filtered_image, transformed_image)

# 绘制图像
plt.imshow(fused_image)

import numpy as np
import matplotlib.pyplot as plt

# 读取光学数据
def read_optical_data(file_path):
    return np.genfromtxt(file_path, delimiter=',')

# 光学测距
def photon_distance(data, distance):
    return data / (1 + distance / wavelength)

# 光学测角
def photon_angle(data, angle):
    return data * np.cos(angle)

# 读取光学数据
data = read_optical_data('data.csv')

# 光学测距
distance = 1.0e-2
distance_data = photon_distance(data, distance)

# 光学测角
angle = np.radians(45)
angle_data = photon_angle(data, angle)

# 绘制数据
plt.plot(distance_data, angle_data)
plt.show()
```

## 4.4 光学测量

import numpy as np
import matplotlib.pyplot as plt

# 读取光学数据
def read_optical_data(file_path):
    return np.genfromtxt(file_path, delimiter=',')

# 光学光度
def photon_irradiance(data, irradiance):
    return data * irradiance

# 光学强度
def photon_intensity(data, intensity):
    return data * intensity

# 光学弧度
def photon_radiance(data, radiance):
    return data * radiance

radiance_data = photon_radiance(data, radiance)
