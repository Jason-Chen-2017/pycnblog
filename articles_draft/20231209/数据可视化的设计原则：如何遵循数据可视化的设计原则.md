```python
import matplotlib.pyplot as plt
import pandas as pd

# 读取数据
data = pd.read_csv('traffic_data.csv')

# 创建数据可视化
fig, ax = plt.subplots()

# 设置标签、标题、图例
ax.set_title('Traffic Congestion in City')
ax.set_xlabel('Time')
ax.set_ylabel('Location')
ax.set_zlabel('Congestion Level')
ax.legend(['Red', 'Yellow', 'Green'])

# 设置颜色、尺寸、图案等元素
colors = ['red', 'yellow', 'green']
sizes = [20, 30, 40]
markers = ['o', 's', 'd']

# 绘制数据可视化
for i in range(len(data)):
    ax.scatter(data['Time'][i], data['Location'][i], data['Congestion Level'][i],
               color=colors[data['Color'][i]], s=sizes[data['Size'][i]], marker=markers[data['Marker'][i]])

# 设置布局、组织、排列等方法
plt.show()
```
