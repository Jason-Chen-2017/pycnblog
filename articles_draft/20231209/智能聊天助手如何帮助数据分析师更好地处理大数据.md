## 4.1 文本预处理
```python
import re
import nltk

def preprocess_text(text):
    # 删除标点符号
    text = re.sub(r'[^\w\s]', '', text)
    # 将文本拆分为单词
    words = nltk.word_tokenize(text)
    # 将单词转换为小写
    words = [word.lower() for word in words]
    # 返回预处理后的文本
    return words
```

## 4.2 词性标注
```python
import nltk

def tag_words(words):
    # 使用 NLTK 库进行词性标注
    tagged_words = nltk.pos_tag(words)
    # 返回标注后的单词
    return tagged_words
```

## 4.3 命名实体识别
```python
import nltk

def named_entity_recognition(words):
    # 使用 NLTK 库进行命名实体识别
    named_entities = nltk.ne_chunk(words)
    # 返回命名实体
    return named_entities
```

## 4.4 语义分析
```python
import spacy

def semantic_analysis(text):
    # 加载 spacy 模型
    nlp = spacy.load('en_core_web_sm')
    # 使用 spacy 进行语义分析
    doc = nlp(text)
    # 返回语义分析结果
    return doc
```

## 4.5 自然语言理解
```python
def natural_language_understanding(question):
    # 预处理问题
    words = preprocess_text(question)
    # 进行词性标注
    tagged_words = tag_words(words)
    # 进行命名实体识别
    named_entities = named_entity_recognition(words)
    # 进行语义分析
    doc = semantic_analysis(question)
    # 返回自然语言理解结果
    return (tagged_words, named_entities, doc)
```
