2. Java代码实例：

```java
import javax.crypto.Cipher;
import javax.crypto.KeyGenerator;
import javax.crypto.SecretKey;
import javax.crypto.spec.IvParameterSpec;
import javax.crypto.spec.SecretKeySpec;
import java.nio.charset.StandardCharsets;
import java.security.SecureRandom;
import java.util.Base64;

public class Encryption {
    public static void main(String[] args) throws Exception {
        String message = "Hello, World!";
        SecretKey secretKey = generateSecretKey();
        byte[] encrypted = encrypt(message.getBytes(StandardCharsets.UTF_8), secretKey);
        byte[] decrypted = decrypt(encrypted, secretKey);
        System.out.println(new String(decrypted, StandardCharsets.UTF_8));
    }

    private static SecretKey generateSecretKey() throws Exception {
        KeyGenerator keyGenerator = KeyGenerator.getInstance("AES");
        SecureRandom secureRandom = new SecureRandom();
        keyGenerator.init(128, secureRandom);
        return keyGenerator.generateKey();
    }

    private static byte[] encrypt(byte[] message, SecretKey secretKey) throws Exception {
        Cipher cipher = Cipher.getInstance("AES/CBC/PKCS5Padding");
        IvParameterSpec ivParameterSpec = new IvParameterSpec(new byte[16]);
        cipher.init(Cipher.ENCRYPT_MODE, secretKey, ivParameterSpec);
        return cipher.doFinal(message);
    }

    private static byte[] decrypt(byte[] encrypted, SecretKey secretKey) throws Exception {
        Cipher cipher = Cipher.getInstance("AES/CBC/PKCS5Padding");
        IvParameterSpec ivParameterSpec = new IvParameterSpec(new byte[16]);
        cipher.init(Cipher.DECRYPT_MODE, secretKey, ivParameterSpec);
        return cipher.doFinal(encrypted);
    }
}
```

3. C++代码实例：

```cpp
#include <iostream>
#include <openssl/aes.h>
#include <openssl/evp.h>
#include <openssl/rand.h>
#include <cstring>
#include <base64.h>

std::string encrypt(const std::string& message) {
    EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
    EVP_EncryptInit_ex(ctx, EVP_aes_256_cbc(), NULL, NULL, NULL);
    EVP_EncryptUpdate(ctx, NULL, NULL, message.data(), message.size());
    unsigned char* encrypted = (unsigned char*)malloc(EVP_CIPHER_CTX_get_output_length(ctx));
    unsigned int encrypted_len = EVP_EncryptUpdate(ctx, encrypted, NULL, message.data(), message.size());
    EVP_CIPHER_CTX_cleanup(ctx);
    std::string result(encrypted, encrypted + encrypted_len);
    free(encrypted);
    return result;
}

std::string decrypt(const std::string& encrypted) {
    EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
    EVP_DecryptInit_ex(ctx, EVP_aes_256_cbc(), NULL, NULL, NULL);
    EVP_DecryptUpdate(ctx, NULL, NULL, encrypted.data(), encrypted.size());
    unsigned char* decrypted = (unsigned char*)malloc(EVP_CIPHER_CTX_get_output_length(ctx));
    unsigned int decrypted_len = EVP_DecryptUpdate(ctx, decrypted, NULL, encrypted.data(), encrypted.size());
    EVP_CIPHER_CTX_cleanup(ctx);
    std::string result(decrypted, decrypted + decrypted_len);
    free(decrypted);
    return result;
}

int main() {
    std::string message = "Hello, World!";
    std::string encrypted = encrypt(message);
    std::string decrypted = decrypt(encrypted);
    std::cout << decrypted << std::endl;
    return 0;
}
```
