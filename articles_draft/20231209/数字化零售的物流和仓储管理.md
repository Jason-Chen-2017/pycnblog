### 1.4.1 物流路径规划的具体代码实例

```python
import math

def calculate_distance(x1, y1, x2, y2):
    return math.sqrt((x1 - x2)**2 + (y1 - y2)**2)

def find_shortest_path(x1, y1, x2, y2):
    distances = []
    for x, y in [(x1, y1), (x2, y2)]:
        for x_, y_ in [(x1, y1), (x2, y2)]:
            if x != x_ and y != y_:
                distances.append(calculate_distance(x, y, x_, y_))
    return min(distances)

x1, y1 = 1, 1
x2, y2 = 2, 2
shortest_distance = find_shortest_path(x1, y1, x2, y2)
print(shortest_distance)
```

### 1.4.2 仓储资源分配的具体代码实例

```python
def calculate_demand(demand_list):
    total_demand = 0
    for demand in demand_list:
        total_demand += demand
    return total_demand

def allocate_resources(resources, demand_list):
    allocated_resources = {}
    for resource, capacity in resources.items():
        allocated_resources[resource] = 0
        for demand in demand_list:
            if demand in allocated_resources:
                allocated_resources[demand] += 1
            elif demand not in allocated_resources:
                allocated_resources[demand] = min(capacity, demand)
    return allocated_resources

resources = {'A': 10, 'B': 20, 'C': 30}
demand_list = ['A', 'B', 'C', 'D', 'E']
allocated_resources = allocate_resources(resources, demand_list)
print(allocated_resources)
```
