### 4.3 遗传算法

```python
import numpy as np

# 目标函数
def f(x):
    return -x[0]**2 - 5*x[1]**2 + 10*x[0]*x[1] + 10

# 初始化种群
pop_size = 100
x_min, x_max = -5, 5
pop = np.random.uniform(x_min, x_max, (pop_size, 2))

# 初始化参数
alpha = 0.5
beta = 1
num_generations = 100

# 遗传算法
for _ in range(num_generations):
    # 计算适应度
    fitness = np.array([f(x) for x in pop])
    # 选择
    selected = np.random.choice(pop_size, pop_size, p=fitness/fitness.sum())
    # 交叉
    crossover_points = np.random.randint(1, 3, (pop_size, 2))
    offspring = np.concatenate([pop[selected][:, :crossover_points[:, 0]], pop[selected][:, crossover_points[:, 1]:]], axis=1)
    # 变异
    for i in range(pop_size):
        if np.random.rand() < 0.1:
            offspring[i] += np.random.uniform(-1, 1, 2)
    # 更新种群
    pop = offspring

# 最佳解
best_x = pop[np.argmin(fitness)]
print(best_x)
```
