数据清洗的代码实例如下：

-- 插值缺失值
UPDATE table SET column = (SELECT AVG(column) FROM table) WHERE column IS NULL;

-- 数据类型转换

-- 数据格式转换
SELECT DATE_FORMAT(column, '%Y-%m-%d') FROM table;

-- 数据归一化

-- 最小-最大归一化
SELECT (column - MIN(column)) / (MAX(column) - MIN(column)) FROM table;
