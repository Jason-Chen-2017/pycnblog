def pagerank(G, alpha=0.85, max_iter=1000, tol=1e-6):
    N = len(G.nodes())
    PR = [1/N] * N
    for _ in range(max_iter):
        new_PR = [sum(PR[nbr] for nbr in G[node].iter_inbound()) * alpha + (1 - alpha)/N for node in G.nodes()]
        diff = np.linalg.norm(new_PR - PR)
        if diff < tol:

        PR = new_PR

# 5.未来发展趋势与挑战

1.人工智能技术的深入融入教育，为教育提供更加智能化、个性化、适应性的支持。
