```python
import nltk

# 定义文本分类任务
def classify_text(text):
    # 预处理文本
    tokens = nltk.word_tokenize(text)
    # 提取特征
    features = nltk.FeatureExtractor(tokens)
    # 训练模型
    model = nltk.NaiveBayesClassifier.train(features)
    # 预测类别
    category = model.classify(tokens)
    return category
```

predictions = algo.test(trainset)

# 评估推荐系统
evaluator = surprise.RecallEvaluator()
rmse = evaluator.compute(predictions, verbose=True)
