```python
pragma solidity ^0.5.16;

contract SimpleContract {
    address public owner;
    uint256 public totalSupply;

    constructor() public {
        owner = msg.sender;
        totalSupply = 1000000 * (10 ** uint256(decimals));
    }

    function transfer(address payable recipient, uint256 amount) public payable {
        require(msg.value == amount, "SimpleContract: Amount must match sent value.");
        recipient.transfer(amount);
    }

    function withdraw(uint256 amount) public {
        require(msg.sender == owner, "SimpleContract: Only the owner can withdraw.");
        owner.transfer(amount);
    }
}
```

- 构造函数：构造函数用于初始化合约的状态，包括合约的所有者和总供应量。
- transfer 函数：transfer 函数用于将 Ether 从发送者转到接收者的地址，并将金额记录在合约的状态中。
- withdraw 函数：withdraw 函数用于从合约的地址中提现金额，并将金额转到合约的所有者的地址。

这个智能合约的代码实现了以下算法原理：
