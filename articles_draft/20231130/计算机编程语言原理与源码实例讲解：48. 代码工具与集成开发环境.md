```python
from pygments.lexer import PygmentsLexer
from pygments.token import *

def format_code(code, language):
    lexer = PygmentsLexer(language)
    tokens = lexer.get_tokens(code)
    formatted_code = []
    indent_level = 0
    for token in tokens:
        if token.type == NAME:
            if token.string in language_keywords:
                formatted_code.append(token.string)
            else:
                formatted_code.append(' ' * indent_level + token.string)
                indent_level += 1
        else:
            formatted_code.append(token.string)
    return ''.join(formatted_code)
```
