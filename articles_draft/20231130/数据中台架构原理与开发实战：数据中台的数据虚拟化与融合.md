class DataVirtualization:
    def __init__(self):
        self.data_sources = []
        self.data_applications = []

    def add_data_source(self, data_source):
        self.data_sources.append(data_source)

    def add_data_application(self, data_application):
        self.data_applications.append(data_application)

    def get_data(self, data_application_name):
        for data_application in self.data_applications:
            if data_application.name == data_application_name:
                return data_application.get_data()
        return None
