class UserManager:
    def __init__(self):
        self.users = {}

    def authenticate(self, username, password):
        if username in self.users and self.users[username] == password:
            return True
        return False

    def assign_permission(self, username, permission):
        if username in self.users:
            self.users[username].append(permission)
        else:
            self.users[username] = [permission]

    def audit_permission(self, username):
        if username in self.users:
            print(self.users[username])

class ProcessManager:
    def __init__(self, user_manager):
        self.users = user_manager
        self.processes = {}

    def create_process(self, username, process):
        if username in self.users and self.users[username].has_permission('create_process'):
            self.processes[process.id] = process
            return True
        return False

    def destroy_process(self, process_id):
        if process_id in self.processes:
            del self.processes[process_id]
            return True
        return False

    def control_resource(self, process_id, resource_id):
        if process_id in self.processes and resource_id in self.processes[process_id].resources:
            return True
        return False

class ResourceManager:
    def __init__(self):
        self.resources = {}

    def monitor_resource(self, resource_id):
        if resource_id in self.resources:
            print(self.resources[resource_id])

    def protect_resource(self, resource_id):
        if resource_id in self.resources:
            self.resources[resource_id].protect()

    def recover_resource(self, resource_id):
        if resource_id in self.resources:
            self.resources[resource_id].recover()
