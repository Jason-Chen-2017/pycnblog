from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import Pipeline

# 数据预处理
X = data['date'].values.reshape(-1, 1)

# 数据标准化
scaler = StandardScaler()

# 创建SVR模型

# 训练模型
model.fit(X, y)

# 预测价格
predicted_price = model.predict(X)

# 输出结果
print('预测价格：', predicted_price)
