import numpy as np
import random

def fitness_function(x):
    return np.sum(np.abs(x))

def distance(x1, x2):
    return np.abs(x1 - x2)

def main():
    population_size = 100
    max_iterations = 1000
    x_bounds = [-10, 10]

    x_population = np.random.uniform(x_bounds[0], x_bounds[1], population_size)
    antibodies = np.array([fitness_function(x) for x in x_population])

    for _ in range(max_iterations):
        antibody_probabilities = np.exp(-antibodies) / np.sum(np.exp(-antibodies))
        activated_antibodies = np.random.choice(population_size, size=population_size, p=antibody_probabilities)

        for i in range(population_size):
            if np.random.rand() < 0.1:
                x_population[i] += random.uniform(-1, 1)

        antibodies = np.array([fitness_function(x) for x in x_population])

    best_antibody_index = np.argmin(antibodies)
    best_antibody = x_population[best_antibody_index]

    print("Best antibody:", best_antibody)
    print("Fitness:", fitness_function(best_antibody))

if __name__ == "__main__":
    main()
