import numpy as np

def f(x):
    return -(x[0]**2 + x[1]**2)

def R(x):
    return np.random.rand(2)

def S(x):
    return x + R(x)

def WHO(n, max_iter):
    X = np.random.rand(n, 2)
    best_x = X[0]
    best_f = f(best_x)

    for iter in range(max_iter):
        for i in range(n):
            x_new = S(X[i])
            f_new = f(x_new)
            if f_new > best_f:
                best_x = x_new
                best_f = f_new

            X[i] = x_new

        if np.linalg.norm(np.diff(X).ravel()) < 1e-6:
            break

    return best_x, best_f

n = 20
max_iter = 1000
best_x, best_f = WHO(n, max_iter)
print("最佳鲸鱼位置：", best_x)
print("最佳适应度：", best_f)
