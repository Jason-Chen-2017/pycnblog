```python
import numpy as np
import pandas as pd

# 数据去噪
def denoise(data):
    return np.median(data)

# 数据填充
def fill(data, value):
    return np.where(np.isnan(data), value, data)

# 数据转换
def transform(data):
    return np.log(data + 1)

# 数据归一化
def normalize(data):
    return (data - np.min(data)) / (np.max(data) - np.min(data))

# 读取数据
data = pd.read_csv('data.csv')

# 数据清洗和预处理
data['data'] = data['data'].apply(denoise)
data['data'] = data['data'].apply(fill, value=0)
data['data'] = data['data'].apply(transform)
data['data'] = data['data'].apply(normalize)
```
