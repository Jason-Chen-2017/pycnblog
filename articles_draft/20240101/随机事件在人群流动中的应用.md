import matplotlib.pyplot as plt

d = np.array([1, 2, 3, 4, 5])
rho = np.array([1, 2, 3, 4, 5])

# 随机事件检测
def detect_random_event(v, d, rho):
    delta_v = v[1:] - v[:-1]
    delta_d = d[1:] - d[:-1]
    delta_rho = rho[1:] - rho[:-1]
    return np.any(np.abs(delta_v) > 1) or np.any(np.abs(delta_d) > 1) or np.any(np.abs(delta_rho) > 1)

# 人群流动模拟
def simulate_pedestrian_flow(v, d, rho):
    v = np.append(v, v[-1])
    d = np.append(d, d[-1])
    rho = np.append(rho, rho[-1])
    return v, d, rho

# 人群安全评估
def evaluate_safety(v, d, rho):
    safety = np.sum(np.abs(v) < 1) / len(v)
    return safety

# 主程序
if detect_random_event(v, d, rho):
    v, d, rho = simulate_pedestrian_flow(v, d, rho)
    safety = evaluate_safety(v, d, rho)
    print("人群安全度：", safety)
else:
    print("未发生随机事件")
