## 4.1 多语言支持的代码实例

```python
languages = ['en', 'zh', 'ja', 'fr']

# 翻译函数
def translate(text, from_lang, to_lang):
    # 翻译逻辑
    # ...
    return translated_text

# 多语言支持功能
def support_multiple_languages(text, languages):
    for from_lang, to_lang in zip(languages, languages[1:]):
        translated_text = translate(text, from_lang, to_lang)
        print(f'{from_lang} -> {to_lang}: {translated_text}')

# 使用多语言支持功能
support_multiple_languages('Hello, world!', languages)
```
