def fish_swarm_optimization(x0, c1, c2, w, n_iter):
    x_best = x0
    g_best = x0
    v = np.random.rand(len(x0))
    for t in range(n_iter):
        r1 = np.random.rand()
        r2 = np.random.rand()
        v = w * v + c1 * r1 * (x_best - x0) + c2 * r2 * (g_best - x0)
        x0 = x0 + v
        if np.linalg.norm(x0 - x_best) < np.linalg.norm(x_best - g_best):
            x_best = x0
        if np.linalg.norm(x0 - g_best) < np.linalg.norm(x_best - g_best):
            g_best = x0
    return g_best
