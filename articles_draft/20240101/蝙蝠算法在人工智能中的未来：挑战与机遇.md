    # 更新蝙蝠的位置
    new_position = position + velocity
    new_energy = energy + alpha * (food_position - position) + gamma * np.random.randn(1, 1) * (position - velocity) + beta * np.random.randn(1, 1) * (velocity)

    new_food_position = food_position + update_value
