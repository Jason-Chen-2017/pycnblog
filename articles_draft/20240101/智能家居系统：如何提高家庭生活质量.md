# 4. 具体代码实例和详细解释说明

在本节中，我们将通过一个具体的智能家居系统实例来详细解释代码实例和解释说明。

假设我们有一个智能家居系统，包括智能灯泡、智能空气净化器、智能门锁、智能摄像头等设备。我们将通过一个简单的Python程序来实现这些设备的控制和监控。

```python
import time
import requests

class SmartHome:
    def __init__(self):
        self.light = Light()
        self.air_purifier = AirPurifier()
        self.lock = Lock()
        self.camera = Camera()

    def control_light(self, action):
        if action == 'on':
            self.light.on()
        elif action == 'off':
            self.light.off()
        elif action == 'brightness':
            brightness = input('请输入亮度值（0-100）：')
            self.light.set_brightness(int(brightness))
        elif action == 'color':
            color = input('请输入颜色（红、绿、蓝、白）：')
            self.light.set_color(color)

    def control_air_purifier(self, action):
        if action == 'on':
            self.air_purifier.on()
        elif action == 'off':
            self.air_purifier.off()
        elif action == 'strength':
            strength = input('请输入滤气强度（1-10）：')
            self.air_purifier.set_strength(int(strength))

    def control_lock(self, action):
        if action == 'open':
            self.lock.open()
        elif action == 'close':
            self.lock.close()

    def control_camera(self, action):
        if action == 'start':
            self.camera.start()
        elif action == 'stop':
            self.camera.stop()

if __name__ == '__main__':
    smart_home = SmartHome()
    while True:
        action = input('请输入操作命令（开、关、亮度、颜色、滤气强度、开始、停止）：')
        if action in ['开', '关', '亮度', '颜色', '滤气强度', '开始', '停止']:
            smart_home.control_light(action)
        elif action in ['开', '关']:
            smart_home.control_air_purifier(action)
        elif action in ['开', '关']:
            smart_home.control_lock(action)
        elif action in ['开始', '停止']:
            smart_home.control_camera(action)
        else:
            print('未知操作命令，请重新输入。')
```

在这个Python程序中，我们首先导入了`time`和`requests`库。然后定义了一个`SmartHome`类，该类包含了智能灯泡、智能空气净化器、智能门锁、智能摄像头等设备的对象。在`control_light`、`control_air_purifier`、`control_lock`和`control_camera`方法中，我们根据不同的操作命令调用了对应的设备方法来实现设备的控制和监控。在主程序中，我们创建了一个`SmartHome`对象，并根据用户输入的操作命令来调用对应的方法。
