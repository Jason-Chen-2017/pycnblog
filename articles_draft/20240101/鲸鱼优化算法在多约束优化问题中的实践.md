    f = -np.sum(x**2)
    return f

def update_position(whale_group, A, r):
    new_whale_group = whale_group.copy()
    for i in range(whale_group.shape[0]):
        new_whale_group[i] = whale_group[i] + A * r * (whale_group[i] - whale_group[np.argmin(whale_group[:, 0])])

    best_solution = np.min(whale_group, axis=0)
    return best_solution
