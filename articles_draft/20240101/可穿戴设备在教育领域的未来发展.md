```python
import datetime

class Student:
    def __init__(self, name):
        self.name = name
        self.learning_time = 0
        self.learning_content = []
        self.learning_effect = 0

    def learn(self, content, time):
        self.learning_time += time
        self.learning_content.append(content)
        self.learning_effect += content.score

class Content:
    def __init__(self, name, score):
        self.name = name
        self.score = score

student = Student("Alice")
english = Content("English", 100)
student.learn(english, 2)
```
