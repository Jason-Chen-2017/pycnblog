```python
import numpy as np

def random_state():
    return np.random.randint(0, 100)

def cost(state, action):
    return np.abs(state - action)

def mc_path_planning(start, goal, T):
    state = start
    best_state = start
    best_cost = float('inf')

    for _ in range(T):
        action = random_state()
        cost_value = cost(state, action)

        if cost_value < best_cost:
            best_state = state
            best_cost = cost_value

        state = action

    return best_state

start = 50
goal = 90
T = 1000

best_state = mc_path_planning(start, goal, T)
print("Best state:", best_state)
```
