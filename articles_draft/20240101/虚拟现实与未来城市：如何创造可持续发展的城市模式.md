# 4.具体代码实例和详细解释说明

在这里，我们以Unity引擎为例，给出一个简单的虚拟建筑创建和交互的代码实例。

```csharp
using UnityEngine;
using System.Collections;

public class Building : MonoBehaviour
{
    public string buildingName;
    public GameObject entrance;

    void Start()
    {
        // 创建虚拟建筑
        GameObject building = new GameObject(buildingName);
        building.AddComponent<Renderer>();
        building.AddComponent<Collider>();

        // 设置物理属性
        Rigidbody rb = building.AddComponent<Rigidbody>();
        rb.mass = 1000.0f;

        // 设置交互属性
        building.AddComponent<BuildingInteraction>();
    }

    void OnTriggerEnter(Collider other)
    {
        // 当玩家进入虚拟建筑的触发区域时，触发交互事件
        if (other.gameObject.CompareTag("Player"))
        {
            entrance.SetActive(true);
        }
    }

    void OnTriggerExit(Collider other)
    {
        // 当玩家离开虚拟建筑的触发区域时，取消交互事件
        if (other.gameObject.CompareTag("Player"))
        {
            entrance.SetActive(false);
        }
    }
}
```
