### 4.1 推荐系统代码实例

在这个例子中，我们将使用Python编程语言来实现一个基于协同过滤算法的推荐系统。代码如下：

```python
import numpy as np

# 用户评分矩阵
user_rating_matrix = np.array([[4, 3, 2],
                               [3, 4, 5],
                               [2, 5, 4]])

# 计算用户之间的相似度
def similarity(user1, user2):
    return np.dot(user1, user2) / (np.linalg.norm(user1) * np.linalg.norm(user2))

# 计算物品之间的相似度
def similarity_items(item1, item2):
    return np.dot(item1, item2) / (np.linalg.norm(item1) * np.linalg.norm(item2))

# 获取用户的评分
def get_user_rating(user, item):
    return np.dot(user, item) / np.linalg.norm(user)

# 获取物品的评分
def get_item_rating(user, item):
    return np.dot(user, item) / np.linalg.norm(item)

# 获取推荐列表
def recommend(user, items, k=3):
    user_similarities = {}
    for i in range(len(items)):
        item = items[i]
        similarity = similarity_items(user, item)
        if similarity not in user_similarities:
            user_similarities[similarity] = []
        user_similarities[similarity].append(item)

    sorted_similarities = sorted(user_similarities.items(), key=lambda x: x[0], reverse=True)
    recommended_items = []
    for similarity, items in sorted_similarities[:k]:
        for item in items:
            recommended_items.append(item)

    return recommended_items

# 测试推荐系统
user = user_rating_matrix[0, :]
items = user_rating_matrix[:, 1]
recommended_items = recommend(user, items)
print(recommended_items)
```

### 4.2 搜索引擎代码实例

在这个例子中，我们将使用Python编程语言来实现一个基本的搜索引擎。代码如下：

```python
import re

# 文档列表
documents = [
    {'id': 1, 'content': '人工智能是人类创造的智能'},
    {'id': 2, 'content': '人工智能是机器学习的一种'},
    {'id': 3, 'content': '人工智能可以帮助人类解决问题'},
    {'id': 4, 'content': '机器学习是人工智能的一个分支'},
]

# 创建索引
def create_index(documents):
    index = {}
    for document in documents:
        words = re.split(r'\W+', document['content'].lower())
        for word in words:
            if word not in index:
                index[word] = []
            index[word].append(document['id'])
    return index

# 搜索
def search(query, index):
    words = re.split(r'\W+', query.lower())
    results = []
    for word in words:
        if word in index:
            results.extend(index[word])
    return results

# 测试搜索引擎
index = create_index(documents)
search_query = '人工智能'
results = search(search_query, index)
print(results)
```
