def logistic_regression(X, y, lambda_):
    m, n = X.shape
    theta = np.zeros(n)
    learning_rate = 0.01
    iterations = 1000

    for i in range(iterations):
        prediction = X.dot(theta)

        gradient = (1 / m) * X.T.dot((prediction - y))

    for i, s in enumerate(S):
