import numpy as np

# 定义目标函数
def f(x):
    return np.sum(x**2)

# 随机搜索
def random_search(x_min, x_max, n_iter):
    x = (x_min + x_max) / 2
    for _ in range(n_iter):
        x = np.random.uniform(x_min, x_max)
        yield x

# 梯度下降
def gradient_descent(x_min, x_max, n_iter, learning_rate):
    x = (x_min + x_max) / 2
    for _ in range(n_iter):
        gradient = np.array([2 * x_i for x_i in x])
        x -= learning_rate * gradient
        yield x

# 粒子群优化
def particle_swarm_optimization(x_min, x_max, n_iter, n_particles):
    x = np.zeros((n_particles, n))
    v = np.zeros((n_particles, n))
    p_best = np.zeros((n_particles, n))
    g_best = np.zeros(n)
    for _ in range(n_iter):
        for i in range(n_particles):
            r1, r2 = np.random.rand(2)
            v[i] += r1 * (p_best[i] - x[i]) + r2 * (g_best - x[i])
            x[i] += v[i]
            if f(x[i]) < f(p_best[i]):
                p_best[i] = x[i]
                if f(x[i]) < f(g_best):
                    g_best = x[i]
        yield g_best

# 评估算法性能
def evaluate_performance(algorithms, x_min, x_max, n_iter):
    results = []
    for algorithm in algorithms:
        results.append(list(algorithm(x_min, x_max, n_iter)))
    return results

        print("Best solution:", results[i][-1])
