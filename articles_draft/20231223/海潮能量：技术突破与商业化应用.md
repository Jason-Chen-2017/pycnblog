def tides_energy(tides_amplitude, tides_frequency, device_efficiency):
    mechanical_power = tides_amplitude**2 * tides_frequency
    electrical_power = mechanical_power * device_efficiency
    return electrical_power

tides_amplitude = 1.0
tides_frequency = 1.0
device_efficiency = 0.5

output_power = tides_energy(tides_amplitude, tides_frequency, device_efficiency)
print("Output power:", output_power)
