### 4.2 动态规划示例

```python
def dynamic_programming(n, w, v):
    dp = [0] * (n + 1)

    for i in range(1, n + 1):
        for j in range(i, n + 1):
            if w[j] <= w[i - 1]:
                dp[j] = max(dp[j], dp[i - 1] + v[j])
            else:
                dp[j] = max(dp[j], dp[i - 1])

    return dp[n]

n = 5
w = [2, 3, 4, 5, 6]
v = [6, 5, 4, 3, 2]

print(dynamic_programming(n, w, v))
```

### 4.3 模拟优化示例

```python
import numpy as np

def simulated_annealing(f, x0, T, alpha, cool_time):
    x = x0
    T_current = T

    for i in range(cool_time):
        T_next = T_current * alpha
        x_next = x + np.random.randn(len(x)) * np.sqrt(T_current)
        if f(x_next) < f(x):
            x = x_next
        T_current = T_next

    return x

def f(x):
    return -(x[0]**2 + x[1]**2)

x0 = np.array([0, 0])
T = 100
alpha = 0.99
cool_time = 1000

x = simulated_annealing(f, x0, T, alpha, cool_time)

print(x)
```
