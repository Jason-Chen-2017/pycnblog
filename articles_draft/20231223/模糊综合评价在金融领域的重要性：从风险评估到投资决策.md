import numpy as np

# 评价指标和评价级别

# 将评价指标转换为模糊语言
def fuzzify(criterion):
    if criterion <= 0.2:
        return '非常差'
    elif criterion <= 0.4:
        return '差'
    elif criterion <= 0.6:
        return '一般'
    elif criterion <= 0.8:
        return '好'
    else:
        return '非常好'

# 将模糊语言转换为模糊数
def membership_function(level):
    if level == '非常好':
        return np.array([0.8, 1])
    elif level == '好':
        return np.array([0.6, 0.8])
    elif level == '一般':
        return np.array([0.4, 0.6])
    elif level == '差':
        return np.array([0.2, 0.4])
    else:
        return np.array([0, 0.2])

# 将模糊数进行综合评价
def aggregate(criteria_membership):
    aggregated_membership = np.max(criteria_membership, axis=0)
    return aggregated_membership

# 对模糊评价结果进行解释和应用
def defuzzify(aggregated_membership):
    defuzzified_result = levels[np.argmax(aggregated_membership)]
    return defuzzified_result

# 示例数据
criterion1 = 0.3
criterion2 = 0.5

# 将示例数据转换为模糊语言
criteria_fuzzified = [fuzzify(criterion1), fuzzify(criterion2)]

# 将模糊语言转换为模糊数
criteria_membership = [membership_function(level) for level in criteria_fuzzified]

# 将模糊数进行综合评价
aggregated_membership = aggregate(criteria_membership)

# 对模糊评价结果进行解释和应用
defuzzified_result = defuzzify(aggregated_membership)
print(defuzzified_result)
