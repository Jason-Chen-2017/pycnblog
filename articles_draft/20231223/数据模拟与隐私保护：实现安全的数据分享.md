### 4.1 数据掩码方法代码实例

以下是一个使用Python语言实现的数据掩码方法代码实例：

import random

def data_mask(data):
    masked_data = []
    for d in data:
        masked_d = {}
        for k, v in d.items():
            masked_d[k] = v if random.random() < 0.5 else '*'
        masked_data.append(masked_d)
    return masked_data

data = [{'name': 'Alice', 'age': 25}, {'name': 'Bob', 'age': 30}]
masked_data = data_mask(data)
print(masked_data)
