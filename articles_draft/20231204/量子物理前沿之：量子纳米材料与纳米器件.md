## 4.2量子门的实现

```python
class HadamardGate:
    def __init__(self):
        pass

    def apply(self, qubit):
        qubit.state = (qubit.state * 1j + 1) / np.sqrt(2)

class CNOTGate:
    def __init__(self):
        pass

    def apply(self, control_qubit, target_qubit):
        if control_qubit.state == 1:
            target_qubit.state = 1 - target_qubit.state
```
