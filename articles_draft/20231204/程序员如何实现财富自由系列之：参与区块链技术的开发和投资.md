```python
import hashlib
import time

class Block:
    def __init__(self, index, previous_hash, timestamp, data, hash):
        self.index = index
        self.previous_hash = previous_hash
        self.timestamp = timestamp
        self.data = data
        self.hash = hash

    @staticmethod
    def calculate_hash(index, previous_hash, timestamp, data):
        block_string = str(index) + previous_hash + str(timestamp) + data
        return hashlib.sha256(block_string.encode()).hexdigest()

    @property
    def nonce(self):
        guess_hash = self.calculate_hash(self.index, self.previous_hash, self.timestamp, self.data)
        nonce = 0
        while not guess_hash.startswith('000000':
            nonce += 1
            guess_hash = self.calculate_hash(self.index, self.previous_hash, self.timestamp, self.data + str(nonce))
        return nonce

    def __str__(self):
        return 'Block #{}: {}'.format(self.index, self.data)

def create_genesis_block():
    return Block(0, '0', time.time(), 'Genesis Block', '0')

def create_next_block(last_block):
    data = 'New Block'
    index = last_block.index + 1
    previous_hash = last_block.hash
    timestamp = time.time()
    nonce = last_block.nonce + 1
    hash = Block.calculate_hash(index, previous_hash, timestamp, data, nonce)
    return Block(index, previous_hash, timestamp, data, hash)

def create_blockchain():
    genesis_block = create_genesis_block()
    blockchain = [genesis_block]
    return blockchain

def add_block(blockchain, data):
    last_block = blockchain[-1]
    new_block = create_next_block(last_block)
    blockchain.append(new_block)
    return blockchain

if __name__ == '__main__':
    blockchain = create_blockchain()
    print(blockchain)
    for _ in range(5):
        blockchain = add_block(blockchain, 'New Block')
        print(blockchain)
```

- 创建区块：通过`Block`类来创建一个区块，包括区块的索引、前一个区块的哈希、时间戳、交易记录、哈希值等信息。
- 计算哈希值：通过`calculate_hash`方法来计算区块的哈希值，使用SHA-256算法。
- 计算不可篡改性：通过`nonce`属性来计算区块的不可篡改性，使用PoW算法。
- 创建区块链：通过`create_blockchain`方法来创建一个区块链，包括一个初始区块。
- 添加区块：通过`add_block`方法来添加新的区块到区块链中，包括新的交易记录。
