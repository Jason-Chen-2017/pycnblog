class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __add__(self, other):
        return self._get_vector_add(other.x, other.y)

    def __mul__(self, other):
        return self._get_vector_mul(other.x, other.y)

    def _get_vector_add(self, x, y):
        return Vector(x + y, 0)

    def _get_vector_mul(self, x, y):
        return Vector(0, x * y)

class VectorSpace:
    def __init__(self, n):
        self.size = n
        self.base = Vector(0, 0)

    def is_base_vector(self, v):
        return v.x == 0 and v.y == 0

    def add(self, v):
        return self.base + v

    def __mul__(self, other):
        if self.is_base_vector(other):
            return self.base + other.base
        else:
            return self.base + other.x * other.y

    def __truediv__(self, other):
        if self.is_base_vector(other):
            return self.base / other.base
        else:
            return self.base / (other.x * other.y)

    def __repr__(self):
        return f"VectorSpace({self.size}, {self.base.x}, {self.base.y})"
