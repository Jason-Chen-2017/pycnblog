import numpy as np
import pandas as pd
from sklearn.preprocessing import StandardScaler
from keras.preprocessing.text import Tokenizer
from keras.preprocessing import StandardScaler
from keras.models import Sequential
from keras.layers import Dense

class Data:
    def __init__(self, dataframe):
        self.dataframe = dataframe

    def to_array(self):
        return np.array(self.dataframe.values)

    def to_token(self):
        tokens = []
        for index, value in self.dataframe.iteritems():
            tokens.append([word.lower() for word in value.split()])
        return tokens

# 清洗、去重、标准化
def clean_data(data):
    data = Data(data)
    data.dataframe = data.to_array()
    data.dataframe = data.to_token()
    data.dataframe = data.dataframe.astype(int)
    return data.dataframe

# 构建知识图谱
def build_knowledge_graph(data):
    scaler = StandardScaler()
    data = clean_data(data)
    data = scaler.fit_transform(data)
    return data

# 获取用户历史订单数据
user_history_orders = []
for user_id in range(1, 10):
    user_data = clean_data("user_history_orders_{}".format(user_id))
    user_history_orders.append(user_data.to_array())

# 构建知识图谱
knowledge_graph = build_knowledge_graph(user_history_orders)
