import torch
import numpy as np
import matplotlib.pyplot as plt

# 定义神经网络模型
class MLP(torch.nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim):
        super(MLP, self).__init__()
        self.layer1 = torch.nn.Linear(input_dim, hidden_dim)
        self.layer2 = torch.nn.Linear(hidden_dim, output_dim)

    def forward(self, x):
        out = torch.relu(self.layer1(x))
        out = torch.relu(self.layer2(out))
        return out

# 定义随机过程模型
class ARIMA(torch.nn.Module):
    def __init__(self, p, d, q):
        super(ARIMA, self).__init__()
        self.p = p
        self.d = d
        self.q = q

    def forward(self, x):
        out = np.random.arange(0, len(x), dtype=float)
        out = np.exp(-0.1 * out ** 2) * np.sin(0.1 * out ** 3) + np.random.randn(len(x)) * 0.5
        return out

# 定义数据集
train_data = np.random.randn(1000, 1)
test_data = np.random.randn(200, 1)

# 定义神经网络模型参数
input_dim = 1
hidden_dim = 64
output_dim = 1

# 训练神经网络模型
model = MLP(input_dim, hidden_dim, output_dim)
criterion = torch.nn.MSELoss()
optimizer = torch.optim.SGD(model.parameters(), lr=0.01)

# 训练随机过程模型
model_arima = ARIMA(p=1, d=2, q=3)

# 模型训练
for epoch in range(100):
    for i, data in enumerate(train_data):
        x = torch.from_numpy(data).float()
        y_pred = model(x)
        loss = criterion(y_pred.data - test_data[i], test_data[i])
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
    print('Epoch {}'.format(epoch + 1))
