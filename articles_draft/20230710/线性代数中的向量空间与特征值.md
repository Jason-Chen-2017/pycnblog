4.3. 核心代码实现

```
# 导入所需库
import numpy as np
import pandas as pd
from scipy.sparse import csr_matrix

# 读取数据
data = pd.read_csv('word_data.csv')

# 预处理数据
words = data.apply(lambda x: x.lower().split())
stopwords = set(pd.read_csv('stop_words.csv'))
words = [word for word in words if word not in stopwords]

# 构建矩阵
A = csr_matrix(words)

# 计算投影长度
投影长度 = A.apply(lambda x: np.linalg.norm(x, axis=1), axis=0)

# 计算聚类系数和聚类中心
Z = np.kron(A.T, A)
Cl = np.argmax(Z, axis=0)
C = np.sum(Z, axis=0, keepdims=True) / Cl

# 输出结果
print("聚类系数：")
print(Cl)
print("聚类中心：")
print(C)
```
