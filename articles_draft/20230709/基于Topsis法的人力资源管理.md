以下是Thesis法的代码实例，主要包括以下几个部分：

```python
import numpy as np
import random

# 1. 收集人力资源需求信息
employee_info = []
title_info = []
salary_info = []
for i in range(100):
    employee_info.append({'name': f'员工{i}', 'title': f'经理{i}','salary': random.random()})
    title_info.append({'name': f'员工{i}', 'title': f'助理{i}','salary': random.random()})
    salary_info.append({'name': f'员工{i}', 'title': f'程序员{i}','salary': random.random()})

# 2. 对需求信息进行分析
skills_info = []
for info in employee_info:
    skills_info.append(info['skills'])

# 3. 确定人力资源策略
strategy_info = []
for info in title_info:
    strategy_info.append({'title': info['name'],'salary': info['salary']})

for info in skills_info:
    strategy_info.append({'skill': info})

# 4. 对策略进行实施
action_info = []
for info in strategy_info:
    action_info.append({'title': info['title'],'salary': info['salary']})

# 5. 对实施效果进行评估
results = []
for info in action_info:
    results.append({'name': info['title'], 'value': info['salary']})

# 6. 绘制评估结果
import matplotlib.pyplot as plt
plt.bar(results)
plt.title('Thesis法人力资源管理评估结果')
plt.xlabel('人力资源策略')
plt.ylabel('评估结果')
plt.show()
```
