```python
import requests
import json

class User:
    def __init__(self, username, password):
        self.username = username
        self.password = password

#获取用户行为数据
url = "https://api.example.com/user_behavior"
params = {
    "username": "user1",
    "password": "password1"
}
response = requests.get(url, params=params)
data = json.loads(response.content)

#分析用户行为
for behavior in data["results"]:
    if behavior["action"] == "purchase":
        print(f"用户购买了运动鞋：{behavior['price']}元")
    else:
        print(f"用户做了其他行为：{behavior['action']}")
```
