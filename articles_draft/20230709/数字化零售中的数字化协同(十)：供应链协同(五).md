4.3. 核心代码实现

# 导入需要的库
import pandas as pd
import numpy as np
from sklearn.cluster import KMeans
from sklearn.linear_model import LinearRegression

# 读取数据
data = pd.read_csv('data.csv')

# 进行数据清洗和整合
data = data.dropna()
data = data.drop(columns=['id', 'name'], axis=1)
data = data.rename(columns={'id': 'row', 'name': 'col'})

# 进行数据分析和决策
X = data[['production_time', 'warehouse_time', 'delivery_time']]
y = data['price']

# 使用KMeans聚类分析
kmeans = KMeans(n_clusters=3, n_features=X.shape[1], n_clusters_per_class=1).fit(X)
labels = kmeans.labels_

# 使用线性回归分析
X = data[['production_time', 'warehouse_time', 'delivery_time']]
y = data['price']
X_train = np.delete(X, labels.index)
X_train = np.insert(X_train, 0, labels.index)
X_train = np.delete(X_train, -1)
X_train = np.insert(X_train, -1, 0)

X_test = data[['production_time', 'warehouse_time', 'delivery_time']]
y_test = data['price']
X_test = np.delete(X_test, labels.index)
X_test = np.insert(X_test, 0, labels.index)
X_test = np.delete(X_test, -1)
X_test = np.insert(X_test, -1, 0)

# 计算模型的指标
accuracy = np.mean(y_test == X_train)
print('Accuracy:', accuracy)

# 计算预测的销售额
forecast = np.predict(X_test)
print('Forecast:', forecast)
