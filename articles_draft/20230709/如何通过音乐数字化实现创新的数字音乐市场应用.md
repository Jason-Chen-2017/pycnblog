import numpy as np
import pandas as pd
from keras.models import Sequential
from keras.layers import Dense, Dropout
from keras.optimizers import Adam

# 加载数据集
data = pd.read_csv('music_data.csv')

# 特征提取
features = []
for col in data.columns:
    if '旋律' in col or '节奏' in col or '和弦' in col:
        features.append(col)

# 建立模型
model = Sequential()
model.add(Dense(32, input_shape=(features.shape[0],), activation='relu'))
model.add(Dropout(0.2))
model.add(Dense(16, activation='relu'))
model.add(Dropout(0.2))
model.add(Dense(1, activation='linear'))

# 编译模型
model.compile(optimizer=Adam(lr=0.001), loss='mean_squared_error')

# 训练模型
model.fit(data.values, epochs=50, batch_size=32, validation_split=0.2)

# 生成相似音乐
 similar_music = model.predict(data.values)
