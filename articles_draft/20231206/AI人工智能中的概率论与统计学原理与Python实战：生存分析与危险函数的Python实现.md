## 4.1生存分析的Python代码实例

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
n = 100
t = np.random.uniform(0, 10, n)
censored = np.random.binomial(1, 0.5, n)

# 生存曲线
plt.figure(figsize=(8, 6))
plt.plot(t[censored == 0], np.exp(-np.sum(t[censored == 0]) / 10), 'o-', label='Survival curve')
plt.plot(t[censored == 1], np.ones(n - np.sum(censored)) / 2, 'o-', label='Censored points')
plt.xlabel('Time')
plt.ylabel('Survival probability')
plt.legend()
plt.show()

# 生存率
def survival_rate(t, censored):
    return np.exp(-np.sum(t[censored == 0]) / 10)

# 生存时间
def survival_time(t, censored):
    return np.sum(t[censored == 0]) / 10

print('生存率:', survival_rate(t, censored))
print('生存时间:', survival_time(t, censored))
```

## 4.2危险函数的Python代码实例

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
n = 100
t = np.random.uniform(0, 10, n)
censored = np.random.binomial(1, 0.5, n)

# 危险函数
def hazard_function(t, censored):
    return -np.sum(t[censored == 0]) / (n - np.sum(censored))

# 绘制危险函数图
plt.figure(figsize=(8, 6))
plt.plot(t[censored == 0], hazard_function(t, censored), 'o-', label='Hazard function')
plt.xlabel('Time')
plt.ylabel('Hazard rate')
plt.legend()
plt.show()

print('危险函数:', hazard_function(t, censored))
```
