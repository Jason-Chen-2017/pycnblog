# 设计算法

clf = RandomForestClassifier(n_estimators=100, random_state=42)
clf.fit(X_train, y_train)

# 实现算法
def predict(X):
    return clf.predict(X)

# 测试算法
y_pred = predict(X_test)
print('Accuracy:', np.mean(y_pred == y_test))
