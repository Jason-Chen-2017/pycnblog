```python

# 数据连接
sc = SparkContext('local', 'DataIntegration')
data1 = sc.textFile('data1.csv').map(lambda x: (x.split(',')[0], x.split(',')[1]))
data2 = sc.textFile('data2.csv').map(lambda x: (x.split(',')[0], x.split(',')[1]))

# 数据转换
data1_transformed = data1.map(lambda x: (x[0], int(x[1])))
data2_transformed = data2.map(lambda x: (x[0], int(x[1])))

# 数据集成
integrated_data = data1_transformed.join(data2_transformed)
integrated_data.saveAsTextFile('integrated_data.csv')
```
