1.4.1 分布式系统的一致性算法实例

```python
import threading
import time

class DistributedSystem:
    def __init__(self):
        self.lock = threading.Lock()
        self.data = {}

    def get(self, key):
        with self.lock:
            return self.data.get(key)

    def set(self, key, value):
        with self.lock:
            self.data[key] = value

if __name__ == '__main__':
    system = DistributedSystem()
    threading.Thread(target=system.set, args=('key', 'value')).start()
    time.sleep(0.1)
    print(system.get('key'))
```
