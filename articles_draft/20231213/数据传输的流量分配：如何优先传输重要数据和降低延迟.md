```python
import random

# 数据包类
class DataPacket:
    def __init__(self, type, size, source):
        self.type = type
        self.size = size
        self.source = source
        self.importance = random.random()

# 重要性值的计算公式
def calculate_importance(data_packet):
    return data_packet.type * 0.3 + data_packet.size * 0.4 + data_packet.source * 0.3

# 排序算法的实现
def quick_sort(data_packets):
    if len(data_packets) <= 1:
        return data_packets
    pivot = data_packets[0]
    left = [x for x in data_packets[1:] if calculate_importance(x) < calculate_importance(pivot)]
    right = [x for x in data_packets[1:] if calculate_importance(x) >= calculate_importance(pivot)]
    return quick_sort(left) + [pivot] + quick_sort(right)

# 流量控制策略的调整
def adjust_traffic_control(data_packets):
    for data_packet in data_packets:
        if data_packet.importance > 0.5:
            # 增加发送速率
            data_packet.speed = 1.5

# 主程序
data_packets = [
    DataPacket(1, 100, 1),
    DataPacket(2, 200, 2),
    DataPacket(3, 300, 3),
    DataPacket(4, 400, 4),
    DataPacket(5, 500, 5),
]

# 排序数据包
sorted_data_packets = quick_sort(data_packets)

# 调整流量控制策略
adjust_traffic_control(sorted_data_packets)

# 输出结果
for data_packet in sorted_data_packets:
    print(data_packet.importance, data_packet.speed)
```
