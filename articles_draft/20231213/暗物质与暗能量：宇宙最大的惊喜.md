```python
import numpy as np
from astropy.cosmology import FlatLambdaCDM

# 初始化宇宙模型
cosmo = FlatLambdaCDM(H0=67.74, Om0=0.3089, h=0.6774)

# 计算红移z的函数
def z(a):
    return cosmo.angular_diameter_distance(a).value / cosmo.comoving_distance(a).value - 1

# 计算膨胀率的函数
def H(z):
    a = cosmo.comoving_distance(z).value / cosmo.comoving_distance(0).value
    return cosmo.h(a) * cosmo.H0

# 计算暗物质与暗能量的比例
def Omega(z):
    h = H(z)
    return 1 - h**2

# 测量红移z的数据
z_data = np.array([0.01, 0.03, 0.1, 0.3, 1.0])

# 计算膨胀率和暗物质与暗能量的比例
H_data = np.array([H(z) for z in z_data])
Omega_data = np.array([Omega(z) for z in z_data])

# 绘制膨胀率和暗物质与暗能量的比例的图像
import matplotlib.pyplot as plt
plt.plot(z_data, H_data, label='膨胀率')
plt.plot(z_data, Omega_data, label='暗物质与暗能量的比例')
plt.xlabel('红移z')
plt.ylabel('值')
plt.legend()
plt.show()
```
