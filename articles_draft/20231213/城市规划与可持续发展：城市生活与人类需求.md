### 4.1 代码实例

import numpy as np

# 地域规划
def domain_planning(population, economy, society):
    a = 0.5
    b = 0.5
    c = -10
    score = a * population**2 + b * population + c
    return score

# 城市规划
def city_planning(demand, environment):
    d = 0.5
    e = 0.5
    f = -10
    score = d * demand**2 + e * demand + f
    return score

# 可持续发展
def sustainable_development(economy, society, environment):
    g = 0.5
    k = 0.5
    l = -10
    score = g * economy**2 + k * economy + l
    return score

# 主函数
def main():
    # 收集数据
    population = 100000
    economy = 10000
    society = 5000
    demand = 5000
    environment = 80

    # 地域规划
    score = domain_planning(population, economy, society)
    print("地域规划得分:", score)

    # 城市规划
    score = city_planning(demand, environment)
    print("城市规划得分:", score)

    # 可持续发展
    score = sustainable_development(economy, society, environment)
    print("可持续发展得分:", score)

if __name__ == "__main__":
    main()
