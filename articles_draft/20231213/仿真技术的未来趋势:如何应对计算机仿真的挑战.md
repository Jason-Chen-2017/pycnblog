### 4.1 代码实例

```python
import numpy as np
import matplotlib.pyplot as plt

# 建立数学模型
def heat_equation(x, t, T0, k, sigma):
    return k * sigma * np.gradient(T0, x) ** 2

# 选择算法
def solve_heat_equation(x, t, T0, k, sigma):
    dt = 0.1
    dx = 0.1
    N = int(t / dt)
    M = int(x / dx)
    T = np.zeros((N + 1, M + 1))
    T[0, :] = T0

    for n in range(N):
        for m in range(M):
            T[n + 1, m] = T[n, m] + dt * heat_equation(dx * m, dt * n, T0, k, sigma)

    return T

# 准备数据
x = np.linspace(0, 1, 100)
t = np.linspace(0, 1, 100)
T0 = np.sin(np.pi * x)
k = 0.1
sigma = 1

# 执行仿真计算
T = solve_heat_equation(x, t, T0, k, sigma)

# 分析结果
plt.plot(x, T[:, 0])
plt.xlabel('x')
plt.ylabel('T(x,t)')
plt.title('Heat Equation Solution')
plt.show()
```
