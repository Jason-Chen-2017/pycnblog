    images.append(img_flip)

    # 旋转

    rotation_matrix = cv2.getRotationMatrix2D(center, 90, 1.0)
    img_rotate = cv2.warpAffine(img, rotation_matrix, (w, h), flags=cv2.INTER_CUBIC, borderMode=cv2.BORDER_REPLICATE)
