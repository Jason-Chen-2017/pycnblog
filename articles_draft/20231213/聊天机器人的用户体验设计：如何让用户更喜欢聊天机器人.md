### 4.1.3 文本生成

```python
from sklearn.metrics.pairwise import cosine_similarity

def text_generation(X, words):
    similarity = cosine_similarity(X, X)
    response = []
    for i in range(len(similarity)):
        max_similarity = max(similarity[i])
        max_index = similarity[i].tolist().index(max_similarity)
        response.append(words[max_index])
    return response
```

```python
import keras

def neural_network_training(X_train, y_train):
    model = keras.models.Sequential()
    model.add(keras.layers.Dense(128, activation='relu', input_shape=(X_train.shape[1],)))
    model.add(keras.layers.Dense(64, activation='relu'))
    model.add(keras.layers.Dense(1, activation='sigmoid'))
    model.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy'])
    model.fit(X_train, y_train, epochs=10, batch_size=32)
    return model
```
