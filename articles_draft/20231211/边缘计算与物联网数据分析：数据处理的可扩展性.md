```python
import multiprocessing
import threading

def edge_computing_task(data):
    # 对数据进行处理
    result = process_data(data)
    return result

# 创建进程池
pool = multiprocessing.Pool(processes=4)

# 创建线程池
thread_pool = threading.ThreadPoolExecutor(max_workers=4)

# 分发任务
tasks = [(data1,), (data2,), (data3,), (data4,)]
results = pool.map(edge_computing_task, tasks)

# 等待任务完成
pool.close()
pool.join()

# 执行线程任务
thread_tasks = [(data5,), (data6,), (data7,), (data8,)]
thread_results = thread_pool.map(edge_computing_task, thread_tasks)

# 等待线程任务完成
thread_pool.shutdown(wait=True)

# 合并结果
final_results = [result1 + result2 + result3 + result4 for result1, result2, result3, result4 in zip(results, thread_results)]
```
