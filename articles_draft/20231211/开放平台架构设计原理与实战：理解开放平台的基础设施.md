import json
import requests

def encrypt_data(data, key):
    iv = base64.b64encode(key.encode('utf-8'))

        json.dumps(data).encode('utf-8') ^ iv

def decrypt_data(encrypted_data, key):

    decrypted_data = json.loads(
        base64.b64decode(encrypted_data).decode('utf-8') ^ iv
