### 4.1.3代码实例

以下是计算器软件的代码实例：

```python
class Calculator:
    def __init__(self):
        self.input = ""
        self.result = 0

    def input_number(self, num):
        self.input += num

    def select_operator(self, op):
        self.operator = op

    def calculate(self):
        num1 = float(self.input[:self.input.find(self.operator)])
        num2 = float(self.input[self.input.find(self.operator) + 1:])
        if self.operator == "+":
            self.result = num1 + num2
        elif self.operator == "-":
            self.result = num1 - num2
        elif self.operator == "*":
            self.result = num1 * num2
        elif self.operator == "/":
            self.result = num1 / num2

    def output_result(self):
        print("Result: ", self.result)


calculator = Calculator()
calculator.input_number("7")
calculator.select_operator("+")
calculator.input_number("5")
calculator.calculate()
calculator.output_result()
```
