from gremlin_python.structure.graph import Graph

from gremlin_python.process.graph_traversal import __

# 创建图数据库实例

# 插入节点
g.add_vertex(label="energy", properties={"name": "electricity"})

# 查询节点

# 更新节点
g.V().has_label("energy").has("name", "electricity").value_map(["name"], "name").next().value("name", "electricity_updated")

# 删除节点
g.V().has_label("energy").has("name", "electricity_updated").drop()
