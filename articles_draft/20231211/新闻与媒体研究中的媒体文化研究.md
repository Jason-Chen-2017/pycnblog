import numpy as np
import pandas as pd

def linear_regression(X, y):
    # 线性回归
    X = np.hstack((np.ones((X.shape[0], 1)), X))
    coefficients = np.linalg.inv(X.T.dot(X)).dot(X.T).dot(y)
    return coefficients

def multiple_regression(X, y):
    # 多元回归
    X = np.hstack((np.ones((X.shape[0], 1)), X))
    coefficients = np.linalg.inv(X.T.dot(X)).dot(X.T).dot(y)
    return coefficients

def logistic_regression(X, y):
    # 逻辑回归
    X = np.hstack((np.ones((X.shape[0], 1)), X))
    coefficients = np.linalg.inv(X.T.dot(X)).dot(X.T).dot(y)
    return coefficients

    # 随机森林
