```python

import pandas as pd
from sklearn.linear_model import LinearRegression

# 加载数据

# 数据预处理
X = data[['age', 'lifestyle', 'diet', 'environment', 'individual_difference']]
y = data['need']
X = X.fillna(0)
y = y.fillna(0)

# 训练模型
model = LinearRegression()
model.fit(X, y)

# 预测
pred = model.predict(X)
```
