### 4.1 文本预处理

```python
import re
import nltk
from nltk.corpus import stopwords
from nltk.stem import PorterStemmer

# 文本数据
text_data = ["我非常喜欢这个电影！", "这部电影真的很烂！"]

# 去除停用词
stop_words = set(stopwords.words("english"))
text_data_clean = [word for word in text_data if word.lower() not in stop_words]

# 词干提取
stemmer = PorterStemmer()
text_data_stem = [stemmer.stem(word) for word in text_data_clean]

# 词汇扩展
word_list = set(text_data_stem)
word_list.update(nltk.corpus.wordnet.all_synsets())

# 文本预处理完成
text_data_processed = text_data_stem
```

### 4.2 词嵌入

```python
from gensim.models import Word2Vec

# 训练词嵌入模型
model = Word2Vec(text_data_processed, min_count=1, size=100, window=5, workers=4)

# 获取词向量
word_vectors = model[text_data_processed]

# 词嵌入完成
word_vectors = word_vectors.tolist()
```

### 4.3 情感分析

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score

# 训练数据
train_data = ["我非常喜欢这个电影！", "这部电影真的很烂！"]

# 情感标签
labels = [1, 0]

# 特征提取
vectorizer = TfidfVectorizer(vocabulary=word_vectors)
X = vectorizer.fit_transform(train_data)

# 模型训练
X_train, X_test, y_train, y_test = train_test_split(X, labels, test_size=0.2, random_state=42)
classifier = LogisticRegression()
classifier.fit(X_train, y_train)

# 预测
y_pred = classifier.predict(X_test)

# 情感分析完成
accuracy = accuracy_score(y_test, y_pred)
print("Accuracy:", accuracy)
```
