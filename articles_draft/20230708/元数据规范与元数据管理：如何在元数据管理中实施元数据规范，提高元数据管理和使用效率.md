## 4.2. 应用实例分析

```python
import sqlite3
from xml.dom import minidom
import json

class User:
    def __init__(self, user_id):
        self.user_id = user_id
        self.first_name = ""
        self.last_name = ""
        self.email = ""
        self.address = ""
        self.phone = ""

# Connect to the database
conn = sqlite3.connect("users.db")

# Create the table to store user data
conn.execute("CREATE TABLE IF NOT EXISTS users (user_id INTEGER PRIMARY KEY AUTOINCREMENT, first_name TEXT, last_name TEXT, email TEXT, address TEXT, phone TEXT)")

# Insert user data
user_id = 1
conn.execute("INSERT INTO users (user_id, first_name, last_name, email, address, phone) VALUES (?,?,?,?,?)", (user_id, "John", "Doe", "johndoe@example.com", "123 Main St.", "555-555-5555"))

# Retrieve user data
user_data = conn.execute("SELECT * FROM users WHERE user_id =?", (user_id,))
for row in user_data:
    print("User ID:", row[0])
    print("First Name:", row[1])
    print("Last Name:", row[2])
    print("Email:", row[3])
    print("Address:", row[4])
    print("Phone:", row[5])

# Close the database connection
conn.close()
```
