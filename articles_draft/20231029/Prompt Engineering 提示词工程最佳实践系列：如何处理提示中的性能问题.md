from pyro import distos

class MCModel(distos.BaseModel):
    def __init__(self, vocab_size):
        super().__init__()
        self.vocab_size = vocab_size
    
    def logprob(self, x):
        q = torch.zeros((x.shape[0], self.vocab_size))
        for i in range(x.shape[1] - 1):
            c = torch.LongTensor([i])
            v = torch.randint(0, self.vocab_size, (1,))
            q[0][v] += 1
            q[i+1][v] += torch.log(torch.exp(self.parameters['alpha']) + 1e-9)
