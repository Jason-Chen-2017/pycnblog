import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# 数据处理
def preprocess_data(data):
    data = data.dropna()
    data = data.fillna(method='ffill')
    return data

# 人流量预测模型
def predict_traffic(data):
    X = data[['time']]
    y = data['traffic']
    model = LinearRegression()
    model.fit(X, y)
    return model

# 等待时间预测模型
def predict_waiting_time(model, data):
    X = data[['time']]
    y = data['waiting_time']
    y_pred = model.predict(X)
    mse = mean_squared_error(y, y_pred)
    return y_pred, mse

# 消费推荐模型
def recommend_consumption(data):
    user_preference = data['user_preference']
    product_sales = data['product_sales']
    recommendation = user_preference.multiply(product_sales).sum(axis=1) / (user_preference.pow(2).sum(axis=1).multiply(product_sales.pow(2).sum(axis=1)).pow(0.5))
    return recommendation

# 主程序
if __name__ == '__main__':
    # 数据处理
    data = pd.read_csv('data.csv')
    data = preprocess_data(data)

    # 人流量预测模型
    model = predict_traffic(data)

    # 等待时间预测模型
    y_pred, mse = predict_waiting_time(model, data)

    # 消费推荐模型
    recommendation = recommend_consumption(data)

    # 输出结果
    print('人流量预测模型:', model)
    print('等待时间预测模型:', y_pred, mse)
    print('消费推荐模型:', recommendation)
