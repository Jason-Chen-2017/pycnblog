    [10, 20, 30, 40],
    [20, 10, 40, 30],
    [30, 40, 10, 20],
    [40, 30, 20, 10]
])

trade_volume = np.array([
    [50, 60, 70, 80],
    [60, 50, 80, 70],
    [70, 80, 50, 60],
    [80, 70, 60, 50]
])

# 初始化文明的战争力量

    [10, 20, 30, 40],
    [20, 10, 40, 30],
    [30, 40, 10, 20],
    [40, 30, 20, 10]
])

# 计算文明的竞赛结果

# 输出文明的竞赛结果
