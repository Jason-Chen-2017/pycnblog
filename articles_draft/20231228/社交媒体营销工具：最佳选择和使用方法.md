```python
import numpy as np

def cosine_similarity(vec1, vec2):
    dot_product = np.dot(vec1, vec2)
    norm1 = np.linalg.norm(vec1)
    norm2 = np.linalg.norm(vec2)
    return dot_product / (norm1 * norm2)

def recommend(user_vector, content_vectors):
    similarities = []
    for content_vector in content_vectors:
        similarity = cosine_similarity(user_vector, content_vector)
        similarities.append(similarity)
    recommended_content_indices = np.argsort(similarities)[::-1]
    return recommended_content_indices
```
