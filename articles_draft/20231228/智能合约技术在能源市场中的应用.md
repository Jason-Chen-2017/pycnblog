```solidity

contract EnergyContract {
    uint public price;
    uint public volume;

    event PriceUpdated(uint newPrice);
    event VolumeUpdated(uint newVolume);

    function setPrice(uint _price) public {
        require(_price > 0);
        price = _price;
        emit PriceUpdated(_price);
    }

    function setVolume(uint _volume) public {
        require(_volume > 0);
        volume = _volume;
        emit VolumeUpdated(_volume);
    }
}
```
