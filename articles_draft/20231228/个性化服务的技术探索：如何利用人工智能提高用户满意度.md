```python
from scipy.spatial.distance import cosine

def similarity(u, v):
    sim = 0
    for i in u:
        for j in v:
            if i == j:
                sim += 1
    return cosine(u, v)

u = [1, 2, 3]
v = [2, 3, 4]
print(similarity(u, v))
```

```python
import networkx as nx

def social_recommendation(G, u, k):

G.add_edge(1, 3, weight=2)
```
