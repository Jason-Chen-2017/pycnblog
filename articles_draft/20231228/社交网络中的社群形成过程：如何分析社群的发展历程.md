    followers1 = set(G.neighbors(user1))
    followers2 = set(G.neighbors(user2))
    return len(followers1.intersection(followers2))

# 计算用户之间的共同关注相似度
def similarity(user1, user2, G):
    followers1 = set(G.neighbors(user1))
    followers2 = set(G.neighbors(user2))
    return cosine_similarity([followers1], [followers2])[0][0]

# 检测社群
def detect_community(G, threshold):
    communities = []
    for user in G.nodes():
        community = []
        for other_user in G.neighbors(user):
            similarity = similarity(user, other_user, G)
            if similarity >= threshold:
                community.append(other_user)
        communities.append(community)
    return communities

# 检测社群

# 打印社群
