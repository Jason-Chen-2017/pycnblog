import numpy as np

def photoelectric_conversion(light_intensity, sensitivity):
    """
    A function to simulate the photoelectric conversion process.

    Parameters
    ----------
    light_intensity : float
        The intensity of the incident light.
    sensitivity : float
        The sensitivity of the photodetector.

    Returns
    -------
    electrical_signal : float
        The output electrical signal.
    """
    electrical_signal = sensitivity * light_intensity
    return electrical_signal

def filtering(electrical_signal, filter_resistance, filter_capacitance, frequency):
    """
    A function to simulate the filtering process.

    Parameters
    ----------
    electrical_signal : float
        The input electrical signal.
    filter_resistance : float
        The resistance of the filter.
    filter_capacitance : float
        The capacitance of the filter.
    frequency : float
        The frequency of the signal.

    Returns
    -------
    filtered_signal : float
        The filtered electrical signal.
    """
    denominator = 1 + electrical_signal / (filter_resistance * filter_capacitance * frequency)
    filtered_signal = electrical_signal / denominator
    return filtered_signal

light_intensity = 1000
sensitivity = 0.1

print("Output electrical signal: {:.2f}".format(electrical_signal))
