```python
class RuleEngine:
    def __init__(self):
        self.rules = []
        self.rule_monitor = RuleMonitor()

    def add_rule(self, rule):
        self.rules.append(rule)

    def execute_rules(self, facts):
        for rule in self.rules:
            if rule.condition(facts):
                self.rule_monitor.start_monitor(rule)
                rule.action(facts)
                self.rule_monitor.stop_monitor(rule)

class RuleMonitor:
    def __init__(self):
        self.monitors = {}

    def start_monitor(self, rule):
        if rule.name not in self.monitors:
            self.monitors[rule.name] = Monitor(rule.name)

    def stop_monitor(self, rule):
        monitor = self.monitors[rule.name]
        monitor.stop()

class Monitor:
    def __init__(self, name):
        self.name = name
        self.counter = 0
        self.start_time = None
        self.end_time = None

    def start(self):
        self.start_time = time.time()

    def stop(self):
        self.end_time = time.time()
        self.counter += 1

```
