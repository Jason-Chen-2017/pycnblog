def calculate_gravity(mass1, mass2, distance):
    G = 6.67430e-11
    return G * (mass1 * mass2) / (distance ** 2)

def calculate_force(gravity, mass):
    return gravity * mass

def calculate_acceleration(force, mass):
    return force / mass

def calculate_time(distance, acceleration):
    return distance / acceleration

# 示例使用

mass2 = 5.972e24
distance = 1.496e11

time = calculate_time(distance, acceleration)

print("引力场对目标天体的作用:", force)

print("到达目的地所需的时间:", time)
