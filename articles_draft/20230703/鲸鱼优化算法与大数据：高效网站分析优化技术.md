import numpy as np
import tensorflow as tf
from tensorflow.keras.layers import Input, Dense, GlobalAveragePooling2D

from tensorflow.keras.preprocessing.text import Tokenizer
from tensorflow.keras.preprocessing.sequence import pad_sequences
from sklearn.model_selection import train_test_split

# 数据预处理
def preprocess_data(sequences):
    # 将文本数据转换为数组
    data = pad_sequences(sequences, maxlen=100)
    # 添加标签
    data = np.concatenate([data, [1]])
    # 将数据输入到神经网络中
    return data

# 生成训练数据集和测试数据集
train_data, test_data = train_test_split(data, test_size=0.2, shuffle=0)

# 准备数据
train_sequences = train_data[0]
test_sequences = test_data[0]

# 定义模型
input_layer = Input(shape=(100,))
x = GlobalAveragePooling2D()(input_layer)
model = Model(inputs=input_layer, outputs=x)

# 定义优化器
optimizer = Adam(0.001)

# 定义损失函数
loss = tf.reduce_mean(tf.nn.sparse_softmax_cross_entropy_with_logits(labels=test_sequences, logits=x))

# 训练模型
model.compile(optimizer=optimizer, loss=loss, metrics=['accuracy'])

# 测试模型
