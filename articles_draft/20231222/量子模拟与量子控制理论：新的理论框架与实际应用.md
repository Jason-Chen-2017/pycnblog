import numpy as np

def quantum_trajectory_algorithm(H, rho, t):
    # 计算H的矩阵表示
    H_matrix = np.array([[0, 1], [1, 0]])
    # 计算H的幂
    H_power = np.linalg.matrix_power(H_matrix, t)
