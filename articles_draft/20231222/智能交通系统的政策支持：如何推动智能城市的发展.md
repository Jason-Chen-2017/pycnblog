import numpy as np

# 车辆位置收集
def collect_vehicle_position(vehicles, time):
    positions = []
    for vehicle in vehicles:
        x = vehicle['x'] + vehicle['v'] * time * np.cos(vehicle['theta'])
        y = vehicle['y'] + vehicle['v'] * time * np.sin(vehicle['theta'])
        positions.append((x, y))
    return positions

# 交通流量预测
def predict_traffic_flow(traffic_data, time_step):
    mu = np.mean(traffic_data)
    sigma = np.std(traffic_data)
    predicted_traffic = (1 / (sigma * np.sqrt(2 * np.pi))) * np.exp(-((traffic_data - mu) ** 2) / (2 * sigma ** 2))
    return predicted_traffic

# PID控制
def pid_control(error, Kp, Ki, Kd):
    dt = 1
    u = Kp * error + Ki * np.integrate.integrate(error, 0, dt) + Kd * (error - np.diff(error, n=2) / 2)
    return u

# 主程序
if __name__ == '__main__':
    vehicles = [{'x': 0, 'y': 0, 'v': 10, 'theta': np.pi / 4}]
    time = 10

    positions = collect_vehicle_position(vehicles, time)
    print('Vehicle positions:', positions)

    traffic_data = [100, 150, 200, 250, 300, 350, 400, 450, 500, 550, 600]
    predicted_traffic = predict_traffic_flow(traffic_data, time)
    print('Predicted traffic flow:', predicted_traffic)

    error = 10

    Kd = 1
    control_output = pid_control(error, Kp, Ki, Kd)
