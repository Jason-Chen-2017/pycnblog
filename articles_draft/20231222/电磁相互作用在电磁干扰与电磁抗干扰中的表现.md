1. 电磁干扰信号生成：

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成电磁干扰信号
def generate_emi_signal(amplitude, frequency, duration, sample_rate):
    t = np.linspace(0, duration, int(sample_rate * duration), endpoint=False)
    signal = amplitude * np.cos(2 * np.pi * frequency * t)
    return signal

# 绘制电磁干扰信号
def plot_emi_signal(signal, sample_rate):
    plt.plot(signal)
    plt.xlabel('Time (s)')
    plt.ylabel('Amplitude')
    plt.title('Electromagnetic Interference Signal')
    plt.show()

# 参数设置
amplitude = 1
frequency = 50
duration = 0.01
sample_rate = 1000

# 生成电磁干扰信号
emi_signal = generate_emi_signal(amplitude, frequency, duration, sample_rate)

# 绘制电磁干扰信号
plot_emi_signal(emi_signal, sample_rate)
```
