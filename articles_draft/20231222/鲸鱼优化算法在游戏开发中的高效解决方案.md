class WhaleOptimization:

        self.puzzle_size = puzzle_size
        self.whales = [np.random.rand(puzzle_size) for _ in range(num_whales)]
        self.best_whale = self.whales[0]
        self.best_score = np.sum(self.best_whale)

    def evaluate(self, whale):
        score = np.sum(whale)
        if score > self.best_score:
            self.best_score = score
            self.best_whale = whale

    def mutate(self, whale):
        mutation_rate = 0.1
        for i in range(self.puzzle_size):
            if np.random.rand() < mutation_rate:
                whale[i] += np.random.rand() - 0.5

    def optimize(self, iterations):
        for _ in range(iterations):
            for i in range(self.num_whales):
                self.evaluate(self.whales[i])
                self.mutate(self.whales[i])
