```python
import numpy as np

# 定义一个向量空间V
V = np.array([[1, 2], [3, 4]])

# 定义一个线性映射T：V→W
def T(v):
    return np.dot(v, [1, -1])

# 定义一个对偶空间V*
V_star = []
for v in V:
    T_v = T(v)
    V_star.append(T_v)

# 定义一个双对偶空间V**
V_double_star = []
for v_star in V_star:
    def T_v_star(v):
        return np.dot(v, v_star)
    V_double_star.append(T_v_star)

# 测试
v = np.array([[1, 2], [3, 4]])
w = T(v)
w_star = V_star[0](v)
w_double_star = V_double_star[0](v)

print("V:", V)
print("T(v):", w)
print("V*:", V_star)
print("V**:", V_double_star)
print("v:", v)
print("T(v):", w)
print("T(v)*(v):", w_star)
print("T(v)**(v):", w_double_star)
```
