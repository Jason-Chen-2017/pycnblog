```python
from rdflib.namespace import RDF, RDFS

# 构建知识图谱
g = Graph()

# 添加实体
g.add((RDF.type, RDFS.Class, "Person"))
g.add((RDF.type, RDFS.Class, "Organization"))
g.add((RDF.type, RDFS.Class, "Location"))
g.add((RDF.type, RDFS.Class, "Product"))

# 添加关系
g.add((RDF.type, RDFS.Property, "worksFor"))
g.add((RDF.type, RDFS.Property, "locatedIn"))
g.add((RDF.type, RDFS.Property, "produces"))

# 添加实例
g.add((RDF.type, "Person", "Alice"))
g.add((RDF.type, "Organization", "Google"))
g.add((RDF.type, "Location", "Mountain View"))
g.add((RDF.type, "Product", "Android"))

# 添加实例关系
g.add((RDF.type, "Alice", "worksFor", "Google"))
g.add((RDF.type, "Google", "locatedIn", "Mountain View"))
g.add((RDF.type, "Google", "produces", "Android"))
```
