```python
import numpy as np
import matplotlib.pyplot as plt

def filter_signal(signal, filter_function):
    filtered_signal = np.convolve(signal, filter_function)
    return filtered_signal

signal = np.sin(np.linspace(0, 2 * np.pi, 100))
filter_function = np.ones(5) / 5
filtered_signal = filter_signal(signal, filter_function)

plt.plot(signal, label='Original Signal')
plt.plot(filtered_signal, label='Filtered Signal')
plt.legend()
plt.show()
```
