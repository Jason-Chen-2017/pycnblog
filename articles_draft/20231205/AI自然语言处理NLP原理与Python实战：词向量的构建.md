```python
import numpy as np
import gensim
from gensim.models import Word2Vec

# 加载语料库
text = [
    "I love you.",
    "You love me.",
    "He loves her.",
    "She loves him.",
    "We love you.",
    "You love us.",
    "They love each other."
]

# 构建词向量矩阵
model = Word2Vec(text, min_count=1, size=100, window=5, workers=4)

# 计算词向量相似性
similarity = model.wv.most_similar("love")
print(similarity)
```
