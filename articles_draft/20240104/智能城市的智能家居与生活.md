        self.light.on()

    def turn_off_light(self):
        self.light.off()

    def set_temperature(self, temp):
        self.air_conditioner.set_temperature(temp)

    def start_air_conditioner(self):
        self.air_conditioner.start()

    def stop_air_conditioner(self):
        self.air_conditioner.stop()

    smart_home.open_door()
    smart_home.turn_on_light()
    smart_home.set_temperature(25)
    smart_home.start_air_conditioner()
    time.sleep(60)
    smart_home.stop_air_conditioner()
    smart_home.turn_off_light()
    smart_home.close_door()
