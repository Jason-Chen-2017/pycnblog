## 4.1 线性回归

线性回归是一种通过计算机程序预测连续型变量的方法。以下是一个Python代码实例：

```python
import numpy as np

# 生成数据
np.random.seed(0)
X = np.random.rand(100, 1)
y = 2 * X + 1 + np.random.randn(100, 1) * 0.5

# 定义损失函数
def squared_loss(y_true, y_pred):
    return np.mean((y_true - y_pred) ** 2)

# 定义梯度下降算法
def gradient_descent(X, y, learning_rate, iterations):
    m, n = X.shape
    w = np.zeros(n)
    for _ in range(iterations):
        y_pred = X.dot(w)
        loss = squared_loss(y, y_pred)
        gradient = X.T.dot(y_pred - y) / m
        w -= learning_rate * gradient
    return w

# 训练模型
X = np.column_stack((np.ones(100), X))
w = gradient_descent(X, y, learning_rate=0.01, iterations=1000)

# 预测
X_new = np.array([[0], [1]])
X_new = np.column_stack((np.ones(100), X_new))
y_pred = X_new.dot(w)
print(y_pred)
```

## 4.2 逻辑回归

逻辑回归是一种通过计算机程序预测分类型变量的方法。以下是一个Python代码实例：

```python
import numpy as np

# 生成数据
np.random.seed(0)
X = np.random.rand(100, 1)
y = (X > 0.5).astype(int)

# 定义损失函数
def binary_cross_entropy_loss(y_true, y_pred):
    return -(y_true * np.log(y_pred) + (1 - y_true) * np.log(1 - y_pred)) / len(y_true)

# 定义梯度下降算法
def gradient_descent(X, y, learning_rate, iterations):
    m, n = X.shape
    w = np.zeros(n)
    for _ in range(iterations):
        y_pred = 1 / (1 + np.exp(-X.dot(w)))
        loss = binary_cross_entropy_loss(y, y_pred)
        gradient = -X.T.dot(y_pred - y) / m
        w -= learning_rate * gradient
    return w

# 训练模型
X = np.column_stack((np.ones(100), X))
w = gradient_descent(X, y, learning_rate=0.01, iterations=1000)

# 预测
X_new = np.array([[0], [1]])
X_new = np.column_stack((np.ones(100), X_new))
y_pred = 1 / (1 + np.exp(-X_new.dot(w)))
print(y_pred)
```
