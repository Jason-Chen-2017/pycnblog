### 4.1 模糊集的实现

```python
import numpy as np
from scipy.interpolate import interp1d

class FuzzySet:
    def __init__(self, x, membership_function):
        self.x = x
        self.membership_function = membership_function
        self.interp = interp1d(x, self.membership_function, kind='linear', bounds_error=False, fill_value='extrapolate')

    def __call__(self, x):
        return self.interp(x)
```

### 4.2 模糊决策规则的实现

```python
class FuzzyRule:
    def __init__(self, input_fuzzy_sets, output_fuzzy_set, rule):
        self.input_fuzzy_sets = input_fuzzy_sets
        self.output_fuzzy_set = output_fuzzy_set
        self.rule = rule

    def __call__(self, input_values):
        output_values = []
        for input_value in input_values:
            output_value = self.rule(self.input_fuzzy_sets, input_value)
            output_values.append(output_value)
        return output_values
```

### 4.3 模糊决策分析的实现

```python
class FuzzyDecisionAnalysis:
    def __init__(self, fuzzy_rules, fuzzy_sets):
        self.fuzzy_rules = fuzzy_rules
        self.fuzzy_sets = fuzzy_sets

    def defuzzify(self, output_fuzzy_sets, method='centroid'):
        if method == 'centroid':
            return self._defuzzify_centroid(output_fuzzy_sets)
        else:
            raise NotImplementedError()

    def _defuzzify_centroid(self, output_fuzzy_sets):
        x = np.linspace(min(output_fuzzy_sets.x), max(output_fuzzy_sets.x), 100)
        y = []
        for output_fuzzy_set in output_fuzzy_sets.x:
            y.append(output_fuzzy_set(x))
        y = np.array(y).T
        centroid = np.sum(y * x, axis=1) / np.sum(y, axis=1)
        return centroid
```
