```python
import numpy as np

# 定义输入数据

# 定义输出数据
Y = np.array([[0], [1], [1], [0]])

# 初始化权重和偏置
np.random.seed(1)

bias = np.random.randn(1, 2)

# 设置学习率
learning_rate = 0.01

# 训练神经网络
for iteration in range(10000):
    # 前向传播
    XP = np.dot(X, weights) + bias
    ZP = 1 / (1 + np.exp(-XP))
    
    # 后向传播
    error = Y - ZP
    dZP = ZP * (1 - ZP)
    dweights = np.dot(X.T, dZP)
    dbias = np.sum(dZP, axis=0, keepdims=True)
    
    # 更新权重和偏置
    weights += learning_rate * dweights
    bias += learning_rate * dbias

# 输出预测结果
print(ZP)
```
