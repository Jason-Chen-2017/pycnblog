import numpy as np

    transfer_time = time.time()
    return data, transfer_time

    analysis_result = np.mean(data)
    return analysis_result

# 实时同步

    sync_time = time.time()
    return sync_time, analysis_result

    data = collect_data()
    data, transfer_time = transfer_data(data)
    analysis_result = analyze_data(data)
    sync_time, analysis_result = sync(data, analysis_result)
    print('数据收集时间：', transfer_time)

    print('同步时间：', sync_time)
