```python
import numpy as np

class Neuron:
    def __init__(self):
        self.weight = np.random.rand(1)

    def activate(self, x):
        return 1 / (1 + np.exp(-x * self.weight))

class NeuralNetwork:
    def __init__(self, input_size, output_size):
        self.input_size = input_size
        self.output_size = output_size
        self.neurons = [Neuron() for _ in range(input_size)]

    def train(self, input_data, target_output):
        input_data = np.array(input_data)
        target_output = np.array(target_output)

        for input_vector in input_data:
            output_vector = np.zeros(self.output_size)
            for neuron in self.neurons:
                output_vector += neuron.activate(np.dot(input_vector, neuron.weight))

            error = target_output - output_vector
            for neuron in self.neurons:
                neuron.weight += error * input_vector

    def predict(self, input_data):
        input_data = np.array(input_data)
        output_vector = np.zeros(self.output_size)
        for neuron in self.neurons:
            output_vector += neuron.activate(np.dot(input_data, neuron.weight))
        return output_vector
```
