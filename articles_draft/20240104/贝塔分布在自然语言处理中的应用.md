# 2.核心概念与联系

## 2.1 贝塔分布的定义

贝塔分布是一种连续概率分布，用于描述随机变量的不确定性。它的概率密度函数（PDF）定义为：

$$
f(x) = \frac{\Gamma(\alpha + \beta)}{\Gamma(\alpha) \Gamma(\beta)} \cdot \frac{x^{\alpha - 1} \cdot (1 - x)^{\beta - 1}}{\beta^\alpha \cdot (1 - \beta)^\beta}
$$

其中，$\alpha$ 和 $\beta$ 是贝塔分布的参数，$\Gamma$ 是伽马函数。

$$

import numpy as np
from scipy.special import gamma, loggamma

def beta_pdf(x, alpha, beta):
    return (gamma(alpha + beta) / (gamma(alpha) * gamma(beta))) * (x**(alpha - 1) * (1 - x)**(beta - 1))

def estimate_alpha(x, n):
    return sum(x[i] * np.log(x[i]) for i in range(n))

def estimate_beta(x, n):
    return sum((1 - x[i]) * np.log(1 - x[i]) for i in range(n))

def sentiment_analysis(text):
    words = text.split()
    word_sentiments = {'happy': [0, 1], 'sad': [1, 0]}
    alpha = estimate_alpha(word_sentiments.values(), len(word_sentiments))
    beta = estimate_beta(word_sentiments.values(), len(word_sentiments))
    for word in words:
        if word in word_sentiments:
            sentiment_prob = beta_pdf(word_sentiments[word][0], alpha, beta) + beta_pdf(word_sentiments[word][1], alpha, beta)
            print(f"{word}: {sentiment_prob}")
        else:
            print(f"{word}: unknown")

text = "I am happy today but sad yesterday"
sentiment_analysis(text)
