### 4.2 决策树

以下是一个简单的决策树示例，通过构建决策树，表示不同选项的可能结果：

```python
class DecisionTreeNode:
    def __init__(self, value):
        self.value = value
        self.children = []

    def add_child(self, child):
        self.children.append(child)

    def make_decision(self, data):
        if len(self.children) == 0:
            return data
        else:
            for child in self.children:
                result = child.make_decision(data)
                if result is not None:
                    return result

# 构建决策树
root = DecisionTreeNode("是否购买保险")
child1 = DecisionTreeNode("购买保险")
child2 = DecisionTreeNode("不购买保险")

root.add_child(child1)
root.add_child(child2)

# 使用决策树进行决策
data = {"年龄": 25, "收入": 60000, "是否购买保险": None}
result = root.make_decision(data)
print(result)
```
