```python
import heapq

def huffman_encoding(data):
    # 统计数据中每个符号的出现次数
    symbol_count = {}
    for symbol in data:
        if symbol not in symbol_count:
            symbol_count[symbol] = 0
        symbol_count[symbol] += 1

    # 创建优先级队列，存储所有符号及其出现次数
    priority_queue = [[weight, [symbol, ""]] for symbol, weight in symbol_count.items()]
    heapq.heapify(priority_queue)

    # 创建哈夫曼树
    while len(priority_queue) > 1:
        left = heapq.heappop(priority_queue)
        right = heapq.heappop(priority_queue)
        for pair in left[1:]:
            pair[1] = '0' + pair[1]
        for pair in right[1:]:
            pair[1] = '1' + pair[1]
        heapq.heappush(priority_queue, [left[0] + right[0]] + left[1:] + right[1:])

    # 获取哈夫曼树的根节点
    root = priority_queue[0][1:]

    # 根据哈夫曼树生成编码表
    huffman_code = {symbol: code for symbol, code in root}

    # 对数据进行哈夫曼编码
    encoded_data = ""
    for symbol in data:
        encoded_data += huffman_code[symbol]

    return huffman_code, encoded_data
```

```python
def lzw_encoding(data):
    # 创建字典，存储已经编码过的数据
    dictionary = {chr(i): i for i in range(256)}
    next_index = 256

    # 对数据进行编码
    encoded_data = []
    for symbol in data:
        if symbol not in dictionary:
            dictionary[chr(next_index)] = next_index
            encoded_data.append(chr(next_index))
            next_index += 1
        encoded_data.append(dictionary[symbol])

    return bytes(encoded_data)
```
