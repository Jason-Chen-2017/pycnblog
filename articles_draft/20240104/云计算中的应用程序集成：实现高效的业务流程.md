## 4.1 服务组合

我们可以使用Python的Flask框架来实现一个简单的服务组合示例。首先，我们需要定义两个独立的服务：

```python
from flask import Flask, request, jsonify

app = Flask(__name__)

@app.route('/service1', methods=['GET'])
def service1():
    data = request.args.get('data')
    result = {'result': 'data from service1: ' + data}
    return jsonify(result)

@app.route('/service2', methods=['GET'])
def service2():
    data = request.args.get('data')
    result = {'result': 'data from service2: ' + data}
    return jsonify(result)
```

然后，我们可以使用Python的requests库来组合这两个服务：

```python
import requests

def service_combination():
    response1 = requests.get('http://localhost:5000/service1', params={'data': 'hello'})
    response2 = requests.get('http://localhost:5000/service2', params={'data': 'world'})
    result = {'result': 'combined result: ' + response1.json()['result'] + ' ' + response2.json()['result']}
    return result
```

```python
import pandas as pd

data1 = {'name': ['Alice', 'Bob', 'Charlie'], 'age': [25, 30, 35]}
data2 = {'name': ['Alice', 'Bob', 'Charlie'], 'gender': ['F', 'M', 'M']}
```

```python
def data_conversion(df1, df2):
    df = pd.merge(df1, df2, on='name')
    result = {'name': df['name'], 'age': df['age'], 'gender': df['gender']}
    return result
```

```python
from workflow import Workflow

wf = Workflow()

@wf.function(name='service_combination')
def service_combination():
    response1 = requests.get('http://localhost:5000/service1', params={'data': 'hello'})
    response2 = requests.get('http://localhost:5000/service2', params={'data': 'world'})
    result = {'result': 'combined result: ' + response1.json()['result'] + ' ' + response2.json()['result']}
    return result

@wf.function(name='data_conversion')
def data_conversion(df1, df2):
    df = pd.merge(df1, df2, on='name')
    result = {'name': df['name'], 'age': df['age'], 'gender': df['gender']}
    return result

@wf.function(name='business_flow')
def business_flow(result1, result2):
    return result1['result'] + ' ' + result2['result']

wf.run(input={'service_combination': {'data': 'hello'}, 'data_conversion': {'df1': df1, 'df2': df2}}, output=['business_flow'])
```
