```python
import cv2
import numpy as np

def preprocess(image):
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    return blur

def extract_features(image):
    gradient = cv2.Sobel(image, cv2.CV_64F, 1, 0, ksize=5)
    return gradient

def match_features(query_features, gallery_features):
    similarity = np.dot(query_features, gallery_features.T) / (np.linalg.norm(query_features) * np.linalg.norm(gallery_features))
    return similarity
```

```python
import cv2
import dlib

detector = dlib.get_frontal_face_detector()
predictor = dlib.shape_predictor("shape_predictor_68_face_landmarks.dat")

def detect_faces(image):
    faces = detector(image, 1)
    return faces

def extract_features(image, faces):
    landmarks = [predictor(image, i) for i in faces]
    features = []
    for landmark in landmarks:
        x = [landmark.part(i).x for i in range(18, 68)]
        y = [landmark.part(i).y for i in range(18, 68)]
        features.append(np.array([x, y]))
    return np.array(features)

def match_features(query_features, gallery_features):

    return similarity
```
