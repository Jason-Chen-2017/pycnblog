## 4.2 代码实例

### 4.2.1 边缘节点代码

```python
import time
import requests

class EdgeNode:
    def __init__(self, sensor_id, data_url):
        self.sensor_id = sensor_id
        self.data_url = data_url

    def collect_data(self):
        data = requests.get(self.data_url).json()
        return data

    def send_data(self, edge_computing_platform):
        data = self.collect_data()
        edge_computing_platform.receive_data(self.sensor_id, data)

    def run(self):
        while True:
            self.send_data(edge_computing_platform)
            time.sleep(1)
```

### 4.2.2 边缘计算平台代码

```python
from threading import Thread

class EdgeComputingPlatform:
    def __init__(self, edge_nodes):
        self.edge_nodes = edge_nodes
        self.data_cache = {}

    def receive_data(self, sensor_id, data):
        if sensor_id not in self.data_cache:
            self.data_cache[sensor_id] = []
        self.data_cache[sensor_id].append(data)

    def process_data(self):
        for sensor_id, data_list in self.data_cache.items():
            avg_data = sum(data_list) / len(data_list)
            print(f"sensor_id: {sensor_id}, avg_data: {avg_data}")
            self.data_cache[sensor_id] = []

    def run(self):
        threads = []
        for edge_node in self.edge_nodes:
            thread = Thread(target=edge_node.run)
            threads.append(thread)
            thread.start()

        while True:
            self.process_data()
            time.sleep(1)
```
