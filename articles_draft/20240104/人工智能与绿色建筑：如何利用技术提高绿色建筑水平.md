```python
import numpy as np

def temperature_control(temperature, target_temperature):
    error = temperature - target_temperature
    if error > 1:
        control_action = 1
    elif error < -1:
        control_action = -1
    else:
        control_action = 0
    return control_action

temperature = np.random.randint(-30, 30)
target_temperature = 20
control_action = temperature_control(temperature, target_temperature)
print(f"控制动作：{control_action}")
```

## 4.2优化算法代码实例

我们通过一个简单的遗传算法代码实例来说明人工智能在绿色建筑中的应用。

```python
import numpy as np

def fitness(x):
    return -(x[0]**2 + x[1]**2)

def crossover(parent1, parent2):
    crossover_point = np.random.randint(1, len(parent1))
    child1 = np.concatenate((parent1[:crossover_point], parent2[crossover_point:]))
    child2 = np.concatenate((parent2[:crossover_point], parent1[crossover_point:]))
    return child1, child2

def mutation(x, mutation_rate):
    if np.random.rand() < mutation_rate:
        index = np.random.randint(0, len(x))
        x[index] = np.random.rand()
    return x

population_size = 10
generations = 100
mutation_rate = 0.1

population = np.random.rand(population_size, 2)
fitness_values = np.array([fitness(x) for x in population])

for generation in range(generations):
    sorted_indices = np.argsort(fitness_values)
    parents = population[sorted_indices[:2]]
    offspring = []
    for i in range(0, population_size, 2):
        parent1, parent2 = parents[i], parents[i+1]
        child1, child2 = crossover(parent1, parent2)
        child1 = mutation(child1, mutation_rate)
        child2 = mutation(child2, mutation_rate)
        offspring.extend([child1, child2])
    population = np.vstack(offspring)
    fitness_values = np.array([fitness(x) for x in population])

print(f"最佳解：{population[np.argmax(fitness_values)]}")
```

在这个代码实例中，我们定义了一个简单的遗传算法，用于实现绿色建筑的节能设计。通过遗传算法，我们可以找到一个最佳的节能解，从而帮助绿色建筑实现更高的节能效率。
