```python

def get_load():
    # 模拟获取系统负载情况
    return 0.5

def adjust_resource(load):
    # 根据负载情况自动调整资源分配
    if load > 0.8:
        return 100
    else:
        return 50

    load = get_load()

    print(f'负载情况：{load}, 资源分配：{resource}')
```
