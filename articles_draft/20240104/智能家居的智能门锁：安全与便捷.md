## 1.2 智能门锁的发展历程

智能门锁的发展可以分为以下几个阶段：

1. 传统门锁阶段：传统门锁主要通过钥匙或手机来实现门锁的开锁功能。这种门锁的安全性较低，易受到盗窃者的攻击。

2. 电子门锁阶段：电子门锁通过电子密码或卡片来实现门锁的开锁功能。这种门锁的安全性较高，但需要购买额外的卡片或密码，使用成本较高。

3. 智能门锁阶段：智能门锁通过互联网和移动应用来实现门锁的开锁功能。这种门锁的安全性更高，使用方便性更强，但需要互联网连接，可能存在网络安全问题。

1. 使用Arduino开发门锁硬件：

#include <WiFi.h>
#include <WiFiClient.h>

const char* ssid = "your_SSID";
const char* password = "your_PASSWORD";

const char* host = "your_HOST";
const int port = 3000;

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }
  Serial.println("Connected to WiFi");
}

void loop() {
  WiFiClient client = WiFi.client();
  if (client.connect(host, port)) {
    Serial.println("Connect to server");
    String password = "your_PASSWORD";
    String postData = "password=" + password;
    client.print(String("POST /unlock HTTP/1.1\r\n") +
                 "Host: " + host + "\r\n" +
                 "Content-Type: application/x-www-form-urlencoded\r\n" +
                 "Content-Length: " + postData.length() + "\r\n" +
                 "Connection: close\r\n\r\n" +
                 postData);
    while (client.available()) {
      String line = client.readStringUntil('\r');
      Serial.print(line);
    }
    Serial.println("\r\n");
    client.stop();
  } else {
    Serial.println("Failed to connect to server");
  }
  delay(10000);
}
