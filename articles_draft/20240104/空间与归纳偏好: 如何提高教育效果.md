```python
from collections import Counter

# 学生的空间和归纳偏好数据
students = [
    {'space': '实验室', 'subject': '数学'},
    {'space': '教室', 'subject': '语文'},
    {'space': '实验室', 'subject': '语文'},
    {'space': '教室', 'subject': '数学'},
    {'space': '实验室', 'subject': '数学'},
]

# 计算空间的统计
space_counter = Counter(student['space'] for student in students)
print(space_counter)

# 计算归纳偏好的统计
subject_counter = Counter(student['subject'] for student in students)
print(subject_counter)
```
