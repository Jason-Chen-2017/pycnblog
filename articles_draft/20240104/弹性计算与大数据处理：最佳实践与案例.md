```python
from multiprocessing import Pool

# 任务调度
def process_data(data):
    # 数据处理
    processed_data = data.process()
    return processed_data

if __name__ == '__main__':
    # 创建进程池
    pool = Pool(processes=4)
    
    # 分配任务
    results = pool.map(process_data, [data_train, data_test])
    
    # 关闭进程池
    pool.close()
    pool.join()
```
