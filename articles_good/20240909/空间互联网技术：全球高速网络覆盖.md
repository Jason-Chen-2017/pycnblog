#### 算法编程题 1：编写一个程序，实现卫星轨道的计算。

**答案：** 

```python
import math

def calculate_satellite_orbit(radius, inclination, eccentricity):
    # 计算轨道周期
    period = 2 * math.pi * math.sqrt(radius**3 / (gravitational_constant * (1 - eccentricity**2)))
    
    # 计算轨道倾角
    inclination_angle = math.atan(inclination / (1 - eccentricity))
    
    # 计算轨道半长轴
    semi_major_axis = radius / (1 - eccentricity)
    
    return period, inclination_angle, semi_major_axis

gravitational_constant = 6.674 * 10**-11
radius = 6.371 * 10**6  # 地球半径
inclination = 0.00668  # 轨道倾角
eccentricity = 0.0167  # 轨道偏心率

period, inclination_angle, semi_major_axis = calculate_satellite_orbit(radius, inclination, eccentricity)
print("轨道周期：", period)
print("轨道倾角：", inclination_angle)
print("轨道半长轴：", semi_major_axis)
```
