```python
import torch
from torch import nn

class TransformerModel(nn.Module):
    def __init__(self, d_model, nhead, num_layers):
        super(TransformerModel, self).__init__()

        self.fc = nn.Linear(d_model, 1)
```

```python
import torch
from torch import nn

class TransformerRecommender(nn.Module):
    def __init__(self, d_model, nhead, num_layers):
        super(TransformerRecommender, self).__init__()
        self.user_embedding = nn.Embedding(d_model, nhead)
        self.item_embedding = nn.Embedding(d_model, nhead)
        self.transformer = nn.Transformer(d_model, nhead, num_layers)
        self.fc = nn.Linear(d_model, 1)

    def forward(self, user_sequence, item_sequence):
        user_embedding = self.user_embedding(user_sequence)
        item_embedding = self.item_embedding(item_sequence)
        output = self.transformer(user_embedding, item_embedding)
        output = self.fc(output)
        return output
```
