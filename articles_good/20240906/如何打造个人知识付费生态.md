    private Map<String, List<String>> interests;

    public UserProfile() {
        this.attributes = new HashMap<>();
        this.interests = new HashMap<>();
    }

    public void addAttribute(String key, String value) {
        attributes.put(key, value);
    }

    public void addInterest(String category, String interest) {
        if (interests.containsKey(category)) {
            interests.get(category).add(interest);
        } else {
            List<String> interestsList = new ArrayList<>();
            interestsList.add(interest);
            interests.put(category, interestsList);
        }
    }

    public Map<String, String> getAttributes() {
        return attributes;
    }

    public Map<String, List<String>> getInterests() {
        return interests;
    }
}
