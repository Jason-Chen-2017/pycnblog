    return -x - y - z

def constraint_function(x, y, z):
    return x + y + z - 100

# 求解线性规划问题
def solve_linear_program(x_min, x_max, y_min, y_max, z_min, z_max):
    x = np.linspace(x_min, x_max, 100)
    y = np.linspace(y_min, y_max, 100)
    z = np.linspace(z_min, z_max, 100)

    objective_values = np.zeros((x.shape[0], y.shape[0], z.shape[0]))
    constraint_values = np.zeros((x.shape[0], y.shape[0], z.shape[0]))

    for i in range(x.shape[0]):
        for j in range(y.shape[0]):
            for k in range(z.shape[0]):
                objective_values[i, j, k] = objective_function(x[i], y[j], z[k])
                constraint_values[i, j, k] = constraint_function(x[i], y[j], z[k])

    plt.contourf(x, y, z, objective_values)
    plt.colorbar()
    plt.contour(x, y, constraint_values, colors='r')
    plt.xlabel('研发投入')
    plt.ylabel('市场推广投入')
    plt.title('目标函数与约束条件的等高线图')
    plt.show()

# 求解结果
solve_linear_program(0, 100, 0, 100, 0, 100)
