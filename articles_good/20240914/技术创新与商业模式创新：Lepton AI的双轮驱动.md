  其中，$x$为输入特征，$W$为全连接权重，$b$为偏置。

def conv_net(x, weights, biases):
    # 第一层卷积

    rel_1 = tf.nn.relu(conv_1 + biases['bc1'])

    # 第二层卷积
    conv_2 = tf.nn.conv2d(rel_1, weights['wc2'], strides=[1, 1, 1, 1], padding='VALID')
    rel_2 = tf.nn.relu(conv_2 + biases['bc2'])

    fc_1 = tf.reshape(rel_2, [-1, weights['wd'].get_shape().as_list()[0]])
    fc_1 = tf.matmul(fc_1, weights['wd']) + biases['bd']
    out = tf.nn.relu(fc_1)
