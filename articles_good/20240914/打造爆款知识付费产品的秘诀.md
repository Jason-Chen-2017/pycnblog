#### 5.2.2 内容推荐

```python
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

# 加载用户行为数据
user_behavior = pd.read_csv('user_behavior.csv')

# 构建用户行为矩阵
behavior_matrix = user_behavior.pivot(index='UserID', columns='ItemID', values='Behavior')

# 计算用户行为矩阵的余弦相似度
cosine_similarity_matrix = cosine_similarity(behavior_matrix)

# 根据用户ID，获取相似用户
def get_similar_users(user_id, similarity_matrix, top_n=5):
    return np.argsort(similarity_matrix[user_id])[1:top_n+1]

# 为用户推荐内容
def content_recommendation(user_id, behavior_matrix, similarity_matrix, top_n=5):
    similar_users = get_similar_users(user_id, similarity_matrix, top_n)
    recommended_items = behavior_matrix.loc[similar_users].sum().sort_values(ascending=False).head(top_n)
    return recommended_items

# 测试推荐效果
user_id = 1
recommended_items = content_recommendation(user_id, behavior_matrix, cosine_similarity_matrix)
print(recommended_items)
```

代码解释：

1. 导入所需的库。
2. 读取用户行为数据。
3. 构建用户行为矩阵。
4. 计算用户行为矩阵的余弦相似度。
5. 根据用户ID，获取相似用户。
6. 为用户推荐内容。
