### 5.2 源代码详细实现
```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# 成本核算模型
class CostCalculationModel:
    def __init__(self, gpu_hours, storage_size, bandwidth, manpower_days, maintenance_days):
        self.gpu_hours = gpu_hours
        self.storage_size = storage_size
        self.bandwidth = bandwidth
        self.manpower_days = manpower_days
        self.maintenance_days = maintenance_days
    
    def calculate_total_cost(self, gpu_price, storage_price, bandwidth_price, manpower_price, maintenance_price):
        compute_cost = self.gpu_hours * gpu_price
        storage_cost = self.storage_size * storage_price
        network_cost = self.bandwidth * bandwidth_price
        manpower_cost = self.manpower_days * manpower_price
        maintenance_cost = self.maintenance_days * maintenance_price
        total_cost = compute_cost + storage_cost + network_cost + manpower_cost + maintenance_cost
        return total_cost
    
    def plot_cost_distribution(self):
        costs = [self.gpu_hours * self.gpu_price, self.storage_size * self.storage_price,
                 self.bandwidth * self.bandwidth_price, self.manpower_days * self.manpower_price,
                 self.maintenance_days * self.maintenance_price]
        labels = ['计算资源成本', '存储成本', '网络成本', '人力资源成本', '维护成本']
        plt.pie(costs, labels=labels, autopct='%.1f%%')
        plt.title('成本分布图')
        plt.show()

# 测试代码
if __name__ == '__main__':
    gpu_price = 100
    storage_price = 10
    bandwidth_price = 1
    manpower_price = 2000
    maintenance_price = 3000

    model = CostCalculationModel(gpu_hours=100, storage_size=100, bandwidth=10, manpower_days=10, maintenance_days=2)
    total_cost = model.calculate_total_cost(gpu_price, storage_price, bandwidth_price, manpower_price, maintenance_price)
    print(f"总成本：{total_cost}元")

    model.plot_cost_distribution()
```
