```python
import spacy

# 加载spaCy模型
nlp = spacy.load("en_core_web_sm")

# 定义意图分类模型
model = spacy.training.IntentClassifier(nlp)

# 训练模型
data = [{"text": "I want to book a flight", "label": "BOOK_FLIGHT"},
         {"text": "Can you show me the weather forecast?", "label": "GET_WEATHER"},
         {"text": "I need to check my email", "label": "CHECK_EMAIL"}]
model.update(data)

# 保存模型
model.save("intent_classifier.model")

# 加载训练好的模型
nlp = spacy.load("intent_classifier.model")

# 定义智能对话系统
def chatbot(text):
    doc = nlp(text)
    intent = model.predict_one(doc)
    return intent

# 与用户进行对话
while True:
    user_input = input("User: ")
    bot_response = chatbot(user_input)
    print("Bot:", bot_response)
```
