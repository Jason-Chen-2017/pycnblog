# 导入所需库
import requests
import json

# 定义API接口
API_URL = "https://api.example.com/innovation"

# 获取需求信息
def get_demand():
    response = requests.get(API_URL + "/demand")
    if response.status_code == 200:
        return json.loads(response.text)
    else:
        return None

# 获取资源信息
def get_resources():
    response = requests.get(API_URL + "/resources")
    if response.status_code == 200:
        return json.loads(response.text)
    else:
        return None

# 获取知识付费信息
def get_payment():
    response = requests.get(API_URL + "/payment")
    if response.status_code == 200:
        return json.loads(response.text)
    else:
        return None

# 实现资源匹配
def match_resources(demand, resources):
    matched_resources = []
    for resource in resources:
        if demand["field"] == resource["field"]:
            matched_resources.append(resource)
    return matched_resources

# 实现知识付费
def make_payment(matched_resources, payment):
    for resource in matched_resources:
        resource["payment"] = payment
    return matched_resources

# 主函数
def main():
    demand = get_demand()
    resources = get_resources()
    payment = get_payment()

    if demand and resources and payment:
        matched_resources = match_resources(demand, resources)
        paid_resources = make_payment(matched_resources, payment)
        print("知识付费完成，匹配资源如下：")
        print(paid_resources)
    else:
        print("获取信息失败，请检查网络连接或API接口。")

# 运行主函数
if __name__ == "__main__":
    main()
