1.线性回归：

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
x = np.random.rand(100)
y = 2 * x + 1 + np.random.randn(100)

# 训练线性回归模型
w = np.linalg.inv(x.T @ x) @ x.T @ y
b = np.mean(y) - w @ x.mean(axis=0)

# 预测
x_test = np.array([0, 1])
y_predict = w @ x_test + b

# 绘制图像
plt.scatter(x, y)
plt.plot(x_test, y_predict, 'r-')
plt.show()
```

2.逻辑回归：

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
x = np.random.rand(100)
y = 2 * x + 1 + np.random.randn(100)

# 训练逻辑回归模型
w = np.linalg.inv(x.T @ x + np.eye(x.shape[1]) * 0.01) @ x.T @ (y > 0).astype(int)
b = np.mean(y[y > 0]) - w @ x.mean(axis=0)

# 预测
x_test = np.array([0, 1])
y_predict = 1 / (1 + np.exp(-w @ x_test - b))

# 绘制图像
plt.scatter(x, y)
plt.plot(x_test, y_predict, 'r-')
plt.show()
```

4.决策树：

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
x = np.random.rand(100)
y = 2 * x + 1 + np.random.randn(100)

# 训练决策树模型
# 这里使用sklearn库中的DecisionTreeRegressor进行训练
from sklearn.tree import DecisionTreeRegressor
tree = DecisionTreeRegressor(max_depth=3)
tree.fit(x.reshape(-1, 1), y)

# 预测
x_test = np.array([0, 1])
y_predict = tree.predict(x_test.reshape(-1, 1))

# 绘制图像
plt.scatter(x, y)
plt.plot(x_test, y_predict, 'r-')
plt.show()
```

5.随机森林：

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
x = np.random.rand(100)
y = 2 * x + 1 + np.random.randn(100)

# 训练随机森林模型
# 这里使用sklearn库中的RandomForestRegressor进行训练
from sklearn.ensemble import RandomForestRegressor
forest = RandomForestRegressor(n_estimators=100, max_depth=3)
forest.fit(x.reshape(-1, 1), y)

# 预测
x_test = np.array([0, 1])
y_predict = forest.predict(x_test.reshape(-1, 1))

# 绘制图像
plt.scatter(x, y)
plt.plot(x_test, y_predict, 'r-')
plt.show()
```
