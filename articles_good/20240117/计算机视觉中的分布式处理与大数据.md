import os

def distribute_data(data, num_nodes):
    hash_func = hashlib.md5()
    for i in range(len(data)):
        hash_func.update(data[i].encode('utf-8'))
        node_id = int(hash_func.hexdigest(), 16) % num_nodes
        data[i].node_id = node_id
        data[i].save()
```

2. 任务分配：

from multiprocessing import Pool

def process_data(data):
    return data.process()

def assign_tasks(data, num_nodes):
    pool = Pool(num_nodes)
    results = pool.map(process_data, data)
    pool.close()
    pool.join()
    return results
```

3. 并行处理：

```python
from multiprocessing import Pipe

def process_data(data, conn):
    data.process()
    conn.send(data)

def parallel_process(data, num_nodes):
    connections = []
    for i in range(num_nodes):
        parent_conn, child_conn = Pipe()
        connections.append((child_conn, i))
        process = Process(target=process_data, args=(data, parent_conn))
        process.start()
    for conn, node_id in connections:
        conn.close()
        data.node_id = node_id
```
