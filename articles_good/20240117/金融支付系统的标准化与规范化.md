### 4.1 技术标准实例

```python
import hashlib

def encrypt(data, key):
    return hashlib.sha256(data.encode('utf-8') + key.encode('utf-8')).hexdigest()

data = "Hello, World!"
key = "123456"
encrypted_data = encrypt(data, key)
print(encrypted_data)
```

### 4.2 业务规范实例

```python
class Payment:
    def __init__(self, amount, currency, payer, payee):
        self.amount = amount
        self.currency = currency
        self.payer = payer
        self.payee = payee

    def validate(self):
        if self.amount <= 0 or self.currency == "":
            return False
        return True

payment = Payment(100, "USD", "Alice", "Bob")
if payment.validate():
    print("Payment is valid.")
else:
    print("Payment is invalid.")
```
