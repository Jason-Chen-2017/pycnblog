### 1.4.1 学生数据管理

```python
import sqlite3

# 连接数据库
conn = sqlite3.connect('student.db')

# 创建表
cursor = conn.cursor()
cursor.execute('''
    CREATE TABLE IF NOT EXISTS student (
        id INTEGER PRIMARY KEY,
        name TEXT,
        age INTEGER,
        score INTEGER
    )
''')

# 插入数据
cursor.execute('''
    INSERT INTO student (name, age, score) VALUES (?, ?, ?)
''', ('张三', 20, 80))

# 查询数据
cursor.execute('''
    SELECT * FROM student
''')

# 更新数据
cursor.execute('''
    UPDATE student SET score = ? WHERE id = ?
''', (90, 1))

# 删除数据
cursor.execute('''
    DELETE FROM student WHERE id = ?
''', (1,))

# 关闭连接
conn.close()
```
