```sql
  COUNT(*) AS query_count,

FROM
  performance_schema.events_statements_summary_by_digest
WHERE
  DIGEST_TEXT LIKE '%SELECT%'
  AND DATABASE_ID = 1
  AND DATETIME(start_time) BETWEEN '2021-01-01 00:00:00' AND '2021-01-01 23:59:59';
```

2. 性能瓶颈分析：

```sql
-- 查询性能瓶颈
SELECT
  t.table_name,
  t.index_name,
  s.query_time,
  s.lock_time,
  s.rows_examined,
  s.rows_sent,
  s.num_rows,
  s.created_tmp_disk_tables,
  s.created_tmp_tables,
  s.extra
FROM
  information_schema.tables t
JOIN
  information_schema.stats_statements s
ON
  t.table_schema = s.table_schema
  AND t.table_name = s.table_name
WHERE
  t.table_schema = 1
  AND s.query_time > 10;
```

3. 性能优化：

