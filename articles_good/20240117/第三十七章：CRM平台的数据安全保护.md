## 4.1 数据加密与解密示例

以下是一个AES数据加密与解密示例：

```python
from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes
from Crypto.Util.Padding import pad, unpad

# 生成密钥
key = get_random_bytes(16)

# 生成加密对象
cipher = AES.new(key, AES.MODE_ECB)

# 生成明文
plaintext = "Hello, World!"

# 加密明文
ciphertext = cipher.encrypt(pad(plaintext.encode(), AES.block_size))

# 生成解密对象
cipher = AES.new(key, AES.MODE_ECB)

# 解密密文
plaintext = unpad(cipher.decrypt(ciphertext), AES.block_size)

print(plaintext)
```

## 4.3 数据安全审计示例

以下是一个数据安全审计示例：

```python
import os

# 检查文件是否存在
def file_exists(file_path):
    return os.path.exists(file_path)

# 检查文件是否可读
def file_readable(file_path):
    return os.access(file_path, os.R_OK)

# 检查文件是否可写
def file_writable(file_path):
    return os.access(file_path, os.W_OK)

# 检查文件是否可执行
def file_executable(file_path):
    return os.access(file_path, os.X_OK)

# 检查文件是否存在、可读、可写、可执行
def check_file(file_path):
    return file_exists(file_path) and file_readable(file_path) and file_writable(file_path) and file_executable(file_path)

# 审计文件
file_path = "data.txt"
if check_file(file_path):
    print("文件存在、可读、可写、可执行")
else:
    print("文件不存在或不可读、不可写、不可执行")
```

## 4.4 数据安全与业务连接示例

以下是一个数据安全与业务连接示例：

```python
import os

# 生成数据文件
data = "Hello, World!"
with open("data.txt", "wb") as f:
    f.write(data.encode())

# 加密数据文件
with open("data.txt", "rb") as f:
    data = f.read()
    cipher = AES.new(key, AES.MODE_ECB)
    ciphertext = cipher.encrypt(pad(data, AES.block_size))

# 保存加密数据文件
with open("data_encrypted.txt", "wb") as f:
    f.write(ciphertext)

# 恢复加密数据文件
with open("data_encrypted.txt", "rb") as f:
    ciphertext = f.read()
    cipher = AES.new(key, AES.MODE_ECB)
    plaintext = unpad(cipher.decrypt(ciphertext), AES.block_size)

print(plaintext)
```
