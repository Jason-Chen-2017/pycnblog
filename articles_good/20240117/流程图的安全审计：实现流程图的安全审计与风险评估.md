# 创建流程图

# 添加节点
G.add_node("A")
G.add_node("B")
G.add_node("C")
G.add_node("D")

# 添加边
G.add_edge("A", "B")
G.add_edge("B", "C")
G.add_edge("C", "D")

# 绘制流程图
pos = nx.spring_layout(G)
nx.draw(G, pos, with_labels=True)
plt.show()
