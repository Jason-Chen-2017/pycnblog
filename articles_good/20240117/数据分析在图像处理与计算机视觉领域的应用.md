import numpy as np

# 获取图像的最小和最大灰度值

## 4.6 行为分析中的数据分析

以下是一个使用Python和OpenCV库实现的人行为识别（Human Behavior Recognition）的代码示例：

```python
import cv2
import numpy as np

# 读取视频
video = cv2.VideoCapture('video.mp4')

# 创建一个SVM分类器
svm = cv2.ml.SVM_create()

# 训练SVM分类器
svm.train(np.array([[1, 2], [3, 4], [5, 6], [7, 8]], np.array([0, 1, 1, 0])))

# 读取视频帧
while True:
    ret, frame = video.read()
    if not ret:
        break

    # 使用Haar分类器进行人脸检测
    face_cascade = cv2.CascadeClassifier('haarcascade_frontalface_default.xml')
    faces = face_cascade.detectMultiScale(frame, 1.1, 4)

    # 对每个人脸进行行为识别
    for (x, y, w, h) in faces:
        roi = frame[y:y+h, x:x+w]
        roi_gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
        roi_features = svm.predict(roi_gray.ravel())

        # 根据行为识别结果进行标注
        if roi_features == 1:
            cv2.putText(frame, 'Walking', (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.9, (255, 255, 255), 2)
        else:
            cv2.putText(frame, 'Standing', (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.9, (255, 255, 255), 2)

    # 显示视频帧
    cv2.imshow('Video', frame)

    # 按任意键退出
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

# 释放资源
video.release()
cv2.destroyAllWindows()
```
