```javascript
import { useFlow, useElements } from 'react-flow-renderer';

const FlowComponent = () => {
  const [nodes, setNodes] = useState([]);
  const [edges, setEdges] = useState([]);
  const flowRef = useFlow();
  const onConnect = useElements((els) => (parent, child) => {
    setEdges((eds) => getEdges(eds, parent.id, child.id));
  });

  useEffect(() => {
    setNodes([
      { id: '1', data: { label: 'Start' } },
      { id: '2', data: { label: 'Process' } },
      { id: '3', data: { label: 'End' } },
    ]);
    setEdges([
      { id: 'e1-2', source: '1', target: '2', animated: true },
      { id: 'e2-3', source: '2', target: '3', animated: true },
    ]);
  }, []);

  return (
    <div>
      <button onClick={() => flowRef.zoomToFit()}>Zoom</button>
      <div style={{ height: '100%' }}>
        <ReactFlow elements={nodes} onConnect={onConnect} />
      </div>
    </div>
  );
};

export default FlowComponent;
```
