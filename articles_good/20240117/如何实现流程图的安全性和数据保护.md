def safe_input(prompt):
    while True:
        try:
            value = input(prompt)
            if value.isdigit():
                return int(value)
            else:
                raise ValueError
        except ValueError:
            print("Invalid input. Please enter a number.")

def safe_output(value):
    return str(value).encode("utf-8")

value = safe_input("Please enter a number: ")
print(safe_output(value))
