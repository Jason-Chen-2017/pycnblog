### 4.1.1 线性回归

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
x = np.random.rand(100)
y = 2 * x + 1 + np.random.randn(100)

# 线性回归模型
a = 2
b = 1

# 计算梯度下降
learning_rate = 0.01
n_iterations = 1000

# 训练数据
x_train = x.reshape(-1, 1)
y_train = y.reshape(-1, 1)

# 训练模型
for i in range(n_iterations):
    gradients = 2/len(x_train) * np.dot(x_train.T, (a*x_train + b - y_train))
    a -= learning_rate * gradients
    b -= learning_rate * np.mean(y_train - a*x_train)

# 预测数据
x_test = np.linspace(0, 1, 100)
y_test = a*x_test + b

# 绘制图像
plt.scatter(x_train, y_train, color='blue')
plt.plot(x_test, y_test, color='red')
plt.show()
```

### 4.1.2 逻辑回归

```python
import numpy as np
import matplotlib.pyplot as plt

# 生成随机数据
np.random.seed(0)
x = np.random.rand(100)
y = 2 * x + 1 + np.random.randn(100)

# 逻辑回归模型
a = 2
b = 1

# 训练数据
x_train = x.reshape(-1, 1)
y_train = y.reshape(-1, 1)

# 训练模型
n_iterations = 1000
learning_rate = 0.01

for i in range(n_iterations):
    gradients = 2/len(x_train) * np.dot(x_train.T, (np.log(y_train) - np.log(1 - y_train) - a*x_train - b))
    a -= learning_rate * gradients
    b -= learning_rate * np.mean(np.log(y_train) - np.log(1 - y_train) - a*x_train)

# 预测数据
x_test = np.linspace(0, 1, 100)
y_test = 1 / (1 + np.exp(-a*x_test - b))

# 绘制图像
plt.scatter(x_train, y_train, color='blue')
plt.plot(x_test, y_test, color='red')
plt.show()
```

from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

# 生成随机数据
np.random.seed(0)
x = np.random.rand(100, 2)
y = 2 * x[:, 0] + 1 + np.random.randn(100)

x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.2, random_state=0)

clf.fit(x_train, y_train)
