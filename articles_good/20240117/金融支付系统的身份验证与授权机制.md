    if b == 0:
        return 1, 0
    x, y = ext_gcd(b, a % b)
    return y, x - a // b * y

def rsa_key_gen(p, q):
    n = p * q
    phi = (p - 1) * (q - 1)
    e = 65537
    g, x = ext_gcd(e, phi)
    assert g == 1
    d = (x % phi + phi) % phi
    return n, e, d
