```python
def retry_function(max_retries=3):
    for i in range(max_retries):
        try:
            # 模拟一个可能失败的操作
            time.sleep(1)
            return "Success"

    return "Failed"

print(result)
```

# 4.5 一致性哈希算法实例

```python
import hashlib

def consistent_hash(key, nodes):
    hash_value = hashlib.sha1(key.encode()).digest()
    index = int(hash_value[-1]) % len(nodes)
    return nodes[index]

nodes = ["node1", "node2", "node3"]
key = "Hello, World!"
node = consistent_hash(key, nodes)
print(f"Consistent Hash: {node}")
```
