2.数据挖掘算法实例

```python
from sklearn.associate import Apriori
from sklearn.metrics import accuracy_score

# 生成购物篮数据
data = [
    ['苹果', '香蕉', '橙子'],
    ['苹果', '香蕉', '梨子'],
    ['香蕉', '橙子', '梨子'],
    ['苹果', '梨子'],
    ['香蕉', '梨子']
]

# 训练关联规则挖掘模型
model = Apriori(min_support=0.5, min_confidence=0.7)
model.fit(data)

# 预测购物篮
basket = ['苹果', '香蕉']
prediction = model.predict(basket)

# 计算准确度
correct = accuracy_score(data, prediction)
print(f'准确度：{correct:.2f}')
```

```python
import numpy as np

# 生成路径数据
data = [
    [0, 0],
    [1, 1],
    [2, 2],
    [3, 3],
    [4, 4]
]

# 训练A*算法
def a_star(start, goal, graph):
    # 初始化
    came_from = {}
    cost_so_far = {start: 0}
    f_score = {start: heuristic(start, goal)}
    visited = set()

    # 搜索
    while True:
        current = min(filter(lambda x: x in visited, f_score), key=lambda x: f_score[x])
        if current == goal:
            break
        visited.add(current)
        for next in graph.neighbors(current):
            new_cost = cost_so_far[current] + graph.cost(current, next)
            if next not in visited:
                if next not in cost_so_far or new_cost < cost_so_far[next]:
                    cost_so_far[next] = new_cost
                    came_from[next] = current
                    f_score[next] = new_cost + heuristic(next, goal)
        cost_so_far.remove(current)

    # 回溯
    path = []
    while current != start:
        path.append(current)
        current = came_from[current]
    path.append(start)
    path.reverse()

    return path, cost_so_far[goal]

# 计算曼哈顿距离
def heuristic(a, b):
    return abs(a[0] - b[0]) + abs(a[1] - b[1])

# 生成邻接表
graph = {
    (0, 0): set([(1, 1), (2, 2), (3, 3), (4, 4)]),
    (1, 1): set([(0, 0), (2, 2), (3, 3), (4, 4)]),
    (2, 2): set([(0, 0), (1, 1), (3, 3), (4, 4)]),
    (3, 3): set([(0, 0), (1, 1), (2, 2), (4, 4)]),
    (4, 4): set([(0, 0), (1, 1), (2, 2), (3, 3)])
}

# 计算最短路径
start = (0, 0)
goal = (4, 4)
path, cost = a_star(start, goal, graph)
print(f'最短路径：{path}')
print(f'最短路径长度：{cost}')
```
