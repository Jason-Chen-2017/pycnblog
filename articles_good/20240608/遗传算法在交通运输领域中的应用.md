## 3. 核心算法原理具体操作步骤
遗传算法的基本操作包括选择、交叉和变异。选择操作根据个体的适应度选择个体进入下一代；交叉操作将两个个体的基因进行交换，产生新的个体；变异操作随机改变个体的基因。遗传算法的具体操作步骤如下：
1. 初始化种群：随机生成一组初始个体，每个个体代表一个可能的解。
2. 计算适应度：根据目标函数计算每个个体的适应度。
3. 选择操作：根据个体的适应度选择个体进入下一代。
4. 交叉操作：对选择的个体进行交叉操作，产生新的个体。
5. 变异操作：对新产生的个体进行变异操作，增加种群的多样性。
6. 重复步骤 2-5，直到满足停止条件。

max_cycles = 100  # 最大迭代次数

# 定义交通信号控制的状态
class TrafficLightState:
    GREEN = 0  # 绿灯
    YELLOW = 1  # 黄灯
    RED = 2  # 红灯

# 定义交通信号控制的动作
class TrafficLightAction:
    CONTINUE = 0  # 保持当前状态
    CHANGE_TO_GREEN = 1  # 切换到绿灯
    CHANGE_TO_YELLOW = 2  # 切换到黄灯
    CHANGE_TO_RED = 3  # 切换到红灯

# 定义交通信号控制的模型
class TrafficLightModel:
    def __init__(self, num_lanes, num_vehicles, max_cycles, mutation_rate):
        self.num_lanes = num_lanes
        self.num_vehicles = num_vehicles
        self.max_cycles = max_cycles
        self.mutation_rate = mutation_rate
        self.current_state = [TrafficLightState.RED] * num_lanes  # 初始状态为红灯
        self.next_state = [TrafficLightState.RED] * num_lanes  # 下一状态

    def update_state(self):
        # 模拟车辆到达和通过路口
        for lane in range(self.num_lanes):
            if self.current_state[lane] == TrafficLightState.RED:
                continue
            if random.random() < 0.1:  # 车辆随机到达
                self.next_state[lane] = TrafficLightState.YELLOW
            else:
                self.next_state[lane] = TrafficLightState.GREEN

        # 计算下一状态的适应度
        fitness = [0] * self.num_lanes
        for lane in range(self.num_lanes):
            fitness[lane] = self.calculate_fitness(self.next_state[lane])

        # 选择个体
        selected = random.sample(range(self.num_lanes), k=self.num_vehicles)

        # 交叉操作
        for _ in range(self.num_vehicles // 2):
            parent1 = selected[random.randint(0, len(selected) - 1)]
            parent2 = selected[random.randint(0, len(selected) - 1)]
            child1, child2 = self.crossover(self.next_state[parent1], self.next_state[parent2])

            # 变异操作
            if random.random() < self.mutation_rate:
                child1 = self.mutate(child1)

            # 更新种群
            self.current_state[parent1] = child1
            self.current_state[parent2] = child2

    def calculate_fitness(self, state):
        # 计算适应度
        fitness = 0
        for lane in range(self.num_lanes):
            if state[lane] == TrafficLightState.GREEN:
                fitness += 1
        return fitness

    def crossover(self, state1, state2):
        # 交叉操作
        child1 = [0] * self.num_lanes
        child2 = [0] * self.num_lanes
        for lane in range(self.num_lanes):
            if random.random() < 0.5:
                child1[lane] = state1[lane]
                child2[lane] = state2[lane]
            else:
                child1[lane] = state2[lane]
                child2[lane] = state1[lane]
        return child1, child2

    def mutate(self, state):
        # 变异操作
        mutated_state = [0] * self.num_lanes
        for lane in range(self.num_lanes):
            if random.random() < 0.5:
                mutated_state[lane] = random.choice([TrafficLightState.GREEN, TrafficLightState.YELLOW, TrafficLightState.RED])
        return mutated_state
