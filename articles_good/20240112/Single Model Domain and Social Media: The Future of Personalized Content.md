```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

# 构建域模型
users = ['user1', 'user2', 'user3']
features = [
    {'age': 25, 'gender': 'male', 'location': 'New York'},
    {'age': 30, 'gender': 'female', 'location': 'Los Angeles'},
    {'age': 28, 'gender': 'female', 'location': 'Chicago'}
]

# 构建用户关系模型
relations = {
    'user1': ['user2', 'user3'],
    'user2': ['user1'],
    'user3': ['user1']
}

# 构建用户行为模型
interactions = {
    'user1': ['item1', 'item2', 'item3'],
    'user2': ['item2', 'item3'],
    'user3': ['item1', 'item3']
}

# 构建推荐模型
def recommend(user, items):
    user_features = [f for f in features if f['user'] == user]
    item_features = [{'item': i, 'similarity': 0} for i in items]

    for item in items:
        for user_feature in user_features:
            item_similarity = cosine_similarity([user_feature], [item_features[item]['item']])[0][0]
            item_features[item]['similarity'] = item_similarity

    item_scores = [item['similarity'] for item in item_features]
    return sorted(zip(items, item_scores), key=lambda x: x[1], reverse=True)

# 生成个性化推荐结果
user = 'user1'
items = ['item1', 'item2', 'item3', 'item4', 'item5']
recommendations = recommend(user, items)
print(recommendations)
```
