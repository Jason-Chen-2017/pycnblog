```python
    # 将字符串倒序
    num1, num2 = num1[::-1], num2[::-1]
    max_len = max(len(num1), len(num2))
    # 补齐长度
    num1 += ['0'] * (max_len - len(num1))
    num2 += ['0'] * (max_len - len(num2))
    carry = 0

    return ''.join(result[::-1])
```

```javascript
function encrypt(data, key) {
    let encrypted = '';
    for (let i = 0; i < data.length; i++) {
        encrypted += String.fromCharCode(data.charCodeAt(i) ^ key.charCodeAt(i % key.length));
    }
    return encrypted;
}

function decrypt(encrypted, key) {
    return encrypt(encrypted, key);
}
```

**解析：** `encrypt` 函数通过异或运算加密字符串。`decrypt` 函数实质上是`encrypt`函数的别名，因为异或运算具有可逆性，加密和解密使用相同的密钥。

```python
    subarrays = set()

    for i in range(n):
        for j in range(i, n):
```

```java
public class ExtractPrimes {

        List<Integer> primes = new ArrayList<>();
        for (int num : arr) {
            if (isPrime(num)) {
                primes.add(num);

                return false;
            }
        }
        return true;
    }
}
```

**解析：** 这个函数首先定义一个`isPrime`辅助函数来判断一个数是否为素数，然后遍历整数数组，将素数添加到列表中。

```java
import java.util.ArrayList;
import java.util.List;

public class ExtractPrimes {
    public static List<Integer> extract_primes(int[] arr) {
        List<Integer> primes = new ArrayList<>();
        for (int num : arr) {
            if (isPrime(num)) {
                primes.add(num);
            }
        }
        return primes;
    }

    private static boolean isPrime(int num) {
        if (num <= 1) {
            return false;
        }
        for (int i = 2; i <= num / 2; i++) {
            if (num % i == 0) {
                return false;
            }
        }
        return true;
    }
}
```
