**示例代码：**

```go
package main

import (
    "fmt"
    "time"
)

type TrafficLight struct {
    State string
}

func NewTrafficLight() *TrafficLight {
    return &TrafficLight{State: "green"}
}

func (tl *TrafficLight) ChangeState() {
    switch tl.State {
    case "green":
        tl.State = "yellow"
    case "yellow":
        tl.State = "red"
    case "red":
        tl.State = "green"
    }
}

func main() {
    trafficLight := NewTrafficLight()

    for {
        trafficLight.ChangeState()
        fmt.Println("Traffic Light:", trafficLight.State)
        time.Sleep(2 * time.Second)
    }
}
```

#### 2. 如何实现一个基于历史数据的交通流量预测算法？

**答案：** 可以采用以下步骤实现一个基于历史数据的交通流量预测算法：

1. 收集历史交通流量数据。
2. 对历史数据进行预处理，如去噪、归一化等。
3. 选择合适的机器学习模型，如线性回归、决策树等，进行训练。
4. 使用训练好的模型对未来的交通流量进行预测。

**示例代码：**

```go
package main

import (
    "fmt"
    "math"
    "time"
)

func linearRegression(x, y []float64) (slope, intercept float64) {
    n := float64(len(x))
    sumX := 0.0
    sumY := 0.0
    sumXY := 0.0
    sumXX := 0.0

    for i := 0; i < n; i++ {
        sumX += x[i]
        sumY += y[i]
        sumXY += x[i] * y[i]
        sumXX += x[i] * x[i]
    }

    slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX)
    intercept = (sumY - slope * sumX) / n

    return
}

func predictTrafficFlow(slope, intercept, x float64) float64 {
    return slope*x + intercept
}

func main() {
    // 示例数据
    x := []float64{0, 1, 2, 3, 4, 5}
    y := []float64{2, 4, 6, 8, 10, 12}

    // 训练模型
    slope, intercept := linearRegression(x, y)

    // 预测未来交通流量
    futureX := float64(6)
    predictedY := predictTrafficFlow(slope, intercept, futureX)
    fmt.Printf("Predicted traffic flow for time %v: %v\n", futureX, predictedY)

    // 等待一段时间
    time.Sleep(5 * time.Second)
}
```
