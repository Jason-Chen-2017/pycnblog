```go
package main

import (
    "fmt"
    "sort"
)

// 弹幕结构体
type Bullet struct {
    Content string
    Time    int
}

// 按时间排序的弹幕切片
type BulletSlice []Bullet

// 实现sort.Interface接口
func (bs BulletSlice) Len() int {
    return len(bs)
}

func (bs BulletSlice) Less(i, j int) bool {
    return bs[i].Time < bs[j].Time
}

func (bs BulletSlice) Swap(i, j int) {
    bs[i], bs[j] = bs[j], bs[i]
}

// 弹幕发送函数
func sendBullet(bullets *BulletSlice, content string, time int) {
    *bullets = append(*bullets, Bullet{Content: content, Time: time})
    sort.Sort(*bullets)
}

// 弹幕显示函数
func showBullet(bullets BulletSlice) {
    for _, b := range bullets {
        fmt.Println(b.Content)
    }
}

func main() {
    // 创建弹幕切片
    var bullets BulletSlice

    // 发送弹幕
    sendBullet(&bullets, "大家好！", 0)
    sendBullet(&bullets, "感谢支持！", 5)
    sendBullet(&bullets, "谢谢大家！", 10)

    // 显示弹幕
    showBullet(bullets)
}
```

**3. 如何实现直播间的抽奖功能？**

**答案：**

```go
package main

import (
    "fmt"
    "math/rand"
    "time"
)

// 抽奖结构体
type Lottery struct {
    Prize     string
    Probability float64
}

// 抽奖函数
func drawLottery(lotteries []Lottery) (string, bool) {
    // 初始化随机数生成器
    rand.Seed(time.Now().UnixNano())

    // 计算总概率
    totalProbability := 0.0
    for _, l := range lotteries {
        totalProbability += l.Probability
    }

    // 抽奖
    r := rand.Float64()
    cumulativeProbability := 0.0
    for _, l := range lotteries {
        cumulativeProbability += l.Probability
        if r < cumulativeProbability {
            return l.Prize, true
        }
    }

    return "", false
}

// 抽奖显示函数
func showLottery(prize string, success bool) {
    if success {
        fmt.Printf("恭喜您，抽中了奖品：%s\n", prize)
    } else {
        fmt.Println("很遗憾，没有抽中奖品。")
    }
}

func main() {
    // 创建抽奖列表
    lotteries := []Lottery{
        {"一等奖", 0.1},
        {"二等奖", 0.3},
        {"三等奖", 0.5},
    }

    // 抽奖
    prize, success := drawLottery(lotteries)

    // 显示抽奖结果
    showLottery(prize, success)
}
```
