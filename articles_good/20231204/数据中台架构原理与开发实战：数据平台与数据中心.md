# 4.2 数据清洗示例
```python
import pandas as pd

# 读取数据源
data = pd.read_csv('data.csv')

# 缺失值处理
data['column1'].fillna(data['column1'].mean(), inplace=True)

# 重复值处理
data.drop_duplicates(inplace=True)

# 异常值处理
data = data[(abs(data['column1'] - data['column1'].mean()) <= 2 * data['column1'].std())]

# 噪声值处理
data['column1'] = data['column1'].rolling(window=3).mean()

# 错误值处理
data = data[data['column1'] > 0]

# 数据类型转换
data['column1'] = data['column1'].astype('int')

# 存储数据源
data.to_csv('data_cleaned.csv')
```
