```cpp
#include <iostream>
#include <unistd.h>
#include <cstdlib>
#include <wiringPi.h>

#define LED_PIN 18

void control_light(int state) {
    if (state == 1) {
        digitalWrite(LED_PIN, HIGH);
    } else {
        digitalWrite(LED_PIN, LOW);
    }
}

int main() {
    wiringPiSetup();
    pinMode(LED_PIN, OUTPUT);

    while (true) {
        control_light(1);
        sleep(1);
        control_light(0);
        sleep(1);
    }

    return 0;
}
```
