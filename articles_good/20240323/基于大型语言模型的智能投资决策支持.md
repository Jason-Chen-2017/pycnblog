import pandas as pd
import numpy as np
from transformers import BertTokenizer, BertModel
from sklearn.linear_model import LinearRegression
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, LSTM

# 1. 数据预处理
stock_data = pd.read_csv('stock_data.csv')
news_data = pd.read_csv('news_data.csv')

# 2. 文本特征提取
tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
model = BertModel.from_pretrained('bert-base-uncased')

news_embeddings = []
for text in news_data['text']:
    input_ids = tokenizer.encode(text, return_tensors='pt')
    output = model(input_ids)[1]
    news_embeddings.append(output.detach().numpy()[0])

news_features = np.array(news_embeddings)

# 3. 时间序列建模
X_train, y_train = stock_data[:-10], stock_data['close_price'][:-10]
X_test, y_test = stock_data[-10:], stock_data['close_price'][-10:]

model = Sequential()
model.add(LSTM(64, input_shape=(X_train.shape[1], X_train.shape[2])))
model.add(Dense(1))
model.compile(optimizer='adam', loss='mean_squared_error')
model.fit(X_train, y_train, epochs=50, batch_size=32, validation_data=(X_test, y_test))

# 4. 融合建模
X = np.concatenate((stock_data, news_features), axis=1)
model = LinearRegression()
model.fit(X[:-10], y_train)
predictions = model.predict(X[-10:])
