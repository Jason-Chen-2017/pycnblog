### 源代码详细实现

1. **智能合约**：

   ```solidity
   // SPDX-License-Identifier: MIT
   pragma solidity ^0.8.0;

   contract MetaverseGovernance {
       struct Proposal {
           string title;
           string description;
           address proposer;
           uint256 startDate;
           uint256 endDate;
           bool isClosed;
           mapping(address => bool) voters;
           uint256 totalVotes;
       }

       Proposal[] public proposals;
       mapping(uint256 => mapping(address => bool)) public votes;

       event ProposalCreated(uint256 id, string title, string description);
       event Voted(uint256 id, address voter, bool vote);

       function createProposal(string memory title, string memory description, uint256 duration) public {
           proposals.push(Proposal({
               title: title,
               description: description,
               proposer: msg.sender,
               startDate: block.timestamp,
               endDate: block.timestamp + duration,
               isClosed: false
           }));
           emit ProposalCreated(proposals.length - 1, title, description);
       }

       function vote(uint256 id, bool voteOption) public {
           require(!votes[id][msg.sender], "Already voted");
           require(block.timestamp <= proposals[id].endDate, "Voting is closed");
           votes[id][msg.sender] = voteOption;
           if (voteOption) {
               proposals[id].totalVotes++;
           }
           emit Voted(id, msg.sender, voteOption);
       }
   }
   ```
