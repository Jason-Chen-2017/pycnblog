import pandas as pd
import matplotlib.pyplot as plt

# 参数设置
knowledge_paid_frequency = 5
learning_time_per_course = 3
knowledge-absorption_rate = 0.8
career_demand = 2

# 知识积累函数
def knowledge_accumulation(frequency, time_per_course, absorption_rate):
    return frequency * time_per_course * absorption_rate

# 职业晋升速度函数
def career_progress_speed(demand, knowledge_accumulation):
    return (demand - knowledge_accumulation) / knowledge_accumulation

# 执行函数
knowledge_accumulated = knowledge_accumulation(knowledge_paid_frequency, learning_time_per_course, knowledge-absorption_rate)
career_progress_speed_value = career_progress_speed(career_demand, knowledge_accumulated)

# 结果展示
print("知识积累量：", knowledge_accumulated)
print("职业晋升速度：", career_progress_speed_value)

# 数据可视化
data = pd.DataFrame({'知识付费频率': [knowledge_paid_frequency], '学习时间（小时）': [learning_time_per_course], '知识吸收率': [knowledge-absorption_rate], '职业需求': [career_demand], '知识积累量': [knowledge_accumulated], '职业晋升速度': [career_progress_speed_value]})
data.plot(x='知识付费频率', y='知识积累量', kind='line')
