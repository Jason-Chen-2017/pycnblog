### 1.5.1 推荐算法的Python代码实例

```python
import numpy as np
from scipy.spatial.distance import cosine

def recommend(user_id, user_data, item_data, user_similarity, item_similarity):
    user_similarity = user_similarity[user_id]
    item_similarity = item_similarity[item_data['item_id']]

    user_similar_items = user_similarity[user_similarity > 0]
    user_similar_item_ids = user_similar_items.index[user_similarity > 0]

    item_similar_users = item_similarity[item_similarity > 0]
    item_similar_user_ids = item_similar_users.index[item_similarity > 0]

    user_item_similarity = user_similar_items.dot(item_similar_users.T)
    user_item_similarity = user_item_similarity[user_similar_item_ids].dot(item_similar_user_ids)

    recommended_item_ids = user_item_similarity.argsort()[-10:][::-1]

    recommended_items = item_data.loc[recommended_item_ids]

    return recommended_items
```

### 1.5.2 搜索算法的Python代码实例

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

def search(query, documents, top_n=10):
    vectorizer = TfidfVectorizer()
    document_vectors = vectorizer.fit_transform(documents)

    query_vector = vectorizer.transform([query])

    cosine_similarities = cosine_similarity(query_vector, document_vectors)

    sorted_indices = np.argsort(cosine_similarities[0])[::-1]

    top_n_indices = sorted_indices[:top_n]

    top_n_documents = documents.iloc[top_n_indices]

    return top_n_documents
```
