```python
# 模块1：数据预处理
def preprocess_data(data):
    # 数据清洗
    data = data.dropna()

    # 归一化处理
    mean_temp = data['temperature'].mean()
    std_temp = data['temperature'].std()
    data['temperature'] = (data['temperature'] - mean_temp) / std_temp

    mean_press = data['pressure'].mean()
    std_press = data['pressure'].std()
    data['pressure'] = (data['pressure'] - mean_press) / std_press

    return data

# 模块2：模型训练
def train_model(X, y):
    model = LinearRegression()
    model.fit(X, y)
    return model

# 模块3：预测
def predict(X, model):
    predictions = model.predict(X)
    return predictions

# 模块4：主函数
def main():
    data = pd.read_csv('deep_sea_data.csv')
    data = preprocess_data(data)
    X = data[['temperature', 'pressure']]
    y = data['resource_type']
    model = train_model(X, y)
    predictions = predict(X, model)
    print(predictions)

if __name__ == '__main__':
    main()
```
