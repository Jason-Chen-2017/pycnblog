import numpy as np

def calculate_digital_literacy(score_info_retrieval, score_digital_security, score_network_security):
    """
    计算数字素养水平。

    参数：
    - score_info_retrieval：信息检索能力得分
    - score_digital_security：数字安全意识得分
    - score_network_security：网络安全知识得分
