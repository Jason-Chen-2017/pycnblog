### 5.2 源代码详细实现

以下是一个简单的Python代码实例，用于设计一种高温合金：

```python
import numpy as np
import scipy.optimize as opt

# 定义材料性能预测模型
def performance(S, C, T):
    P = 0.5 * (S ** 2 + C ** 2 + T ** 2)
    return P

# 定义优化目标函数
def objective(x):
    S, C = x
    P = performance(S, C, 1000) # 设定温度为1000K
    return -P # 最大化性能

# 定义约束条件
def constraint(x):
    S, C = x
    return S + C - 1 # 约束条件：S + C = 1

# 定义优化算法
def optimize():
    x0 = [0.5, 0.5] # 初始猜测值
    cons = ({'type': 'ineq', 'fun': constraint})
    result = opt.minimize(objective, x0, constraints=cons)
    return result.x

# 执行优化
x_opt = optimize()
print("最优材料比例：S:", x_opt[0], ", C:", x_opt[1])
```
