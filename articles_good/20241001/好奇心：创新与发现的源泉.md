def curiosity_function(x):

    optimizer = tf.keras.optimizers.Adam()
    optimizer.minimize(loss, var_list=[x])
    return loss

def test(x, y):
    curiosity = curiosity_function(x)
    innovation = innovation_function(x, curiosity)

    loss = tf.keras.losses.MAE(y, discovery)
    return loss

    loss = train(x_train, y_train)
