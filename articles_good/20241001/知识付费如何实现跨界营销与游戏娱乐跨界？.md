以下是一个简单的示例代码，展示如何实现用户积分系统：

```javascript
// 用户积分系统

const express = require('express');
const mongoose = require('mongoose');
const app = express();

// 连接MongoDB数据库
mongoose.connect('mongodb://localhost:27017/knowledgeGame', {
  useNewUrlParser: true,
  useUnifiedTopology: true,
});

// 用户积分模型
const User = mongoose.model('User', new mongoose.Schema({
  username: String,
  points: Number,
}));

// 用户获取积分
app.post('/api/points', async (req, res) => {
  const { username, activityId } = req.body;
  let user = await User.findOne({ username });

  if (!user) {
    user = new User({ username, points: 0 });
    await user.save();
  }

  user.points += 100; // 增加积分
  await user.save();

  res.json({ success: true, points: user.points });
});

// 用户获取当前积分
app.get('/api/points', async (req, res) => {
  const { username } = req.query;
  let user = await User.findOne({ username });
  res.json({ points: user ? user.points : 0 });
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
```

上述代码实现了用户积分系统的基本功能，包括用户获取积分和查询当前积分。具体解读如下：

- **连接MongoDB数据库**：使用mongoose库连接到本地MongoDB数据库，并设置使用新的连接选项。
- **用户积分模型**：定义一个名为`User`的模型，包含`username`和`points`两个字段。
- **用户获取积分**：通过`/api/points`接口，接收用户名和活动ID，更新用户积分，并返回新的积分值。
- **用户获取当前积分**：通过`/api/points`接口，接收用户名，查询用户积分，并返回积分值。
