    while (getline(input, line)) {
        data.push_back(line);
    }

    for (string line : data) {
        string[] words = line.split(",");
        int num = atoi(words[0]);
        string item = words[1];
        data.push_back(item);
    }

    // 机器学习模型
    vector<vector<double>> model;
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});
    model.push_back(vector<double>{0, 0});

    // 计算结果
    int numModel = model.size();
    for (int i = 0; i < numModel; i++) {
        int pos = i - 2;
        double sum = 0;
        for (int j = 1; j < numModel; j++) {
            sum += model[i][j] * model[j][pos];
        }
        double avg = sum / (double)pos;
        model[i][0] = avg;
        model[i][1] = 1;
    }

    // 输出结果
    for (int i = 0; i < numModel; i++) {
        cout << "Model " << i << ": " << model[i][1] << endl;
    }

    return 0;
}
