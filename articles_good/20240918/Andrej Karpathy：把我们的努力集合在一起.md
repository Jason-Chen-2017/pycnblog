\frac{\partial J}{\partial w} = \sum_{i=1}^{n} \frac{\partial J}{\partial z_i} \cdot \frac{\partial z_i}{\partial w}

    tf.keras.layers.Flatten(),

    tf.keras.layers.Dense(10, activation='softmax')
])

# 编译模型
