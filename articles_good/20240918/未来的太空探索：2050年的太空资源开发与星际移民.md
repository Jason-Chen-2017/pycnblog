```python
# 导入必要的库
import numpy as np
import matplotlib.pyplot as plt

# 地质模型
def geology_model(x, y, z):
    d = np.sqrt(x**2 + y**2 + z**2)
    return 1 / d

# 资源模型
def resource_model(x, y, z, t):
    R = np.sin(x) * np.cos(y) * np.exp(-z/t)
    return R

# 开采模型
def mining_model(x, y, z, t):
    C = geology_model(x, y, z) + resource_model(x, y, z, t)
    return C

# 可视化
def visualize(C):
    x = np.linspace(0, 10, 100)
    y = np.linspace(0, 10, 100)
    X, Y = np.meshgrid(x, y)
    Z = mining_model(X, Y, 5, 2)

    plt.contourf(X, Y, Z, cmap='viridis')
    plt.colorbar()
    plt.show()

# 运行可视化
visualize(mining_model)

```
