### 4.1. 后端代码实现

以下是一个简化的后端代码实现示例：

```python
from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///washingshop.db'
db = SQLAlchemy(app)

class Order(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    customer_id = db.Column(db.Integer, nullable=False)
    device_id = db.Column(db.Integer, nullable=False)
    status = db.Column(db.String(50), nullable=False, default='待处理')

@app.route('/orders', methods=['GET', 'POST'])
def orders():
    if request.method == 'POST':
        new_order = Order(customer_id=request.json['customer_id'], device_id=request.json['device_id'])
        db.session.add(new_order)
        db.session.commit()
        return jsonify({'id': new_order.id}), 201
    elif request.method == 'GET':
        orders = Order.query.all()
        return jsonify([{'id': order.id, 'customer_id': order.customer_id, 'device_id': order.device_id, 'status': order.status} for order in orders])

if __name__ == '__main__':
    db.create_all()
    app.run(debug=True)
```

### 4.2. 前端代码实现

以下是一个简化的前端代码实现示例：

```html
<!DOCTYPE html>
<html>
<head>
    <title>Washing Shop Management System</title>

</head>
<body>
    <h1>Washing Shop Management System</h1>
    <div id="orders">
        <h2>Orders</h2>
        <ul id="order-list"></ul>
    </div>
    <script>
        function fetchOrders() {
            $.getJSON('/orders', function(data) {
                var orderList = $('#order-list');
                orderList.empty();
                data.forEach(function(order) {
                    var listItem = $('<li>').text('Order ID: ' + order.id + ', Customer ID: ' + order.customer_id + ', Device ID: ' + order.device_id + ', Status: ' + order.status);
                    orderList.append(listItem);
                });
            });
        }

        $(document).ready(function() {
            fetchOrders();
        });
    </script>
</body>
</html>
```
