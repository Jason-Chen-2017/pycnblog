    doc = nlp(text)

def relation_extraction(text, entities):
    relations = []
    for s, o in combinations(entities, 2):
        # 根据本体中定义的关系模式进行匹配
        for p in g.predicates():
            if g.value(subject=URIRef(s[1]), predicate=p, object=URIRef(o[1])):
                relations.append((s[0], p, o[0]))
    return relations

# 属性标注
def attribute_annotation(entities):
    attributes = defaultdict(dict)
    for e, label in entities:
        # 查找实体在本体中的属性
        for p, o in g.predicate_objects(subject=URIRef(label)):
            attributes[e][p] = o

print("Attributes:", attributes)
