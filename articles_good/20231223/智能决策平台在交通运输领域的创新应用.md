```python
import networkx as nx

# 创建图
G = nx.DiGraph()

# 添加节点
nodes = ['A', 'B', 'C', 'D', 'E', 'F']
G.add_nodes_from(nodes)

# 添加边
edges = [('A', 'B'), ('B', 'C'), ('C', 'D'), ('D', 'E'), ('E', 'F')]
G.add_edges_from(edges)

# 设置边的容量
capacity = {'A': 30, 'B': 20, 'C': 10, 'D': 25, 'E': 15, 'F': 20}
G.edges[('A', 'B')]['capacity'] = capacity['A']
G.edges[('B', 'C')]['capacity'] = capacity['B']
G.edges[('C', 'D')]['capacity'] = capacity['C']
G.edges[('D', 'E')]['capacity'] = capacity['D']
G.edges[('E', 'F')]['capacity'] = capacity['E']

# 最小费用流算法
cost = 0
flow = 0
while True:
    # 找到可行路径
    path = nx.min_path(G, 'A', 'F', content='capacity')
    if not path:
        break
    flow += 1
    cost += nx.get_edge_data(G, path).values()

print('流量:', flow)
print('成本:', cost)
```
