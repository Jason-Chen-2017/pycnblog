```python
from pulp import *

# 创建优化问题
prob = LpProblem("EducationPolicy", LpMinimize)

# 定义决策变量
x = LpVariable("x", lowBound=0)

# 定义成本向量
c = [1, 2, 3]

# 定义限制矩阵和限制向量
A = [[1, 1, 1], [2, 2, 2]]
b = [10, 5]

# 添加限制条件
prob += A * x <= b

# 设置目标函数
prob += c * x

# 解决优化问题
prob.solve()

# 输出结果
print("决策变量的值：", x.varValue)
print("目标函数的值：", prob.objective.value())
```
