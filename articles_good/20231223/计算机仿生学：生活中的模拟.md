```python
import numpy as np
import matplotlib.pyplot as plt

def dN_dt(N, r, K):
    return r * N * (1 - N / K)

def simulate_growth(r, K, t_max, dt):
    N = 0
    t = 0
    t_values = []
    N_values = []

    while t < t_max:
        N = N + dt * dN_dt(N, r, K)
        t = t + dt
        t_values.append(t)
        N_values.append(N)

    plt.ylabel('Cell Number')
    plt.title('Growth and Division Model')
```
