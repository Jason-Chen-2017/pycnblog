## 4.1 语言模型的估计

```python
import random

class LanguageModel:
    def __init__(self, vocabulary, model_parameters):
        self.vocabulary = vocabulary
        self.model_parameters = model_parameters

    def estimate_probability(self, word_sequence):
        probability = 1.0
        for i in range(len(word_sequence)):
            word = word_sequence[i]
            probability *= self.model_parameters[word][word_sequence[:i]]
        return probability

# 初始化词汇表和模型参数
vocabulary = ['apple', 'banana', 'cherry']
model_parameters = {
    'apple': {'apple': 0.5, 'banana': 0.3, 'cherry': 0.2},
    'banana': {'apple': 0.4, 'banana': 0.5, 'cherry': 0.1},
    'cherry': {'apple': 0.3, 'banana': 0.2, 'cherry': 0.5}
}

# 创建语言模型实例
language_model = LanguageModel(vocabulary, model_parameters)

# 生成一个词序列
word_sequence = ['apple', 'banana', 'apple']

# 估计词序列的概率
probability = language_model.estimate_probability(word_sequence)
print(f"The probability of the word sequence is: {probability}")
```

在上述代码中，我们首先定义了一个`LanguageModel`类，其中包含了一个`estimate_probability`方法用于估计给定一个词序列的概率。然后，我们初始化了一个词汇表和模型参数，并创建了一个`LanguageModel`实例。最后，我们生成了一个词序列，并使用`estimate_probability`方法来估计其概率。

```python
import random

class SentenceModel:
    def __init__(self, vocabulary, model_parameters):
        self.vocabulary = vocabulary
        self.model_parameters = model_parameters

    def generate_sentence(self, sentence_length):
        sentence = []
        for _ in range(sentence_length):
            word = random.choices(self.vocabulary, weights=self.model_parameters.values())[0]
            sentence.append(word)
        return sentence

# 初始化词汇表和模型参数
vocabulary = ['apple', 'banana', 'cherry']
model_parameters = {
    'apple': {'apple': 0.5, 'banana': 0.3, 'cherry': 0.2},
    'banana': {'apple': 0.4, 'banana': 0.5, 'cherry': 0.1},
    'cherry': {'apple': 0.3, 'banana': 0.2, 'cherry': 0.5}
}

# 创建句子生成模型实例
sentence_model = SentenceModel(vocabulary, model_parameters)

# 生成一个句子
sentence = sentence_model.generate_sentence(5)
print(f"The generated sentence is: {sentence}")
```

在上述代码中，我们首先定义了一个`SentenceModel`类，其中包含了一个`generate_sentence`方法用于生成一个句子。然后，我们初始化了一个词汇表和模型参数，并创建了一个`SentenceModel`实例。最后，我们使用`generate_sentence`方法来生成一个句子。
