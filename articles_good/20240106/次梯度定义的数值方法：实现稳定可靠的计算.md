    return np.array([x**3 - y**2])

def J(x, y):
    return np.array([[3*x**2], [-2*y]])

def new_mexico(f, J, x0, y0, tol=1e-6, max_iter=1000):
    x, y = x0, y0
    for k in range(max_iter):
        d = -np.linalg.inv(J(x, y)) @ f(x, y)
        x_new = x + alpha * d[0]
        y_new = y + alpha * d[1]
        if np.linalg.norm(d) < tol:
            break
        x, y = x_new, y_new
    return x, y

print("x =", x, "y =", y)
