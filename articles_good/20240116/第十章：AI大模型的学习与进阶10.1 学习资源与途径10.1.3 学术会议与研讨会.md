4. 推荐系统中的协同过滤：

```python
import numpy as np

# 生成随机数据
users = ['user1', 'user2', 'user3', 'user4', 'user5']
ratings = np.random.randint(1, 5, (5, 5))

# 计算协同过滤
similarities = {}
for i in range(len(users)):
    similarities[users[i]] = {}
    for j in range(i + 1, len(users)):
        similarity = np.dot(ratings[i], ratings[j]) / (np.linalg.norm(ratings[i]) * np.linalg.norm(ratings[j]))
        similarities[users[i]][users[j]] = similarity

# 打印协同过滤结果
for user, similarities in similarities.items():
    print(f'{user}: {similarities}')
```
