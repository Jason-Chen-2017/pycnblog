# 定义图像输入层

fc2 = Dense(512, activation='relu')(fc1)
fc3 = Dense(256, activation='relu')(fc2)
# 定义图像输出层
output_image = Dense(num_classes, activation='softmax')(fc3)

# 定义文本输入层

# 定义文本嵌入层
