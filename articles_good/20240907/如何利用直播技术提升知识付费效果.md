```python
import cv2

# 读取视频文件
cap = cv2.VideoCapture('input.mp4')

# 创建输出视频文件
fourcc = cv2.VideoWriter_fourcc('H', '2', '6', '5')
out = cv2.VideoWriter('output.mp4', fourcc, 30.0, (640, 480))

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break

    # 对帧进行编码优化
    frame = cv2.resize(frame, (640, 480))

    # 写入优化后的帧到输出文件
    out.write(frame)

# 释放资源
cap.release()
out.release()
cv2.destroyAllWindows()
```
