#### **1. 数据清洗与预处理**

**题目：** 编写一个 Python 函数，对给定的员工数据进行清洗和预处理，包括去除缺失值、标准化数值型特征、分类编码等。

**答案：** 
```python
import pandas as pd
from sklearn.preprocessing import StandardScaler, OneHotEncoder

def preprocess_data(data):
    # 去除缺失值
    data = data.dropna()

    # 标准化数值型特征
    numerical_features = data.select_dtypes(include=['number'])
    scaler = StandardScaler()
    numerical_features_scaled = scaler.fit_transform(numerical_features)
    data[numerical_features.columns] = numerical_features_scaled

    # 分类编码
    categorical_features = data.select_dtypes(include=['object'])
    encoder = OneHotEncoder()
    categorical_encoded = encoder.fit_transform(categorical_features)
    data_encoded = pd.DataFrame(categorical_encoded.toarray(), columns=encoder.get_feature_names_out())
    data = data.drop(categorical_features.columns, axis=1)
    data = pd.concat([data, data_encoded], axis=1)

    return data

# 示例数据
data = pd.DataFrame({
    'age': [25, 30, 35, None],
    'gender': ['male', 'female', 'male', 'unknown'],
    'department': ['HR', 'IT', 'Finance', 'HR']
})

preprocessed_data = preprocess_data(data)
print(preprocessed_data)
```

#### **2. 员工推荐系统**

**题目：** 编写一个基于协同过滤的员工推荐系统，实现相似度计算和推荐结果生成。

**答案：**
```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

def collaborative_filtering(user_ratings, similarity_threshold=0.5):
    # 计算用户之间的相似度矩阵
    similarity_matrix = cosine_similarity(user_ratings)

    # 根据相似度阈值筛选邻居用户
    neighbors = []
    for i, row in enumerate(similarity_matrix):
        neighbors.append([index for index, value in enumerate(row) if value >= similarity_threshold])

    # 计算推荐结果
    recommendations = []
    for neighbor in neighbors:
        neighbor_ratings = user_ratings[neighbor].mean(axis=0)
        recommendations.append(neighbor_ratings)

    return recommendations

# 示例数据
user_ratings = np.array([
    [5, 3, 0, 1],
    [4, 0, 0, 1],
    [1, 5, 0, 0],
    [1, 4, 4, 3],
    [0, 2, 2, 4]
])

recommendations = collaborative_filtering(user_ratings)
print(recommendations)
```

#### **3. 模型训练与预测**

**题目：** 使用 Keras 编写一个简单的神经网络模型，实现员工绩效预测。

**答案：**
```python
from keras.models import Sequential
from keras.layers import Dense

def build_performance_prediction_model(input_shape):
    model = Sequential()
    model.add(Dense(64, input_shape=input_shape, activation='relu'))
    model.add(Dense(32, activation='relu'))
    model.add(Dense(1, activation='sigmoid'))

    model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
    return model

# 示例数据
X_train = np.array([[0.5, 0.5, 0.5], [0.5, 0.5, 0.5], [0.5, 0.5, 0.5]])
y_train = np.array([1, 0, 1])

model = build_performance_prediction_model(X_train.shape[1:])
model.fit(X_train, y_train, epochs=10, batch_size=1)

predictions = model.predict(X_train)
print(predictions)
```
