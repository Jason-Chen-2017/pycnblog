from sklearn.metrics.pairwise import cosine_similarity

def process_text_query(text_query):

    search_results = search_engine.search(text_query)

g.add_nodes_from(['商品A', '商品B', '品牌A', '类别A', '类别B'])
g.add_edges_from([('商品A', '品牌A'), ('商品B', '品牌A'), ('品牌A', '类别A'), ('类别A', '类别B')])

# 图谱查询
def query_graph(graph, start_node, depth):
    visited = set()
    results = []
    traverse(graph, start_node, depth, visited, results)
    return results

def traverse(graph, node, depth, visited, results):
    if depth == 0 or node in visited:
        return
    visited.add(node)
    results.append(node)
    for neighbor in graph.neighbors(node):
        traverse(graph, neighbor, depth - 1, visited, results)

# 推荐算法
def recommend_from_graph(graph, start_node, depth, user_interest):
    related_nodes = query_graph(graph, start_node, depth)
    recommendations = []
    for node in related_nodes:
        if node in user_interest:
            recommendations.append(node)
    return recommendations
