class VirtualMarket:
    def __init__(self):
        self.currencies = {'gold': 0}
        self.items = {'food': {'price': 10, 'quantity': 100},
                      'toy': {'price': 20, 'quantity': 100}}

    def add_currency(self, currency, amount):
        self.currencies[currency] = amount

    def buy_item(self, item_name, currency):
        if currency in self.currencies and self.items[item_name]['quantity'] > 0:
            if self.currencies[currency] >= self.items[item_name]['price']:
                self.currencies[currency] -= self.items[item_name]['price']

            else:

            print("Invalid item or currency.")

3. **查询优化：** 对于排名前 N 位的查询，可以采用优化算法，如二分查找等。

**代码实例：**

```python
class Leaderboard:
    def __init__(self):
        self.scores = []

    def add_score(self, player, score):
        self.scores.append((player, score))
        self.scores.sort(key=lambda x: x[1], reverse=True)

    def get_rank(self, player):
        for i, (p, _) in enumerate(self.scores):
            if p == player:
                return i + 1
        return -1

    def get_top_n_players(self, n):
        return self.scores[:n]

# 创建排行榜实例
board = Leaderboard()
board.add_score('Alice', 1000)
board.add_score('Bob', 500)
board.add_score('Charlie', 1500)
print(board.get_top_n_players(2))  # 输出：[('Charlie', 1500), ('Alice', 1000)]
```

        self.friends = {}
        self.chats = {}

    def add_friend(self, player, friend):
        if player not in self.friends:
            self.friends[player] = []
        self.friends[player].append(friend)

        if player in self.friends and friend in self.friends[player]:
            self.friends[player].remove(friend)
