```python
import numpy as np
import matplotlib.pyplot as plt
from brainpy import neural
from spacecomm import SpaceComm

# 5.2.1 创建神经网络
net = neural.Network()
layer1 = neural.LIFNeuron(100)
layer2 = neural.LIFNeuron(50)
net.add_layer(layer1, layer2)

# 5.2.2 初始化网络参数
np.random.seed(0)
net.layers[0].weights.value = np.random.normal(0, 1, (100, 50))
net.layers[0]. biases.value = np.random.normal(0, 1, 50)

# 5.2.3 创建通信网络
space_comm = SpaceComm()

# 5.2.4 训练神经网络
for i in range(1000):
    x = np.random.normal(0, 1, 100)
    y = space_comm.communicate(x)
    net.layers[0].input(x)
    net.run()
    net.layers[1].input(y)
    net.run()

# 5.2.5 测试神经网络
x_test = np.random.normal(0, 1, 100)
y_pred = space_comm.communicate(x_test)
net.layers[0].input(x_test)
net.run()
net.layers[1].input(y_pred)
net.run()

plt.scatter(x_test, y_pred)
plt.xlabel('Input')
plt.ylabel('Output')
plt.show()
```
