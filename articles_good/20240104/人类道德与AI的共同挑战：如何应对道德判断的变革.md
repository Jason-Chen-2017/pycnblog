### 4.1 道德判断算法实现

```python
import numpy as np
from sklearn.linear_model import LogisticRegression

# 人类道德规则
rules = [
    {'attribute': 'age', 'value': 18, 'op': '>='},
    {'attribute': 'alcohol', 'value': 0.6, 'op': '<='},
]

# 数据集
data = np.array([
    [20, 0.5],
    [18, 0.6],
    [20, 0.7],
    [18, 0.6],
])

# 编码人类道德规则
def encode_rules(rules):
    f = np.vectorize(lambda x, op: op(x) if op in ['>=', '<='] else 1 - op(x))
    return np.apply_along_axis(f, 1, data, **rules)

# 道德判断
def ethical_judgment(x):
    f = encode_rules(rules)
    return np.all(f >= 1)

# 训练模型
model = LogisticRegression()
model.fit(data, ethical_judgment(data))

# 预测
x = np.array([[20, 0.5], [18, 0.6], [20, 0.7], [18, 0.6]])
y_pred = model.predict(x)
print(y_pred)
```

```python
X_test = scaler.transform(X_test)

# 道德学习
def ethical_learning(X_train, y_train, X_test, y_test):
    model = LogisticRegression()
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)
    return model, y_pred

model, y_pred = ethical_learning(X_train, y_train, X_test, y_test)
print(y_pred)
```
