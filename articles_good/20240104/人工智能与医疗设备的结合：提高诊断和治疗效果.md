### 4.1.1 图像增强

import cv2
import numpy as np

def image_enhancement(image_path):
    # 读取原始图像
    image = cv2.imread(image_path)

    # 对比度调整
    contrast_image = cv2.convertScaleAbs(image, alpha=2, beta=50)

    # 亮度调整
    brightness_image = cv2.add(image, np.array([30]))

    # 锐化
    sharp_image = cv2.GaussianBlur(image, (5, 5), 0)
    sharp_image = cv2.add(image, sharp_image)

    # 显示增强后的图像
    cv2.imshow('Enhanced Image', np.hstack([image, contrast_image, brightness_image, sharp_image]))
    cv2.waitKey(0)
    cv2.destroyAllWindows()

```

### 4.1.2 图像分割

import cv2
import numpy as np

def image_segmentation(image_path):
    # 读取原始图像
    image = cv2.imread(image_path)

    # 转换为灰度图像
    gray_image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # 二值化处理
    _, binary_image = cv2.threshold(gray_image, 127, 255, cv2.THRESH_BINARY)

    # 边缘检测
    edges = cv2.Canny(binary_image, 50, 150)

    # 图像分割
    markers = np.zeros_like(gray_image, dtype=np.uint8)
    markers[edges > 0] = 1
    segmented_image = cv2.watershed(gray_image, markers)

    # 显示分割后的图像
    cv2.imshow('Segmented Image', segmented_image)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

```
