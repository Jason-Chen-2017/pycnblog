## 4.1 权值分配代码实例

```python
def assign_weight(device, functionality, type, quality):
    weight = functionality * 0.4 + type * 0.3 + quality * 0.3
    return weight

device2 = {'id': 2, 'functionality': 70, 'type': 4, 'quality': 80}

device1_weight = assign_weight(device1, device1['functionality'], device1['type'], device1['quality'])
device2_weight = assign_weight(device2, device2['functionality'], device2['type'], device2['quality'])

print("Device1 weight:", device1_weight)
print("Device2 weight:", device2_weight)
```

```python
def convert_weight(device, platform_standard):
    if platform_standard == 'PlatformA':
```
