```python
import hashlib

def consistent_hash(key, nodes):
    # 计算key的哈希值
    hash_value = hashlib.sha1(key.encode('utf-8')).hexdigest()

    # 计算哈希值对应的节点索引
    index = int(hash_value, 16) % len(nodes)

    # 返回对应的节点
    return nodes[index]

nodes = ['node1', 'node2', 'node3']
key = 'example'
node = consistent_hash(key, nodes)
print(node)  # Output: node1
```
