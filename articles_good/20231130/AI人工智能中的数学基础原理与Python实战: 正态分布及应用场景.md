# 4.1正态分布的Python代码实例
在Python中，可以使用numpy库来计算正态分布的概率、期望、方差、标准差等信息。以下是一个正态分布的Python代码实例：

```python
import numpy as np

# 计算正态分布的概率
def normal_probability(x, mu, sigma):
    return (1 / (np.sqrt(2 * np.pi * sigma**2))) * np.exp(-(x - mu)**2 / (2 * sigma**2))

# 计算正态分布的期望
def normal_expectation(mu, sigma):
    return mu

# 计算正态分布的方差
def normal_variance(sigma):
    return sigma**2

# 计算正态分布的标准差
def normal_standard_deviation(sigma):
    return sigma

# 测试正态分布的代码
mu = 0
sigma = 1
x = np.linspace(-3, 3, 100)

# 计算正态分布的概率
probability = normal_probability(x, mu, sigma)

# 计算正态分布的期望
expectation = normal_expectation(mu, sigma)

# 计算正态分布的方差
variance = normal_variance(sigma)

# 计算正态分布的标准差
standard_deviation = normal_standard_deviation(sigma)

# 绘制正态分布的曲线
plt.plot(x, probability)
plt.title('Normal Distribution')
plt.xlabel('x')
plt.ylabel('Probability')
plt.show()
```
