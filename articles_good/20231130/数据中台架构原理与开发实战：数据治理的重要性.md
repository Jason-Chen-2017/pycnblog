    return pd.read_csv(file_path)

    future_data = executor.submit(read_data, 'data.csv')
    data = future_data.result()

# 数据分布
def distribute_data(data, hosts):
    for host in hosts:
        data.to_csv(f'{host}/data.csv')

hosts = ['host1', 'host2', 'host3']
distribute_data(data, hosts)

# 数据一致性
def check_consistency(data1, data2):
