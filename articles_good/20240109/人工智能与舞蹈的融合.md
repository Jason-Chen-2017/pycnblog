```python
def motion_analysis(x, t, a):
    v = np.diff(x) / np.diff(t)
    a = np.diff(v) / np.diff(t)
    return v, a

x = np.array([0, 1, 2, 3, 4])
t = np.array([0, 1, 2, 3, 4])
a = np.array([0, 1, 2, 3, 4])

v, a = motion_analysis(x, t, a)
print("Speed: ", v)
print("Acceleration: ", a)
```
