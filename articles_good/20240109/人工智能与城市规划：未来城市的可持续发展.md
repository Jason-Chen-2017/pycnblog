class CityModel(abm.Model):
    def __init__(self, population, area):

        self.area = area
        self.agents = []

    def step(self):
        for agent in self.agents:
            agent.move()

    def add_agent(self, agent):
        self.agents.append(agent)

# 定义城市中的代理
class Agent:
    def __init__(self, x, y):

        self.y = y

    def move(self):
        self.x += random.choice([-1, 1])
        self.y += random.choice([-1, 1])

# 创建城市模型
