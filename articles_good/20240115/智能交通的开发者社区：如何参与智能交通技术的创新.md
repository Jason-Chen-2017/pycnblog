# 4.3 机器学习的代码实例
机器学习的代码实例如下：

```python
import numpy as np

def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def gradient_descent(x, y, learning_rate, n_iterations):
    m = len(y)
    weights = np.zeros(len(x[0]))
    for _ in range(n_iterations):
        predictions = sigmoid(np.dot(x, weights))
        errors = y - predictions
        gradient = np.dot(x.T, errors) / m
        weights -= learning_rate * gradient
    return weights

x = np.array([[1, 2], [2, 3], [3, 4], [4, 5]])
y = np.array([0, 0, 1, 1])

learning_rate = 0.01
n_iterations = 1000

weights = gradient_descent(x, y, learning_rate, n_iterations)
print("Weights:", weights)
```

```python
import numpy as np

def pid_controller(x, kp, ki, kd):
    n = len(x)
    error = np.zeros(n)
    integral_error = 0
    derivative_error = np.zeros(n)
    u = np.zeros(n)

    for i in range(1, n):
        error[i] = x[i] - x[i-1]
        integral_error += error[i]
        derivative_error[i] = error[i] - error[i-1]
        u[i] = kp * error[i] + ki * integral_error + kd * derivative_error[i]

    return u

x = np.array([1, 2, 3, 4, 5])
kp = 1
ki = 1
kd = 1

u = pid_controller(x, kp, ki, kd)
print("Control Output:", u)
```
