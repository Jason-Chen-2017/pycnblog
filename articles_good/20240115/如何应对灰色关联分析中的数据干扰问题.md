import math

# 计算支持
def support(data, itemset):
    itemset_count = sum(1 for transaction in data if set(transaction).issubset(itemset))
    total_transactions = len(data)
    return itemset_count / total_transactions

# 计算信息增益
def gain(data, itemset):
    I = lambda x: math.log2(len(data) / support(data, x))
    item_entropy = sum(I(item) for item in itemset)
    itemset_entropy = I(itemset)
    return itemset_entropy - item_entropy

# 找到最佳关联规则
