```solidity
pragma solidity ^0.8.0;

contract ArtifactsRegistry {
    struct Artifacts {
        string name;
        string artist;
        uint256 creationDate;
        address owner;
    }

    mapping(string => Artifacts) artifacts;

    function registerArtifact(
        string memory name,
        string memory artist,
        uint256 creationDate
    ) public {
        artifacts[name] = Artifacts({
            name: name,
            artist: artist,
            creationDate: creationDate,
            owner: msg.sender
        });
    }

    function transferArtifactOwnership(
        string memory name,
        address newOwner
    ) public {
        require(artifacts[name].owner == msg.sender, "Not authorized");
        artifacts[name].owner = newOwner;
    }

    function getArtifactDetails(
        string memory name
    ) public view returns (
        string memory,
        string memory,
        uint256,
        address
    ) {
        require(
            keccak256(abi.encodePacked(name)) == keccak256(
                bytes(artifacts[name].name)
            ),
            "Artifact not found"
        );
        return (
            artifacts[name].name,
            artifacts[name].artist,
            artifacts[name].creationDate,
            artifacts[name].owner
        );
    }
}
```
