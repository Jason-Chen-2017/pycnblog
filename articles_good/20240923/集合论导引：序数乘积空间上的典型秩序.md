import networkx as nx
import matplotlib.pyplot as plt

def build_ordinal_product_space(alpha, beta):
    G = nx.Graph()
    for i in range(alpha):
        for j in range(beta):
            G.add_node(i, label=f'{i}')
    for i in range(alpha):
        for j in range(beta):
            G.add_edge(i, j, weight=min(i, j))
    return G

def find_ordinal_order(G):
    order = []
    for i in range(len(G.nodes)):
        min_weight = float('inf')
        min_node = None
        for j in G.nodes:
            if G[i][j]['weight'] < min_weight:
                min_weight = G[i][j]['weight']
                min_node = j
        order.append(min_node)
    return order

def draw_ordinal_order(G, order):
    pos = nx.spring_layout(G)
    nx.draw(G, pos, with_labels=True, node_color='blue', node_size=2000, edge_color='black', width=2)
    for i in range(len(order)):
        nx.draw(G, pos, with_labels=True, node_color='red', node_size=2000, edge_color='black', width=2)
        plt.pause(1)
        nx.draw(G, pos, with_labels=True, node_color='blue', node_size=2000, edge_color='black', width=2)

if __name__ == '__main__':
    alpha = 3
    beta = 4
    G = build_ordinal_product_space(alpha, beta)
    order = find_ordinal_order(G)
