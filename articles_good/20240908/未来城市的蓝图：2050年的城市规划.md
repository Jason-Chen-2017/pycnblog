import numpy as np
import scipy.optimize as opt

# 能源使用数据（单位：kWh）
energy_usage = np.array([100, 120, 130, 110, 90, 100, 150, 170, 140, 130])

# 能源供应能力（单位：kWh）
energy_supply = np.array([200, 200, 200, 200, 200, 200, 200, 200, 200, 200])

# 能源分配策略（单位：kWh）
energy_allocation = np.zeros_like(energy_usage)

# 线性规划模型
def objective_function(allocation):
    # 能源供应能力限制
    supply_constraint = np.sum(allocation) - np.sum(energy_usage)
    if supply_constraint < 0:
        return supply_constraint
    
    # 能源需求满足度
    demand_constraint = np.sum(allocation) - np.sum(energy_usage)
    if demand_constraint < 0:
        return demand_constraint
    
    # 能源浪费
    waste = np.sum(allocation) - np.sum(energy_usage)
    return waste

# 优化算法
result = opt.minimize(objective_function, energy_allocation, method='SLSQP', constraints=constraints)

            emergency_measures[disaster] = "紧急疏散，确保人员安全"
        elif level == 5:
            emergency_measures[disaster] = "全力救援，确保灾区安全"
    
    # 动态调整应急资源
    adjusted_resources = resources.copy()
    for disaster, measure in emergency_measures.items():
        if disaster in ["地震救援队伍", "洪水救援队伍", "医疗设施", "物资储备"]:
            adjusted_resources[disaster] -= 1
    
    return emergency_measures, adjusted_resources
