```python
import tensorflow as tf

# 初始化客户端
client_model = tf.keras.Sequential([
  tf.keras.layers.Dense(10, activation='relu', input_shape=(784,)),
  tf.keras.layers.Dense(1, activation='sigmoid')
])

# 初始化中央服务器
server_model = client_model.clone().wrap_for_reuse()

# 定义训练过程
def federated_train(client_models, server_model, optimizer, num_epochs):
  for epoch in range(num_epochs):
    for client_model in client_models:
      client_optimizer = optimizer
      server_model = client_model.update_state(server_model, client_optimizer)
      loss = client_model.get_loss(server_model)
      print(f'Epoch {epoch}, Loss: {loss}')
  
  return server_model

# 定义优化器和损失函数
optimizer = tf.keras.optimizers.Adam(learning_rate=0.01)
loss_fn = tf.keras.losses.BinaryCrossentropy()

# 运行联邦学习过程
server_model = federated_train(client_models, server_model, optimizer, num_epochs=10)
```
