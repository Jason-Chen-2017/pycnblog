```go
    "sort"

    Denom int64
```

##### 题目 11：字符串转有理数

**题目描述：** 编写一个函数，将字符串表示的有理数转换为 `big.Rat` 类型。

**答案：**

```go
package main

import (
    "fmt"
    "math/big"
    "strings"
)

func stringToRational(s string) *big.Rat {
    parts := strings.Split(s, "/")
    numer := big.NewInt(0)
    denom := big.NewInt(0)
    numer.SetString(parts[0], 10)
    if len(parts) > 1 {
        denom.SetString(parts[1], 10)
    }
    return new(big.Rat).SetInt(numer, denom)
}

func main() {
    s := "3/4"
    r := stringToRational(s)
    fmt.Println("Rational:", r)
}
```

**解析：** 将字符串按照斜杠分割为两部分，分别转换为 `int64` 类型的分子和分母，然后创建 `big.Rat` 对象。

```go
    n := new(big.Rat).SetInt64(int64(len(numbers)))
    mean := new(big.Rat).Quo(sum, n)
    variance := varianceRationals(numbers, mean)
    stdDev := new(big.Rat).Sqrt(variance)
    return sum, mean, stdDev, variance
}

func main() {
    numbers := []*big.Rat{
        big.NewRat(1, 2),
        big.NewRat(2, 3),
        big.NewRat(3, 4),
    }
    sum, mean, stdDev, variance := stats(numbers)
    fmt.Println("Sum:", sum)
    fmt.Println("Mean:", mean)
    fmt.Println("Standard Deviation:", stdDev)
    fmt.Println("Variance:", variance)
}
```

**解析：** 遍历一组有理数，求和，然后使用之前定义的 `averageOfRationals`、`varianceRationals` 和 `sqrt` 函数计算平均值、标准差和方差。

##### 题目 29：求一组有理数的最大值和最小值

**题目描述：** 编写一个函数，求一组有理数的最大值和最小值。

**答案：**

```go
package main

import (
    "fmt"
    "math/big"
)

func findMinMax(numbers []*big.Rat) (*big.Rat, *big.Rat) {
    if len(numbers) == 0 {
        return nil, nil
    }
    min := numbers[0]
    max := numbers[0]
    for _, number := range numbers {
        if number.Cmp(min) < 0 {
            min = number
        }
        if number.Cmp(max) > 0 {
            max = number
        }
    }
    return min, max
}

func main() {
    numbers := []*big.Rat{
        big.NewRat(1, 2),
        big.NewRat(2, 3),
        big.NewRat(3, 4),
    }
    min, max := findMinMax(numbers)
    fmt.Println("Minimum:", min)
    fmt.Println("Maximum:", max)
}
```

**解析：** 遍历一组有理数，比较每个有理数与当前最小值和最大值，更新最小值和最大值。

##### 题目 30：求一组有理数的总和与每个元素之和

**题目描述：** 编写一个函数，求一组有理数的总和以及每个元素的和。

**答案：**

```go
package main

import (
    "fmt"
    "math/big"
)

func sumAndElementSums(numbers []*big.Rat) (*big.Rat, []*big.Rat) {
    totalSum := new(big.Rat).SetInt64(0)
    elementSums := make([]*big.Rat, len(numbers))
    for i, number := range numbers {
        elementSums[i] = new(big.Rat).Set(number)
        totalSum = new(big.Rat).Add(totalSum, number)
    }
    return totalSum, elementSums
}

func main() {
    numbers := []*big.Rat{
        big.NewRat(1, 2),
        big.NewRat(2, 3),
        big.NewRat(3, 4),
    }
    totalSum, elementSums := sumAndElementSums(numbers)
    fmt.Println("Total Sum:", totalSum)
    fmt.Println("Element Sums:", elementSums)
}
```

**解析：** 遍历一组有理数，计算总和和每个元素的和。
