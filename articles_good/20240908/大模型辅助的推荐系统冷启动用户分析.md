### 二、算法编程题库

**6. 编写一个Python程序，用于提取新用户的标签特征。**

```python
def extract_features(user_data):
    # 提取用户行为数据
    behaviors = user_data['behaviors']
    # 提取用户标签特征
    features = []
    for behavior in behaviors:
        feature = behavior['feature']
        features.append(feature)
    return features

user_data = {
    'behaviors': [
        {'feature': '阅读'},
        {'feature': '音乐'},
        {'feature': '旅游'}
    ]
}

features = extract_features(user_data)
print(features)
```

**7. 编写一个Python程序，用于计算用户之间的相似度。**

```python
from sklearn.metrics.pairwise import cosine_similarity

def compute_similarity(user1, user2):
    # 提取用户特征向量
    features1 = extract_features(user1)
    features2 = extract_features(user2)
    # 计算相似度
    similarity = cosine_similarity([features1], [features2])[0][0]
    return similarity

user1 = {
    'behaviors': [
        {'feature': '阅读'},
        {'feature': '音乐'},
        {'feature': '旅游'}
    ]
}

user2 = {
    'behaviors': [
        {'feature': '音乐'},
        {'feature': '电影'},
        {'feature': '旅游'}
    ]
}

similarity = compute_similarity(user1, user2)
print(similarity)
```

**8. 编写一个Python程序，用于生成推荐列表。**

```python
def generate_recommendations(users, top_n):
    # 计算用户之间的相似度
    similarities = []
    for i in range(len(users)):
        for j in range(i + 1, len(users)):
            similarity = compute_similarity(users[i], users[j])
            similarities.append((i, j, similarity))

    # 对相似度进行排序
    sorted similarities = sorted(similarities, key=lambda x: x[2], reverse=True)

    # 生成推荐列表
    recommendations = []
    for i, j, _ in sorted similarities[:top_n]:
        recommendations.append((i, j))

    return recommendations

    ...
]
```
