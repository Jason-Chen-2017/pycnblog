#### 4. 编写一个基于TF-IDF算法的搜索系统。

**答案：** 

```python
import math
from collections import defaultdict

def compute_tf(document, total_words):
    tf = defaultdict(int)
    for word in document:
        tf[word] += 1
    for word in tf:
        tf[word] = tf[word] / total_words
    return tf

def compute_idf(documents):
    idf = defaultdict(int)
    N = len(documents)
    for document in documents:
        unique_words = set(document)
        for word in unique_words:
            idf[word] += 1
    for word in idf:
        idf[word] = math.log(N / idf[word])
    return idf

def compute_tf_idf(document, documents, idf):
    tf = compute_tf(document, len(document))
    tf_idf = defaultdict(float)
    for word in tf:
        tf_idf[word] = tf[word] * idf[word]
    return tf_idf

documents = [['iphone', 'camera', 'phone'], ['camera', 'phone', 'camera'], ['phone', 'camera', 'camera']]
tf_idf = compute_tf_idf(documents[0], documents, compute_idf(documents))
print(tf_idf)
```

**解析：** 这个问题考察了应聘者对TF-IDF算法的理解和实现能力，以及如何处理文本数据。

#### 5. 编写一个基于BERT模型的搜索系统。

**答案：**

```python
from transformers import BertTokenizer, BertModel
import torch

tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
model = BertModel.from_pretrained('bert-base-uncased')

def searchBERT(query, corpus):
    query_embedding = model(torch.tensor([tokenizer.encode(query)]))[0]
    corpus_embeddings = []
    for document in corpus:
        corpus_embeddings.append(model(torch.tensor([tokenizer.encode(document)]))[0])
    similarity_scores = []
    for embedding in corpus_embeddings:
        similarity_scores.append(torch.cosine_similarity(query_embedding, embedding).item())
    return similarity_scores

corpus = ['This is an example document.', 'This document is another example.', 'This is a third example document.']
query = 'example document'
print(searchBERT(query, corpus))
```

**解析：** 这个问题考察了应聘者对BERT模型的理解和应用能力，以及如何使用深度学习模型来处理文本数据。
