from sklearn.cluster import KMeans

# 用户推荐算法
def recommend_courses(user_interests, all_courses):
    recommended_courses = []
    for course in all_courses:
        if np.any(np.isin(course, user_interests)):
            recommended_courses.append(course)
    return recommended_courses

# 示例数据
user_interests = [1, 1]
all_courses = [["编程", "英语"], ["健身", "营养"], ["编程", "营养"], ["健身", "英语"]]

# 推荐课程
