### 4.2 基于Transformer的推荐系统

**题目：** 实现基于Transformer的推荐系统。

**答案：** 基于Transformer的推荐系统模型架构如下：

1. **输入层：** 输入用户和物品的嵌入向量。
2. **编码器层：** 使用多个Transformer编码器块，实现特征提取和编码。
3. **解码器层：** 使用多个Transformer解码器块，实现特征融合和生成推荐列表。

**伪代码：**

```python
# 定义Transformer编码器
class Encoder(nn.Module):
    def __init__(self, input_dim, hidden_dim):
        super(Encoder, self).__init__()
        self.embedding = nn.Embedding(input_dim, hidden_dim)
        self.encoder = nn.Sequential(
            nn.LayerNorm(hidden_dim),
            nn.Linear(hidden_dim, hidden_dim),
            nn.GELU(),
            nn.Dropout(p=0.1)
        )

    def forward(self, x):
        x = self.embedding(x)
        x = self.encoder(x)
        return x

# 定义Transformer解码器
class Decoder(nn.Module):
    def __init__(self, hidden_dim, output_dim):
        super(Decoder, self).__init__()
        self.embedding = nn.Embedding(output_dim, hidden_dim)
        self.decoder = nn.Sequential(
            nn.LayerNorm(hidden_dim),
            nn.Linear(hidden_dim, hidden_dim),
            nn.GELU(),
            nn.Dropout(p=0.1)
        )

    def forward(self, x):
        x = self.embedding(x)
        x = self.decoder(x)
        return x

# 定义推荐系统模型
class RecommenderModel(nn.Module):
    def __init__(self, user_vocab_size, item_vocab_size, hidden_dim):
        super(RecommenderModel, self).__init__()
        self.encoder = Encoder(user_vocab_size, hidden_dim)
        self.decoder = Decoder(item_vocab_size, hidden_dim)
        self.output_layer = nn.Linear(hidden_dim, item_vocab_size)

    def forward(self, user_ids, item_ids):
        user_embeddings = self.encoder(user_ids)
        item_embeddings = self.decoder(item_ids)
        output = self.output_layer(item_embeddings)
        return output
```
