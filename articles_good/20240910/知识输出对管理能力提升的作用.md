// Golang 示例：优先队列实现
package main

import (
    "container/heap"
    "fmt"
)

type PriorityQueue struct {
    items []Item
}

type Item struct {
    Value    interface{}
    Priority float64
    Index    int
}

// 实现sort.Interface接口，用于排序
func (pq PriorityQueue) Len() int {
    return len(pq.items)
}

func (pq PriorityQueue) Less(i, j int) bool {
    // 小根堆
    return pq.items[i].Priority < pq.items[j].Priority
}

func (pq PriorityQueue) Swap(i, j int) {
    pq.items[i], pq.items[j] = pq.items[j], pq.items[i]
    pq.items[i].Index = i
    pq.items[j].Index = j
}

func (pq *PriorityQueue) Push(x interface{}) {
    n := len(pq.items)
    item := x.(Item)
    item.Index = n
    pq.items = append(pq.items, item)
}

func (pq *PriorityQueue) Pop() interface{} {
    old := pq.items
    n := len(old)
    item := old[n-1]
    item.Index = -1 // for safety
    pq.items = old[0 : n-1]
    return item
}

// 更新元素的优先级
func (pq *PriorityQueue) update(item Item) {
    heap.Fix(pq, item.Index)
}

func main() {
    pq := &PriorityQueue{}
    heap.Init(pq)

    pq.Push(Item{Value: "知识点1", Priority: 2.0})
    pq.Push(Item{Value: "知识点2", Priority: 1.0})
    pq.Push(Item{Value: "知识点3", Priority: 3.0})

    for pq.Len() > 0 {
        item := heap.Pop(pq).(Item)
        fmt.Printf("输出知识点：%v，优先级：%v\n", item.Value, item.Priority)
    }
}
