import numpy as np

# 内容基于的推荐算法
def content_based_recommendation(content_matrix, user_id, num_recommendations):
    # 计算内容基于的欧氏距离相似性
    similarity_matrix = np.linalg.norm(content_matrix - content_matrix.dot(np.ones((content_matrix.shape[0], 1))) / content_matrix.sum(axis=1)[:, None], axis=1)
    # 计算用户的兴趣向量
    user_interest_vector = content_matrix[user_id]

    similar_items = np.argsort(-similarity_matrix[user_id])[:num_recommendations]

    return recommended_items
