def dijkstra(graph, start, end):
    # 初始化距离
    distance = {node: float('inf') for node in graph.nodes}
    distance[start] = 0
    # 初始化前驱
    predecessor = {node: None for node in graph.nodes}
    # 初始化已访问节点
    visited = set()
    # 初始化待访问节点
    queue = [(0, start)]
    while queue:
        # 获取距离最近的节点
        current_distance, current_node = heapq.heappop(queue)
        # 如果当前节点已访问，跳过
        if current_node in visited:
            continue
        # 更新距离
        distance[current_node] = current_distance
        # 更新前驱
        predecessor[current_node] = start
        # 更新已访问节点
        visited.add(current_node)
        # 更新待访问节点
        for neighbor, weight in graph[current_node].items():
            new_distance = current_distance + weight
            if new_distance < distance[neighbor]:
                distance[neighbor] = new_distance
                predecessor[neighbor] = current_node
                heapq.heappush(queue, (new_distance, neighbor))
    return distance, predecessor
```

### 4.2 物流资源分配算法实例

def assign_resources(orders, resources):
    # 初始化分配结果
    allocation = {}
    # 初始化剩余资源
    remaining_resources = resources.copy()
    # 初始化待分配订单
    pending_orders = orders.copy()
    # 循环分配资源
    while pending_orders:
        # 获取最优订单
        best_order = min(pending_orders, key=lambda order: order['demand'])
        # 获取最优资源
        best_resource = min(remaining_resources, key=lambda resource: resource['capacity'])
        # 分配资源
        allocation[best_order['id']] = best_resource
        # 更新剩余资源
        remaining_resources[best_resource['id']] -= best_order['demand']
        # 更新待分配订单
        pending_orders.remove(best_order)
    return allocation
```
