```python
# 创建一个知识图谱实例
kg = KnowledgeGraph()

# 添加实体
kg.add_entity("艾伦·戴克", "人")
kg.add_entity("纽约", "地点")
kg.add_entity("世界大战二", "事件")

# 添加关系
kg.add_relation("艾伦·戴克", "职业", "演员")
kg.add_relation("纽约", "地理位置", "美国东部沿海")
kg.add_relation("世界大战二", "时间", "20世纪20年代")

# 添加实体与关系的联系
kg.add_entity_relation_link("艾伦·戴克", "演员")
kg.add_entity_relation_link("纽约", "美国东部沿海")
kg.add_entity_relation_link("世界大战二", "20世纪20年代")
```
