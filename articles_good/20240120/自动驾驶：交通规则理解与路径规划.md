在这个实例中，我们将使用一个基于A*算法的路径规划模型，以选择最佳路径。

```python
import heapq

# 定义A*算法
def a_star_search(graph, start, goal):
    open_set = []
    heapq.heappush(open_set, (0, start))
    came_from = {}
    g_score = {node: 0 for node in graph}
    f_score = {node: 0 for node in graph}

    while open_set:
        current = heapq.heappop(open_set)[1]

        if current == goal:
            break

        for neighbor in graph[current].neighbors():
            new_g_score = g_score[current] + graph[current][neighbor].weight()
            if neighbor not in g_score or new_g_score < g_score[neighbor]:
                came_from[neighbor] = current
                f_score[neighbor] = f_score[current] + heuristic(current, neighbor)
                g_score[neighbor] = new_g_score
                heapq.heappush(open_set, (f_score[neighbor], neighbor))

    return came_from, g_score, f_score

# 定义曼哈顿距离作为启发式函数
def heuristic(a, b):
    return abs(a.x - b.x) + abs(a.y - b.y)

# 使用A*算法选择最佳路径
came_from, g_score, f_score = a_star_search(graph, start, goal)
```
