### 4.1 能源供应

```python
import rospy
from sensor_msgs.msg import BatteryState

def callback(data):
    battery_state = data.state
    if battery_state == 1:
        rospy.loginfo("Battery is charging")
    elif battery_state == 2:
        rospy.loginfo("Battery is discharging")
    elif battery_state == 3:
        rospy.loginfo("Battery is full")
    elif battery_state == 4:
        rospy.loginfo("Battery is low")

def listener():
    rospy.init_node('battery_state_listener', anonymous=True)
    rospy.Subscriber('battery_state', BatteryState, callback)
    rospy.spin()

if __name__ == '__main__':
    listener()
```
