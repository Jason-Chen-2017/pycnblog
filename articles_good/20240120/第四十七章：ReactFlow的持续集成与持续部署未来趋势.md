以下是一个ReactFlow的代码实例：

```javascript
import React from 'react';
import { useNodes, useEdges } from '@react-flow/core';

const MyFlow = () => {
  const nodes = useNodes();
  const edges = useEdges();

  return (
    <div>
      <h1>My Flow</h1>
      <div>
        {nodes.map((node) => (
          <div key={node.id}>{node.data.label}</div>
        ))}
      </div>
      <div>
        {edges.map((edge) => (
          <div key={edge.id}>{edge.data.label}</div>
        ))}
      </div>
    </div>
  );
};

export default MyFlow;
```
