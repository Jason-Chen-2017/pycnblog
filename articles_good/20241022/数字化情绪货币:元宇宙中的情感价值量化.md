    "function executeOrder(uint256 orderId) external returns (bool)",
];

async function createOrder(web3Provider, wallet, tokenA, amountA, tokenB, amountB) {
    const tokenAContract = new Contract(tokenA, ERC20_ABI, web3Provider);
    const tokenBContract = new Contract(tokenB, ERC20_ABI, web3Provider);

    await tokenAContract.transfer(exchangeContract.address, amountA);
    await tokenBContract.transfer(exchangeContract.address, amountB);

    const tx = await exchangeContract.createOrder(tokenA, amountA, tokenB, amountB, { from: wallet.address });
    return tx;
}

async function executeOrder(web3Provider, wallet, orderId) {
