```python
from jython.jar.rules.RuleEngine import RuleEngine
from jython.jar.rules.RuleSet import RuleSet
from jython.jar.rules.Rule import Rule
from jython.jar.rules.Fact import Fact

# 创建规则引擎
rule_engine = RuleEngine()

# 创建规则集
rule_set = RuleSet()

# 创建规则
rule1 = Rule(
    "如果年龄大于等于 18 且性别为男，则满足年轻人的定义",
    "age >= 18 and gender == 'male'",
    "定义为年轻人"
)
rule_set.add_rule(rule1)

# 创建事实
fact1 = Fact("年龄", 20)
fact2 = Fact("性别", "male")

# 添加事实到规则引擎
rule_engine.add_fact(fact1)
rule_engine.add_fact(fact2)

# 执行规则引擎
rule_engine.fire(rule_set)
```
