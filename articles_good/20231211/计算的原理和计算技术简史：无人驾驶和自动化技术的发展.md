from scipy.spatial import distance

# 起点和终点
start = np.array([0, 0])
end = np.array([10, 10])

# 障碍物
obstacles = [np.array([1, 1]), np.array([9, 1]), np.array([1, 9])]

distance_start_end = distance.euclidean(start, end)

# 计算起点和障碍物之间的距离
distances_start_obstacles = [distance.euclidean(start, obstacle) for obstacle in obstacles]

# 计算终点和障碍物之间的距离
distances_end_obstacles = [distance.euclidean(end, obstacle) for obstacle in obstacles]

# 判断是否可以到达终点
if min(distances_start_obstacles) > distance_start_end and min(distances_end_obstacles) > distance_start_end:
    print('可以到达终点')
else:
    print('无法到达终点')

    error = set_point - current_value
