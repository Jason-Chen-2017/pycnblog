2. 数据清洗：

```python
# 消除数据中的噪声
def remove_noise(data):
    return data.dropna()

# 消除数据中的错误
def remove_error(data):
    return data.replace('', np.nan).dropna()

# 消除数据中的缺失值
def remove_missing(data):
    return data.fillna(data.mean())

# 整理数据
def normalize_data(data):
    return (data - data.min()) / (data.max() - data.min())

# 清洗数据
def clean_data(data):
    return remove_noise(remove_error(remove_missing(normalize_data(data))))

# 对所有数据进行清洗
climate_data = clean_data(climate_data)
soil_data = clean_data(soil_data)
crop_data = clean_data(crop_data)
agriculture_data = clean_data(agriculture_data)
```
