### 1.4.5 文本摘要

```python
import spacy
from spacy.lang.en.stop_words import STOP_WORDS

# 加载语言模型
nlp = spacy.load('en_core_web_sm')

# 文本摘要
def text_summarization(text):
    doc = nlp(text)
    sentences = list(doc.sents)
    sentence_scores = [len(token.lemma_ for token in sent.tokens) for sent in sentences]
    avg_sentence_length = sum(sentence_scores) / len(sentence_scores)
    sentence_scores = [score / avg_sentence_length for score in sentence_scores]
    summarized_sentences = [sent for sent, score in zip(sentences, sentence_scores) if score > 0.5]
    summarized_text = ' '.join([sent.text for sent in summarized_sentences])
    return summarized_text

# 使用文本摘要

print(summarized_text)
```

### 1.4.6 情感分析

from spacy.lang.en.stop_words import STOP_WORDS

# 加载语言模型
nlp = spacy.load('en_core_web_sm')

# 情感分析
def sentiment_analysis(text):
    doc = nlp(text)
    sentiment_scores = [token.pos_ for token in doc]
    avg_sentiment_score = sum(sentiment_scores) / len(sentiment_scores)
    sentiment = 'positive' if avg_sentiment_score > 0.5 else 'negative'
    return sentiment
