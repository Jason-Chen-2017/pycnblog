$$

$$

### 4.1线性回归

```python

from sklearn.linear_model import LinearRegression

# 数据预处理
x = np.array([[1, 2], [3, 4], [5, 6], [7, 8]])
y = np.array([1, 2, 3, 4])

# 算法训练

model.fit(x, y)

# 算法测试

y_pred = model.predict(x_test)
print(y_pred)
```
