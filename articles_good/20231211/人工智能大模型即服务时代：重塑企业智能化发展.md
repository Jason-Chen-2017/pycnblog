encoder_inputs = Input(shape=(input_dim,))
encoder = Dense(latent_dim, activation='relu')(encoder_inputs)
encoder = Dense(latent_dim, activation='sigmoid')(encoder)

# 构建解码器

# 构建模型

output = decoder(z)

# 编译模型
