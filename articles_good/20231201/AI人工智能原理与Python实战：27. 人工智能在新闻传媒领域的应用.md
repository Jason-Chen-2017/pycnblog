```python
import re
import jieba

def preprocess(text):
    # 去除标点符号
    text = re.sub(r'[^\w\s]', '', text)
    # 停用词去除
    text = ' '.join(jieba.cut(text))
    # 词干提取
    text = ' '.join([word for word in jieba.cut(text) if not jieba.is_stopword(word)])
    return text
```
