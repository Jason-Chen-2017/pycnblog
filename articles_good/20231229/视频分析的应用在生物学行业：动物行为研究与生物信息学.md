import numpy as np

# 加载视频流

while(cap.isOpened()):
    ret, frame = cap.read()

    # 如果帧不存在，则退出循环

        break

    if event == cv2.EVENT_LBUTTONDOWN:
        # 获取目标的位置
        x, y, w, h = cv2.boundingRect(contours)

        # 获取目标的速度
        dx = x - last_x
        dy = y - last_y
        speed = np.sqrt(dx**2 + dy**2)

        # 绘制目标的位置和速度
        cv2.rectangle(frame, (x, y), (x + w, y + h), (255, 0, 0), 2)
        cv2.putText(frame, 'Speed: ' + str(int(speed)), (x, y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 0, 0), 2)

        # 更新上一次的位置
        last_x = x
        last_y = y

    # 显示帧
    cv2.imshow('Behavior analysis', frame)

    # 如果按下'q'键，则退出循环
