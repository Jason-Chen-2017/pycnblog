## 1.1 背景介绍
推荐系统是一种基于数据挖掘、机器学习和人工智能技术的应用，目的是为用户提供个性化的内容或产品推荐。随着互联网的发展，推荐系统已经成为各种在线平台（如电商、社交网络、新闻门户等）的核心功能之一。

在推荐系统中，以下几个核心概念是必须要理解的：

5. 推荐策略：推荐策略是推荐系统中的核心，它决定了如何根据用户需求和内容特征，为用户提供推荐。

推荐系统的核心概念之间存在着密切的联系。例如，用户行为会影响用户需求的预测，而内容特征会影响内容筛选与排序。这些概念的联系使得推荐系统能够实现个性化推荐，从而提高用户满意度和用户活跃度。

import numpy as np

def cosine_similarity(u, v):
    dot_product = np.dot(u, v)
    norm_u = np.linalg.norm(u)
    norm_v = np.linalg.norm(v)
    return dot_product / (norm_u * norm_v)

def collaborative_filtering(ratings, k=5):
    user_similarity = {}
    for user, user_ratings in ratings.items():
        user_similarity[user] = {}
        for other_user, other_ratings in ratings.items():
            if user != other_user:
                user_similarity[user][other_user] = cosine_similarity(user_ratings, other_ratings)
    recommended_items = {}
    for user, user_ratings in ratings.items():
        recommended_items[user] = {}
        similar_users = sorted(user_similarity[user], key=lambda x: user_similarity[user][x], reverse=True)
        for other_user in similar_users[:k]:
            for item, rating in ratings[other_user].items():
                if item not in user_ratings:
                    recommended_items[user][item] = rating
    return recommended_items

ratings = {
    'user1': {'item1': 5, 'item2': 3, 'item3': 4},
    'user2': {'item1': 4, 'item2': 5, 'item3': 3},
    'user3': {'item1': 3, 'item2': 4, 'item3': 5},
}

print(collaborative_filtering(ratings))
