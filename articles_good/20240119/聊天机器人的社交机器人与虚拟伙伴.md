### 4.2 情感识别实例

```python
import tensorflow as tf
from tensorflow.keras.preprocessing.text import Tokenizer
from tensorflow.keras.preprocessing.sequence import pad_sequences
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Embedding, LSTM, Dense

# 文本数据
texts = ["I love this movie", "This is a terrible movie", "I am happy today", "I am sad today"]
labels = [1, 0, 1, 0]  # 1表示正面情感，0表示负面情感

# 文本预处理
tokenizer = Tokenizer(num_words=1000)
tokenizer.fit_on_texts(texts)
sequences = tokenizer.texts_to_sequences(texts)
X = pad_sequences(sequences, maxlen=100)

# 模型构建
model = Sequential()
model.add(Embedding(1000, 64, input_length=100))
model.add(LSTM(64))
model.add(Dense(1, activation="sigmoid"))

# 模型训练
model.compile(optimizer="adam", loss="binary_crossentropy", metrics=["accuracy"])
model.fit(X, labels, epochs=10, batch_size=32)

# 预测
predictions = model.predict(X)
```

### 4.3 人机交互实例

```python
import random

# 对话数据
dialogues = [
    {"role": "user", "content": "Hello, how are you?"},
    {"role": "assistant", "content": "Hello! I'm fine, thank you. How can I help you today?"},
    {"role": "user", "content": "I'm looking for a movie to watch."},
    {"role": "assistant", "content": "Sure, what kind of movie do you like?"}
]

# 对话管理
def generate_response(dialogue, user_role="user"):
    if user_role == "user":
        return "user"
    else:
        responses = ["Sure, what can I do for you?", "I'm here to help you."]
        return random.choice(responses)

# 对话生成
def generate_dialogue(dialogue, user_role="user"):
    if user_role == "user":
        return dialogue
    else:
        response = generate_response(dialogue, user_role)
        dialogue.append({"role": "assistant", "content": response})
        return dialogue

# 测试
dialogue = generate_dialogue(dialogues)
for item in dialogue:
    print(f"{item['role']}: {item['content']}")
```
