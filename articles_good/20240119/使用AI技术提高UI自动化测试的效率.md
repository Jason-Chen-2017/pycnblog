def analyze_test_results(test_results):
    # 根据测试结果进行深入的分析
    analysis_report = ""
    for result in test_results:
        if result.status == "failed":
            analysis_report += f"{result.step} failed with error: {result.error}\n"
    return analysis_report

# 示例测试结果

        self.status = status
        self.error = error
