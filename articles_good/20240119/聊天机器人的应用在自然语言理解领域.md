import nltk

def parse_sentence(sentence):
    grammar = "NP: {<DT>?<JJ>*<NN>}"
    parser = nltk.RegexpParser(grammar)
    parsed_sentence = parser.parse(nltk.word_tokenize(sentence))
    return parsed_sentence
```

from spacy.lang.en import English

nlp = English()

def recognize_entity(sentence):
    doc = nlp(sentence)
    return [(ent.text, ent.label_) for ent in doc.ents]
```

def extract_relation(sentence):
    doc = nlp(sentence)
    return [(ent1.text, ent2.text, rel) for ent1, ent2, rel in doc.ents]
