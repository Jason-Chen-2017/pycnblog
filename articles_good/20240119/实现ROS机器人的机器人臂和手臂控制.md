#!/usr/bin/env python

import rospy
from sensor_msgs.msg import JointState
from std_msgs.msg import Float64

class RobotArmController:
    def __init__(self):
        self.joint_states_pub = rospy.Publisher('joint_states', JointState, queue_size=10)
        self.joint_state_sub = rospy.Subscriber('joint_states', JointState, self.joint_state_callback)

    def joint_state_callback(self, msg):
        joint_states = msg.position
        joint_states.append(Float64(0.0))  # 添加额外的控制量
        self.joint_states_pub.publish(joint_states)

if __name__ == '__main__':
    rospy.init_node('robot_arm_controller')
    controller = RobotArmController()
    rospy.spin()
