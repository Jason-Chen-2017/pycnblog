以下是一个具体的代码实例：

```python
import os
import hashlib

def hash_file(file_path):
    with open(file_path, 'rb') as f:
        file_data = f.read()
        file_hash = hashlib.sha256(file_data).hexdigest()
    return file_hash

def store_file(file_path, storage_path):
    file_hash = hash_file(file_path)
    os.makedirs(storage_path, exist_ok=True)
    with open(os.path.join(storage_path, file_hash), 'wb') as f:
        with open(file_path, 'rb') as g:
            f.write(g.read())

def manage_content(file_path, storage_path):
    file_hash = hash_file(file_path)
    if os.path.exists(os.path.join(storage_path, file_hash)):
        print(f"File {file_hash} already exists.")
    else:
        store_file(file_path, storage_path)
        print(f"File {file_hash} stored successfully.")

if __name__ == "__main__":
    file_path = "example.txt"
    storage_path = "storage"
    manage_content(file_path, storage_path)
```
