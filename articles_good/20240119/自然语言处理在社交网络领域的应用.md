### 4.2 情感分析

```python
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.pipeline import Pipeline

# 情感词典
positive_words = ["好", "喜欢", "满意", "愉快", "快乐"]
negative_words = ["不好", "不喜欢", "不满意", "不愉快", "不快乐"]

# 情感分数
def sentiment_score(text, positive_words, negative_words):
    positive_count = sum([1 for word in positive_words if word in text])
    negative_count = sum([1 for word in negative_words if word in text])
    return positive_count - negative_count

# 情感模型
def sentiment_analysis(text, model):
    return model.predict([text])[0]

# 测试
text = "自然语言处理是一门研究如何让计算机理解和生成人类语言的科学。"
sentiment_score(text, positive_words, negative_words)

# 训练情感模型
corpus = ["自然语言处理是一门研究如何让计算机理解和生成人类语言的科学。", "自然语言处理技术在社交网络领域的应用主要包括文本挖掘、情感分析和语言翻译。"]
model = Pipeline([
    ("vectorizer", CountVectorizer()),
    ("classifier", MultinomialNB()),
])
model.fit(corpus)

# 情感分析
sentiment_analysis(text, model)
```
