        self.symbol = initial_symbol

            if transition:

                self.tape.append(transition[2])

# 定义图灵机的状态集合、输入集合、输出集合、转移规则

    ("q1", "1"): ("q1", "1", "R"),
    ("q2", "0"): ("q2", "0", "R"),
    ("q2", "1"): ("q2", "1", "R")
