class SmartPayment:
    def __init__(self):
        self.payment_methods = ['cash', 'credit_card', 'wechat_pay', 'alipay']

    def check_payment_method(self, method):
        if method in self.payment_methods:
            return True
        else:
            return False

    def calculate_payment(self, amount, method):
        if method == 'cash':
            return amount
        elif method == 'credit_card':
            return amount * 0.95
        elif method == 'wechat_pay':
            return amount * 0.98
        elif method == 'alipay':
            return amount * 0.97

    def pay(self, amount, method):
        if self.check_payment_method(method):
            return self.calculate_payment(amount, method)
        else:
            return 'Payment method is not supported.'
```

class SmartTicket:
    def __init__(self):
        self.tickets = []

    def generate_ticket(self, ticket_id, ticket_info):
        ticket = {'ticket_id': ticket_id, 'ticket_info': ticket_info}
        self.tickets.append(ticket)

    def encrypt_ticket(self, ticket_id, encryption_key):
        encrypted_ticket = encrypt(ticket_id, encryption_key)
        return encrypted_ticket

    def decrypt_ticket(self, encrypted_ticket, decryption_key):
        decrypted_ticket = decrypt(encrypted_ticket, decryption_key)
        return decrypted_ticket

    def get_ticket(self, ticket_id):
        for ticket in self.tickets:
            if ticket['ticket_id'] == ticket_id:
                return ticket
        return 'Ticket not found.'
```
