```python
import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression

# 加载数据
data = pd.read_csv('star_features.csv')

# 预处理数据
data = data.dropna()
data = (data - data.mean()) / data.std()

# 训练模型
X = data[['distance', 'pressure', 'temperature', 'humidity', 'weather']]
y = data['life_possibility']
model = LinearRegression()
model.fit(X, y)

# 预测生命可能性
new_star = np.array([[0.5, 0.8, 0.6, 0.9, 0.7]])
predicted_life_possibility = model.predict(new_star)

print(predicted_life_possibility)
```
