class HomogeneousPolynomial:

        return HomogeneousPolynomial(new_coefficients)

        new_coefficients = [scalar * coeff for coeff in self.coefficients]
        return HomogeneousPolynomial(new_coefficients)

# 将一个数乘以一个齐次单项式
scalar_poly = 2 * poly1
print(scalar_poly)  # 输出: 4x^2
