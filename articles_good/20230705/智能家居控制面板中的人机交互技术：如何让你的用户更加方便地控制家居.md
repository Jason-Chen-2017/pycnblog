#### 4.3.1. 连接家庭电路

```
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define PORT 9000
#define MAX_BUF 256

int main(int argc, char **argv) {
    int sockfd, client_sockfd, read_size;
    struct sockaddr_in server, client;
    char buffer[MAX_BUF];
    char *str = "123456789012345678901234567890";
    int len = strlen(str);
    
    // 初始化套接字
    sockfd = socket(AF_INET, SOCK_STREAM, 0);
    if (sockfd == -1) {
        perror("Error creating socket");
        exit(1);
    }
    
    // 配置套接字
    server.sin_family = AF_INET;
    server.sin_port = htons(PORT);
    server.sin_addr.s_addr = INADDR_ANY;
    
    // 绑定套接字
    if (bind(sockfd, (struct sockaddr *)&server, sizeof(server)) == -1) {
        perror("Error binding socket");
        exit(1);
    }
    
    // 列表头
    printf("    Software: %s
", "My Smart Home Controller");
    printf("    Version: 1.0
");
    printf("    Copyright (c) 2023 My Smart Home Controller");
    
    // 接收客户端连接
    read_size = sizeof(struct sockaddr_in);
    while ((client_sockfd = accept(sockfd, (struct sockaddr *)&client, (socklen_t*)&read_size))!= -1) {
        // 接收数据
        printf("Client connected
");
        
        // 发送数据
        send(client_sockfd, buffer, MAX_BUF, 0);
        
        // 接收数据
        while ((read_size = recv(client_sockfd, buffer, MAX_BUF, 0)) > 0) {
            // 解析数据
            printf("Received %s
", buffer[0]);
            
            // 控制家电
            int cmd = atoi(buffer[0]);
            switch (cmd) {
                case 1:
                    control_device(1);
                    break;
                case 2:
                    control_device(2);
                    break;
                case 3:
                    control_device(3);
                    break;
                case 4:
                    control_device(4);
                    break;
                case 5:
                    control_device(5);
                    break;
                default:
                    break;
            }
        }
        
        // 关闭连接
        close(client_sockfd);
    }
    
    return 0;
}
```
