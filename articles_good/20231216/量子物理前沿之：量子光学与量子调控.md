def quantum_filter(signal, tau):
    sigma = 1
    N = len(signal)
    z = np.zeros(N)
    for i in range(N):
        z[i] = (1 - 1/(2*sigma**2)) * signal[i] + (1/(2*sigma**2)) * (signal[i-1] if i > 0 else 0)
        z[i] += tau * (signal[i] - signal[i-1] if i > 0 else 0)
    return z

# 测试量子滤波算法

tau = 0.1
