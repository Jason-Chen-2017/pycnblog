```python
def draw_mind_map(nodes, edges, center, radius, angle, fontsize):
    plt.figure(figsize=(6, 6))
    plt.axis('equal')

    theta = [angle + 2 * np.pi * i / len(nodes) for i in range(len(nodes))]
    x = [center + radius * np.cos(theta[i]) for i in range(len(nodes))]
    y = [center + radius * np.sin(theta[i]) for i in range(len(nodes))]

    plt.plot(x, y, 'o-', linewidth=2)
    for i, node in enumerate(nodes):
        plt.text(x[i], y[i], node, fontsize=fontsize)

    for i, edge in enumerate(edges):
        plt.plot([x[edge[0]], x[edge[1]]], [y[edge[0]], y[edge[1]]], 'o-', linewidth=1)

    plt.show()

nodes = ['A', 'B', 'C', 'D']
edges = [[0, 1], [1, 2], [2, 3]]
center = 0.5
radius = 1
angle = np.pi / 4
fontsize = 12

draw_mind_map(nodes, edges, center, radius, angle, fontsize)
```
