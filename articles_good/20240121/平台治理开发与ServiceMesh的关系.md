# 创建服务实例
service1 = Service("service1", "127.0.0.1:8081")

# 注册服务
sm.register(service1)

# 发现服务
services = sm.discover()
print(services)

# 负载均衡
load_balancer = sm.load_balancer()

# 故障转移
