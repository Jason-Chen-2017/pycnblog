在实际应用中，我们可以使用以下代码实例来展示软件系统架构黄金法则在机器学习和人工智能领域的应用：

```python
import numpy as np
import tensorflow as tf

# 定义神经网络的层次结构
class NeuralNetwork:
    def __init__(self, input_size, hidden_size, output_size):
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size
        self.weights = {
            'hidden': tf.Variable(tf.random.normal([input_size, hidden_size])),
            'output': tf.Variable(tf.random.normal([hidden_size, output_size]))
        }
        self.biases = {
            'hidden': tf.Variable(tf.zeros([hidden_size])),
            'output': tf.Variable(tf.zeros([output_size]))
        }

    def forward(self, x):
        hidden_layer = tf.nn.relu(tf.matmul(x, self.weights['hidden']) + self.biases['hidden'])
        output_layer = tf.matmul(hidden_layer, self.weights['output']) + self.biases['output']
        return output_layer

# 定义训练函数
def train(network, x, y, learning_rate, epochs):
    with tf.GradientTape() as tape:
        predictions = network.forward(x)
        loss = tf.reduce_mean(tf.square(predictions - y))
    gradients = tape.gradient(loss, network.weights + network.biases)
    optimizer = tf.optimizers.SGD(learning_rate)
    optimizer.apply_gradients(zip(gradients, network.weights + network.biases))

# 定义分布式训练函数
def distributed_train(network, x, y, learning_rate, epochs, num_nodes):
    # 将数据集划分为多个部分
    data_size = x.shape[0]
    chunk_size = data_size // num_nodes
    for i in range(num_nodes):
        start = i * chunk_size
        end = (i + 1) * chunk_size
        x_chunk = x[start:end]
        y_chunk = y[start:end]
        train(network, x_chunk, y_chunk, learning_rate, epochs)

# 定义模块化优化函数
def module_optimize(network, x, y, learning_rate, epochs, num_modules):
    for i in range(num_modules):
        # 训练第 i 个模块
        train(network, x, y, learning_rate, epochs)

# 测试代码
input_size = 10
hidden_size = 5
output_size = 1
x = np.random.rand(100, input_size)
y = np.random.rand(100, output_size)
learning_rate = 0.01
epochs = 100
num_nodes = 4
num_modules = 2

network = NeuralNetwork(input_size, hidden_size, output_size)
distributed_train(network, x, y, learning_rate, epochs, num_nodes)
module_optimize(network, x, y, learning_rate, epochs, num_modules)
```
