```python
import numpy as np
import networkx as nx

# 创建路网
G = nx.DiGraph()

# 添加路段
G.add_edge('A', 'B', capacity=10, cost=1)
G.add_edge('B', 'C', capacity=15, cost=2)
G.add_edge('C', 'D', capacity=20, cost=3)
G.add_edge('D', 'A', capacity=5, cost=4)

# 最小费用流算法
def min_cost_flow(G, s, t, flow):
    cost = 0
    while flow > 0:
        potential = np.zeros(G.number_of_nodes())
        visited = np.zeros(G.number_of_nodes())
        parent = np.zeros(G.number_of_nodes())
        dist = np.inf * np.ones(G.number_of_nodes())

        dist[s] = 0
        visited[s] = 1

        while not visited[t]:
            for node in G.nodes():
                if not visited[node] and dist[node] > dist[s] + potential[node]:
                    dist[node] = dist[s] + potential[node]
                    parent[node] = s
                    visited[node] = 1
                    for edge in G.out_edges(node):
                        if G.flow(edge[0]) < G.capacity(edge[0]):
                            potential[G.destination(edge[0])] = min(potential[node], G.cost(edge[0]))

        if flow > 0 and dist[t] == np.inf:
            return -1

        delta = min(flow, G.capacity(parent[t]), G.cost(parent[t]))
        flow -= delta
        cost += delta * dist[t]
        for node in G.nodes():
            potential[node] += dist[node]

    return cost

# 测试
print(min_cost_flow(G, 'A', 'D', 20))
```
