```python
import matplotlib.pyplot as plt
import numpy as np

def animate(i):
    x = np.linspace(0, 1, 100)
    y = np.sin(2 * np.pi * x)
    plt.plot(x, y)
    plt.xlim(0, 1)
    plt.ylim(-1, 1)
    plt.title(f'Frame {i}')
    plt.pause(0.1)

ani = plt.animation(animate, frames=100, interval=50, blit=True)
plt.show()
```

### 4.2 过渡效果实现

```python
import matplotlib.pyplot as plt
import numpy as np

def draw_line(x, y):
    plt.plot(x, y)
    plt.xlim(0, 1)
    plt.ylim(-1, 1)

x = np.linspace(0, 1, 100)
y = np.zeros_like(x)

plt.figure(figsize=(8, 6))
plt.plot(x, y, 'k-')
plt.title('Initial Line')
plt.show()

y = np.sin(2 * np.pi * x)
plt.figure(figsize=(8, 6))
draw_line(x, y)
plt.title('Transition Line')
plt.show()

y = np.cos(2 * np.pi * x)
plt.figure(figsize=(8, 6))
draw_line(x, y)
plt.title('Final Line')
plt.show()
```
