以下是一个简单的代码实例，演示了如何使用Apache Shiro进行安全事件处理和响应：

```java
import org.apache.shiro.SecurityUtils;
import org.apache.shiro.authc.UsernamePasswordToken;
import org.apache.shiro.authz.UnauthenticatedException;
import org.apache.shiro.crypto.hash.SimpleHash;
import org.apache.shiro.subject.Subject;

public class ShiroExample {
    public static void main(String[] args) {
        // 初始化Shiro环境
        SecurityUtils.setSecurityManager(new MySecurityManager());

        // 创建用户和密码
        String username = "admin";
        String password = "123456";

        // 创建令牌
        UsernamePasswordToken token = new UsernamePasswordToken(username, password);

        // 获取当前用户
        Subject currentUser = SecurityUtils.getSubject();

        try {
            // 登录
            currentUser.login(token);

            // 获取用户密码的散列值
            SimpleHash hash = new SimpleHash("SHA-256", password, null, 2);

            // 验证用户密码
            currentUser.checkPassword(hash.toHex());

            // 获取用户角色和权限
            System.out.println("角色：" + currentUser.getRoles());
            System.out.println("权限：" + currentUser.getAuthorities());
        } catch (UnauthenticatedException e) {
            System.out.println("登录失败：" + e.getMessage());
        }
    }
}
```
