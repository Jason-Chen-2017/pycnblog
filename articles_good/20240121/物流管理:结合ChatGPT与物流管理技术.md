### 4.1 自然语言指令处理

```python
import spacy

nlp = spacy.load("en_core_web_sm")

def process_instruction(instruction):
    doc = nlp(instruction)
    tokens = [token.text for token in doc]
    tags = [token.tag_ for token in doc]
    return tokens, tags

instruction = "Please provide the shipping status of order number 123456."
tokens, tags = process_instruction(instruction)
print(tokens, tags)
```

### 4.3 智能决策支持

```python
from sklearn.linear_model import LogisticRegression

def train_decision_model(data):
    X = data.drop("status", axis=1)
    y = data["status"]
    model = LogisticRegression()
    model.fit(X, y)
    return model

def predict_decision(model, data):
    X = data.drop("status", axis=1)
    y = model.predict(X)
    return y

data = [
    {"order_id": 123456, "days_overdue": 0, "status": "shipped"},
    {"order_id": 123457, "days_overdue": 2, "status": "processing"},
]
model = train_decision_model(data)
predictions = predict_decision(model, data)
print(predictions)
```
