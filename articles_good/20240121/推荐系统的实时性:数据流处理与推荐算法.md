## 4. 具体最佳实践：代码实例和详细解释说明

### 4.1 基于协同过滤的推荐算法实例

```python
import numpy as np

def cosine_similarity(a, b):
    dot_product = np.dot(a, b)
    norm_a = np.linalg.norm(a)
    norm_b = np.linalg.norm(b)
    return dot_product / (norm_a * norm_b)

user_matrix = np.array([[1, 0, 0, 1],
                        [0, 1, 1, 0],
                        [1, 1, 0, 0],
                        [0, 0, 1, 1]])

user_similarity = np.zeros(user_matrix.shape)

for i in range(user_matrix.shape[0]):
    for j in range(i + 1, user_matrix.shape[0]):
        user_similarity[i, j] = cosine_similarity(user_matrix[i, :], user_matrix[j, :])
        user_similarity[j, i] = user_similarity[i, j]

recommended_user = 2
user_similarity = user_similarity[recommended_user, :]

recommended_items = np.argsort(-user_similarity)
print(recommended_items)
```

### 4.2 基于内容的推荐算法实例

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

movie_matrix = np.array([
    ["action", "adventure", "fantasy"],
    ["action", "adventure", "horror"],
    ["comedy", "romance", "fantasy"],
    ["comedy", "romance", "horror"]])

tfidf = TfidfVectorizer()
movie_matrix = tfidf.fit_transform(movie_matrix)

movie_similarity = cosine_similarity(movie_matrix)

recommended_movie = 0
movie_similarity = movie_similarity[recommended_movie, :]

recommended_items = np.argsort(-movie_similarity)
print(recommended_items)
```
