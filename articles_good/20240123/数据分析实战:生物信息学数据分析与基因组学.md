def needleman_wunsch(seq1, seq2):
    m, n = len(seq1), len(seq2)
    d = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1):
        for j in range(n + 1):
            if i == 0 or j == 0:
                d[i][j] = 0
            elif seq1[i - 1] == seq2[j - 1]:
                d[i][j] = d[i - 1][j - 1] + 1
            else:
                d[i][j] = max(d[i - 1][j], d[i][j - 1], d[i - 1][j - 1]) - 1
    align1, align2 = "", ""
    i, j = m, n
    while i > 0 and j > 0:
        if seq1[i - 1] == seq2[j - 1]:
            align1 = seq1[i - 1] + align1
            align2 = seq2[j - 1] + align2
            i -= 1
            j -= 1
        elif d[i - 1][j] > d[i][j - 1]:
            align1 = seq1[i - 1] + align1
            align2 = "-" + align2
            i -= 1
        else:
            align1 = "-" + align1
            align2 = seq2[j - 1] + align2
            j -= 1
    return align1, align2
```

from collections import defaultdict

def de_bruijn(seq, k):
    nodes = set()
    edges = defaultdict(set)
    for i in range(len(seq) - k + 1):
        node = tuple(seq[i:i+k])
        nodes.add(node)
        edges[node].add((seq[i+k-1], node[1:]))
    return nodes, edges
```

## 5. 实际应用场景

生物信息学数据分析与基因组学的实际应用场景包括：

- 基因组比较：比较不同物种的基因组序列，以探索物种之间的演化关系和特征差异。
- 基因表达谱分析：研究不同条件下基因表达水平的变化，以探索基因功能和生物过程的调控机制。
- 基因功能预测：根据基因序列和基因表达谱数据，预测基因的功能和作用。

这些实际应用场景可以帮助生物科学研究人员更好地理解生物过程和物种演化，为生物技术和医疗科学的发展提供有力支持。
