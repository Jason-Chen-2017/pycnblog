```python
from collections import deque

# 创建库存队列
stock_queue = deque()

# 库存增加
def add_stock(goods_id, stock):
    stock_queue.append((goods_id, stock))

# 库存减少
def reduce_stock(goods_id, stock):
    for goods, s in stock_queue:
        if goods == goods_id:
            if s >= stock:
                s -= stock
                return s
            else:
                return s
    return 0

# 库存查询
def query_stock(goods_id):
    for goods, s in stock_queue:
        if goods == goods_id:
            return s
    return 0
```
