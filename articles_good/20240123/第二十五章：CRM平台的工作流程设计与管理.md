以下是一个使用n8n工作流程管理工具的具体代码实例：

```javascript
const n8n = require('n8n');

const workflow = new n8n.Workflow();

workflow
  .nodes([
    {
      id: '1',
      name: 'Create Lead',
      type: 'n8n-nodes-base.createObject',
      credentials: ['salesforce'],
      options: {
        object: 'Lead',
        fields: {
          firstName: 'John',
          lastName: 'Doe',
          company: 'Acme Corp',
          email: 'john.doe@acme.com',
          phone: '123-456-7890',
        },
      },
    },
    {
      id: '2',
      name: 'Update Lead',
      type: 'n8n-nodes-base.updateObject',
      credentials: ['salesforce'],
      options: {
        object: 'Lead',
        fields: {
          status: 'Qualified',
        },
        search: [
          {
            field: 'id',
            value: '{{$global.leadId}}',
            operator: '==',
          },
        ],
      },
    },
  ])
  .edges([
    {
      id: '1',
      source: '1',
      target: '2',
      action: 'workflow',
    },
  ])
  .run((error, result) => {
    if (error) throw error;
    console.log(result);
  });
```
