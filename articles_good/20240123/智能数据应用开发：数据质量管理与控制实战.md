### 4.1 数据质量评估

```python
import pandas as pd

# 加载数据
data = pd.read_csv('data.csv')

# 计算准确性
accuracy = data['target'].apply(lambda x: 1 if x == data.loc[data['target'] == x, 'feature'].iloc[0] else 0).sum() / len(data)

# 计算完整性
completeness = data.isnull().sum().sum() / data.isnull().sum().sum()

# 计算一致性
consistency = data.duplicated().sum() / len(data)

# 计算及时性
timeliness = data['timestamp'].max() - data['timestamp'].min() / data['timestamp'].max() - data['timestamp'].min()

# 计算可靠性
reliability = data.loc[data['source'] == 'reliable'].isnull().sum().sum() / data.loc[data['source'] == 'reliable'].isnull().sum().sum()

# 打印结果
print('准确性:', accuracy)
print('完整性:', completeness)
print('一致性:', consistency)
print('及时性:', timeliness)
print('可靠性:', reliability)
```

### 4.2 数据清洗

```python
# 数据检查
def check_data(data):
    errors = []
    for col in data.columns:
        if data[col].isnull().sum() > 0:
            errors.append((col, data[col].isnull().sum()))
    return errors

# 数据纠正
def correct_data(data, errors):
    for col, count in errors:
        data[col].fillna(data[col].mode()[0], inplace=True)

# 数据验证
def validate_data(data):
    valid = True
    for col in data.columns:
        if data[col].isnull().sum() > 0:
            valid = False
    return valid

# 主程序
data = pd.read_csv('data.csv')
errors = check_data(data)
correct_data(data, errors)
valid = validate_data(data)
print('数据清洗完成，数据有效性:', valid)
```

### 4.4 数据质量改进

```python
import pandas as pd

# 加载数据
data = pd.read_csv('data.csv')

# 数据检查
def check_data(data):
    errors = []
    for col in data.columns:
        if data[col].isnull().sum() > 0:
            errors.append((col, data[col].isnull().sum()))
    return errors

# 数据纠正
def correct_data(data, errors):
    for col, count in errors:
        data[col].fillna(data[col].mode()[0], inplace=True)

# 数据验证
def validate_data(data):
    valid = True
    for col in data.columns:
        if data[col].isnull().sum() > 0:
            valid = False
    return valid

# 数据质量改进
def improve_data_quality(data):
    errors = check_data(data)
    correct_data(data, errors)
    valid = validate_data(data)
    if not valid:
        improve_data_quality(data)
    return data

# 主程序
data = pd.read_csv('data.csv')
improved_data = improve_data_quality(data)
print('数据质量改进完成，数据有效性:', improved_data.isnull().sum().sum() / len(improved_data))
```
