### 4.2 持续集成

使用Jenkins持续集成技术，将开发和运维过程紧密结合，实现开发和运维之间的无缝沟通和协作。

```
pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                sh 'mvn clean install'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Deploy') {
            steps {
                withCredentials([usernamePassword(credentialsId: 'deploy-creds', usernameVariable: 'DEPLOY_USERNAME', passwordVariable: 'DEPLOY_PASSWORD')]) {
                    sh 'sshpass -p $DEPLOY_PASSWORD ssh -o StrictHostKeyChecking=no $DEPLOY_USERNAME@$DEPLOY_HOST "scp target/your-app.jar $DEPLOY_USERNAME@$DEPLOY_HOST:/home/$DEPLOY_USERNAME/your-app.jar"'
                    sh 'sshpass -p $DEPLOY_PASSWORD ssh -o StrictHostKeyChecking=no $DEPLOY_USERNAME@$DEPLOY_HOST "java -jar /home/$DEPLOY_USERNAME/your-app.jar"'
                }
            }
        }
    }
}
```
