```python
def min_job_time(jobs, resources):
    min_time = float('inf')
    for resource in resources:
        resource_jobs = [job for job in jobs if job.resource == resource]
        resource_time = sum([job.time for job in resource_jobs])
        if resource_time < min_time:
            min_time = resource_time
            min_resource = resource
    return min_resource
```

def max_resource_utilization(jobs, resources):
    max_utilization = 0

        resource_utilization = sum([job.time for job in resource_jobs]) / len(resource_jobs)
