$$
\text{平均值} = \frac{\sum_{i=1}^{n}x_i}{n}
$$

$$
\text{和差} = \sum_{i=1}^{n}x_i - n \times \text{平均值}
$$

$$
\text{百分比} = \frac{x}{y} \times 100\%
$$

import pandas as pd
