class Customer:
    def __init__(self, name, phone):
        self.name = name
        self.phone = phone

class Order:
    def __init__(self, customer, product, quantity):
        self.customer = customer
        self.product = product
        self.quantity = quantity

def process_order(order):
    print(f"Processing order for {order.customer.name} from {order.product} with quantity {order.quantity}")
    time.sleep(2)
    print(f"Order processed for {order.customer.name}")

customer1 = Customer("John Doe", "1234567890")
order1 = Order(customer1, "Laptop", 1)
process_order(order1)
