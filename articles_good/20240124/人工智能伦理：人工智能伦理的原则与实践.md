```python
def encrypt_data(data):
    """
    对数据进行加密
    """

    return encrypted_data

def mask_data(data):
    """
    对数据进行掩码处理
    """
    masked_data = '*' * len(data)
    return masked_data

def save_data(data):
    """
    保存加密后的数据
    """
    with open('data.txt', 'w') as f:
        f.write(data)

def load_data():
    """
    加载加密后的数据
    """
    with open('data.txt', 'r') as f:
        data = f.read()
    return data

if __name__ == '__main__':
    data = 'John Doe'
    encrypted_data = encrypt_data(data)
    masked_data = mask_data(data)
    save_data(encrypted_data)
    loaded_data = load_data()
    print(loaded_data)
```
