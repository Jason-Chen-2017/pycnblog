```python
import math

def calculate_power(P_t, G, lambda_, d):
    return (P_t * G ** 2 * lambda_ ** 2) / (4 * math.pi ** 2 * d ** 2)

P_t = 100  # 发射功率，单位为W
G = 10  # 发射天线增益，单位为无单位
lambda_ = 1  # 波长，单位为m
d = 10  # 距离，单位为m

power = calculate_power(P_t, G, lambda_, d)
print("接收功率：", power, "W")
```
