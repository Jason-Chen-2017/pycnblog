```python
import json

def load_events():
    with open('events.json', 'r') as f:
        events_str = f.readlines()
    events = [json.loads(event) for event in events_str]
    return events

def analyze_events(events):
    event_count = len(events)
    event_types = {}
    for event in events:
        event_type = event['type']
        if event_type not in event_types:
            event_types[event_type] = 0
        event_types[event_type] += 1
    return event_count, event_types

events = load_events()
event_count, event_types = analyze_events(events)
print(event_count)
print(event_types)
```
