data = {

}
df = pd.DataFrame(data)

# 创建数据立方体
dimensions = ['时间', '地理位置', '产品']
hierarchies = [['2021-01-01', '2021-01-02', '2021-01-03'], ['北京', '上海', '广州'], ['产品A', '产品B', '产品C']]
