import numpy as np
from scipy.optimize import minimize

# 环保材料性能数据
data = np.array([[100, 50, 80], [120, 60, 90], [110, 70, 85]])

# 性能指标
performance_indicators = ['strength', 'durability', 'environmental_performance']

# 目标函数
def objective_function(x):
    x = np.array(x)
    return np.sum([x[i] * data[:, i] for i in range(3)])

# 约束条件
def constraint_function(x):
    x = np.array(x)
    return np.sum([x[i]**2 for i in range(3)]) - 1000

# 优化参数
initial_guess = [1, 1, 1]
bounds = [(0, 100), (0, 100), (0, 100)]

# 优化设计参数
result = minimize(objective_function, initial_guess, bounds=bounds, constraints=constraint_function)

# 输出结果
print('Optimized design parameters:', result.x)
print('Optimized performance:', objective_function(result.x))
```
