### 4.1 独立成件分析具体代码实例

```python
import scipy.optimize as opt

# 混合信号
x = np.array([[1, 2], [3, 4]])

# 独立成件分析算法
def icap(x):
    s = opt.ifc(x)
    return s

# 独立成件分解
s = icap(x)

# 独立成件恢复
W = np.array([[1, -1], [-2, 1]])
y = np.dot(W, s)

print("独立成件分解: ", s)
print("独立成件恢复: ", y)
```
