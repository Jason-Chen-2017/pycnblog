from Crypto.PublicKey import RSA
from Crypto.Signature import pkcs1_15
from Crypto.Hash import SHA256
import numpy as np
import requests

# 数字身份认证
def digital_id():
    # 生成密钥对
    key = RSA.generate(2048)
    private_key = key.export_key()
    public_key = key.publickey().export_key()

    # 用户登录请求
    login_request = {
        "username": "user123",
        "password": "password456"
    }

    # 对登录请求进行签名
    message = str(login_request).encode()
    signature = pkcs1_15.new(key).sign(SHA256.new(message))

    # 验证签名
    public_key_bytes = public_key.decode()
    public_key = RSA.import_key(public_key_bytes)
    verified = pkcs1_15.new(public_key).verify(SHA256.new(message), signature)

    return verified

# 数据隐私保护
def data_privacy(file_path, epsilon):
    # 读取文件
    with open(file_path, 'rb') as file:
        data = file.read()

    # 生成高斯噪声
    mu = 0
    sigma = 1 / np.sqrt(2 * np.pi * epsilon)
    noise = np.random.normal(mu, sigma, data.shape)

    # 加噪后的数据
    noised_data = data + noise

    return noised_data

# 测试代码
if __name__ == "__main__":
