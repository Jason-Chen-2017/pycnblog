### 5.2 源代码详细实现

```python
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier

def calculate_demand(row):
    # 根据活动频率、兴趣爱好和社交需求计算需求值
    activity = row['activity']
    hobby = row['hobby']
    social = row['social']
    return activity * hobby * social

# 读取数据
data = pd.read_csv('data.csv')

# 数据清洗
data = data[data['activity'] != '']

# 特征提取
data['demand'] = data.apply(lambda x: calculate_demand(x), axis=1)

# 划分训练集和测试集
X_train, X_test, y_train, y_test = train_test_split(data[['activity', 'hobby', 'social']], data['demand'], test_size=0.2, random_state=42)

# 训练模型
model = RandomForestClassifier()
model.fit(X_train, y_train)

# 预测
predictions = model.predict(X_test)

# 评估模型
accuracy = model.score(X_test, y_test)
print(f"模型准确率：{accuracy}")
```
