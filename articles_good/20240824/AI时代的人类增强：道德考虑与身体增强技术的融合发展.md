import numpy as np
import pandas as pd
from scipy.optimize import linear_sum_assignment

def edit_distance(s1, s2):
    """
    计算两个字符串的编辑距离。
    """
    m, n = len(s1), len(s2)
    dp = np.zeros((m+1, n+1))

    for i in range(m+1):
        for j in range(n+1):
            if i == 0:
                dp[i][j] = j
            elif j == 0:
                dp[i][j] = i
            elif s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])

    return dp[m][n]

# 示例

s2 = "AGTGC"
distance = edit_distance(s1, s2)
print(f"编辑距离：{distance}")

```python
import numpy as np
import pandas as pd
from scipy.optimize import linear_sum_assignment

def edit_distance(s1, s2):
    """
    计算两个字符串的编辑距离。
    """
    m, n = len(s1), len(s2)
    dp = np.zeros((m+1, n+1))

    for i in range(m+1):
        for j in range(n+1):
            if i == 0:
                dp[i][j] = j
            elif j == 0:
                dp[i][j] = i
            elif s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])

    return dp[m][n]

# 示例
s1 = "AGTC"
s2 = "AGTGC"
distance = edit_distance(s1, s2)
print(f"编辑距离：{distance}")
```

2. **示例**：计算两个示例字符串的编辑距离，并打印结果。
