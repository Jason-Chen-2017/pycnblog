from qiskit import QuantumCircuit, Aer, execute
from qiskit.visualization import plot_bloch_multivector

# 创建量子电路
qc = QuantumCircuit(2)

# 初始化量子系统A和B处于量子纠缠态
qc.h(0)

# 测量量子系统A
