```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# 用户画像构建函数
def build_user_profile(data):
    # 年龄分布
    age_distribution = data['age'].value_counts(normalize=True)
    
    # 性别比例
    gender_ratio = data['gender'].value_counts(normalize=True)
    
    # 职业分布
    job_distribution = data['job'].value_counts(normalize=True)
    
    # 兴趣偏好
    interest_preferences = data['interest'].value_counts(normalize=True)
    
    # 地理位置分布
    location_distribution = data['location'].value_counts(normalize=True)
    
    # 绘制用户画像
    fig, axes = plt.subplots(2, 2, figsize=(10, 8))
    axes[0, 0].bar(age_distribution.index, age_distribution.values)
    axes[0, 0].set_title('Age Distribution')
    
    axes[0, 1].bar(gender_ratio.index, gender_ratio.values)
    axes[0, 1].set_title('Gender Ratio')
    
    axes[1, 0].bar(job_distribution.index, job_distribution.values)
    axes[1, 0].set_title('Job Distribution')
    
    axes[1, 1].bar(interest_preferences.index, interest_preferences.values)
    axes[1, 1].set_title('Interest Preferences')
    
    plt.tight_layout()
    plt.show()
    
    # 返回用户画像数据
    return {
        'age_distribution': age_distribution,
        'gender_ratio': gender_ratio,
        'job_distribution': job_distribution,
        'interest_preferences': interest_preferences,
        'location_distribution': location_distribution
    }

# 市场定位函数
def market_positioning(target_market, competitive_analysis, user_profile):
    # 根据用户画像和市场分析确定市场定位
    positioning = f"{target_market} - 专业性/实战性 - {user_profile['age_distribution'].index[0]}-45岁，IT/金融/互联网行业，技术趋势/编程语言/职业发展兴趣偏好"
    return positioning

# 示例数据
data = {
    'age': [25, 30, 35, 40, 45, 50],
    'gender': ['M', 'F', 'M', 'F', 'M', 'F'],
    'job': ['IT', 'Finance', 'Internet', 'IT', 'Finance', 'Internet'],
    'interest': ['Tech Trends', 'Programming', 'Career Development', 'Tech Trends', 'Programming', 'Career Development'],
    'location': ['Beijing', 'Shanghai', 'Beijing', 'Shanghai', 'Guangzhou', 'Shanghai']
}

# 构建用户画像
user_profile = build_user_profile(pd.DataFrame(data))

# 市场定位
positioning = market_positioning('IT领域', '专业性/实战性竞争优势', user_profile)
print("市场定位：", positioning)
```

   - 通过`value_counts(normalize=True)`统计各特征的分布情况，并绘制柱状图以可视化用户画像。

2. **市场定位**：
