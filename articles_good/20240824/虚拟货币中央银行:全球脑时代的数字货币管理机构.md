```solidity
pragma solidity ^0.8.0;

contract VirtualCurrencyCentralBank {
    mapping(address => uint256) public balances;
    uint256 public totalSupply;
    uint256 public constant INCREMENT = 1000; // 每期新增发行量

    event Issued(address beneficiary, uint256 amount);

    constructor() public {
        totalSupply = 10000; // 初始总供应量
        balances[msg.sender] = totalSupply;
    }

    function issue() public {
        require(msg.sender == owner, "只有管理员可以发行虚拟货币");
        balances[owner] += INCREMENT;
        totalSupply += INCREMENT;
        emit Issued(owner, INCREMENT);
    }

    function transfer(address recipient, uint256 amount) public {
        require(balances[msg.sender] >= amount, "余额不足");
        balances[msg.sender] -= amount;
        balances[recipient] += amount;
    }
}
```

这个智能合约定义了虚拟货币中央银行的发行、转账等功能。其中，`issue`函数用于发行虚拟货币，`transfer`函数用于虚拟货币的转账。
