## 4.具体代码实例和详细解释说明

### 4.1 数据流处理的具体代码实例

```python
import numpy as np
import pandas as pd

# 读取数据流
def read(source):
    data = []
    while True:
        chunk = source.read(1024)
        if not chunk:
            break
        data.append(chunk)
    return np.concatenate(data)

# 处理数据流
def process(data):
    result = []
    for d in data:
        if d > 100:
            result.append(d)
    return result

# 存储数据流
def store(data, filename):
    with open(filename, 'wb') as f:
        f.write(data)

# 输出数据流
def output(data):
    print(data)

# 主函数
def main():
    source = open('data.bin', 'rb')
    data = read(source)
    result = process(data)
    store(result, 'result.bin')
    output(result)

if __name__ == '__main__':
    main()
```
