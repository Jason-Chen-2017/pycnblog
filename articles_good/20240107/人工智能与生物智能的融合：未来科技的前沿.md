def seq_alignment(seq1, seq2):
    len1 = len(seq1)
    len2 = len(seq2)
    matrix = [[0] * (len2 + 1) for _ in range(len1 + 1)]
    for i in range(1, len1 + 1):
        for j in range(1, len2 + 1):
            match = matrix[i - 1][j - 1] + (seq1[i - 1] == seq2[j - 1])
            insert = matrix[i - 1][j]
            delete = matrix[i][j - 1]
            matrix[i][j] = max(match, insert, delete)
    return matrix
