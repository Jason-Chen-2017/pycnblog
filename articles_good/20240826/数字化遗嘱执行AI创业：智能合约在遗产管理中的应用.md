```solidity
pragma solidity ^0.8.0;

contract Will {
    address public executor;
    mapping(address => bool) public heirs;

    constructor() {
        executor = msg.sender;
    }

    function addHeir(address heir) public {
        require(msg.sender == executor, "Only executor can add heirs");
        heirs[heir] = true;
    }

    function executeWill() public {
        require(msg.sender == executor, "Only executor can execute will");
        for (heir in heirs) {
            if (heirs[heir]) {
                payable(heir).transfer(1 ether); // Transfer 1 ether to the heir
            }
        }
    }
}
```
