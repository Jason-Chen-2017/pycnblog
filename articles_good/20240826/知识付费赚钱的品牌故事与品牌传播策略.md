### 5.2 源代码详细实现

以下是用于计算用户满意度和品牌影响力的Python代码：

```python
import pandas as pd
import numpy as np

# 用户满意度计算函数
def calculate_user_satisfaction(content_quality, service_experience, brand_value):
    w1, w2, w3 = 0.3, 0.2, 0.5
    s = w1 * content_quality + w2 * service_experience + w3 * brand_value
    return s

# 品牌影响力计算函数
def calculate_brand_influence(brand_knowledge, brand_prestige, user_loyalty):
    x1, x2, x3 = 0.4, 0.3, 0.3
    i = x1 * brand_knowledge + x2 * brand_prestige + x3 * user_loyalty
    return i

# 示例数据
data = {
    'content_quality': [90, 85, 80],
    'service_experience': [95, 90, 85],
    'brand_value': [80, 75, 70],
    'brand_knowledge': [70, 65, 60],
    'brand_prestige': [85, 80, 75],
    'user_loyalty': [90, 85, 80]
}

df = pd.DataFrame(data)

# 计算用户满意度和品牌影响力
df['user_satisfaction'] = df.apply(lambda row: calculate_user_satisfaction(row['content_quality'], row['service_experience'], row['brand_value']), axis=1)
df['brand_influence'] = df.apply(lambda row: calculate_brand_influence(row['brand_knowledge'], row['brand_prestige'], row['user_loyalty']), axis=1)

print(df)
```
