```python
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 假设我们有一个用户行为数据集
user_data = {
    'user_id': [1, 2, 3],
    'content_id': [101, 202, 303],
    'read_time': [300, 150, 200],
    'like_count': [15, 5, 10],
    'comment_count': [3, 1, 2]
}

# 创建DataFrame
df = pd.DataFrame(user_data)

# 计算平均阅读时长、平均点赞数和平均评论数
user_behavior_stats = df.groupby('user_id').mean()

# 创建TF-IDF向量器
tfidf_vectorizer = TfidfVectorizer()

# 使用TF-IDF向量器转换内容
tfidf_matrix = tfidf_vectorizer.fit_transform(df['content_id'])

# 计算用户行为特征向量
user_behavior_vector = tfidf_matrix[:, user_behavior_stats.index]

# 计算用户之间的相似度
similarity_matrix = cosine_similarity(user_behavior_vector)

# 推荐相似的用户喜欢的内容
content_recommendations = df[df['user_id'].isin(user_behavior_stats.index)]
content_recommendations['similarity_score'] = similarity_matrix[0]

# 对相似度分数进行降序排序，推荐相似度最高的内容
recommended_contents = content_recommendations.sort_values(by='similarity_score', ascending=False)

print(recommended_contents)
```
