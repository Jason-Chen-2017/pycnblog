import numpy as np
import matplotlib.pyplot as plt

def f(x, y, z):
    a = 1
    b = 1
    c = 1
    return a * x**2 + b * y**2 + c * z**2

def simulate(x0, y0, z0, t_max):
    x = x0
    y = y0
    z = z0
    t = 0
    times = []
    values = []

    while t < t_max:
        t += 0.1
        dx = f(x, y, z) * 0.1
        x += dx
        y += 0.1
        z += 0.1
        times.append(t)
        values.append(x)

    return times, values

x0 = 0
y0 = 1
z0 = 1
t_max = 10

times, values = simulate(x0, y0, z0, t_max)

plt.plot(times, values)
plt.xlabel('Time (s)')
plt.ylabel('X')
plt.title('Inspiration Simulation')
plt.show()
