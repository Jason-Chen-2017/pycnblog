class PowerSet:
    def __init__(self, set):
        self.set = set
        self.power_set = self._generate_power_set()

    def _generate_power_set(self):
        power_set = [Set() for _ in range(2 ** len(self.set))]
        for i, element in enumerate(self.set):
            for j, subset in enumerate(power_set):
                if j & (1 << i):
                    new_subset = Set()
                    new_subset.elements.extend(subset.elements)
                    new_subset.add(element)
                    power_set.append(new_subset)
        return [subset for subset in power_set if subset.elements]
