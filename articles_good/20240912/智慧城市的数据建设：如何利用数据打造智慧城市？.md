def create_connection():

        conn = sqlite3.connect(':memory:')  # 创建一个内存数据库连接
        print(f"连接成功：{conn}")
    except Error as e:
        print(e)
    return conn

def create_table(conn):
    try:
        sql_create_person_table = """
        CREATE TABLE IF NOT EXISTS person (
            id INTEGER PRIMARY KEY,
            name TEXT NOT NULL,
            age INTEGER,
            address TEXT
        );
        """
        cursor = conn.cursor()
        cursor.execute(sql_create_person_table)
        print("Table created successfully")
    except Error as e:
        print(e)

def insert_person(conn, person):
    try:
        sql_insert_person = """
        INSERT INTO person(name, age, address)
        VALUES(?,?,?)
        """
        cursor = conn.cursor()
        cursor.execute(sql_insert_person, person)
        conn.commit()
        print("Person inserted successfully")
    except Error as e:
        print(e)
