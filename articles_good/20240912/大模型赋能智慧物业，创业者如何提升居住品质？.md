2. 算法实现：使用队列结构实现投诉请求的排队和处理，使用哈希表存储用户信息和投诉请求，以便快速查找和处理。

3. 示例代码（Python）：

```python
class User:
    def __init__(self, name, complaints):
        self.name = name
        self.complaints = complaints

class Complaint:
    def __init__(self, content, status, handler):
        self.content = content
        self.status = status
        self.handler = handler

class ComplaintSystem:
    def __init__(self):
        self.users = {}
        self.complaints_queue = []

    def add_complaint(self, user, complaint):
        if user not in self.users:
            self.users[user] = []
        self.users[user].append(complaint)
        self.complaints_queue.append(complaint)

    def process_complaint(self, complaint):
        complaint.status = "处理中"
        complaint.handler.handle(complaint)

    def get_user_complaints(self, user):
        return self.users[user]

system = ComplaintSystem()
user1 = User("张三", [])
complaint1 = Complaint("电梯故障", "待处理", "李四")
system.add_complaint(user1, complaint1)
system.process_complaint(complaint1)
print(system.get_user_complaints(user1))
```

contract Payment {
    struct PaymentRecord {
        uint256 amount;
        uint256 timestamp;
        bool status;
    }

    mapping(address => PaymentRecord[]) public payments;

    function makePayment() public payable {
        payments[msg.sender].push(PaymentRecord(msg.value, block.timestamp, false));
    }

    function confirmPayment(uint256 index) public {
        require(index < payments[msg.sender].length, "Invalid index");
        PaymentRecord storage payment = payments[msg.sender][index];
        require(!payment.status, "Payment already confirmed");
        payment.status = true;
    }
}

pragma solidity ^0.8.0;

contract Voting {
    struct Vote {
        address voter;
        bool hasVoted;
        uint256 vote;
    }

    mapping(uint256 => mapping(address => Vote)) public votes;

    function vote(uint256 questionId, uint256 choice) public {
        require(!votes[questionId][msg.sender].hasVoted, "Already voted");
        votes[questionId][msg.sender] = Vote(msg.sender, true, choice);
    }

    function getVoteCount(uint256 questionId) public view returns (uint256) {
        uint256 count = 0;
        for (uint256 i = 0; i < votes[questionId].length; i++) {
            if (votes[questionId][i].hasVoted) {
                count++;
            }
        }
        return count;
    }
}
