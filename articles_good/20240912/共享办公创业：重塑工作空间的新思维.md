import heapq

def schedule_appointments(booking_requests):
    # 创建一个优先级队列
    priority_queue = []
    for request in booking_requests:
        heapq.heappush(priority_queue, (-request.timestamp, request))
    # 处理预约请求
    processed_requests = []
    while priority_queue:
        _, request = heapq.heappop(priority_queue)
        processed_requests.append(request)
    return processed_requests
