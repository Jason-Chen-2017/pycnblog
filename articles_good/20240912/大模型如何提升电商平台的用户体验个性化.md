```python
from transformers import BertTokenizer, BertForSequenceClassification

# 自然语言处理与意图识别

model = BertForSequenceClassification.from_pretrained('bert-base-chinese')

user_question = "我想要购买一件羽绒服，有没有推荐？"
input_ids = tokenizer.encode(user_question, add_special_tokens=True, return_tensors='pt')

# 提取关键信息与意图识别
with torch.no_grad():
    outputs = model(input_ids)
    logits = outputs.logits
    predicted_intent = logits.argmax(-1).item()
print("Predicted Intent:", predicted_intent)

# 回答生成
if predicted_intent == 1:
    response = "我们为您推荐以下羽绒服：1. XXX品牌羽绒服，2. YYY品牌羽绒服，您可以根据自己的需求选择。"
else:
    response = "对不起，我无法理解您的问题，请您重新描述一下您的需求。"

print("Response:", response)
```
