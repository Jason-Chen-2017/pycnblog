import Course from './Course';

class CourseList extends Component {
  constructor(props) {
    super(props);
    this.state = {
      courses: [],
      isLoading: true
    };
  }

  componentDidMount() {
    fetch('/api/courses')
      .then(response => response.json())
      .then(data => {

      });

    if (isLoading) {

        ))}
      </div>
    );
  }
}

export default CourseList;
