```python
import numpy as np
import matplotlib.pyplot as plt

# 定义仿真模型的参数
params = {
    'T0': 20,  # 初始温度
    'T1': 30,  # 目标温度
    'L': 100,  # 温度变化的长度
    'dt': 1,   # 时间步长
}

# 初始化温度数组
T = np.full(params['L'], params['T0'])

# 定义温度的变化方程
def update_temperature(T, dt, params):
    T_new = np.zeros(params['L'])
    for i in range(params['L']):
        T_new[i] = params['T0'] + (params['T1'] - params['T0']) * (i / params['L'])
    T = T_new
    return T

# 更新温度数组
for _ in range(params['L']):
    T = update_temperature(T, params['dt'], params)

# 绘制温度变化图
plt.plot(T)
plt.xlabel('位置')
plt.ylabel('温度')
plt.title('温度变化')
plt.show()
```
