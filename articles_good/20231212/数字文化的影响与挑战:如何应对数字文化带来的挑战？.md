## 4.1 数字信息处理代码实例

在这个代码实例中，我们将实现一个简单的数字信息压缩算法，即 Huffman 编码。

```python
from collections import Counter, heapq

def huffman_encoding(data):
    # 计算字符出现的频率
    freq = Counter(data)

    # 构建优先级队列
    heap = [[weight, [char, [char, 0]]] for char, weight in freq.items()]
    heapq.heapify(heap)

    # 构建Huffman树
    while len(heap) > 1:
        lo = heapq.heappop(heap)
        hi = heapq.heappop(heap)
        for pair in lo[1:]:
            pair[1] += [char, lo[0]]
        for pair in hi[1:]:
            pair[1] += [char, hi[0]]
        heapq.heappush(heap, [lo[0] + hi[0]] + lo[1:] + hi[1:])

    # 构建Huffman编码
    codes = dict()
    def build_codes(branch, prefix):
        if len(branch) == 1:
            codes[branch[0]] = prefix
        else:
            build_codes(branch[1], prefix + [0])
            build_codes(branch[2], prefix + [1])
    build_codes(heap[0], [])

    # 编码数据
    encoded_data = ''
    for char in data:
        encoded_data += codes[char]
    return encoded_data

data = 'hello world'
encoded_data = huffman_encoding(data)
print(encoded_data)
```

这个代码实例中，我们首先计算字符出现的频率，然后构建一个优先级队列，将字符和其出现频率一起放入队列中。接着，我们构建一个Huffman树，并计算每个字符的Huffman编码。最后，我们使用Huffman编码对数据进行编码。

```python
from io import BytesIO

def jpeg_compression(image_path):
    # 读取图像
    image = Image.open(image_path)

    compressed_image = image_bytes.read()
    return compressed_image

compressed_image = jpeg_compression(image_path)
```

```python
from pydub import AudioSegment
from pydub.playback import play

def mp3_compression(audio_path):
    # 读取音频
    audio = AudioSegment.from_mp3(audio_path)
    audio_bytes = audio.to_bytes()

    # 压缩音频
    compressed_audio = audio.compress(bitrate='128k')
    compressed_audio_bytes = compressed_audio.to_bytes()

    # 保存压缩音频
    with open('compressed_audio.mp3', 'wb') as f:
        f.write(compressed_audio_bytes)

audio_path = 'guitar.mp3'
mp3_compression(audio_path)
play('compressed_audio.mp3')
```

这个代码实例中，我们首先读取音频，然后使用MP3格式进行压缩。最后，我们得到压缩后的音频数据。
