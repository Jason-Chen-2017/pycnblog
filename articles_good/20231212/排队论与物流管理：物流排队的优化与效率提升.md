def m(lambda_, rho):

# 计算平均等待时间

    return 1 / (lambda_ - rho)

# 计算平均服务时间

    return 1 / mu

    return lambda_ / (lambda_ + rho)

# 参数设置

mu = 12  # 服务资源平均处理时间

# 计算结果
L = m(lambda_, rho)
W = W(lambda_, rho)
S = S(mu)
X = X(lambda_, rho)

# 输出结果
print("排队长度：", L)
print("平均等待时间：", W)
