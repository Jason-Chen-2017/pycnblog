def quantum_phase_estimation(qubit_num=3, precision=4):
    qc = QuantumCircuit(qubit_num, 1)
    qc.h(qubit_num-1)
    for i in range(precision):
        qc.cx(qubit_num-1, qubit_num-i-1)
