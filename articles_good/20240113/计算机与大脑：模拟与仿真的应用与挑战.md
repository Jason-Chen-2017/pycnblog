## 4.1 神经网络模型代码实例

```python
import numpy as np

# 构建神经网络
input_layer = np.array([[1, 0], [0, 1]])
hidden_layer = np.array([[0.5, 0.5], [0.5, -0.5]])
output_layer = np.array([[1, 0], [0, 1]])

# 初始化权重
weights_input_hidden = np.random.rand(2, 2)
weights_hidden_output = np.random.rand(2, 2)

# 训练神经网络
for _ in range(1000):
    # 前向传播
    hidden_layer = np.dot(input_layer, weights_input_hidden) + hidden_layer
    hidden_layer = np.tanh(hidden_layer)

    output_layer = np.dot(hidden_layer, weights_hidden_output) + output_layer
    output_layer = np.tanh(output_layer)

    # 反向传播
    error = output_layer - np.array([[1, 0], [0, 1]])
    output_delta = error * (1 - output_layer ** 2)

    hidden_error = output_layer.dot(weights_hidden_output.T) * (1 - hidden_layer ** 2)
    hidden_delta = hidden_error.dot(weights_input_hidden.T * output_delta)

    # 更新权重
    weights_hidden_output += hidden_layer.T.dot(output_delta)
    weights_input_hidden += input_layer.T.dot(hidden_delta)

# 测试神经网络
test_input = np.array([[0, 0], [1, 1]])
test_output = np.dot(test_input, weights_input_hidden) + hidden_layer
test_output = np.tanh(test_output)
test_output = np.dot(test_output, weights_hidden_output) + output_layer
test_output = np.tanh(test_output)
print(test_output)
```
