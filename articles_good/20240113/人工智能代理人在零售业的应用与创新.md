    tokens = word_tokenize(sentence)
    grammar = r"""
    NP: {<DT>?<JJ>*<NN>}
    """
    parser = nltk.RegexpParser(grammar)
    parsed_sentence = parser.parse(tokens)
    return parsed_sentence

# 语义分析
def semantic_analysis(sentence):
    tokens = word_tokenize(sentence)
    tagged_tokens = nltk.pos_tag(tokens)
