## 5.1 自动化法律咨询

以下是一个简单的自动化法律咨询的代码实例：

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

def get_similarity(question, corpus):
    vectorizer = TfidfVectorizer()
    question_vector = vectorizer.fit_transform([question])
    corpus_vector = vectorizer.fit_transform(corpus)
    similarity = cosine_similarity(question_vector, corpus_vector).flatten()
    return similarity

corpus = ["合同法规定，合同的一方违反合同条款，应承担违约责任。",
          "合同法规定，合同的一方违反合同条款，应承担违约责任。",
          "合同法规定，合同的一方违反合同条款，应承担违约责任。"]

question = "合同一方违反合同条款，应承担违约责任。"
similarity = get_similarity(question, corpus)
print(similarity)
```

```python
from transformers import pipeline

def generate_legal_document(content):
    model = pipeline("text-generation", model="gpt-2")
    generated_text = model(content, max_length=100, num_return_sequences=1)
    return generated_text[0]["generated_text"]

content = "合同法规定，合同的一方违反合同条款，应承担违约责任。"
generated_text = generate_legal_document(content)
print(generated_text)
```
