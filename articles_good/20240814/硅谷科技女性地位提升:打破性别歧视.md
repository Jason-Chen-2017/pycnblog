R_F = \frac{1}{N_F}\sum_{i=1}^{N_F}\sum_{j=1}^{m}p_{ij}

mean_s = np.mean(S.values())
var_s = np.var(S.values())

# 职业晋升率计算
P = data['promotion_rates'].to_dict()
R_F = (P.values().sum() / P_F).to_dict()
print('职业晋升率:', R_F)
