import matplotlib.pyplot as plt

# 随机传播模型
def random_propagation(x, p):
    y = []
    for xi in x:
        yi = np.random.choice(x, p=p)
        y.append(yi)
    return y

# 生成随机数据

import networkx as nx
import matplotlib.pyplot as plt

## 4.3 医疗资源分配模型实现

### 4.3.1 构建医疗资源分配模型

```python
import numpy as np
import matplotlib.pyplot as plt

# 构建医疗资源分配模型
def build_medical_resource_allocation_model(resources, demands, costs):
    G = nx.DiGraph()
    G.add_nodes_from(resources)
    G.add_nodes_from(demands)
    G.add_edges_from(zip(resources, demands))
    for r, d in zip(resources, demands):
        G.edges[(r, d)]['cost'] = costs[r][d]
    return G

# 生成随机数据
resources = np.random.rand(100)
demands = np.random.rand(100)
costs = np.random.rand(100, 100)

# 构建医疗资源分配模型
G = build_medical_resource_allocation_model(resources, demands, costs)

# 绘制图像
nx.draw(G, with_labels=True)
plt.xlabel('Resources')
plt.ylabel('Demands')
plt.title('Medical Resource Allocation Model')
plt.show()
```
