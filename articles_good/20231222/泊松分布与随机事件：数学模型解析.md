```python
import math

# 平均雨天次数
lambda = 1.5

# 计算某一天雨天的概率
def rain_probability(k):
    probability = math.exp(-lambda) * (lambda ** k) / math.factorial(k)
    return probability

# 计算某一天晴天的概率
def clear_probability(k):
    if k == 0:
        return 1
    else:
        return 1 - rain_probability(k)

# 输入某一天的雨天次数
k = int(input("请输入某一天的雨天次数："))

# 输出某一天雨天或晴天的概率
print("某一天雨天的概率：", rain_probability(k))
print("某一天晴天的概率：", clear_probability(k))
```
