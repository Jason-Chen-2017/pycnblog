```python
# 确定问题
def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

# 确定假设
assumption = "所有的偶数都是合数"

# 评估假设
if assumption == "所有的偶数都是合数":
    for n in range(2, 100):
        if is_prime(n):
            print(f"{n} 是素数")
        else:
            print(f"{n} 不是素数")

# 处理问题
def is_even(n):
    return n % 2 == 0

def is_odd(n):
    return not is_even(n)

def main():
    n = 10
    if is_even(n):
        print(f"{n} 是偶数")
    else:
        print(f"{n} 是奇数")

if __name__ == "__main__":
    main()
```

在这个代码实例中，我们首先确定了问题，即判断一个数是否为素数。然后，我们确定了原假设，即所有的偶数都是合数。接下来，我们评估了原假设的合理性，并发现原假设是不合理的，因为所有的偶数都不是素数。最后，我们处理了问题，并输出了结果。
