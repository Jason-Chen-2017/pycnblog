```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.decomposition import LatentDirichletAllocation

def analyze_text(texts, word_list):
    # 计算单词在文本数据中的出现频率和逆向文件（IDF）
    tfidf_vectorizer = TfidfVectorizer(vocabulary=word_list)
    tfidf_matrix = tfidf_vectorizer.fit_transform(texts)
    # 使用主题建模算法将文本数据划分为多个主题
    lda_model = LatentDirichletAllocation(n_components=5)
    lda_model.fit(tfidf_matrix)
    return lda_model
```

## 4.2 文本挖掘
### 4.2.1 文本分类
```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.svm import LinearSVC
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score

def text_classification(texts, labels):
    # 构建文本特征
    tfidf_vectorizer = TfidfVectorizer()
    tfidf_matrix = tfidf_vectorizer.fit_transform(texts)
    # 构建文本分类模型
    classifier = LinearSVC()
    # 划分训练集和测试集
    X_train, X_test, y_train, y_test = train_test_split(tfidf_matrix, labels, test_size=0.2, random_state=42)
    # 训练模型
    classifier.fit(X_train, y_train)
    # 预测结果
    y_pred = classifier.predict(X_test)
    # 评估模型性能
    accuracy = accuracy_score(y_test, y_pred)
    precision = precision_score(y_test, y_pred)
    recall = recall_score(y_test, y_pred)
    f1 = f1_score(y_test, y_pred)
    return classifier, accuracy, precision, recall, f1
```

### 4.2.2 文本聚类

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.cluster import KMeans
from sklearn.metrics import silhouette_score

def text_clustering(texts):
    # 构建文本特征
    tfidf_vectorizer = TfidfVectorizer()
    tfidf_matrix = tfidf_vectorizer.fit_transform(texts)
    # 构建文本聚类模型
    kmeans = KMeans(n_clusters=3)
    # 训练模型
    kmeans.fit(tfidf_matrix)
    # 预测结果
    labels = kmeans.labels_
    # 评估模型性能
    silhouette_avg = silhouette_score(tfidf_matrix, labels)
    return kmeans, silhouette_avg
```
