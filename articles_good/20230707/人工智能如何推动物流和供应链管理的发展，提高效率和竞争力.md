    def add_item(self, item):
        for stock in self.inventory:
            if stock.name == item.name:
                stock.quantity += 1
                return
        self.inventory.append(Stock(item.name, item.quantity, item.cost))

    def remove_item(self, item):
        for stock in self.inventory:
            if stock.name == item.name:
                stock.quantity -= 1
                return
        self.inventory.remove(Stock(item.name, item.quantity, item.cost))

    def search_item(self, item):
        for stock in self.inventory:
            if stock.name == item.name:
                return stock
        return None
