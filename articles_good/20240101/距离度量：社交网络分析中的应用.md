def breadth_first_search(graph, start):
    visited = set()
    queue = deque([start])
    while queue:
        current = queue.popleft()
        if current not in visited:
            visited.add(current)
            for neighbor in graph[current]:
                if neighbor not in visited:
                    queue.append(neighbor)
