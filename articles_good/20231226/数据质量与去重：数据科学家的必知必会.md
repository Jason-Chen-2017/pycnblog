### 4.2.1 基于哈希表的方法

```python
def remove_duplicates_hash(data):
    hash_table = {}
    result = []
    for record in data:
        key = tuple(record)
        if key not in hash_table:
            hash_table[key] = record
            result.append(record)
    return result

data = [(1, 2), (2, 3), (1, 2), (3, 4)]
data_unique = remove_duplicates_hash(data)
print(data_unique)
```

### 4.2.2 基于排序的方法

```python
def remove_duplicates_sort(data):
    data.sort()
    result = []
    for i in range(len(data)):
        if i == 0 or data[i] != data[i - 1]:
            result.append(data[i])
    return result

data = [(1, 2), (2, 3), (1, 2), (3, 4)]
data_unique = remove_duplicates_sort(data)
print(data_unique)
```

    for i in range(len(data)):
        if i == 0 or data[i] != data[i - 1]:
            result.append(data[i])
    return result

data_unique = remove_duplicates_sort(data)
