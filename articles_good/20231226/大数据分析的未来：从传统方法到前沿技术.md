G.add_edge(2, 3)

visited = []
def dfs(G, node, visited):
    visited.append(node)
    for neighbor in G.neighbors(node):
        if neighbor not in visited:
            dfs(G, neighbor, visited)

dfs(G, 1, visited)
print(visited)
