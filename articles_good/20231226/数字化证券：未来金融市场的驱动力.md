from web3 import Web3

class DigitalAsset(object):
    def __init__(self, web3, address, balance):
        self.web3 = web3
        self.address = address
        self.balance = balance

    def transfer(self, recipient, amount):
        transaction = {
            'from': self.address,
            'to': recipient,
            'value': amount,
        }
        signed_transaction = self.web3.eth.account.sign_transaction(transaction, self.private_key)
        raw_transaction = signed_transaction.rawTransaction
        transaction_hash = self.web3.eth.send_raw_transaction(raw_transaction)
        transaction_receipt = self.web3.eth.waitForTransactionReceipt(transaction_hash)
        return transaction_receipt
