```python
import time

class SmartLock:
    def __init__(self):
        self.locked = True

    def control_smart_lock(self, action):
        if action == 'open':
            self.locked = False
        elif action == 'close':
            self.locked = True
        return self.locked

if __name__ == '__main__':
    lock = SmartLock()
    while True:
        action = input('请输入门锁操作（open/close）:')
        locked = lock.control_smart_lock(action)
        print('门锁状态:', '锁定' if locked else '解锁')
        time.sleep(1)
```
