```python
import random

def collect_data():
```

```python
import pandas as pd

def process_data(data):
    df = pd.DataFrame(data)
    df['temperature'] = df['temperature'].dropna()
    df['pressure'] = df['pressure'].dropna()
    df['flow_rate'] = df['flow_rate'].dropna()
    df['temperature'] = df['temperature'].astype('float64')
    df['pressure'] = df['pressure'].astype('float64')
    df['flow_rate'] = df['flow_rate'].astype('float64')
    df['temperature'] = df['temperature'] * 10
    df['pressure'] = df['pressure'] * 100
    df['flow_rate'] = df['flow_rate'] * 1000
    return df
```
