import numpy as np

def spatial_weight_matrix(distance_matrix):
    """
    计算公共自行车数据的空间权重矩阵
    """
    # 计算空间距离
    distance_matrix = np.sqrt(np.sum((distance_matrix - distance_matrix.mean()) ** 2, axis=1))
