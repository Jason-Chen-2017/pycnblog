from tensorflow.keras.layers import Input, Dense, Reshape, Flatten

    z = Input(shape=(z_dim,))
    x = Dense(128, activation='relu')(z)
    x = Dense(28 * 28 * 1, activation='sigmoid')(x)
    x = Reshape((28, 28, 1))(x)
    model = Model(z, x)
    return model

    img = Input(shape=img_shape)

    validity = Dense(1, activation='sigmoid')(x)

    return model
