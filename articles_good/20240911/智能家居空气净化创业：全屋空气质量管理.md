```python
from typing import List, Tuple

def findPollutionSource(aqi_data: List[int]) -> Tuple[int, int]:
    max_aqi = max(aqi_data)
    max_index = aqi_data.index(max_aqi) + 1
    return (max_index, max_aqi)
```

**解析：** 这个函数首先使用 `max` 函数找到 AQI 最高的房间，然后使用 `index` 方法找到其索引。由于房间编号从 1 开始，所以需要将索引加 1。最后，函数返回一个元组，包含房间编号和 AQI 值。

#### 4. 室内空气污染扩散模拟

**题目：** 请设计一个算法，用于模拟室内空气污染的扩散过程。给定一个二维房间网格，每个单元格表示一个房间，其值为当前单元格的空气质量指数 (AQI)。请编写一个函数 `simulateAirPollution(aqi_grid: List[List[int]], steps: int) -> List[List[int]]`，模拟在给定步骤后，每个单元格的 AQI 变化情况。

**答案：**

```python
def simulateAirPollution(aqi_grid: List[List[int]], steps: int) -> List[List[int]]:
    rows, cols = len(aqi_grid), len(aqi_grid[0])
    for _ in range(steps):
        new_aqi_grid = [[0] * cols for _ in range(rows)]
        for i in range(rows):
            for j in range(cols):
                new_aqi = aqi_grid[i][j]
                # 考虑周围八个单元格的 AQI，取平均值
                for di in range(-1, 2):
                    for dj in range(-1, 2):
                        ni, nj = i + di, j + dj
                        if 0 <= ni < rows and 0 <= nj < cols:
                            new_aqi += aqi_grid[ni][nj]
                new_aqi /= 9
                new_aqi_grid[i][j] = new_aqi
        aqi_grid = new_aqi_grid
    return aqi_grid
```

**解析：** 这个函数使用了一个简单的扩散模型，将当前单元格的 AQI 与周围八个单元格的 AQI 进行平均。在给定步骤后，函数返回新的 AQI 网格。

```python
class AirQualityControl:
    def __init__(self, aqi: int, user_demand: str):
        self.aqi = aqi
        self.user_demand = user_demand
        self.control_settings = {
            "mode": "自动",
            "wind_speed": 2,
            "timer_on": None,
            "timer_off": None,
        }

    def adjustControlSettings(self):
        if self.user_demand == "健康模式":
            if self.aqi > 100:
                self.control_settings["mode"] = "自动"
                self.control_settings["wind_speed"] = 3

            else:
                self.control_settings["mode"] = "自动"
```

```python
def generateMaintenancePlan(last_clean_date: datetime, last_check_date: datetime) -> List[str]:
```
