   # 计算商品间的相似度矩阵
   similarity_matrix = cosine_similarity(tfidf_matrix)

   # 为用户推荐相关的商品
   def content_based_recommender(item_id, similarity_matrix, k=5):

       item_index = item_id - 1

       top_k_indices = np.argsort(similarity_matrix[item_index])[1:k+1]
