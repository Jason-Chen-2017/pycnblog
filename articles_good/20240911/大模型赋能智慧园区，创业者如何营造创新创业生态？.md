def predict_highest_traffic(heat_map):
    max_traffic = 0
    highest_traffic_area = None
    
    for area, traffic in heat_map.items():
        if traffic > max_traffic:
            max_traffic = traffic
            highest_traffic_area = area

from sklearn.metrics.pairwise import cosine_similarity

# 加载物业管理数据
