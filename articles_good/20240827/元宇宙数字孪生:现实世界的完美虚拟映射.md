import numpy as np
import matplotlib.pyplot as plt

# 实体建模
def create_entity_model(n_points):
    points = np.random.rand(n_points, 2)
    return points

# 数据建模
def create_data_model(n_points):
    data = np.random.rand(n_points)
    return data

# 交互建模
def create_interaction_model(n_entities):
    edges = np.random.choice(n_entities, size=n_entities, replace=False)
    return edges

# 主函数
def main():
    # 创建实体模型
    n_points = 10
    points = create_entity_model(n_points)

    # 创建数据模型
    n_points = 10
    data = create_data_model(n_points)

    # 创建交互模型
    n_entities = 10
    edges = create_interaction_model(n_entities)

    # 绘制实体模型
    plt.scatter(points[:, 0], points[:, 1], label='Entity Model')

    # 绘制数据模型
    plt.plot(data, label='Data Model')

    # 绘制交互模型
    for i in range(len(edges)):
        plt.plot([points[edges[i], 0], points[edges[i+1], 0]], [points[edges[i], 1], points[edges[i+1], 1]], 'r--', label='Interaction Model')

    # 显示图形
    plt.legend()
    plt.show()

# 调用主函数
if __name__ == '__main__':
    main()
