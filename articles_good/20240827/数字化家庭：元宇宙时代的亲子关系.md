### 5.3 代码解读与分析

在代码中，我们定义了两个类：`Parent`和`Child`。`Parent`类实现了Q-learning算法，`Child`类实现了随机游走算法。通过这些算法，家长和孩子在虚拟世界中互动。

```csharp
// Parent类
public class Parent : MonoBehaviour
{
    // Q-learning算法参数
    private float learningRate = 0.1f;
    private float discountFactor = 0.9f;
    private float[] qValues;

    // 初始化Q值矩阵
    private void InitializeQValues()
    {
        int numActions = 4; // 4个可能的行为
        qValues = new float[numActions];
        for (int i = 0; i < numActions; i++)
        {
            qValues[i] = 0;
        }
    }

    // 更新Q值
    private void UpdateQValues(int action, float reward)
    {
        int currentQValue = (int)qValues[action];
        int nextQValue = GetMaxQValue();
        qValues[action] = currentQValue + learningRate * (reward + discountFactor * nextQValue - currentQValue);
    }

    // 获取最大Q值
    private int GetMaxQValue()
    {
        int maxQValue = int.MinValue;
        for (int i = 0; i < qValues.Length; i++)
        {
            if (qValues[i] > maxQValue)
            {
                maxQValue = qValues[i];
            }
        }
        return maxQValue;
    }

    // 家长行为
    private void ParentBehavior()
    {
        // 获取当前状态
        int state = GetState();

        // 根据Q值选择行为
        int action = GetBestAction(state);

        // 执行行为
        switch (action)
        {
            case 0:
                // 与孩子互动
                InteractWithChild();
                break;
            case 1:
                // 教育孩子
                EducateChild();
                break;
            case 2:
                // 给孩子讲故事
                TellStory();
                break;
            case 3:
                // 陪孩子玩耍
                PlayWithChild();
                break;
        }
    }

    // 初始化
    private void Start()
    {
        InitializeQValues();
    }

    // 更新
    private void Update()
    {
        if (IsGoalState())
        {
            // 更新Q值
            UpdateQValues(action, reward);
        }
        else
        {
            // 更新行为
            ParentBehavior();
        }
    }
}

// Child类
public class Child : MonoBehaviour
{
    // 随机游走算法参数
    private float moveSpeed = 5.0f;

    // 孩子行为
    private void ChildBehavior()
    {
        // 随机移动
        float moveX = Random.Range(-1.0f, 1.0f);
        float moveZ = Random.Range(-1.0f, 1.0f);
        transform.Translate(new Vector3(moveX * moveSpeed * Time.deltaTime, 0, moveZ * moveSpeed * Time.deltaTime));
    }

    // 初始化
    private void Start()
    {
        // 开始孩子行为
        ChildBehavior();
    }

    // 更新
    private void Update()
    {
        // 更新孩子行为
        ChildBehavior();
    }
}
```
