noisy_gate = qiskit.PauliGate('X', 2)  # 噪声干扰概率为0.1
qcs.append(noisy_gate, [0, 1])

# 量子纠错
qcs.h(0)
qcs.cx(0, 1)
qcs.measure_all()

# 执行量子电路
backend = qiskit.BasicAer.get_backend("qasm_simulator")
results = qcs.run(backend, shots=1024)

# 解码量子密钥
key = results.get_counts(qcs)
print("密钥：", key)
