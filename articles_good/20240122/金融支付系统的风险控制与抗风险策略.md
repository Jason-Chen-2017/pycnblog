```python
from pomegranate import BayesianNetwork

# 加载数据
data = pd.read_csv('transaction.csv')

y = data['is_fraud']

# 构建贝叶斯网络

bn.add_nodes(X.columns)
bn.add_edges(X.corr().index, X.corr().values)

# 训练贝叶斯网络
bn.fit(X)

# 评估风险

```
