```python
import hashlib
import time
import redis

# 创建一个新的区块
def create_block(index, previous_hash, timestamp, data, nonce):
    block = {
        'index': index,
        'previous_hash': previous_hash,
        'timestamp': timestamp,
        'data': data,
        'nonce': nonce
    }
    return block

# 计算哈希值
def calculate_hash(block):
    block_string = str(block)
    return hashlib.sha256(block_string.encode()).hexdigest()

# 解决难度调整问题
def mine_block(block, difficulty):
    nonce = 0
    while calculate_hash(block).startswith('0' * difficulty):
        nonce += 1
    block['nonce'] = nonce
    return block

# 创建一个新的区块链
def create_blockchain():
    blockchain = []
    blockchain.append(create_block(0, '0', time.time(), 'Genesis Block', 0))
    for i in range(1, 6):
        previous_hash = calculate_hash(blockchain[-1])
        block = create_block(i, previous_hash, time.time(), f'Block {i}', i)
        block = mine_block(block, 4)
        blockchain.append(block)
    return blockchain

# 存储区块链数据到Redis
def store_blockchain_to_redis(blockchain, redis_client):
    for block in blockchain:
        redis_client.hset('blockchain', block['index'], block)

# 从Redis中获取区块链数据
def get_blockchain_from_redis(redis_client):
    blockchain = []
    for i in range(len(blockchain)):
        block = redis_client.hgetall(f'blockchain:{i}')
        block = dict(block)
        blockchain.append(block)
    return blockchain

# 初始化Redis客户端
redis_client = redis.StrictRedis(host='localhost', port=6379, db=0)

# 创建一个新的区块链
blockchain = create_blockchain()

# 存储区块链数据到Redis
store_blockchain_to_redis(blockchain, redis_client)

# 从Redis中获取区块链数据
blockchain_from_redis = get_blockchain_from_redis(redis_client)
```
