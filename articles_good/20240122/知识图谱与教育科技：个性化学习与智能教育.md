### 4.1 知识图谱构建实例

以下是一个简单的知识图谱构建实例：

```python
from rdflib import Graph, Namespace, Literal, URIRef
from rdflib.namespace import RDF, RDFS

# 创建一个知识图谱对象
g = Graph()

# 创建一个命名空间
ns = Namespace("http://example.org/")

# 创建实体
person = URIRef(ns.person)
name = URIRef(ns.name)
age = URIRef(ns.age)

# 创建属性
name_label = Literal("John")
age_label = Literal(30)

# 创建实体关系
g.add((person, RDF.type, RDFS.Class))
g.add((person, RDF.label, name_label))
g.add((person, age, age_label))
```

### 4.2 个性化学习算法实例

以下是一个简单的个性化学习算法实例：

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 收集学习资源
resources = ["Python编程入门", "数据挖掘技术", "人工智能基础"]

# 学生特点和需求
student_profile = {"年龄": 25, "性别": "男", "兴趣": "编程"}

# 学习资源筛选
def filter_resources(resources, profile):
    return [resource for resource in resources if profile["兴趣"] in resource]

# 学习资源排序
def sort_resources(resources):
    vectorizer = TfidfVectorizer()
    tfidf_matrix = vectorizer.fit_transform(resources)
    cosine_similarities = cosine_similarity(tfidf_matrix, tfidf_matrix)
    sorted_resources = sorted(zip(resources, cosine_similarities.flatten()), key=lambda x: x[1], reverse=True)
    return [resource for resource, _ in sorted_resources]

# 个性化学习资源推荐
recommended_resources = filter_resources(resources, student_profile)
recommended_resources = sort_resources(recommended_resources)
```
