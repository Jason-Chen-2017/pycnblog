    encoded_data = ""
    for i in range(len(data)):
        if i < encoding_rate * len(data):
            encoded_data += data[i]
        else:
            encoded_data += "0"
    return encoded_data

def decode(encoded_data, encoding_rate):
    data = ""
    for i in range(len(encoded_data)):
        if i < encoding_rate * len(encoded_data):
            data += encoded_data[i]
        else:
            data += "1"
    return data

encoding_rate = 0.5
