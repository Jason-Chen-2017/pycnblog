其中，$S(i,u)$ 表示娱乐内容 $i$ 对用户 $u$ 的推荐得分；$r_{ui}$ 表示用户 $u$ 对娱乐内容 $i$ 的评分；$C$ 表示娱乐内容的特征集合；$d(i)$ 表示娱乐内容 $i$ 的特征向量。

import numpy as np
import pandas as pd
from scipy.spatial.distance import cosine
from sklearn.metrics.pairwise import cosine_similarity

# 加载数据
data = pd.read_csv('movie_data.csv')

# 计算用户之间的相似度
user_similarity = pd.DataFrame(index=data['user_id'].unique(), columns=data['user_id'].unique())
user_similarity.fillna(0)
for u in data['user_id'].unique():
    similarities = cosine_similarity(data[data['user_id'] == u].T, data[data['user_id'] == u].T)
    user_similarity.loc[u, :] = similarities.mean(axis=1)

# 计算娱乐内容之间的相似度
item_similarity = pd.DataFrame(index=data['item_id'].unique(), columns=data['item_id'].unique())
item_similarity.fillna(0)
for i in data['item_id'].unique():
    similarities = cosine_similarity(data[data['item_id'] == i].T, data[data['item_id'] == i].T)
    item_similarity.loc[i, :] = similarities.mean(axis=1)

# 推荐娱乐内容
def recommend(user_id, n_recommend=10):
    user_ratings = data[data['user_id'] == user_id].drop('user_id', axis=1)
    user_similarity_ratings = user_similarity[user_id].drop(user_id).values.reshape(1, -1)
    similarity_ratings = cosine_similarity(user_ratings, user_similarity_ratings)
    similarity_ratings = similarity_ratings.flatten()
    weighted_ratings = np.multiply(similarity_ratings, item_similarity.loc[user_id])
    weighted_ratings = weighted_ratings.sum(axis=1)
    recommendations = data.nlargest(n_recommend, 'item_id').sort_values(by='item_id')
    return recommendations

# 测试推荐
recommended_items = recommend(1)
print(recommended_items)
