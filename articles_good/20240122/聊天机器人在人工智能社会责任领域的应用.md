```python
    def __init__(self):
        self.state = None

    def set_state(self, state):
        self.state = state

    def get_state(self):
        return self.state

    def process_input(self, input_text):
        if self.state == "greeting":
            output_text = "你好，有什么可以帮助你吗？"
        elif self.state == "farewell":
            output_text = "再见，祝你一切顺利！"
        else:
            output_text = "抱歉，我没有理解你的意思。"
        return output_text
```
