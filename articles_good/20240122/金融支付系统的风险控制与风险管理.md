```python
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score

# 加载数据

accuracy = accuracy_score(y_test, y_pred)
```

### 5.3 区块链技术

```python
from hashlib import sha256
from time import time

# 创建区块
def create_block(index, previous_hash, timestamp, data, nonce):
    block = {
        'index': index,
        'previous_hash': previous_hash,
        'timestamp': timestamp,
        'data': data,
        'nonce': nonce
    }
    return block

# 计算哈希值
def calculate_hash(block):
    block_string = str(block['index']) + str(block['previous_hash']) + str(block['timestamp']) + str(block['data']) + str(block['nonce'])
    return sha256(block_string.encode()).hexdigest()

# 区块链创建
def create_blockchain():
    blockchain = [create_block(0, '0', time(), 'Genesis Block', '0')]
    return blockchain
```
