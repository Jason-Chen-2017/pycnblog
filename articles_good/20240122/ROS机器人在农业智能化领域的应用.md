## 4. 具体最佳实践：代码实例和详细解释说明

### 4.1 使用ROS实现机器人定位与导航

在农业智能化领域，机器人需要具备高效的定位与导航能力。以下是一个使用ROS实现机器人定位与导航的代码实例：

```python
#!/usr/bin/env python
import rospy
from nav_msgs.msg import Odometry
from tf.msg import tf
from geometry_msgs.msg import PoseStamped

def callback(odom_msg):
    global last_pose
    last_pose = odom_msg.pose.pose

def main():
    rospy.init_node('robot_localization')
    rospy.Subscriber('/odom', Odometry, callback)
    rate = rospy.Rate(10)
    while not rospy.is_shutdown():
        # 获取机器人当前的位置
        current_pose = last_pose
        # 计算机器人的速度和方向
        linear_velocity = current_pose.pose.pose.position.x
        angular_velocity = current_pose.pose.pose.orientation.z
        # 更新机器人的位置
        last_pose = current_pose
        # 输出机器人的位置和速度
        print('Current position: {:.2f}, {:.2f}'.format(current_pose.pose.pose.position.x, current_pose.pose.pose.position.y))
        print('Current orientation: {:.2f}'.format(current_pose.pose.pose.orientation.z))
        print('Linear velocity: {:.2f}'.format(linear_velocity))
        print('Angular velocity: {:.2f}'.format(angular_velocity))
        rate.sleep()

if __name__ == '__main__':
    main()
```

### 4.2 使用ROS实现机器人控制

在农业智能化领域，机器人需要具备高效的控制能力。以下是一个使用ROS实现机器人控制的代码实例：

```python
#!/usr/bin/env python
import rospy
from geometry_msgs.msg import Twist

def callback(twist_msg):
    global linear_velocity, angular_velocity
    linear_velocity = twist_msg.linear.x
    angular_velocity = twist_msg.angular.z

def main():
    rospy.init_node('robot_control')
    rospy.Subscriber('/cmd_vel', Twist, callback)
    rate = rospy.Rate(10)
    while not rospy.is_shutdown():
        # 获取机器人的速度和方向
        linear_velocity = 0.0
        angular_velocity = 0.0
        # 更新机器人的速度和方向
        print('Current linear velocity: {:.2f}'.format(linear_velocity))
        print('Current angular velocity: {:.2f}'.format(angular_velocity))
        rate.sleep()

if __name__ == '__main__':
    main()
```
