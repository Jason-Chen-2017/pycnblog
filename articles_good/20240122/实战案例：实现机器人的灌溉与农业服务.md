```python
from gps import GPSD

def get_gps_coordinates():
    # 使用GPS技术获取设备位置
    gpsd = GPSD()
    gpsd.connect()
    while not gpsd.waiting():
        pass
    wpt = gpsd.get_wpt()
    latitude = wpt.lat
    longitude = wpt.lon
    return latitude, longitude
```
