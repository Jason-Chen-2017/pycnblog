```python
import pandas as pd
import numpy as np

# 数据清洗
def clean_data(df):
    df = df.dropna()  # 删除缺失值
    df = df[df['age'] > 0]  # 删除异常值
    df = df.astype(float)  # 数据类型转换
    return df

# 数据分析
def analyze_data(df):
    df['age'] = df['age'].astype(int)  # 数据类型转换
    df['age_group'] = pd.cut(df['age'], bins=[0, 18, 35, 50, 65, 80, 100], labels=['0-18', '19-34', '35-49', '50-64', '65-79', '80-99', '100+'])
    df['age_group'].value_counts().plot(kind='bar')

data = pd.read_csv('data.csv')
cleaned_data = clean_data(data)
analyze_data(cleaned_data)
```

### 4.3 安全性保障

```python
import hashlib
import os

# 身份验证
def hash_password(password):
    salt = os.urandom(16)
    hashed_password = hashlib.pbkdf2_hmac('sha256', password.encode('utf-8'), salt, 100000)
    return salt + hashed_password

# 授权
def check_password(password, hashed_password):
    salt = hashed_password[:16]
    hashed_input_password = hashlib.pbkdf2_hmac('sha256', password.encode('utf-8'), salt, 100000)
    return hashed_input_password == hashed_password[16:]

password = '123456'
hashed_password = hash_password(password)
print(check_password(password, hashed_password))
```
