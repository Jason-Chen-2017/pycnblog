def random_select(images, weights):
    indices = np.random.choice(len(images), size=len(images), p=weights)
    selected_images = [images[i] for i in indices]
    return selected_images

def weighted_mix(images, weights):
    mixed_images = []
    for image, weight in zip(images, weights):
        mixed_images.append(image * weight)
    return np.sum(mixed_images, axis=0)
