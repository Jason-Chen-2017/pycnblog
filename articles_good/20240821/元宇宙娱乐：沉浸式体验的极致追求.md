```csharp
using UnityEngine;

public class JointDriving : MonoBehaviour
{
    public Transform target;
    public Transform bone;
    public float distanceThreshold = 0.1f;

    private void FixedUpdate()
    {
        Vector3 targetPos = target.position;
        Vector3 bonePos = bone.position;

        Vector3 boneTarget = targetPos - bonePos;
        float boneDistance = Vector3.Distance(boneTarget, bone.rotation * boneTarget);

        if (boneDistance < distanceThreshold)
        {
            bone.rotation = Quaternion.Slerp(bone.rotation, target.rotation, boneDistance / distanceThreshold);
        }
    }
}
```
