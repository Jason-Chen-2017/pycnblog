### 3.1.1 一阶求导法则

一阶求导法则是用于计算函数的一阶导数的求导法则。它的基本思想是通过对函数进行分差Approximation，从而得到导数的近似值。一阶求导法则的公式如下：

$$
f'(x) \approx \frac{f(x + h) - f(x - h)}{2h}
$$

### 3.1.2 二阶求导法则

二阶求导法则是用于计算函数的二阶导数的求导法则。它的基本思想是通过对函数进行分差Approximation，从而得到导数的近似值。二阶求导法则的公式如下：

$$

import numpy as np
import cv2

def gradient(image):
    dx = np.array([[-1, 0, 1], [-2, 0, 2], [-1, 0, 1]])
    dy = np.array([[-1, -2, -1], [0, 0, 0], [1, 2, 1]])

    Gx = np.zeros(image.shape)
    Gy = np.zeros(image.shape)

    for i in range(1, image.shape[0] - 1):
        for j in range(1, image.shape[1] - 1):
            Gx[i, j] = dx[i, j] * image[i, j]
            Gy[i, j] = dy[i, j] * image[i, j]

    return Gx, Gy

def smooth(image, kernel_size):
    kernel = np.ones((kernel_size, kernel_size)) / (kernel_size * kernel_size)
    return cv2.filter2D(image, -1, kernel)

Gx, Gy = gradient(image)
smooth_image = smooth(Gx, 3)

def FAST(image, threshold, hessian_threshold):
    corners = []

    for i in range(1, image.shape[0] - 1):
        for j in range(1, image.shape[1] - 1):
            hessian = np.array([[image[i - 1, j - 1], image[i - 1, j], image[i - 1, j + 1]],

                                [image[i + 1, j - 1], image[i + 1, j], image[i + 1, j + 1]]])
            hessian_matrix = np.linalg.inv(hessian.T @ hessian)
            lambda1, lambda2, lambda3 = np.linalg.eigvals(hessian_matrix)
