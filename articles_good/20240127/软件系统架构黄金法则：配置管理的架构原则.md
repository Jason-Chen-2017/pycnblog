# 模块化设计
class Calculator:
    def add(self, a, b):
        return a + b

    def subtract(self, a, b):
        return a - b

# 抽象设计
class AbstractCalculator:
    def add(self, a, b):
        raise NotImplementedError

    def subtract(self, a, b):
        raise NotImplementedError

class ConcreteCalculator(AbstractCalculator):
    def add(self, a, b):
        return a + b

    def subtract(self, a, b):
        return a - b

# 可扩展性设计
class ExtendableCalculator(ConcreteCalculator):
    def multiply(self, a, b):
        return a * b

# 可维护性设计
class MaintainableCalculator(ExtendableCalculator):
    def divide(self, a, b):
        return a / b

# 可靠性设计
class ReliableCalculator(MaintainableCalculator):
    def validate_input(self, a, b):
        if a < 0 or b < 0:
            raise ValueError("Input values must be non-negative.")
        return a, b

# 使用
calculator = ReliableCalculator()
a, b = calculator.validate_input(10, 5)
result = calculator.add(a, b)
