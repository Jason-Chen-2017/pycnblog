```python
def load_balancer(url):
    parsed_url = urlparse(url)
    hostname = parsed_url.hostname
    port = parsed_url.port or 80
    if hostname in ['www.example.com', 'www.example.net']:
        return f'http://{hostname}:{port}'
    elif hostname in ['www.example.org', 'www.example.com']:
        return f'http://{hostname}:{port}'
    else:
        return None

balanced_url = load_balancer(url)
print(balanced_url)
```
