    def set_language(self, language):
        self.internationalization_strategy.set_language(language)

    def set_currency(self, currency):
        self.internationalization_strategy.set_currency(currency)

    def set_payment(self, payment):
        self.internationalization_strategy.set_payment(payment)

    def set_shipping(self, shipping):
        self.internationalization_strategy.set_shipping(shipping)

    def set_customer_service(self, customer_service):
