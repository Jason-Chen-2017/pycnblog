```python
from sklearn.cluster import KMeans

# 市场需求分析
def market_need_analysis(data):
    # 使用KMeans聚类算法对市场需求进行分析
    kmeans = KMeans(n_clusters=3, random_state=0).fit(data[['需求']])
    data['需求分类'] = kmeans.labels_

    # 分析市场需求分类的分布
    need_distribution = data.groupby('需求分类').size().plot(kind='bar')
    need_distribution.set_title('市场需求分类分布')
    need_distribution.set_xlabel('需求分类')
    need_distribution.set_ylabel('数量')
    need_distribution.show()

market_need_analysis(data)
```
