### 3.1图像增强的核心算法

#### 3.1.1对比度调整

对比度调整是图像增强中最基本的操作之一，其主要目标是调整图像的对比度，使图像更加明显。对比度调整可以通过以下公式实现：

$$
I_{out}(x,y) = I_{in}(x,y) + k
$$

其中，$I_{out}(x,y)$ 表示输出图像的灰度值，$I_{in}(x,y)$ 表示输入图像的灰度值，$k$ 是一个常数，用于调整对比度。

#### 3.1.2锐化

锐化是图像增强中另一个常见的操作之一，其主要目标是增强图像中的边缘信息，使图像更加锐利。锐化可以通过以下公式实现：

$$
I_{out}(x,y) = I_{in}(x,y) * G(x,y)
$$

其中，$I_{out}(x,y)$ 表示输出图像的灰度值，$I_{in}(x,y)$ 表示输入图像的灰度值，$G(x,y)$ 是一个滤波器，用于增强边缘信息。

### 3.2图像改进的核心算法

#### 3.2.1二值化

二值化是图像改进中一个常见的操作之一，其主要目标是将图像转换为二值图像，使图像更加简洁。二值化可以通过以下公式实现：

$$
I_{out}(x,y) =
\begin{cases}
255, & \text{if } I_{in}(x,y) \geq T \\
0, & \text{otherwise}
\end{cases}
$$

其中，$I_{out}(x,y)$ 表示输出图像的灰度值，$I_{in}(x,y)$ 表示输入图像的灰度值，$T$ 是一个阈值，用于将图像分为两个区域。

#### 3.2.2边缘检测

边缘检测是图像改进中另一个常见的操作之一，其主要目标是检测图像中的边缘信息，使图像更加明确。边缘检测可以通过以下公式实现：

$$
\nabla I(x,y) = \begin{bmatrix} \frac{\partial I}{\partial x} \\ \frac{\partial I}{\partial y} \end{bmatrix}
$$

其中，$\nabla I(x,y)$ 表示图像的梯度向量，$\frac{\partial I}{\partial x}$ 和 $\frac{\partial I}{\partial y}$ 分别表示图像在x和y方向的梯度。

import cv2
import numpy as np

def contrast_adjust(image, k):
    height, width = image.shape[:2]
    output_image = np.zeros((height, width), dtype=np.uint8)
    for y in range(height):
        for x in range(width):
            output_image[y, x] = image[y, x] + k
    return output_image

import cv2
import numpy as np

def sharpen(image, kernel):
    height, width = image.shape[:2]
    output_image = np.zeros((height, width), dtype=np.uint8)
    for y in range(height):
        for x in range(width):
            sum = 0
            for dy in range(-1, 2):
                for dx in range(-1, 2):
                    if dx == 0 and dy == 0:
                        continue
                    output_image[y, x] += image[y + dy, x + dx] * kernel[dy, dx]
                    sum += kernel[dy, dx]
            output_image[y, x] = output_image[y, x] / sum
    return output_image

kernel = np.array([[-1, -1, -1],
                   [-1,  8, -1],
                   [-1, -1, -1]])

output_image = sharpen(image, kernel)
cv2.imshow('Output Image', output_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

import cv2
import numpy as np

def binary_image(image, threshold):
    height, width = image.shape[:2]
    output_image = np.zeros((height, width), dtype=np.uint8)
    for y in range(height):
        for x in range(width):
            if image[y, x] >= threshold:
                output_image[y, x] = 255
            else:
                output_image[y, x] = 0
    return output_image

threshold = 128
output_image = binary_image(image, threshold)
cv2.imshow('Output Image', output_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

#### 4.2.2边缘检测

import cv2
import numpy as np

def edge_detection(image, kernel):
    height, width = image.shape[:2]
    output_image = np.zeros((height, width), dtype=np.uint8)
    for y in range(height):
        for x in range(width):
            gradient_x = 0
            gradient_y = 0
            for dy in range(-1, 2):
                for dx in range(-1, 2):
                    if dx == 0 and dy == 0:
                        continue
                    gradient_x += image[y + dy, x + dx] * kernel[dy, dx][0]
                    gradient_y += image[y + dy, x + dx] * kernel[dy, dx][1]
            gradient = np.sqrt(gradient_x**2 + gradient_y**2)
            output_image[y, x] = gradient
    return output_image

kernel = np.array([[-1, -1, -1],
                   [-1,  0, -1],
                   [-1, -1, -1]])

output_image = edge_detection(image, kernel)
cv2.imshow('Output Image', output_image)
cv2.waitKey(0)
cv2.destroyAllWindows()
