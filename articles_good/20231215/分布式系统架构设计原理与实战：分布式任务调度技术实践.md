```python
import threading
import time

class Task:
    def __init__(self, id, execution_time):
        self.id = id
        self.execution_time = execution_time

class Node:
    def __init__(self, id):
        self.id = id
        self.tasks = []

    def execute_task(self, task):
        start_time = time.time()
        time.sleep(task.execution_time)
        end_time = time.time()
        print(f"Node {self.id} executed task {task.id} in {end_time - start_time} seconds")

class TaskScheduler:
    def __init__(self, nodes):
        self.nodes = nodes
        self.tasks = []

    def add_task(self, task):
        self.tasks.append(task)

    def distribute_tasks(self):
        for task in self.tasks:
            node = self.select_node(task)
            node.tasks.append(task)
            threading.Thread(target=node.execute_task, args=(task,)).start()

    def select_node(self, task):
        min_execution_time = float("inf")
        node = None
        for node in self.nodes:
            if task.execution_time < min_execution_time:
                min_execution_time = task.execution_time
                node = node
        return node

# 初始化节点
nodes = [Node(i) for i in range(5)]

# 初始化任务调度中心
task_scheduler = TaskScheduler(nodes)

# 添加任务
task_scheduler.add_task(Task(1, 2))
task_scheduler.add_task(Task(2, 1))
task_scheduler.add_task(Task(3, 3))

# 分配任务
task_scheduler.distribute_tasks()
```
