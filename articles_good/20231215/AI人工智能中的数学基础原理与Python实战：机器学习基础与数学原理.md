```python
import numpy as np

# 初始化权重
w = np.random.randn(1, X_train.shape[1])

# 学习率
alpha = 0.01

# 迭代次数
iterations = 1000

# 训练模型
for i in range(iterations):
    # 计算梯度
    grad = 2 * (X_train.T.dot(X_train.dot(w) - y_train)).mean(axis=0)

    # 更新权重
    w -= alpha * grad

# 预测
y_pred = X_test.dot(w)
```
