    key = random.choice(['C', 'D', 'E', 'F', 'G', 'A', 'B'])

    # 将生成的旋律、节奏、音调等组合成一个完整的音乐作品
    music = {
        'tempo': tempo,
        'key': key,
        'time_signature': time_signature,
        'melody': melody,
        'rhythm': rhythm
    }

    return music
