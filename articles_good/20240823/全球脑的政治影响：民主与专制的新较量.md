```python
import numpy as np
from sklearn.linear_model import LinearRegression

# 数据收集
data = np.array([[1, 1], [1, 2], [2, 2], [2, 3]])

# 数据预处理
X = data[:, :1]

# 模型训练

# 决策生成

print(prediction)
```
