pragma solidity ^0.8.0;

contract MetaVerseArbitration {
    struct Case {
        address plaintiff;
        address defendant;
        bytes32 evidenceHash;
        bytes32 lawHash;
        bool resolved;
    }

    mapping(bytes32 => Case) public cases;

    function createCase(
        address _plaintiff,
        address _defendant,
        bytes32 _evidenceHash,
        bytes32 _lawHash
    ) external {
        require(!cases[_evidenceHash].resolved, "Case already resolved");

        Case memory newCase = Case({
            plaintiff: _plaintiff,
            defendant: _defendant,
            evidenceHash: _evidenceHash,
            lawHash: _lawHash,
            resolved: false
        });

        cases[_evidenceHash] = newCase;
    }

    function resolveCase(bytes32 _evidenceHash, bytes32 _裁决结果) external {
        require(!cases[_evidenceHash].resolved, "Case already resolved");

        bytes32 calculatedResult = hashEvidenceAndLaw(
            cases[_evidenceHash].evidenceHash,
            cases[_evidenceHash].lawHash
        );

        require(calculatedResult == _裁决结果, "Invalid result");

        cases[_evidenceHash].resolved = true;
    }

    function hashEvidenceAndLaw(bytes32 _evidenceHash, bytes32 _lawHash) private pure returns (bytes32) {
        return keccak256(abi.encodePacked(_evidenceHash, _lawHash));
    }
}
