import json

def send_request(url, method, data=None):
    """发送HTTP请求"""
    headers = {'Content-Type': 'application/json'}
    if method == 'POST':
        response = requests.post(url, headers=headers, json=data)
    elif method == 'GET':
        response = requests.get(url, headers=headers, params=data)
    return response.json()

def create_arbitration_case(data):
    """创建仲裁案件"""
    url = 'https://api.metaverse.com/arbitration/case'
    response = send_request(url, 'POST', data)
    return response

def query_arbitration_case(case_id):
    """查询仲裁案件"""
    url = f'https://api.metaverse.com/arbitration/case/{case_id}'
    response = send_request(url, 'GET')
    return response
