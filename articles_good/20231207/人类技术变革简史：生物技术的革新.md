    Y_str = ''.join(Y)

    # 然后，将两个字符串分别转换为数字形式
    X_num = [ord(x) for x in X_str]
    Y_num = [ord(y) for y in Y_str]

    # 接着，将两个数字序列分别转换为矩阵形式
    matrix = [[0 for _ in range(len(Y_num) + 1)] for _ in range(len(X_num) + 1)]

    for i in range(1, len(X_num) + 1):
        for j in range(1, len(Y_num) + 1):
            if X_num[i - 1] == Y_num[j - 1]:
                matrix[i][j] = matrix[i - 1][j - 1] + 1
            else:
                matrix[i][j] = max(matrix[i - 1][j], matrix[i][j - 1])

    # 最后，将最长公共子序列转换回字符串形式，并计算出它们之间的相似性
    max_length = matrix[len(X_num)][len(Y_num)]
    alignment = [None] * max_length
    i, j = len(X_num), len(Y_num)
    while i > 0 and j > 0:
        if X_num[i - 1] == Y_num[j - 1]:
            alignment[max_length - 1] = X_num[i - 1]
            i -= 1
            j -= 1
            max_length -= 1
        elif matrix[i - 1][j] > matrix[i][j - 1]:
            i -= 1
        else:
            j -= 1

    return ''.join(alignment)
