matches = bf.knnMatch(descriptors, descriptors, k=2)

# 显示图像
good = []
for m, n in matches:
    if m.distance < 0.75 * n.distance:
        good.append([m])

img2 = cv2.drawMatches(img, keypoints[0], img, keypoints[1], good, None, flags=2)
