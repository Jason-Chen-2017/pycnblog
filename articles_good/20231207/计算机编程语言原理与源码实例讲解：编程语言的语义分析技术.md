```python
import re

def tokenize(source):
    tokens = []
    pattern = r"[a-zA-Z_][a-zA-Z0-9_]+"
    for match in re.finditer(pattern, source):
        token = match.group(0)
        if token.isupper():
            tokens.append(("IDENTIFIER", token))
        else:
            tokens.append(("KEYWORD", token))
    return tokens
```
