```python
from mpi4py import MPI

def distributed_computing(comm):
    # 获取当前进程的ID和名称
    rank = comm.Get_rank()
    name = comm.Get_name()

    # 执行并行计算任务
    x = np.random.rand(100)
    y = np.random.rand(100)
    z = x + y
    result = np.sum(z**2)

    # 将结果发送给其他进程
    if rank == 0:
        comm.Send(result, dest=1, tag=0)
    elif rank == 1:
        result_received = comm.Recv(source=0, tag=0)
        print(f"Rank {rank} received result: {result_received}")

# 初始化MPI环境
comm = MPI.COMM_WORLD

# 执行分布式计算任务
distributed_computing(comm)
```
