    {'age': 30, 'gender': 'female', 'interest': 'romance'},
    {'age': 40, 'gender': 'male', 'interest': 'comedy'}

# 物品特征
movie_features = [
    {'description': 'action movie', 'type': 'action'},
    {'description': 'romance movie', 'type': 'romance'},
    {'description': 'comedy movie', 'type': 'comedy'}

similarity_matrix = cosine_similarity(user_features, movie_features)

# 根据相似度来为新用户和新物品提供个性化的推荐
