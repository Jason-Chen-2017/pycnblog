## 4. 具体代码实例和详细解释说明

以下是一个简单的遗传算法实现示例，用于优化投资组合的风险管理：

```python
import numpy as np

# 初始化种群
population_size = 100
population = np.random.rand(population_size, 5)

# 适应度函数
def fitness_function(x):
    # 计算风险
    risk = np.sum(x**2)
    # 最小化风险
    return 1 / (1 + risk)

# 选择操作
def selection(population, fitness):
    # 计算适应度
    fitness_values = np.array([fitness(x) for x in population])
    # 轮盘赌选择
    probabilities = fitness_values / np.sum(fitness_values)
    selected_indices = np.random.choice(np.arange(len(population)), size=population_size, p=probabilities, replace=True)
    selected_population = population[selected_indices]
    return selected_population

# 交叉操作
def crossover(population):
    # 随机选择两个父亲
    parent1_index = np.random.randint(0, population_size)
    parent2_index = np.random.randint(0, population_size)
    # 随机选择交叉点
    crossover_point = np.random.randint(1, 5)
    # 生成子女
    child1 = np.concatenate((population[parent1_index][:crossover_point], population[parent2_index][crossover_point:]))
    child2 = np.concatenate((population[parent2_index][:crossover_point], population[parent1_index][crossover_point:]))
    return np.array([child1, child2])

# 变异操作
def mutation(population):
    # 随机选择一个基因
    gene_index = np.random.randint(0, population_size)
    # 随机选择一个值
    value = np.random.rand()
    # 生成变异后的基因
    mutated_gene = population[gene_index] + value
    # 限制基因范围
    mutated_gene = np.clip(mutated_gene, 0, 1)
    return mutated_gene

# 主程序
for _ in range(1000):
    # 评估适应度
    fitness_values = np.array([fitness_function(x) for x in population])
    # 选择
    selected_population = selection(population, fitness_values)
    # 交叉
    new_population = np.concatenate((selected_population, crossover(selected_population)))
    # 变异
    mutated_population = np.array([mutation(x) for x in new_population])
    # 更新种群
    population = mutated_population

# 输出最佳解
best_solution = population[np.argmax(fitness_values)]
print("最佳解:", best_solution)
```

上述代码实现了一个简单的遗传算法，用于优化投资组合的风险管理。代码首先初始化种群，然后使用适应度函数评估种群中每个个体的适应度。接着使用选择、交叉和变异操作生成新的种群。最后输出最佳解。
