domain.createXML(xml)
domain.define(name, memory, vcpu, disk)

domain.start()

# 监控虚拟机
while True:
    state = domain.info()[libvirt_domain.VIR_DOMAIN_INFO_RUNNING]
    if state == libvirt_domain.VIR_DOMAIN_RUNNING:
        print('虚拟机正在运行')
    else:
