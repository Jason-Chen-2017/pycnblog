import pandas as pd
import numpy as np

# 缺失值处理
def fill_missing_values(data, method='mean'):
    if method == 'mean':
        data.fillna(data.mean(), inplace=True)
    elif method == 'median':
        data.fillna(data.median(), inplace=True)
    elif method == 'mode':
        data.fillna(data.mode().iloc[0], inplace=True)
    elif method == 'interpolate':
        data.interpolate(inplace=True)
    elif method == 'ffill':
        data.ffill(inplace=True)
    elif method == 'bfill':
        data.bfill(inplace=True)
    return data

# 数据类型转换
def convert_data_type(data, column, data_type='int'):
    if data_type == 'int':
        data[column] = data[column].astype(int)
    elif data_type == 'float':
        data[column] = data[column].astype(float)
    elif data_type == 'str':
        data[column] = data[column].astype(str)
    elif data_type == 'datetime':
        data[column] = pd.to_datetime(data[column])
    return data

# 数据格式转换
def convert_data_format(data, column, data_format='str'):
    if data_format == 'str':
        data[column] = data[column].astype(str)
    elif data_format == 'date':
        data[column] = pd.to_datetime(data[column]).dt.date
    elif data_format == 'datetime':
        data[column] = pd.to_datetime(data[column])

def aggregate_calculate(data, column, operation='mean'):

        result = data[column].mean()
    elif operation == 'median':
        result = data[column].median()
    elif operation == 'mode':
        result = data[column].mode()
    elif operation == 'var':
        result = data[column].var()
    elif operation == 'std':
        result = data[column].std()
    return result
