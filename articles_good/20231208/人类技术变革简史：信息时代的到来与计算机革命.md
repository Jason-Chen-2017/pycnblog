        return '1', 'move_right'

        return '1', 'move_right'

def turing_machine(input_string):
    tape = list(input_string)
    while tape:
        state, direction = rule_transform(state, tape[0])
        tape.pop(0)
        if direction == 'move_left':
            tape.insert(0, state)
        else:
            tape.append(state)
    return tape

# 计算
