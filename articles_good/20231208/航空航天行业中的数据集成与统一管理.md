  data = pd.concat([data1, data2], axis=0)  # 纵向连接
  data = pd.concat([data1, data2], axis=1)  # 横向连接

  # 数据合并
  data = pd.merge(data1, data2, on='key', how='inner')  # 内连接
  data = pd.merge(data1, data2, on='key', how='outer')  # 外连接
  data = pd.merge(data1, data2, on='key', how='left')  # 左连接
  data = pd.merge(data1, data2, on='key', how='right')  # 右连接

  # 数据聚合
  data = data.groupby('key').mean()  # 平均值聚合
  data = data.groupby('key').sum()  # 总和聚合
  data = data.groupby('key').max()  # 最大值聚合
