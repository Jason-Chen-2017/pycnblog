### 4.1 数据清洗代码实例

```python
import pandas as pd
import numpy as np

# 数据预处理
def preprocess_data(data):
    # 去除噪声
    data = data.apply(lambda x: np.mean(x))
    # 填充缺失
    data = data.fillna(data.mean())
    # 转换类型
    data = data.astype(np.float32)
    # 标准化
    data = (data - data.mean()) / data.std()
    return data

# 数据检查
def check_data(data):
    # 异常值检测
    z_scores = np.abs(np.std(data, axis=0) / np.mean(data, axis=0))
    outliers = data[z_scores > 3]
    # 错误值检测
    errors = data[data < 0]
    return outliers, errors

# 数据清洗
def clean_data(data, outliers, errors):
    # 修复异常值
    data = data.replace(outliers, np.nan)
    # 修复错误值
    data = data.replace(errors, 0)
    # 填充缺失
    data = data.fillna(data.mean())
    return data
```
