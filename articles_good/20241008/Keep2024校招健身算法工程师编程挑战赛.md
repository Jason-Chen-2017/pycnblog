#### 4.1 线性回归模型

线性回归是一种常见的数学模型，用于分析两个或多个变量之间的线性关系。在健身算法中，线性回归模型可以用于预测用户的体重、消耗的卡路里等指标。

线性回归模型的基本公式为：

\[ y = \beta_0 + \beta_1 \cdot x \]

其中，\( y \) 是因变量（如体重），\( x \) 是自变量（如时间），\( \beta_0 \) 是截距，\( \beta_1 \) 是斜率。

**示例**：

\[ y = 50 + 0.5 \cdot x \]

当用户每天运动30分钟时，预测体重为：

\[ y = 50 + 0.5 \cdot 30 = 65 \text{ kg} \]

```python
import numpy as np
import pandas as pd

from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# 5.2.1 数据收集
def collect_data():
    # 从文件中读取数据
    data = pd.read_csv('fitness_data.csv')
    return data

# 5.2.2 数据预处理
def preprocess_data(data):
    # 清洗数据，处理缺失值和异常值
    data = data.dropna()
    data = data[(data['weight'] > 30) & (data['weight'] < 150)]
    return data

# 5.2.3 特征提取
def extract_features(data):
    # 提取特征
    features = data[['age', 'height', 'duration']]
    return features

# 5.2.4 目标变量提取
def extract_target(data):
    # 提取目标变量（如消耗的卡路里）
    target = data['calories']
    return target

# 5.2.5 模型训练
def train_model(features, target):
    # 分割数据集
    X_train, X_test, y_train, y_test = train_test_split(features, target, test_size=0.2, random_state=42)
    
    # 创建线性回归模型
    model = LinearRegression()
    
    # 训练模型
    model.fit(X_train, y_train)
    
    return model

# 5.2.6 模型评估
def evaluate_model(model, X_test, y_test):
    # 评估模型性能
    y_pred = model.predict(X_test)
    mse = mean_squared_error(y_test, y_pred)
    return mse

# 5.2.7 结果输出
def generate_plan(model, new_user_data):
    # 生成个性化健身计划
    calories = model.predict([new_user_data])
    return calories

# 主函数
if __name__ == '__main__':
    # 收集数据
    data = collect_data()
    
    # 预处理数据
    data = preprocess_data(data)
    
    # 提取特征和目标变量
    features = extract_features(data)
    target = extract_target(data)
    
    # 训练模型
    model = train_model(features, target)
    
    # 评估模型
    mse = evaluate_model(model, features, target)
    print(f'Mean Squared Error: {mse}')
    
    # 生成个性化健身计划
    new_user_data = np.array([[25, 170, 60]])
    calories = generate_plan(model, new_user_data)
    print(f'Predicted Calories: {calories[0]}')
```
