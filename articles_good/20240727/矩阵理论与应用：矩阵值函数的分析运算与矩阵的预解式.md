4. 分析矩阵的稳定性：判断特征值的实部和虚部来分析矩阵的稳定性。

import numpy as np

def matrix_exponential(A, t):
    n = np.size(A)
    M = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            M[i, j] = sum([(A ** k)[i, j] / (k!) for k in range(t + 1)])
    return M

# 测试矩阵指数函数
