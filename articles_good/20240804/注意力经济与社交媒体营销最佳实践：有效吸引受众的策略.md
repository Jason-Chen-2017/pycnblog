```python
import pandas as pd

app = Flask(__name__)

# 内容创作
@app.route('/create_content', methods=['POST'])
def create_content():
    title = request.form['title']
    description = request.form['description']
    content = request.form['content']
    return render_template('create_content.html', title=title, description=description, content=content)

# 内容分发
@app.route('/publish_content', methods=['POST'])
def publish_content():
    content_id = request.form['content_id']
    platform = request.form['platform']
    return render_template('publish_content.html', content_id=content_id, platform=platform)

# 用户互动
@app.route('/user_interaction', methods=['POST'])
def user_interaction():
    content_id = request.form['content_id']
    interaction_type = request.form['interaction_type']
    user_id = request.form['user_id']
    return render_template('user_interaction.html', content_id=content_id, interaction_type=interaction_type, user_id=user_id)

if __name__ == '__main__':
    app.run(debug=True)
```
