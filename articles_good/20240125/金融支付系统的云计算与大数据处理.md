```python
import boto3

ec2 = boto3.client('ec2')

# 创建一个新的实例

    ImageId='ami-0c55b159cbfafe1f0',
    MinCount=1,
    MaxCount=1,
    InstanceType='t2.micro',
    KeyName='my-key-pair',
    SecurityGroupIds=['sg-087654321']
)

# 获取实例的ID
instance_id = response['Instances'][0]['InstanceId']

# 打印实例的ID
print(f'创建的实例ID为：{instance_id}')
```

### 4.2 大数据处理的具体最佳实践：代码实例和详细解释说明

以下是一个简单的大数据处理代码实例：

```python
from pyspark import SparkConf, SparkContext

# 创建一个Spark配置对象
conf = SparkConf().setAppName('bigdata_example').setMaster('local')

# 创建一个Spark上下文对象
sc = SparkContext(conf=conf)

# 读取一个大数据文件
data = sc.textFile('hdfs://localhost:9000/user/cloudera/data.txt')

# 对数据进行过滤、清洗、转换等操作
filtered_data = data.filter(lambda line: 'error' not in line)
cleaned_data = filtered_data.map(lambda line: line.strip())
converted_data = cleaned_data.map(lambda line: line.split(','))

# 对数据进行挖掘和分析
result = converted_data.countByValue()

# 打印结果
print(result)
```
