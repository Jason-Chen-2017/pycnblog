```python
from qiskit import QuantumCircuit, execute, Aer

# 创建一个量子电路
qc = QuantumCircuit(2)

# 添加Pauli门操作
qc.x(0)
qc.y(1)

# 添加Hadamard门操作
qc.h(0)

# 添加CNOT门操作
qc.cx(0, 1)

# 使用Aer后端执行量子电路

result = job.result()

# 打印结果

print(counts)
```
