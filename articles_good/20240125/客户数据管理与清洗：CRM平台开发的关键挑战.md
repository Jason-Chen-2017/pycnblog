### 4.1 客户数据收集

```python
import requests
from bs4 import BeautifulSoup

url = 'https://www.example.com/customers'
response = requests.get(url)
soup = BeautifulSoup(response.text, 'html.parser')
customers = soup.find_all('tr')

for customer in customers:
    name = customer.find('td', class_='name').text
    email = customer.find('td', class_='email').text
    phone = customer.find('td', class_='phone').text
    address = customer.find('td', class_='address').text
    customer_data = {
        'name': name,
        'email': email,
        'phone': phone,
        'address': address
    }
    print(customer_data)
```

### 4.2 客户数据存储

```python
import sqlite3

connection = sqlite3.connect('customers.db')
cursor = connection.cursor()

cursor.execute('''
    CREATE TABLE IF NOT EXISTS customers (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT,
        email TEXT,
        phone TEXT,
        address TEXT
    )
''')
```

### 4.3 客户数据处理

```python
import pandas as pd

customer_data_df = pd.DataFrame(customer_data_list)
customer_data_df = customer_data_df.drop_duplicates()
customer_data_df = customer_data_df.dropna()
customer_data_df['phone'] = customer_data_df['phone'].str.replace(r'\D', '')
customer_data_df['email'] = customer_data_df['email'].str.lower()
customer_data_df['address'] = customer_data_df['address'].str.strip()
customer_data_df.to_csv('cleaned_customers.csv', index=False)
```
