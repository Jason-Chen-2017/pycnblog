### 5.1 加密算法的实例

以AES加密算法为例，我们可以使用以下代码实现：

```java
import javax.crypto.Cipher;
import javax.crypto.KeyGenerator;
import javax.crypto.SecretKey;
import javax.crypto.spec.SecretKeySpec;

public class AESExample {
    public static void main(String[] args) throws Exception {
        // 生成AES密钥
        KeyGenerator keyGenerator = KeyGenerator.getInstance("AES");
        keyGenerator.init(128);
        SecretKey secretKey = keyGenerator.generateKey();

        // 使用AES加密
        Cipher cipher = Cipher.getInstance("AES");
        cipher.init(Cipher.ENCRYPT_MODE, secretKey);
        byte[] plaintext = "Hello, World!".getBytes();
        byte[] ciphertext = cipher.doFinal(plaintext);

        // 使用AES解密
        cipher.init(Cipher.DECRYPT_MODE, secretKey);
        byte[] decryptedText = cipher.doFinal(ciphertext);

        System.out.println("Plaintext: " + new String(plaintext));
        System.out.println("Ciphertext: " + new String(ciphertext));
        System.out.println("Decrypted Text: " + new String(decryptedText));
    }
}
```

### 5.2 身份验证算法的实例

以JWT身份验证算法为例，我们可以使用以下代码实现：

```java
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.Keys;

public class JWTExample {
    public static void main(String[] args) {
        // 生成JWT令牌
        String jwt = Jwts.builder()
                .setSubject("user")
                .setIssuedAt(new Date())
                .signWith(Keys.hmacShaKeyFor("secret".getBytes()))
                .compact();

        System.out.println("JWT: " + jwt);

        // 验证JWT令牌
        boolean isValid = Jwts.parser()
                .setSigningKey(Keys.hmacShaKeyFor("secret".getBytes()))
                .parseClaimsJws(jwt)
                .getBody()
                .getExpiration()
                .after(new Date());

        System.out.println("Is Valid: " + isValid);
    }
}
```

### 5.3 通信协议的实例

以HTTPS通信协议为例，我们可以使用以下代码实现：

```java
import java.net.HttpURLConnection;
import java.net.URL;

public class HTTPSExample {
    public static void main(String[] args) throws Exception {
        // 创建URL对象
        URL url = new URL("https://api.example.com/pay");

        // 创建HttpURLConnection对象
        HttpURLConnection connection = (HttpURLConnection) url.openConnection();

        // 设置请求方法
        connection.setRequestMethod("POST");

        // 设置请求头
        connection.setRequestProperty("Content-Type", "application/json");

        // 设置请求体
        String requestBody = "{\"amount\":100,\"currency\":\"USD\"}";
        connection.setDoOutput(true);
        connection.getOutputStream().write(requestBody.getBytes());

        // 获取响应状态码
        int responseCode = connection.getResponseCode();
        System.out.println("Response Code: " + responseCode);

        // 获取响应体
        StringBuilder response = new StringBuilder();
        try (java.io.BufferedReader reader = new java.io.BufferedReader(new java.io.InputStreamReader(connection.getInputStream()))) {
            String line;
            while ((line = reader.readLine()) != null) {
                response.append(line);
            }
        }
        System.out.println("Response: " + response.toString());

        // 关闭连接
        connection.disconnect();
    }
}
```
