from keras.models import Sequential
from keras.layers import Conv2D, Dense, Flatten

# 数据集
X = np.array([[[1, 2], [3, 4]], [[5, 6], [7, 8]]])
y = np.array([0, 1])

- 优化算法：

- 梯度下降：

```python
import numpy as np

# 数据集
X = np.array([[1, 2], [3, 4]])
y = np.array([1, 2])

# 模型
theta = np.array([0, 0])
alpha = 0.01

# 训练
for _ in range(1000):
    grad = np.dot(X.T, (X * np.dot(X.T, theta) - y))
    theta = theta - alpha * grad

# 预测
pred = np.dot(X, theta)
```

- 随机梯度下降：

```python
import numpy as np
import random

# 数据集
X = np.array([[1, 2], [3, 4]])
y = np.array([1, 2])

# 模型
theta = np.array([0, 0])
alpha = 0.01

# 训练
for _ in range(1000):
    i = random.randint(0, len(X) - 1)
    grad = 2 * (X[i] * (np.dot(X[i].T, theta) - y[i]))
    theta = theta - alpha * grad

# 预测
pred = np.dot(X, theta)
```
