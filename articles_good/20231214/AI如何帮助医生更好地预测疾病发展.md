    result = model.predict([[age, blood_pressure, blood_sugar]])

import pickle

model = pickle.load(open('model.pkl', 'rb'))

# 创建Flask应用
app = Flask(__name__)

# 定义预测接口
@app.route('/predict', methods=['POST'])
def predict():
    data = request.get_json()
    age = data['age']
    blood_pressure = data['blood_pressure']

if __name__ == '__main__':
    app.run(debug=True)
