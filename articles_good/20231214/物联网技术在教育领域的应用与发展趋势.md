import time
from datetime import datetime

class Student:
    def __init__(self, name):
        self.name = name
        self.study_records = []

    def add_study_record(self, study_record):
        self.study_records.append(study_record)

    def get_study_records(self):
        return self.study_records

class StudyRecord:
    def __init__(self, study_time, study_content, study_result):
        self.study_time = study_time
        self.study_content = study_content
        self.study_result = study_result

# 创建学生对象
student = Student("张三")

# 添加学习记录
study_record = StudyRecord(datetime.now(), "学习Python编程", "学习完成")
student.add_study_record(study_record)

# 获取学习记录
study_records = student.get_study_records()
for record in study_records:
    print(record)

from datetime import datetime

class Student:
    def __init__(self, name):
        self.name = name
        self.study_resources = []

    def allocate_study_resource(self, study_resource):
        self.study_resources.append(study_resource)

    def use_study_resource(self, study_resource):
        for resource in self.study_resources:
            if resource == study_resource:
                resource.use()
                return True
        return False

    def get_study_resources(self):

class StudyResource:

        self.name = name
        self.capacity = capacity
        self.used_capacity = 0

    def use(self):
        self.used_capacity += 1

    def __str__(self):
        return "资源名称：%s，容量：%d，已使用容量：%d" % (self.name, self.capacity, self.used_capacity)

# 创建学生对象

# 分配学习资源
study_resource = StudyResource("笔记本电脑", 1)
student.allocate_study_resource(study_resource)

student.use_study_resource(study_resource)

# 获取学习资源
study_resources = student.get_study_resources()
for resource in study_resources:
    print(resource)
