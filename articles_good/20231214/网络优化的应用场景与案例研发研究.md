### 4.1 流量优化算法实例

在这个实例中，我们将实现一个基于距离的流量分配算法，即最短路径算法。

```python
import networkx as nx

def shortest_path(graph, start, end):
    visited = set()
    queue = [(start, [start])]

    while queue:
        path, current_path = queue.pop(0)
        visited.add(path)

        if path == end:
            return current_path

        for neighbor in graph[path]:
            if neighbor not in visited:
                queue.append((neighbor, current_path + [neighbor]))

graph = nx.Graph()
graph.add_nodes_from(['A', 'B', 'C', 'D', 'E'])
graph.add_edges_from([('A', 'B'), ('A', 'C'), ('B', 'D'), ('C', 'D'), ('C', 'E')])

start = 'A'
end = 'E'

path = shortest_path(graph, start, end)
print(path)  # Output: ['A', 'C', 'E']
```

class LRUCache:
    def __init__(self, capacity):
        self.capacity = capacity
        self.cache = {}
        self.access_time = {}

    def get(self, key):
        if key not in self.cache:
            return -1

        self.access_time[key] = time.time()
        return self.cache[key]

    def put(self, key, value):
        if key in self.cache:
            self.access_time[key] = time.time()
        else:
            if len(self.cache) >= self.capacity:
                oldest_key = min(self.access_time, key for key in self.access_time)
                del self.cache[oldest_key]
                del self.access_time[oldest_key]

            self.cache[key] = value
            self.access_time[key] = time.time()

### 4.3 安全优化算法实例

在这个实例中，我们将实现一个基于规则的防火墙算法，即基于规则的状态防火墙。

```python
import re

def rule_based_firewall(traffic, rules):
    for rule in rules:
        pattern = re.compile(rule['pattern'])
        if pattern.match(traffic):
            if rule['action'] == 'allow':
                return rule['action']
            else:
                return rule['action']

    return 'deny'

rules = [
    {'pattern': '^GET', 'action': 'allow'},
    {'pattern': '^POST', 'action': 'deny'},
    {'pattern': '^PUT', 'action': 'allow'},
    {'pattern': '^DELETE', 'action': 'deny'}
]

traffic = 'GET /index.html HTTP/1.1'

action = rule_based_firewall(traffic, rules)
print(action)  # Output: 'allow'
```
