import numpy as np
import scipy.signal as signal

# 读取音频文件
audio_file = 'audio.wav'
audio_data = np.fromfile(audio_file, dtype=np.int16)

# 设置滤波器参数
filter_order = 4
cutoff_frequency = 0.1

# 设置压缩器参数
compressor_order = 4
compression_ratio = 0.5

# 设置补偿器参数
compensator_order = 4
compensation_gain = 0.5

filter_coefficients = signal.firwin(filter_order, cutoff_frequency, window='hamming')

# 设计压缩器
compressor_coefficients = signal.lfilter_fir(compressor_order, 1, audio_data, 0.5)

# 设计补偿器
compensator_coefficients = signal.lfilter_fir(compensator_order, 1, audio_data, 0.5)

# 恢复音频信号
restored_audio_data = signal.lfilter_fir(filter_coefficients, 1, compressor_coefficients)

# 写入恢复后的音频文件
restored_audio_file = 'restored_audio.wav'
