```python
import numpy as np

def hsv_to_rgb(h, s, v):
    r, g, b = 0, 0, 0
    if s == 0:
        r, g, b = v, v, v
    else:
        h1 = h / 60
        x = v * (1 - s)
        r1, g1, b1 = v, v * (1 - h1), v * (1 - h1)
        if 0 <= h1 < 1:
            r, g, b = r1, g, b1
        elif 1 <= h1 < 2:
            r, g, b = g1, x, b1
        elif 2 <= h1 < 3:
            r, g, b = x, b1, r1
        elif 3 <= h1 < 4:
            r, g, b = b1, r1, g1
        elif 4 <= h1 < 5:
            r, g, b = g1, x, r1
        elif 5 <= h1 < 6:
            r, g, b = r1, b1, g1
    return r, g, b

h, s, v = 30, 0.5, 0.5
r, g, b = hsv_to_rgb(h, s, v)
print(r, g, b)
```
