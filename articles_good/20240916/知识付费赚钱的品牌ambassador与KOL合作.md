import numpy as np
import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

# 1. 数据预处理
def preprocess_data(data):
    # 数据清洗、去重等操作
    return data

# 2. 影响力计算
def compute_influence(data):
    # 加权平均法计算影响力得分
    scores = {}
    for user in data:
        score = 0.5 * user['followers'] + 0.3 * user['interactions'] + 0.2 * user['content_quality']
        scores[user['id']] = score
    return scores

# 3. 协同过滤推荐
def collaborative_filtering(user_interest, content_data):
    # 计算相似度
    similarities = cosine_similarity([user_interest], content_data)
    # 计算加权平均相似度
    weighted_average = similarities * content_data
    return weighted_average

# 示例数据
data = [
    {'id': 1, 'followers': 1000, 'interactions': 0.1, 'content_quality': 0.8},
    {'id': 2, 'followers': 2000, 'interactions': 0.15, 'content_quality': 0.9},
    {'user_interest': [0.8, 0.2, 0.1]},
    {'content': [
        {'id': 1, 'score': 0.6, 'weight': 0.5},
        {'id': 2, 'score': 0.4, 'weight': 0.5}
    ]}
]

# 数据预处理
preprocessed_data = preprocess_data(data)

# 计算影响力得分
influence_scores = compute_influence(preprocessed_data['users'])

# 计算协同过滤推荐
user_interest = preprocessed_data['user_interest']
content_data = preprocessed_data['content']
weighted_average = collaborative_filtering(user_interest, content_data)

print("影响力得分：", influence_scores)
print("协同过滤推荐：", weighted_average)
