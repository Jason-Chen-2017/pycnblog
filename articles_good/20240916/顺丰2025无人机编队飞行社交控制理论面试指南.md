class Drone:
    def __init__(self, x, y, v):
        self.x = x
        self.y = y
        self.v = v

    def update(self, dt):
        self.x += self.v * np.cos(self.y) * dt
        self.y += self.v * np.sin(self.y) * dt

class DroneTeam:
    def __init__(self, drones):

    def update(self, dt):
        for drone in self.drones:
            drone.update(dt)

    def plot(self):
        plt.figure()
        for drone in self.drones:
            plt.plot(drone.x, drone.y, 'o')
        plt.show()

drones = [

    Drone(10, 0, 1),
    Drone(10, 10, 1),
    Drone(0, 10, 1),
]

team = DroneTeam(drones)

dt = 0.1
for _ in range(100):
