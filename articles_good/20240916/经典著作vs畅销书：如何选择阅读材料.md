def calculate_utility(depth, width, practicality, interest):
    return depth * 0.4 + width * 0.3 + practicality * 0.2 + interest * 0.1

def select_books(books, depth_weights, width_weights, practicality_weights, interest_weights):
    utilities = []
    for book in books:

import numpy as np

# 定义书籍类
class Book:
    def __init__(self, name, depth, width, practicality, interest):
        self.name = name
        self.depth = depth
        self.width = width
        self.practicality = practicality
        self.interest = interest

# 构建效用函数
def calculate_utility(book, weights):
    utility = (weights[0] * book.depth + weights[1] * book.width +
               weights[2] * book.practicality + weights[3] * book.interest)
    return utility

# 优先级排序函数
def select_books(books, weights):
    utilities = [calculate_utility(book, weights) for book in books]
    sorted_books = np.array(books)[np.argsort(utilities)]
    return sorted_books

# 参数设置
weights = [0.3, 0.2, 0.3, 0.2]  # 深度、广度、实用性和兴趣的权重

# 书籍实例
books = [

    Book('畅销书C', 0.6, 0.9, 0.8, 0.8)
]

# 选择阅读材料
sorted_books = select_books(books, weights)

# 打印排序结果
for book in sorted_books:
    print(book.name)
