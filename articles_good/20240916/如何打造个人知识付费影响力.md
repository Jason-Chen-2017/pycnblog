### 5.2 源代码详细实现

```python
import requests
import json
import tkinter as tk
from tkinter import ttk

# 登录功能
def login(username, password):
    url = "https://example.com/login"
    payload = {
        "username": username,
        "password": password
    }
    response = requests.post(url, data=payload)
    if response.status_code == 200:
        return json.loads(response.text)["token"]
    else:
        return None

# 购买课程
def buy_course(course_id, token):
    url = f"https://example.com/course/{course_id}/buy"
    headers = {
        "Authorization": f"Bearer {token}"
    }
    response = requests.post(url, headers=headers)
    if response.status_code == 200:
        return json.loads(response.text)["message"]
    else:
        return "购买失败"

# 主界面
root = tk.Tk()
root.title("知识付费系统")

# 创建表单
frame = ttk.Frame(root, padding="10")
frame.pack(fill="both", expand=True)

username_label = ttk.Label(frame, text="用户名：")
username_label.grid(row=0, column=0, padx=5, pady=5)
username_entry = ttk.Entry(frame)
username_entry.grid(row=0, column=1, padx=5, pady=5)

password_label = ttk.Label(frame, text="密码：")
password_label.grid(row=1, column=0, padx=5, pady=5)
password_entry = ttk.Entry(frame, show="*")
password_entry.grid(row=1, column=1, padx=5, pady=5)

login_button = ttk.Button(frame, text="登录", command=lambda: login(username_entry.get(), password_entry.get()))
login_button.grid(row=2, column=0, columnspan=2, padx=5, pady=5)

course_id_label = ttk.Label(frame, text="课程ID：")
course_id_label.grid(row=3, column=0, padx=5, pady=5)
course_id_entry = ttk.Entry(frame)
course_id_entry.grid(row=3, column=1, padx=5, pady=5)

buy_button = ttk.Button(frame, text="购买", command=lambda: buy_course(course_id_entry.get(), token))
buy_button.grid(row=4, column=0, columnspan=2, padx=5, pady=5)

# 运行主界面
root.mainloop()
```

### 5.3 代码解读与分析

1. **登录功能**：通过requests库向登录接口发送POST请求，获取token。
2. **购买课程**：通过requests库向购买接口发送POST请求，实现购买功能。
3. **主界面**：使用tkinter库创建用户界面，包括登录、购买等操作。
