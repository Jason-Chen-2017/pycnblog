```jsx
import React, { useState, useEffect } from 'react';
import axios from 'axios';

const PrivateBoard = () => {
  const [content, setContent] = useState([]);

  useEffect(() => {

      setContent(response.data);
    };
    fetchData();

  return (

      {content.map((item) => (

      ))}

  );
};

export default PrivateBoard;
```
