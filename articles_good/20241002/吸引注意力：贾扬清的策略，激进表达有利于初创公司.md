def analyze_user_profile(df):
    # 年龄分布
    age_distribution = df['年龄'].value_counts()

    # 性别比例
    gender_ratio = df['性别'].value_counts(normalize=True)
    print("性别比例：", gender_ratio)
    
    # 地域分布
    region_distribution = df['地域'].value_counts()
    print("地域分布：", region_distribution)
    
    # 职业分布
    job_distribution = df['职业'].value_counts()
    print("职业分布：", job_distribution)

# 精准营销策略
def precise_marketing_strategy(df):
    # 根据年龄推荐产品
    age_segment = df['年龄'].mean()
    if age_segment < 30:
        print("推荐产品：学生产品")
    elif age_segment < 40:
        print("推荐产品：职场产品")
    else:
        print("推荐产品：老年产品")

# 执行用户画像分析和精准营销策略
analyze_user_profile(df)
precise_marketing_strategy(df)
