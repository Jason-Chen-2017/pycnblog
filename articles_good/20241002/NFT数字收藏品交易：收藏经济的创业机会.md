import "@openzeppelin/contracts/access/Ownable.sol";

contract NFTMarketplace is ERC721, Ownable {
    mapping(uint256 => address) private _owners;
    mapping(address => uint256) private _balances;
    mapping(uint256 => bool) private _isToken Listed;
    mapping(uint256 => uint256) private _listPrices;

    event NFTListed(uint256 tokenId, uint256 price);
    event NFTBought(uint256 tokenId, address buyer);

    constructor() ERC721("NFT Marketplace", "NFTM") {}

    function listNFT(uint256 tokenId, uint256 price) external {
        require(_owners[tokenId] == msg.sender, "Not owner of the NFT");
        require(_isToken[tokenId] == false, "NFT already listed");
        _isToken[tokenId] = true;
        _listPrices[tokenId] = price;
        emit NFTListed(tokenId, price);
    }

    function buyNFT(uint256 tokenId) external payable {
        require(_isToken[tokenId] == true, "NFT not listed");
        require(msg.value >= _listPrices[tokenId], "Insufficient payment");
        _isToken[tokenId] = false;
        _transferOwnership(msg.sender, tokenId);
        _balances[msg.sender]++;
        emit NFTBought(tokenId, msg.sender);
    }

    function getNFTListPrice(uint256 tokenId) external view returns (uint256) {
        return _listPrices[tokenId];
    }

    function getNFTOwner(uint256 tokenId) external view returns (address) {
        return _owners[tokenId];
    }
}
