在马尔可夫过程中,随机变量{X_n}的取值只与前面的随机变量{X}_{n-1}有关,而与具体的取值无关。这使得马尔可夫过程具有无后效性的特点。同时,马尔可夫过程具有马尔可夫性质,即随机变量{X_n}的 future values{X}_{n+k}只与 current values{X_n}有关,而与具体的取值无关。

import numpy as np

# 定义马尔可夫过程的参数

beta = 0.4  # β 控制扩散概率
gamma = 0.1  # γ 控制激励项概率

# 定义随机变量的取值范围

 discrete_gamma = np.arange(0, 1, 0.1)

# 定义随机变量的概率分布
def probability_distribution(discrete_alpha, discrete_beta, discrete_gamma):
    return (
        (alpha ** discrete_alpha) / (alpha ** discrete_alpha + np.exp(-gamma * (discrete_beta ** 2 / 2))),
        (1 - alpha) * (1 - beta) * (1 - gamma) ** discrete_alpha,
        (1 - alpha) * (beta ** discrete_alpha) * (1 - gamma) ** (discrete_alpha - 1),
        (alpha ** (discrete_alpha - 1)) * (1 - gamma) ** (discrete_alpha - 1),
        (1 - alpha) * (1 - beta) * (beta ** (discrete_alpha - 1)) * (1 - gamma) ** (discrete_alpha - 1),
    )

# 定义随机变量的数学期望和方差
def expected_value(discrete_alpha, discrete_beta, discrete_gamma):
    return np.sum(probability_distribution(discrete_alpha, discrete_beta, discrete_gamma) * x) / np.sum(probability_distribution(discrete_alpha, discrete_beta, discrete_gamma))

def variance(discrete_alpha, discrete_beta, discrete_gamma):
    return np.sum((probability_distribution(discrete_alpha, discrete_beta, discrete_gamma) * (np.log(probability_distribution(discrete_alpha, discrete_beta, discrete_gamma)))) ** 2) / np.sum(probability_distribution(discrete_alpha, discrete_beta, discrete_gamma))

# 定义随机变量的中心极限定理
def central_limit_theorem(discrete_alpha, discrete_beta, discrete_gamma):
    return (np.sqrt(np.sum((discrete_alpha - expected_value(discrete_alpha, discrete_beta, discrete_gamma)) ** 2)) / (np.sqrt(np.sum(probability_distribution(discrete_alpha, discrete_beta, discrete_gamma)))) ** 0.5) * np.exp(-(discrete_alpha - expected_value(discrete_alpha, discrete_beta, discrete_gamma)) ** 2 / (2 * variance(discrete_alpha, discrete_beta, discrete_gamma))))

# 基于马尔可夫过程的随机过程建模

discrete_beta = discrete_beta[:-1]

# 利用中心极限定理计算随机变量的3倍方差

# 输出结果
