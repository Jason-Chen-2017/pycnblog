import re
import nltk
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer

def preprocess(text):
    # 去除HTML标签
    text = re.sub('<.*?>', '', text)
    # 去除停用词
    stop_words = set(stopwords.words('english'))
    words = [word for word in text.lower().split() if word not in stop_words]
    # 词干化
    lemmatizer = WordNetLemmatizer()
    words = [lemmatizer.lemmatize(word) for word in words]
    return''.join(words)
```

import requests
from bs4 import BeautifulSoup
import numpy as np

def get_recommendations(url):
    # 发送请求，获取推荐内容
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    # 解析推荐内容
    items = soup.find_all('div', {'class': 'item'})
    recommendations = []
    for item in items:
        # 提取推荐内容
        content = item.find('a', {'class': 'text'})['h3']
        # 去除HTML标签
        content = re.sub('<.*?>', '', content)
        # 去除停用词
        stop_words = set(stopwords.words('english'))
        words = [word for word in content.lower().split() if word not in stop_words]
        # 词干化
        lemmatizer = WordNetLemmatizer()
        words = [lemmatizer.lemmatize(word) for word in words]
        # 计算推荐分数
        score = 0.0
        for word in words:
            score += len(word) / (len(words) + 10)
        # 取平均推荐分数
        recommendations.append({
            'content_id': '1234567890',
           'recommendation': f'Score: {score:.2f}',
            'user_id': '1234567890'
        })
    return recommendations
```
