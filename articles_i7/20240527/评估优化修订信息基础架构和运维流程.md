### 5.2 遗传算法求解 TSP 问题

```python
import numpy as np

def fitness(route):
    # 计算路径长度作为适应度函数
    return -np.sum(dist_matrix[route[:-1], route[1:]])

def selection(population, scores, k=3):
    # 锦标赛选择
    selection_ix = np.random.randint(len(population))
    for ix in np.random.randint(0, len(population), k-1):
        if scores[ix] < scores[selection_ix]:
            selection_ix = ix
    return population[selection_ix]

def crossover(parent1, parent2):
    # 有序交叉
    start, end = np.sort(np.random.randint(0, len(parent1), 2))
    child = [-1] * len(parent1)
    child[start:end] = parent1[start:end]
    for i in parent2:
        if i not in child:
            for j in range(len(child)):
                if child[j] == -1:
                    child[j] = i
                    break
    return child

def mutation(route, prob=0.01):
    # 交换变异
    for i in range(len(route)):
        if np.random.rand() < prob:
            j = np.random.randint(0, len(route))
            route[i], route[j] = route[j], route[i]
    return route

# 遗传算法主循环
def ga(dist_matrix, n_population=100, n_iter=1000):
    population = [np.random.permutation(len(dist_matrix)) for _ in range(n_population)]
    best = None
    for _ in range(n_iter):
        scores = [fitness(route) for route in population]
        if best is None or scores[-1] > fitness(best):
            best = population[-1]
        selected = [selection(population, scores) for _ in range(n_population - 1)]
        children = []
        for i in range(0, n_population - 1, 2):
            child1, child2 = crossover(selected[i], selected[i+1])
            child1 = mutation(child1)
            child2 = mutation(child2)
            children.append(child1)
            children.append(child2)
        population = children + [best]
    return best

# 测试
dist_matrix = np.random.rand(20, 20)
dist_matrix = (dist_matrix + dist_matrix.T) / 2
np.fill_diagonal(dist_matrix, 0)
route = ga(dist_matrix)
print(route)
```

### 5.3 最短路径算法的 C++ 实现

```cpp
#include <iostream>
#include <vector>
#include <queue>
using namespace std;

typedef pair<int, int> PII;

const int INF = 1e9;

vector<vector<PII>> adj;

void dijkstra(int s, vector<int> & d) {
    d.assign(adj.size(), INF);
    d[s] = 0;
    priority_queue<PII, vector<PII>, greater<PII>> pq;
    pq.push({0, s});
    while (!pq.empty()) {
        int v = pq.top().second;
        int d_v = pq.top().first;
        pq.pop();
        if (d_v != d[v]) continue;
        for (auto edge : adj[v]) {
            int to = edge.first;
            int len = edge.second;
            if (d[v] + len < d[to]) {
                d[to] = d[v] + len;
                pq.push({d[to], to});
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;
    adj.resize(n);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    vector<int> d;
    dijkstra(0, d);
    for (int i = 0; i < n; i++) {
        cout << d[i] << " ";
    }
    return 0;
}
```

## 9. 附录：常见问题与解答

### Q1: 如何选择合适的评估指标?
A1: 评估指标的选择要结合业务目标、技术特点、数据可获得性等因素综合考虑。常见的指标包括性能、可用性、安全性、成本等。

### Q2: 优化和修订应该多久进行一次?
A2: 优化和修订的频率取决于业务和技术环境的变化速度。一般建议每季度或每半年进行一次全面评估,同时建立定期review机制。

### Q3: 如何平衡优化成本和收益?
A3: 可以通过成本效益分析来权衡优化的投入产出比。对于收益大、风险小的优化项可以优先推进。

### Q4: 优化和修订过程中如何控制风险?
A4: 风险控制的关键是渐进式变更和充分验证。可以先在测试环境进行验证,再逐步推广到生产环境。同时要制定完善的回滚预案。

### Q5: 如何处理优化过程中的阻力?
A5: 优化和修订往往涉及多方利益,需要通过有效沟通来达成共识。要让所有相关方了解优化的必要性和收益,并给予一定的激励。
