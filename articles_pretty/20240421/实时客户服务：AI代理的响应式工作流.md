```python
# 导入所需的库
import numpy as np
from sklearn.linear_model import LogisticRegression
from sklearn.feature_extraction.text import CountVectorizer

# 创建数据
questions = ["How can I reset my password?", "What is the latest version?", "How to cancel my subscription?"]
labels = ["password", "version", "cancel"]
vectorizer = CountVectorizer()
X = vectorizer.fit_transform(questions)
y = np.array(labels)

# 训练模型
clf = LogisticRegression()
clf.fit(X, y)

# 使用模型预测新的问题
new_question = ["How to update my account?"]
X_new = vectorizer.transform(new_question)
y_pred = clf.predict(X_new)
print(y_pred)
```

以上代码首先使用`CountVectorizer`将问题转换为向量，然后使用逻辑回归模型进行训练。训练完成后，我们可以使用模型预测新的问题。
