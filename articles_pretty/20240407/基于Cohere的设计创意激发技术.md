from collections import defaultdict
from sklearn.metrics.pairwise import cosine_similarity

# 初始化Cohere API客户端
co = cohere.Client('YOUR_API_KEY_HERE')

# 定义创意素材库
creative_materials = [
    "Sleek and modern office furniture design",
    "Innovative eco-friendly packaging solutions",
    "Minimalist branding for a high-tech startup",
    "Interactive virtual reality experience for gaming",
    "Captivating social media campaign for a new fashion line",
    # 添加更多创意素材
]

def get_creative_ideas(design_brief):
    """
    根据设计需求,生成创意点子
    """
    # 使用Cohere API计算设计需求与创意素材的语义相似度
    design_brief_embed = co.embed(texts=[design_brief]).embeddings[0]
    material_embeds = [co.embed(texts=[material]).embeddings[0] for material in creative_materials]
    similarities = cosine_similarity([design_brief_embed], material_embeds)[0]

    # 找出最相关的创意素材
    top_material_idxs = similarities.argsort()[-3:]
    top_materials = [creative_materials[idx] for idx in top_material_idxs]

    # 使用Cohere API生成联想概念
    ideas = []
    for material in top_materials:
        prompt = f"Based on the design brief '{design_brief}' and the creative material '{material}', generate 3 creative design ideas."
        response = co.generate(prompt=prompt, max_tokens=100, num_generate=3, temperature=0.7)
        ideas.extend(response.generations)

    return ideas

# 示例用法
design_brief = "Design an engaging website for a sustainable fashion brand"
creative_ideas = get_creative_ideas(design_brief)
for idea in creative_ideas:
    print(idea.text)

1. 使用Cohere API计算设计需求与创意素材的语义相似度,找出最相关的3个创意素材。
2. 对于每个top创意素材,使用Cohere API生成3个基于该素材和设计需求的创意点子。
3. 将生成的创意点子返回给用户。
