### 4.1 基于资源的调度算法实例

```python
import threading
import time

class Task:
    def __init__(self, name, resource_consumption):
        self.name = name
        self.resource_consumption = resource_consumption

class Node:
    def __init__(self, id, resource_status):
        self.id = id
        self.resource_status = resource_status

def SJF_scheduler(tasks, nodes):
    tasks.sort(key=lambda x: x.resource_consumption)
    for task in tasks:
        for node in nodes:
            if node.resource_status >= task.resource_consumption:
                node.resource_status -= task.resource_consumption
                task.status = 'running'
                threading.Thread(target=task.run).start()
                break

tasks = [Task('task1', 10), Task('task2', 5), Task('task3', 15)]
nodes = [Node(1, 20), Node(2, 15), Node(3, 10)]
SJF_scheduler(tasks, nodes)
```
