```python
class Road:
    def __init__(self, id, name, type, speed_limit):
        self.id = id
        self.name = name
        self.type = type
        self.speed_limit = speed_limit

class RoadNetwork:
    def __init__(self):
        self.roads = []

    def add_road(self, road):
        self.roads.append(road)

    def get_road(self, id):
        for road in self.roads:
            if road.id == id:
                return road
        return None

# 创建道路网络
network = RoadNetwork()

# 创建道路
road1 = Road(1, "A Street", "residential", 30)
road2 = Road(2, "B Street", "highway", 70)

# 添加道路到道路网络
network.add_road(road1)
network.add_road(road2)

# 获取道路
road = network.get_road(1)
print(road.name)  # 输出：A Street
```
