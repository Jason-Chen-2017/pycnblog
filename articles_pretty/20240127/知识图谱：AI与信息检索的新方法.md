# 创建一个图
graph = defaultdict(list)
graph['艾伦·埃德勒'].append('澳大利亚足球运动员')

# 定义DFS算法
def dfs(graph, node, visited):
    visited.add(node)
    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs(graph, neighbor, visited)

# 使用DFS算法来找到与“艾伦·埃德勒”和“澳大利亚足球运动员”相关的信息
def find_related_info(graph, entity):
    visited = set()
    dfs(graph, entity, visited)
    return visited

# 测试代码
