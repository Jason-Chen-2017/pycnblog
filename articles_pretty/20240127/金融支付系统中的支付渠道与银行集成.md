```python
import hashlib
import hmac
import json

def encrypt_data(data, key):
    """
    加密数据
    """
    encoded_data = json.dumps(data).encode('utf-8')
    h = hmac.new(key.encode('utf-8'), encoded_data, hashlib.sha256)
    return h.hexdigest()

def decrypt_data(data, key):
    """
    解密数据
    """
    encoded_data = json.dumps(data).encode('utf-8')
    h = hmac.new(key.encode('utf-8'), encoded_data, hashlib.sha256)
    return h.hexdigest()

data = {
    "amount": 100,
    "currency": "USD",
    "bank_account": "1234567890"
}

key = "my_secret_key"

encrypted_data = encrypt_data(data, key)
decrypted_data = decrypt_data(encrypted_data, key)

print(encrypted_data)
print(decrypted_data)
```

在这个代码实例中，我们使用了Python语言来实现支付系统与银行系统之间的资金交易。我们使用了HMAC算法来对数据进行加密和解密，以确保数据的安全传输。在这个例子中，我们定义了一个`encrypt_data`函数来对数据进行加密，并定义了一个`decrypt_data`函数来对数据进行解密。我们使用了HMAC算法来对数据进行加密和解密，以确保数据的安全传输。
