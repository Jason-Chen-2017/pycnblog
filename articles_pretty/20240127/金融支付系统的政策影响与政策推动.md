```python
import hmac
import base64

def encrypt(data, key):
    return base64.b64encode(hmac.new(key, data, hashlib.sha256).digest())

def decrypt(data, key):
    return base64.b64decode(data).decode('utf-8')

def verify(data, key):
    return hmac.compare_digest(data, encrypt(data, key))

def pay(amount, key):
    data = f"{amount}USD"
    encrypted_data = encrypt(data, key)
    return encrypted_data

key = "mysecretkey"
amount = 100
encrypted_amount = pay(amount, key)
print(encrypted_amount)
```
