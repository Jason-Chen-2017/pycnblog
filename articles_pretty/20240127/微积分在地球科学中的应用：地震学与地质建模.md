```python
import numpy as np

def wave_speed(density, elastic_modulus, poisson_ratio):
    """
    Calculate the wave speed in a medium.
    """
    v_p = np.sqrt((elastic_modulus / (density * (1 - poisson_ratio**2))))
    return v_p

# Example usage
density = 2.5e3  # kg/m^3
elastic_modulus = 3e11  # Pa
poisson_ratio = 0.25

v_p = wave_speed(density, elastic_modulus, poisson_ratio)
print("P-wave speed:", v_p, "m/s")
```

```python
import numpy as np
import scipy.linalg

def solve_heat_equation(domain, time, dt, T0, Q):
    """
    Solve the heat equation to obtain the temperature distribution.
    """
    N = domain.shape[0]
    A = scipy.linalg.toeplitz(np.ones(N), -1)
    b = np.ones(N)
    c = -Q
    M = np.eye(N)
    K = A + M
    L = np.linalg.solve(K, b)
    x = np.zeros((time, N))
    x[0] = T0
    for t in range(1, time):
        x[t] = np.dot(L, x[t-1]) + c * dt
    return x

# Example usage
domain = np.linspace(0, 1, 100)  # 1D domain
time = 1000  # Time in seconds
dt = 1  # Time step
T0 = 20  # Initial temperature
Q = 1e-3  # Heat source

x = solve_heat_equation(domain, time, dt, T0, Q)
```
