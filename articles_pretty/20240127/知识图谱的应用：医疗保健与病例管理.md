```python

from umls import Disease
from medline import Patient

# 查找药物的相关信息
drug = Drug.search("ibuprofen")
print(drug.name, drug.side_effects)

# 查找疾病的相关信息
disease = Disease.search("diabetes")
print(disease.name, disease.symptoms)

# 查找患者的病历信息

print(patient.name, patient.medical_history)
```
