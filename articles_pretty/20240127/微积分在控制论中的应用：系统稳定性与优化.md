```python
import numpy as np
import matplotlib.pyplot as plt

# 定义系统动态方程
def system_dynamics(t, y):
    return -2 * y + 3 * t

# 定义微分方程
def differential_equation(t, y, dy_dt):
    return dy_dt - system_dynamics(t, y)

# 初始条件
t0, y0 = 0, 1
t_end, dt = 10, 0.1

# 求解微分方程
t_values = np.arange(t0, t_end, dt)
y_values = np.zeros_like(t_values)
y_values[0] = y0

for i in range(1, len(t_values)):
    t = t_values[i]
    y = y_values[i - 1]
    dy_dt = system_dynamics(t, y)
    y_values[i] = y + dy_dt * dt

# 绘制系统动态曲线
plt.plot(t_values, y_values)
plt.xlabel('时间')
plt.ylabel('系统值')
plt.title('系统动态曲线')
plt.show()
```
