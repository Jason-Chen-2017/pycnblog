import matplotlib.pyplot as plt

# 定义材料性能指标

def thermal_conductivity(conductivity, specific_heat, density):
    return conductivity / (specific_heat * density)

def electrical_conductivity(resistance):
    return 1 / resistance

# 计算材料性能指标

conductivity = 100
specific_heat = 500
density = 2000
thermal_conductivity_value = thermal_conductivity(conductivity, specific_heat, density)

resistance = 100
electrical_conductivity_value = electrical_conductivity(resistance)

# 绘制材料性能指标
