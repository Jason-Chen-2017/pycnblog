```python
import networkx as nx
import numpy as np

# 创建知识图谱
G = nx.DiGraph()

# 添加交通设施
G.add_node("A", flow=100)
G.add_node("B", flow=200)
G.add_node("C", flow=300)

# 添加关系
G.add_edge("A", "B", capacity=200)
G.add_edge("B", "C", capacity=200)

# 预测交通流量
def predict_traffic(G, source, target, capacity):
    flow = 0
    while flow < capacity:
        flow += 1
        if nx.maximum_flow(G, source=source, target=target, capacity=flow) == flow:
            break
    return flow

# 输出预测结果
print(predict_traffic(G, "A", "C", 500))
```
