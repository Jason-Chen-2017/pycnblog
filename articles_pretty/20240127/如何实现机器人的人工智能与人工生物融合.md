```python
from scipy.optimize import minimize

# 定义目标函数
def objective_function(x):
    return x[0]**2 + x[1]**2

# 定义约束条件
def constraint(x):
    return x[0] + x[1] - 10

# 定义初始值
x0 = [0, 0]

# 调用优化算法
result = minimize(objective_function, x0, constraints=constraint)
```
