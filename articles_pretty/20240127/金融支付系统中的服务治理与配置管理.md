```yaml

kind: VirtualService
metadata:
  name: hello-world
spec:
  hosts:
  - hello-world
  gateways:
  - hello-world-gateway
  http:
  - match:
    - uri:
        exact: /hello
    route:
    - destination:
        host: hello-world
```
