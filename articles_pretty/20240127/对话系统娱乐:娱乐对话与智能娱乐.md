### 4.2 智能娱乐实例

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 音乐推荐系统
def recommend_music(user_preferences, music_data):
    # 创建TF-IDF向量化器
    vectorizer = TfidfVectorizer()

    # 将用户喜好和音乐数据转换为向量
    user_vector = vectorizer.fit_transform([user_preferences])
    music_matrix = vectorizer.fit_transform(music_data)

    # 计算音乐之间的相似度
    similarity_matrix = cosine_similarity(music_matrix, music_matrix)

    # 根据用户喜好推荐音乐
    recommended_indices = (similarity_matrix * user_vector).sum(axis=1)
    recommended_indices = recommended_indices.argsort()[0][::-1]

    # 返回推荐的音乐列表
    return music_data[recommended_indices]

# 示例音乐数据
music_data = ['I Will Always Love You', 'Bohemian Rhapsody', 'Billie Jean', 'Hallelujah', 'Imagine']

# 示例用户喜好
user_preferences = 'I love Whitney Houston and Queen'

# 推荐音乐
recommended_music = recommend_music(user_preferences, music_data)
print(recommended_music)
```
