4. 将输出结果与原始输入数据进行比较，并得到最终结果。

import numpy as np

# 创建量子计算机
qc = QuantumCircuit(2, 2)

# 创建量子态
qc.h(0)

# 编译和执行量子计算机
qc = transpile(qc, Aer.get_backend('statevector_simulator'))
qobj = assemble(qc)
result = Aer.get_backend('statevector_simulator').run(qobj).result()

# 解析结果
