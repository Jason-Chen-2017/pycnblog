以下是一个C++类的例子：

```cpp
#include <iostream>
using namespace std;

class MyClass {
public:
    void print() {
        cout << "Hello, World!" << endl;
    }
};

int main() {
    MyClass myClass;
    myClass.print();
    return 0;
}
```
