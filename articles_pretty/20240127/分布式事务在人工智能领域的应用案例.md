```python
class DistributedTransaction:
    def __init__(self):
        self.locks = {}

    def acquire(self, key):
        if key not in self.locks:
            self.locks[key] = 0
        self.locks[key] += 1

    def release(self, key):
        self.locks[key] -= 1
        if self.locks[key] == 0:
            del self.locks[key]

    def execute(self, key, value):
        self.acquire(key)
        try:
            # 执行事务
            print(f"执行事务: {key} = {value}")
            return value
        finally:
            self.release(key)

# 使用分布式事务
transaction = DistributedTransaction()
result1 = transaction.execute("counter", 1)
result2 = transaction.execute("counter", 2)
result3 = transaction.execute("counter", 3)
print(f"最终结果: {result1 + result2 + result3}")
```
