```python
import numpy as np
import matplotlib.pyplot as plt

# 生成波动数据
np.random.seed(42)
t = np.linspace(0, 1, 100)
x = np.sin(2 * np.pi * t)
y = np.cos(2 * np.pi * t)

# 计算波动分析
dx = np.gradient(x, t)
dy = np.gradient(y, t)
d2x = np.gradient(dx, t)
d2y = np.gradient(dy, t)

# 绘制波动分析
plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.plot(t, x, label='x(t)')
plt.plot(t, dx, label='dx/dt')
plt.plot(t, d2x, label='d²x/dt²')
plt.legend()

plt.subplot(1, 2, 2)
plt.plot(t, y, label='y(t)')
plt.plot(t, dy, label='dy/dt')
plt.plot(t, d2y, label='d²y/dt²')
plt.legend()

plt.show()
```
