```python
# 连接MongoDB数据库

# 查询设备数据

# 根据设备数据实现自动化控制和管理

    collection.update_one({'device_id': '123456'}, {'$set': {'status': 'online'}})
    print('设备已经线上')
else:
    print('设备已经线上')
```
