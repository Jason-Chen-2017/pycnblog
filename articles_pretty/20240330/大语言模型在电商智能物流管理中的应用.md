```python
import torch.nn as nn

class DemandForecastModel(nn.Module):
    def __init__(self, vocab_size, embed_dim, num_layers, num_heads, hidden_dim):
        super(DemandForecastModel, self).__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim)
        self.transformer = nn.Transformer(embed_dim, num_heads, num_layers, batch_first=True)
        self.fc = nn.Linear(embed_dim, 1)

    def forward(self, x):
        embedded = self.embedding(x)
        output, _ = self.transformer(embedded, embedded)
        forecast = self.fc(output[:, -1, :])
        return forecast
```
