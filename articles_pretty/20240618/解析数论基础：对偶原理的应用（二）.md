2. **验证对偶定理**：通过证明这个新陈述，实际上需要证明的是，对于任意\\(n\\)和\\(b\\)（\\(n > b\\)且\\(n\\)和\\(b\\)互质），\\(n^b\\)除以\\(b\\)的余数为1。

import sympy

def euler_theorem_check(a, n):
    phi_n = sympy.euler_phi(n)
    if sympy.gcd(a, n) == 1:
        return pow(a, phi_n, n) == 1
    else:
        return False

def dual_euler_theorem_check(b, n):
    return pow(n, b, b) == 1

# 示例
a = 3

b = 5
