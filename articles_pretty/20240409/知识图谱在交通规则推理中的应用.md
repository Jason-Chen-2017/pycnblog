road = Node("Road", name="Main Street")

# 创建节点之间的关系
graph.create(road)
graph.create(vehicle)
graph.create(traffic_light)
graph.create(Relationship(vehicle, "ON", road))
graph.create(Relationship(traffic_light, "AT", road))

# 查询知识图谱

for item in result:
