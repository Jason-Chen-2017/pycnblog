$$
\begin{aligned}
&\text{每个节点至少有 }d\text{ 个子节点，至多有 }2d\text{ 个子节点（根节点除外）。}\\
&\text{每个非叶子节点至少有 }d\text{ 个键，至多有 }2d - 1\text{ 个键。}\\

\end{aligned}
$$

```python
class Disc:
    def __init__(self, name, capacity):
        self.name = name
        self.capacity = capacity
        self.used = 0
        self.data = {}

    def write(self, file_name, content):
        if self.used + len(content) > self.capacity:
            print("Not enough space!")
            return
        self.data[file_name] = content
        self.used += len(content)

    def read(self, file_name):
        if file_name in self.data:
            return self.data[file_name]
        else:
            print("File not found!")
            return None
```
