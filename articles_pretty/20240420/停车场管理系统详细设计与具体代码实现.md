import time

        self.license_plate = license_plate
        self.enter_time = time.time()

class ParkingLot:
    def __init__(self, size):
        self.size = size
        self.spaces = [0] * size

    def park(self, car):
        for i in range(self.size):
            if self.spaces[i] == 0:
                self.spaces[i] = 1
                return i
        return -1

    def leave(self, spot):
        if spot >= 0 and spot < self.size:
            self.spaces[spot] = 0

class FeeCalculator:
    def __init__(self, rate):
        self.rate = rate

    def calculate_fee(self, car):
        park_time = time.time() - car.enter_time
        return self.rate * park_time
