```python
# 设置GPIO的针脚编号

# 设置针脚为输入模式，用于读取传感器数据

# 循环读取传感器数据
while True:
    # 读取传感器数据
    data = GPIO.input(PIN)
    # 打印数据
    print("Sensor Data: ", data)
    # 等待1秒
    time.sleep(1)
```
