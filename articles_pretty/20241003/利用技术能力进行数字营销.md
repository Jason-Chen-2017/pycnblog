```python
from sklearn.metrics.pairwise import cosine_similarity

# 计算用户之间的相似度
user_similarity = cosine_similarity(data['user_id'], data['user_id'])

# 计算物品之间的相似度
item_similarity = cosine_similarity(data['item_id'], data['item_id'])

# 预测评分
def predict_rating(user_id, item_id):
    user_index = data[data['user_id'] == user_id].index[0]
    item_index = data[data['item_id'] == item_id].index[0]
    user_similarities = user_similarity[user_index]
    item_similarities = item_similarity[item_index]
    weighted_ratings = user_similarities * item_similarities
    return weighted_ratings.sum() / weighted_ratings.sum()
```
