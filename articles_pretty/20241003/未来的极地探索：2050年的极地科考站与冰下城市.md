import matplotlib.pyplot as plt

# 定义能量平衡方程
def energy_balance(E_solar, E_wind, E_geothermal):
    E_total = E_solar + E_wind + E_geothermal
    return E_total

# 定义自动化控制方程
def automation_control(e, K_p, K_i, K_d):
    y = K_p * e + K_i * np.trapz(e) + K_d * np.gradient(e)
    return y

# 定义材料性能方程
def material_performance(E, epsilon):
    sigma = E * epsilon
    return sigma

# 定义环境参数方程
def environment_parameters(T_0, alpha, beta, omega, phi, t):
    T = T_0 + alpha * t + beta * np.sin(omega * t + phi)
    return T

# 测试代码

E_wind = 50
E_geothermal = 20
E_total = energy_balance(E_solar, E_wind, E_geothermal)
print("Total Energy:", E_total)

K_p = 1
K_i = 0.1
K_d = 0.01

E = 10000
epsilon = 0.01
sigma = material_performance(E, epsilon)
print("Material Performance:", sigma)

T_0 = 20
alpha = 0.1
beta = 5
omega = 0.1
phi = 0

T = environment_parameters(T_0, alpha, beta, omega, phi, t)
plt.plot(t, T)
plt.xlabel('Time')
plt.ylabel('Temperature')
