#### 定义自伴算子

在离散场景中，自伴算子可以看作是节点之间的相互作用矩阵或邻接矩阵，满足\\(A = A^*\\)的性质，其中\\(A\\)是矩阵，\\(A^*\\)是\\(A\\)的共轭转置。

import numpy as np

class DiscreteVNAlgebra:
    def __init__(self, graph):
        self.graph = graph
        self.n = len(graph)

    def build_algebra(self):
        # 构建自伴矩阵集合
        self.A = np.array([np.eye(self.n) for _ in range(self.n)])
        for i, j in self.graph.edges():
            self.A[i][j] = self.A[j][i]

    def compute_product(self, index1, index2):
        # 计算两节点之间的相互作用矩阵乘积
        return np.dot(self.A[index1], self.A[index2])

# 示例图结构
