import numpy as np

def sl2_coadjoint_orbit():
    # 定义生成元
    e, f, h = np.array([1, 0]), np.array([-1, 0]), np.array([0, 1])
    
    # 计算李乘积
    def commutator(x, y):
        return np.cross(x, y)
    
    # 根的验证
    def is_root(x, y):
        return np.dot(commutator(e, f), x) == 2 * np.dot(h, x) and np.dot(commutator(e, f), y) == -2 * np.dot(h, y)
    
    # 构建根系
    roots = []
    for alpha in [e, f]:
        for beta in [e, f]:
            if is_root(alpha, beta):
                roots.append(alpha)
    
    return roots
