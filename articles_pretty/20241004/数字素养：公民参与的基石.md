### 5.2 源代码详细实现和代码解读
以下是一个简单的Python代码示例，展示了如何实现信息获取、评估和传播：

```python
import requests
from bs4 import BeautifulSoup
import re

def searchEngineQuery(query):
    url = f"https://www.google.com/search?q={query}"
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    results = soup.find_all('div', class_='tF2Cxc')
    return [result.text for result in results]

def evaluateSource(source):
    # 假设source是一个URL
    response = requests.get(source)
    soup = BeautifulSoup(response.text, 'html.parser')
    text = soup.get_text()
    if re.search(r'广告|赞助', text):
        return False
    return True

def factCheck(fact):
    # 假设fact是一个陈述
    if fact == "地球是平的":
        return False
    return True

def createContent(contentType, content):
    # 假设contentType是"blog"，content是博客内容
    with open(f"{contentType}.md", "w") as file:
        file.write(content)

def shareContent(platform, content):
    # 假设platform是"twitter"
    if platform == "twitter":
        print(f"Tweeting: {content}")
    else:
        print(f"Sharing on {platform}: {content}")

if __name__ == "__main__":
    query = "数字素养"
    results = searchEngineQuery(query)
    for result in results:
        if evaluateSource(result):
            if factCheck(result):
                createContent("blog", result)
                shareContent("twitter", result)
```

- **factCheck**：检查信息的准确性，例如检查是否为虚假信息。
- **createContent**：创建内容并保存为Markdown文件。
- **shareContent**：通过社交媒体分享内容。
