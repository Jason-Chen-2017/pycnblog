### 5.2 源代码详细实现和代码解读
我们将实现一个简单的迭代函数系统（IFS）生成分形几何形状的代码。具体代码如下：
```python
import numpy as np
import matplotlib.pyplot as plt

def ifs(x, y, A1, b1, A2, b2, A3, b3, A4, b4, p):
    """IFS算法实现"""
    r = np.random.rand()
    if r < p[0]:
        x, y = A1 @ np.array([x, y]) + b1
    elif r < p[0] + p[1]:
        x, y = A2 @ np.array([x, y]) + b2
    elif r < p[0] + p[1] + p[2]:
        x, y = A3 @ np.array([x, y]) + b3
    else:
        x, y = A4 @ np.array([x, y]) + b4
    return x, y

def generate_fractal(num_points, A1, b1, A2, b2, A3, b3, A4, b4, p):
    """生成分形几何形状"""
    x, y = 0, 0
    points = [(x, y)]
    for _ in range(num_points):
        x, y = ifs(x, y, A1, b1, A2, b2, A3, b3, A4, b4, p)
        points.append((x, y))
    return np.array(points)

def plot_fractal(points):
    """绘制分形几何形状"""
    plt.figure(figsize=(10, 10))
    plt.scatter(points[:, 0], points[:, 1], s=1, c='b')
    plt.axis('equal')
    plt.show()

# 定义迭代函数
A1 = np.array([[0.5, 0], [0, 0.5]])
b1 = np.array([0, 0])
A2 = np.array([[0.5, 0], [0, 0.5]])
b2 = np.array([0.5, 0.5])
A3 = np.array([[0.5, 0], [0, 0.5]])
b3 = np.array([0.5, 0])
A4 = np.array([[0.5, 0], [0, 0.5]])
b4 = np.array([0, 0.5])
p = [0.25, 0.25, 0.25, 0.25]

# 生成分形几何形状
points = generate_fractal(10000, A1, b1, A2, b2, A3, b3, A4, b4, p)

# 绘制分形几何形状
plot_fractal(points)
```
