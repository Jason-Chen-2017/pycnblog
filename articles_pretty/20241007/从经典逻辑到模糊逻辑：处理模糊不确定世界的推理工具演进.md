```python
import numpy as np
import skfuzzy as fuzz
from skfuzzy import control as ctrl

# 定义模糊集
temperature = ctrl.Antecedent(np.arange(0, 110, 1), 'temperature')
humidity = ctrl.Antecedent(np.arange(0, 101, 1), 'humidity')
comfort = ctrl.Consequent(np.arange(0, 11, 1), 'comfort')

# 定义隶属度函数
temperature['cold'] = fuzz.trimf(temperature.universe, [0, 0, 30])
temperature['mild'] = fuzz.trimf(temperature.universe, [20, 50, 80])
temperature['hot'] = fuzz.trimf(temperature.universe, [70, 100, 100])

humidity['low'] = fuzz.trimf(humidity.universe, [0, 0, 30])
humidity['medium'] = fuzz.trimf(humidity.universe, [20, 50, 80])
humidity['high'] = fuzz.trimf(humidity.universe, [70, 100, 100])

comfort['uncomfortable'] = fuzz.trimf(comfort.universe, [0, 0, 5])
comfort['comfortable'] = fuzz.trimf(comfort.universe, [2, 5, 8])
comfort['very_comfortable'] = fuzz.trimf(comfort.universe, [5, 10, 10])

# 定义模糊推理规则
rule1 = ctrl.Rule(temperature['cold'] & humidity['low'], comfort['uncomfortable'])
rule2 = ctrl.Rule(temperature['cold'] & humidity['medium'], comfort['comfortable'])
rule3 = ctrl.Rule(temperature['cold'] & humidity['high'], comfort['very_comfortable'])
rule4 = ctrl.Rule(temperature['mild'] & humidity['low'], comfort['comfortable'])
rule5 = ctrl.Rule(temperature['mild'] & humidity['medium'], comfort['very_comfortable'])
rule6 = ctrl.Rule(temperature['mild'] & humidity['high'], comfort['very_comfortable'])
rule7 = ctrl.Rule(temperature['hot'] & humidity['low'], comfort['very_comfortable'])
rule8 = ctrl.Rule(temperature['hot'] & humidity['medium'], comfort['very_comfortable'])
rule9 = ctrl.Rule(temperature['hot'] & humidity['high'], comfort['very_comfortable'])

# 创建控制系统
comfort_ctrl = ctrl.ControlSystem([rule1, rule2, rule3, rule4, rule5, rule6, rule7, rule8, rule9])

# 创建仿真
comfort_sim = ctrl.ControlSystemSimulation(comfort_ctrl)

# 设置输入值
comfort_sim.input['temperature'] = 30
comfort_sim.input['humidity'] = 50

# 运行仿真
comfort_sim.compute()

# 输出结果
print(comfort_sim.output['comfort'])
```
