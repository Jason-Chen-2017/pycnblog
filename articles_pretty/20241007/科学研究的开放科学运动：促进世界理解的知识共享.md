```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

    'A': np.random.randn(100),
    'B': np.random.randn(100)
})

# 数据分析

# 结果发布

# 透明度
plt.figure(figsize=(10, 6))
plt.plot(data['A'], label='A')
plt.plot(data['B'], label='B')
plt.legend()
plt.show()
```
