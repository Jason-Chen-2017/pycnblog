```python
import numpy as np
from scipy.optimize import minimize

# 输入数据
asset_returns = np.array([0.08, 0.12, 0.05, 0.10])
asset_risks = np.array([0.15, 0.20, 0.12, 0.18])
correlation_matrix = np.array([[1.0, 0.6, 0.4, 0.5],
                              [0.6, 1.0, 0.3, 0.7],
                              [0.4, 0.3, 1.0, 0.4],
                              [0.5, 0.7, 0.4, 1.0]])

# 均值-方差模型优化
def mean_variance_optimization(expected_return):
    def objective(weights):
        return np.dot(weights, np.dot(correlation_matrix * np.outer(asset_risks, asset_risks), weights))
    
    cons = ({'type': 'eq', 'fun': lambda x: np.dot(x, asset_returns) - expected_return},
           {'type': 'eq', 'fun': lambda x: np.sum(x) - 1})
    
    res = minimize(objective, [0.25, 0.25, 0.25, 0.25], method='SLSQP', constraints=cons)
    return res.x

# 调用优化函数
target_return = 0.09
optimal_weights = mean_variance_optimization(target_return)
print(f"Optimal portfolio weights: {optimal_weights}")
```

该代码实现了基于均值-方差模型的投资组合优化,输入了4个资产的预期收益率和风险,以及它们之间的相关性矩阵。通过优化算法,得到了满足目标收益率的最优资产权重分配。
