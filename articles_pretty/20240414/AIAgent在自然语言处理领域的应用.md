```python
from transformers import MarianMTModel, MarianTokenizer

# 加载预训练的多语言翻译模型
model = MarianMTModel.from_pretrained('Helsinki-NLP/opus-mt-en-zh')
tokenizer = MarianTokenizer.from_pretrained('Helsinki-NLP/opus-mt-en-zh')

# 定义翻译函数
def translate(text, src_lang='en', tgt_lang='zh'):
    # 编码输入文本
    input_ids = tokenizer.encode(text, return_tensors='pt')
    
    # 生成目标语言文本
    output_ids = model.generate(input_ids, max_length=100, num_beams=4, early_stopping=True)
    
    # 解码输出
    translated_text = tokenizer.decode(output_ids[0], skip_special_tokens=True)
    
    return translated_text

# 示例翻译
en_text = "Hello, how are you?"
zh_text = translate(en_text)
print("English:", en_text)
print("Chinese:", zh_text)
```
