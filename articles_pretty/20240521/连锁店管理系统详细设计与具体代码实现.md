```python
# 创建库存表
c.execute('''
    CREATE TABLE inventory (
        product_id INTEGER PRIMARY KEY,
        quantity INTEGER
    )
''')

# 添加商品
def add_product(product_id, quantity):
    c.execute('''
        INSERT INTO inventory (product_id, quantity)
        VALUES (?, ?)
    ''', (product_id, quantity))
    conn.commit()

# 出售商品
def sell_product(product_id, quantity):
    c.execute('''
        UPDATE inventory
        SET quantity = quantity - ?
        WHERE product_id = ?
    ''', (quantity, product_id))
    conn.commit()

# 检查库存

    c.execute('''
        SELECT quantity
        FROM inventory
        WHERE product_id = ?
    ''', (product_id,))
    quantity = c.fetchone()[0]
    return quantity
```
