```python
# 协同过滤算法示例代码
import numpy as np

# 用户-物品评分矩阵
ratings = np.array([
    [4, 0, 2, 0, 1],
    [0, 2, 3, 0, 0],
    [1, 0, 2, 4, 0],
    [5, 0, 0, 3, 1],
    [0, 0, 1, 5, 1]
])

# 计算用户相似度矩阵
user_similarity = np.corrcoef(ratings)

# 预测用户对物品的评分
def predict_rating(user_id, item_id):
    similar_users = np.argsort(user_similarity[user_id])[::-1][1:]
    similar_ratings = ratings[similar_users, item_id]
    similarity_scores = user_similarity[user_id][similar_users]
    
    pred_rating = np.dot(similar_ratings, similarity_scores) / np.sum(similarity_scores)
    return pred_rating

# 预测用户1对物品3的评分
user_id = 0
item_id = 2
pred_rating = predict_rating(user_id, item_id)
print(f"用户{user_id}对物品{item_id}的预测评分为：{pred_rating:.2f}")
```

以上示例代码展示了一个简单的协同过滤算法实现。通过计算用户之间的相似度，我们可以预测用户对未评分物品的评分。这种基于用户相似度的推荐方法，可以为学生推荐与其兴趣和学习风格相似的学习资源。
