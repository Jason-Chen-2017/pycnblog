### 4.1 余弦相似度

余弦相似度用于衡量两个用户或商品之间的相似程度，其计算公式为：

$$
\cos(\theta) = \frac{\vec{a} \cdot \vec{b}}{|\vec{a}| \cdot |\vec{b}|}

其中， $\vec{a}$ 和 $\vec{b}$ 分别表示两个用户或商品的特征向量。

import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

# 加载用户评分数据
ratings = pd.read_csv('ratings.csv')

# 计算用户相似度矩阵
user_similarity = cosine_similarity(ratings.pivot(index='userId', columns='movieId', values='rating'))

# 预测用户评分
def predict_rating(user_id, movie_id):
    # 获取与目标用户相似的用户
    similar_users = user_similarity[user_id].argsort()[::-1][1:]
    
    # 计算相似用户的平均评分
    similar_ratings = ratings[(ratings['userId'].isin(similar_users)) & (ratings['movieId'] == movie_id)]['rating']
    
    # 预测目标用户的评分
    predicted_rating = similar_ratings.mean()
    
    return predicted_rating
