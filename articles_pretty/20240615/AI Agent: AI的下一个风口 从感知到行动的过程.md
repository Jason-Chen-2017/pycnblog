```python
def update_Q_value(Q, state, action, reward, new_state, alpha, gamma):
    max_future_Q = max(Q[new_state].values())
    current_Q = Q[state][action]
    new_Q = (1 - alpha) * current_Q + alpha * (reward + gamma * max_future_Q)
    Q[state][action] = new_Q
```
