```python
import numpy as np
import matplotlib.pyplot as plt

def f(z):
    return np.sin(z)

def compute_divisor(f, x, y, eps=1e-6):
    n = len(x)
    div = []
    for i in range(n):
        p = x[i] + y[i]*1j
        z = p
        fz = f(z)
        if abs(fz) < eps:
            continue
        n = 0
        while abs(fz) > eps:
            z = z - fz / f.deriv()(z)
            fz = f(z)
            n += 1
        div.append((p, n))
    return div

x = np.linspace(-np.pi, np.pi, 100)
y = np.linspace(-np.pi, np.pi, 100)
X, Y = np.meshgrid(x, y)
Z = X + Y*1j
W = f(Z)

plt.contourf(X, Y, np.real(W), levels=20)
plt.colorbar()
plt.show()

div = compute_divisor(f, x, y)
print(div)
```
