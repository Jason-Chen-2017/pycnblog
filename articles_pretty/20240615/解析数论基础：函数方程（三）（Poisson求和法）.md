```python
import matplotlib.pyplot as plt

# 定义一个周期为 N 的函数
N = 100
def f(n):
    return np.sin(2*np.pi*n/N)

# 定义一个离散的数列
x = np.arange(N)
y = f(x)

# 对这个数列进行Poisson求和
F = np.fft.fft(y)
F = np.fft.fftshift(F)
F = F/N

# 将傅里叶变换得到的连续函数绘制出来
freq = np.fft.fftfreq(N)
freq = np.fft.fftshift(freq)
plt.plot(freq, np.abs(F))
plt.show()
```
