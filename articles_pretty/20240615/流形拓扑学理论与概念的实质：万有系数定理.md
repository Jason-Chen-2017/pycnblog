import scipy.sparse as sp
import scipy.sparse.linalg as spla

def cochain_complex(X, U):
    """
    构造一个复合复形
    """
    n = len(X)
    m = len(U)
    d = np.zeros((n, m))
    for i in range(n):
        for j in range(m):
            if X[i].intersection(U[j]):
                d[i, j] = 1
    D = sp.csr_matrix(d)
    C = D.T @ D
    return C

def cohomology(X, U):
    """
    计算同调群
    """
    C = cochain_complex(X, U)
    H = spla.eigsh(C, k=1, which='SM')
    return H[0]

def chern_class(E):
    """
    计算向量丛的第一陈类
    """
    X = E.base
    U = [E.local_trivialization(x) for x in X]
    c1 = cohomology(X, U)
    return c1

def subbundle(E, F):
    """
    计算向量丛的子丛
    """
    X = E.base
    U = [E.local_trivialization(x).intersection(F.local_trivialization(x)) for x in X]
    F = E.__class__(X, U)
    return F

def universal_coefficient_theorem(E):
    """
    计算万有系数定理
    """
    F = subbundle(E, E)
    c1 = chern_class(E)
    c2 = chern_class(F)
    return c1 / (2 * np.pi) - c2 / (4 * np.pi ** 2)
