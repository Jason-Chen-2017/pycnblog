# 定义一个算子

    return np.array([[1, 1], [1, 0]])

# 计算算子的谱截面
def spectral_section(A, num_sections):
    eigenvalues, eigenvectors = eigh(A)
    # 构建谱截面
    spectral_section = []
    for eigenvalue in eigenvalues:
        if eigenvalue.real > 0:
            eigenvector = eigenvectors[:, np.argmax(eigenvector.real)]

    return spectral_section

# 打印谱截面

    print(f"Section {i+1}:")
    print(eigenvector)
