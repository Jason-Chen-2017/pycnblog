## 4. 项目实践:代码实例和详细解释说明

以下是一个基于以太坊的供应链追溯系统的代码示例:

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract SupplyChainTracker {
    struct Product {
        string productId;
        string manufacturer;
        string currentOwner;
        uint256 timestamp;
    }

    mapping(string => Product) public products;

    function trackProduct(string memory _productId, string memory _manufacturer, string memory _currentOwner) public {
        require(bytes(products[_productId].productId).length == 0, "Product already exists");
        products[_productId] = Product(_productId, _manufacturer, _currentOwner, block.timestamp);
    }

    function getProductHistory(string memory _productId) public view returns (Product[] memory) {
        uint256 count = 0;
        for (uint256 i = 0; i < 100; i++) {
            if (bytes(products[string(abi.encodePacked(_productId, i.toString()))].productId).length > 0) {
                count++;
            }
        }

        Product[] memory history = new Product[](count);
        uint256 index = 0;
        for (uint256 i = 0; i < 100; i++) {
            string memory key = string(abi.encodePacked(_productId, i.toString()));
            if (bytes(products[key].productId).length > 0) {
                history[index++] = products[key];
            }
        }

        return history;
    }
}
```

该合约实现了产品的追踪功能,包括记录产品的ID、制造商、当前所有者以及时间戳信息。开发者可以通过`trackProduct`函数添加新的产品信息,通过`getProductHistory`函数查询产品的整个流转历史。
