## 4. 项目实践：代码实例和详细解释说明
以下是一个基于PyTorch的语言模型在社交媒体危机公关中的应用实例:

```python
import torch
import torch.nn as nn
import torch.optim as optim
from transformers import GPT2LMHeadModel, GPT2Tokenizer

# 1. 负面信息检测
class SentimentClassifier(nn.Module):
    def __init__(self, model_name):
        super().__init__()
        self.model = GPT2LMHeadModel.from_pretrained(model_name)
        self.classifier = nn.Linear(self.model.config.hidden_size, 2)

    def forward(self, input_ids):
        outputs = self.model(input_ids)[0]
        pooled_output = outputs[:, 0, :]
        logits = self.classifier(pooled_output)
        return logits

# 2. 公关回应生成
class CrisisResponseGenerator(nn.Module):
    def __init__(self, model_name):
        super().__init__()
        self.model = GPT2LMHeadModel.from_pretrained(model_name)
        self.tokenizer = GPT2Tokenizer.from_pretrained(model_name)

    def forward(self, input_text, max_length=100):
        input_ids = self.tokenizer.encode(input_text, return_tensors='pt')
        output_ids = self.model.generate(input_ids, max_length=max_length, num_return_sequences=1, do_sample=True, top_k=50, top_p=0.95, num_iterations=1)
        return self.tokenizer.decode(output_ids[0], skip_special_tokens=True)

# 3. 传播趋势预测
class TrendPredictionModel(nn.Module):
    def __init__(self, model_name):
        super().__init__()
        self.model = GPT2LMHeadModel.from_pretrained(model_name)
        self.fc = nn.Linear(self.model.config.hidden_size, 1)

    def forward(self, input_ids):
        outputs = self.model(input_ids)[0]
        pooled_output = outputs[:, 0, :]
        trend_score = self.fc(pooled_output)
        return trend_score
```

这些代码实现了语言模型在社交媒体危机公关中的三个关键功能:负面信息检测、公关回应生成和传播趋势预测。通过对这些模块的组合使用,企业可以快速有效地应对社交媒体危机事件,维护企业声誉。
