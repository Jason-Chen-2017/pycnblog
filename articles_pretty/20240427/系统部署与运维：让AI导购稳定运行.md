import pandas as pd
from sklearn.neighbors import NearestNeighbors

# 加载用户行为数据
data = pd.read_csv('user_behavior.csv')

# 构建用户-商品评分矩阵
ratings_matrix = data.pivot_table(index='user_id', columns='item_id', values='rating')

# 使用KNN算法计算用户相似度
model = NearestNeighbors(n_neighbors=10)
model.fit(ratings_matrix)

# 获取目标用户的相似用户
user_id = 123
distances, neighbors = model.kneighbors(ratings_matrix.loc[user_id].values.reshape(1, -1))

# 获取相似用户喜欢的商品
recommendations = []
for neighbor in neighbors[0]:
    recommendations.extend(ratings_matrix.loc[neighbor].sort_values(ascending=False).index.tolist())

# 推荐排名前10的商品
top_recommendations = list(set(recommendations))[:10]

print(top_recommendations)
