我们可以编写如下代码：

```python
class ShoppingCart:
    def __init__(self):
        self.items = []

    def add_item(self, item):
        self.items.append(item)

    def remove_item(self, item):
        self.items.remove(item)

class View:
    def display_items(self, items):
        for item in items:
            print(item)

class Controller:
    def __init__(self, model, view):
        self.model = model
        self.view = view

    def add_item(self, item):
        self.model.add_item(item)
        self.view.display_items(self.model.items)

    def remove_item(self, item):
        self.model.remove_item(item)
        self.view.display_items(self.model.items)

cart = ShoppingCart()
view = View()
controller = Controller(cart, view)
controller.add_item("apple")
controller.add_item("banana")
controller.remove_item("apple")
```
