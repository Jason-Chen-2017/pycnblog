                self.resources[i] -= demand

class TaskScheduler:
    def __init__(self):
        self.tasks = []

    def schedule(self, task):
        priority = task.priority / task.demand
        self.tasks.append((priority, task))
        self.tasks.sort(reverse=True)

class Task:
