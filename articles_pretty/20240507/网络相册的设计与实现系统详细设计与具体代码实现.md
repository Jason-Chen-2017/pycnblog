## 5. 项目实践：代码实例与详解
### 5.1 前端代码实现
#### 5.1.1 图片上传组件
```html
<template>
  <div>
    <input type="file" @change="handleFileChange" />
    <button @click="uploadFile">上传</button>
  </div>
</template>

<script>
export default {
  methods: {
    handleFileChange(e) {
      this.file = e.target.files[0];
    },
    uploadFile() {
      let formData = new FormData();
      formData.append("file", this.file);
      // 调用上传API
      axios.post("/api/upload", formData);
    }
  }
};
</script>
```
#### 5.1.2 图片展示组件
```html
<template>
  <div class="photo-wall">
    <img v-for="(img, index) in imgs" :src="img.url" :key="index" @click="openModal(index)">
  </div>
</template>

<script>
export default {
  data() {
    return {
      imgs: []
    }
  },
  methods: {
    getPhotos() {
      // 调用获取图片列表API
      axios.get("/api/photos").then(res => {
        this.imgs = res.data;
      });
    },
    openModal(index) {
      // 打开图片详情弹窗
      this.$refs.modal.open(this.imgs[index]);
    }
  },
  mounted() {
    this.getPhotos();
  }
};
</script>
```
### 5.2 后端代码实现
#### 5.2.1 图片上传接口
```java
@PostMapping("/upload")
public Result upload(@RequestParam("file") MultipartFile file) {
    // 文件存储逻辑
    String url = fileService.save(file);
    
    Photo photo = new Photo();
    photo.setUrl(url);
    photo.setUploadTime(new Date());
    // 保存图片信息到数据库
    photoService.save(photo);

    return Result.ofSuccess(url);
}
```
#### 5.2.2 图片列表接口
```java
@GetMapping("/photos")
public Result getAll() {
    // 查询图片列表
    List<Photo> photos = photoService.getAll();
    return Result.ofSuccess(photos);
}
```
