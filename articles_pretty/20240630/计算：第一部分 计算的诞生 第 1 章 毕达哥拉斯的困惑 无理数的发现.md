def calculate_pi(n):
  """
  使用蒙特卡洛方法计算 π 的近似值。

  Args:
    n: 随机点数量。

  Returns:
    π 的近似值。
  """

  for i in range(n):
    x = random.uniform(0, 1)
    y = random.uniform(0, 1)
    distance = math.sqrt(x**2 + y**2)
    if distance <= 1:
      inside_circle += 1
  return 4 * (inside_circle / n)

# 计算 π 的近似值
n = 1000000
pi = calculate_pi(n)

* 然后，我们使用 `random.uniform(0, 1)` 生成 `n` 个随机点，每个点的坐标都在单位正方形内。
* 对于每个点，我们计算其到原点的距离，如果距离小于等于 1，则该点落在了单位圆内。
