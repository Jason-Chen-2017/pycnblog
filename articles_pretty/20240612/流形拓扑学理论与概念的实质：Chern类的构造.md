def chern_class(E, M):
    """
    Compute the Chern class of a vector bundle E over a manifold M.

    # Define a connection on E
    A = np.zeros((len(M), len(E), len(E)))
    for i in range(len(M)):
        A[i] = np.eye(len(E))
    # Compute the curvature of the connection

        F[i] = np.dot(A[i], A[i]) - np.dot(A[i], A[i]).T

    c = np.zeros(len(M))
    for i in range(len(M)):
        c[i] = np.trace(np.linalg.matrix_power(F[i], len(E))) / (2 * np.pi * 1j)
    return c
