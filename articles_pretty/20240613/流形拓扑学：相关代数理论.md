import numpy as np
import scipy.sparse as sp

def boundary_operator(dim, simplices):
    num_simplices = simplices.shape[0]
    num_vertices = simplices.max() + 1
    data = np.ones(num_simplices)
    indptr = np.arange(num_simplices+1)
    indices = simplices.ravel()
    return sp.csr_matrix((data, indices, indptr), shape=(num_simplices, num_vertices))

def homology(dim, simplices):
    boundary = boundary_operator(dim+1, simplices)
    kernel = sp.linalg.null_space(boundary)
    image = boundary_operator(dim, simplices) * kernel
    return kernel.shape[1] - image.shape[1]

simplices = np.array([[0, 1, 2], [1, 2, 3], [2, 3, 4], [3, 4, 5]])
print(homology(1, simplices))
