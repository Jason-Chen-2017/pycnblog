```java
@GetMapping("/products")
public List<Product> searchProducts(@RequestParam(required = false) String keyword,
                                    @RequestParam(required = false) String brand,
                                    @RequestParam(required = false) String model,
                                    @RequestParam(required = false) BigDecimal minPrice,
                                    @RequestParam(required = false) BigDecimal maxPrice) {
  // 构建查询条件
  List<Predicate> predicates = new ArrayList<>();
  if (StringUtils.isNotBlank(keyword)) {
    predicates.add(criteriaBuilder.like(root.get("name"), "%" + keyword + "%"));
  }
  if (StringUtils.isNotBlank(brand)) {
    predicates.add(criteriaBuilder.equal(root.get("brand"), brand));
  }
  if (StringUtils.isNotBlank(model)) {
    predicates.add(criteriaBuilder.equal(root.get("model"), model));
  }
  if (minPrice != null) {
    predicates.add(criteriaBuilder.greaterThanOrEqualTo(root.get("price"), minPrice));
  }
  if (maxPrice != null) {
    predicates.add(criteriaBuilder.lessThanOrEqualTo(root.get("price"), maxPrice));
  }
  // 执行查询
  CriteriaQuery<Product> query = criteriaBuilder.createQuery(Product.class);
  query.where(predicates.toArray(new Predicate[0]));
  return entityManager.createQuery(query).getResultList();
}
```

```java
  // 校验商品库存
  Product product = productRepository.findById(orderRequest.getProductId())
      .orElseThrow(() -> new NotFoundException("Product not found"));

    throw new BadRequestException("Insufficient stock");
  }
  // 扣减商品库存
  product.setStock(product.getStock() - orderRequest.getQuantity());
  productRepository.save(product);
  // 创建订单
  Order order = new Order();
  order.setUserId(orderRequest.getUserId());
  order.setProductId(orderRequest.getProductId());
  order.setQuantity(orderRequest.getQuantity());
```
