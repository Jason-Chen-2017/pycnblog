## 4.项目实践：代码实例和详细解释说明

下面，我们将通过一个简单的示例，来演示如何使用Python和Flask框架来实现卷烟物价管理系统的后端。本示例主要展示了如何使用Flask和SQLAlchemy来实现CRUD操作和RESTful API设计。

```python
from flask import Flask, request
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:////tmp/test.db'
db = SQLAlchemy(app)

class Cigarette(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(50), unique=True, nullable=False)
    price = db.Column(db.Float, nullable=False)

@app.route('/cigarettes', methods=['POST'])
def create_cigarette():
    name = request.json['name']
    price = request.json['price']
    cigarette = Cigarette(name=name, price=price)
    db.session.add(cigarette)
    db.session.commit()
    return {'id': cigarette.id}, 201

@app.route('/cigarettes/<int:id>', methods=['GET'])
def get_cigarette(id):
    cigarette = Cigarette.query.get(id)
    if cigarette is None:
        return {'error': 'not found'}, 404
    return {'name': cigarette.name, 'price': cigarette.price}

@app.route('/cigarettes/<int:id>', methods=['PUT'])
def update_cigarette(id):
    cigarette = Cigarette.query.get(id)
    if cigarette is None:
        return {'error': 'not found'}, 404
    if 'name' in request.json:
        cigarette.name = request.json['name']
    if 'price' in request.json:
        cigarette.price = request.json['price']
    db.session.commit()
    return {'name': cigarette.name, 'price': cigarette.price}

@app.route('/cigarettes/<int:id>', methods=['DELETE'])
def delete_cigarette(id):
    cigarette = Cigarette.query.get(id)
    if cigarette is None:
        return {'error': 'not found'}, 404
    db.session.delete(cigarette)
    db.session.commit()
    return {}, 204
```

这段代码实现了卷烟物价管理系统的基本功能，包括创建、读取、更新和删除卷烟的价格信息。通过定义RESTful API，前端可以通过HTTP请求来调用这些功能。
