$$
Price = RoomTypePrice * RoomNum * Days
$$

其中，RoomTypePrice表示客房的价格，RoomNum表示客房的数量，Days表示客户的入住天数。

财务管理公式如下：

$$
Profit = Income - Expense
$$

```python
class Room:
    def __init__(self, room_type, room_num, room_price):
        self.room_type = room_type
        self.room_num = room_num
        self.room_price = room_price

class Order:
    def __init__(self, customer_name, room_type, room_num, check_in_date, check_out_date):
        self.customer_name = customer_name
        self.room_type = room_type
        self.room_num = room_num
        self.check_in_date = check_in_date
        self.check_out_date = check_out_date

    def get_price(self):
        room = Room(self.room_type, self.room_num, self.room_price)
        days = (self.check_out_date - self.check_in_date).days
        return room.room_price * room.room_num * days

class Hotel:
    def __init__(self, name, rooms):
        self.name = name
        self.rooms = rooms
        self.orders = []

    def add_order(self, order):
        self.orders.append(order)

    def get_income(self):
        income = 0
        for order in self.orders:
            income += order.get_price()
        return income

    def get_expense(self):
        expense = 0
        for room in self.rooms:
            expense += room.room_price * room.room_num
        return expense

    def get_profit(self):
        return self.get_income() - self.get_expense()
```
