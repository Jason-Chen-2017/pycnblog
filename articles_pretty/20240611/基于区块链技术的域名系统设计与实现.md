```
pragma solidity ^0.8.0;

contract DomainRegistry {
    mapping(string => address) domains;

    function register(string memory domain) public payable {
        require(domains[domain] == address(0), "Domain already registered");
        require(msg.value >= 1 ether, "Registration fee is 1 ether");
        domains[domain] = msg.sender;
    }

    function getOwner(string memory domain) public view returns (address) {
        return domains[domain];
    }
}
```
