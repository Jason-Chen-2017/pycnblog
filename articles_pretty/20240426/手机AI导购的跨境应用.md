import pandas as pd

# 加载用户评分数据
ratings = pd.read_csv('ratings.csv')

# 计算用户之间的相似度
def user_similarity(user1, user2):
    # 获取用户共同评分的物品
    common_items = ratings[ratings['userId'].isin([user1, user2])]
    # 计算相似度
    similarity = common_items.corr().loc[user1, user2]
    return similarity

# 获取与目标用户最相似的用户
def get_similar_users(user, n):
    # 计算目标用户与其他用户的相似度
    similarities = ratings['userId'].apply(lambda x: user_similarity(user, x))
    # 排序并返回前 n 个相似用户
    return similarities.sort_values(ascending=False).head(n)

# 为目标用户推荐商品
def recommend_items(user, n):
    # 获取与目标用户最相似的用户
    similar_users = get_similar_users(user, n)
    # 获取相似用户喜欢的商品
    similar_user_items = ratings[ratings['userId'].isin(similar_users.index)]['itemId']
    # 统计商品出现的次数
    item_counts = similar_user_items.value_counts()
    # 返回出现次数最多的 n 个商品
    return item_counts.head(n)
