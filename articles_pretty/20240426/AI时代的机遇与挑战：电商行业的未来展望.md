similar_users = user_similarity[current_user_id].sort_values(ascending=False)

# 获取相似用户购买过的商品，并推荐给当前用户
recommended_items = []
for user_id, similarity in similar_users.items():
    if user_id != current_user_id:
        user_items = users[users['user_id'] == user_id]['item_id']
        recommended_items.extend(user_items)

# 去重并推荐
