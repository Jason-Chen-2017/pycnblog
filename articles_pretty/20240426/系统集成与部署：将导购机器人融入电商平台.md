$$
\begin{aligned}

& \text{预测评分}(u, i) = \bar{r}_u + \frac{\sum_{v \in N_u(i)} \text{相似度}(u, v)(r_{vi} - \bar{r}_v)}{\sum_{v \in N_u(i)} \text{相似度}(u, v)}

import random

class Product:
    def __init__(self, name, price, category):
        self.name = name
        self.price = price
        self.category = category

class User:
    def __init__(self, name, preferences):
        self.name = name
        self.preferences = preferences

class ShoppingBot:
    def __init__(self, products):
        self.products = products

    def recommend_products(self, user):
        recommended_products = []
        for product in self.products:
            if product.category in user.preferences:
                recommended_products.append(product)
        return random.sample(recommended_products, 3)

# 创建商品
products = [
    Product("T恤", 50, "服装"),
    Product("牛仔裤", 100, "服装"),
    Product("运动鞋", 200, "鞋子"),
    Product("笔记本电脑", 5000, "电子产品"),
    Product("手机", 3000, "电子产品"),
]

# 创建用户
user = User("张三", ["服装", "鞋子"])

# 创建导购机器人
bot = ShoppingBot(products)

# 推荐商品
recommended_products = bot.recommend_products(user)

# 打印推荐商品
for product in recommended_products:
    print(product.name)
