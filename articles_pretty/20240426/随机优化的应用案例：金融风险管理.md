def portfolio_optimization(mu, Sigma, lambda_):
    """
    投资组合优化
    """
    w = np.linalg.solve(Sigma, mu) / lambda_
    return w / w.sum()

                  [0.002, 0.01, 0.09]])

# 风险厌恶系数
