```python
from rdkit import Chem
from rdkit.Chem import AllChem

# 加载靶点蛋白结构
protein = Chem.MolFromPDBFile("protein.pdb")

# 准备化合物库
ligands = [Chem.MolFromSmiles(smi) for smi in open("ligands.smi").readlines()]

# 对每个化合物进行分子对接
for ligand in ligands:
    # 准备配体分子
    AllChem.EmbedMolecule(ligand)
    AllChem.MMFFOptimizeMolecule(ligand)

    # 进行分子对接
    dock_result = AllChem.Dock(ligand, protein)

    # 获取打分
    score = dock_result.GetScore()

    # 打印结果
    print(f"Ligand: {ligand.GetProp('_Name')}, Score: {score}")
```
