```sql
CREATE TABLE User (
    UserID INT PRIMARY KEY,
    UserName VARCHAR(20),
    Password VARCHAR(20),
    PhoneNumber VARCHAR(20)
);

CREATE TABLE Car (
    CarID INT PRIMARY KEY,
    CarBrand VARCHAR(20),
    CarType VARCHAR(20),
    RentPrice FLOAT
);

CREATE TABLE Order (
    OrderID INT PRIMARY KEY,
    UserID INT,
    CarID INT,
    StartDate DATE,
    EndDate DATE,
    TotalPrice FLOAT
);

CREATE TABLE Finance (
    FinanceID INT PRIMARY KEY,
    OrderID INT,
    PaymentStatus BOOLEAN
);
```
