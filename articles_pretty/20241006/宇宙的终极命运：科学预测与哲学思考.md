import matplotlib.pyplot as plt
from scipy.integrate import odeint

def general_relativity(y, t, H0, Omega_Lambda, Omega_m, Omega_k):
    a = y[0]
    da_dt = y[1]
    d2a_dt2 = -H0**2 * (Omega_Lambda * a**2 + Omega_m * a + Omega_k)
    return [da_dt, d2a_dt2]

# 求解微分方程

plt.xlabel('时间 (Gyr)')
