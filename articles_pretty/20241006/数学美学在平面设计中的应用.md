```python
import math

def golden_ratio(length):
    phi = (1 + math.sqrt(5)) / 2
    a = length / phi
    b = length - a
    return a, b

# 示例

a, b = golden_ratio(total_length)
print(f"较小部分长度: {a}, 较大部分长度: {b}")
```

#### 5.2.2 斐波那契数列代码实现
```python
def fibonacci_sequence(n):
    a, b = 0, 1
    sequence = []
    for _ in range(n):
        sequence.append(a)
        a, b = b, a + b
    return sequence

# 示例
n = 10
fib_sequence = fibonacci_sequence(n)
print(f"斐波那契数列: {fib_sequence}")
```
