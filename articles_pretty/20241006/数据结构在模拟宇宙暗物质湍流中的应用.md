### 4.3 公式与详细讲解
#### 4.3.1 牛顿引力方程
$$
F = G \frac{m_1 m_2}{r^2}
$$
其中，$F$ 为引力，$G$ 为引力常数，$m_1$ 和 $m_2$ 分别为两个物体的质量，$r$ 为两个物体之间的距离。

#### 4.3.2 流体动力学方程
$$
\frac{\partial \rho}{\partial t} + \nabla \cdot (\rho \mathbf{v}) = 0
$$
$$
\frac{\partial (\rho \mathbf{v})}{\partial t} + \nabla \cdot (\rho \mathbf{v} \otimes \mathbf{v}) + \nabla p = \nabla \cdot \mathbf{F}
$$
其中，$\rho$ 为密度，$\mathbf{v}$ 为速度，$p$ 为压力，$\mathbf{F}$ 为外力。

```python
import numpy as np
import time

def initialize_simulation(n, m, p):
    data = np.zeros((n, m, p))
    for i in range(n):
        for j in range(m):
            for k in range(p):
                data[i, j, k] = np.random.uniform(0, 1)
    return data

def time_step(data, dt):
    for i in range(n):
        for j in range(m):
            for k in range(p):
                data[i, j, k] += dt * calculate_velocity(data, i, j, k)
    return data

def calculate_velocity(data, i, j, k):
    # 计算每个点的引力和流体动力学速度
    velocity = 0
    for dx in [-1, 0, 1]:
        for dy in [-1, 0, 1]:
            for dz in [-1, 0, 1]:
                if dx == 0 and dy == 0 and dz == 0:
                    continue
                ni, nj, nk = i + dx, j + dy, k + dz
                if 0 <= ni < n and 0 <= nj < m and 0 <= nk < p:
                    velocity += data[ni, nj, nk] * (dx, dy, dz)
    return velocity

def update_data(data):
    for i in range(n):
        for j in range(m):
            for k in range(p):
                data[i, j, k] = update_velocity(data, i, j, k)
    return data

def update_velocity(data, i, j, k):
    # 更新每个点的速度
    velocity = 0
    for dx in [-1, 0, 1]:
        for dy in [-1, 0, 1]:
            for dz in [-1, 0, 1]:
                if dx == 0 and dy == 0 and dz == 0:
                    continue
                ni, nj, nk = i + dx, j + dy, k + dz
                if 0 <= ni < n and 0 <= nj < m and 0 <= nk < p:
                    velocity += data[ni, nj, nk] * (dx, dy, dz)
    return velocity

def store_data(data, filename):
    np.save(filename, data)

if __name__ == "__main__":
    n, m, p = 100, 100, 100
    dt = 0.01
    data = initialize_simulation(n, m, p)
    start_time = time.time()
    for t in range(1000):
        data = time_step(data, dt)
        data = update_data(data)
    end_time = time.time()
    print(f"Total time: {end_time - start_time} seconds")
    store_data(data, "output.npy")
```
