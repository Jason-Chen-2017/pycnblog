def encrypt(data):
    # 使用SHA-256进行加密
    encrypted_data = hashlib.sha256(data.encode()).hexdigest()
    return encrypted_data

def generate_key():
    # 生成随机密钥
    key = hashlib.sha256(os.urandom(16)).hexdigest()
    return key

def protect_key(key):
    # 保护密钥
    protected_key = hashlib.sha256(key.encode()).hexdigest()
    return protected_key
