### 5.2 源代码详细实现和代码解读
#### 5.2.1 积分系统实现
```python
# 积分系统实现
积分规则 = {任务1: 10, 任务2: 20, 任务3: 30}

def 计算积分(任务完成情况):
    积分 = 0
    for 任务, 权重 in 积分规则.items():
        if 任务完成情况[任务]:
            积分 += 权重
    return 积分

def 展示积分(玩家列表):
    排行榜 = []
    for 玩家 in 玩家列表:
        排行榜.append((玩家, 计算积分(玩家任务完成情况)))
    排行榜.sort(key=lambda x: x[1], reverse=True)
    for 玩家, 积分 in 排行榜:
        print(f"{玩家} 的积分: {积分}")

# 示例
玩家列表 = ["玩家1", "玩家2", "玩家3"]
玩家任务完成情况 = {
    "玩家1": {任务1: True, 任务2: False, 任务3: True},
    "玩家2": {任务1: False, 任务2: True, 任务3: True},
    "玩家3": {任务1: True, 任务2: True, 任务3: False}
}
展示积分(玩家列表)
```
