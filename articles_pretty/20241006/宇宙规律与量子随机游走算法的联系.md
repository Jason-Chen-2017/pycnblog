def quantum_random_walk(n_steps):
    # 初始化量子态
    qc = QuantumCircuit(1)
    qc.h(0)
    
    # 量子步操作
    for _ in range(n_steps):
        qc.cx(0, 1)
        qc.h(0)
    
    # 测量
    qc.measure_all()
    
    # 仿真
    backend = Aer.get_backend('qasm_simulator')
    job = execute(qc, backend, shots=1024)
    result = job.result()
    
    # 获取结果
    counts = result.get_counts(qc)
    return counts

# 运行量子随机游走
