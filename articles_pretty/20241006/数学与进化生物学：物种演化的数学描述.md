### 5.2 源代码详细实现和代码解读
```python
import numpy as np
import matplotlib.pyplot as plt

# 初始化种群
population_size = 1000
gene_type_A = 0.6
gene_type_B = 0.4

# 计算适应度
fitness_A = 0.8
fitness_B = 0.6

# 选择适应度较高的个体进行繁殖
def select_individuals(population_size, fitness_A, fitness_B):
    individuals = np.random.choice(['A', 'B'], size=population_size, p=[fitness_A, fitness_B])
    return individuals

# 生成新的种群
def generate_new_population(individuals):
    new_population = np.random.choice(individuals, size=population_size, replace=True)
    return new_population

# 计算基因频率
def calculate_gene_frequency(new_population):
    gene_type_A_count = np.sum(new_population == 'A')
    gene_type_B_count = np.sum(new_population == 'B')
    gene_type_A_frequency = gene_type_A_count / population_size
    gene_type_B_frequency = gene_type_B_count / population_size
    return gene_type_A_frequency, gene_type_B_frequency

# 主函数
def main():
    gene_type_A_frequency_list = []
    gene_type_B_frequency_list = []

    for i in range(100):
        individuals = select_individuals(population_size, fitness_A, fitness_B)
        new_population = generate_new_population(individuals)
        gene_type_A_frequency, gene_type_B_frequency = calculate_gene_frequency(new_population)
        gene_type_A_frequency_list.append(gene_type_A_frequency)
        gene_type_B_frequency_list.append(gene_type_B_frequency)

    plt.plot(gene_type_A_frequency_list, label='Gene Type A')
    plt.plot(gene_type_B_frequency_list, label='Gene Type B')
    plt.xlabel('Generation')
    plt.ylabel('Gene Frequency')
    plt.legend()
    plt.show()

if __name__ == '__main__':
    main()
```
