def adversarial_training(model, criterion, optimizer, inputs, labels, epsilon):
    # 对输入进行对抗扰动
    adv_inputs = generate_adversarial_examples(inputs, model, epsilon)
    
    # 对抗样本的预测结果
    adv_outputs = model(adv_inputs)
    
    # 计算对抗样本的损失
    adv_loss = criterion(adv_outputs, labels)
    
    # 反向传播，更新模型参数
    optimizer.zero_grad()
    adv_loss.backward()
    optimizer.step()
    
    return adv_loss.item()
