```python
import sympy as sp
import numpy as np

def compute_characteristic_polynomial(matrix):
    \"\"\"
    计算矩阵的特征多项式。
    \"\"\"
    # 创建符号变量
    x = sp.symbols('x')
    char_poly = sp.characteristic_polynomial(matrix, x)
    return char_poly

def identify_irreducible_factors(poly):
    \"\"\"
    识别多项式的不可约因子。
    \"\"\"
    irreducible_factors = sp.factor(poly)
    return irreducible_factors

def construct_root_system(poly, matrix):
    \"\"\"
    基于特征多项式构建根系统。
    \"\"\"
    roots = []
    for factor in poly.factor():
        if sp.is_polynomial(factor) and not sp.is_number(factor):
            root_space = find_root_space(matrix, factor)
            roots.extend(root_space)
    return roots

def find_root_space(matrix, factor):
    \"\"\"
    计算根空间。
    \"\"\"
    eigenvalues = sp.solve(factor, matrix)
    root_space = []
    for eigenvalue in eigenvalues:
        eigenvectors = sp.eigenvects(matrix - eigenvalue * sp.eye(matrix.shape[0]))
        for eigenvector in eigenvectors:
            root_space.append(eigenvector)
    return root_space

def main():
    # 创建一个2x2矩阵作为示例
    matrix = np.array([[1, 2], [3, 4]])
    char_poly = compute_characteristic_polynomial(matrix)
    irreducible_factors = identify_irreducible_factors(char_poly)
    root_system = construct_root_system(irreducible_factors, matrix)
    print(f\"特征多项式：{char_poly}\")
    print(f\"不可约因子：{irreducible_factors}\")
    print(f\"根系统：{root_system}\")

if __name__ == \"__main__\":
    main()
```
