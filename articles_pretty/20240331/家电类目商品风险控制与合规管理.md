```python
# 基于异常检测的安全隐患预警算法 
import numpy as np
from scipy.stats import norm

def safety_risk_detection(monitoring_data):
    """
    输入: 产品安全监测数据 monitoring_data
    输出: 安全隐患预警结果
    """
    # 计算数据的均值和方差
    mu = np.mean(monitoring_data)
    sigma = np.std(monitoring_data)
    
    # 计算每个样本的异常度得分
    scores = norm.pdf(monitoring_data, loc=mu, scale=sigma)
    
    # 设置异常阈值, 识别安全隐患
    threshold = 0.05
    warnings = monitoring_data[scores < threshold]
    
    return warnings
```

```python
import rdflib
from rdflib import Graph, Namespace, Literal
from rdflib.plugins.sparql import prepareQuery

def compliance_check(product_features, regulations):
    """
    输入: 产品特征数据 product_features, 行政监管要求 regulations
    输出: 合规性检查结果
    """
    # 构建知识图谱
    g = Graph()
    eg = Namespace("http://example.org/")
    
    # 添加产品特征知识
    for f, v in product_features.items():
        g.add((eg[f], eg["hasValue"], Literal(v)))
    
    # 添加监管要求知识
    for r, c in regulations.items():
        g.add((eg[r], eg["requires"], eg[c]))
    
    # 定义合规性推理规则
    query = prepareQuery("""
        SELECT ?product ?nonCompliant
        WHERE {
            ?product eg:hasValue ?v .
            ?regulation eg:requires ?c .
            FILTER NOT EXISTS { ?product eg:hasValue ?c }
            BIND(CONCAT(?product, " is not compliant with ", ?regulation) AS ?nonCompliant)
        }
    """)
    
    # 执行推理并返回结果
    results = g.query(query)
    return list(results)
```
