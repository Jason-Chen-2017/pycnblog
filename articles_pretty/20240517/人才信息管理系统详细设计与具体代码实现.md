- 性别
- 年龄


## 4.项目实践：代码实例和详细解释说明

下面我们开始实践部分，首先是后端的Python代码。

### 4.1 Python后端代码

```python
from flask import Flask, request
import sqlite3

app = Flask(__name__)

@app.route('/employees', methods=['GET', 'POST', 'PUT', 'DELETE'])
def handle_employees():
    conn = sqlite3.connect('employees.db')
    cur = conn.cursor()

    if request.method == 'GET':
        cur.execute('SELECT * FROM Employees')
        employees = cur.fetchall()
        return {'employees': employees}

    elif request.method == 'POST':
        new_employee = request.json
        cur.execute('''
            INSERT INTO Employees (Name, Gender, Age, Position, Salary, HireDate)
            VALUES (?, ?, ?, ?, ?, ?)
        ''', (new_employee['Name'], new_employee['Gender'], new_employee['Age'], 
              new_employee['Position'], new_employee['Salary'], new_employee['HireDate']))
        conn.commit()
        return {'message': 'New employee added.'}

    elif request.method == 'PUT':
        updated_employee = request.json
        cur.execute('''
            UPDATE Employees
            SET Name = ?, Gender = ?, Age = ?, Position = ?, Salary = ?, HireDate = ?
            WHERE Name = ?
        ''', (updated_employee['Name'], updated_employee['Gender'], updated_employee['Age'], 
              updated_employee['Position'], updated_employee['Salary'], updated_employee['HireDate'], 
              updated_employee['OldName']))
        conn.commit()
        return {'message': 'Employee information updated.'}

    elif request.method == 'DELETE':
        employee_to_delete = request.json
        cur.execute('''
            DELETE FROM Employees
            WHERE Name = ?
        ''', (employee_to_delete['Name'],))
        conn.commit()
        return {'message': 'Employee deleted.'}

if __name__ == '__main__':
    app.run()
```
