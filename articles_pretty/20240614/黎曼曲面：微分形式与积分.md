```python
# 定义复数坐标
z = sp.symbols('z')

# 定义函数
f = 2 * z

# 定义积分路径
path = sp.exp(sp.I * sp.symbols('t'))

# 计算积分
integral = sp.integrate(f.subs(z, path) * sp.diff(path, sp.symbols('t')), (sp.symbols('t'), 0, 2 * sp.pi))

print(f"积分结果: {integral}")
```
