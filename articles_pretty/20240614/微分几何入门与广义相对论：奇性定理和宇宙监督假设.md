import numpy as np

def christoffel_symbols(metric):
    """
    Calculate the Christoffel symbols for a given metric.
    """
    inverse_metric = np.linalg.inv(metric)
    christoffel = np.zeros(metric.shape + (metric.shape[0],))
    for lambda_ in range(metric.shape[0]):
        for mu in range(metric.shape[1]):
            for nu in range(metric.shape[1]):
                for sigma in range(metric.shape[1]):
                    christoffel[lambda_][mu][nu] += (
                        inverse_metric[lambda_][sigma] * 0.5 * (
                            np.gradient(metric[sigma][mu], axis=nu) +
                            np.gradient(metric[sigma][nu], axis=mu) -
                            np.gradient(metric[mu][nu], axis=sigma)
                        )
                    )
    return christoffel

# Example metric for a 2D space (e.g., surface of a sphere)
