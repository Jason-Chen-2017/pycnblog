```python
import numpy as np

def length_on_surface(curve, surface, metric):
    """
    计算曲线在曲面上的长度
    """
    length = 0
    for i in range(len(curve)-1):
        x1, y1, z1 = curve[i]
        x2, y2, z2 = curve[i+1]
        dx = x2 - x1
        dy = y2 - y1
        dz = z2 - z1
        ds = np.sqrt(dx**2 + dy**2 + dz**2)
        g11 = metric(x1, y1, z1)[0][0]
        g22 = metric(x1, y1, z1)[1][1]
        g33 = metric(x1, y1, z1)[2][2]
        length += np.sqrt(g11*dx**2 + g22*dy**2 + g33*dz**2) * ds
    return length
```
