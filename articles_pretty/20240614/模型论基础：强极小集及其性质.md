    partial_order = []
    for s in model.subsets():
        if all(not s.issubset(t) for t in partial_order):
            partial_order.append(s)

    strong_minimal_set = set()
    for s in maximal_element:

            strong_minimal_set.add(s)

    return strong_minimal_set
