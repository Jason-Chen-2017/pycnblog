```python
import numpy as np

def christoffel_symbols(g, g_inv):
    n = g.shape[0]
    symbols = np.zeros((n, n, n))
    for i in range(n):
        for j in range(n):
            for k in range(n):
                symbols[i, j, k] = 0.5 * (g_inv[i, k] * (partial(g[k, j], i) + partial(g[i, k], j) - partial(g[i, j], k)))
    return symbols

def curvature_tensor(symbols):
    n = symbols.shape[0]
    curvature = np.zeros((n, n, n, n))
    for i in range(n):
        for j in range(n):
            for k in range(n):
                for l in range(n):
                    curvature[i, j, k, l] = partial(symbols[i, j, l], k) - partial(symbols[i, j, k], l)
                    for m in range(n):
                        curvature[i, j, k, l] += symbols[i, m, l] * symbols[m, j, k] - symbols[i, m, k] * symbols[m, j, l]
    return curvature

def partial(f, i):
    h = 1e-6
    x = np.zeros(f.shape)
    x[i] = h
    return (f(x + h) - f(x - h)) / (2 * h)

# Example usage
g = np.array([[1, 0, 0], [0, 1, 0], [0, 0, 1]])
g_inv = np.linalg.inv(g)

curvature = curvature_tensor(symbols)
```
