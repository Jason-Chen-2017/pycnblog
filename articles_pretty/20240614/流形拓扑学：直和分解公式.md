```python
import numpy as np
import scipy as sp
import scipy.sparse.linalg as spla
import scipy.sparse as sps
import networkx as nx

def homology(G):
    # 计算同调群
    n = len(G)
    A = nx.adjacency_matrix(G)
    B = sps.eye(n) - A
    C = sps.csr_matrix(B)
    ker = spla.spsolve(C, np.ones(n))
    ker = np.round(ker)
    ker = ker.astype(int)
    ker = ker.reshape((n, 1))
    ker = sps.csr_matrix(ker)
    H = spla.spsolve(A, ker)
    H = np.round(H)
    H = H.astype(int)
    H = H.reshape((n, 1))
    return H

def cohomology(G):
    # 计算上同调群
    n = len(G)
    A = nx.adjacency_matrix(G)
    B = sps.eye(n) - A
    C = sps.csr_matrix(B.transpose())
    ker = spla.spsolve(C, np.ones(n))
    ker = np.round(ker)
    ker = ker.astype(int)
    ker = ker.reshape((n, 1))
    ker = sps.csr_matrix(ker)
    H = spla.spsolve(A.transpose(), ker)
    H = np.round(H)
    H = H.astype(int)
    H = H.reshape((n, 1))
    return H

# 测试代码
G = nx.Graph()
G.add_edges_from([(0, 1), (1, 2), (2, 3), (3, 0)])
H = homology(G)
print("同调群：")
print(H)
C = cohomology(G)
print("上同调群：")
print(C)
```
