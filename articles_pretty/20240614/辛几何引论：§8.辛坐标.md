import numpy as np

def symplectic_transformation(A, B):
    """
    计算辛变换矩阵
    """

    M = np.block([[A, B], [-B.T, A.T]])
    J = np.block([[np.zeros((n, n)), np.eye(n)], [-np.eye(n), np.zeros((n, n))]])
    return np.linalg.inv(M) @ J @ M

def symplectic_coordinates(q, p, A, B):
    """
    计算辛坐标系下的向量场
    """
    n = q.shape[0]
    X = np.zeros((n, 1))
    Y = np.zeros((n, 1))
    for i in range(n):
        X[i] = np.dot(A[i, :], p) - np.dot(B[i, :], q)
        Y[i] = np.dot(B[i, :], p) + np.dot(A[i, :], q)
    return np.vstack((X, Y))
