```python
from transformers import GPT2LMHeadModel, GPT2Tokenizer

# 加载预训练的GPT2模型
model = GPT2LMHeadModel.from_pretrained('gpt2')
tokenizer = GPT2Tokenizer.from_pretrained('gpt2')

def generate_quantum_key(quantum_state):
    # 将量子态编码为token序列
    input_ids = tokenizer.encode(str(quantum_state), return_tensors='pt')
    
    # 使用GPT2模型生成随机比特串
    output = model.generate(input_ids, max_length=256, num_return_sequences=1, 
                            do_sample=True, top_k=50, top_p=0.95, num_beams=1)
    
    # 解码输出为字符串
    quantum_key = tokenizer.decode(output[0], skip_special_tokens=True)
    
    return quantum_key
```
