```python
import RPi.GPIO as GPIO
import time

# 设置GPIO引脚
GPIO.setmode(GPIO.BCM)

# 控制LED灯
def turn_on_led():
    GPIO.output(18, GPIO.HIGH)

def turn_off_led():
    GPIO.output(18, GPIO.LOW)

# 主程序
while True:
    # 获取用户输入
    command = input("请输入指令 (on/off): ")

    # 根据指令控制LED灯
    if command == "on":
        turn_on_led()
    elif command == "off":
        turn_off_led()
    else:
        print("无效指令")

    # 延时
    time.sleep(1)
```

- 代码首先设置GPIO引脚，然后定义了两个函数 `turn_on_led()` 和 `turn_off_led()` 来控制LED灯的开关。
- 主程序循环获取用户输入，并根据指令调用相应的函数控制LED灯。
