import sympy as sp

# 定义布尔代数
B = sp.BooleanAlgebra()

# 定义逻辑公式
phi = sp.symbols('phi')

# 构建集合
D = {x for x in B if phi.subs(x)}

# 验证集合
def verify_set(D, phi):
    for x in D:
        if not phi.subs(x):
            return False
    return True
