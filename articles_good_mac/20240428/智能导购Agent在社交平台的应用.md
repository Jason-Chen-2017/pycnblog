**操作步骤:**

1. 收集用户的行为数据和相似用户的行为数据。
2. 计算用户之间的相似度。

```python
import numpy as np

def cosine_similarity(a, b):
    """计算两个向量的余弦相似度"""
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

def collaborative_filtering(ratings, user_id, k):
    """协同过滤算法"""
    # 计算用户之间的相似度
    similarities = []
    for other_user_id in range(ratings.shape[0]):
        if other_user_id != user_id:
            similarity = cosine_similarity(ratings[user_id], ratings[other_user_id])
            similarities.append((other_user_id, similarity))
    
    # 选择 k 个最相似的用户
    similarities.sort(key=lambda x: x[1], reverse=True)
    top_k_users = similarities[:k]

    # 预测用户对未评分商品的评分
    predictions = []
    for item_id in range(ratings.shape[1]):
        if ratings[user_id, item_id] == 0:
            weighted_sum = 0
            similarity_sum = 0
            for other_user_id, similarity in top_k_users:
                if ratings[other_user_id, item_id] != 0:
                    weighted_sum += similarity * ratings[other_user_id, item_id]
                    similarity_sum += similarity
            if similarity_sum > 0:
                prediction = weighted_sum / similarity_sum
                predictions.append((item_id, prediction))
    
    # 返回预测结果
    predictions.sort(key=lambda x: x[1], reverse=True)
    return predictions
```
