from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 定义停用词列表
stop_words = set(stopwords.words('english'))

# 定义手机信息列表
phones = [
    {'brand': 'Apple', 'model': 'iPhone 14', 'features': ['5G', 'A16 Bionic chip', '48MP camera']},
    {'brand': 'Samsung', 'model': 'Galaxy S23', 'features': ['5G', 'Snapdragon 8 Gen 2', '200MP camera']},
    # ...
]

# 定义用户需求
user_query = "I want a phone with a good camera and 5G."

# 对用户需求进行预处理
user_query = user_query.lower()
user_query_tokens = nltk.word_tokenize(user_query)
user_query_filtered = [word for word in user_query_tokens if word not in stop_words]

# 创建 TF-IDF 向量化器
vectorizer = TfidfVectorizer()

# 将手机信息和用户需求转化为 TF-IDF 向量
phone_features = [' '.join(phone['features']) for phone in phones]
tfidf_matrix = vectorizer.fit_transform(phone_features + [user_query])

# 计算用户需求与手机信息之间的余弦相似度
cosine_similarities = cosine_similarity(tfidf_matrix[-1:], tfidf_matrix[:-1]).flatten()

# 根据相似度排序，推荐最合适的手机
recommended_phones = [phones[i] for i in cosine_similarities.argsort()[::-1]]

# 打印推荐结果

    print(f"- {phone['brand']} {phone['model']}")
