$$L(f) = \\sup_{z \\in \\mathbb{C}} \\left| \\frac{\\partial f}{\\partial z}(z) \\right|$$

import numpy as np
import matplotlib.pyplot as plt

def leibeg_constant(f, x_range=(-1, 1), y_range=(-1, 1)):
    \"\"\"
    计算勒贝格常数
    \"\"\"
    dx, dy = 0.01, 0.01
    x = np.arange(x_range[0], x_range[1], dx)
    y = np.arange(y_range[0], y_range[1], dy)
    X, Y = np.meshgrid(x, y)
    f_x = f(X + 1j * Y)
    grad = np.gradient(f_x)
    return np.max(np.abs(grad))

def plot_riemann_surface(f, x_range=(-1, 1), y_range=(-1, 1), levels=50):
    \"\"\"
    绘制黎曼曲面
    \"\"\"
    dx, dy = 0.01, 0.01
    x = np.arange(x_range[0], x_range[1], dx)
    y = np.arange(y_range[0], y_range[1], dy)
    X, Y = np.meshgrid(x, y)
    Z = f(X + 1j * Y)
    plt.contourf(X, Y, np.abs(Z), levels=levels)
    plt.colorbar()
    plt.title('Riemann Surface')
    plt.show()

# 示例：f(z) = z^2
f = lambda z: z**2
plot_riemann_surface(f)
print(\"Leibeg constant for f(z) = z^2:\", leibeg_constant(f))
