```python
# 科学传播游戏化项目
from flask import Flask, render_template, request
import mysql.connector

app = Flask(__name__)

# 数据库连接
db = mysql.connector.connect(
    host="localhost",
    user="root",
    password="password",
    database="science_game"
)

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/submit', methods=['POST'])
def submit():
    username = request.form['username']
    task_id = request.form['task_id']
    # 插入数据
    cursor = db.cursor()
    cursor.execute("INSERT INTO user_tasks (username, task_id) VALUES (%s, %s)", (username, task_id))
    db.commit()
    return "提交成功"

if __name__ == '__main__':
    app.run(debug=True)
```
