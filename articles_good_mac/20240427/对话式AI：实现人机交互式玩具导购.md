```python
# 导入必要的库
import nltk
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.svm import SVC

# 定义训练数据
train_data = [
    ("我想买一个玩具车", "询问"),
    ("给我推荐一些适合 5 岁男孩的玩具", "推荐"),
    ("乐高积木和拼图哪个更适合 8 岁的女孩", "比较"),
]

# 将文本数据转换为特征向量
vectorizer = TfidfVectorizer()
X_train = vectorizer.fit_transform([text for text, label in train_data])

# 训练意图识别模型
model = SVC()
model.fit(X_train, [label for text, label in train_data])

# 预测新文本的意图
new_text = "我想给我的儿子买一个生日礼物"
X_new = vectorizer.transform([new_text])
predicted_intent = model.predict(X_new)[0]

print(f"预测意图: {predicted_intent}")
```

*   定义训练数据，每个样本包含一个文本和对应的意图标签。
*   使用 `TfidfVectorizer` 将文本数据转换为特征向量。
*   使用支持向量机 (SVM) 训练意图识别模型。
