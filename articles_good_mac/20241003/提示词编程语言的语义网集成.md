# 5.2.1 自然语言指令到RDF triples的转化
def natural_language_to_rdf_triples(natural_language):
    # 分词与词性标注
    tokens = nltk.word_tokenize(natural_language)
    pos_tags = nltk.pos_tag(tokens)
    
    # 语法分析
    parser = nltk.RegexpParser(r"""
        NP: {<DT|JJ|NN.*>+}          # Noun phrase
        VP: {<VB.*><NP|PP|CLAUSE>+$}  # Verb phrase
        PP: {<IN><NP>}                # Prepositional phrase
        CLAUSE: {<NP><VP>}            # Clause
    """)
    tree = parser.parse(pos_tags)
    
    # 语义解析
    rdf_triples = []
    for subtree in tree.subtrees():
        if subtree.label() == 'NP':
            subject = subtree.leaves()[0][0]
            predicate = 'is'
            object = subtree.leaves()[1][0]
            rdf_triples.append((subject, predicate, object))
        elif subtree.label() == 'VP':
            subject = subtree.leaves()[0][0]
            predicate = subtree.leaves()[1][0]
            object = subtree.leaves()[2][0]
            rdf_triples.append((subject, predicate, object))
    
    return rdf_triples
