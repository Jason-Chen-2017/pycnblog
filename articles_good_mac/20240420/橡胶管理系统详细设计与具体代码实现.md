```python
from flask import Flask, request
from flask_sqlalchemy import SQLAlchemy

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'mysql://user:pass@localhost/db'
db = SQLAlchemy(app)

class Rubber(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    type = db.Column(db.String(100))
    quality = db.Column(db.String(100))
    price = db.Column(db.Numeric(10, 2))

@app.route('/rubber', methods=['POST'])
def add_rubber():
    type = request.form.get('type')
    quality = request.form.get('quality')
    price = request.form.get('price')
    rubber = Rubber(type=type, quality=quality, price=price)
    db.session.add(rubber)
    db.session.commit()
    return 'OK'
```
