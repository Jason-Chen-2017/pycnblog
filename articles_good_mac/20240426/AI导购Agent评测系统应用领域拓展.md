import pandas as pd
from sklearn.metrics.pairwise import cosine_similarity

# 加载数据
data = pd.read_csv('ratings.csv')

# 计算用户相似度矩阵
user_similarity = cosine_similarity(data.pivot_table(index='userId', columns='movieId', values='rating'))

# 为用户推荐商品
def recommend(userId, num_recommendations):
    # 找到与目标用户最相似的用户
    similar_users = user_similarity[userId].argsort()[-num_recommendations-1:-1]
    
    # 获取相似用户评价过的商品
    similar_items = data[data['userId'].isin(similar_users)]['movieId'].unique()
    
    # 推荐相似用户评价过的商品
    return similar_items
