# 定义空调控制函数
def control_ac(temperature):
    if temperature > target_temperature + 1:
        # 打开空调
        print("打开空调")
    elif temperature < target_temperature - 1:
        # 关闭空调
        print("关闭空调")

# 主循环
while True:
    try:
        # 读取温度
        temperature = dht_device.temperature

        # 控制空调
        control_ac(temperature)

        # 等待一段时间
        time.sleep(1)
    except RuntimeError as error:
        # 处理传感器错误
        print(error.args[0])
        time.sleep(2.0)
        continue
    except Exception as error:
        # 处理其他错误
        dht_device.exit()
        raise error
