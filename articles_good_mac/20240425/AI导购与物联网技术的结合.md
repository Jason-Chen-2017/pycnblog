# 计算用户之间的相似度矩阵
user_similarity = cosine_similarity(ratings)

# 预测用户对未评分物品的评分
def predict_rating(user_id, item_id):
    # 找到与目标用户相似的用户
    similar_users = user_similarity[user_id].argsort()[::-1][1:]

    # 计算相似用户对目标物品的平均评分
    similar_user_ratings = ratings.loc[similar_users, item_id]
    return similar_user_ratings.mean()

# 推荐商品
def recommend_items(user_id, num_recommendations):
    # 预测用户对所有未评分物品的评分
    predicted_ratings = ratings.apply(lambda x: predict_rating(user_id, x.name), axis=1)

    # 返回评分最高的 num_recommendations 个物品
    return predicted_ratings.sort_values(ascending=False)[:num_recommendations]
