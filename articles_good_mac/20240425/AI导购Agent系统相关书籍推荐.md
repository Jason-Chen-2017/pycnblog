### 4.2 协同过滤算法

协同过滤算法可以使用以下公式计算用户之间的相似度：

$$
sim(u, v) = \frac{\sum_{i \in I_{uv}}(r_{ui} - \bar{r}_u)(r_{vi} - \bar{r}_v)}{\sqrt{\sum_{i \in I_{uv}}(r_{ui} - \bar{r}_u)^2}\sqrt{\sum_{i \in I_{uv}}(r_{vi} - \bar{r}_v)^2}}

其中，$u$ 和 $v$ 表示两个用户，$I_{uv}$ 表示两个用户都评价过的商品集合，$r_{ui}$ 表示用户 $u$ 对商品 $i$ 的评分，$\bar{r}_u$ 表示用户 $u$ 的平均评分。

import nltk
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 定义商品知识库
products = {
    "p1": {"name": "T恤", "category": "服装", "price": 99},
    "p2": {"name": "牛仔裤", "category": "服装", "price": 199},
    "p3": {"name": "运动鞋", "category": "鞋子", "price": 399},
    # ...
}

# 定义用户输入
user_input = "我想买一件T恤"

# 使用 nltk 进行分词
tokens = nltk.word_tokenize(user_input)

# 使用 TF-IDF 提取文本特征
vectorizer = TfidfVectorizer()
product_vectors = vectorizer.fit_transform(products.values())
user_vector = vectorizer.transform([user_input])

# 计算商品与用户之间的相似度
similarities = cosine_similarity(user_vector, product_vectors)

# 获取相似度最高的商品
recommended_product = list(products.keys())[similarities.argmax()]

# 输出推荐结果
print(f"推荐商品: {products[recommended_product]['name']}")
