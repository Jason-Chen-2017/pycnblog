def assign_responsibilities(tasks, team_members):
    # 责任分配矩阵
    responsibility_matrix = [[0] * len(tasks) for _ in range(len(team_members))]
    
    # 分配任务
    for i, task in enumerate(tasks):
        for j, member in enumerate(team_members):
            if member == task['responsible']:
                responsibility_matrix[j][i] = 1
    
    return responsibility_matrix

# 代码解读
def main():
    # 任务列表
    tasks = [
        {'name': '任务1', 'responsible': '成员1'},
        {'name': '任务2', 'responsible': '成员2'},
        {'name': '任务3', 'responsible': '成员3'}
    ]
    
    # 团队成员列表
    team_members = ['成员1', '成员2', '成员3']
    
    # 责任分配
    responsibility_matrix = assign_responsibilities(tasks, team_members)
    
    # 打印责任分配矩阵
    for row in responsibility_matrix:
        print(row)

if __name__ == '__main__':
    main()
