class Block:
    def __init__(self, index, previous_hash, timestamp, data, hash):
        self.index = index
        self.previous_hash = previous_hash
        self.timestamp = timestamp
        self.data = data
        self.hash = hash

class Blockchain:
    def __init__(self):
        self.chain = [self.create_genesis_block()]
    
    def create_genesis_block(self):
        return Block(0, "0", datetime.now(), "Genesis Block", self.calculate_hash(0, "0", datetime.now(), "Genesis Block"))
    
    def calculate_hash(self, index, previous_hash, timestamp, data):
        return hashlib.sha256((str(index) + previous_hash + str(timestamp) + data).encode()).hexdigest()
    
    def add_block(self, data):
        last_block = self.chain[-1]
        index = last_block.index + 1
        previous_hash = last_block.hash
        timestamp = datetime.now()
        hash = self.calculate_hash(index, previous_hash, timestamp, data)
        new_block = Block(index, previous_hash, timestamp, data, hash)
        self.chain.append(new_block)

## 4. 数学模型和公式 & 详细讲解 & 举例说明
### 4.1 机器学习数学模型
机器学习数学模型可以用于处理和分析数字化遗产中的数据。以下是一个简单的机器学习数学模型公式示例：

$$
\hat{y} = \text{sign}(\mathbf{w}^T \mathbf{x} + b)
$$

其中，$\hat{y}$ 是预测值，$\mathbf{w}$ 是权重向量，$\mathbf{x}$ 是特征向量，$b$ 是偏置项。

### 4.2 区块链数学模型
区块链数学模型可以用于确保数字化遗产的安全性和不可篡改性。以下是一个简单的区块链数学模型公式示例：

$$
\text{hash} = \text{SHA256}(\text{index} + \text{previous\_hash} + \text{timestamp} + \text{data})
$$

其中，hash 是区块的哈希值，index 是区块的索引，previous\_hash 是前一个区块的哈希值，timestamp 是区块的生成时间，data 是区块的数据。

### 4.3 情感计算数学模型
情感计算数学模型可以用于分析用户的情感状态。以下是一个简单的情感计算数学模型公式示例：

$$
\text{emotion} = \text{softmax}(\mathbf{W} \mathbf{x} + \mathbf{b})
$$

其中，emotion 是用户的情感状态，$\mathbf{W}$ 是权重矩阵，$\mathbf{x}$ 是特征向量，$\mathbf{b}$ 是偏置项。
