### 5.2 源代码详细实现和代码解读
假设我们有一个情感词典，包含以下情感词汇及其情感标签：
- 快乐：正面
- 悲伤：负面
- 中性：中性

假设我们有一个文本：“我今天很快乐，但是有些悲伤。”

```python
import nltk
from nltk.tokenize import word_tokenize
from nltk.corpus import sentiwordnet as swn

# 加载情感词典
nltk.download('sentiwordnet')
nltk.download('punkt')

# 情感词汇的识别
def identify_sentiment_words(text):
    tokens = word_tokenize(text)
    sentiment_words = []
    for token in tokens:
        synsets = swn.senti_synsets(token)
        if synsets:
            sentiment_words.append((token, synsets[0].pos_score(), synsets[0].neg_score()))
    return sentiment_words

# 情感标签的分配
def assign_sentiment_labels(sentiment_words):
    sentiment_labels = []
    for word, pos_score, neg_score in sentiment_words:
        if pos_score > neg_score:
            sentiment_labels.append((word, '正面'))
        elif neg_score > pos_score:
            sentiment_labels.append((word, '负面'))
        else:
            sentiment_labels.append((word, '中性'))
    return sentiment_labels

# 情感倾向的计算
def calculate_sentiment_tendency(sentiment_labels):
    positive_count = 0
    negative_count = 0
    neutral_count = 0
    for _, label in sentiment_labels:
        if label == '正面':
            positive_count += 1
        elif label == '负面':
            negative_count += 1
        else:
            neutral_count += 1
    total_count = positive_count + negative_count + neutral_count
    positive_tendency = positive_count / total_count
    negative_tendency = negative_count / total_count
    neutral_tendency = neutral_count / total_count
    return positive_tendency, negative_tendency, neutral_tendency

# 情感分析的流程
def sentiment_analysis(text):
    sentiment_words = identify_sentiment_words(text)
    sentiment_labels = assign_sentiment_labels(sentiment_words)
    positive_tendency, negative_tendency, neutral_tendency = calculate_sentiment_tendency(sentiment_labels)
    return positive_tendency, negative_tendency, neutral_tendency

# 测试
text = "我今天很快乐，但是有些悲伤。"
positive_tendency, negative_tendency, neutral_tendency = sentiment_analysis(text)
print(f"正面情感比例：{positive_tendency:.2f}")
print(f"负面情感比例：{negative_tendency:.2f}")
print(f"中性情感比例：{neutral_tendency:.2f}")
```
