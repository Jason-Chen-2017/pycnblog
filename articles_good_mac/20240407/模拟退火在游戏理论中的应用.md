```python
import numpy as np
import math

# 定义博弈论模型
def game_model(strategy):
    # 计算各参与者的收益
    player1_reward = 2 * strategy[0] * (1 - strategy[1])
    player2_reward = 2 * strategy[1] * (1 - strategy[0])
    return [player1_reward, player2_reward]

# 模拟退火算法
def simulated_annealing(init_strategy, max_iter, init_temp, alpha):
    strategy = init_strategy
    temp = init_temp
    
    for i in range(max_iter):
        # 在解空间内随机产生新解
        new_strategy = strategy + np.random.uniform(-0.1, 0.1, size=2)
        new_strategy = np.clip(new_strategy, 0, 1)
        
        # 计算新旧解的收益
        old_reward = game_model(strategy)
        new_reward = game_model(new_strategy)
        
        # 以概率接受新解
        delta_reward = sum(new_reward) - sum(old_reward)
        prob = min(1, math.exp(delta_reward / temp))
        if np.random.rand() < prob:
            strategy = new_strategy
        
        # 降低温度
        temp *= alpha
    
    return strategy

# 测试
init_strategy = np.array([0.5, 0.5])
optimal_strategy = simulated_annealing(init_strategy, 1000, 100, 0.95)
print("Optimal strategy:", optimal_strategy)
print("Payoffs:", game_model(optimal_strategy))
```

3. 降低温度,重复步骤1-2,直到满足终止条件。
4. 输出最终的最优策略组合。
