#### 代码实现

```python
import numpy as np
from scipy.integrate import odeint
import matplotlib.pyplot as plt

# 定义微分方程
def element_cycle(E, t, alpha, beta, gamma, delta):
    dE_dt = alpha * E + beta * E + gamma * E + delta * E
    return dE_dt

# 定义参数
alpha = 0.1
beta = 0.2
gamma = 0.3
delta = 0.4
E0 = 1.0

# 定义时间范围
t = np.linspace(0, 10, 100)

# 求解微分方程
E = odeint(element_cycle, E0, t, args=(alpha, beta, gamma, delta))

# 绘制结果
plt.plot(t, E)
plt.xlabel('时间 (t)')
plt.ylabel('元素浓度 (E)')
plt.title('元素循环过程')
plt.show()
```
