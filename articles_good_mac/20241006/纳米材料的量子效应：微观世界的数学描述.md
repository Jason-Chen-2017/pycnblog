import numpy as np
from scipy.sparse import diags
from scipy.sparse.linalg import eigsh
import matplotlib.pyplot as plt

# 定义量子点模型
def quantum_dot_model(N, L, V0, hbar, m, t):
    # 离散化空间
    x = np.linspace(0, L, N+1)
    dx = x[1] - x[0]
    
    # 建立哈密顿算子
    H = diags([-1, 2, -1], [-1, 0, 1], shape=(N, N)).toarray()
    H[0, 0] = H[-1, -1] = 0  # 边界条件
    H = H / (2 * m * dx**2) + V0 * np.eye(N)  # 加上势能
    
    # 求解薛定谔方程
    E, psi = eigsh(H, k=1, which='SM', sigma=0)  # 选择最低能量态
    psi = psi[:, 0]
    
    # 计算波函数的模平方
    psi_squared = np.abs(psi)**2
    
    # 绘制结果
    plt.plot(x, psi_squared)
    plt.xlabel('x')
    plt.ylabel('|ψ|^2')
    plt.title('Quantum Dot Model')
    plt.show()
    
    return E, psi

# 定义量子隧穿模型
def quantum_tunnelling_model(N, L, V0, hbar, m, t):
    # 离散化空间
    x = np.linspace(0, L, N+1)
    dx = x[1] - x[0]
    
    # 建立哈密顿算子

    H = H / (2 * m * dx**2) + V0 * np.eye(N)  # 加上势能

    # 绘制结果

    plt.xlabel('x')

    plt.title('Quantum Tunnelling Model')
    plt.show()
