\frac{dC_s}{dt} = k_{a,s} C_a - k_{s,a} C_s + k_{s,r} C_s - k_{r,s} C_r

def runge_kutta(C, k, dt):
    k1 = dt * (-k[0,1] * C[0] + k[1,0] * C[1] + k[0,2] * C[0] - k[2,0] * C[2])
    k2 = dt * (k[0,1] * C[0] - k[1,0] * C[1] + k[1,3] * C[1] - k[3,1] * C[3])
    k3 = dt * (k[2,0] * C[2] - k[0,2] * C[0] + k[2,3] * C[2] - k[3,2] * C[3])
    k4 = dt * (k[3,1] * C[3] - k[1,3] * C[1])
    C[0] += (k1 + 2 * k2 + 2 * k3 + k4) / 6
    C[1] += (k1 + 2 * k2 + 2 * k3 + k4) / 6

    C[3] += (k1 + 2 * k2 + 2 * k3 + k4) / 6

import matplotlib.pyplot as plt
