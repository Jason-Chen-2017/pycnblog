def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# 测试代码

import numpy as np
from scipy.integrate import quad

def lagrangian(q, q_dot, t):
    return 0.5 * q_dot**2 - q**2

def path_integral(t1, t2, q1, q2):
    def integrand(t):
        q = q1 + (q2 - q1) * (t - t1) / (t2 - t1)
        q_dot = (q2 - q1) / (t2 - t1)
        return np.exp(-lagrangian(q, q_dot, t))
    return quad(integrand, t1, t2)[0]

# 测试代码
