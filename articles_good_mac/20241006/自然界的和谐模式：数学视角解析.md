### 2.1 斐波那契数列
斐波那契数列是一个经典的数学序列，其定义如下：
$$
F(n) = \begin{cases} 
0 & \text{if } n = 0 \\
1 & \text{if } n = 1 \\
F(n-1) + F(n-2) & \text{if } n > 1 
\end{cases}
$$

#### 3.1.1 递归方法
```python
def fibonacci_recursive(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci_recursive(n-1) + fibonacci_recursive(n-2)
```

#### 3.1.2 迭代方法
```python
def fibonacci_iterative(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    a, b = 0, 1
    for _ in range(2, n+1):
        a, b = b, a + b
    return b
```

```python
import numpy as np
import matplotlib.pyplot as plt

def bezier_curve(points, t):
    if len(points) == 2:
        return (1 - t) * points[0] + t * points[1]
    else:
        return bezier_curve([a + (1 - t) * (b - a) for a, b in zip(points[:-1], points[1:])], t)

# 生成贝塞尔曲线
points = [(0, 0), (1, 1), (2, 0)]
t_values = np.linspace(0, 1, 100)
curve_points = [bezier_curve(points, t) for t in t_values]
plt.plot([p[0] for p in curve_points], [p[1] for p in curve_points])
plt.show()
```

#### 5.2.4 傅里叶变换的实现
```python
import numpy as np
import matplotlib.pyplot as plt

def fourier_transform(signal):
    n = len(signal)
    if n == 1:
        return signal
    else:
        even = fourier_transform(signal[::2])
        odd = fourier_transform(signal[1::2])
        factor = [np.exp(-2j * np.pi * k / n) for k in range(n // 2)]
        return [even[k] + factor[k] * odd[k] for k in range(n // 2)] + [even[k] - factor[k] * odd[k] for k in range(n // 2)]

# 生成信号
signal = np.sin(2 * np.pi * 5 * np.linspace(0, 1, 1000))
# 计算傅里叶变换
transformed_signal = fourier_transform(signal)
# 绘制信号和傅里叶变换
plt.figure(figsize=(12, 6))
plt.subplot(2, 1, 1)
plt.plot(signal)
plt.title('Original Signal')
plt.subplot(2, 1, 2)
plt.plot(np.abs(transformed_signal))
plt.title('Fourier Transform')
plt.show()
```
