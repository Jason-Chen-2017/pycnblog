```python
import numpy as np

def analyze_data(df):
    # 数据整理
    data = df.values
    # 数据分析
    mean_temp = np.mean(data[:, 0])
    mean_humidity = np.mean(data[:, 1])
    mean_precipitation = np.mean(data[:, 2])
    # 结果解释
    print(f"平均温度：{mean_temp}°C")
    print(f"平均湿度：{mean_humidity}%")
    print(f"平均降水量：{mean_precipitation}mm")
    # 结论形成
    if mean_temp > 27:
        print("存在气候变化趋势")
    else:
        print("气候变化趋势不明显")
```
