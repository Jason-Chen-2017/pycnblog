def cosine_similarity(ratings):
    """
    计算用户之间的余弦相似度。

    Args:
        ratings: 用户评分矩阵。

    Returns:
        用户相似度矩阵。
    """
    # 计算用户之间的余弦相似度
    similarity = np.dot(ratings, ratings.T)
    # 归一化
