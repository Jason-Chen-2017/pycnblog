import pandas as pd
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans

# 1. 数据收集
data = pd.read_csv('user_behavior.csv')

# 2. 数据清洗
data = data.dropna()

# 3. 数据可视化
plt.plot(data['阶段'], data['行为路径'])
plt.xlabel('阶段')
plt.ylabel('行为路径')
plt.title('用户旅程路径')
plt.show()

# 4. 用户画像构建
kmeans = KMeans(n_clusters=3)
kmeans.fit(data[['阶段', '行为路径']])
data['用户画像'] = kmeans.labels_

# 5. 瓶颈识别
data.groupby('用户画像')['行为路径'].mean().plot(kind='bar')
plt.xlabel('用户画像')
plt.ylabel('行为路径')
plt.title('用户画像与行为路径的关系')
plt.show()
