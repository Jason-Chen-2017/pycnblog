```R
# 数据录入
data <- data.frame(
  Group = c("A", "B", "C", "D", "E"),
  Value = c(5, 10, 15, 20, 25)
)

# 建立t检验模型
model <- lm(Value ~ Group, data = data)

# 进行t检验
t_result <- summary(model)$coefficients
print(t_result)
```
