# 定义一个映射
class Mapping:
    def __init__(self, source, target, function):
        self.source = source
        self.target = target
        self.function = function

    # 计算映射的连续性
    def is_continuous(self):
        for point in self.source.points:
            if not self.function(point) in self.target.points:
                return False
        return True

    # 计算映射的同胚性
    def is_homeomorphism(self):

    def __init__(self, mapping):
        self.mapping = mapping

    # 计算障碍类的值
    def compute(self):
        if self.mapping.is_homeomorphism():
            return 0
        else:
            return np.inf
