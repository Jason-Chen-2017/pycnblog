def forcing(A, B):
    # 初始化力迫过程
    F = A.copy()
    # 添加新元素
    for x in range(len(A)):
        F.append(x + len(A))
    # 寻找SCH最小反例
    for x in F:
        if x not in B:
            return True
    return False

# 测试代码
