\[ \Psi[\phi] = \frac{1}{Z} \int D[\phi] \, e^{iS[\phi]} \]

\[ \langle \phi(x) \rangle = \frac{1}{Z} \int D[\phi] \, \phi(x) \, e^{iS[\phi]} \]
