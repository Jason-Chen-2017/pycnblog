```python
import numpy as np
import matplotlib.pyplot as plt

def heart_rate_model(t, hr_base, amplitude, omega, phi):
    """
    心率模型函数。
    
    :param t: 时间序列。
    :param hr_base: 基础心率。
    :param amplitude: 心率幅值。
    :param omega: 角频率。
    :param phi: 相位。
    :return: 心率序列。
    """
    hr_t = hr_base + amplitude * np.sin(omega * t + phi)
    return hr_t

# 参数设置
hr_base = 80  # 基础心率
amplitude = 10  # 心率幅值
omega = np.pi / 30  # 角频率
phi = 0  # 相位

# 时间序列
t = np.linspace(0, 120, 1200)

# 模型计算
hr_t = heart_rate_model(t, hr_base, amplitude, omega, phi)

# 绘图
plt.plot(t, hr_t)
plt.xlabel('Time (s)')
plt.ylabel('Heart Rate (bpm)')
plt.title('Heart Rate Model')
plt.show()
```
