```csharp
using UnityEngine;

public class VirtualBeach : MonoBehaviour
{
    public GameObject patientAvatar;
    public GameObject therapistAvatar;
    public Material beachMaterial;

    // 开始治疗的准备工作
    void Start()
    {
        // 创建患者虚拟化身
        patientAvatar = Instantiate<GameObject>(patientAvatar);
        patientAvatar.transform.position = new Vector3(0, 1, 0);

        // 创建治疗师虚拟化身
        therapistAvatar = Instantiate<GameObject>(therapistAvatar);
        therapistAvatar.transform.position = new Vector3(5, 1, 0);

        // 设置海滩材质
        beachMaterial.mainTextureScale = new Vector2(10, 10);
        beachMaterial.mainTextureWrapMode = TextureWrapMode.Repeat;
    }

    // 每帧更新
    void Update()
    {
        // 如果患者和治疗师距离太远，提示患者接近治疗师
        float distance = Vector3.Distance(patientAvatar.transform.position, therapistAvatar.transform.position);
        if (distance > 10)
        {
            Debug.Log("患者和治疗师距离太远，请患者接近治疗师。");
        }
    }
}
```

- **设置海滩材质**：为了创建逼真的海滩环境，我们使用了重复纹理材质，使海滩看起来更加广阔。
- **更新交互状态**：每帧更新时，计算患者和治疗师之间的距离，如果距离太远，就提示患者接近治疗师。
