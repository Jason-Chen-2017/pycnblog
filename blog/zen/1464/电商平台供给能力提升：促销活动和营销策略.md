def pricing_strategy(demand_model, behavior_model, data, cost):
    demand_data = data[['S', 'P', 'T']]
    behavior_data = data[['U', 'I', 'R']]
    
    demand_pred = demand_model.predict(demand_data[['S']])

    price = cost + demand_pred * behavior_pred
    return price
