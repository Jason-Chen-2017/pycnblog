from qiskit import IBMQ
from qiskit.providers.ibmq import least_busy

# 连接到量子计算机
provider = IBMQ-provider()

# 获取测量结果

# 解码测量结果
key = result.get_counts(qc)

# 绘制量子电路
qc.draw(output='mpl')
