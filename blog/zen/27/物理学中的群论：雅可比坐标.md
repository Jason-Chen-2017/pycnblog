import numpy as np
from scipy.integrate import solve_ivp

def potential_energy(r):
    """双原子分子的势能函数"""
    k = 1.0e7  # 弹簧常数
    return k * r**(-1)

def hamiltonian(J, theta, k=1.0e7):
    """哈密顿量"""
    pJ = np.sqrt(k * J)  # 动量的雅可比坐标表示
    pT = np.sqrt(k * np.sin(theta)**2)  # 动量的旋转坐标表示
    return pJ**2 / (2 * k) + pT**2 / (2 * k) + potential_energy(J)

def dhamiltonian_dJ(J, theta):
    """关于J的偏导数"""
    return -(k * np.sqrt(potential_energy(J)))

def dhamiltonian_dtheta(J, theta):
    """关于theta的偏导数"""
    return -(np.sqrt(k * J) * np.cos(theta))

def main():
    J0 = 1.0e-10  # 初始伸缩坐标值
    theta0 = np.pi / 4  # 初始旋转角度值
    t_span = (0, 10)
    t_eval = np.linspace(t_span[0], t_span[1], 1000)

    sol = solve_ivp(lambda t, y: [-dhamiltonian_dJ(y[0], y[1]), -dhamiltonian_dtheta(y[0], y[1])], t_span, [J0, theta0], t_eval=t_eval)

    # 可视化轨迹
    import matplotlib.pyplot as plt
    plt.figure(figsize=(10, 6))
    plt.plot(sol.t, sol.y[0], label="伸缩坐标轨迹")
    plt.plot(sol.t, sol.y[1], label="旋转角度轨迹")
    plt.xlabel("时间")
    plt.ylabel("坐标")
    plt.legend()
    plt.show()

if __name__ == "__main__":
    main()
