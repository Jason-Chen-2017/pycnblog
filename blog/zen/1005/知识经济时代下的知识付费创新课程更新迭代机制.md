import numpy as np
import tensorflow as tf

# 构建用户特征矩阵和课程特征矩阵
U = tf.Variable(tf.random.normal([m, k]))
V = tf.Variable(tf.random.normal([n, k]))

# 定义损失函数
loss = tf.reduce_mean(tf.square(R - UV^T))

# 定义优化器
optimizer = tf.optimizers.Adam()

# 模型训练
for epoch in range(num_epochs):
  with tf.GradientTape() as tape:
    predictions = UV^T
    loss_value = loss(predictions, R)
  grads = tape.gradient(loss_value, [U, V])
  optimizer.apply_gradients(zip(grads, [U, V]))

# 模型评估
accuracy = tf.reduce_mean(tf.cast(tf.equal(predictions, R), tf.float32))
print(f"Accuracy: {accuracy.numpy()}")
