### 5.2 源代码详细实现

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# 知识创造模型
def create_knowledge(n, innovation_rates):
    knowledge_counts = np.random.choice(n, size=n, p=innovation_rates)
    return knowledge_counts

# 知识演化模型
def evolve_knowledge(t, initial_knowledge, evolution_rates):
    evolved_knowledge = initial_knowledge * (1 + evolution_rates ** t)
    return evolved_knowledge

# 知识过时模型
def outdated_knowledge(t, initial_knowledge, decay_rates):
    outdated_knowledge = initial_knowledge * (1 - decay_rates ** t)
    return outdated_knowledge

# 知识更新模型
def update_knowledge(t, initial_knowledge, decay_rates, innovation_rates):
    evolved_knowledge = evolve_knowledge(t, initial_knowledge, innovation_rates)
    outdated_knowledge = outdated_knowledge(t, initial_knowledge, decay_rates)
    updated_knowledge = evolved_knowledge - outdated_knowledge
    return updated_knowledge

# 示例数据
n = 100  # 知识总数
innovation_rates = np.random.rand(n)  # 创新度概率分布
evolution_rates = np.random.rand(n)  # 演化度概率分布
decay_rates = np.random.rand(n)  # 过时度概率分布

# 知识创造
knowledge_counts = create_knowledge(n, innovation_rates)

# 知识演化
evolved_knowledge = evolve_knowledge(10, knowledge_counts, evolution_rates)

# 知识过时
outdated_knowledge = outdated_knowledge(10, knowledge_counts, decay_rates)

# 知识更新
updated_knowledge = update_knowledge(10, knowledge_counts, decay_rates, innovation_rates)

# 结果展示
plt.figure(figsize=(10, 5))
plt.plot(knowledge_counts, label='原始知识')
plt.plot(evolved_knowledge, label='演化知识')
plt.plot(outdated_knowledge, label='过时知识')
plt.plot(updated_knowledge, label='更新知识')
plt.xlabel('知识编号')
plt.ylabel('知识值')
plt.legend()
plt.show()
```
