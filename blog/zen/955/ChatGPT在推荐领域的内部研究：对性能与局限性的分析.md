# 生成推荐文案
def generate_recommendation(user, content):
    prompt = f"根据用户{user}的特征和商品{content}的特征，生成一条推荐文案。"
    response = openai.Completion.create(
        engine="text-davinci-002",

        max_tokens=50,
        n=1,
        stop=None,
        temperature=0.7,
    )
    return response.choices[0].text.strip()

def generate_recommendation(user, content):
