```python
# 导入必要的库
import pymongo
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# 连接MongoDB数据库
client = pymongo.MongoClient("mongodb://localhost:27017/")
db = client["knowledge_matrix"]
users_collection = db["users"]
contents_collection = db["contents"]

# 从数据库中提取用户行为数据和内容数据
users_data = pd.DataFrame(list(users_collection.find()))
contents_data = pd.DataFrame(list(contents_collection.find()))

# 对内容进行标签化处理
vectorizer = TfidfVectorizer()
content_matrix = vectorizer.fit_transform(contents_data["content"])

# 构建用户画像
user_profiles = []
for user in users_data.iterrows():
    user行为的文本 = " ".join([行为["text"] for 行为 in user行为的列表])
    user_profile = vectorizer.transform([user行为的文本])
    user_profiles.append(user_profile)

# 计算用户与内容的相似度
user_content_similarity = cosine_similarity(user_profiles, content_matrix)

# 推荐结果
recommends = []
for i, user_similarity in enumerate(user_content_similarity):
    recommended_contents = []
    for j, similarity in enumerate(user_similarity):
        if similarity > 0.5:
            recommended_contents.append(contents_data.iloc[j]["title"])
    recommends.append(recommended_contents)

# 存储推荐结果
for i, recommend in enumerate(recommends):
    users_collection.update_one({"_id": users_data.iloc[i]["_id"]}, {"$set": {"recommends": recommend}})

print("推荐结果已更新到数据库。")
```
