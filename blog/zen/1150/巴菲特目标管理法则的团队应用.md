import pandas as pd

# 定义目标管理类
class BuffettGoalManagement:
    def __init__(self, team_members, goals):
        self.team_members = team_members
        self.goals = goals

    def set_goals(self, member, goal):
        self.goals[member] = goal

    def get_goals(self, member):
        return self.goals[member]

    def update_goals(self, member, new_goal):
        self.goals[member] = new_goal

    def evaluate_performance(self):
        performance = {}
        for member, goal in self.goals.items():
            performance[member] = goal['completed'] / goal['total']
        return performance

bmg = BuffettGoalManagement(team_members, goals)

# 更新目标

# 评估绩效
