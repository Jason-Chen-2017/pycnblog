2. 对除菌设备进行实时监控，确保除菌效果最优。

3. 调整除菌设备的工作参数，如温度、湿度等，以适应不同的除菌需求。

#### 3.2.4 除菌效果评估

1. 对除菌过程进行实时监控，记录除菌设备的运行参数。

2. 根据除菌设备的运行参数和传感器收集的数据，评估除菌效果。

3. 若除菌效果未达到预期，重新调整除菌策略和控制参数。

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from mermaid import Mermaid

# 数据采集与处理
def data_collection():
    # 假设传感器采集的数据为细菌和病毒浓度、温度、湿度等
    data = {
        'Bacteria': [100, 150, 200, 250, 300],
        'Temperature': [30, 35, 40, 45, 50],
        'Humidity': [60, 65, 70, 75, 80]
    }
    df = pd.DataFrame(data)
    return df

def data_preprocessing(df):
    # 数据预处理，如滤波、去噪等
    dffiltered = df.rolling(window=3).mean()
    return dffiltered

# 除菌策略选择
def strategy_selection(df):
    # 根据细菌和病毒浓度、温度、湿度等选择除菌策略
    strategy = 'High Temperature'
    if df['Bacteria'].mean() < 150:
        strategy = 'Chemical Disinfection'
    elif df['Temperature'].mean() > 40:
        strategy = 'UV Disinfection'
    return strategy

# 除菌过程控制
def process_control(strategy, df):
    # 根据选择的除菌策略，调整除菌设备的工作参数
    if strategy == 'High Temperature':
        temperature = df['Temperature'].mean() + 10
        humidity = df['Humidity'].mean()
    elif strategy == 'Chemical Disinfection':
        temperature = df['Temperature'].mean()
        humidity = df['Humidity'].mean() + 10
    elif strategy == 'UV Disinfection':
        temperature = df['Temperature'].mean()
        humidity = df['Humidity'].mean()
    return temperature, humidity

# 除菌效果评估
def effect_evaluation(df, temperature, humidity):
    # 根据除菌设备的运行参数和传感器收集的数据，评估除菌效果
    effect = df['Bacteria'].mean() < 50
    return effect

# 主函数
def main():
    df = data_collection()
    dffiltered = data_preprocessing(df)
    strategy = strategy_selection(dffiltered)
    temperature, humidity = process_control(strategy, dffiltered)
    effect = effect_evaluation(dffiltered, temperature, humidity)
    print(f"Strategy: {strategy}")
    print(f"Temperature: {temperature}^\circ C")
    print(f"Humidity: {humidity}%")
    print(f"Effect: {'Satisfied' if effect else 'Unsatisfied'}")

    # 生成 Mermaid 流程图
    mermaid = Mermaid()
    mermaid.add_node('Start', 'Start')
    mermaid.add_node('Data Collection', 'Data Collection')
    mermaid.add_node('Data Preprocessing', 'Data Preprocessing')
    mermaid.add_node('Strategy Selection', 'Strategy Selection')
    mermaid.add_node('Process Control', 'Process Control')
    mermaid.add_node('Effect Evaluation', 'Effect Evaluation')
    mermaid.add_node('End', 'End')
    mermaid.add_edge('Start', 'Data Collection')
    mermaid.add_edge('Data Collection', 'Data Preprocessing')
    mermaid.add_edge('Data Preprocessing', 'Strategy Selection')
    mermaid.add_edge('Strategy Selection', 'Process Control')
    mermaid.add_edge('Process Control', 'Effect Evaluation')
    mermaid.add_edge('Effect Evaluation', 'End')
    print(mermaid.generate_html())

if __name__ == '__main__':
    main()
```
