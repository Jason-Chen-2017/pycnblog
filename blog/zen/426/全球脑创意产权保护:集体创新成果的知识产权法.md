### 5.2 源代码详细实现

```python
from ethereum import utils
from web3 import Web3

# 创建Web3实例
web3 = Web3(Web3.HTTPProvider('http://localhost:8545'))

# 创建智能合约
contract_address = '0x1234567890'
contract = web3.eth.contract(address=contract_address, abi='...')

# 注册创意作品
def register_work(title, content, contributors):
    # 生成数据指纹
    fingerprint = utils.keccak(title + content)
    # 记录时间戳
    timestamp = utils.timestamp()
    # 记录贡献度
    contribution = utils.keccak(contributors)
    # 调用智能合约注册
    tx_hash = contract.functions.registerWork(title, content, fingerprint, timestamp, contribution).transact({'send': '0xabcdef'})
    tx_receipt = web3.eth.getTransactionReceipt(tx_hash)
    return tx_receipt

# 分配版权收益
def distribute_royalty(tx_receipt):
    # 获取版权收益
    royalty = contract.functions.getRoyalty(tx_receipt['tx_hash']).transact({'send': '0xabcdef'})
    royalty_receipt = web3.eth.getTransactionReceipt(royalty)
    # 分配版权收益
    for contributor in royalty_receipt['contract_addr']:
        # 提取贡献度与版权收益
        contribution = contract.functions.getContribution(contributor).transact({'send': '0xabcdef'})
        royalty = contract.functions.getRoyalty(contribution['tx_hash']).transact({'send': '0xabcdef'})
        distribute_royalty(contribution['tx_hash'], royalty)

# 配置DRM系统
def configure_drm(work_id, license, usage_rule):
    # 调用智能合约配置DRM
    tx_hash = contract.functions.configureDRM(work_id, license, usage_rule).transact({'send': '0xabcdef'})
    tx_receipt = web3.eth.getTransactionReceipt(tx_hash)
    return tx_receipt
```
