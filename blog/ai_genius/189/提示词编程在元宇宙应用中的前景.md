```python
# 初始化nltk库
nltk.download('punkt')

# 用户输入提示词

    elif "友好" in tokens:

    else:
        behavior_code = "role explore()"
    return behavior_code

# 调用生成函数
behavior_code = generate_role_behavior(tokens)
print(behavior_code)
```
