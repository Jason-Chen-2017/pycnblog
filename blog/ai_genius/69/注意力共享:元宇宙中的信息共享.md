# 注意力模型构建
def build_attention_model(input_dim, hidden_size, num_users, num_items):
    # 用户嵌入层
    user_embedding = Embedding(input_dim, hidden_size)
    # 项目嵌入层
    item_embedding = Embedding(num_items, hidden_size)
    
    # 用户嵌入表示
    user_vector = user_embedding(tf.keras.layers.Input(shape=(1,)))
    # 项目嵌入表示
    item_vector = item_embedding(tf.keras.layers.Input(shape=(1,)))
    
    # 注意力机制
    attention = tf.keras.layers.Dot(activation='sigmoid')(user_vector, item_vector)
    # 加权求和
    weighted_item_vector = tf.reduce_sum(attention * item_vector, axis=1)
    
    # 全连接层
    merged_vector = tf.keras.layers.Concatenate()([weighted_item_vector, user_vector])
    output = Dense(1, activation='sigmoid')(merged_vector)
    
    # 构建模型
    model = Model(inputs=[user_vector, item_vector], outputs=output)
    model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
    
    return model

from tensorflow.keras.models import Model

# 注意力模型构建
def build_attention_model(input_dim, hidden_size, num_users, num_items):
    # 用户嵌入层
    user_embedding = Embedding(input_dim, hidden_size)
    # 项目嵌入层
    item_embedding = Embedding(num_items, hidden_size)
    
    # 用户嵌入表示
    user_vector = user_embedding(tf.keras.layers.Input(shape=(1,)))
    # 项目嵌入表示
    item_vector = item_embedding(tf.keras.layers.Input(shape=(1,)))
    
    # 注意力机制
    attention = tf.keras.layers.Dot(activation='sigmoid')(user_vector, item_vector)
    # 加权求和
    weighted_item_vector = tf.reduce_sum(attention * item_vector, axis=1)
    
    # 全连接层
    merged_vector = tf.keras.layers.Concatenate()([weighted_item_vector, user_vector])
    output = Dense(1, activation='sigmoid')(merged_vector)
    
    # 构建模型
    model = Model(inputs=[user_vector, item_vector], outputs=output)
    model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
    
    return model

# 模型训练

    model.fit([user_data, item_data], labels, epochs=epochs, batch_size=batch_size)
