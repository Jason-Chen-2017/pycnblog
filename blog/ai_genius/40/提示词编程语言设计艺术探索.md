class Interpreter:
    def __init__(self):
        self.env = {}  # 存储变量和其值

    def interpret(self, ast):
        for node in ast:
            if isinstance(node, AssignmentNode):
                self.assign(node.variable, node.value)
            elif isinstance(node, PrintNode):
                self.print_value(node.value)

    def assign(self, variable, value):
        self.env[variable] = value

```python
class Compiler:
    def __init__(self):
        self.symbol_table = {}  # 存储变量和其类型

    def compile(self, ast):
        bytecode = []
        for node in ast:
            if isinstance(node, AssignmentNode):
                bytecode.append(self.compile_assignment(node))
            elif isinstance(node, PrintNode):
                bytecode.append(self.compile_print(node))
        return bytecode

    def compile_assignment(self, node):
        # 编译赋值语句
        bytecode = [self.compile_variable(node.variable), self.compile_value(node.value)]
        bytecode.append("store")
        return bytecode

    def compile_print(self, node):
        # 编译打印语句
        bytecode = [self.compile_value(node.value)]
        bytecode.append("print")
        return bytecode

    def compile_variable(self, variable):
        # 编译变量引用
        return self.symbol_table[variable]

    def compile_value(self, value):
        # 编译值
        if isinstance(value, int):
            return ["int", value]
        elif isinstance(value, float):
            return ["float", value]
        elif isinstance(value, str):
            return ["string", value]
```

在这个示例中，`Compiler`类负责将AST编译为字节码。`compile`函数遍历AST，并调用相应的函数编译每个节点。
