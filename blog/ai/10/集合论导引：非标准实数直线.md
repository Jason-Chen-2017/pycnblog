class HyperReal:
    def __init__(self, sequence):
        self.sequence = sequence

    def __add__(self, other):
        return HyperReal([a + b for a, b in zip(self.sequence, other.sequence)])

    def __mul__(self, other):
        return HyperReal([a * b for a, b in zip(self.sequence, other.sequence)])

# 示例
