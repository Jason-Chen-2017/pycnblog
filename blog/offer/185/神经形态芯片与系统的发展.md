```python
import numpy as np

# 初始化权重和偏置
weights1 = np.random.rand(3, 3)
biases1 = np.random.rand(3, 1)

weights2 = np.random.rand(3, 3)
biases2 = np.random.rand(3, 1)

# 前向传播
def forward_propagation(x):
    layer1 = np.dot(x, weights1) + biases1
    layer2 = np.dot(layer1, weights2) + biases2
    return layer2

# 输入数据
x = np.random.rand(3, 1)

# 计算输出
output = forward_propagation(x)
print(output)
```
