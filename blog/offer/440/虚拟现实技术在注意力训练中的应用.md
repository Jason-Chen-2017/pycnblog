##### 3.1 编写一个程序，模拟用户在虚拟现实环境中进行目标追踪的任务。

**题目：** 编写一个程序，模拟用户在虚拟现实环境中进行目标追踪的任务。程序应包括以下功能：
- 初始化目标位置；
- 用户输入方向键，控制目标移动；
- 显示目标当前位置。

```python
import pygame
import sys

# 初始化pygame
pygame.init()

# 设置屏幕尺寸
screen_width = 800
screen_height = 600
screen = pygame.display.set_mode((screen_width, screen_height))

# 设置目标初始位置
target_x = 400
target_y = 300

# 设置目标移动速度
speed = 5

# 设置目标颜色
color = (0, 0, 255) # BGR模式，即蓝色

# 设置字体
font = pygame.font.Font(None, 36)

# 主循环
running = True
while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # 用户输入
    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT]:
        target_x -= speed
    if keys[pygame.K_RIGHT]:
        target_x += speed
    if keys[pygame.K_UP]:
        target_y -= speed
    if keys[pygame.K_DOWN]:
        target_y += speed

    # 更新屏幕
    screen.fill((255, 255, 255)) # 设置屏幕背景色
    pygame.draw.circle(screen, color, (target_x, target_y), 20) # 绘制目标
    text = font.render("目标位置：({}, {})".format(target_x, target_y), True, (0, 0, 0))
    screen.blit(text, (10, 10)) # 显示目标位置
    pygame.display.flip() # 更新屏幕

# 退出程序
pygame.quit()
sys.exit()
```

##### 3.2 编写一个程序，模拟用户在虚拟现实环境中进行多任务处理。

**题目：** 编写一个程序，模拟用户在虚拟现实环境中进行多任务处理。程序应包括以下功能：
- 初始化任务列表；
- 用户输入数字键，分配任务给用户；
- 显示任务进度。

```python
import pygame
import sys

# 初始化pygame
pygame.init()

# 设置屏幕尺寸
screen_width = 800
screen_height = 600
screen = pygame.display.set_mode((screen_width, screen_height))

# 设置字体
font = pygame.font.Font(None, 36)

# 初始化任务列表
tasks = [
    {"name": "任务1", "progress": 0},
    {"name": "任务2", "progress": 0},
    {"name": "任务3", "progress": 0},
]

# 设置任务颜色
task_colors = [(0, 255, 0), (0, 0, 255), (255, 0, 0)]

# 主循环
running = True
while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # 用户输入
    keys = pygame.key.get_pressed()
    if keys[pygame.K_1]:
        tasks[0]["progress"] += 10
    if keys[pygame.K_2]:
        tasks[1]["progress"] += 10
    if keys[pygame.K_3]:
        tasks[2]["progress"] += 10

    # 更新屏幕
    screen.fill((255, 255, 255)) # 设置屏幕背景色
    for i, task in enumerate(tasks):
        text = font.render(task["name"], True, (0, 0, 0))
        bar_width = (screen_width - 20) * (task["progress"] / 100)
        bar_color = task_colors[i]
        pygame.draw.rect(screen, bar_color, (10, 50 + i*40, bar_width, 30))
        screen.blit(text, (10, 50 + i*40))
    pygame.display.flip() # 更新屏幕

# 退出程序
pygame.quit()
sys.exit()
```
