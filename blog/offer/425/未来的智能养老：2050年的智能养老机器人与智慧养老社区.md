def min_difference(nums):
    nums.sort()
    return min(nums[i+1] - nums[i] for i in range(len(nums) - 1))

**5. 编写一个基于计算机视觉的智能养老机器人，用于识别老年人的情绪状态。**

**题目：**

请使用 Python 和 OpenCV 库编写一个简单的智能养老机器人，通过计算机视觉技术识别老年人的情绪状态。

**答案解析：**

```python
import cv2
import numpy as np

# 加载情绪分类器
face_cascade = cv2.CascadeClassifier("haarcascade_frontalface_default.xml")
emotion_classifier = cv2.ml.SVM_load("emotion_svm_model.yml")

# 识别情绪状态
def recognize_emotion(frame):
    gray_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    faces = face_cascade.detectMultiScale(gray_frame, scaleFactor=1.1, minNeighbors=5, minSize=(30, 30), flags=cv2.CASCADE_SCALE_IMAGE)
    emotions = []
    for (x, y, w, h) in faces:
        roi = gray_frame[y:y+h, x:x+w]
        roi_flatten = roi.flatten()
        emotion_pred, _ = emotion_classifier.predict(roi_flatten)
        emotions.append(emotion_pred)
    return emotions

# 处理视频流并识别情绪状态
cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    if ret:
        emotions = recognize_emotion(frame)
        for (x, y, w, h), emotion in zip(faces, emotions):
            cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 0, 255), 2)
            cv2.putText(frame, str(emotion), (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 255, 255), 2)
        cv2.imshow("Emotion Recognition", frame)

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```
