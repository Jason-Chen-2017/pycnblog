**答案：** 

```python
def most_frequent_char(s: str) -> str:
    freq = {}
    for c in s:
        freq[c] = freq.get(c, 0) + 1
    max_freq = max(freq.values())
    return [c for c, f in freq.items() if f == max_freq][0]
```

##### **3. 实现一个函数，求两个有序数组的中间值。**

**答案：** 

```python
def findMiddle(nums1: List[int], nums2: List[int]) -> int:
    m, n = len(nums1), len(nums2)
    if m > n:
        nums1, nums2, m, n = nums2, nums1, n, m
    imin, imax, half_len = 0, m, (m + n + 1) // 2
    while imin <= imax:
        i = (imin + imax) // 2
        j = half_len - i
        if i < m and nums2[j - 1] > nums1[i]:
            imin = i + 1
        elif i > 0 and nums1[i - 1] > nums2[j]:
            imax = i - 1
        else:
            if i == 0:
                return nums2[j - 1]
            if j == 0:
                return nums1[i - 1]
            return min(nums1[i], nums2[j])
```

def longest_common_prefix(strs: List[str]) -> str:

##### **11. 实现一个函数，求一个字符串中的最长公共子序列。**

**答案：** 

```python
def longest_common_subsequence(s1: str, s2: str) -> str:
    m, n = len(s1), len(s2)
    dp = [["" for _ in range(n + 1)] for _ in range(m + 1)]
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i - 1] == s2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + s1[i - 1]
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
    return dp[-1][-1]
```

##### **12. 实现一个函数，求一个字符串中的最长重复子串。**

**答案：** 

```python
def longest_repeated_substring(s: str) -> str:
    n = len(s)
    lps = [0] * n
    length = 0
    i = 1
    while i < n:
        if s[i] == s[length]:
            length += 1
            lps[i] = length
            i += 1
        else:
            if length != 0:
                length = lps[length - 1]
            else:
                i += 1
    return s[:length]
```

```python
def longest_consecutive_sequence(arr: List[int]) -> int:
    arr.sort()
    longest_sequence = 1
    current_sequence = 1
    for i in range(1, len(arr)):
        if arr[i] == arr[i - 1] + 1:
            current_sequence += 1
            longest_sequence = max(longest_sequence, current_sequence)
        else:
            current_sequence = 1
    return longest_sequence
```
