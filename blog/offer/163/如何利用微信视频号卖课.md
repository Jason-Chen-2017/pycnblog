```go
// 假设用户行为数据存储在数据库中，这里使用简单的内存数据结构表示
var userBehavior = map[int]map[int]int{
    1: {101: 5, 102: 3},
    2: {102: 4, 103: 2},
    3: {101: 4, 103: 5},
}

// 用户推荐函数
func recommendCourse(userId int) []int {
    // 计算用户行为的相似度
    similarityMatrix := make(map[int]map[int]float64)
    for u1, behaviors1 := range userBehavior {
        for u2, behaviors2 := range userBehavior {
            if u1 == u2 {
                continue
            }
            commonBehaviors := 0
            for c1, score1 := range behaviors1 {
                if score2, exists := behaviors2[c1]; exists {
                    commonBehaviors++
                }
            }
            similarity := float64(commonBehaviors) / 10.0 // 这里假设相似度与共同行为数量成正比
            similarityMatrix[u1][u2] = similarity
            similarityMatrix[u2][u1] = similarity
        }
    }

    // 根据相似度推荐课程
    recommendedCourses := make([]int, 0)
    highestSimilarity := 0.0
    for u2, similarity := range similarityMatrix[userId] {
        if similarity > highestSimilarity {
            highestSimilarity = similarity
            recommendedCourses = []int{u2}
        } else if similarity == highestSimilarity {
            recommendedCourses = append(recommendedCourses, u2)
        }
    }

    return recommendedCourses
}
```

**解析：** 上述代码实现了一个简单的用户推荐系统，通过计算用户行为相似度，推荐相似的课程。实际应用中，可能需要更复杂的算法和更丰富的数据。
