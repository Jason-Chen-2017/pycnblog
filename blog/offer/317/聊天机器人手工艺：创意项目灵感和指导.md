```python
class ChatBot:
    def __init__(self):
        self.last_message = ""

    def respond_to_message(self, message):
        self.last_message = message.lower()
        if "hello" in self.last_message:
            return "Hello! How can I help you today?"
        elif "weather" in self.last_message:
            return "The weather today is sunny."
        else:
            return "I'm not sure how to respond to that."

# 使用示例
bot = ChatBot()
print(bot.respond_to_message("Hello!"))
print(bot.respond_to_message("What's the weather like?"))
print(bot.respond_to_message("Can you tell me a joke?"))
```

```python
from nltk import pos_tag
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize

class IntentRecognizer:
    def __init__(self):
        self.stop_words = set(stopwords.words('english'))

    def preprocess_text(self, text):
        words = word_tokenize(text.lower())
        filtered_words = [word for word in words if word not in self.stop_words]
        return filtered_words

    def recognize_intent(self, text):
        preprocessed_text = self.preprocess_text(text)
        if "find" in preprocessed_text:
            return "I can help you find information."
        elif "order" in preprocessed_text:
            return "I can help you place an order."
        else:
            return "I'm not sure what you want."

# 使用示例
recognizer = IntentRecognizer()
print(recognizer.recognize_intent("Can you help me find a restaurant?"))
print(recognizer.recognize_intent("I want to order a pizza."))
print(recognizer.recognize_intent("Can you tell me the weather?"))
```

**3. 实现一个聊天机器人，能够理解用户的情绪并给出相应的回复。**

**答案：**

```python
class EmotionDetector:
    def __init__(self):
        self.emotions = {
            "happy": ["joy", "happy", "great", "amazing"],
            "sad": ["sad", "sorrow", "unhappy", "bad"],
            "angry": ["angry", "rage", "mad", "frustrated"],
            "neutral": []
        }

    def detect_emotion(self, text):
        text = text.lower()
        for emotion, keywords in self.emotions.items():
            if any(keyword in text for keyword in keywords):
                return emotion
        return "neutral"

    def respond_to_emotion(self, emotion):
        if emotion == "happy":
            return "I'm glad to hear that! How can I assist you further?"
        elif emotion == "sad":
            return "I'm sorry to hear that. Is there anything I can do to help?"
        elif emotion == "angry":
            return "I understand you're frustrated. Let's try to resolve this calmly."
        else:
            return "It seems like you're in a neutral mood. How can I help you?"

# 使用示例
detector = EmotionDetector()
print(detector.detect_emotion("I'm so happy to see you!"))
print(detector.detect_emotion("I'm really upset about this."))
print(detector.detect_emotion("I'm just fine, thanks."))
print(detector.respond_to_emotion("happy"))
print(detector.respond_to_emotion("sad"))
print(detector.respond_to_emotion("neutral"))
```
