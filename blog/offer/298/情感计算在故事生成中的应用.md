```python
class Character:
    def __init__(self, name, emotions):
        self.name = name
        self.emotions = emotions

def generate_story(character):
    emotions = character.emotions
    if 'happy' in emotions:
        story = f"{character.name} 在这个美好的日子里，感到非常快乐。"
    elif 'sad' in emotions:
        story = f"{character.name} 看到了一件令人伤心的事情，他感到非常难过。"
    else:
        story = f"{character.name} 站在原地，情绪低落，不知道该怎么做。"
    return story

# 创建角色
character = Character("小明", ["happy", "sad"])

# 生成故事
story = generate_story(character)
print(story)
```

**解析：** 该示例程序定义了一个 `Character` 类，用于表示故事中的角色，并为角色赋予情感特征。`generate_story` 函数根据角色的情感特征生成相应的情节。运行该程序，将输出一个简单的情感故事。

### 编程题：

**7.4 编写一个情感识别的简单程序，实现对文本数据的情感分类。**

**答案：**

```python
# 情感识别示例

import nltk
from nltk.corpus import movie_reviews

# 加载电影评论数据集
nltk.download('movie_reviews')
nltk.download('stopwords')
nltk.download('wordnet')

def preprocess_text(text):
    # 删除停用词
    stopwords = set(nltk.corpus.stopwords.words('english'))
    words = nltk.word_tokenize(text)
    filtered_words = [word for word in words if word not in stopwords]
    # 词性标注
    pos_tags = nltk.pos_tag(filtered_words)
    # 提取名词和动词
    nouns_verbs = [word for word, tag in pos_tags if tag.startswith('NN') or tag.startswith('VB')]
    return ' '.join(nouns_verbs)

def train_classifier():
    # 训练情感分类器
    fileids = movie_reviews.fileids()
    labels = ['positive' if 'pos' in fileid else 'negative' for fileid in fileids]
    features = []
    for fileid, label in zip(fileids, labels):
        text = movie_reviews.raw(fileid)
        processed_text = preprocess_text(text)
        words = nltk.word_tokenize(processed_text)
        features.append({word: True for word in words})
    classifier = nltk.NaiveBayesClassifier.train(features)
    return classifier

def classify_text(text, classifier):
    # 对文本进行情感分类
    processed_text = preprocess_text(text)
    words = nltk.word_tokenize(processed_text)
    features = {word: True for word in words}
    return classifier.classify(features)

# 训练分类器
classifier = train_classifier()

# 测试文本
test_text = "I really enjoyed this movie, the plot was exciting and the actors performed well."
result = classify_text(test_text, classifier)
print("The sentiment of the text is:", result)
```

**解析：** 该示例程序使用 NLTK 库加载电影评论数据集，对文本数据进行预处理（去除停用词、提取名词和动词），训练一个朴素贝叶斯分类器，然后对输入的文本进行情感分类。运行该程序，将输出文本的情感分类结果。

```python
# 情感计算聊天机器人示例

import random

# 情感词汇库
happy_phrases = ["I'm happy too!", "That's great to hear!", "I'm glad for you!"]
sad_phrases = ["I'm sorry to hear that.", "That must be tough.", "Let me know if you need someone to talk to."]
angry_phrases = ["I can understand your frustration.", "That's not cool.", "Calm down and let's talk."]

    if "happy" in text.lower():
        return "happy"
    elif "sad" in text.lower() or "unhappy" in text.lower():
        return "sad"
    elif "angry" in text.lower() or "mad" in text.lower():
        return "angry"
    else:
        return "neutral"

# 聊天机器人函数
def chat_bot(text):
    emotion = detect_emotion(text)
    if emotion == "happy":
        response = random.choice(happy_phrases)
    elif emotion == "sad":
        response = random.choice(sad_phrases)
    elif emotion == "angry":
        response = random.choice(angry_phrases)
    else:
        response = "I'm here to chat whenever you need it."
    return response

# 测试聊天
user_input = "I'm really happy because I got a new job!"
print("User:", user_input)
print("Bot:", chat_bot(user_input))
```
