```python
import matplotlib.pyplot as plt

def calculate_release_curve(concentration, time_constant, time):
    release_rate = concentration / time_constant
    accumulated_release = 0
    time_points = []
    release_points = []
    
    for t in range(1, int(time) + 1):
        accumulated_release += release_rate
        time_points.append(t)
        release_points.append(accumulated_release)
    
    return time_points, release_points

concentration = float(input("请输入药物浓度（单位：mg/cm³）："))
time_constant = float(input("请输入药物释放的时间常数（单位：小时）："))
time = float(input("请输入药物释放的时间（单位：小时）："))
time_points, release_points = calculate_release_curve(concentration, time_constant, time)

plt.plot(time_points, release_points)
plt.xlabel("时间（小时）")
plt.ylabel("药物累积释放量（mg）")
plt.title("纳米颗粒药物释放曲线")
plt.show()
```
