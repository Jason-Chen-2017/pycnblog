# 转换图像数据类型

# 计算图像的归一化值
scaled_image = scaled_image / 255.0

# 将图像数据添加到批次中
batched_image = np.expand_dims(scaled_image, axis=0)

# 进行商品识别
model.setInput(batched_image)
predictions = model.forward()

# 获取最高置信度的类别
top_prediction = np.argmax(predictions[0])

# 获取对应的类别标签
label_map = {0: 'apple', 1: 'orange', 2: 'banana'} # 根据实际类别标签进行映射
