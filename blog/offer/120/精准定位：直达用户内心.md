**算法编程题：** 请使用 Python 编写一个基于协同过滤的个性化推荐算法。

```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

# 加载用户评分数据
data = pd.read_csv('user_data.csv')

# 计算用户和物品的相似度矩阵
user_similarity = cosine_similarity(data['ratings'].values)

# 生成推荐列表
def recommend(user_id, similarity_matrix, data, top_n=5):
    # 计算用户与其他用户的相似度
    user_similarity = similarity_matrix[user_id]

    # 计算用户与其他用户的相似度得分
    scores = user_similarity.dot(data['ratings'].values) / np.linalg.norm(data['ratings'].values, axis=1)

    # 对相似度得分进行降序排序
    sorted_scores = np.argsort(scores)[::-1]

    # 获取推荐列表
    recommendations = []
    for i in sorted_scores[1:top_n+1]:
        recommendations.append(data['item_id'][i])

    return recommendations

# 测试推荐系统
user_id = 0
recommendations = recommend(user_id, user_similarity, data)
print("推荐列表：", recommendations)
```

```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

# 加载用户评分数据
data = pd.read_csv('user_data.csv')

# 计算用户和物品的相似度矩阵
user_similarity = cosine_similarity(data['ratings'].values)

# 生成推荐列表
def recommend(user_id, similarity_matrix, data, top_n=5):
    # 计算用户与其他用户的相似度
    user_similarity = similarity_matrix[user_id]

    # 计算用户与其他用户的相似度得分
    scores = user_similarity.dot(data['ratings'].values) / np.linalg.norm(data['ratings'].values, axis=1)

    # 对相似度得分进行降序排序
    sorted_scores = np.argsort(scores)[::-1]

    # 获取推荐列表
    recommendations = []
    for i in sorted_scores[1:top_n+1]:
        recommendations.append(data['item_id'][i])

    return recommendations

# 测试推荐系统
user_id = 0
recommendations = recommend(user_id, user_similarity, data)
print("推荐列表：", recommendations)
