def calculate_recovery_prices(products):
    prices = []
    for product in products:
        price = 0
        if product['condition'] == '良好':
            price += 100
        elif product['condition'] == '较差':
            price += 50
        if product['material'] == '丝绸':
            price += 50
        prices.append(price)
    return prices

products = [{'name': '品牌A', 'material': '丝绸', 'condition': '良好'}, {'name': '品牌B', 'material': '棉布', 'condition': '较差'}, ...]

import bcrypt

def hash_password(password):
    salt = bcrypt.gensalt()
    hashed_password = bcrypt.hashpw(password.encode(), salt)
    return hashed_password

def verify_password(hashed_password, input_password):
    return bcrypt.checkpw(input_password.encode(), hashed_password)

# 用户注册

hashed_password = hash_password(password)
print(f"用户密码加密：{hashed_password}")
