### 算法编程题库

#### 1. 设计一个在线课程评价系统，实现以下功能：

* 用户可以发布评价。
* 用户可以查看某门课程的平均评价。
* 系统可以根据评价给出课程推荐。

```python
class CourseEvaluation:
    def __init__(self):
        self.courses = {}

    def add_evaluation(self, course_id, user_id, rating):
        if course_id not in self.courses:
            self.courses[course_id] = []
        self.courses[course_id].append((user_id, rating))

    def get_average_rating(self, course_id):
        if course_id not in self.courses:
            return 0
        ratings = [rating for _, rating in self.courses[course_id]]
        return sum(ratings) / len(ratings)

    def recommend_courses(self, user_id):
        recommended_courses = []
        for course_id, evaluations in self.courses.items():
            if user_id in [user_id for user_id, _ in evaluations]:
                continue
            average_rating = self.get_average_rating(course_id)
            recommended_courses.append((course_id, average_rating))
        recommended_courses.sort(key=lambda x: x[1], reverse=True)
        return recommended_courses
```

#### 2. 实现一个知识付费平台，支持以下功能：

* 用户注册、登录、充值、购买课程。
* 课程发布、更新、删除。
* 支持课程分类、标签、搜索功能。
* 支持课程评论、评分、推荐功能。

```python
class KnowledgePlatform:
    def __init__(self):
        self.users = {}
        self.courses = {}

    def register(self, username, password):
        if username in self.users:
            return "用户已存在"
        self.users[username] = password

    def login(self, username, password):
        if username not in self.users or self.users[username] != password:
            return "登录失败"
        return "登录成功"

    def recharge(self, username, amount):
        if username not in self.users:
            return "用户不存在"
        self.users[username]["balance"] += amount

    def purchase_course(self, username, course_id):
        if username not in self.users or self.users[username]["balance"] < 100:
            return "余额不足"
        self.users[username]["courses"].append(course_id)
        self.courses[course_id]["students"].append(username)

    def publish_course(self, course_id, title, description, price):
        self.courses[course_id] = {
            "title": title,
            "description": description,
            "price": price,
            "students": [],
        }

    def update_course(self, course_id, title, description, price):
        if course_id not in self.courses:
            return "课程不存在"
        self.courses[course_id]["title"] = title
        self.courses[course_id]["description"] = description
        self.courses[course_id]["price"] = price

    def delete_course(self, course_id):
        if course_id not in self.courses:
            return "课程不存在"
        del self.courses[course_id]

    def search_courses(self, keyword):
        results = []
        for course_id, course in self.courses.items():
            if keyword in course["title"] or keyword in course["description"]:
                results.append(course)
        return results

    def rate_course(self, username, course_id, rating):
        if username not in self.users or course_id not in self.courses:
            return "用户或课程不存在"
        self.courses[course_id]["ratings"].append(rating)

    def recommend_courses(self, username):
        recommended_courses = []
        for course_id, course in self.courses.items():
            if username in course["students"]:
                continue
            average_rating = sum(course["ratings"]) / len(course["ratings"])
            recommended_courses.append((course_id, average_rating))
        recommended_courses.sort(key=lambda x: x[1], reverse=True)
        return recommended_courses
```
