```python
class SeatingLayout:
    def __init__(self, num_rows, num_cols):
        self.layout = [[' ' for _ in range(num_cols)] for _ in range(num_rows)]
        self.seat_ids = []

    def assign_seat(self, row, col, seat_id):
        if self.layout[row][col] == ' ':
            self.layout[row][col] = seat_id
            self.seat_ids.append(seat_id)
            return True
        return False

    def generate_layout(self):
        layout_str = "座位布局：\n"
        for row in self.layout:
            layout_str += ''.join(row) + "\n"
        return layout_str

# 示例使用

print(layout.generate_layout())
```

class CostBenefitAnalysis:
    def __init__(self):
        self.costs = []
        self.profits = []

    def add_cost(self, cost):
        self.costs.append(cost)

    def add_profit(self, profit):
        self.profits.append(profit)

    def calculate_total_cost(self):
        return sum(self.costs)

    def calculate_total_profit(self):
        return sum(self.profits)

    def assess_cost_benefit_ratio(self):
        if self.calculate_total_profit() == 0:
            return 0
        return self.calculate_total_profit() / self.calculate_total_cost()

    def generate_report(self):
        report = "成本效益分析报告：\n"
        report += f"总成本：{self.calculate_total_cost()}元\n"
        report += f"总收益：{self.calculate_total_profit()}元\n"
        report += f"成本效益比：{self.assess_cost_benefit_ratio()}\n"
        return report

# 示例使用

analysis.add_cost(5000)
