```python
import numpy as np
from sklearn.feature_extraction.text import TfidfVectorizer

def train_tfidf(texts, labels):
    # 初始化TF-IDF向量器
    vectorizer = TfidfVectorizer()
    # 训练向量器
    vectorizer.fit(texts)
    # 转换文本数据为TF-IDF特征向量
    X = vectorizer.transform(texts)
    # 返回特征向量和标签
    return X, labels

def classify_tfidf(X_train, y_train, text):
    # 创建TF-IDF向量器，使用训练集的参数
    vectorizer = TfidfVectorizer()
    vectorizer.fit(X_train, y_train)
    # 转换文本数据为TF-IDF特征向量
    X_test = vectorizer.transform([text])
    # 使用训练集的模型进行分类
    from sklearn.linear_model import LogisticRegression
    classifier = LogisticRegression()
    classifier.fit(X_train, y_train)
    # 返回分类结果
    return classifier.predict(X_test)[0]

# 示例
texts = ["This is the first document.", "This document is the second document.", "And this is the third one.", "Is this the first document?"]
labels = [0, 0, 1, 1]
X, y = train_tfidf(texts, labels)
print("Classification of new text:", classify_tfidf(X, y, "This is a new document."))
```

**2. 请编写一个Python函数，实现一个基于K-means算法的聚类分析。**

**题目描述：** 编写一个Python函数，实现一个基于K-means算法的聚类分析，能够对给定数据集进行聚类，并输出每个簇的中心点和簇内成员。

**答案：**

```python
from sklearn.cluster import KMeans
import matplotlib.pyplot as plt

def kmeans_clustering(data, n_clusters):
    # 创建K-means聚类模型
    kmeans = KMeans(n_clusters=n_clusters)
    # 模型拟合数据
    kmeans.fit(data)
    # 返回聚类结果
    return kmeans

def plot_clusters(clustering, data):
    # 绘制聚类结果
    colors = ["g", "r", "c", "b", "k"]
    for i in range(len(colors)):
        # 过滤属于当前簇的数据点
        points = data[clustering.labels_ == i]
        # 绘制数据点
        plt.scatter(points[:, 0], points[:, 1], s=30, c=colors[i], edgecolor="k")
    # 绘制簇中心点
    centers = clustering.cluster_centers_
    plt.scatter(centers[:, 0], centers[:, 1], s=100, c="yellow", marker="s", edgecolor="k")
    plt.show()

# 示例
data = np.array([[1, 2], [1, 4], [1, 0], [4, 2], [4, 4], [4, 0]])
kmeans = kmeans_clustering(data, 2)
plot_clusters(kmeans, data)
```
