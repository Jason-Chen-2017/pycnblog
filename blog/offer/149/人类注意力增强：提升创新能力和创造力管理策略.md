```python
class Solution:
    def findShortestPath(self, graph: List[List[int]], start: int, target: int) -> int:
        def dfs(node, target, depth):
            if node == target:
                nonlocal ans
                ans = depth
                return
            for next_node in graph[node]:
                dfs(next_node, target, depth + 1)

        ans = float('inf')
        dfs(start, target, 0)
        return ans if ans != float('inf') else -1
```

```python
class BinarySearch:
    def __init__(self, nums):
        self.nums = sorted(nums)
    
    def find(self, num):
        left, right = 0, len(self.nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if self.nums[mid] == num:
                return True
            elif self.nums[mid] < num:
                left = mid + 1
            else:
                right = mid - 1
        return False
```

```python
from collections import deque

def solve_maze(maze):
    rows, cols = len(maze), len(maze[0])
    start, target = (0, 0), (rows - 1, cols - 1)
    visited = [[False] * cols for _ in range(rows)]
    q = deque([start])
    visited[start[0]][start[1]] = True

    while q:
        node = q.popleft()
        if node == target:
            return True
        for next_node in get_neighbors(maze, node, visited):
            q.append(next_node)
            visited[next_node[0]][next_node[1]] = True

    return False

def get_neighbors(maze, node, visited):
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    rows, cols = len(maze), len(maze[0])
    result = []
    for d in directions:
        new_node = (node[0] + d[0], node[1] + d[1])
        if 0 <= new_node[0] < rows and 0 <= new_node[1] < cols and not visited[new_node[0]][new_node[1]]:
            result.append(new_node)
    return result
```
