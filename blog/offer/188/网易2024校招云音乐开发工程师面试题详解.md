```go
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
```

**题目3：实现一个LRU缓存机制。**

**答案：**

```go
type LRUCache struct {
    cmap     map[int]*list.Element
    capacity int
    lruList  *list.List
}

func NewLRUCache(capacity int) *LRUCache {
    cache := &LRUCache{
        cmap:     make(map[int]*list.Element),
        capacity: capacity,
        lruList:  list.New(),
    }
    return cache
}

func (this *LRUCache) Get(key int) int {
    if element, found := this.cmap[key]; found {
        this.lruList.MoveToFront(element)
        return element.Value.(int)
    }
    return -1
}

func (this *LRUCache) Put(key int, value int) {
    if element, found := this.cmap[key]; found {
        this.lruList.MoveToFront(element)
        element.Value = value
    } else {
        element := this.lruList.PushFront(value)
        this.cmap[key] = element
        if len(this.cmap) > this.capacity {
            evict := this.lruList.Back()
            delete(this.cmap, evict.Value.(int))
            this.lruList.Remove(evict)
        }
    }
}
```

**解析：** 这道题用于考察链表和哈希表的使用，以及如何实现一个LRU缓存。

**题目4：实现一个排序算法。**

**答案：**

```go
func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}
```

**解析：** 这是一道简单的排序算法题，用于考察基本的编程和算法知识。

**题目5：简述TCP和UDP协议的区别。**

**答案：**

- TCP（传输控制协议）是一种面向连接、可靠的、基于字节流的传输层通信协议。
- UDP（用户数据报协议）是一种无连接、不可靠、基于数据报的传输层通信协议。

**解析：** 这道题用于考察对TCP和UDP协议的理解。
