   ```python
   import numpy as np

   def linear_regression(train_data, test_data):
       # 训练模型
       X_train, y_train = train_data
       X_test, y_test = test_data
       w = np.linalg.inv(X_train.T.dot(X_train)).dot(X_train.T).dot(y_train)
       # 预测
       y_pred = X_test.dot(w)
       return y_pred
   ```

```python
# 假设我们已经有了用户行为数据，以下是一个简化的数据集

    # ...更多用户行为数据
])

# 使用K-means算法进行聚类
kmeans = KMeans(n_clusters=3, random_state=0).fit(user_behavior_data)

# 输出聚类结果
print("Cluster centers:", kmeans.cluster_centers_)
print("Cluster labels:", kmeans.labels_)

# 根据聚类结果为每个用户打上标签
user_clusters = kmeans.predict(user_behavior_data)
print("User clusters:", user_clusters)
```
