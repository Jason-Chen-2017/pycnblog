    # 根据历史数据计算当前红绿灯变换周期

    green_light_duration = calculate_green_light_duration(history_data)

    print("红灯持续时间：{}秒，绿灯持续时间：{}秒".format(red_light_duration, green_light_duration))

    return max(0, 60 - history_data["green_time"])
