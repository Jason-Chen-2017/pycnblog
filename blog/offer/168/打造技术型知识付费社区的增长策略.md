class CollaborativeFilteringRecommender:
    def __init__(self, user_rated_items, content_tags):
        self.user_rated_items = user_rated_items
        self.content_tags = content_tags
        self.user_similarity_matrix = None
        self.user_item_matrix = None

    def build_user_similarity_matrix(self):
        user_item_matrix = self.build_user_item_matrix()
        self.user_similarity_matrix = cosine_similarity(user_item_matrix)

    def build_user_item_matrix(self):
        user_item_matrix = []
        for user in self.user_rated_items:
            user_ratings = [0] * len(self.content_tags)
            for rated_item in user['rated_items']:
                index = self.content_tags.index(rated_item)
                user_ratings[index] = 1
            user_item_matrix.append(user_ratings)
        return user_item_matrix

    def predict_rating(self, user_index, item_index):
        if self.user_similarity_matrix is None:
            self.build_user_similarity_matrix()
        similarity = self.user_similarity_matrix[user_index][item_index]
        return similarity * self.user_item_matrix[user_index][item_index]

    def recommend(self, user_index, content_pool, k=5):
        predicted_ratings = []
        for item_index, content in enumerate(content_pool):
            predicted_rating = self.predict_rating(user_index, item_index)
            predicted_ratings.append((predicted_rating, content))
        predicted_ratings.sort(reverse=True)
        return [content for _, content in predicted_ratings[:k]]
