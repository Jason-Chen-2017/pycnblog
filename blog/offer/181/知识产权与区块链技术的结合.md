```python
# 基于Python的简化版版权保护系统

import json
from hashlib import sha256

# 区块类
class Block:
    def __init__(self, index, transactions, timestamp, previous_hash):
        self.index = index
        self.transactions = transactions
        self.timestamp = timestamp
        self.previous_hash = previous_hash
        self.hash = self.compute_hash()

    # 计算块的哈希值
    def compute_hash(self):
        block_string = json.dumps(self.__dict__, sort_keys=True)
        return sha256(block_string.encode()).hexdigest()

# 区块链类
class Blockchain:
    def __init__(self):
        self.unconfirmed_transactions = []
        self.chain = []
        self.create_genesis_block()

    # 创建创世块
    def create_genesis_block(self):
        genesis_block = Block(0, [], "2019-01-01", "0")
        genesis_block.hash = genesis_block.compute_hash()
        self.chain.append(genesis_block)

    # 添加新块到链中
    def add_new_block(self, block):
        if self.is_chain_valid():
            block.previous_hash = self.chain[-1].hash
            block.hash = block.compute_hash()
            self.chain.append(block)
            self.unconfirmed_transactions = []
        else:
            print("链验证失败，不能添加新块。")

    # 检查链是否有效
    def is_chain_valid(self):
        for i in range(1, len(self.chain)):
            current = self.chain[i]
            previous = self.chain[i - 1]
            if current.hash != current.compute_hash():
                return False
            if current.previous_hash != previous.hash:
                return False
        return True

    # 添加新交易到未确认交易池
    def add_new_transaction(self, transaction):
        self.unconfirmed_transactions.append(transaction)

    # 矿工挖矿
    def mine(self):
        if not self.unconfirmed_transactions:
            return False
        last_block = self.chain[-1]
        new_block = Block(len(self.chain), self.unconfirmed_transactions, "2019-01-02", last_block.hash)
        new_block.hash = new_block.compute_hash()
        self.add_new_block(new_block)
        return new_block.index

# 创建区块链实例
blockchain = Blockchain()

# 添加新交易
blockchain.add_new_transaction("版权登记：作品A，作者：张三")
blockchain.add_new_transaction("版权交易：作品A，买家：李四")

# 挖矿
print("矿工开始挖矿...")
blockchain.mine()

# 打印区块链
print("当前区块链：")
for block in blockchain.chain:
    print(json.dumps(block.__dict__, indent=4))
```

* **Blockchain 类：** 定义了一个区块链，包括未确认交易列表、链本身、创建创世块、添加新块、验证链有效性、添加新交易和挖矿等功能。
* **主程序：** 创建了一个区块链实例，添加了一些交易，然后进行挖矿，最后打印了区块链。

```solidity
// 合约：IPRights.sol
contract IPRights {
    // 结构体：版权信息
    struct Copyright {
        bytes32 id;
        string title;
        string author;
        uint timestamp;
        address owner;
    }

    // 字段：版权信息映射
    mapping(bytes32 => Copyright) public copyrights;

    // 字段：版权交易记录映射
    mapping(bytes32 => mapping(address => uint)) public transactions;

    // 事件：版权登记
    event RegisteredCopyright(bytes32 id, string title, string author, address owner);

    // 事件：版权交易
    event TransferredCopyright(bytes32 id, address from, address to);

    // 函数：登记版权
    function registerCopyright(bytes32 id, string memory title, string memory author) public {
        require(copyrights[id].id == 0, "版权已存在");
        copyrights[id] = Copyright(id, title, author, block.timestamp, msg.sender);
        emit RegisteredCopyright(id, title, author, msg.sender);
    }

    // 函数：版权交易
    function transferCopyright(bytes32 id, address to) public {
        require(copyrights[id].id != 0, "版权不存在");
        require(msg.sender == copyrights[id].owner, "非版权所有者");
        transactions[id][msg.sender] += 1;
        copyrights[id].owner = to;
        emit TransferredCopyright(id, msg.sender, to);
    }

    // 函数：获取版权信息
    function getCopyright(bytes32 id) public view returns (string memory, string memory, uint, address) {
        require(copyrights[id].id != 0, "版权不存在");
        return (copyrights[id].title, copyrights[id].author, copyrights[id].timestamp, copyrights[id].owner);
    }

    // 函数：获取交易记录
    function getTransactionHistory(bytes32 id) public view returns (address[] memory) {
        require(copyrights[id].id != 0, "版权不存在");
        address[] memory history = new address[](transactions[id][msg.sender]);
        uint count = 0;
        for (address owner : transactions[id]) {
            if (owner != address(0)) {
                history[count] = owner;
                count++;
            }
        }
        return history;
    }
}
```

   - **transferCopyright：** 用于版权交易，要求调用者必须是版权所有者。
