package main

import (
    "fmt"
    "regexp"
)

func extractData(input string) (nums []int, letters []string) {
    // 提取数字
    numsRegex := regexp.MustCompile(`\d+`)
    numsStr := numsRegex.FindAllString(input, -1)
    for _, numStr := range numsStr {
        num, _ := strconv.Atoi(numStr)
        nums = append(nums, num)
    }

    // 提取字母
    lettersRegex := regexp.MustCompile(`[a-zA-Z]+`)
    lettersStr := lettersRegex.FindAllString(input, -1)
    for _, letterStr := range lettersStr {
        letters = append(letters, letterStr)
    }

    return
}

func main() {
    input := "123abc456"
    nums, letters := extractData(input)
    fmt.Println("Numbers:", nums)
    fmt.Println("Letters:", letters)
}

**题目：** 请解释 Go 语言并发编程的基本概念，并编写一个示例程序，展示如何使用 Goroutines 和 Channels 来优化信息处理效率。

**答案：** Go 语言并发编程的基本概念包括 Goroutines、Channels 和 WaitGroups。

1. **Goroutines**：Go 语言的轻量级线程，由 Go 运行时系统管理。每个 Goroutine 独立运行，互不干扰。
2. **Channels**：用于在 Goroutines 之间传递数据的通道。通道具有类型，可以是带缓冲的或不带缓冲的。
3. **WaitGroups**：用于等待多个 Goroutine 结束的同步工具。

**示例程序：**

```go
package main

import (
    "fmt"
    "time"
)

// 处理信息的方法
func processInfo(info string, ch chan<- string) {
    // 模拟信息处理时间
    time.Sleep(2 * time.Second)
    // 处理后的信息
    processedInfo := "Processed: " + info
    // 将处理后的信息发送到通道
    ch <- processedInfo
}

func main() {
    // 创建通道
    ch := make(chan string)

    // 启动 3 个 Goroutines，分别处理信息
    go processInfo("info1", ch)
    go processInfo("info2", ch)
    go processInfo("info3", ch)

    // 使用 for 循环接收处理后的信息
    for i := 0; i < 3; i++ {
        // 从通道中接收信息
        processedInfo := <-ch
        // 输出处理后的信息
        fmt.Println(processedInfo)
    }

    // 使用 WaitGroup 等待所有 Goroutine 结束
    var wg sync.WaitGroup
    wg.Add(3)
    for i := 0; i < 3; i++ {
        go func() {
            processInfo("info" + strconv.Itoa(i), ch)
            wg.Done()
        }()
    }
    wg.Wait()

    // 关闭通道
    close(ch)
}
```

**答案解析：**

1. 在 `processInfo` 方法中，我们模拟了信息处理所需的时间，并将处理后的信息发送到通道 `ch`。
2. 在 `main` 方法中，我们启动了 3 个 Goroutines，分别调用 `processInfo` 方法处理信息。每个 Goroutine 都将处理后的信息发送到通道 `ch`。
3. 我们使用一个 `for` 循环从通道 `ch` 中接收处理后的信息，并输出。
4. 使用 `WaitGroup` 等待所有 Goroutine 结束。

通过使用并发编程，我们可以优化信息处理效率。多个 Goroutines 同时处理信息，降低了单个 Goroutine 的处理时间，从而提高了整体处理效率。同时，通道 `ch` 保证了数据在 Goroutines 之间的同步传输，避免了数据竞争和死锁等问题。

### 7. 如何使用缓存技术提高信息检索速度？

**题目：** 请解释缓存技术的基本概念，并列举两种常见的缓存策略。

**答案：** 缓存技术是一种存储技术，用于减少对原始数据的访问次数，提高数据检索速度。缓存技术的基本概念包括：

1. **缓存**：将经常访问的数据存储在高速缓存中，以便快速读取。
2. **缓存失效**：缓存中的数据在一定时间后失效，需要重新从原始数据源读取。

**常见的缓存策略：**

1. **时间戳缓存**：根据缓存数据的有效时间来管理缓存，超过有效时间的数据将自动失效。
2. **写回缓存**：在缓存数据发生修改时，将修改后的数据同步回原始数据源，以保证数据一致性。

**示例代码：**

**时间戳缓存**：

```go
package main

import (
    "fmt"
    "time"
)

type CacheEntry struct {
    Value   interface{}
    Expiry  time.Time
}

type Cache struct {
    entries map[string]CacheEntry
}

func NewCache() *Cache {
    return &Cache{
        entries: make(map[string]CacheEntry),
    }
}

func (c *Cache) Set(key string, value interface{}, expiry time.Duration) {
    c.entries[key] = CacheEntry{
        Value:   value,
        Expiry:  time.Now().Add(expiry),
    }
}

func (c *Cache) Get(key string) (interface{}, bool) {
    entry, exists := c.entries[key]
    if !exists || time.Now().After(entry.Expiry) {
        return nil, false
    }
    return entry.Value, true
}

func main() {
    cache := NewCache()
    cache.Set("user1", "John Doe", 10*time.Minute)

    value, exists := cache.Get("user1")
    if exists {
        fmt.Println("Cache hit:", value)
    } else {
        fmt.Println("Cache miss")
    }

    time.Sleep(11 * time.Minute)
    value, exists = cache.Get("user1")
    if exists {
        fmt.Println("Cache hit:", value)
    } else {
        fmt.Println("Cache miss")
    }
}
```

**写回缓存**：

```go
package main

import (
    "fmt"
    "time"
)

type CacheEntry struct {
    Value   interface{}
    Updated time.Time
}

type Cache struct {
    entries   map[string]CacheEntry
    dataStore func(key string, value interface{}) error
}

func NewCache(dataStore func(key string, value interface{}) error) *Cache {
    return &Cache{
        entries:   make(map[string]CacheEntry),
        dataStore: dataStore,
    }
}

func (c *Cache) Set(key string, value interface{}) error {
    c.entries[key] = CacheEntry{
        Value:   value,
        Updated: time.Now(),
    }
    return c.dataStore(key, value)
}

func (c *Cache) Get(key string) (interface{}, error) {
    entry, exists := c.entries[key]
    if !exists {
        return nil, fmt.Errorf("key not found")
    }
    return entry.Value, nil
}

func (c *Cache) Update(key string, value interface{}) error {
    c.entries[key] = CacheEntry{
        Value:   value,
        Updated: time.Now(),
    }
    return c.dataStore(key, value)
}

func main() {
    // 模拟数据存储函数
    storeData := func(key string, value interface{}) error {
        fmt.Println("Storing data:", key, value)
        return nil
    }

    cache := NewCache(storeData)
    err := cache.Set("user1", "John Doe")
    if err != nil {
        panic(err)
    }

    value, err := cache.Get("user1")
    if err != nil {
        panic(err)
    }
    fmt.Println("Retrieved data:", value)

    // 更新缓存中的数据
    err = cache.Update("user1", "Jane Doe")
    if err != nil {
        panic(err)
    }

    // 从缓存中获取更新后的数据
    value, err = cache.Get("user1")
    if err != nil {
        panic(err)
    }
    fmt.Println("Updated data:", value)
}
```

通过使用缓存技术，我们可以显著提高信息检索速度，减少对原始数据源的访问压力。同时，合理的缓存策略有助于保证缓存数据的准确性，避免数据不一致的问题。

**答案：** 信息安全是指保护信息资产免受各种威胁和攻击的措施，确保信息的完整性、保密性和可用性。信息安全的基本概念包括：

2. **身份认证**：验证用户的身份，确保只有授权用户才能访问系统。
3. **访问控制**：限制用户对系统资源的访问权限，防止未授权访问。

**三种常见的信息安全技术：**

1. **对称加密**：加密和解密使用相同的密钥，如 AES。
2. **非对称加密**：加密和解密使用不同的密钥，如 RSA。
3. **多因素认证**：结合多种认证方式，如密码、短信验证码、指纹等。

**示例代码：**

**对称加密**：

```python
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad, unpad

# 加密
key = b'mysecretkey12345'
cipher = AES.new(key, AES.MODE_CBC)
pt = b'plaintext message'
ct = cipher.encrypt(pad(pt, AES.block_size))
iv = cipher.iv
print("Cipher text:", ct)
print("IV:", iv)

# 解密
cipher = AES.new(key, AES.MODE_CBC, iv)
pt = unpad(cipher.decrypt(ct), AES.block_size)
print("Decrypted text:", pt)
```

**非对称加密**：

```python
from Crypto.PublicKey import RSA
from Crypto.Cipher import PKCS1_OAEP

private_key = key.export_key()
public_key = key.publickey().export_key()

# 加密
cipher = PKCS1_OAEP.new(RSA.import_key(public_key))
ct = cipher.encrypt(b'plaintext message')

# 解密
cipher = PKCS1_OAEP.new(RSA.import_key(private_key))
pt = cipher.decrypt(ct)
print("Decrypted text:", pt)
```

# 创建 MapReduce 任务
class WordCount(mrjob.MRJob):
    def mapper(self, _, line):

    def reducer(self, word, counts):
