**参考代码：**

```go
package main

import (
    "fmt"
    "sync"
)

var (
    cpuUsage int
    mu       sync.Mutex
)

func allocateResource() {
    mu.Lock()
    defer mu.Unlock()
    cpuUsage += 10
    fmt.Printf("Allocated resource: CPU usage %d\n", cpuUsage)
}

func deallocateResource() {
    mu.Lock()
    defer mu.Unlock()
    cpuUsage -= 10
    fmt.Printf("Deallocated resource: CPU usage %d\n", cpuUsage)
}

func main() {
    go allocateResource()
    time.Sleep(time.Second)
    deallocateResource()
}
```

func sendDataToCloud(url string, data Data) error {
    payloadBytes, err := json.Marshal(data)

        return err
    }

    resp, err := http.Post(url, "application/json", bytes.NewBuffer(payloadBytes))

    }

    }

    return nil
}
