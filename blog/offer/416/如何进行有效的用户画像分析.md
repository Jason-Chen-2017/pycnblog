4. 将用户分配到距离最近的聚类中心。
5. 更新聚类中心。
6. 重复步骤3至5，直到聚类中心不再变化。
7. 将聚类结果保存到文件中。

**参考代码（Python）：**

```python
import numpy as np
import json

# 用户特征数据结构
class UserFeature:
    def __init__(self, user_id, features):
        self.user_id = user_id
        self.features = np.array(features)

# K-均值算法
def kmeans聚类(data, K, max_iter=100):
    n_samples, n_features = data.shape
    centroids = data[np.random.choice(n_samples, K, replace=False)]

    for _ in range(max_iter):
        # 计算用户与聚类中心的距离
        distances = np.linalg.norm(data - centroids, axis=1)

        # 将用户分配到最近的聚类中心
        labels = np.argmin(distances, axis=1)

        # 更新聚类中心
        new_centroids = np.array([data[labels == k].mean(axis=0) for k in range(K)])

        # 判断聚类中心是否发生变化
        if np.all(centroids == new_centroids):
            break

        centroids = new_centroids

    return labels, centroids

    labels, centroids = kmeans聚类(data, K)
```

```python
import pandas as pd
from mlxtend.frequent_patterns import apriori
from mlxtend.frequent_patterns import association_rules

# 用户行为数据结构
class UserBehavior:
    def __init__(self, user_id, behaviors):
        self.user_id = user_id
        self.behaviors = behaviors

# 挖掘关联规则
def挖掘关联规则(data, min_support=0.1, min_confidence=0.5):
    # 构建事务集
    transactions = [behavior for user in data for behavior in user.behaviors]

    # 挖掘频繁项集
    frequent_itemsets = apriori(transactions, min_support=min_support, use_colnames=True)

    # 构建关联规则
    rules = association_rules(frequent_itemsets, metric="confidence", min_threshold=min_confidence)
```
