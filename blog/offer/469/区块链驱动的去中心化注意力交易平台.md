#### 1. 创建区块链节点

**题目：** 编写一个简单的区块链节点创建脚本，实现节点的基本功能。

**答案：**
```python
import hashlib
import json
from time import time

class BlockChainNode:
    def __init__(self):
        self.chain = []
        self.current_transactions = []
        self.create_block(previous_hash='1', proof=100)

    def create_block(self, proof, previous_hash):
        block = {
            'index': len(self.chain) + 1,
            'timestamp': time(),
            'transactions': self.current_transactions,
            'proof': proof,
            'previous_hash': previous_hash,
        }
        self.current_transactions = []
        self.chain.append(block)
        return block

    def get_latest_block(self):
        return self.chain[-1]

    def add_transaction(self, sender, recipient, amount):
        self.current_transactions.append({'sender': sender, 'recipient': recipient, 'amount': amount})

    @staticmethod
    def hash(block):
        block_string = json.dumps(block, sort_keys=True)
        return hashlib.sha256(block_string.encode()).hexdigest()

    @staticmethod
    def proof_of_work(last_proof):
        proof = 0
        while not BlockChainNode.valid_proof(last_proof, proof):
            proof += 1
        return proof

    @staticmethod
    def valid_proof(last_proof, proof):
        guess = f'{last_proof}{proof}'.encode()
        guess_hash = hashlib.sha256(guess).hexdigest()
        return guess_hash[:4] == "0000"

    def resolve_conflicts(self, other_chain):
        longest_chain = self.chain
        if len(other_chain) > len(longest_chain):
            longest_chain = other_chain
        return longest_chain

if __name__ == '__main__':
    node = BlockChainNode()
    node.add_transaction("Alice", "Bob", 10)
    node.add_transaction("Bob", "Charlie", 20)
    print(node.chain)
    node2 = BlockChainNode()
    node2.chain = node.resolve_conflicts(node2.chain)
    print(node2.chain)
```

**题目：** 编写一个智能合约，实现用户通过支付注意力货币来获取内容创作者提供的注意力资源。

**答案：**
```solidity
pragma solidity ^0.8.0;

contract AttentionResourcePlatform {
    mapping(address => uint256) public balances;
    mapping(address => mapping(address => uint256)) public allocations;

    event Transaction(address sender, address recipient, uint256 amount);
    event Allocation(address owner, address recipient, uint256 amount);

    function deposit() public payable {
        balances[msg.sender()] += msg.value;
    }

    function allocateAttention(address recipient, uint256 amount) public {
        require(balances[msg.sender()] >= amount, "Insufficient balance");
        balances[msg.sender()] -= amount;
        allocations[msg.sender()][recipient] += amount;
        emit Allocation(msg.sender(), recipient, amount);
    }

    function transferAttention(address recipient, uint256 amount) public {
        require(allocations[msg.sender()][recipient] >= amount, "Insufficient allocated attention");
        allocations[msg.sender()][recipient] -= amount;
        balances[recipient] += amount;
        emit Transaction(msg.sender(), recipient, amount);
    }

    function getBalance(address account) public view returns (uint256) {
        return balances[account];
    }

    function getAllocation(address owner, address recipient) public view returns (uint256) {
        return allocations[owner][recipient];
    }
}

```

**解析：** 该智能合约实现了用户（内容创作者）通过支付以太币（注意力货币）来获取注意力资源，并能够将注意力资源分配给其他用户（购买者）。合约包括存款、分配注意力资源、转移注意力资源以及获取余额和分配情况的函数。事件用于记录交易和分配行为，便于外部监控和审计。
