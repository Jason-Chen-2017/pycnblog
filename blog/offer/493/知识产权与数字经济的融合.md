pragma solidity ^0.8.0;

contract CopyrightTrade {
    struct Trade {
        address seller;
        address buyer;
        string contentHash;
        uint256 price;
        bool isCompleted;
    }

    mapping(string => Trade) public trades;

    function registerCopyright(string memory contentHash) public {
        trades[contentHash].seller = msg.sender;
        trades[contentHash].contentHash = contentHash;
    }

    function makeTrade(string memory contentHash, uint256 price) public payable {
        require(trades[contentHash].seller != address(0), "No copyright registered");
        require(msg.value == price, "Incorrect payment amount");
        trades[contentHash].buyer = msg.sender;
        trades[contentHash].price = price;
        trades[contentHash].isCompleted = true;
        trades[contentHash].seller.transfer(price);
    }

    function trackCopyright(string memory contentHash) public view returns (address, address, uint256, bool) {
        return (trades[contentHash].seller, trades[contentHash].buyer, trades[contentHash].price, trades[contentHash].isCompleted);
    }
}
