    bestOption := ""
    highestScore := -1
    for _, option := range options {
        score := 0
        for k, v := range knowledgeBase {
            if strings.Contains(option, k) {
                score += len(v)
            }
        }
        if score > highestScore {

            bestOption = option
        }
    }
    return bestOption
}
