```go
func generateMindMap(input []string) *Node {
    var root *Node
    var current *Node
    
    for _, item := range input {
        parts := strings.Split(item, "->")
        if len(parts) == 1 {
            // 新的节点
            current = &Node{Title: parts[0]}
            if root == nil {
                root = current
            }
        } else {
            // 子节点
            child := &Node{Title: parts[1]}
            current.Children = append(current.Children, child)
            current = child
        }
    }
    
    return root
}
```

#### 1. 阿里巴巴：实现一个思维导图生成算法

**题目描述：** 请设计一个思维导图生成算法，给定一组节点和它们之间的父子关系，生成一个树状结构的思维导图。

**输入：**
- 节点列表：`nodes = [{"id": 1, "title": "根节点"}, {"id": 2, "title": "子节点1"}, {"id": 3, "title": "子节点2"}, {"id": 4, "title": "子节点1-1"}]`
- 父子关系列表：`parentChild = [[1, 2], [1, 3], [2, 4]]`

**输出：**
- 一个树状结构的思维导图，其中根节点为 `1`，子节点为 `2`、`3`，`2` 的子节点为 `4`。

**答案解析：**

```go
package main

import (
    "encoding/json"
    "fmt"
)

type Node struct {
    Id   int    `json:"id"`
    Title string `json:"title"`
    Children []*Node `json:"children"`
}

func buildMindMap(nodes map[int]*Node, parentChild [][]int) {
    for _, pc := range parentChild {
        parentId, childId := pc[0], pc[1]
        if node, ok := nodes[parentId]; ok {
            node.Children = append(node.Children, &Node{Id: childId, Title: "子节点"})
        }
    }
}

func main() {
    nodes := []map[string]int{
        {"id": 1, "title": "根节点"},
        {"id": 2, "title": "子节点1"},
        {"id": 3, "title": "子节点2"},
        {"id": 4, "title": "子节点1-1"},
    }
    parentChild := [][]int{{1, 2}, {1, 3}, {2, 4}}

    nodeMap := make(map[int]*Node)
    for _, n := range nodes {
        nodeMap[n["id"]] = &Node{Id: n["id"], Title: n["title"]}
    }

    buildMindMap(nodeMap, parentChild)

    root := nodeMap[1]
    fmt.Println("思维导图：")
    printMindMap(root)
}

func printMindMap(node *Node) {
    if node == nil {
        return
    }
    fmt.Println(node.Title)
    for _, child := range node.Children {
        printMindMap(child)
    }
}
```

**源代码实例：** 上述代码实现了根据节点列表和父子关系列表生成树状结构的思维导图，并通过递归打印出树状结构。

#### 2. 百度：实现一个思维导图的遍历算法

**题目描述：** 请实现一个函数 `breadthFirstSearch`，使用广度优先搜索遍历一个给定的思维导图，并返回遍历的结果。

**输入：**
- 思维导图的根节点：`root`

**输出：**
- 遍历结果：`result`，格式为 `[根节点，子节点1，子节点2，子节点1-1]`

**答案解析：**

```go
func breadthFirstSearch(root *Node) []string {
    if root == nil {
        return []string{}
    }
    
    result := []string{root.Title}
    queue := []*Node{root}
    
    for len(queue) > 0 {
        node := queue[0]
        queue = queue[1:]
        
        for _, child := range node.Children {
            result = append(result, child.Title)
            queue = append(queue, child)
        }
    }
    
    return result
}
```

**源代码实例：** 上述代码实现了广度优先搜索遍历思维导图的算法，返回遍历的结果。

- 思维导图的根节点：`root`

```go
func depthFirstSearch(root *Node) []string {
    if root == nil {
        return []string{}
    }
    
    result := []string{root.Title}
    for _, child := range root.Children {
        result = append(result, depthFirstSearch(child)...)
    }
    
    return result
}
```

#### 4. 字节跳动：实现一个思维导图的路径查找算法

**题目描述：** 请实现一个函数 `findPath`，给定一个思维导图的根节点和一个目标节点，找出从根节点到目标节点的路径。

**输入：**
- 思维导图的根节点：`root`
- 目标节点：`target`

**输出：**
- 路径：`path`，格式为 `[根节点，子节点1，子节点1-1，目标节点]`

**答案解析：**

```go
func findPath(root *Node, target string) []string {
    if root == nil {
        return nil
    }
    
    if root.Title == target {
        return []string{root.Title}
    }
    
    for _, child := range root.Children {
        path := findPath(child, target)
        if path != nil {
            path = append([]string{root.Title}, path...)
            return path
        }
    }
    
    return nil
}
```

**源代码实例：** 上述代码实现了根据目标节点查找从根节点到目标节点的路径的算法。
