```python
import math

def morse_derivatives(n, x):
    # 一阶导数：f'(x) = -1/2 * (x1/x1, x2/x2, ..., xn/xn)
    # 二阶导数：f''(x) = 1/4 * (1/x1^2, 1/x2^2, ..., 1/xn^2)
```

```python
import numpy as np

def morse_hessian(n, x):
    # 海森矩阵是二阶导数矩阵，即 H = [df''/dx_i dx_j]
    # 对于f(x) = 1 - √(x1^2 + x2^2 + ... + xn^2)，我们有：
    # H = [1/(2x1^2), ..., 1/(2x1^2), 1/(2x2^2), ..., 1/(2x2^2), ..., 1/(2xn^2), ..., 1/(2xn^2)]
    hessian = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            hessian[i][j] = 1 / (2 * x[i] ** 2) if i == j else 0
    return hessian

```

**输入：** 一个正整数n和一个n维列表x

**答案与解析：**
```python
import numpy as np

def is_morse_local_max(n, x):
    # 判断该点是否为局部极大值点，需要满足以下条件：
    # 1. 一阶导数小于0，即 df/dx_i < 0 对于所有的 i
    # 2. 二阶导数矩阵的特征值均大于0
    gradient = np.array([(-1/2 * xi) for xi in x])
    hessian = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            hessian[i][j] = 1 / (2 * x[i] ** 2) if i == j else 0
    eigenvalues, _ = np.linalg.eig(hessian)
    return np.all(gradient < 0) and np.all(eigenvalues > 0)

print(is_morse_local_max(2, [0.0, 0.0]))  # 输出应为 True
```
**代码说明：** 我们使用Python和NumPy编写了一个函数，首先计算了一阶导数，然后计算了二阶导数的特征值。如果一阶导数小于0且二阶导数的特征值均大于0，则该点为局部极大值点。

```python
import numpy as np
```

```python
def is_morse_saddle_point(n, x):
```
```python
    # 判断该点是否为鞍点，需要满足以下条件：
    # 1. 一阶导数不等于0，即 df/dx_i ≠ 0 对于至少一个 i
```

```python
import numpy as np

def is_morse_local_extrema(n, x):
    gradient = [-1/(2 * np.sqrt(np.sum(np.square(x)))) for _ in range(n)]
    hessian = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j:
                hessian[i][j] = 1 / (4 * x[i] ** 4)
            else:
                hessian[i][j] = -1 / (4 * x[i] ** 2 * x[j] ** 2)
    eigenvalues, _ = np.linalg.eig(hessian)
    return np.all(gradient == 0) and (np.all(eigenvalues > 0) or np.all(eigenvalues < 0))
```
