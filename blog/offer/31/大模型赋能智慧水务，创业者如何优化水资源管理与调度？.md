### **1. 利用 K-Means 算法对水资源数据进行聚类分析**

**题目：** 利用 K-Means 算法对一组水资源数据进行聚类分析，输出每个簇的中心点和簇内数据点的数量。

**答案：** 
```python
from sklearn.cluster import KMeans
import numpy as np

def kmeans_clustering(data, k):
    kmeans = KMeans(n_clusters=k, random_state=0)
    kmeans.fit(data)
    
    centroids = kmeans.cluster_centers_
    labels = kmeans.labels_
    
    cluster_counts = {i: 0 for i in range(k)}
    for label in labels:
        cluster_counts[label] += 1
    
    return centroids, cluster_counts

# 示例数据
data = np.array([[1, 2], [1, 4], [1, 0], [4, 2], [4, 4], [4, 0]])

# 聚类分析
centroids, cluster_counts = kmeans_clustering(data, 2)
print("Cluster Centroids:", centroids)
print("Cluster Counts:", cluster_counts)
```

### **2. 利用决策树对水资源供需数据进行分类**

**题目：** 利用决策树算法对一组水资源供需数据进行分类，输出每个分类的准确率。

**答案：** 
```python
from sklearn.tree import DecisionTreeClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score

def decision_tree_classification(data, labels):
    X_train, X_test, y_train, y_test = train_test_split(data, labels, test_size=0.3, random_state=0)
    classifier = DecisionTreeClassifier()
    classifier.fit(X_train, y_train)
    y_pred = classifier.predict(X_test)
    
    accuracy = accuracy_score(y_test, y_pred)
    return accuracy

# 示例数据
data = np.array([[1, 2], [2, 3], [3, 1], [4, 2], [5, 3]])
labels = np.array([0, 0, 0, 1, 1])

# 分类分析
accuracy = decision_tree_classification(data, labels)
print("Accuracy:", accuracy)
```

## **三、算法编程题库**

### **1. 利用深度学习模型预测水资源供需**

**题目：** 使用深度学习模型（如 LSTM）预测某地区未来 24 小时内的水资源供需。

**答案：**
```python
import numpy as np
import pandas as pd
from sklearn.preprocessing import MinMaxScaler
from keras.models import Sequential
from keras.layers import LSTM, Dense

# 加载数据
data = pd.read_csv('water_resource_data.csv')
X = data[['temperature', 'precipitation', 'water_usage']].values
y = data['water_supply'].values

# 数据预处理
scaler = MinMaxScaler(feature_range=(0, 1))
X_scaled = scaler.fit_transform(X)
y_scaled = scaler.fit_transform(y.reshape(-1, 1))

# 切分数据为训练集和测试集
X_train, y_train = [], []
for i in range(24, len(y_scaled)):
    X_train.append(X_scaled[i-24:i, :])
    y_train.append(y_scaled[i, :])
X_train, y_train = np.array(X_train), np.array(y_train)

X_train = np.reshape(X_train, (X_train.shape[0], X_train.shape[1], X_train.shape[2]))

# 建立模型
model = Sequential()
model.add(LSTM(units=50, return_sequences=True, input_shape=(24, 3)))
model.add(LSTM(units=50))
model.add(Dense(units=1))

model.compile(optimizer='adam', loss='mean_squared_error')
model.fit(X_train, y_train, epochs=100, batch_size=32)

# 预测
predicted_supply = model.predict(X_train[-1:])
predicted_supply = scaler.inverse_transform(predicted_supply)

print("Predicted Water Supply:", predicted_supply)
```

# 定义 GNN 模型
input_node = Input(shape=(None,))  # 输入为时间序列数据

# 训练模型
model.fit(X_train, y_train, epochs=10, batch_size=32)

predicted_consumption = model.predict(X_train[-1:])
