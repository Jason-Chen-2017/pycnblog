from transformers import BertTokenizer, BertModel
import torch

tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
model = BertModel.from_pretrained('bert-base-uncased')

def text_similarity(text1, text2):
    inputs1 = tokenizer.encode_plus(text1, return_tensors='pt', max_length=512, padding='max_length', truncation=True)
    inputs2 = tokenizer.encode_plus(text2, return_tensors='pt', max_length=512, padding='max_length', truncation=True)

    outputs1 = model(**inputs1)
    outputs2 = model(**inputs2)

    # 计算两个文本的隐藏状态的平均值
    hidden_state1 = outputs1.pooler_output.mean(dim=1)
    hidden_state2 = outputs2.pooler_output.mean(dim=1)

    # 计算两个隐藏状态的相似度
