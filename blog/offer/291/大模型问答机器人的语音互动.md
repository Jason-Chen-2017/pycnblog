```python
def str_match(s: str, p: str) -> int:
    n, m = len(s), len(p)
    j = 0
    for i in range(n):
        if j == m:
            break
        if p[j] == '*':
            j += 1
            if j == m:
                break
            while i + 1 < n and (s[i + 1] == p[j] or p[j] == '.'):
                i += 1
        elif s[i] == p[j] or p[j] == '.':
            i, j = i + 1, j + 1
        else:
            j = max(0, j - m + i - n)
    return i == n or j == m
```

```python
def edit_distance(s1: str, s2: str) -> int:
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(m + 1):
        for j in range(n + 1):
            if i == 0:
                dp[i][j] = j
            elif j == 0:
                dp[i][j] = i
            elif s1[i - 1] == s2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                dp[i][j] = min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]) + 1

    return dp[m][n]
```
