## 二、算法编程题库

### 1. 实现向量数据库中的相似度计算

**题目：** 请使用 Python 实现余弦相似度计算。

**答案：** 

```python
import numpy as np

def cosine_similarity(v1, v2):
    return np.dot(v1, v2) / (np.linalg.norm(v1) * np.linalg.norm(v2))

# 示例
vec1 = np.array([1, 2, 3])
vec2 = np.array([4, 5, 6])
similarity = cosine_similarity(vec1, vec2)
print("余弦相似度：", similarity)
```

### 2. 实现向量数据库的索引结构

**题目：** 请使用 Python 实现 K-D 树。

**答案：**

```python
import numpy as np
from heapq import heappush, heappop

class Node:
    def __init__(self, axis, point, left=None, right=None):
        self.axis = axis
        self.point = point
        self.left = left
        self.right = right

class KDTree:
    def __init__(self, points):
        self.root = self.build_tree(points, 0)

    def build_tree(self, points, depth):
        if not points:
            return None
        points.sort(key=lambda x: x[depth % len(x)])
        median = len(points) // 2
        return Node(depth, points[median], self.build_tree(points[:median], depth + 1), self.build_tree(points[median + 1:], depth + 1))

    def query(self, point, depth=0):
        node = self.root
        results = []
        while node:
            if np.linalg.norm(node.point - point) < 1e-9:
                results.append(node.point)
                return results
            if point[node.axis] < node.point[node.axis]:
                node = node.left
            else:
                node = node.right
        return results

# 示例
points = [[1, 2], [3, 4], [5, 6], [7, 8]]
tree = KDTree(points)
print("最近邻查询结果：", tree.query([2, 3]))
```
