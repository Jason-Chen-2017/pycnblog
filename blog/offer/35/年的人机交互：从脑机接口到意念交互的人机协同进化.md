import numpy as np
from scipy.signal import butter, filtfilt

def butter_bandpass_filter(data, lowcut, highcut, fs, order=4):
    nyq = 0.5 * fs
    low = lowcut / nyq
    high = highcut / nyq
    b, a = butter(order, [low, high], btype='band')
    filtered = filtfilt(b, a, data)
    return filtered

### 1. 脑机接口算法实现
**题目：** 实现一个简单的脑机接口算法，通过脑电信号预测用户的意图。

**答案：** 实现一个简单的脑机接口算法，通过频域分析提取特征，然后使用机器学习算法进行意图预测。

**解析：**
```python
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split

def extract_features(data):
    # 提取频域特征
    fft = np.fft.fft(data)
    freqs = np.fft.fftfreq(len(data))
    features = np.abs(fft[1:-1]) * freqs[1:-1]
    return features

def train_predictor(train_data, train_labels):
    # 训练随机森林分类器
    model = RandomForestClassifier()
    model.fit(train_data, train_labels)
    return model

def predict_intent(model, test_data):
    # 应用训练好的分类器进行意图预测
    return model.predict(test_data)

# 示例
train_data = np.random.rand(100, 1000)  # 假设训练数据
train_labels = np.random.randint(0, 2, size=(100, 1))  # 假设训练标签
test_data = np.random.rand(10, 1000)  # 假设测试数据

# 提取特征
train_features = extract_features(train_data)
test_features = extract_features(test_data)

# 训练预测器
predictor = train_predictor(train_features, train_labels)

# 预测意图
predictions = predict_intent(predictor, test_features)
```
