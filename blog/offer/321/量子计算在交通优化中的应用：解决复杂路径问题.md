from pyspo import QuantumParticleSwarm
from qiskit import QuantumCircuit
from qiskit.visualization import plot_histogram

# 初始化量子粒子群优化算法参数
n_bits = 3
n_qubits = n_bits + 1
n_particles = 10
max_iterations = 100
w = 0.5
c1 = 1.0
c2 = 1.0

# 定义适应度函数
def fitness(individual):
    qc = QuantumCircuit(n_qubits)
    qc.h([i for i in range(n_bits)])
    for i, particle in enumerate(individual):
        qc.cx(i, n_bits)
