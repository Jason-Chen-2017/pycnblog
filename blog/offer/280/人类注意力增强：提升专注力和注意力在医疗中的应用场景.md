### 5. 构建注意力模型

**题目：** 编写一个简单的注意力模型，用于关注医疗数据中的关键信息。

**答案：**

```python
import tensorflow as tf

class AttentionModel(tf.keras.Model):
    def __init__(self, embed_dim, hidden_dim):
        super(AttentionModel, self).__init__()
        self.W1 = tf.keras.layers.Dense(hidden_dim)
        self.W2 = tf.keras.layers.Dense(hidden_dim)
        self.V = tf.keras.layers.Dense(1)

    def call(self, inputs, training=False):
        # inputs 是一个包含医疗数据的序列，维度为 (batch_size, sequence_length)
        query = self.W1(inputs)
        keys = self.W2(inputs)
        
        # 计算注意力分数
        attention_scores = self.V(tf.matmul(tf.tanh(query + keys), tf.transpose(keys, [0, 2, 1])))
        
        # 应用softmax函数得到注意力权重
        attention_weights = tf.nn.softmax(attention_scores, axis=1)
        
        # 计算注意力输出
        context_vector = tf.reduce_sum(attention_weights * inputs, axis=1)
        
        return context_vector

# 示例
model = AttentionModel(embed_dim=64, hidden_dim=32)
print(model.summary())
```

### 7. 注意力模型优化

**题目：** 编写一个函数，用于优化注意力模型的性能。

**答案：**

```python
import tensorflow as tf

def optimize_model(model, x_train, y_train, x_val, y_val, epochs=10, learning_rate=0.001):
    optimizer = tf.keras.optimizers.Adam(learning_rate=learning_rate)
    loss_fn = tf.keras.losses.SparseCategoricalCrossentropy()

    for epoch in range(epochs):
        with tf.GradientTape() as tape:
            predictions = model(x_train)
            loss = loss_fn(y_train, predictions)

        grads = tape.gradient(loss, model.trainable_variables)
        optimizer.apply_gradients(zip(grads, model.trainable_variables))

        # 计算验证集上的损失
        val_predictions = model(x_val)
        val_loss = loss_fn(y_val, val_predictions)

        print(f"Epoch {epoch+1}, Loss: {loss.numpy()}, Val Loss: {val_loss.numpy()}")
```
