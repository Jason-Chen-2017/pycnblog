def calculate_emission(time, production, transportation):
    """
    计算虚拟时装的碳排放量
    :param time: 生产时间（天）
    :param production: 生产过程中的碳排放量（kg CO2）
    :param transportation: 运输过程中的碳排放量（kg CO2）
    :return: 碳排放量（kg CO2）
