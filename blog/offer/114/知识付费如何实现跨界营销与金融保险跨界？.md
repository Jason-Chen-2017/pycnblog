class MembershipSystem:
    def __init__(self):
        self.membership_levels = {'普通会员': 0, '铜牌会员': 100, '银牌会员': 200, '金牌会员': 500}
        self.members = {}

    def register_member(self, username):
        self.members[username] = {'level': '普通会员', 'points': 0}

    def upgrade_membership(self, username):
        member = self.members[username]
        if member['points'] >= self.membership_levels[member['level']]:
            member['level'] = next((level for level, points in self.membership_levels.items() if points > member['points']), member['level'])
            member['points'] = 0
            print(f"{username}升级为{member['level']}会员")

    def add_points(self, username, points):
        self.members[username]['points'] += points

class SearchEngine:
    def __init__(self, courses):
        self.courses = courses

    def search(self, query):
        results = []
        for course in self.courses:
            if query in course['title'] or query in course['description']:
                results.append(course)
        return results

    def sort_results(self, results, user_id=None):
        # 根据用户历史，个性化排序
        if user_id:
            # 模拟个性化排序逻辑
            results.sort(key=lambda x: x['rating'], reverse=True)
        else:
            # 按照课程热度排序
            results.sort(key=lambda x: x['views'], reverse=True)
        return results

    def suggest_search(self, query):
        # 提供实时搜索建议
        suggestions = []
        for course in self.courses:
            if query in course['title']:
                suggestions.append(course['title'])
        return suggestions

        self.min_purchase = min_purchase

            return total - self.discount
        return total

class PointSystem:
    def __init__(self):
        self.members = {}

    def add_points(self, user_id, points):
        if user_id not in self.members:
            self.members[user_id] = {'points': 0}
        self.members[user_id]['points'] += points

    def spend_points(self, user_id, points):
        if user_id in self.members and self.members[user_id]['points'] >= points:
            self.members[user_id]['points'] -= points
            print(f"{user_id}消耗{points}积分")
        else:
            print(f"{user_id}积分不足")

    def get_points(self, user_id):
        if user_id in self.members:
            return self.members[user_id]['points']
        return 0

# 示例
