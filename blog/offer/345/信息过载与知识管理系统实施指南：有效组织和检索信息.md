```python
import numpy as np

def kmeans(data, k, max_iters=100):
    centroids = data[np.random.choice(data.shape[0], k, replace=False)]
    for _ in range(max_iters):
        clusters = []
        for point in data:
            distances = np.linalg.norm(point - centroids, axis=1)
            closest_cluster = np.argmin(distances)
            clusters.append(closest_cluster)
        new_centroids = np.array([data[clusters.count(i)] for i in range(k)])
        if np.all(centroids == new_centroids):
            break
        centroids = new_centroids
    return centroids, clusters

data = np.array([[1, 1], [1, 2], [1, 3], [1, 4], [2, 2], [2, 3], [2, 4]])
centroids, clusters = kmeans(data, 2)
print("Centroids:", centroids)
print("Clusters:", clusters)
```

### 2. 信息过载导致的后果有哪些？

#### **面试题：** 请列举信息过载可能带来的几种后果。

**答案：**

信息过载可能导致的后果包括：

1. **注意力分散**：面对大量信息，个体难以集中注意力，导致工作效率下降。
2. **焦虑和压力**：信息过载可能导致个体产生焦虑感和压力，影响心理健康。
3. **疲劳**：长时间处理大量信息可能导致身体和心理疲劳，影响工作质量和生活质量。
4. **信息过时**：未能及时处理和筛选信息，可能导致信息过时，失去其原有的价值和意义。
5. **决策困难**：在信息过载的环境下，个体可能难以从大量信息中筛选出关键信息，导致决策困难。
6. **信息孤岛**：信息分散在多个来源和渠道中，难以整合和利用，形成信息孤岛，影响知识的共享和利用。

```python
from datetime import datetime

def calculate_obsolete_ratio(information_list, current_time):
    current_date = datetime.strptime(current_time, "%Y-%m-%d")
    expired_count = 0
    
    for info in information_list:
        info_date = datetime.strptime(info["expires"], "%Y-%m-%d")
        if info_date < current_date:
            expired_count += 1
    
    ratio = expired_count / len(information_list)
    return ratio
```

### 3. 如何评估信息价值？

#### **面试题：** 请简述评估信息价值的几个关键因素。

**答案：**

评估信息价值的关键因素包括：

1. **相关性**：信息与个人或组织目标、需求的相关性。高度相关的信息通常具有更高的价值。
2. **可靠性**：信息的来源和内容是否可信、可靠。权威机构、专家或知名媒体发布的信息通常具有较高的可靠性。
3. **时效性**：信息的时效性对其价值有很大影响。对于时效性要求高的领域，如新闻、股市等，及时的信息尤为重要。
4. **完整性**：信息是否全面、完整，能够为决策或问题解决提供充足的信息支持。
5. **可验证性**：信息是否可以通过其他来源或方法进行验证，以确保其准确性。
6. **实用性**：信息是否能够实际应用，解决具体问题或带来实际效益。

```python
def build_hierarchy_categories(topics):
    category_tree = {}
    for parent_topic, sub_topics in topics.items():
        category_tree[parent_topic] = {sub_topic: [] for sub_topic in sub_topics}
    return category_tree
```

**答案示例：**
```python
from collections import defaultdict

def collaborative_filtering(reviews, user_id):
    user_ratings = reviews[user_id]
    similarity_scores = defaultdict(float)
    for other_user, other_ratings in reviews.items():
        if other_user != user_id:
            common_ratings = set(user_ratings).intersection(set(other_ratings))
            if not common_ratings:
                continue
            sim = 1 / len(common_ratings)
            for item in common_ratings:
                sim *= (user_ratings[item] - other_ratings[item]) ** 2
            similarity_scores[other_user] = sim
    
    recommended_items = []
    for other_user, sim in similarity_scores.items():
        recommended_items.extend([(other_user, item, rating) for item, rating in reviews[other_user] if (other_user, item) not in user_ratings])
    
    recommended_items.sort(key=lambda x: x[2], reverse=True)
    return recommended_items[:5]

# 测试函数

    "user1": [{"item_id": 1, "rating": 5}, {"item_id": 2, "rating": 3}, {"item_id": 3, "rating": 4}],
```

```python
from collections import Counter

def recommend_tags(tags, user_tags):
    tag_counts = Counter(tags)
    user_tag_counts = Counter(user_tags)
    similar_tags = []
    
    for tag, count in tag_counts.items():
        if tag in user_tag_counts:

        if count >= user_tag_counts.most_common(1)[0][1] / 2:
            similar_tags.append(tag)
    
    return similar_tags[:3]  # 返回前三个相似标签
```

```python
from scipy.spatial.distance import cosine

def collaborative_filtering(reviews, user_id):
    user_ratings = reviews[user_id]
    user_ratings_matrix = np.array([user_ratings ratings for ratings in user_ratings.values()])
    similar_users = []
    
    for other_user, other_ratings in reviews.items():
        if other_user != user_id:
            other_ratings_matrix = np.array([other_ratings ratings for ratings in other_ratings.values()])
            similarity = 1 - cosine(user_ratings_matrix, other_ratings_matrix)
            similar_users.append((other_user, similarity))
    
    similar_users.sort(key=lambda x: x[1], reverse=True)
    recommended_items = []
    
    for user, similarity in similar_users[:5]:
        recommended_items.extend([(user, item_id, rating) for item_id, rating in reviews[user] if (user, item_id) not in user_ratings])
    
    recommended_items.sort(key=lambda x: x[2], reverse=True)
    return recommended_items[:5]
```

**输入示例：**

```python
    {"id": 1, "title": "编程语言", "content": "Python是一种编程语言。"},

```

**输出示例：**
```python

    {"id": 1, "title": "编程语言", "content": "Python是一种编程语言。", "status": "通过"},
```

**答案示例：**
```python
def knowledge_audit(knowledge, rules):
    audit_results = []
    for item in knowledge:
        result = {"id": item["id"], "title": item["title"], "content": item["content"], "status": "通过"}
        for rule in rules:
            if rule["condition"] == ">":
                if len(item[rule["field"]]) <= rule["value"]:
                    result["status"] = "未通过"
                    break
            elif rule["condition"] == "contains":
                if rule["value"] not in item[rule["field"]]:
                    result["status"] = "未通过"
                    break
            else:
                raise ValueError("未知审核规则")
        audit_results.append(result)
    return audit_results
```
