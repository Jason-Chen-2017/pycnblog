import aes "crypto/aes"
import crypto "crypto/cipher"

func EncryptData(data string, key string) string {
  block, _ := aes.NewCipher([]byte(key))
  ciphertext := make([]byte, aes.BlockSize+len(data))
  iv := ciphertext[:aes.BlockSize]
  crypto.Random(iv)
  cipher := cipher.NewCFBEncrypter(block, iv)
  cipher.XORKeyStream(ciphertext[aes.BlockSize:], []byte(data))
  return base64.StdEncoding.EncodeToString(ciphertext)
}

// 权限控制（使用JWT进行身份验证）
import jwt "github.com/dgrijalva/jwt-go"

func GenerateToken(username, password string) string {

  token.Claims["password"] = password
