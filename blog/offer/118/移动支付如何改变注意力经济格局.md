#### 问题 4：安全移动支付系统的实现

**解析：** 
为了实现一个安全的移动支付系统，需要关注数据的安全传输、存储以及交易的验证。

**源代码实例：**

```go
package main

import (
    "crypto/rand"
    "crypto/rsa"
    "crypto/sha256"
    "encoding/json"
    "encoding/pem"
)

// 生成RSA密钥对
func generateRSAKeyPair() (*rsa.PrivateKey, error) {
    privateKey, err := rsa.GenerateKey(rand.Reader, 2048)
    if err != nil {
        return nil, err
    }
    return privateKey, nil
}

// 签名
func signData(data []byte, privateKey *rsa.PrivateKey) ([]byte, error) {
    hasher := sha256.New()
    hasher.Write(data)
    hash := hasher.Sum(nil)
    return rsa.SignPKCS1v15(rand.Reader, privateKey, crypto.SHA256, hash)
}

// 验证签名
func verifySignature(data []byte, signature []byte, publicKey *rsa.PublicKey) (bool, error) {
    hasher := sha256.New()
    hasher.Write(data)
    hash := hasher.Sum(nil)
    return rsa.VerifyPKCS1v15(publicKey, crypto.SHA256, hash, signature)
}

// 支付请求结构体
type PaymentRequest struct {
    Amount    float64 `json:"amount"`
    Timestamp int64   `json:"timestamp"`
}

// 支付应答结构体
type PaymentResponse struct {
    Status    string `json:"status"`
    Signature string `json:"signature"`
}

func main() {
    // 生成RSA密钥对
    privateKey, err := generateRSAKeyPair()
    if err != nil {
        panic(err)
    }

    // 公钥
    publicKey := &privateKey.PublicKey

    // 创建支付请求
    req := PaymentRequest{
        Amount:    100.0,
        Timestamp: time.Now().Unix(),
    }
    reqJSON, err := json.Marshal(req)
    if err != nil {
        panic(err)
    }

    // 签名
    signature, err := signData(reqJSON, privateKey)
    if err != nil {
        panic(err)
    }

    // 构建支付应答
    resp := PaymentResponse{
        Status:    "SUCCESS",
        Signature: string(signature),
    }
    respJSON, err := json.Marshal(resp)
    if err != nil {
        panic(err)
    }

    // 验证签名
    verified, err := verifySignature(respJSON, signature, publicKey)
    if err != nil {
        panic(err)
    }

    if verified {
        fmt.Println("Payment verified successfully.")
    } else {
        fmt.Println("Payment verification failed.")
    }
}
```
