**1. 给定一个整数数组，找出所有出现次数大于一半的数字。**

```python
def find_majority_element(nums):
    count = 0
    candidate = None
    for num in nums:
        if count == 0:
            candidate = num
        count += (1 if num == candidate else -1)
    return candidate
```

**2. 设计一个算法，找出数组中第二大的元素。**

```python
def find_second_largest(nums):
    if len(nums) < 2:
        return None
    first_max, second_max = float('-inf'), float('-inf')
    for num in nums:
        if num > first_max:
            second_max = first_max
            first_max = num
        elif num > second_max and num != first_max:
            second_max = num
    return second_max
```

**3. 给定一个字符串，找出其中第一个只出现一次的字符。**

```python
def first_unique_char(s):
    char_count = {}
    for char in s:
        char_count[char] = char_count.get(char, 0) + 1
    for char in s:
        if char_count[char] == 1:
            return char
    return None
```

**5. 给定一个无序数组，找出其中的最长递增子序列。**

```python
def longest_increasing_subsequence(nums):
    if not nums:
        return []
    dp = [1] * len(nums)
    for i in range(1, len(nums)):
        for j in range(i):
            if nums[i] > nums[j]:
                dp[i] = max(dp[i], dp[j] + 1)
    max_len = max(dp)
    indices = []
    for i in range(len(dp)):
        if dp[i] == max_len:
            indices.append(i)
            max_len -= 1
    return indices
```

**14. 给定一个字符串，找出其中最长的回文子串。**

```python
def longest_palindromic_substring(s):
    start, max_len = 0, 1
    for i in range(len(s)):
        len1 = helper(s, i, i)
        len2 = helper(s, i, i + 1)
        max_len = max(max_len, len1, len2)
        start = i - ((max_len - 1) // 2)
    return s[start:start + max_len]

def helper(s, left, right):
    while left >= 0 and right < len(s) and s[left] == s[right]:
        left -= 1
        right += 1
    return right - left - 1
```

```python
def longest_common_prefix(s):
    if not s:
        return ""
    prefix = ""
    for i, char in enumerate(s[0]):
        for s in s[1:]:
            if i >= len(s) or s[i] != char:
                return prefix
        prefix += char
    return prefix
```

```python
        a = a ^ b
        b = carry << 1
```

**20. 给定一个整数数组，找出其中的最大连续子序列和。**

```python
def max_subarray_sum(nums):
    if not nums:
        return 0
    max_sum = nums[0]
    current_sum = nums[0]
    for num in nums[1:]:
        current_sum = max(num, current_sum + num)
        max_sum = max(max_sum, current_sum)
    return max_sum
```
