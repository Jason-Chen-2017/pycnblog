**解决方案：**

```python
import hashlib

class User:
    def __init__(self, username, password, email):
        self.username = username
        self.password = self.encrypt_password(password)
        self.email = email

    def encrypt_password(self, password):
        return hashlib.sha256(password.encode('utf-8')).hexdigest()

    def verify_password(self, password):
        return self.encrypt_password(password) == self.password

# 示例
```

```python
def assign_task(creators, tasks):
    assigned_tasks = {}
    for creator in creators:
        for task in tasks:
            if creator.skill_level >= task.difficulty and creator.not_assigned:
                assigned_tasks[creator] = task
                creator.not_assigned = False
                break
    return assigned_tasks

class Creator:
    def __init__(self, skill_level):
        self.skill_level = skill_level
        self.not_assigned = True

class Task:
    def __init__(self, difficulty):
        self.difficulty = difficulty

creators = [Creator(5), Creator(7), Creator(8)]
tasks = [Task(3), Task(5), Task(8)]

```

```python
class Review:
    def __init__(self, rating, comment):
        self.rating = rating
        self.comment = comment

class ReviewSystem:
    def __init__(self):
        self.reviews = []

    def add_review(self, review):
        self.reviews.append(review)

    def average_rating(self):
        if not self.reviews:
            return 0
        total_rating = sum(review.rating for review in self.reviews)
        return total_rating / len(self.reviews)

# 示例
review_system = ReviewSystem()
review_system.add_review(Review(4, 'Excellent work!'))
review_system.add_review(Review(5, 'Wonderful project!'))
print(review_system.average_rating())  # 输出：4.5
```
