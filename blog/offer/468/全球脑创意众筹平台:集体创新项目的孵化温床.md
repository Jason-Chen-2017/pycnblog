    def __init__(self):
        self.projects = {}
        self.user_funds = {}

    def create_project(self, project_id, owner_id, target_amount):
        if owner_id in self.user_funds:
            self.projects[project_id] = {
                'owner_id': owner_id,
                'target_amount': target_amount,
                'current_amount': 0
            }
            print(f"Project {project_id} created successfully.")
        else:
            print("Error: User not found.")

    def donate_to_project(self, project_id, donor_id, amount):
        if project_id in self.projects:
            self.projects[project_id]['current_amount'] += amount
            self.user_funds[donor_id] -= amount
            print(f"Donation successful. Amount: {amount}")
        else:
            print("Error: Project not found.")

    def check_project_status(self, project_id):
        if project_id in self.projects:
            project = self.projects[project_id]
            if project['current_amount'] >= project['target_amount']:
                print("Project funded successfully.")
            else:
                print(f"Project funding in progress. Target amount: {project['target_amount']}, Current amount: {project['current_amount']}")
        else:
            print("Error: Project not found.")

class ContentBasedRecommender:
    def __init__(self, projects):
        self.projects = projects
        self.project_tags = self.extract_project_tags()

    def extract_project_tags(self):
        project_tags = {}
        for project in self.projects:
            project_tags[project['id']] = project['tags']
        return project_tags

    def recommend_projects(self, user_interests, top_n=5):
        recommended_projects = []
        for project_id, project_tags in self.project_tags.items():
            if any(tag in project_tags for tag in user_interests):
                recommended_projects.append(project_id)
        recommended_projects = sorted(recommended_projects, key=lambda x: -len(self.project_tags[x]))
        return recommended_projects[:top_n]

class PaymentEncryption:
    def __init__(self, key):
        self.key = key
        self.fernet = Fernet(key)

    def encrypt_payment_data(self, data):
        encrypted_data = self.fernet.encrypt(data.encode('utf-8'))
        return encrypted_data

    def decrypt_payment_data(self, encrypted_data):
        decrypted_data = self.fernet.decrypt(encrypted_data).decode('utf-8')
        return decrypted_data

        return project_tags

        categorized_projects = []
        for project_id, project_tags in self.project_tags.items():
            if any(tag in project_tags for tag in category_tags):
                categorized_projects.append(project_id)

            project = self.projects[project_id]
            category = project['category']
            if category in stats:
                stats[category]['count'] += 1
                stats[category]['total_amount'] += project['target_amount']
            else:
                stats[category] = {'count': 1, 'total_amount': project['target_amount']}
        return stats

import smtplib
from email.mime.text import MIMEText
from email.header import Header

def check_project_progress():
    # 示例：获取项目众筹进度数据
    current_amount = 2000
    target_amount = 2000
    progress = current_amount / target_amount * 100

    # 示例：设定报警规则
    if progress > 90:
        send_alarm("Project progress exceeds 90%.")

def send_alarm(message):
    sender = 'platform@example.com'
    receiver = 'manager@example.com'
    subject = 'Project Crowdfunding Alarm'
    content = f"{message}\n\nPlease check the project crowdfunding progress."

    message = MIMEText(content, 'plain', 'utf-8')
    message['From'] = Header('创意众筹平台', 'utf-8')
    message['To'] = Header('管理员', 'utf-8')
    message['Subject'] = Header(subject, 'utf-8')

    try:
        smtp_obj = smtplib.SMTP('smtp.example.com', 587)
        smtp_obj.starttls()
        smtp_obj.sendmail(sender, receiver, message.as_string())
        print("Alarm sent successfully.")
    except Exception as e:
        print(f"Error sending alarm: {e}")

# 测试代码
schedule.every(1).minutes.do(check_project_progress)

while True:
    schedule.run_pending()
    time.sleep(1)
