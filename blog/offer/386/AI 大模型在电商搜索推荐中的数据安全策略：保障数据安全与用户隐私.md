    # 计算预测准确率
    accuracy = np.mean(true_labels == predicted_labels)
    print("Algorithm Accuracy:", accuracy)
    
    # 计算不同群体的准确率
    group_accuracies = {}
    for i in range(len(true_labels)):

    # 计算每个群体的准确率

        group_accuracy = np.mean(labels)
        print(f"Group {group} Accuracy:", group_accuracy)
