def string_matching(pattern, text):
    n, m = len(pattern), len(text)
    dp = [[False] * (m + 1) for _ in range(n + 1)]

    for i in range(n + 1):
        for j in range(m + 1):
            if i == 0:
                dp[i][j] = j == 0
            elif j == 0:
                dp[i][j] = False
            elif pattern[i - 1] == text[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                dp[i][j] = dp[i - 1][j - 1] or dp[i][j - 1]

    return dp[n][m]
