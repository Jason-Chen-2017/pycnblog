     ```python
     import cv2

     def detect_change(image1, image2):
         image1 = cv2.cvtColor(image1, cv2.COLOR_GRAY2BGR)
         image2 = cv2.cvtColor(image2, cv2.COLOR_GRAY2BGR)
         change_mask = cv2.absdiff(image1, image2)
         _, change_mask = cv2.threshold(change_mask, 30, 255, cv2.THRESH_BINARY)
     ```

     ```python
     import json

     def annotate(change_mask, annotations):
         contours, _ = cv2.findContours(change_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
         for contour in contours:
             x, y, w, h = cv2.boundingRect(contour)
             annotations["shapes"].append({
                 "label": "change",
                 "points": [[x, y], [x+w, y], [x+w, y+h], [x, y+h]],
                 "group_id": 1
             })

         with open("annotations.json", "w") as f:
             json.dump(annotations, f)

     annotate(change_mask, annotations)
     ```
