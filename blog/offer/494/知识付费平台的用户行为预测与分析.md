```python
def preprocess_data(data):
    # 去除缺失值
    data = data.dropna()

    # 填充缺失值
    data['age'] = data['age'].fillna(data['age'].mean())
    data['income'] = data['income'].fillna(data['income'].mean())

    # 数据类型转换
    data['age'] = data['age'].astype(int)
    data['income'] = data['income'].astype(float)

    return data
```

**3. 实现用户留存率计算函数**

**题目：** 编写一个函数，用于计算知识付费平台的用户留存率。输入用户登录数据，输出日留存率、周留存率、月留存率。

```python

    # 请在这里编写代码
    pass
```

**答案：**

```python
def calculate_retention_rate(data):
    # 计算日留存率
    dailyRetention = data.groupby('day')['user_id'].nunique() / data['user_id'].nunique()

    # 计算周留存率
    weeklyRetention = data.groupby('week')['user_id'].nunique() / data['user_id'].nunique()

    # 计算月留存率
    monthlyRetention = data.groupby('month')['user_id'].nunique() / data['user_id'].nunique()

    return dailyRetention, weeklyRetention, monthlyRetention
```

```python
def detect_fraud(data):
    # 创建孤立森林模型
    model = IsolationForest(contamination=0.01)

    # 训练模型
    model.fit(data)

    # 预测欺诈用户
    fraud_users = model.predict(data)

    # 标记欺诈用户
    data['is_fraud'] = fraud_users == -1

    return data
```
