```go
package main

import (
    "fmt"
    "strings"
)

func findMostFrequentChar(s string) (char rune, count int) {
    charFrequency := make(map[rune]int)
    maxCount := 0

    for _, char := range s {
        charFrequency[char]++
        if charFrequency[char] > maxCount {
            maxCount = charFrequency[char]
        }
    }

    for char, count := range charFrequency {
        if count == maxCount {
            return char, count
        }
    }

    return ' ', 0
}

func main() {
    s := "aabbbccccddddd"
    char, count := findMostFrequentChar(s)
    fmt.Printf("The most frequent char is '%c', with a count of %d\n", char, count)
}
```

```go
package main

import (
    "fmt"
    "strings"
)

func compressString(s string) string {
    compressed := ""
    count := 1

    for i := 1; i < len(s); i++ {
        if s[i] == s[i-1] {
            count++
        } else {
            compressed += string(s[i-1]) + strconv.Itoa(count)
            count = 1
        }
    }

    compressed += string(s[len(s)-1]) + strconv.Itoa(count)
    return compressed
}

func decompressString(s string) string {
    decompressed := ""
    for i := 0; i < len(s); i++ {
        if i < len(s)-1 && isDigit(s[i+1]) {
            count := 0
            for isDigit(s[i+1]) {
                count = count*10 + int(s[i+1]-'0')
                i++
            }
            decompressed += strings.Repeat(string(s[i]), count)
        } else {
            decompressed += string(s[i])
        }
    }
    return decompressed
}

func isDigit(char rune) bool {
    return char >= '0' && char <= '9'
}

func main() {
```
