def filter_useless_content(content_list, keywords):
    """
    过滤掉包含指定关键词的内容
    :param content_list: 内容列表
    :param keywords: 关键词列表
    :return: 过滤后的内容列表

**代码示例：**

```python
from collections import defaultdict

def collaborative_filtering(train_data, user_history, k=5):
    """
    基于协同过滤的推荐算法
    :param train_data: 训练数据（用户-物品评分矩阵）
    :param user_history: 用户历史行为数据
    :param k: 附近邻居的数量
    :return: 推荐结果列表
    """
    sim_matrix = defaultdict(dict)
    for user_a, items_a in train_data.items():
        for user_b, items_b in train_data.items():
            if user_a == user_b:
                continue
            sim_sum = 0
            common_items = 0
            for item in items_a:
                if item in items_b:
                    sim_sum += (items_a[item] - items_b[item]) ** 2
                    common_items += 1
            if common_items > 0:
                sim_matrix[user_a][user_b] = 1 - (sim_sum / (common_items * common_items))
    
    user_similarity = defaultdict(dict)
    for user_a, items_a in train_data.items():
        for user_b, items_b in train_data.items():
            if user_a == user_b:
                continue
            user_similarity[user_a][user_b] = sum(sim_matrix[user_a][u_b] for u_b in items_b) / k
    
    user_profile = defaultdict(set)
    for user_a, items_a in train_data.items():
        for item in items_a:
            for user_b, items_b in train_data.items():
                if item in items_b:
                    user_profile[user_a].add(user_b)
    
    recommendations = []
    for user in user_history:
        similarity_scores = defaultdict(float)
        for neighbor in user_profile[user]:
            for item in train_data[neighbor]:
                if item not in user_history:
                    similarity_scores[item] += user_similarity[user][neighbor]
        recommended_items = sorted(similarity_scores.items(), key=lambda x: x[1], reverse=True)
        recommendations.append(recommended_items[:k])
    
    return recommendations
```
